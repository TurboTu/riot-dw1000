121 static void test_sixlowpan_ctx_lookup_addr__same_addr(void)
122 {
123  ipv6_addr_t addr = DEFAULT_TEST_PREFIX;
124  gnrc_sixlowpan_ctx_t *ctx;
125 
126  /* add context DEFAULT_TEST_PREFIX to DEFAULT_TEST_ID */
127  test_sixlowpan_ctx_update__success();
128  TEST_ASSERT_NOT_NULL((ctx = gnrc_sixlowpan_ctx_lookup_addr(&addr)));
129  TEST_ASSERT_EQUAL_INT(GNRC_SIXLOWPAN_CTX_FLAGS_COMP | DEFAULT_TEST_ID, ctx->flags_id);
130  TEST_ASSERT_EQUAL_INT(DEFAULT_TEST_PREFIX_LEN, ctx->prefix_len);
131  TEST_ASSERT(TEST_UINT16 >= ctx->ltime);
132  TEST_ASSERT(DEFAULT_TEST_PREFIX_LEN >= ipv6_addr_match_prefix(&addr, &ctx->prefix));
133 }
