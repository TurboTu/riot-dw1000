57 static void clk_init(void)
58 {
59  /* Reset the RCC clock configuration to the default reset state(for debug purpose) */
60  /* Set MSION bit */
61  RCC->CR |= RCC_CR_MSION;
62  /* Reset SW, HPRE, PPRE1, PPRE2, MCOSEL and MCOPRE bits */
63  RCC->CFGR &= ~(RCC_CFGR_PLLSRC | RCC_CFGR_PLLDIV | RCC_CFGR_PLLMUL);
64  /* Reset HSION, HSEON, CSSON and PLLON bits */
65  RCC->CR &= ~(RCC_CR_HSION | RCC_CR_HSEON | RCC_CR_HSEBYP | RCC_CR_CSSON | RCC_CR_PLLON);
66  /* Disable all interrupts */
67  RCC->CIR = 0x0;
68 
69  /* SYSCLK, HCLK, PCLK2 and PCLK1 configuration */
70  /* Enable high speed clock source */
71  RCC->CR |= CLOCK_CR_SOURCE;
72  /* Wait till the high speed clock source is ready
73  * NOTE: the MCU will stay here forever if you use an external clock source and it's not connected */
74  while (!(RCC->CR & CLOCK_CR_SOURCE_RDY)) {}
75  FLASH->ACR |= FLASH_ACR_ACC64;
76  /* Enable Prefetch Buffer */
77  FLASH->ACR |= FLASH_ACR_PRFTEN;
78  /* Flash 1 wait state */
79  FLASH->ACR |= CLOCK_FLASH_LATENCY;
80  /* Power enable */
81  periph_clk_en(APB1, RCC_APB1ENR_PWREN);
82  /* Select the Voltage Range 1 (1.8 V) */
83  PWR->CR = PWR_CR_VOS_0;
84  /* Wait Until the Voltage Regulator is ready */
85  while((PWR->CSR & PWR_CSR_VOSF) != 0) {}
86  /* HCLK = SYSCLK */
87  RCC->CFGR |= (uint32_t)CLOCK_AHB_DIV;
88  /* PCLK2 = HCLK */
89  RCC->CFGR |= (uint32_t)CLOCK_APB2_DIV;
90  /* PCLK1 = HCLK */
91  RCC->CFGR |= (uint32_t)CLOCK_APB1_DIV;
92  /* PLL configuration: PLLCLK = CLOCK_SOURCE / PLL_DIV * PLL_MUL */
93  RCC->CFGR &= ~((uint32_t)(RCC_CFGR_PLLSRC | RCC_CFGR_PLLDIV | RCC_CFGR_PLLMUL));
94  RCC->CFGR |= (uint32_t)(CLOCK_PLL_SOURCE | CLOCK_PLL_DIV | CLOCK_PLL_MUL);
95  /* Enable PLL */
96  RCC->CR |= RCC_CR_PLLON;
97  /* Wait till PLL is ready */
98  while ((RCC->CR & RCC_CR_PLLRDY) == 0) {}
99  /* Select PLL as system clock source */
100  RCC->CFGR &= ~((uint32_t)(RCC_CFGR_SW));
101  RCC->CFGR |= (uint32_t)RCC_CFGR_SW_PLL;
102  /* Wait till PLL is used as system clock source */
103  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL) {}
104 }
