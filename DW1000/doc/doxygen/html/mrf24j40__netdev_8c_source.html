86 static int _send(netdev2_t *netdev, const struct iovec *vector, unsigned count)
87 {
88  mrf24j40_t *dev = (mrf24j40_t *)netdev;
89  const struct iovec *ptr = vector;
90  size_t len = 0;
91 
92  mrf24j40_tx_prepare(dev);
93 
94  /* load packet data into FIFO */
95  for (int i = 0; i < count; i++, ptr++) {
96  /* current packet data + FCS too long */
97  if ((len + ptr->iov_len + 2) > IEEE802154_FRAME_LEN_MAX) {
98  DEBUG("[mrf24j40] error: packet too large (%u byte) to be send\n",
99  (unsigned)len + 2);
100  return -EOVERFLOW;
101  }
102 
103 #ifdef MODULE_NETSTATS_L2
104  netdev->stats.tx_bytes += len;
105 #endif
106  len = mrf24j40_tx_load(dev, ptr->iov_base, ptr->iov_len, len);
107  if (i == 0) {
108  dev->header_len = len;
109  }
110 
111  }
112 
113  /* send data out directly if pre-loading is disabled */
114  if (!(dev->netdev.flags & MRF24J40_OPT_PRELOADING)) {
115  mrf24j40_tx_exec(dev);
116  }
117  /* return the number of bytes that were actually send out */
118  return (int)len;
119 
120 }

122 static int _recv(netdev2_t *netdev, void *buf, size_t len, void *info)
123 {
124  mrf24j40_t *dev = (mrf24j40_t *)netdev;
125  uint8_t phr;
126  size_t pkt_len;
127 
128  /* Disable receiving while reading the RX fifo (datasheet sec. 3.11.4) */
129  mrf24j40_reg_write_short(dev, MRF24J40_REG_BBREG1, MRF24J40_BBREG1_RXDECINV );
130 
131  /* get the size of the received packet */
132  phr = mrf24j40_reg_read_long(dev, MRF24J40_RX_FIFO);
133 
134  pkt_len = (phr & 0x7f) - 2;
135 
136  /* just return length when buf == NULL */
137  if (buf == NULL) {
138  return pkt_len;
139  }
140 #ifdef MODULE_NETSTATS_L2
141  netdev->stats.rx_count++;
142  netdev->stats.rx_bytes += pkt_len;
143 #endif
144  /* not enough space in buf */
145  if (pkt_len > len) {
146  DEBUG("[mrf24j40] No space in receive buffers\n");
147  mrf24j40_reg_write_short(dev, MRF24J40_REG_RXFLUSH, MRF24J40_RXFLUSH_RXFLUSH);
148  /* Turn on reception of packets off the air */
149  mrf24j40_reg_write_short(dev, MRF24J40_REG_BBREG1, 0x00);
150  return -ENOBUFS;
151  }
152  /* copy payload */
153  mrf24j40_rx_fifo_read(dev, 1, (uint8_t *)buf, pkt_len);
154 
155  if (info != NULL) {
156  netdev2_ieee802154_rx_info_t *radio_info = info;
157  /* Read LQI and RSSI values from the RX fifo */
158  mrf24j40_rx_fifo_read(dev, phr + 1, &(radio_info->lqi), 1);
159  mrf24j40_rx_fifo_read(dev, phr + 2, &(radio_info->rssi), 1);
160  }
161 
162  /* Turn on reception of packets off the air */
163  mrf24j40_reg_write_short(dev, MRF24J40_REG_BBREG1, 0x00);
164  return pkt_len;
165 }

182 static int _get(netdev2_t *netdev, netopt_t opt, void *val, size_t max_len)
183 {
184  mrf24j40_t *dev = (mrf24j40_t *) netdev;
185 
186  if (netdev == NULL) {
187  return -ENODEV;
188  }
189 
190  int res;
191  switch (opt) {
192  case NETOPT_CHANNEL_PAGE:
193  if (max_len < sizeof(uint16_t)) {
194  res = -EOVERFLOW;
195  }
196  else {
197  ((uint8_t *)val)[1] = 0;
198  ((uint8_t *)val)[0] = 0;
199  res = sizeof(uint16_t);
200  }
201  break;
202 
203  case NETOPT_MAX_PACKET_SIZE:
204  if (max_len < sizeof(int16_t)) {
205  res = -EOVERFLOW;
206  }
207  else {
208  *((uint16_t *)val) = IEEE802154_FRAME_LEN_MAX - _MAX_MHR_OVERHEAD;
209  res = sizeof(uint16_t);
210  }
211  break;
212 
213  case NETOPT_STATE:
214  if (max_len < sizeof(netopt_state_t)) {
215  res = -EOVERFLOW;
216  }
217  else {
218  *((netopt_state_t *)val) = _get_state(dev);
219  res = sizeof(netopt_state_t);
220  }
221  break;
222 
223  case NETOPT_PRELOADING:
224  if (dev->netdev.flags & MRF24J40_OPT_PRELOADING) {
225  *((netopt_enable_t *)val) = NETOPT_ENABLE;
226  }
227  else {
228  *((netopt_enable_t *)val) = NETOPT_DISABLE;
229  }
230  res = sizeof(netopt_enable_t);
231  break;
232 
233  case NETOPT_PROMISCUOUSMODE:
234  if (dev->netdev.flags & MRF24J40_OPT_PROMISCUOUS) {
235  *((netopt_enable_t *)val) = NETOPT_ENABLE;
236  }
237  else {
238  *((netopt_enable_t *)val) = NETOPT_DISABLE;
239  }
240  res = sizeof(netopt_enable_t);
241  break;
242 
243  case NETOPT_RX_START_IRQ:
244  *((netopt_enable_t *)val) =
245  !!(dev->netdev.flags & MRF24J40_OPT_TELL_RX_START);
246  res = sizeof(netopt_enable_t);
247  break;
248 
249  case NETOPT_RX_END_IRQ:
250  *((netopt_enable_t *)val) =
251  !!(dev->netdev.flags & MRF24J40_OPT_TELL_RX_END);
252  res = sizeof(netopt_enable_t);
253  break;
254 
255  case NETOPT_TX_START_IRQ:
256  *((netopt_enable_t *)val) =
257  !!(dev->netdev.flags & MRF24J40_OPT_TELL_TX_START);
258  res = sizeof(netopt_enable_t);
259  break;
260 
261  case NETOPT_TX_END_IRQ:
262  *((netopt_enable_t *)val) =
263  !!(dev->netdev.flags & MRF24J40_OPT_TELL_TX_END);
264  res = sizeof(netopt_enable_t);
265  break;
266 
267  case NETOPT_CSMA:
268  *((netopt_enable_t *)val) =
269  !!(dev->netdev.flags & MRF24J40_OPT_CSMA);
270  res = sizeof(netopt_enable_t);
271  break;
272 
273  case NETOPT_TX_POWER:
274  if (max_len < sizeof(int16_t)) {
275  res = -EOVERFLOW;
276  }
277  else {
278  *((uint16_t *)val) = mrf24j40_get_txpower(dev);
279  res = sizeof(uint16_t);
280  }
281  break;
282 
283  case NETOPT_RETRANS:
284  if (max_len < sizeof(uint8_t)) {
285  res = -EOVERFLOW;
286  }
287  else {
288  *((uint8_t *)val) = MRF24J40_MAX_FRAME_RETRIES;
289  res = sizeof(uint8_t);
290  }
291  break;
292 
293  case NETOPT_IS_CHANNEL_CLR:
294  if (mrf24j40_cca(dev)) {
295  *((netopt_enable_t *)val) = NETOPT_ENABLE;
296  }
297  else {
298  *((netopt_enable_t *)val) = NETOPT_DISABLE;
299  }
300  res = sizeof(netopt_enable_t);
301  break;
302 
303  case NETOPT_CSMA_RETRIES:
304  if (max_len < sizeof(uint8_t)) {
305  res = -EOVERFLOW;
306  }
307  else {
308  *((uint8_t *)val) = mrf24j40_get_csma_max_retries(dev);
309  res = sizeof(uint8_t);
310  }
311  break;
312 
313  case NETOPT_CCA_THRESHOLD:
314  if (max_len < sizeof(int8_t)) {
315  res = -EOVERFLOW;
316  }
317  else {
318  *((int8_t *)val) = mrf24j40_get_cca_threshold(dev);
319  res = sizeof(int8_t);
320  }
321  break;
322 
323  default:
324  /* try netdev2 settings */
325  res = netdev2_ieee802154_get((netdev2_ieee802154_t *)netdev, opt,
326  val, max_len);
327  }
328  return res;
329 }

354 static int _set(netdev2_t *netdev, netopt_t opt, void *val, size_t len)
355 {
356  mrf24j40_t *dev = (mrf24j40_t *) netdev;
357  int res = -ENOTSUP;
358 
359  if (dev == NULL) {
360  return -ENODEV;
361  }
362 
363  switch (opt) {
364  case NETOPT_ADDRESS:
365  if (len > sizeof(uint16_t)) {
366  res = -EOVERFLOW;
367  }
368  else {
369  mrf24j40_set_addr_short(dev, *((uint16_t *)val));
370  /* don't set res to set netdev2_ieee802154_t::short_addr */
371  }
372  break;
373 
374  case NETOPT_ADDRESS_LONG:
375  if (len > sizeof(uint64_t)) {
376  res = -EOVERFLOW;
377  }
378  else {
379  mrf24j40_set_addr_long(dev, *((uint64_t *)val));
380  /* don't set res to set netdev2_ieee802154_t::long_addr */
381  }
382  break;
383 
384  case NETOPT_NID:
385  if (len > sizeof(uint16_t)) {
386  res = -EOVERFLOW;
387  }
388  else {
389  mrf24j40_set_pan(dev, *((uint16_t *)val));
390  /* don't set res to set netdev2_ieee802154_t::pan */
391  }
392  break;
393 
394  case NETOPT_CHANNEL:
395  if (len != sizeof(uint16_t)) {
396  res = -EINVAL;
397  }
398  else {
399  uint8_t chan = ((uint8_t *)val)[0];
400  if (chan < IEEE802154_CHANNEL_MIN ||
401  chan > IEEE802154_CHANNEL_MAX ||
402  dev->netdev.chan == chan) {
403  res = -EINVAL;
404  break;
405  }
406  mrf24j40_set_chan(dev, chan);
407  /* don't set res to set netdev2_ieee802154_t::chan */
408  }
409  break;
410 
411  case NETOPT_CHANNEL_PAGE:
412  if (len != sizeof(uint16_t)) {
413  res = -EINVAL;
414  }
415  else {
416  uint8_t page = ((uint8_t *)val)[0];
417 
418  /* mrf24j40 only supports page 0, no need to configure anything in the driver. */
419  if (page != 0) {
420  res = -EINVAL;
421  }
422  else {
423  res = sizeof(uint16_t);
424  }
425  }
426  break;
427 
428  case NETOPT_TX_POWER:
429  if (len > sizeof(int16_t)) {
430  res = -EOVERFLOW;
431  }
432  else {
433  mrf24j40_set_txpower(dev, *((int16_t *)val));
434  res = sizeof(uint16_t);
435  }
436  break;
437 
438  case NETOPT_STATE:
439  if (len > sizeof(netopt_state_t)) {
440  res = -EOVERFLOW;
441  }
442  else {
443  res = _set_state(dev, *((netopt_state_t *)val));
444  }
445  break;
446 
447  case NETOPT_AUTOACK:
448  mrf24j40_set_option(dev, NETDEV2_IEEE802154_ACK_REQ,
449  ((bool *)val)[0]);
450  /* don't set res to set netdev2_ieee802154_t::flags */
451  break;
452 
453  case NETOPT_PRELOADING:
454  mrf24j40_set_option(dev, MRF24J40_OPT_PRELOADING,
455  ((bool *)val)[0]);
456  res = sizeof(netopt_enable_t);
457  break;
458 
459  case NETOPT_PROMISCUOUSMODE:
460  mrf24j40_set_option(dev, MRF24J40_OPT_PROMISCUOUS,
461  ((bool *)val)[0]);
462  res = sizeof(netopt_enable_t);
463  break;
464 
465  case NETOPT_RX_START_IRQ:
466  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_RX_START,
467  ((bool *)val)[0]);
468  res = sizeof(netopt_enable_t);
469  break;
470 
471  case NETOPT_RX_END_IRQ:
472  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_RX_END,
473  ((bool *)val)[0]);
474  res = sizeof(netopt_enable_t);
475  break;
476 
477  case NETOPT_TX_START_IRQ:
478  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_TX_START,
479  ((bool *)val)[0]);
480  res = sizeof(netopt_enable_t);
481  break;
482 
483  case NETOPT_TX_END_IRQ:
484  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_TX_END,
485  ((bool *)val)[0]);
486  res = sizeof(netopt_enable_t);
487  break;
488 
489  case NETOPT_CSMA:
490  mrf24j40_set_option(dev, MRF24J40_OPT_CSMA,
491  ((bool *)val)[0]);
492  res = sizeof(netopt_enable_t);
493  break;
494 
495  case NETOPT_CSMA_RETRIES:
496  if ((len > sizeof(uint8_t)) ||
497  (*((uint8_t *)val) > 5)) {
498  res = -EOVERFLOW;
499  }
500  else if (dev->netdev.flags & MRF24J40_OPT_CSMA) {
501  /* only set if CSMA is enabled */
502  mrf24j40_set_csma_max_retries(dev, *((uint8_t *)val));
503  res = sizeof(uint8_t);
504  }
505  break;
506 
507  case NETOPT_CCA_THRESHOLD:
508  if (len > sizeof(int8_t)) {
509  res = -EOVERFLOW;
510  }
511  else {
512  mrf24j40_set_cca_threshold(dev, *((int8_t *)val));
513  res = sizeof(int8_t);
514  }
515  break;
516 
517  default:
518  break;
519  }
520  /* try netdev2 building flags */
521  if (res == -ENOTSUP) {
522  res = netdev2_ieee802154_set((netdev2_ieee802154_t *)netdev, opt,
523  val, len);
524  }
525  return res;
526 }

528 static void _isr(netdev2_t *netdev)
529 {
530  mrf24j40_t *dev = (mrf24j40_t *) netdev;
531  /* update pending bits */
532  mrf24j40_update_tasks(dev);
533  DEBUG("[mrf24j40] INTERRUPT (pending: %x),\n", dev->pending);
534  /* Transmit interrupt occured */
535  if (dev->pending & MRF24J40_TASK_TX_READY) {
536  dev->pending &= ~(MRF24J40_TASK_TX_READY);
537  DEBUG("[mrf24j40] EVT - TX_END\n");
538 #ifdef MODULE_NETSTATS_L2
539  if (netdev->event_callback && (dev->netdev.flags & MRF24J40_OPT_TELL_TX_END)) {
540  uint8_t txstat = mrf24j40_reg_read_short(dev, MRF24J40_REG_TXSTAT);
541  /* transmision failed */
542  if (txstat & MRF24J40_TXSTAT_TXNSTAT) {
543  /* TX_NOACK - CCAFAIL */
544  if (txstat & MRF24J40_TXSTAT_CCAFAIL) {
545  netdev->event_callback(netdev, NETDEV2_EVENT_TX_MEDIUM_BUSY);
546  DEBUG("[mrf24j40] TX_CHANNEL_ACCESS_FAILURE\n");
547  }
548  /* check max retries */
549  else if (txstat & MRF24J40_TXSTAT_MAX_FRAME_RETRIES) {
550  netdev->event_callback(netdev, NETDEV2_EVENT_TX_NOACK);
551  DEBUG("[mrf24j40] TX NO_ACK\n");
552  }
553  }
554  else {
555  netdev->event_callback(netdev, NETDEV2_EVENT_TX_COMPLETE);
556  }
557  }
558 #endif
559  }
560  /* Receive interrupt occured */
561  if (dev->pending & MRF24J40_TASK_RX_READY) {
562  DEBUG("[mrf24j40] EVT - RX_END\n");
563  if ((dev->netdev.flags & MRF24J40_OPT_TELL_RX_END)) {
564  netdev->event_callback(netdev, NETDEV2_EVENT_RX_COMPLETE);
565  }
566  dev->pending &= ~(MRF24J40_TASK_RX_READY);
567  }
568  DEBUG("[mrf24j40] END IRQ\n");
569 }
