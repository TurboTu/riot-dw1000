50 void _set_payload(icmpv6_echo_t *hdr, size_t payload_len)
51 {
52  size_t i = 0;
53  uint8_t *payload = (uint8_t *)(hdr + 1);
54 
55  while (i < payload_len) {
56  if (id > 0xff) {
57  payload[i] = (uint8_t)(id >> 8);
58  payload[i + 1] = (uint8_t)(id & 0xff);
59  i += 2;
60  }
61  else {
62  payload[i++] = (uint8_t)(id & 0xff);
63  }
64  }
65 
66  if (i > payload_len) { /* when id > 0xff and payload_len is odd */
67  payload[payload_len - 1] = id >> 8;
68  }
69 }

82 int _handle_reply(gnrc_pktsnip_t *pkt, uint32_t time)
83 {
84  gnrc_pktsnip_t *ipv6, *icmpv6;
85  ipv6_hdr_t *ipv6_hdr;
86  icmpv6_echo_t *icmpv6_hdr;
87  uint16_t seq;
88 
89  ipv6 = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_IPV6);
90  icmpv6 = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_ICMPV6);
91 
92  if ((ipv6 == NULL) || (icmpv6 == NULL)) {
93  puts("error: IPv6 header or ICMPv6 header not found in reply");
94  return 0;
95  }
96 
97  ipv6_hdr = ipv6->data;
98  icmpv6_hdr = icmpv6->data;
99  seq = byteorder_ntohs(icmpv6_hdr->seq);
100 
101  if ((byteorder_ntohs(icmpv6_hdr->id) == id) && _expected_seq(seq)) {
102  if (seq <= min_seq_expected) {
103  min_seq_expected++;
104  }
105 
106  printf("%u bytes from %s: id=%" PRIu16 " seq=%" PRIu16 " hop limit=%u time = %"
107  PRIu32 ".%03" PRIu32 " ms\n", (unsigned) icmpv6->size,
108  ipv6_addr_to_str(ipv6_str, &(ipv6_hdr->src), sizeof(ipv6_str)),
109  byteorder_ntohs(icmpv6_hdr->id), seq, (unsigned)ipv6_hdr->hl,
110  time / US_PER_MS, time % US_PER_MS);
111 #ifdef MODULE_GNRC_IPV6_NC
112  gnrc_ipv6_nc_still_reachable(&ipv6_hdr->src);
113 #endif
114  }
115  else {
116  puts("error: unexpected parameters");
117  return 0;
118  }
119 
120  return 1;
121 }

123 static void _print_stats(char *addr_str, int success, int count, uint64_t total_time,
124  uint64_t sum_rtt, uint32_t min_rtt, uint32_t max_rtt)
125 {
126  printf("--- %s ping statistics ---\n", addr_str);
127 
128  if (success > 0) {
129  uint32_t avg_rtt = (uint32_t)(sum_rtt / count); /* get average */
130  printf("%d packets transmitted, %d received, %d%% packet loss, time %"
131  PRIu32 ".06%" PRIu32 " s\n", count, success,
132  (100 - ((success * 100) / count)),
133  (uint32_t)total_time / US_PER_SEC, (uint32_t)total_time % US_PER_SEC);
134  printf("rtt min/avg/max = "
135  "%" PRIu32 ".%03" PRIu32 "/"
136  "%" PRIu32 ".%03" PRIu32 "/"
137  "%" PRIu32 ".%03" PRIu32 " ms\n",
138  min_rtt / US_PER_MS, min_rtt % US_PER_MS,
139  avg_rtt / US_PER_MS, avg_rtt % US_PER_MS,
140  max_rtt / US_PER_MS, max_rtt % US_PER_MS);
141  }
142  else {
143  printf("%d packets transmitted, 0 received, 100%% packet loss\n", count);
144  }
145 }

147 int _icmpv6_ping(int argc, char **argv)
148 {
149  int count = 3, success = 0, remaining, stat_interval = 0, stat_counter = 0;
150  size_t payload_len = 4;
151  uint32_t delay = 1 * MS_PER_SEC;
152  char *addr_str;
153  ipv6_addr_t addr;
154  kernel_pid_t src_iface;
155  msg_t msg;
156  gnrc_netreg_entry_t my_entry = GNRC_NETREG_ENTRY_INIT_PID(ICMPV6_ECHO_REP,
157  sched_active_pid);
158  uint32_t min_rtt = UINT32_MAX, max_rtt = 0;
159  uint64_t sum_rtt = 0;
160  uint64_t ping_start;
161  int param_offset = 0;
162 
163  if (argc < 2) {
164  usage(argv);
165  return 1;
166  }
167  else if ((argc > 2) && ((count = atoi(argv[1])) > 0)) {
168  param_offset = 1;
169  }
170  else {
171  count = 3;
172  }
173 
174  stat_interval = count;
175 
176  addr_str = argv[1 + param_offset];
177  if (argc > (2 + param_offset)) {
178  payload_len = atoi(argv[2 + param_offset]);
179  }
180  if (argc > (3 + param_offset)) {
181  delay = atoi(argv[3 + param_offset]);
182  }
183  if (argc > (4 + param_offset)) {
184  stat_interval = atoi(argv[4 + param_offset]);
185  }
186 
187  if ((int)payload_len < 0) {
188  usage(argv);
189  return 1;
190  }
191 
192  src_iface = ipv6_addr_split_iface(addr_str);
193  if (src_iface == -1) {
194  src_iface = KERNEL_PID_UNDEF;
195  }
196 
197  if (ipv6_addr_from_str(&addr, addr_str) == NULL) {
198  puts("error: malformed address");
199  return 1;
200  }
201 
202  if (ipv6_addr_is_link_local(&addr) || (src_iface != KERNEL_PID_UNDEF)) {
203  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
204  size_t ifnum = gnrc_netif_get(ifs);
205  if (src_iface == KERNEL_PID_UNDEF) {
206  if (ifnum == 1) {
207  src_iface = ifs[0];
208  }
209  else {
210  puts("error: link local target needs interface parameter (use \"<address>%<ifnum>\")\n");
211  return 1;
212  }
213  }
214  else {
215  int valid = 0;
216  for (size_t i = 0; i < ifnum && i < GNRC_NETIF_NUMOF; i++) {
217  if (ifs[i] == src_iface) {
218  valid = 1;
219  break;
220  }
221  }
222  if (!valid) {
223  printf("error: %"PRIkernel_pid" is not a valid interface.\n", src_iface);
224  return 1;
225  }
226  }
227  }
228 
229  if (gnrc_netreg_register(GNRC_NETTYPE_ICMPV6, &my_entry) < 0) {
230  puts("error: network registry is full");
231  return 1;
232  }
233 
234  remaining = count;
235 
236  ping_start = xtimer_now_usec64();
237 
238  while ((remaining--) > 0) {
239  gnrc_pktsnip_t *pkt;
240  uint32_t start, timeout = 1 * US_PER_SEC;
241 
242  pkt = gnrc_icmpv6_echo_build(ICMPV6_ECHO_REQ, id, ++max_seq_expected,
243  NULL, payload_len);
244 
245  if (pkt == NULL) {
246  puts("error: packet buffer full");
247  continue;
248  }
249 
250  _set_payload(pkt->data, payload_len);
251 
252  pkt = gnrc_ipv6_hdr_build(pkt, NULL, &addr);
253 
254  if (pkt == NULL) {
255  puts("error: packet buffer full");
256  continue;
257  }
258 
259  if (src_iface != KERNEL_PID_UNDEF) {
260  pkt = gnrc_pktbuf_add(pkt, NULL, sizeof(gnrc_netif_hdr_t),
261  GNRC_NETTYPE_NETIF);
262 
263  if (pkt == NULL) {
264  puts("error: packet buffer full");
265  continue;
266  }
267 
268  gnrc_netif_hdr_init(((gnrc_netif_hdr_t *)pkt->data), 0, 0);
269  ((gnrc_netif_hdr_t *)pkt->data)->if_pid = src_iface;
270  }
271 
272  start = xtimer_now_usec();
273  if (!gnrc_netapi_dispatch_send(GNRC_NETTYPE_IPV6, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
274  puts("error: unable to send ICMPv6 echo request\n");
275  gnrc_pktbuf_release(pkt);
276  continue;
277  }
278 
279  /* TODO: replace when #4219 was fixed */
280  if (xtimer_msg_receive_timeout64(&msg, (uint64_t)timeout) >= 0) {
281  uint32_t stop;
282  switch (msg.type) {
283  case GNRC_NETAPI_MSG_TYPE_RCV:
284  stop = xtimer_now_usec() - start;
285 
286  gnrc_pktsnip_t *pkt = msg.content.ptr;
287  success += _handle_reply(pkt, stop);
288  gnrc_pktbuf_release(pkt);
289 
290  if (stop > max_rtt) {
291  max_rtt = stop;
292  }
293 
294  if (stop < min_rtt) {
295  min_rtt = stop;
296  }
297 
298  sum_rtt += stop;
299 
300  break;
301 
302  default:
303  /* requeue wrong packets */
304  msg_send(&msg, thread_getpid());
305  break;
306  }
307  }
308  else {
309  puts("ping timeout");
310  }
311 
312  while (msg_try_receive(&msg) > 0) {
313  if (msg.type == GNRC_NETAPI_MSG_TYPE_RCV) {
314  printf("dropping additional response packet (probably caused by duplicates)\n");
315  gnrc_pktsnip_t *pkt = msg.content.ptr;
316  gnrc_pktbuf_release(pkt);
317  }
318  }
319 
320  if (remaining > 0) {
321  xtimer_usleep64(delay * US_PER_MS);
322  }
323  if ((++stat_counter == stat_interval) || (remaining == 0)) {
324  uint64_t total_time = xtimer_now_usec64() - ping_start;
325  _print_stats(addr_str, success, (count - remaining), total_time, sum_rtt, min_rtt,
326  max_rtt);
327  stat_counter = 0;
328  }
329 
330  }
331 
332  max_seq_expected = 0;
333  id++;
334 
335  gnrc_netreg_unregister(GNRC_NETTYPE_ICMPV6, &my_entry);
336  while (msg_try_receive(&msg) > 0) {
337  if (msg.type == GNRC_NETAPI_MSG_TYPE_RCV) {
338  printf("dropping additional response packet (probably caused by duplicates)\n");
339  gnrc_pktsnip_t *pkt = msg.content.ptr;
340  gnrc_pktbuf_release(pkt);
341  }
342  }
343 
344  return success ? 0 : 1;
345 }
