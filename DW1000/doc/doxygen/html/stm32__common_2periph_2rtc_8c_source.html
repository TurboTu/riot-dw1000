62  PWR->CR |= PWR_CR_DBP;
63 
64 #if defined(CPU_FAM_STM32L1)
65  if (!(RCC->CSR & RCC_CSR_RTCEN)) {
66 #else
67  if (!(RCC->BDCR & RCC_BDCR_RTCEN)) {
68 #endif

70  }

73  RTC->WPR = RTC_WRITE_PROTECTION_KEY1;
74  RTC->WPR = RTC_WRITE_PROTECTION_KEY2;
75

77  RTC->ISR = 0;

129 int rtc_get_time(struct tm *time)
130 {
131  time->tm_year = MCU_YEAR_OFFSET;
132  time->tm_year += (((RTC->DR & RTC_DR_YT) >> 20) * 10) + ((RTC->DR & RTC_DR_YU) >> 16);
133  time->tm_mon = (((RTC->DR & RTC_DR_MT) >> 12) * 10) + ((RTC->DR & RTC_DR_MU) >> 8) - 1;
134  time->tm_mday = (((RTC->DR & RTC_DR_DT) >> 4) * 10) + ((RTC->DR & RTC_DR_DU) >> 0);
135  time->tm_hour = (((RTC->TR & RTC_TR_HT) >> 20) * 10) + ((RTC->TR & RTC_TR_HU) >> 16);
136  if (RTC->TR & RTC_TR_PM) {
137  /* 12PM is noon */
138  if (time->tm_hour != 12) {
139  time->tm_hour += 12;
140  }
141  }
142  else if ((RTC->CR & RTC_CR_FMT) && time->tm_hour == 12) {
143  /* 12AM is midnight */
144  time->tm_hour -= 12;
145  }
146  time->tm_min = (((RTC->TR & RTC_TR_MNT) >> 12) * 10) + ((RTC->TR & RTC_TR_MNU) >> 8);
147  time->tm_sec = (((RTC->TR & RTC_TR_ST) >> 4) * 10) + ((RTC->TR & RTC_TR_SU) >> 0);
148  return 0;
149 }

151 int rtc_set_alarm(struct tm *time, rtc_alarm_cb_t cb, void *arg)
152 {
153  /* Enable write access to RTC registers */
154  periph_clk_en(APB1, RCC_APB1ENR_PWREN);
155  PWR->CR |= PWR_CR_DBP;
156 
157  /* Unlock RTC write protection */
158  RTC->WPR = RTC_WRITE_PROTECTION_KEY1;
159  RTC->WPR = RTC_WRITE_PROTECTION_KEY2;
160 
161  /* Enter RTC Init mode */
162  RTC->ISR |= RTC_ISR_INIT;
163  while ((RTC->ISR & RTC_ISR_INITF) == 0) {}
164 
165  RTC->CR &= ~(RTC_CR_ALRAE);
166  while ((RTC->ISR & RTC_ISR_ALRAWF) == 0) {}
167 
168  RTC->ALRMAR &= ~(RTC_ALRMAR_MSK1 | RTC_ALRMAR_MSK2 | RTC_ALRMAR_MSK3 | RTC_ALRMAR_MSK4);
169  RTC->ALRMAR = ((((uint32_t)byte2bcd(time->tm_mday) << 24) & (RTC_ALRMAR_DT | RTC_ALRMAR_DU)) |
170  (((uint32_t)byte2bcd(time->tm_hour) << 16) & (RTC_ALRMAR_HT | RTC_ALRMAR_HU)) |
171  (((uint32_t)byte2bcd(time->tm_min) << 8) & (RTC_ALRMAR_MNT | RTC_ALRMAR_MNU)) |
172  (((uint32_t)byte2bcd(time->tm_sec) << 0) & (RTC_ALRMAR_ST | RTC_ALRMAR_SU)));
173  /* Enable Alarm A */
174  RTC->CR |= RTC_CR_ALRAE;
175  RTC->CR |= RTC_CR_ALRAIE;
176  RTC->ISR &= ~(RTC_ISR_ALRAF);
177 
178  /* Exit RTC init mode */
179  RTC->ISR &= (uint32_t) ~RTC_ISR_INIT;
180  /* Enable RTC write protection */
181  RTC->WPR = 0xFF;
182 
183  EXTI->IMR |= EXTI_IMR_MR17;
184  EXTI->RTSR |= EXTI_RTSR_TR17;
185 #if defined(CPU_FAM_STM32F0)

188 #else

190  NVIC_EnableIRQ(RTC_Alarm_IRQn);
191 #endif
192 
193  rtc_callback.cb = cb;
194  rtc_callback.arg = arg;
195 
196  return 0;
