129 int cc2538_get_tx_power(void)
130 {
131  int index;
132  int best_index = 0;
133  int best_delta = INT_MAX;
134  int txpower;
135 
136  txpower = RFCORE_XREG_TXPOWER & 0xff;
137 
138  for (index = 0; index < NUM_POWER_LEVELS; index++) {
139  int delta = ABS_DIFF(power_lut[index], txpower);
140 
141  if (delta < best_delta) {
142  best_delta = delta;
143  best_index = index;
144  }
145  }
146 
147  return OUTPUT_POWER_MIN + best_index;
148 }

205 void cc2538_set_state(cc2538_rf_t *dev, netopt_state_t state)
206 {
207  switch (state) {
208  case NETOPT_STATE_OFF:
209  case NETOPT_STATE_SLEEP:
210  cc2538_off();
211  dev->state = state;
212  break;
213 
214  case NETOPT_STATE_RX:
215  case NETOPT_STATE_IDLE:
216  if (!cc2538_is_on()) {
217  cc2538_on();
218  RFCORE_WAIT_UNTIL(RFCORE->XREG_FSMSTAT0bits.FSM_FFCTRL_STATE > FSM_STATE_RX_CALIBRATION);
219  }
220  dev->state = state;
221  break;
222 
223  case NETOPT_STATE_TX:
224  dev->state = NETOPT_STATE_IDLE;
225  break;
226 
227  case NETOPT_STATE_RESET:
228  cc2538_off();
229  cc2538_on();
230  RFCORE_WAIT_UNTIL(RFCORE->XREG_FSMSTAT0bits.FSM_FFCTRL_STATE > FSM_STATE_RX_CALIBRATION);
231  dev->state = NETOPT_STATE_IDLE;
232  break;
233  }
234 }
