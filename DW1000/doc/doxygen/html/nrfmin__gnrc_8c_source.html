58 static int hdr_netif_to_nrfmin(nrfmin_hdr_t *nrfmin, gnrc_pktsnip_t *pkt)
59 {
60  gnrc_netif_hdr_t *netif = (gnrc_netif_hdr_t *)pkt->data;
61 
62  if (!(netif->flags & BCAST) && (netif->dst_l2addr_len != 2)) {
63  return -EINVAL;
64  }
65 
66  nrfmin->len = gnrc_pkt_len(pkt->next) + NRFMIN_HDR_LEN;
67  if (netif->flags & BCAST) {
68  nrfmin->dst_addr = NRFMIN_ADDR_BCAST;
69  }
70  else {
71  memcpy(&nrfmin->dst_addr, gnrc_netif_hdr_get_dst_addr(netif), 2);
72  }
73  nrfmin->src_addr = nrfmin_get_addr();
74  if (pkt->next) {
75  nrfmin->proto = (uint8_t)pkt->next->type;
76  }
77  else {
78  nrfmin->proto = 0;
79  }
80 
81  return 0;
82 }

84 static int gnrc_nrfmin_send(gnrc_netdev2_t *dev, gnrc_pktsnip_t *pkt)
85 {
86  int res;
87  struct iovec *vec;
88  size_t vec_len;
89  gnrc_pktsnip_t *vec_snip;
90  nrfmin_hdr_t nrfmin_hdr;
91 
92  assert(pkt);
93 
94  if (pkt->type != GNRC_NETTYPE_NETIF) {
95  DEBUG("[nrfmin_gnrc] send: first header is not generic netif header\n");
96  return -EBADMSG;
97  }
98 
99  /* build the nrfmin header from the generic netif header */
100  res = hdr_netif_to_nrfmin(&nrfmin_hdr, pkt);
101  if (res < 0) {
102  DEBUG("[nrfmin_gnrc] send: failed to build nrfmin header\n");
103  gnrc_pktbuf_release(pkt);
104  return res;
105  }
106 
107  /* create iovec of data */
108  vec_snip = gnrc_pktbuf_get_iovec(pkt, &vec_len);
109  if (vec_snip == NULL) {
110  DEBUG("[nrfmin_gnrc] send: failed to create IO vector\n");
111  gnrc_pktbuf_release(pkt);
112  return -ENOBUFS;
113  }
114 
115  /* link first entry of the vector to the nrfmin header */
116  vec = (struct iovec *)vec_snip->data;
117  vec[0].iov_base = &nrfmin_hdr;
118  vec[0].iov_len = NRFMIN_HDR_LEN;
119 
120  /* and finally send out the data and release the packet */
121  res = dev->dev->driver->send(dev->dev, vec, vec_len);
122  gnrc_pktbuf_release(vec_snip);
123 
124  return res;
125 }

127 static gnrc_pktsnip_t *gnrc_nrfmin_recv(gnrc_netdev2_t *dev)
128 {
129  int pktsize;
130  nrfmin_hdr_t *nrfmin;
131  gnrc_netif_hdr_t *netif;
132  gnrc_pktsnip_t *pkt_snip;
133  gnrc_pktsnip_t *hdr_snip;
134  gnrc_pktsnip_t *netif_snip;
135 
136  /* get the size of the new packet */
137  pktsize = nrfmin_dev.driver->recv(NULL, NULL, 0, NULL);
138  if (pktsize <= 0) {
139  DEBUG("[nrfmin_gnrc] recv: error: tried to read empty packet\n");
140  return NULL;
141  }
142 
143  /* allocate space in the packet buffer */
144  pkt_snip = gnrc_pktbuf_add(NULL, NULL, pktsize, GNRC_NETTYPE_UNDEF);
145  if (pkt_snip == NULL) {
146  DEBUG("[nrfmin_gnrc] recv: unable to allocate pktsnip\n");
147  return NULL;
148  }
149 
150  /* read the incoming data into the packet buffer */
151  nrfmin_dev.driver->recv(NULL, pkt_snip->data, pktsize, NULL);
152 
153  /* now we mark the nrfmin header */
154  hdr_snip = gnrc_pktbuf_mark(pkt_snip, NRFMIN_HDR_LEN, GNRC_NETTYPE_UNDEF);
155  if (hdr_snip == NULL) {
156  DEBUG("[nrfmin_gnrc] recv: unable to mark the nrfmin header\n");
157  gnrc_pktbuf_release(pkt_snip);
158  return NULL;
159  }
160 
161  /* allocate the generic netif header and populate it with data from the
162  nrfmin header */
163  nrfmin = (nrfmin_hdr_t *)hdr_snip->data;
164  netif_snip = gnrc_netif_hdr_build((uint8_t *)&nrfmin->src_addr, 2,
165  (uint8_t *)&nrfmin->dst_addr, 2);
166  if (netif_snip == NULL) {
167  DEBUG("[nrfmin_gnrc] recv: unable to allocate netif header\n");
168  gnrc_pktbuf_release(pkt_snip);
169  return NULL;
170  }
171 
172  netif = (gnrc_netif_hdr_t *)netif_snip->data;
173  if (nrfmin->dst_addr == NRFMIN_ADDR_BCAST) {
174  netif->flags |= GNRC_NETIF_HDR_FLAGS_BROADCAST;
175  }
176  netif->lqi = 0;
177  netif->rssi = 0;
178  netif->if_pid = plug.pid;
179  pkt_snip->type = nrfmin->proto;
180 
181  /* finally: remove the nrfmin header and append the netif header */
182  gnrc_pktbuf_remove_snip(pkt_snip, hdr_snip);
183  LL_APPEND(pkt_snip, netif_snip);
184 
185  return pkt_snip;
186 }
