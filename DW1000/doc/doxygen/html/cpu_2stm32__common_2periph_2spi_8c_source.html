91 int spi_init_cs(spi_t bus, spi_cs_t cs)
92 {
93  if (bus >= SPI_NUMOF) {
94  return SPI_NODEV;
95  }
96  if (cs == SPI_CS_UNDEF ||
97  (((cs & SPI_HWCS_MASK) == SPI_HWCS_MASK) && (cs & ~(SPI_HWCS_MASK)))) {
98  return SPI_NOCS;
99  }
100 
101  if (cs == SPI_HWCS_MASK) {
102  if (spi_config[bus].cs_pin == GPIO_UNDEF) {
103  return SPI_NOCS;
104  }
105 #ifdef CPU_FAM_STM32F1
106  gpio_init_af(spi_config[bus].cs_pin, GPIO_AF_OUT_PP);
107 #else
108  gpio_init(spi_config[bus].cs_pin, GPIO_OUT);
109  gpio_init_af(spi_config[bus].cs_pin, spi_config[bus].af);
110 #endif
111  }
112  else {
113  gpio_init((gpio_t)cs, GPIO_OUT);
114  gpio_set((gpio_t)cs);
115  }
116 
117  return SPI_OK;
118 }

120 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
121 {
122  #ifdef RIOT_TREK_DW1000_APP /* The maximum SPI clock frequency is defined as 20MHz */
123  assert((clk >= SPI_CLK_100KHZ) && (clk <= SPI_CLK_18MHZ));
124  #else
125  assert((clk >= SPI_CLK_100KHZ) && (clk <= SPI_CLK_10MHZ));
126  #endif
127 
128  /* lock bus */
129  mutex_lock(&locks[bus]);
130  /* enable SPI device clock */
131  periph_clk_en(spi_config[bus].apbbus, spi_config[bus].rccmask);
132  /* enable device */
133  uint8_t br = spi_divtable[spi_config[bus].apbbus][clk];
134  dev(bus)->CR1 = ((br << BR_SHIFT) | mode | SPI_CR1_MSTR);
135  if (cs != SPI_HWCS_MASK) {
136  dev(bus)->CR1 |= (SPI_CR1_SSM | SPI_CR1_SSI);
137  }
138  else {
139  dev(bus)->CR2 |= (SPI_CR2_SSOE);
140  }
141 
142  return SPI_OK;
143 }

150  periph_clk_dis(spi_config[bus].apbbus, spi_config[bus].rccmask);
151  mutex_unlock(&locks[bus]);
152 }

154 void spi_transfer_bytes(spi_t bus, spi_cs_t cs, bool cont,
155  const void *out, void *in, size_t len)
156 {
157  uint8_t *inbuf = (uint8_t *)in;
158  uint8_t *outbuf = (uint8_t *)out;
159 
160  /* make sure at least one input or one output buffer is given */
161  assert(outbuf || inbuf);
162 
163  /* we need to recast the data register to uint_8 to force 8-bit access */
164  volatile uint8_t *DR = (volatile uint8_t*)&(dev(bus)->DR);
165 
166  /* active the given chip select line */
167  dev(bus)->CR1 |= (SPI_CR1_SPE); /* this pulls the HW CS line low */
168  if ((cs != SPI_HWCS_MASK) && (cs != SPI_CS_UNDEF)) {
169  gpio_clear((gpio_t)cs);
170  }
171 
172  /* transfer data, use shortpath if only sending data */
173  if (!inbuf) {
174  for (size_t i = 0; i < len; i++) {
175  while (!(dev(bus)->SR & SPI_SR_TXE));
176  *DR = outbuf[i];
177  }
178  /* wait until everything is finished and empty the receive buffer */
179  while (dev(bus)->SR & SPI_SR_BSY) {}
180  while (dev(bus)->SR & SPI_SR_RXNE) {
181  dev(bus)->DR; /* we might just read 2 bytes at once here */
182  }
183  }
184  else if (!outbuf) {
185  for (size_t i = 0; i < len; i++) {
186  while (!(dev(bus)->SR & SPI_SR_TXE));
187  *DR = 0;
188  while (!(dev(bus)->SR & SPI_SR_RXNE));
189  inbuf[i] = *DR;
190  }
191  }
192  else {
193  for (size_t i = 0; i < len; i++) {
194  while (!(dev(bus)->SR & SPI_SR_TXE));
195  *DR = outbuf[i];
196  while (!(dev(bus)->SR & SPI_SR_RXNE));
197  inbuf[i] = *DR;
198  }
199  }
200 
201  /* make sure the transfer is completed before continuing, see reference
202  * manual(s) -> section 'Disabling the SPI' */
203  while (!(dev(bus)->SR & SPI_SR_TXE)) {}
204  while (dev(bus)->SR & SPI_SR_BSY) {}
205 
206  /* release the chip select if not specified differently */
207  if ((!cont) && (cs != SPI_CS_UNDEF)) {
208  dev(bus)->CR1 &= ~(SPI_CR1_SPE); /* pull HW CS line high */
209  if (cs != SPI_HWCS_MASK) {
210  gpio_set((gpio_t)cs);
211  }
212  }
213 }
