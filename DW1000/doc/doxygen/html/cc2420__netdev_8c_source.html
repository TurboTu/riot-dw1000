100 static int _init(netdev2_t *netdev)
101 {
102  cc2420_t *dev = (cc2420_t *)netdev;
103 
104  uint16_t reg;
105 
106  /* initialize power and reset pins -> put the device into reset state */
107  gpio_init(dev->params.pin_reset, GPIO_OUT);
108  gpio_set(dev->params.pin_reset);
109  gpio_init(dev->params.pin_vrefen, GPIO_OUT);
110  gpio_clear(dev->params.pin_vrefen);
111 
112  /* initialize the input lines */
113  gpio_init(dev->params.pin_cca, GPIO_IN);
114  gpio_init(dev->params.pin_sfd, GPIO_IN);
115  gpio_init(dev->params.pin_fifo, GPIO_IN);
116  gpio_init_int(dev->params.pin_fifop, GPIO_IN, GPIO_RISING, _irq_handler, dev);
117 
118  /* initialize the chip select line and the SPI bus */
119  spi_init_cs(dev->params.spi, dev->params.pin_cs);
120 
121  /* power on and toggle reset */
122  gpio_set(dev->params.pin_vrefen);
123  gpio_clear(dev->params.pin_reset);
124  xtimer_usleep(CC2420_RESET_DELAY);
125  gpio_set(dev->params.pin_reset);
126 
127  /* test the connection to the device by reading MANFIDL register */
128  reg = cc2420_reg_read(dev, CC2420_REG_MANFIDL);
129  if (reg != CC2420_MANFIDL_VAL) {
130  DEBUG("cc2420: init: unable to communicate with device\n");
131  return -1;
132  }
133 
134  /* turn on the oscillator and wait for it to be stable */
135  cc2420_en_xosc(dev);
136  if (!(cc2420_status(dev) & CC2420_STATUS_XOSC_STABLE)) {
137  DEBUG("cc2420: init: oscillator did not stabilize\n");
138  return -1;
139  }
140 
141 #ifdef MODULE_NETSTATS_L2
142  memset(&netdev->stats, 0, sizeof(netstats_t));
143 #endif
144 
145  return cc2420_init((cc2420_t *)dev);
146 }

165 static int _get(netdev2_t *netdev, netopt_t opt, void *val, size_t max_len)
166 {
167  if (netdev == NULL) {
168  return -ENODEV;
169  }
170 
171  cc2420_t *dev = (cc2420_t *)netdev;
172 
173  int ext = netdev2_ieee802154_get(&dev->netdev, opt, val, max_len);
174  if (ext > 0) {
175  return ext;
176  }
177 
178  switch (opt) {
179 
180  case NETOPT_ADDRESS:
181  assert(max_len >= sizeof(uint16_t));
182  cc2420_get_addr_short(dev, val);
183  return sizeof(uint16_t);
184 
185  case NETOPT_ADDRESS_LONG:
186  assert(max_len >= 8);
187  cc2420_get_addr_long(dev, val);
188  return 8;
189 
190  case NETOPT_NID:
191  assert(max_len >= sizeof(uint16_t));
192  return w_u16(val, cc2420_get_pan(dev));
193 
194  case NETOPT_CHANNEL:
195  assert(max_len >= sizeof(uint16_t));
196  return w_u16(val, cc2420_get_chan(dev));
197 
198  case NETOPT_TX_POWER:
199  assert(max_len >= sizeof(int16_t));
200  return w_i16(val, cc2420_get_txpower(dev));
201 
202  case NETOPT_STATE:
203  assert(max_len >= sizeof(netopt_state_t));
204  *((netopt_state_t *)val) = cc2420_get_state(dev);
205  return sizeof(netopt_state_t);
206 
207  case NETOPT_IS_CHANNEL_CLR:
208  return opt_state(val, cc2420_cca(dev));
209 
210  case NETOPT_AUTOACK:
211  return opt_state(val, (dev->options & CC2420_OPT_AUTOACK));
212 
213  case NETOPT_CSMA:
214  return opt_state(val, (dev->options & CC2420_OPT_CSMA));
215 
216  case NETOPT_PRELOADING:
217  return opt_state(val, (dev->options & CC2420_OPT_PRELOADING));
218 
219  case NETOPT_PROMISCUOUSMODE:
220  return opt_state(val, (dev->options & CC2420_OPT_PROMISCUOUS));
221 
222  case NETOPT_RX_START_IRQ:
223  return opt_state(val, (dev->options & CC2420_OPT_TELL_RX_START));
224 
225  case NETOPT_RX_END_IRQ:
226  return opt_state(val, (dev->options & CC2420_OPT_TELL_TX_END));
227 
228  case NETOPT_TX_START_IRQ:
229  return opt_state(val, (dev->options & CC2420_OPT_TELL_RX_START));
230 
231  case NETOPT_TX_END_IRQ:
232  return opt_state(val, (dev->options & CC2420_OPT_TELL_RX_END));
233 
234  default:
235  return -ENOTSUP;
236  }
237 }

239 static int _set(netdev2_t *netdev, netopt_t opt, void *val, size_t val_len)
240 {
241  if (netdev == NULL) {
242  return -ENODEV;
243  }
244 
245  cc2420_t *dev = (cc2420_t *)netdev;
246 
247  int ext = netdev2_ieee802154_set(&dev->netdev, opt, val, val_len);
248 
249  switch (opt) {
250  case NETOPT_ADDRESS:
251  assert(val_len == 2);
252  cc2420_set_addr_short(dev, (uint8_t *)val);
253  return 2;
254 
255  case NETOPT_ADDRESS_LONG:
256  assert(val_len == 8);
257  cc2420_set_addr_long(dev, (uint8_t *)val);
258  return 8;
259 
260  case NETOPT_NID:
261  assert(val_len == sizeof(uint16_t));
262  cc2420_set_pan(dev, to_u16(val));
263  return sizeof(uint16_t);
264 
265  case NETOPT_CHANNEL:
266  assert(val_len == sizeof(uint16_t));
267  return cc2420_set_chan(dev, to_u16(val));
268 
269  case NETOPT_TX_POWER:
270  assert(val_len == sizeof(int16_t));
271  cc2420_set_txpower(dev, to_i16(val));
272  return sizeof(int16_t);
273 
274  case NETOPT_STATE:
275  assert(val_len == sizeof(netopt_state_t));
276  return cc2420_set_state(dev, *((netopt_state_t *)val));
277 
278  case NETOPT_AUTOACK:
279  return cc2420_set_option(dev, CC2420_OPT_AUTOACK, to_bool(val));
280 
281  case NETOPT_CSMA:
282  return cc2420_set_option(dev, CC2420_OPT_CSMA, to_bool(val));
283 
284  case NETOPT_PRELOADING:
285  return cc2420_set_option(dev, CC2420_OPT_PRELOADING, to_bool(val));
286 
287  case NETOPT_PROMISCUOUSMODE:
288  return cc2420_set_option(dev, CC2420_OPT_PROMISCUOUS, to_bool(val));
289 
290  case NETOPT_RX_START_IRQ:
291  return cc2420_set_option(dev, CC2420_OPT_TELL_RX_START, to_bool(val));
292 
293  case NETOPT_RX_END_IRQ:
294  return cc2420_set_option(dev, CC2420_OPT_TELL_RX_END, to_bool(val));
295 
296  case NETOPT_TX_START_IRQ:
297  return cc2420_set_option(dev, CC2420_OPT_TELL_TX_START, to_bool(val));
298 
299  case NETOPT_TX_END_IRQ:
300  return cc2420_set_option(dev, CC2420_OPT_TELL_TX_END, to_bool(val));
301

303  return ext;
304  }
305 
306  return 0;
