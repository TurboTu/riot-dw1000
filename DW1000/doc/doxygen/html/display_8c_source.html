164 char *itoa(uint32_t n, uint8_t digits, uint8_t blanks)
165 {
166  uint8_t digits1 = digits;
167 
168  /* Preset result string */
169  memcpy(itoa_str, "0000000", 7);
170 
171  /* Return empty string if number of digits is invalid (valid range for digits: 1-7) */
172  if ((digits == 0) || (digits > 7)) {
173  return (itoa_str);
174  }
175 
176  /* Numbers 0 .. 180 can be copied from itoa_conversion_table without conversion */
177  if (n <= 180) {
178  if (digits >= 3) {
179  memcpy(itoa_str + (digits - 3), itoa_conversion_table[n], 3);
180  }
181  /* digits == 1 || 2 */
182  else {
183  memcpy(itoa_str, itoa_conversion_table[n] + (3 - digits), digits);
184  }
185  }
186  /* For n > 180 need to calculate string content */
187  else {
188  /* Calculate digits from least to most significant number */
189  do {
190  itoa_str[digits - 1] = n % 10 + '0';
191  n /= 10;
192  }
193  while (--digits > 0);
194  }
195 
196  /* Remove specified number of leading '0', always keep last one */
197  uint8_t i = 0;
198  while ((i < digits1 - 1) && (itoa_str[i] == '0')) {
199  if (blanks > 0) {
200  /* Convert only specified number of leading '0' */
201  itoa_str[i] = ' ';
202  blanks--;
203  }
204 
205  i++;
206  }
207 
208  return (itoa_str);
209 }

238 void display_char(uint8_t segment, char chr, uint8_t mode)
239 {
240  /* Write to single 7-segment character */
241  if ((segment >= LCD_SEG_L1_3) && (segment <= LCD_SEG_L2_DP)) {
242  uint8_t bits; /* Bits to write */
243 
244  /* Get LCD memory address for segment from table */
245  uint8_t *lcdmem = (uint8_t *)segments_lcdmem[segment];
246 
247  /* Get bitmask for character from table */
248  uint8_t bitmask = segments_bitmask[segment];
249 
250  /* Get bits from font set */
251  if ((chr >= 0x30) && (chr <= 0x5A)) {
252  /* Use font set */
253  bits = lcd_font[chr - 0x30];
254  }
255  else if (chr == 0x2D) {
256  /* '-' not in font set */
257  bits = BIT1;
258  }
259  else {
260  /* Other characters map to ' ' (blank) */
261  bits = 0;
262  }
263 
264  /* When addressing LINE2 7-segment characters need to swap high- and low-nibble, */
265  /* because LCD COM/SEG assignment is mirrored against LINE1 */
266  if (segment >= LCD_SEG_L2_5) {
267  uint8_t bits1 = ((bits << 4) & 0xF0) | ((bits >> 4) & 0x0F);
268  bits = bits1;
269 
270  /* When addressing LCD_SEG_L2_5, need to convert ASCII '1' and 'L' to 1 bit, */
271  /* because LCD COM/SEG assignment is special for this incomplete character */
272  if (segment == LCD_SEG_L2_5) {
273  if ((chr == '1') || (chr == 'L')) {
274  bits = BIT7;
275  }
276  }
277  }
278 
279  /* Physically write to LCD memory */
280  write_lcd_mem(lcdmem, bits, bitmask, mode);
281  }
