42 int mma8x5x_init(mma8x5x_t *dev, const mma8x5x_params_t *params)
43 {
44  uint8_t reg;
45 
46  assert(dev && params);
47 
48  /* write device descriptor */
49  memcpy(dev, params, sizeof(mma8x5x_params_t));
50 
51  /* initialize the I2C bus */
52  i2c_acquire(BUS);
53  if (i2c_init_master(BUS, I2C_SPEED) < 0) {
54  i2c_release(BUS);
55  DEBUG("[mma8x5x] init - error: unable to initialize I2C bus\n");
56  return MMA8X5X_NOI2C;
57  }
58 
59  /* test if the target device responds */
60  i2c_read_reg(BUS, ADDR, MMA8X5X_WHO_AM_I, &reg);
61  if (reg != dev->params.type) {
62  i2c_release(BUS);
63  DEBUG("[mma8x5x] init - error: invalid WHO_AM_I value [0x%02x]\n",
64  (int)reg);
65  return MMA8X5X_NODEV;
66  }
67 
68  /* reset the device */
69  i2c_write_reg(BUS, ADDR, MMA8X5X_CTRL_REG2, MMA8X5X_CTRL_REG2_RST);
70  do {
71  i2c_read_reg(BUS, ADDR, MMA8X5X_CTRL_REG2, &reg);
72  } while (reg & MMA8X5X_CTRL_REG2_RST);
73  /* configure the user offset */
74  i2c_write_regs(BUS, ADDR, MMA8X5X_OFF_X, dev->params.offset, 3);
75  /* configure range, rate, and activate the device */
76  reg = (dev->params.range & MMA8X5X_XYZ_DATA_CFG_FS_MASK);
77  i2c_write_reg(BUS, ADDR, MMA8X5X_XYZ_DATA_CFG, reg);
78  reg = ((dev->params.rate & MMA8X5X_CTRL_REG1_DR_MASK) |
79  MMA8X5X_CTRL_REG1_ACTIVE);
80  i2c_write_reg(BUS, ADDR, MMA8X5X_CTRL_REG1, reg);
81 
82  /* finally release the bus */
83  i2c_release(BUS);
84 
85  DEBUG("[mma8x5x] init: successful\n");
86 
87  return MMA8X5X_OK;
88 }

159 {

162  assert(dev);
163 
164  i2c_acquire(BUS);

168  data->x = ((int16_t)(buf[1] << 8 | buf[2])) / (16 >> dev->params.range);
169  data->y = ((int16_t)(buf[3] << 8 | buf[4])) / (16 >> dev->params.range);
