62 int gpio_init(gpio_t pin, gpio_mode_t mode)
63 {
64  GPIO_TypeDef *port = _port(pin);
65  int pin_num = _pin_num(pin);
66 
67  /* enable clock */
68  periph_clk_en(AHB, (RCC_AHBENR_GPIOAEN << _port_num(pin)));
69 
70  /* set mode */
71  port->MODER &= ~(0x3 << (2 * pin_num));
72  port->MODER |= ((mode & 0x3) << (2 * pin_num));
73  /* set pull resistor configuration */
74  port->PUPDR &= ~(0x3 << (2 * pin_num));
75  port->PUPDR |= (((mode >> 2) & 0x3) << (2 * pin_num));
76  /* set output mode */
77  port->OTYPER &= ~(1 << pin_num);
78  port->OTYPER |= (((mode >> 4) & 0x1) << pin_num);
79  /* finally set pin speed to maximum and reset output */
80  port->OSPEEDR |= (3 << (2 * pin_num));
81  port->BRR = (1 << pin_num);
82 
83  return 0;
84 }

86 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
87  gpio_cb_t cb, void *arg)
88 {

90  int port_num = _port_num(pin);
91 
92  /* set callback */
93  isr_ctx[pin_num].cb = cb;
94  isr_ctx[pin_num].arg = arg;
95 
96  /* enable clock of the SYSCFG module for EXTI configuration */
97  periph_clk_en(APB2, RCC_APB2ENR_SYSCFGCOMPEN);
98 
99  /* initialize pin as input */
100  gpio_init(pin, mode);
101 
102  /* enable global pin interrupt */
103  if (pin_num < 2) {
104  NVIC_EnableIRQ(EXTI0_1_IRQn);
105  }
106  else if (pin_num < 4) {
107  NVIC_EnableIRQ(EXTI2_3_IRQn);
108  }
109  else {
110  NVIC_EnableIRQ(EXTI4_15_IRQn);
111  }
112  /* configure the active flank */
113  EXTI->RTSR &= ~(1 << pin_num);
114  EXTI->RTSR |= ((flank & 0x1) << pin_num);
115  EXTI->FTSR &= ~(1 << pin_num);
116  EXTI->FTSR |= ((flank >> 1) << pin_num);
117  /* enable specific pin as exti sources */
118  SYSCFG->EXTICR[pin_num >> 2] &= ~(0xf << ((pin_num & 0x03) * 4));
119  SYSCFG->EXTICR[pin_num >> 2] |= (port_num << ((pin_num & 0x03) * 4));
120 
121  /* clear any pending requests */
122  EXTI->PR = (1 << pin);
123  /* unmask the pins interrupt channel */
124  EXTI->IMR |= (1 << pin);
125 
126  return 0;
127 }

199 void isr_exti(void)
200 {
201  /* only generate interrupts against lines which have their IMR set */
202  uint32_t pending_isr = (EXTI->PR & EXTI->IMR);
203  for (size_t i = 0; i < EXTI_NUMOF; i++) {
204  if (pending_isr & (1 << i)) {
205  EXTI->PR = (1 << i); /* clear by writing a 1 */
206  isr_ctx[i].cb(isr_ctx[i].arg);
