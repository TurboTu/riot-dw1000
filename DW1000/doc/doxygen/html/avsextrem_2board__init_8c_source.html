39 void init_clks1(void)
40 {
41  // Disconnect PLL
42  PLLCON &= ~0x0002;
43  pllfeed();
44 
45  while (PLLSTAT & BIT25); /* wait until PLL is disconnected before
46  * disabling - deadlock otherwise */
47 
48  // Disable PLL
49  PLLCON &= ~0x0001;
50  pllfeed();
51 
52  while (PLLSTAT & BIT24); // wait until PLL is disabled
53 
54  SCS |= 0x10; // main OSC between 15MHz and 24MHz (more stable in tests)
55  SCS |= 0x20; // Enable main OSC
56 
57  while (!(SCS & 0x40)); // Wait until main OSC is usable
58 
59  /* select main OSC, 16MHz, as the PLL clock source */
60  CLKSRCSEL = 0x0001;
61 
62  // Setting Multiplier and Divider values
63  PLLCFG = 0x0008; // M=9 N=1 Fcco = 288 MHz
64  pllfeed();
65 
66  // Enabling the PLL */
67  PLLCON = 0x0001;
68  pllfeed();
69 
70  /* Set clock divider to 4 (value+1) */
71  CCLKCFG = CL_CPU_DIV - 1; // Fcpu = 72 MHz
72 
73 #if USE_USB
74  USBCLKCFG = USBCLKDivValue; /* usbclk = 288 MHz/6 = 48 MHz */
75 #endif
76 }
