198 template <class Tuple>
199 void* thread_proxy(void* vp) {
200  { // without this scope, the objects here are not cleaned up corrctly
201  std::unique_ptr<Tuple> p(static_cast<Tuple*>(vp));
202  auto tmp = std::get<0>(*p);
203  std::unique_ptr<thread_data, thread_data_deleter> data{tmp};
204  // create indices for the arguments, 0 is thread_data and 1 is the function
205  auto indices = detail::get_indices<std::tuple_size<Tuple>::value, 2>();
206  try {
207  detail::apply_args(std::get<1>(*p), indices, *p);
208  }
209  catch (...) {
210  // nop
211  }
212  if (data->joining_thread != thread_uninitialized) {
213  thread_wakeup(data->joining_thread);
214  }
215  }
216  // some riot cleanup code
217  sched_task_exit();
218  return nullptr;
219 }

222 thread::thread(F&& f, Args&&... args)
223  : m_data{new thread_data} {
224  using namespace std;
225  using func_and_args = tuple
226  <thread_data*, typename decay<F>::type, typename decay<Args>::type...>;
227  std::unique_ptr<func_and_args> p(
228  new func_and_args(m_data.get(), forward<F>(f), forward<Args>(args)...));
229  m_handle = thread_create(
230  m_data->stack, stack_size, THREAD_PRIORITY_MAIN - 1, 0,
231  &thread_proxy<func_and_args>, p.get(), "riot_cpp_thread");
232  if (m_handle >= 0) {
233  p.release();
234  } else {
