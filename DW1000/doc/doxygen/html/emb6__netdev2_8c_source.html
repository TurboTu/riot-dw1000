83 static void _event_cb(netdev2_t *dev, netdev2_event_t event)
84 {
85  if (event == NETDEV2_EVENT_ISR) {
86  /* EVENT_TYPE_PCK_LL is supposed to be used by drivers, so use it
87  * (though NETDEV2_EVENT_ISR technically doesn't only signify
88  * incoming packets) */
89  evproc_putEvent(E_EVPROC_HEAD, EVENT_TYPE_PCK_LL, NULL);
90  }
91  else {
92  switch (event) {
93  case NETDEV2_EVENT_RX_COMPLETE: {
94  _get_recv_pkt();
95  }
96  break;
97  default:
98  break;
99  }
100  }
101 }

120 static int8_t _netdev2_init(s_ns_t *p_ns)
121 {
122  if ((_dev != NULL) && (p_ns != NULL) && (p_ns->lmac != NULL)) {
123  _dev->event_callback = _event_cb;
124  _dev->driver->get(_dev, NETOPT_ADDRESS_LONG, &mac_phy_config.mac_address,
125  sizeof(mac_phy_config.mac_address));
126  memcpy(&uip_lladdr, mac_phy_config.mac_address,
127  sizeof(mac_phy_config.mac_address));
128  _dev->driver->get(_dev, NETOPT_NID, &mac_phy_config.pan_id,
129  sizeof(mac_phy_config.pan_id));
130  linkaddr_set_node_addr((linkaddr_t *)&uip_lladdr);
131  _lowmac = p_ns->lmac;
132  evproc_regCallback(EVENT_TYPE_PCK_LL, _emb6_netdev2_callback);
133  return 1;
134  }
135  return 0;
136 }

138 static int8_t _netdev2_send(const void *pr_payload, uint8_t c_len)
139 {
140  if (_dev != NULL) {
141  const struct iovec vector = {
142  .iov_base = (void *)pr_payload,
143  .iov_len = c_len
144  };
145  if (_dev->driver->send(_dev, &vector, 1) < 0) {
146  DEBUG("Error on send\n");
147  return RADIO_TX_ERR;
148  }
149  DEBUG("Packet of length %u was transmitted\n", (unsigned)c_len);
150  return RADIO_TX_OK;
151  }
152  DEBUG("Device was not initialized\n");
153  return RADIO_TX_ERR;
154 }
