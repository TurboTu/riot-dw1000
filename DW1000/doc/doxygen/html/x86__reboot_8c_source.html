59 void NORETURN x86_kbc_reboot(void)
60 {
61  /* First load an empty IDT. Any interrupt will cause a tripple fault. */
62  x86_load_empty_idt();
63 
64  while (1) {
65  if (fail_violently) {
66  __asm__ volatile ("int3"); /* Cause a tripple fault. Won't return. */
67  }
68  fail_violently = true;
69 
70  for (unsigned i = 0; i < 0x100; ++i) {
71  for (unsigned j = 0; j < 0x10000; ++j) {
72  uint8_t c = inb(KBC_STATUS);
73  if (c & KBC_OUTPUT_FULL) {
74  (void) inb(KBC_DATA);
75  }
76  else if (!(c & KBC_OUTPUT_FULL)) {
77  outb(KBC_COMMAND, KBC_RESET);
78  break;
79  }
80  }
81  }
82 
83  __asm__ volatile ("int3"); /* Cause a tripple fault. Won't return. */
84  }
85 }
