35 void gnrc_ndp_host_init(gnrc_ipv6_netif_t *iface)
36 {
37  uint32_t interval = random_uint32_range(0, GNRC_NDP_MAX_RTR_SOL_DELAY * US_PER_SEC);
38  mutex_lock(&iface->mutex);
39  iface->rtr_sol_count = GNRC_NDP_MAX_RTR_SOL_NUMOF;
40  DEBUG("ndp host: delayed initial router solicitation by %" PRIu32 " usec.\n", interval);
41  _reschedule_rtr_sol(iface, interval);
42  mutex_unlock(&iface->mutex);
43 }

45 void gnrc_ndp_host_retrans_rtr_sol(gnrc_ipv6_netif_t *iface)
46 {
47  mutex_lock(&iface->mutex);
48  if (iface->rtr_sol_count > 1) { /* regard off-by-one error */
49  DEBUG("ndp hst: retransmit rtr sol in %d sec\n", GNRC_NDP_MAX_RTR_SOL_INT);
50  iface->rtr_sol_count--;
51  _reschedule_rtr_sol(iface, GNRC_NDP_MAX_RTR_SOL_INT * US_PER_SEC);
52  }
53  mutex_unlock(&iface->mutex);
54  gnrc_ndp_internal_send_rtr_sol(iface->pid, NULL);
55 }
