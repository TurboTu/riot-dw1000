42 int bmp180_init(bmp180_t *dev, i2c_t i2c, uint8_t mode)
43 {
44  dev->i2c_dev = i2c;
45 
46  /* Clamp oversampling mode */
47  if (mode > BMP180_ULTRAHIGHRES) {
48  mode = BMP180_ULTRAHIGHRES;
49  }
50 
51  /* Setting oversampling mode */
52  dev->oversampling = mode;
53 
54  /* Initialize I2C interface */
55  if (i2c_init_master(dev->i2c_dev, I2C_SPEED_NORMAL)) {
56  DEBUG("[Error] I2C device not enabled\n");
57  return -1;
58  }
59 
60  /* Acquire exclusive access */
61  i2c_acquire(dev->i2c_dev);
62 
63  /* Check sensor ID */
64  uint8_t checkid;
65  i2c_read_reg(dev->i2c_dev, BMP180_ADDR, BMP180_REGISTER_ID, &checkid);
66  if (checkid != 0x55) {
67  DEBUG("[Error] Wrong device ID\n");
68  i2c_release(dev->i2c_dev);
69  return -1;
70  }
71 
72  /* adding delay before reading calibration values to avoid timing issues */
73  xtimer_usleep(BMP180_ULTRALOWPOWER_DELAY);
74 
75  uint8_t buffer[22] = {0};
76  /* Read calibration values, using contiguous register addresses */
77  if (i2c_read_regs(dev->i2c_dev, BMP180_ADDR, BMP180_CALIBRATION_AC1, buffer, 22) < 0) {
78  DEBUG("[Error] Cannot read calibration registers.\n");
79  i2c_release(dev->i2c_dev);
80  return -1;
81  }
82  dev->calibration.ac1 = (int16_t)(buffer[0] << 8) | buffer[1];
83  dev->calibration.ac2 = (int16_t)(buffer[2] << 8) | buffer[3];
84  dev->calibration.ac3 = (int16_t)(buffer[4] << 8) | buffer[4];
85  dev->calibration.ac4 = (uint16_t)(buffer[6] << 8) | buffer[7];
86  dev->calibration.ac5 = (uint16_t)(buffer[8] << 8) | buffer[9];
87  dev->calibration.ac6 = (uint16_t)(buffer[10] << 8) | buffer[11];
88  dev->calibration.b1 = (int16_t)(buffer[12] << 8) | buffer[13];
89  dev->calibration.b2 = (int16_t)(buffer[14] << 8) | buffer[15];
90  dev->calibration.mb = (int16_t)(buffer[16] << 8) | buffer[17];
91  dev->calibration.mc = (int16_t)(buffer[18] << 8) | buffer[19];
92  dev->calibration.md = (int16_t)(buffer[20] << 8) | buffer[21];
93 
94  /* Release I2C device */
95  i2c_release(dev->i2c_dev);
96 
97  DEBUG("AC1: %i\n", (int)dev->calibration.ac1);
98  DEBUG("AC2: %i\n", (int)dev->calibration.ac2);
99  DEBUG("AC3: %i\n", (int)dev->calibration.ac3);
100  DEBUG("AC4: %i\n", (int)dev->calibration.ac4);
101  DEBUG("AC5: %i\n", (int)dev->calibration.ac5);
102  DEBUG("AC6: %i\n", (int)dev->calibration.ac6);
103  DEBUG("B1: %i\n", (int)dev->calibration.b1);
104  DEBUG("B2: %i\n", (int)dev->calibration.b2);
105  DEBUG("MB: %i\n", (int)dev->calibration.mb);
106  DEBUG("MC: %i\n", (int)dev->calibration.mc);
107  DEBUG("MD: %i\n", (int)dev->calibration.md);
108  return 0;
109 }

111 int bmp180_read_temperature(bmp180_t *dev, int32_t *temperature)
112 {
113  int32_t ut, b5;
114  /* Acquire exclusive access */
115  i2c_acquire(dev->i2c_dev);
116 
117  /* Read uncompensated value */
118  _read_ut(dev, &ut);
119 
120  /* Compute true temperature value following datasheet formulas */
121  _compute_b5(dev, ut, &b5);

130 int bmp180_read_pressure(bmp180_t *dev, int32_t *pressure)
131 {
132  int32_t ut = 0, up = 0, x1, x2, x3, b3, b5, b6, p;
133  uint32_t b4, b7;
134 
135  /* Acquire exclusive access */
136  i2c_acquire(dev->i2c_dev);
137 
138  /* Read uncompensated values: first temperature, second pressure */
139  _read_ut(dev, &ut);
140  _read_up(dev, &up);
141 
142  /* Compute true pressure value following datasheet formulas */
143  _compute_b5(dev, ut, &b5);
144  b6 = b5 - 4000;
145  x1 = ((int32_t)dev->calibration.b2 * ((b6 * b6) >> 12)) >> 11;
146  x2 = ((int32_t)dev->calibration.ac2 * b6) >> 11;
147  x3 = x1 + x2;
148  b3 = ((((int32_t)dev->calibration.ac1*4 + x3) << dev->oversampling) + 2) >> 2;
149  x1 = ((int32_t)dev->calibration.ac3 * b6) >> 13;
150  x2 = ((int32_t)dev->calibration.b1 * (b6 * b6) >> 12) >> 16;
151  x3 = ((x1 + x2) + 2) >> 2;
152  b4 = (int32_t)dev->calibration.ac4 * (uint32_t)(x3+32768) >> 15;
153  b7 = ((uint32_t)up - b3) * (uint32_t)(50000UL >> dev->oversampling);
154  if (b7 < 0x80000000) {
155  p = (b7 * 2) / b4;
156  }
157  else {
158  p = (b7 / b4) * 2;
159  }
160 
161  x1 = (p >> 8) * (p >> 8);
162  x1 = (x1 * 3038) >> 16;
163  x2 = (-7357 * p) >> 16;
164  *pressure = p + ((x1 + x2 + 3791) >> 4);
165 
166  /* release I2C device */
167  i2c_release(dev->i2c_dev);
168 
169  return 0;
170 }

196 static int _read_ut(bmp180_t *dev, int32_t *output)
197 {
198  /* Read UT (Uncompsensated Temperature value) */
199  uint8_t ut[2] = {0};
200  uint8_t control[2] = { BMP180_REGISTER_CONTROL, BMP180_TEMPERATURE_COMMAND };
201  i2c_write_bytes(dev->i2c_dev, BMP180_ADDR, control, 2);
202  xtimer_usleep(BMP180_ULTRALOWPOWER_DELAY);
203  if (i2c_read_regs(dev->i2c_dev, BMP180_ADDR, BMP180_REGISTER_DATA, ut, 2) < 0) {
204  DEBUG("[Error] Cannot read uncompensated temperature.\n");
205  i2c_release(dev->i2c_dev);
206  return -1;
207  }
208  *output = ( ut[0] << 8 ) | ut[1];
209 
210  DEBUG("UT: %i\n", (int)*output);
211 
212  return 0;
213 }

215 static int _read_up(bmp180_t *dev, int32_t *output)
216 {
217  /* Read UP (Uncompsensated Pressure value) */
218  uint8_t up[3] = {0};
219  uint8_t control[2] = { BMP180_REGISTER_CONTROL, BMP180_PRESSURE_COMMAND | (dev->oversampling & 0x3) << 6 };
220  i2c_write_bytes(dev->i2c_dev, BMP180_ADDR, control, 2);
221  switch (dev->oversampling) {
222  case BMP180_ULTRALOWPOWER:
223  xtimer_usleep(BMP180_ULTRALOWPOWER_DELAY);
224  break;
225  case BMP180_STANDARD:
226  xtimer_usleep(BMP180_STANDARD_DELAY);
227  break;
228  case BMP180_HIGHRES:
229  xtimer_usleep(BMP180_HIGHRES_DELAY);
230  break;
231  case BMP180_ULTRAHIGHRES:
232  xtimer_usleep(BMP180_ULTRAHIGHRES_DELAY);
233  break;
234  default:
235  xtimer_usleep(BMP180_ULTRALOWPOWER_DELAY);
236  break;
237  }
238  if (i2c_read_regs(dev->i2c_dev, BMP180_ADDR, BMP180_REGISTER_DATA, up, 3) < 0) {
239  DEBUG("[Error] Cannot read uncompensated pressure.\n");
240  i2c_release(dev->i2c_dev);
241  return -1;
242  }
243 
244  *output = ((up[0] << 16) | (up[1] << 8) | up[2]) >> (8 - dev->oversampling);
245 
246  DEBUG("UP: %i\n", (int)*output);
247 
248  return 0;
249 }

251 static int _compute_b5(bmp180_t *dev, int32_t ut, int32_t *output)
252 {
253  int32_t x1, x2;
254  x1 = (ut - dev->calibration.ac6) * dev->calibration.ac5 >> 15;
255  x2 = (dev->calibration.mc << 11) / (x1 + dev->calibration.md);
256 
257  *output = x1 + x2;
258 
259  return 0;
260 }
