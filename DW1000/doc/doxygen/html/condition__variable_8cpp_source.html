57 void condition_variable::notify_all() noexcept {
58  unsigned old_state = irq_disable();
59  int other_prio = -1;
60  while (true) {
61  priority_queue_node_t* head = priority_queue_remove_head(&m_queue);
62  if (head == NULL) {
63  break;
64  }
65  thread_t* other_thread = (thread_t*)sched_threads[head->data];
66  if (other_thread) {
67  auto max_prio
68  = [](int a, int b) { return (a < 0) ? b : ((a < b) ? a : b); };
69  other_prio = max_prio(other_prio, other_thread->priority);
70  sched_set_status(other_thread, STATUS_PENDING);
71  }
72  head->data = -1u;
73  }
74  irq_restore(old_state);
75  if (other_prio >= 0) {

80 void condition_variable::wait(unique_lock<mutex>& lock) noexcept {
81  priority_queue_node_t n;
82  n.priority = sched_active_thread->priority;
83  n.data = sched_active_pid;
84  n.next = NULL;
85  // the signaling thread may not hold the mutex, the queue is not thread safe
86  unsigned old_state = irq_disable();
87  priority_queue_add(&m_queue, &n);
88  irq_restore(old_state);
89  mutex_unlock_and_sleep(lock.mutex()->native_handle());
90  if (n.data != -1u) {
91  // on signaling n.data is set to -1u
92  // if it isn't set, then the wakeup is either spurious or a timer wakeup
93  old_state = irq_disable();
94  priority_queue_remove(&m_queue, &n);
95  irq_restore(old_state);
96  }
97  mutex_lock(lock.mutex()->native_handle());
98 }
