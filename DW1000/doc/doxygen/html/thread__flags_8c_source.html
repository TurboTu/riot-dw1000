102 inline int __attribute__((always_inline)) thread_flags_wake(thread_t *thread)
103 {
104  unsigned wakeup = 0;
105  thread_flags_t mask = (uint16_t)(unsigned)thread->wait_data;
106  switch(thread->status) {
107  case STATUS_FLAG_BLOCKED_ANY:
108  wakeup = (thread->flags & mask);
109  break;
110  case STATUS_FLAG_BLOCKED_ALL:
111  wakeup = ((thread->flags & mask) == mask);
112  break;
113  }
114 
115  if (wakeup) {
116  DEBUG("_thread_flags_wake(): wakeing up pid %"PRIkernel_pid"\n", thread->pid);
117  sched_set_status(thread, STATUS_RUNNING);
118  }
119 
120  return wakeup;
121 }
