76 int __attribute__((used)) sched_run(void)
77 {
78  sched_context_switch_request = 0;
79 
80  thread_t *active_thread = (thread_t *)sched_active_thread;
81 
82  /* The bitmask in runqueue_bitcache is never empty,
83  * since the threading should not be started before at least the idle thread was started.
84  */
85  int nextrq = bitarithm_lsb(runqueue_bitcache);
86  thread_t *next_thread = container_of(sched_runqueues[nextrq].next->next, thread_t, rq_entry);
87 
88  DEBUG("sched_run: active thread: %" PRIkernel_pid ", next thread: %" PRIkernel_pid "\n",
89  (active_thread == NULL) ? KERNEL_PID_UNDEF : active_thread->pid,
90  next_thread->pid);
91 
92  if (active_thread == next_thread) {
93  DEBUG("sched_run: done, sched_active_thread was not changed.\n");
94  return 0;
95  }
96 
97 #ifdef MODULE_SCHEDSTATISTICS
98  unsigned long time = _xtimer_now();
99 #endif
100 
101  if (active_thread) {
102  if (active_thread->status == STATUS_RUNNING) {
103  active_thread->status = STATUS_PENDING;
104  }
105 
106 #ifdef SCHED_TEST_STACK
107  if (*((uintptr_t *) active_thread->stack_start) != (uintptr_t) active_thread->stack_start) {
108  LOG_WARNING("scheduler(): stack overflow detected, pid=%" PRIkernel_pid "\n", active_thread->pid);
109  }
110 #endif
111 
112 #ifdef MODULE_SCHEDSTATISTICS
113  schedstat *active_stat = &sched_pidlist[active_thread->pid];
114  if (active_stat->laststart) {
115  active_stat->runtime_ticks += time - active_stat->laststart;
116  }
117 #endif
118  }
119 
120 #ifdef MODULE_SCHEDSTATISTICS
121  schedstat *next_stat = &sched_pidlist[next_thread->pid];
122  next_stat->laststart = time;
123  next_stat->schedules++;
124  if (sched_cb) {
125  sched_cb(time, next_thread->pid);
126  }
127 #endif
128 
129  next_thread->status = STATUS_RUNNING;
130  sched_active_pid = next_thread->pid;
131  sched_active_thread = (volatile thread_t *) next_thread;
132 
133 #ifdef MODULE_MPU_STACK_GUARD
134  mpu_configure(
135  1, /* MPU region 1 */
136  (uintptr_t)sched_active_thread->stack_start + 31, /* Base Address (rounded up) */
137  MPU_ATTR(1, AP_RO_RO, 0, 1, 0, 1, MPU_SIZE_32B) /* Attributes and Size */
138  );
139 
140  mpu_enable();
141 #endif
142 
143  DEBUG("sched_run: done, changed sched_active_thread.\n");
144 
145  return 1;
146 }

155 void sched_set_status(thread_t *process, unsigned int status)
156 {
157  if (status >= STATUS_ON_RUNQUEUE) {
158  if (!(process->status >= STATUS_ON_RUNQUEUE)) {
159  DEBUG("sched_set_status: adding thread %" PRIkernel_pid " to runqueue %" PRIu16 ".\n",
160  process->pid, process->priority);
161  clist_rpush(&sched_runqueues[process->priority], &(process->rq_entry));
162  runqueue_bitcache |= 1 << process->priority;
163  }
164  }
165  else {
166  if (process->status >= STATUS_ON_RUNQUEUE) {
167  DEBUG("sched_set_status: removing thread %" PRIkernel_pid " to runqueue %" PRIu16 ".\n",
168  process->pid, process->priority);
169  clist_lpop(&sched_runqueues[process->priority]);
170 
171  if (!sched_runqueues[process->priority].next) {
172  runqueue_bitcache &= ~(1 << process->priority);
173  }
174  }
175  }
176 
177  process->status = status;
178 }

180 void sched_switch(uint16_t other_prio)
181 {
182  thread_t *active_thread = (thread_t *) sched_active_thread;
183  uint16_t current_prio = active_thread->priority;
184  int on_runqueue = (active_thread->status >= STATUS_ON_RUNQUEUE);
185 
186  DEBUG("sched_switch: active pid=%" PRIkernel_pid" prio=%" PRIu16 " on_runqueue=%i "
187  ", other_prio=%" PRIu16 "\n",
188  active_thread->pid, current_prio, on_runqueue, other_prio);
189 
190  if (!on_runqueue || (current_prio > other_prio)) {
191  if (irq_is_in()) {
192  DEBUG("sched_switch: setting sched_context_switch_request.\n");
193  sched_context_switch_request = 1;
194  }
195  else {
196  DEBUG("sched_switch: yielding immediately.\n");
197  thread_yield_higher();
198  }
199  }
200  else {
201  DEBUG("sched_switch: continuing without yield.\n");
202  }
203 }
