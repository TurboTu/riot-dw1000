40 static void _event_cb(gnrc_netdev_event_t event, void *data)
41 {
42  DEBUG("nomac: event triggered -> %i\n", event);
43  /* NOMAC only understands the RX_COMPLETE event... */
44  if (event == NETDEV_EVENT_RX_COMPLETE) {
45  gnrc_pktsnip_t *pkt;
46 
47  /* get pointer to the received packet */
48  pkt = (gnrc_pktsnip_t *)data;
49  /* send the packet to everyone interested in it's type */
50  if (!gnrc_netapi_dispatch_receive(pkt->type, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
51  DEBUG("nomac: unable to forward packet of type %i\n", pkt->type);
52  gnrc_pktbuf_release(pkt);
53  }
54  }
55 }

64 static void *_nomac_thread(void *args)
65 {
66  gnrc_netdev_t *dev = (gnrc_netdev_t *)args;
67  gnrc_netapi_opt_t *opt;
68  int res;
69  msg_t msg, reply, msg_queue[GNRC_NOMAC_MSG_QUEUE_SIZE];
70 
71  /* setup the MAC layers message queue */
72  msg_init_queue(msg_queue, GNRC_NOMAC_MSG_QUEUE_SIZE);
73  /* save the PID to the device descriptor and register the device */
74  dev->mac_pid = thread_getpid();
75  gnrc_netif_add(dev->mac_pid);
76  /* register the event callback with the device driver */
77  dev->driver->add_event_callback(dev, _event_cb);
78 
79  /* start the event loop */
80  while (1) {
81  DEBUG("nomac: waiting for incoming messages\n");
82  msg_receive(&msg);
83  /* dispatch NETDEV and NETAPI messages */
84  switch (msg.type) {
85  case GNRC_NETDEV_MSG_TYPE_EVENT:
86  DEBUG("nomac: GNRC_NETDEV_MSG_TYPE_EVENT received\n");
87  dev->driver->isr_event(dev, msg.content.value);
88  break;
89  case GNRC_NETAPI_MSG_TYPE_SND:
90  DEBUG("nomac: GNRC_NETAPI_MSG_TYPE_SND received\n");
91  dev->driver->send_data(dev, msg.content.ptr);
92  break;
93  case GNRC_NETAPI_MSG_TYPE_SET:
94  /* TODO: filter out MAC layer options -> for now forward
95  everything to the device driver */
96  DEBUG("nomac: GNRC_NETAPI_MSG_TYPE_SET received\n");
97  /* read incoming options */
98  opt = msg.content.ptr;
99  /* set option for device driver */
100  res = dev->driver->set(dev, opt->opt, opt->data, opt->data_len);
101  DEBUG("nomac: response of netdev->set: %i\n", res);
102  /* send reply to calling thread */
103  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
104  reply.content.value = (uint32_t)res;
105  msg_reply(&msg, &reply);
106  break;
107  case GNRC_NETAPI_MSG_TYPE_GET:
108  /* TODO: filter out MAC layer options -> for now forward
109  everything to the device driver */
110  DEBUG("nomac: GNRC_NETAPI_MSG_TYPE_GET received\n");
111  /* read incoming options */
112  opt = msg.content.ptr;
113  /* get option from device driver */
114  res = dev->driver->get(dev, opt->opt, opt->data, opt->data_len);
115  DEBUG("nomac: response of netdev->get: %i\n", res);
116  /* send reply to calling thread */
117  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
118  reply.content.value = (uint32_t)res;
119  msg_reply(&msg, &reply);
120  break;
121  default:
122  DEBUG("nomac: Unknown command %" PRIu16 "\n", msg.type);
123  break;
124  }
125  }
126  /* never reached */
127  return NULL;
128 }

131  const char *name, gnrc_netdev_t *dev)
132 {
133  kernel_pid_t res;
134

137  return -ENODEV;

139  /* create new NOMAC thread */
