48 static int _send(gnrc_pktsnip_t *pkt)
49 {
50  /* NOTE: Sending Direction: pkt = nw, nw->next = tcp, tcp->next = payload */
51  gnrc_pktsnip_t *tcp;
52 
53  /* Search for tcp header */
54  LL_SEARCH_SCALAR(pkt, tcp, type, GNRC_NETTYPE_TCP);
55  if (tcp == NULL) {
56  DEBUG("gnrc_tcp_eventloop : _send() : tcp header missing.\n");
57  gnrc_pktbuf_release(pkt);
58  return -EBADMSG;
59  }
60 
61  /* Dispatch to network layer */
62  if (!gnrc_netapi_dispatch_send(pkt->type, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
63  DEBUG("gnrc_tcp_eventloop : _send() : network layer not found\n");
64  gnrc_pktbuf_release(pkt);
65  }
66  return 0;
67 }

82 static int _receive(gnrc_pktsnip_t *pkt)
83 {
84  /* NOTE: Receiving direction: pkt = payload, payload->next = tcp, tcp->next = nw */
85  uint16_t ctl = 0;
86  uint16_t src = 0;
87  uint16_t dst = 0;
88  uint8_t hdr_size = 0;
89  uint8_t syn = 0;
90  gnrc_pktsnip_t *ip = NULL;
91  gnrc_pktsnip_t *reset = NULL;
92  gnrc_tcp_tcb_t *tcb = NULL;
93  tcp_hdr_t *hdr;
94 
95  /* Get write access to the TCP Header */
96  gnrc_pktsnip_t *tcp = gnrc_pktbuf_start_write(pkt);
97  if (tcp == NULL) {
98  DEBUG("gnrc_tcp_eventloop.c : _receive() : can't write to packet\n");
99  gnrc_pktbuf_release(pkt);
100  return -EACCES;
101  }
102  pkt = tcp;
103 
104 #ifdef MODULE_GNRC_IPV6
105  /* Get IP Header, discard packet if doesn't contain an ip header */
106  LL_SEARCH_SCALAR(pkt, ip, type, GNRC_NETTYPE_IPV6);
107  if (ip == NULL) {
108  DEBUG("gnrc_tcp_eventloop.c : _receive() : pkt contains no IP Header\n");
109  gnrc_pktbuf_release(pkt);
110  return 0;
111  }
112 #endif
113 
114  /* Get TCP Header */
115  LL_SEARCH_SCALAR(pkt, tcp, type, GNRC_NETTYPE_TCP);
116  if (tcp == NULL) {
117  DEBUG("gnrc_tcp_eventloop.c : _receive() : pkt contains no TCP Header\n");
118  gnrc_pktbuf_release(pkt);
119  return 0;
120  }
121 
122  /* Extract control bits, src and dst ports and check if SYN is set (not SYN+ACK) */
123  hdr = (tcp_hdr_t *)tcp->data;
124  ctl = byteorder_ntohs(hdr->off_ctl);
125  src = byteorder_ntohs(hdr->src_port);
126  dst = byteorder_ntohs(hdr->dst_port);
127  syn = ((ctl & MSK_SYN_ACK) == MSK_SYN);
128 
129  /* Validate Offset */
130  if (GET_OFFSET(ctl) < OPTION_OFFSET_BASE) {
131  DEBUG("gnrc_tcp_eventloop.c : _receive() : unexpected Offset Value\n");
132  gnrc_pktbuf_release(pkt);
133  return -ERANGE;
134  }
135 
136  /* Calculate tcp header size */
137  hdr_size = GET_OFFSET(ctl) * 4;
138 
139  /* Mark TCP-Header, if it contains any payload */
140  if ((pkt->type == GNRC_NETTYPE_TCP) && (pkt->size != hdr_size)) {
141  tcp = gnrc_pktbuf_mark(pkt, hdr_size, GNRC_NETTYPE_TCP);
142  if (tcp == NULL) {
143  DEBUG("gnrc_tcp_eventloop.c : _receive() : Header marking failed\n");
144  gnrc_pktbuf_release(pkt);
145  return -ENOMSG;
146  }
147  pkt->type = GNRC_NETTYPE_UNDEF;
148  }
149 
150  /* Validate Checksum */
151  if (byteorder_ntohs(hdr->checksum) != _pkt_calc_csum(tcp, ip, pkt)) {
152  DEBUG("gnrc_tcp_eventloop.c : _receive() : Invalid checksum\n");
153  gnrc_pktbuf_release(pkt);
154  return -EINVAL;
155  }
156 
157  /* Find tcb to de-multiplex this packet to */
158  mutex_lock(&_list_gnrc_tcp_tcb_lock);
159  tcb = _list_gnrc_tcp_tcb_head;
160  while (tcb) {
161 #ifdef MODULE_GNRC_IPV6
162  /* Check if current tcb is fitting for the incomming packet */
163  if (ip->type == GNRC_NETTYPE_IPV6 && tcb->address_family == AF_INET6) {
164  /* If SYN is set, a connection is listening on that port ... */
165  ipv6_addr_t * tmp_addr = NULL;
166  if (syn && tcb->local_port == dst && tcb->state == GNRC_TCP_FSM_STATE_LISTEN) {
167  /* ... and local addr is unspec or preconfigured */
168  tmp_addr = &((ipv6_hdr_t * )ip->data)->dst;
169  if (ipv6_addr_equal((ipv6_addr_t *) tcb->local_addr, (ipv6_addr_t *) tmp_addr)
170  || ipv6_addr_is_unspecified((ipv6_addr_t *) tcb->local_addr)
171  ) {
172  break;
173  }
174  }
175 
176  /* If SYN is not set and the ports match ... */
177  if (!syn && tcb->local_port == dst && tcb->peer_port == src) {
178  /* .. and the IP-Addresses match */
179  tmp_addr = &((ipv6_hdr_t * )ip->data)->src;
180  if (ipv6_addr_equal((ipv6_addr_t *) tcb->peer_addr, (ipv6_addr_t *) tmp_addr)) {
181  break;
182  }
183  }
184  }
185 #else
186  /* Supress compiler warnings if TCP is build without IP-Layer */
187  (void) syn;
188  (void) src;
189  (void) dst;
190 #endif
191  tcb = tcb->next;
192  }
193  mutex_unlock(&_list_gnrc_tcp_tcb_lock);
194 
195  /* Call FSM with event RCVD_PKT if a fitting connection was found */
196  if (tcb != NULL) {
197  _fsm(tcb, GNRC_TCP_FSM_EVENT_RCVD_PKT, pkt, NULL, 0);
198  }
199  /* No fitting connection has been found. Respond with reset */
200  else {
201  DEBUG("gnrc_tcp_eventloop.c : _receive() : Can't find fitting connection\n");
202  if ((ctl & MSK_RST) != MSK_RST) {
203  _pkt_build_reset_from_pkt(&reset, pkt);
204  gnrc_netapi_send(_gnrc_tcp_pid, reset);
205  }
206  return -ENOTCONN;
207  }
208  gnrc_pktbuf_release(pkt);
209  return 0;
210 }

212 void *_event_loop(__attribute__((unused)) void *arg)
213 {
214  msg_t msg;
215  msg_t reply;
216  msg_t msg_queue[GNRC_TCP_MSG_QUEUE_SIZE];
217 
218  /* Store pid */
219  _gnrc_tcp_pid = thread_getpid();
220 
221  /* Setup reply message */
222  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
223  reply.content.value = (uint32_t)-ENOTSUP;
224 
225  /* Init message queue*/
226  msg_init_queue(msg_queue, GNRC_TCP_MSG_QUEUE_SIZE);
227 
228  /* Register GNRC_tcp in netreg */
229  gnrc_netreg_entry_t entry;
230  gnrc_netreg_entry_init_pid(&entry, GNRC_NETREG_DEMUX_CTX_ALL, _gnrc_tcp_pid);
231  gnrc_netreg_register(GNRC_NETTYPE_TCP, &entry);
232 
233  /* dispatch NETAPI Messages */
234  while (1) {
235  msg_receive(&msg);
236  switch (msg.type) {
237  /* Pass Message up the network stack */
238  case GNRC_NETAPI_MSG_TYPE_RCV:
239  DEBUG("gnrc_tcp_eventloop.c : _event_loop() : GNRC_NETAPI_MSG_TYPE_RCV\n");
240  _receive((gnrc_pktsnip_t *)msg.content.ptr);
241  break;
242 
243  /* Pass Message down the network stack */
244  case GNRC_NETAPI_MSG_TYPE_SND:
245  DEBUG("gnrc_tcp_eventloop.c : _event_loop() : GNRC_NETAPI_MSG_TYPE_SND\n");
246  _send((gnrc_pktsnip_t *)msg.content.ptr);
247  break;
248 
249  /* Option Set and Get Messages*/
250  case GNRC_NETAPI_MSG_TYPE_SET:
251  case GNRC_NETAPI_MSG_TYPE_GET:
252  msg_reply(&msg, &reply);
253  break;
254 
255  /* Retransmission Timer expired -> Call FSM with retransmission event */
256  case MSG_TYPE_RETRANSMISSION:
257  DEBUG("gnrc_tcp_eventloop.c : _event_loop() : MSG_TYPE_RETRANSMISSION\n");
258  _fsm((gnrc_tcp_tcb_t *)msg.content.ptr, GNRC_TCP_FSM_EVENT_TIMEOUT_RETRANSMIT,
259  NULL, NULL, 0);
260  break;
261 
262  /* Time Wait Timer expired -> Call FSM with timewait event */
263  case MSG_TYPE_TIMEWAIT:
264  DEBUG("gnrc_tcp_eventloop.c : _event_loop() : MSG_TYPE_TIMEWAIT\n");
265  _fsm((gnrc_tcp_tcb_t *)msg.content.ptr, GNRC_TCP_FSM_EVENT_TIMEOUT_TIMEWAIT,
266  NULL, NULL, 0);
267  break;
268 
269  default:
270  DEBUG("gnrc_tcp_eventloop.c : _event_loop() : received expected message\n");
271  }
272  }
273  /* never reached */
274  return NULL;
275 }
