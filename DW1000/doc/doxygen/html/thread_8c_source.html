144 {
145  if (priority >= SCHED_PRIO_LEVELS) {
146  return -EINVAL;
147  }
148

151 #else
152  (void) name;
153 #endif
154 
155  /* align the stack on a 16/32bit boundary */
156  uintptr_t misalignment = (uintptr_t) stack % ALIGN_OF(void *);
157  if (misalignment) {
158  misalignment = ALIGN_OF(void *) - misalignment;
159  stack += misalignment;
160  stacksize -= misalignment;
161  }
162 
163  /* make room for the thread control block */
164  stacksize -= sizeof(thread_t);
165 
166  /* round down the stacksize to a multiple of thread_t alignments (usually 16/32bit) */
167  stacksize -= stacksize % ALIGN_OF(thread_t);
168

170  DEBUG("thread_create: stacksize is too small!\n");
171  }
172  /* allocate our thread control block at the top of our stackspace */
173  thread_t *cb = (thread_t *) (stack + stacksize);
174 
175 #if defined(DEVELHELP) || defined(SCHED_TEST_STACK)
176  if (flags & THREAD_CREATE_STACKTEST) {
177  /* assign each int of the stack the value of it's address */
178  uintptr_t *stackmax = (uintptr_t *) (stack + stacksize);
179  uintptr_t *stackp = (uintptr_t *) stack;
180 
181  while (stackp < stackmax) {
182  *stackp = (uintptr_t) stackp;
183  stackp++;
184  }
185  }
186  else {
187  /* create stack guard */
188  *(uintptr_t *) stack = (uintptr_t) stack;
189  }
190 #endif
191 
192  unsigned state = irq_disable();
193 
194  kernel_pid_t pid = KERNEL_PID_UNDEF;
195  for (kernel_pid_t i = KERNEL_PID_FIRST; i <= KERNEL_PID_LAST; ++i) {
196  if (sched_threads[i] == NULL) {
197  pid = i;
198  break;
199  }
200  }
201  if (pid == KERNEL_PID_UNDEF) {
202  DEBUG("thread_create(): too many threads!\n");
203 
204  irq_restore(state);
205 
206  return -EOVERFLOW;
207  }
208

210

212  cb->sp = thread_stack_init(function, arg, stack, stacksize);
213 
214 #if defined(DEVELHELP) || defined(SCHED_TEST_STACK) || defined(MODULE_MPU_STACK_GUARD)
215  cb->stack_start = stack;
216 #endif
217 
218 #ifdef DEVELHELP

220  cb->name = name;
221 #endif
222

224  cb->status = 0;
225

227

234
