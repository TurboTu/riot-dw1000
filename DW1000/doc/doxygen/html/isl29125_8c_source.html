95 void isl29125_read_rgb_lux(isl29125_t *dev, isl29125_rgb_t *dest)
96 {
97  /* acquire exclusive access to the bus */
98  (void) i2c_acquire(dev->i2c);
99 
100  /* read values */
101  uint8_t bytes[6];
102  (void) i2c_read_regs(dev->i2c, ISL29125_I2C_ADDRESS, ISL29125_REG_GDLB, bytes, 6);
103 
104  /* release the I2C bus */
105  (void) i2c_release(dev->i2c);
106 
107  /* possibly shift by 4 to normalize 12 to 16 bit */
108  int resfactor = (dev->res == ISL29125_RESOLUTION_12) ? 4 : 0;
109  /* parse and normalize readings */
110  uint16_t green = (bytes[0] | (bytes[1] << 8)) << resfactor;
111  uint16_t red = (bytes[2] | (bytes[3] << 8)) << resfactor;
112  uint16_t blue = (bytes[4] | (bytes[5] << 8)) << resfactor;
113 
114  DEBUG("isl29125_read_rgb: adjusted, unconverted readings: (%5i / %5i / %5i) \n", red, green, blue);
115 
116  /* convert readings to lux */
117  float luxfactor = (dev->range == ISL29125_RANGE_10K) ? 10000.0 / 65535.0 : 375.0 / 65535.0;
118  dest->red = red * luxfactor;
119  dest->green = green * luxfactor;
120  dest->blue = blue * luxfactor;
121 }
