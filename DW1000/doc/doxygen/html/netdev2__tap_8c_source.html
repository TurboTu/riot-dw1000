105 static int _get(netdev2_t *dev, netopt_t opt, void *value, size_t max_len)
106 {
107  int res = 0;
108 
109  switch (opt) {
110  case NETOPT_ADDRESS:
111  if (max_len < ETHERNET_ADDR_LEN) {
112  res = -EINVAL;
113  }
114  else {
115  _get_mac_addr(dev, (uint8_t*)value);
116  res = ETHERNET_ADDR_LEN;
117  }
118  break;
119  case NETOPT_PROMISCUOUSMODE:
120  *((bool*)value) = (bool)_get_promiscous(dev);
121  res = sizeof(bool);
122  break;
123  default:
124  res = netdev2_eth_get(dev, opt, value, max_len);
125  break;
126  }
127 
128  return res;
129 }

173 static void _continue_reading(netdev2_tap_t *dev)
174 {
175  /* work around lost signals */
176  fd_set rfds;
177  struct timeval t;
178  memset(&t, 0, sizeof(t));
179  FD_ZERO(&rfds);
180  FD_SET(dev->tap_fd, &rfds);
181 
182  _native_in_syscall++; /* no switching here */
183 
184  if (real_select(dev->tap_fd + 1, &rfds, NULL, NULL, &t) == 1) {
185  int sig = SIGIO;
186  extern int _sig_pipefd[2];
187  extern ssize_t (*real_write)(int fd, const void * buf, size_t count);
188  real_write(_sig_pipefd[1], &sig, sizeof(int));
189  _native_sigpend++;
190  DEBUG("netdev2_tap: sigpend++\n");
191  }
192  else {
193  DEBUG("netdev2_tap: native_async_read_continue\n");
194  native_async_read_continue(dev->tap_fd);
195  }
196 
197  _native_in_syscall--;
198 }

200 static int _recv(netdev2_t *netdev2, void *buf, size_t len, void *info)
201 {
202  netdev2_tap_t *dev = (netdev2_tap_t*)netdev2;
203  (void)info;
204 
205  if (!buf) {
206  if (len > 0) {
207  /* no memory available in pktbuf, discarding the frame */
208  DEBUG("netdev2_tap: discarding the frame\n");
209 
210  /* repeating `real_read` for small size on tap device results in
211  * freeze for some reason. Using a large buffer for now. */
212  /*
213  uint8_t buf[4];
214  while (real_read(dev->tap_fd, buf, sizeof(buf)) > 0) {
215  }
216  */
217 
218  static uint8_t buf[ETHERNET_FRAME_LEN];
219 
220  real_read(dev->tap_fd, buf, sizeof(buf));
221 
222  _continue_reading(dev);
223  }
224 
225  /* no way of figuring out packet size without racey buffering,
226  * so we return the maximum possible size */
227  return ETHERNET_FRAME_LEN;
228  }
229 
230  int nread = real_read(dev->tap_fd, buf, len);
231  DEBUG("netdev2_tap: read %d bytes\n", nread);
232 
233  if (nread > 0) {
234  ethernet_hdr_t *hdr = (ethernet_hdr_t *)buf;
235  if (!(dev->promiscous) && !_is_addr_multicast(hdr->dst) &&
236  !_is_addr_broadcast(hdr->dst) &&
237  (memcmp(hdr->dst, dev->addr, ETHERNET_ADDR_LEN) != 0)) {
238  DEBUG("netdev2_tap: received for %02x:%02x:%02x:%02x:%02x:%02x\n"
239  "That's not me => Dropped\n",
240  hdr->dst[0], hdr->dst[1], hdr->dst[2],
241  hdr->dst[3], hdr->dst[4], hdr->dst[5]);
242 
243  native_async_read_continue(dev->tap_fd);
244 
245  return 0;
246  }
247 
248  _continue_reading(dev);
249 
250 #ifdef MODULE_NETSTATS_L2
251  netdev2->stats.rx_count++;
252  netdev2->stats.rx_bytes += nread;
253 #endif
254  return nread;
255  }
256  else if (nread == -1) {
257  if ((errno == EAGAIN) || (errno == EWOULDBLOCK)) {
258  }
259  else {
260  err(EXIT_FAILURE, "netdev2_tap: read");
261  }
262  }
263  else if (nread == 0) {
264  DEBUG("_native_handle_tap_input: ignoring null-event\n");
265  }
266  else {
267  errx(EXIT_FAILURE, "internal error _rx_event");
268  }
269 
270  return -1;
271 }

273 static int _send(netdev2_t *netdev, const struct iovec *vector, unsigned n)
274 {
275  netdev2_tap_t *dev = (netdev2_tap_t*)netdev;
276  int res = _native_writev(dev->tap_fd, vector, n);
277 #ifdef MODULE_NETSTATS_L2
278  size_t bytes = 0;

280  bytes += vector->iov_len;

284 #endif
285  if (netdev->event_callback) {

309 static int _init(netdev2_t *netdev)
310 {
311  DEBUG("%s:%s:%u\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
312 
313  netdev2_tap_t *dev = (netdev2_tap_t*)netdev;
314 
315  /* check device parametrs */
316  if (dev == NULL) {
317  return -ENODEV;
318  }
319 
320  char *name = dev->tap_name;
321 #ifdef __MACH__ /* OSX */
322  char clonedev[255] = "/dev/"; /* XXX bad size */
323  strncpy(clonedev + 5, name, 250);
324 #elif defined(__FreeBSD__)
325  char clonedev[255] = "/dev/"; /* XXX bad size */
326  strncpy(clonedev + 5, name, 250);
327 #else /* Linux */
328  struct ifreq ifr;
329  const char *clonedev = "/dev/net/tun";
330 #endif
331  /* initialize device descriptor */
332  dev->promiscous = 0;
333  /* implicitly create the tap interface */
334  if ((dev->tap_fd = real_open(clonedev, O_RDWR | O_NONBLOCK)) == -1) {
335  err(EXIT_FAILURE, "open(%s)", clonedev);
336  }
337 #if (defined(__MACH__) || defined(__FreeBSD__)) /* OSX/FreeBSD */
338  struct ifaddrs *iflist;
339  if (real_getifaddrs(&iflist) == 0) {
340  for (struct ifaddrs *cur = iflist; cur; cur = cur->ifa_next) {
341  if ((cur->ifa_addr->sa_family == AF_LINK) && (strcmp(cur->ifa_name, name) == 0) && cur->ifa_addr) {
342  struct sockaddr_dl *sdl = (struct sockaddr_dl *)cur->ifa_addr;
343  memcpy(dev->addr, LLADDR(sdl), sdl->sdl_alen);
344  break;
345  }
346  }
347  real_freeifaddrs(iflist);
348  }
349 #else /* Linux */
350  memset(&ifr, 0, sizeof(ifr));
351  ifr.ifr_flags = IFF_TAP | IFF_NO_PI;
352  strncpy(ifr.ifr_name, name, IFNAMSIZ);
353  if (real_ioctl(dev->tap_fd, TUNSETIFF, (void *)&ifr) == -1) {
354  _native_in_syscall++;
355  warn("ioctl TUNSETIFF");
356  warnx("probably the tap interface (%s) does not exist or is already in use", name);
357  real_exit(EXIT_FAILURE);
358  }
359 
360  /* get MAC address */
361  memset(&ifr, 0, sizeof(ifr));
362  snprintf(ifr.ifr_name, sizeof(ifr.ifr_name), "%s", name);
363  if (real_ioctl(dev->tap_fd, SIOCGIFHWADDR, &ifr) == -1) {
364  _native_in_syscall++;
365  warn("ioctl SIOCGIFHWADDR");
366  if (real_close(dev->tap_fd) == -1) {
367  warn("close");
368  }
369  real_exit(EXIT_FAILURE);
370  }
371  memcpy(dev->addr, ifr.ifr_hwaddr.sa_data, ETHERNET_ADDR_LEN);
372 
373  /* change mac addr so it differs from what the host is using */
374  dev->addr[5]++;
375 #endif
376  DEBUG("gnrc_tapnet_init(): dev->addr = %02x:%02x:%02x:%02x:%02x:%02x\n",
377  dev->addr[0], dev->addr[1], dev->addr[2],
378  dev->addr[3], dev->addr[4], dev->addr[5]);
379 
380  /* configure signal handler for fds */
381  native_async_read_setup();
382  native_async_read_add_handler(dev->tap_fd, netdev, _tap_isr);
383 
384 #ifdef MODULE_NETSTATS_L2
385  memset(&netdev->stats, 0, sizeof(netstats_t));
386 #endif
387  DEBUG("gnrc_tapnet: initialized.\n");
388  return 0;
389 }
