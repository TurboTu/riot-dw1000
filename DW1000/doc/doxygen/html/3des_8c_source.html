254 int tripledes_init(cipher_context_t *context, const uint8_t *key,
255  uint8_t keySize)
256 {
257  uint8_t i;
258 
259  // Make sure that context is large enough. If this is not the case,
260  // you should build with -DTHREEDES
261  if(CIPHER_MAX_CONTEXT_SIZE < THREEDES_MAX_KEY_SIZE) {
262  return CIPHER_ERR_BAD_CONTEXT_SIZE;
263  }
264 
265  //key must be at least 24 Bytes long
266  if (keySize < 24) {
267  //fill up by concatenating key to as long as needed
268  for (i = 0; i < 24; i++) {
269  context->context[i] = key[(i % keySize)];
270  }
271  }
272  else {
273  for (i = 0; i < 24; i++) {
274  context->context[i] = key[i];
275  }
276  }
277 
278  return CIPHER_INIT_SUCCESS;
279 }

281 int tripledes_encrypt(const cipher_context_t *context, const uint8_t *plain, uint8_t *crypt)
282 {
283  int res;
284  struct des3_key_s *key = malloc(sizeof(des3_key_s));
285  uint32_t work[2];
286 
287  if (!key) {
288  DEBUG("%s:%d in %s: [ERROR] Could NOT malloc space for the des3_key_s struct.\r\n",
289  RIOT_FILE_RELATIVE, __LINE__, DEBUG_FUNC);
290  return -1;
291  }
292 
293  memset(key, 0, sizeof(des3_key_s));
294  res = des3_key_setup(context->context, key);
295 
296  if (res < 0) {
297  DEBUG("%s:%d in %s: [ERROR] des3_key_setup failed with Code %i\r\n",
298  RIOT_FILE_RELATIVE, __LINE__, DEBUG_FUNC, res);
299  free(key);
300  return -2;
301  }
302 
303  work[0] = WPA_GET_BE32(plain);
304  work[1] = WPA_GET_BE32(plain + 4);
305  desfunc(work, key->ek[0]);
306  desfunc(work, key->ek[1]);
307  desfunc(work, key->ek[2]);
308  WPA_PUT_BE32(crypt, work[0]);
309  WPA_PUT_BE32(crypt + 4, work[1]);
310 
311  free(key);
312  return 1;
313 }

350 static void cookey(const uint32_t *raw1, uint32_t *keyout)
351 {
352  uint32_t *cook;
353  uint32_t dough[32];
354  int i;
355 
356  cook = dough;
357 
358  for (i = 0; i < 16; i++, raw1++) {
359  const uint32_t *raw0 = raw1++;
360  *cook = (*raw0 & 0x00fc0000L) << 6;
361  *cook |= (*raw0 & 0x00000fc0L) << 10;
362  *cook |= (*raw1 & 0x00fc0000L) >> 10;
363  *cook++ |= (*raw1 & 0x00000fc0L) >> 6;
364  *cook = (*raw0 & 0x0003f000L) << 12;
365  *cook |= (*raw0 & 0x0000003fL) << 16;
366  *cook |= (*raw1 & 0x0003f000L) >> 4;
367  *cook++ |= (*raw1 & 0x0000003fL);
368  }
369 
370  memcpy(keyout, dough, sizeof(dough));
371 }

374 static void deskey(const uint8_t *key, int decrypt, uint32_t *keyout)
375 {
376  uint32_t i, j, l, m, kn[32];
377  uint8_t pc1m[56], pcr[56];
378 
379  for (j = 0; j < 56; j++) {
380  l = (uint32_t) pc1[j];
381  m = l & 7;
382  pc1m[j] = (uint8_t)
383  ((key[l >> 3U] & bytebit[m]) == bytebit[m] ? 1 : 0);
384  }
385 
386  for (i = 0; i < 16; i++) {
387  if (decrypt) {
388  m = (15 - i) << 1;
389  }
390  else {
391  m = i << 1;
392  }
393 
394  uint32_t n = m + 1;
395  kn[m] = kn[n] = 0L;
396 
397  for (j = 0; j < 28; j++) {
398  l = j + (uint32_t) totrot[i];
399 
400  if (l < 28) {
401  pcr[j] = pc1m[l];
402  }
403  else {
404  pcr[j] = pc1m[l - 28];
405  }
406  }
407 
408  for (/* j = 28 */; j < 56; j++) {
409  l = j + (uint32_t) totrot[i];
410 
411  if (l < 56) {
412  pcr[j] = pc1m[l];
413  }
414  else {
415  pcr[j] = pc1m[l - 28];
416  }
417  }
418 
419  for (j = 0; j < 24; j++) {
420  if ((int) pcr[(int) pc2[j]] != 0) {
421  kn[m] |= bigbyte[j];
422  }
423 
424  if ((int) pcr[(int) pc2[j + 24]] != 0) {
425  kn[n] |= bigbyte[j];
426  }
427  }
428  }
429 
430  cookey(kn, keyout);
431 }

434 static void desfunc(uint32_t *block, const uint32_t *keys)
435 {
436  uint32_t work, right, leftt;
437  int cur_round;
438 
439  leftt = block[0];
440  right = block[1];
441 
442  work = ((leftt >> 4) ^ right) & 0x0f0f0f0fL;
443  right ^= work;
444  leftt ^= (work << 4);
445 
446  work = ((leftt >> 16) ^ right) & 0x0000ffffL;
447  right ^= work;
448  leftt ^= (work << 16);
449 
450  work = ((right >> 2) ^ leftt) & 0x33333333L;
451  leftt ^= work;
452  right ^= (work << 2);
453 
454  work = ((right >> 8) ^ leftt) & 0x00ff00ffL;
455  leftt ^= work;
456  right ^= (work << 8);
457 
458  right = ROLc(right, 1);
459  work = (leftt ^ right) & 0xaaaaaaaaL;
460 
461  leftt ^= work;
462  right ^= work;
463  leftt = ROLc(leftt, 1);
464 
465  for (cur_round = 0; cur_round < 8; cur_round++) {
466  work = RORc(right, 4) ^ *keys++;
467  leftt ^= SP7[work & 0x3fL]
468  ^ SP5[(work >> 8) & 0x3fL]
469  ^ SP3[(work >> 16) & 0x3fL]
470  ^ SP1[(work >> 24) & 0x3fL];
471  work = right ^ *keys++;
472  leftt ^= SP8[ work & 0x3fL]
473  ^ SP6[(work >> 8) & 0x3fL]
474  ^ SP4[(work >> 16) & 0x3fL]
475  ^ SP2[(work >> 24) & 0x3fL];
476 
477  work = RORc(leftt, 4) ^ *keys++;
478  right ^= SP7[ work & 0x3fL]
479  ^ SP5[(work >> 8) & 0x3fL]
480  ^ SP3[(work >> 16) & 0x3fL]
481  ^ SP1[(work >> 24) & 0x3fL];
482  work = leftt ^ *keys++;
483  right ^= SP8[ work & 0x3fL]
484  ^ SP6[(work >> 8) & 0x3fL]
485  ^ SP4[(work >> 16) & 0x3fL]
486  ^ SP2[(work >> 24) & 0x3fL];
487  }
488 
489  right = RORc(right, 1);
490  work = (leftt ^ right) & 0xaaaaaaaaL;
491  leftt ^= work;
492  right ^= work;
493  leftt = RORc(leftt, 1);
494  work = ((leftt >> 8) ^ right) & 0x00ff00ffL;
495  right ^= work;
496  leftt ^= (work << 8);
497  /* -- */
498  work = ((leftt >> 2) ^ right) & 0x33333333L;
499  right ^= work;
500  leftt ^= (work << 2);
501  work = ((right >> 16) ^ leftt) & 0x0000ffffL;
502  leftt ^= work;
503  right ^= (work << 16);
504  work = ((right >> 4) ^ leftt) & 0x0f0f0f0fL;
505  leftt ^= work;
506  right ^= (work << 4);
507 
508  block[0] = right;
509  block[1] = leftt;
