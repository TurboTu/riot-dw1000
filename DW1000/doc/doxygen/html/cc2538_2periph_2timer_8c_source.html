75 int timer_init(tim_t dev, unsigned long freq, timer_cb_t cb, void *arg)
76 {
77  cc2538_gptimer_t *gptimer = timer_config[dev].dev;
78  unsigned int gptimer_num;
79  uint32_t chan_mode;
80 
81  DEBUG("%s(%u, %lu, %p, %p)\n", __FUNCTION__, dev, freq, cb, arg);
82 
83  if (dev >= TIMER_NUMOF) {
84  return -1;
85  }
86 
87  gptimer_num = GPTIMER_GET_NUM(gptimer);
88 
89  /* Save the callback function: */
90  assert(gptimer_num < GPTIMER_NUMOF);
91  config[gptimer_num].cb = cb;
92  config[gptimer_num].arg = arg;
93 
94  /* Enable the clock for this timer: */
95  SYS_CTRL_RCGCGPT |= (1 << gptimer_num);
96 
97  /* Disable this timer before configuring it: */
98  gptimer->cc2538_gptimer_ctl.CTL = 0;
99 
100  if (timer_config[dev].cfg == GPTMCFG_32_BIT_TIMER) {
101  /* Count up in periodic mode */
102  chan_mode = TnCMIE | TnCDIR | GPTIMER_PERIODIC_MODE;
103 
104  if (timer_config[dev].channels > 1) {
105  DEBUG("Invalid timer_config. Multiple channels are available only in 16-bit mode.");
106  return -1;
107  }
108 
109  if (freq != sys_clock_freq()) {
110  DEBUG("In 32-bit mode, the GPTimer frequency must equal the system clock frequency (%u).", sys_clock_freq());
111  return -1;
112  }
113  } else {
114  /* Count down in periodic mode */
115  chan_mode = TnCMIE | GPTIMER_PERIODIC_MODE;
116  }
117 
118  gptimer->CFG = timer_config[dev].cfg;
119  gptimer->cc2538_gptimer_tamr.TAMR = chan_mode;
120 
121  switch (timer_config[dev].channels) {
122  case 1:
123  /* Enable the timer: */
124  gptimer->cc2538_gptimer_ctl.CTL = TAEN;
125  break;
126 
127  case 2:
128  gptimer->cc2538_gptimer_tbmr.TBMR = chan_mode;
129 
130  gptimer->TAILR = LOAD_VALUE;
131  gptimer->TBILR = LOAD_VALUE;
132 
133  uint32_t prescaler = sys_clock_freq();
134  prescaler += freq / 2;
135  prescaler /= freq;
136  if (prescaler > 0) prescaler--;
137  if (prescaler > 255) prescaler = 255;
138 
139  gptimer->TAPR = prescaler;
140  gptimer->TBPR = prescaler;
141 
142  /* Enable the timer: */
143  gptimer->cc2538_gptimer_ctl.CTL = TBEN | TAEN;
144  break;
145  }
146 
147  /* Enable interrupts for given timer: */
148  _irq_enable(dev);
149 
150  return 0;
151 }

153 int timer_set(tim_t dev, int channel, unsigned int timeout)
154 {
155  /* get timer base register address */
156  cc2538_gptimer_t *gptimer = timer_config[dev].dev;
157 
158  if ( (dev >= TIMER_NUMOF) || (channel >= timer_config[dev].channels) ) {
159  return -1;
160  }
161 
162  switch (channel) {
163  case 0:
164  /* clear any pending match interrupts */
165  gptimer->ICR = TAMIM;
166 
167  /* set timeout value */
168  gptimer->TAMATCHR = (gptimer->CFG == GPTMCFG_32_BIT_TIMER)? (gptimer->TAV + timeout) : (gptimer->TAV - timeout);
169  gptimer->cc2538_gptimer_imr.IMR |= TAMIM;
170  break;
171 
172  case 1:
173  /* clear any pending match interrupts */
174  gptimer->ICR = TBMIM;
175 
176  /* set timeout value */
177  gptimer->TAMATCHR = (gptimer->CFG == GPTMCFG_32_BIT_TIMER)? (gptimer->TBV + timeout) : (gptimer->TBV - timeout);
178  gptimer->cc2538_gptimer_imr.IMR |= TBMIM;
179  break;
180  }
181 
182  return 1;
183 }

300 static void irq_handler_a(int n) {
301  uint32_t mis;
302 
303  /* Latch the active interrupt flags */
304  mis = GPTIMER[n].MIS & TIMER_A_IRQ_MASK;
305 
306  /* Clear the latched interrupt flags */
307  GPTIMER[n].ICR = mis;
308 
309  if (mis & TAMIM) {
310  /* This is a Timer A Match Interrupt */
311 
312  /* Disable further match interrupts for this timer/channel */
313  GPTIMER[n].cc2538_gptimer_imr.IMR &= ~TAMIM;
314 
315  /* Invoke the callback function */
316  assert(config[n].cb != NULL);
317  config[n].cb(config[n].arg, 0);
318  }
319 
320  cortexm_isr_end();
321 }
