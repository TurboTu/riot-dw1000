20 static int compare_str_vs_digest(const char *str,
21  const uint8_t hash[SHA256_DIGEST_LENGTH])
22 {
23  char ch[3] = { 0, 0, 0 };
24  size_t iter_hash = 0;
25  size_t str_length = strlen((char*)str);
26  for (size_t i = 0; i < str_length; i += 2) {
27  ch[0] = str[i];
28  ch[1] = str[i + 1];
29 
30  if (hash[iter_hash++] != strtol(ch, NULL, 16)) {
31  return 0;
32  }
33  }
34  return 1;
35 }
