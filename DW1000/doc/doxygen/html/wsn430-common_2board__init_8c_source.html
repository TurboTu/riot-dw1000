73 void msp430_init_dco(void)
74 {
75  /*----------------------- use external oszillator -------------------------*/
76  uint16_t i;
77 
78  // Stop watchdog
79  WDTCTL = WDTPW + WDTHOLD;
80 
81  BCSCTL1 = RSEL2;
82 
83  // Wait for xtal to stabilize
84  do {
85  IFG1 &= ~OFIFG; // Clear oscillator fault flag
86  for (i = 0xFF; i > 0; i--); // Time for flag to set
87  }
88  while ((IFG1 & OFIFG) != 0); // Oscillator fault flag still set?
89  switch (__msp430_cpu_speed) {
90  case MCLK_2MHZ_SCLK_1MHZ:
91  BCSCTL2 = (SELM_2 | DIVM_2) | (SELS | DIVS_3);
92  break;
93  case MCLK_4MHZ_SCLK_1MHZ:
94  BCSCTL2 = (SELM_2 | DIVM_1) | (SELS | DIVS_3);
95  break;
96  case MCLK_8MHZ_SCLK_1MHZ:
97  BCSCTL2 = SELM_2 | (SELS | DIVS_3);
98  break;
99  default:
100  BCSCTL2 = SELM_2 + SELS; // MCLK and SMCLK = XT2 (safe)
101  break;
102  }
103 }
