103  gpio_cb_t cb, void *arg)
104 {
105  msp_port_isr_t *port = _isr_port(pin);
106 
107  /* check if port, pull resistor and flank configuration are valid */
108  if ((port == NULL) || (flank == GPIO_BOTH)) {
109  return -1;
110  }
111 
112  /* disable any activated interrupt */
113  port->IE &= ~(_pin(pin));
114  /* configure as input */

116  return -1;
