71 int i2c_init_master(i2c_t dev, i2c_speed_t speed)
72 {
73  I2C_TypeDef *i2c;
74  GPIO_TypeDef *port_scl;
75  GPIO_TypeDef *port_sda;
76  int pin_scl = 0, pin_sda = 0;
77  uint32_t presc, scll, sclh, sdadel, scldel, timing;
78 
79  /*
80  * Speed configuration:
81  * Example values can be found in the STM32F3xx Reference manual RM0316
82  * Chapter 28.4.9: Table 148 Examples of timings settings for f_I2CCLK = 8MHz
83  * t_SCLL: SCL low level counter
84  * t_SCLH: SCL high level counter
85  * t_SDADEL: Delay before sending SDA output
86  * t_SCLDEL: SCL low level during setup-time
87  */
88  switch (speed) {
89  case I2C_SPEED_NORMAL:
90  presc = 1;
91  scll = 0x13; /* t_SCLL = 5.0us */
92  sclh = 0xF; /* t_SCLH = 4.0us */
93  sdadel = 0x2; /* t_SDADEL = 500ns */
94  scldel = 0x4; /* t_SCLDEL = 1250ns */
95  break;
96 
97  case I2C_SPEED_FAST:
98  presc = 0;
99  scll = 0x9; /* t_SCLL = 1250ns */
100  sclh = 0x3; /* t_SCLH = 500ns */
101  sdadel = 0x1; /* t_SDADEL = 125ns */
102  scldel = 0x3; /* t_SCLDEL = 500ns */
103  break;
104 
105  case I2C_SPEED_FAST_PLUS:
106  presc = 0;
107  scll = 0x6; /* t_SCLL = 875ns */
108  sclh = 0x3; /* t_SCLH = 500ns */
109  sdadel = 0x0; /* t_SDADEL = 0ns */
110  scldel = 0x1; /* t_SCLDEL = 250ns */
111  break;
112 
113  default:
114  return -2;
115  }
116 
117  /* prepare the timing register value */
118  timing = ((presc << 28) | (scldel << 20) | (sdadel << 16) | (sclh << 8) | scll);
119 
120  /* read static device configuration */
121  switch (dev) {
122 #if I2C_0_EN
123  case I2C_0:
124  i2c = I2C_0_DEV;
125  port_scl = I2C_0_SCL_PORT;
126  pin_scl = I2C_0_SCL_PIN;
127  port_sda = I2C_0_SDA_PORT;
128  pin_sda = I2C_0_SDA_PIN;
129  I2C_0_CLKEN();
130  I2C_0_SCL_CLKEN();
131  I2C_0_SDA_CLKEN();
132  NVIC_SetPriority(I2C_0_ERR_IRQ, I2C_IRQ_PRIO);
133  NVIC_EnableIRQ(I2C_0_ERR_IRQ);
134  break;
135 #endif
136 #if I2C_1_EN
137  case I2C_1:
138  i2c = I2C_1_DEV;
139  port_scl = I2C_1_SCL_PORT;
140  pin_scl = I2C_1_SCL_PIN;
141  port_sda = I2C_1_SDA_PORT;
142  pin_sda = I2C_1_SDA_PIN;
143  I2C_1_CLKEN();
144  I2C_1_SCL_CLKEN();
145  I2C_1_SDA_CLKEN();
146  NVIC_SetPriority(I2C_1_ERR_IRQ, I2C_IRQ_PRIO);
147  NVIC_EnableIRQ(I2C_1_ERR_IRQ);
148  break;
149 #endif
150 
151  default:
152  return -1;
153  }
154 
155  /* configure pins */
156  _pin_config(port_scl, port_sda, pin_scl, pin_sda);
157 
158  /* configure device */
159  _i2c_init(i2c, presc, scll, sclh, sdadel, scldel, timing);
160 
161  return 0;
162 }

164 static void _i2c_init(I2C_TypeDef *i2c, uint32_t presc, uint32_t scll,
165  uint32_t sclh, uint32_t sdadel, uint32_t scldel,
166  uint32_t timing)
167 {
168  /* disable device */
169  i2c->CR1 &= ~(I2C_CR1_PE);
170 
171  /* configure analog noise filter */
172  i2c->CR1 |= I2C_CR1_ANFOFF;
173 
174  /* configure digital noise filter */
175  i2c->CR1 |= I2C_CR1_DNF;
176 
177  /* set timing registers */
178  i2c->TIMINGR = timing;
179 
180  /* configure clock stretching */
181  i2c->CR1 &= ~(I2C_CR1_NOSTRETCH);
182 
183  /* enable device */
184  i2c->CR1 |= I2C_CR1_PE;
185 }

187 static void _pin_config(GPIO_TypeDef *port_scl, GPIO_TypeDef *port_sda,
188  int pin_scl, int pin_sda)
189 {
190  /* Set GPIOs to AF mode */
191  port_scl->MODER &= ~(3 << (2 * pin_scl));
192  port_scl->MODER |= (2 << (2 * pin_scl));
193  port_sda->MODER &= ~(3 << (2 * pin_sda));
194  port_sda->MODER |= (2 << (2 * pin_sda));
195 
196  /* Set speed high*/
197  port_scl->OSPEEDR |= (3 << (2 * pin_scl));
198  port_sda->OSPEEDR |= (3 << (2 * pin_sda));
199 
200  /* Set to push-pull configuration open drain*/
201  port_scl->OTYPER |= (1 << pin_scl);
202  port_sda->OTYPER |= (1 << pin_sda);
203 
204  /* Enable pull-up resistors */
205  port_scl->PUPDR &= ~(3 << (2 * pin_scl));
206  port_scl->PUPDR |= (1 << (2 * pin_scl));
207  port_sda->PUPDR &= ~(3 << (2 * pin_sda));
208  port_sda->PUPDR |= (1 << (2 * pin_sda));
209 
210  /* Configure GPIOs to for the I2C alternate function */
211  if (pin_scl < 8) {
212  port_scl->AFR[0] &= ~(0xf << (4 * pin_scl));
213  port_scl->AFR[0] |= (I2C_0_SCL_AF << (4 * pin_scl));
214  }
215  else {
216  port_scl->AFR[1] &= ~(0xf << (4 * (pin_scl - 8)));
217  port_scl->AFR[1] |= (I2C_0_SCL_AF << (4 * (pin_scl - 8)));
218  }
219 
220  if (pin_sda < 8) {
221  port_sda->AFR[0] &= ~(0xf << (4 * pin_sda));
222  port_sda->AFR[0] |= (I2C_0_SDA_AF << (4 * pin_sda));
223  }
224  else {
225  port_sda->AFR[1] &= ~(0xf << (4 * (pin_sda - 8)));
226  port_sda->AFR[1] |= (I2C_0_SDA_AF << (4 * (pin_sda - 8)));
227  }
228 }

445 static void _start(I2C_TypeDef *dev, uint8_t address, uint8_t length, uint8_t rw_flag)
446 {
447  /* set address mode to 7-bit */
448  dev->CR2 &= ~(I2C_CR2_ADD10);
449 
450  /* set slave address */
451  dev->CR2 &= ~(I2C_CR2_SADD);
452  dev->CR2 |= (address << 1);
453 
454  /* set transfer direction */
455  dev->CR2 &= ~(I2C_CR2_RD_WRN);
456  dev->CR2 |= (rw_flag << 10);
457 
458  /* set number of bytes */
459  dev->CR2 &= ~(I2C_CR2_NBYTES);
460  dev->CR2 |= (length << 16);
461 
462  /* configure autoend configuration */
463  dev->CR2 &= ~(I2C_CR2_AUTOEND);
464 
465  /* generate start condition */
466  DEBUG("Generate start condition\n");
467  dev->CR2 |= I2C_CR2_START;
468 }

470 static inline void _read(I2C_TypeDef *dev, uint8_t *data, int length)
471 {
472  for (int i = 0; i < length; i++) {
473  /* wait for transfer to finish */
474  DEBUG("Waiting for DR to be full\n");
475  while (!(dev->ISR & I2C_ISR_RXNE)) {}
476  DEBUG("DR is now full\n");
477 
478  /* read data from data register */
479  data[i] = dev->RXDR;
480  DEBUG("Read byte %i from DR\n", i);
481  }
482 }

494  DEBUG("Sending data\n");
