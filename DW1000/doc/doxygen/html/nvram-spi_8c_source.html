126 static int nvram_spi_write(nvram_t *dev, const uint8_t *src, uint32_t dst, size_t len)
127 {
128  nvram_spi_params_t *spi_dev = (nvram_spi_params_t *) dev->extra;
129  union {
130  uint32_t u32;
131  char c[4];
132  } addr;
133 
134  /* Address is expected by the device as big-endian, i.e. network byte order,
135  * we utilize the network byte order macros here. */
136  addr.u32 = HTONL(dst);
137 
138  /* Acquire exclusive bus access while configuring clock and mode */
139  spi_acquire(spi_dev->spi, spi_dev->cs, SPI_MODE_0, spi_dev->clk);
140  /* Enable writes */
141  spi_transfer_byte(spi_dev->spi, spi_dev->cs, false, NVRAM_SPI_CMD_WREN);
142  /* Make sure we have a minimum gap between transfers */
143  xtimer_spin(NVRAM_SPI_CS_TOGGLE_TICKS);
144  /* Write command and address */
145  spi_transfer_byte(spi_dev->spi, spi_dev->cs, true, NVRAM_SPI_CMD_WRITE);
146  spi_transfer_bytes(spi_dev->spi, spi_dev->cs, true,
147  &addr.c[sizeof(addr.c) - spi_dev->address_count], NULL,
148  spi_dev->address_count);
149  /* Write data (we still hold the CS line low in between) */
150  spi_transfer_bytes(spi_dev->spi, spi_dev->cs, false, src, NULL, len);
151  /* Release exclusive bus access */
152  spi_release(spi_dev->spi);
153 
154  return (int)len;
155 }

185 static int nvram_spi_write_9bit_addr(nvram_t *dev, const uint8_t *src, uint32_t dst, size_t len)
186 {
187  nvram_spi_params_t *spi_dev = (nvram_spi_params_t *) dev->extra;
188  uint8_t cmd;
189  uint8_t addr;
190  cmd = NVRAM_SPI_CMD_WRITE;
191 
192  /* The upper address bit is mixed into the command byte on certain devices,
193  * probably just to save a byte in the SPI transfer protocol. */
194  if (dst > 0xff) {
195  cmd |= 0x08;
196  }
197  /* LSB of address */
198  addr = (dst & 0xff);
199 
200  spi_acquire(spi_dev->spi, spi_dev->cs, SPI_MODE_0, spi_dev->clk);
201  /* Enable writes */
202  spi_transfer_byte(spi_dev->spi, spi_dev->cs, false, NVRAM_SPI_CMD_WREN);
203  /* Insert needed delay between transactions */
204  xtimer_spin(NVRAM_SPI_CS_TOGGLE_TICKS);
205  /* Write command and address */
206  spi_transfer_byte(spi_dev->spi, spi_dev->cs, true, cmd);
207  spi_transfer_byte(spi_dev->spi, spi_dev->cs, true, addr);
208  /* Keep holding CS and write data */
209  spi_transfer_bytes(spi_dev->spi, spi_dev->cs, false, src, NULL, len);
210  spi_release(spi_dev->spi);
211 
212  /* status contains the number of bytes actually written to the SPI bus. */
213  return (int)len;
214 }
