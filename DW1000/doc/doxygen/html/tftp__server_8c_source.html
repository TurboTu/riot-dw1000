85 static int _tftp_server_data_cb(uint32_t offset, void *data, size_t data_len)
86 {
87  char *c = (char *) data;
88 
89  /* if we are reading return the part of the data that is being requested */
90  if (_tftp_action == TFTP_READ) {
91  /* calculate the length of the data block to transfer */
92  if (offset + data_len > sizeof(_tftp_server_hello)) {
93  data_len -= (offset + data_len) - sizeof(_tftp_server_hello);
94  }
95 
96  /* copy the block to the output buffer */
97  memcpy(data, _tftp_server_hello + offset, data_len);
98  }
99  else {
100  /* we received a data block which we output to the console */
101  printf("\n -- SERVER DATA --\n%.*s\n -- SERVER DATA --\n", (int)data_len, c);
102  }
103 
104  /* return the length of the data block */
