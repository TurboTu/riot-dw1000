19 void lpc2387_pclk_scale(uint32_t source, uint32_t target, uint32_t *pclksel, uint32_t *prescale)
20 {
21  uint32_t pclkdiv;
22  *prescale = source / target;
23 
24  if ((*prescale % 16) == 0) {
25  *pclksel = 3;
26  pclkdiv = 8;
27  }
28  else if ((*prescale % 8) == 0) {
29  *pclksel = 0;
30  pclkdiv = 4;
31  }
32  else if ((*prescale % 4) == 0) {
33  *pclksel = 2;
34  pclkdiv = 2;
35  }
36  else {
37  *pclksel = 1;
38  pclkdiv = 1;
39  }
40 
41  *prescale /= pclkdiv;
42

44  (*prescale)++;
45  }
46 }

71 {
