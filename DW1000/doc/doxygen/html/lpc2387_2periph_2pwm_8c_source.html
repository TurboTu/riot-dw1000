37 uint32_t pwm_init(pwm_t dev, pwm_mode_t mode, uint32_t freq, uint16_t res)
38 {
39  if ((dev != PWM_DEV(0)) && (mode != PWM_LEFT)) {
40  return 0;
41  }
42 
43  /* select function PWM[3] for pins */
44  PWM_PORT &= ~((3 << PWM_CH0_PIN * 2) |
45  (3 << PWM_CH1_PIN * 2) |
46  (3 << PWM_CH2_PIN * 2));
47  PWM_PORT |= (PWM_FUNC << PWM_CH0_PIN * 2) |
48  (PWM_FUNC << PWM_CH1_PIN * 2) |
49  (PWM_FUNC << PWM_CH2_PIN * 2);
50 
51  /* power on PWM1 */
52  pwm_poweron(dev);
53 
54  /* select PWM1 clock */
55  PCLKSEL0 &= ~(BIT13);
56  PCLKSEL0 |= (BIT12);
57 
58  /* reset PWM1s counter */
59  PWM1TCR = BIT1;
60 
61  /* set prescaler */
62  PWM1PR = (CLOCK_CORECLOCK / (freq * res)) - 1;
63 
64  /* set match register */
65  PWM1MR0 = res;
66  PWM_CH0_MR = 0;
67  PWM_CH1_MR = 0;
68  PWM_CH2_MR = 0;
69 
70  /* reset timer counter on MR0 match */
71  PWM1MCR = BIT1;
72 
73  /* enable PWM1 channel 3, 4 and 5 */
74  PWM1PCR = (1 << (8 + PWM_CH0)) | (1 << (8 + PWM_CH1)) | (1 << (8 + PWM_CH2));
75 
76  /* enable PWM1 timer in PWM mode */
77  PWM1TCR = BIT0 + BIT3;
78 
79  /* update match registers */
80  PWM1LER = BIT0 | (1 << PWM_CH0) | (1 << PWM_CH1) | (1 << PWM_CH2);
81 
82  return freq;
83 }
