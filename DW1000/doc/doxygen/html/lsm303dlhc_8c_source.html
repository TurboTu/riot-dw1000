28 int lsm303dlhc_init(lsm303dlhc_t *dev, i2c_t i2c, gpio_t acc_pin, gpio_t mag_pin,
29  uint8_t acc_address,
30  lsm303dlhc_acc_sample_rate_t acc_sample_rate,
31  lsm303dlhc_acc_scale_t acc_scale,
32  uint8_t mag_address,
33  lsm303dlhc_mag_sample_rate_t mag_sample_rate,
34  lsm303dlhc_mag_gain_t mag_gain)
35 {
36  int res;
37  uint8_t tmp;
38 
39  dev->i2c = i2c;
40  dev->acc_address = acc_address;
41  dev->mag_address = mag_address;
42  dev->acc_pin = acc_pin;
43  dev->mag_pin = mag_pin;
44  dev->acc_scale = acc_scale;
45  dev->mag_gain = mag_gain;
46 
47  /* Acquire exclusive access to the bus. */
48  i2c_acquire(dev->i2c);
49  i2c_init_master(i2c, I2C_SPEED_NORMAL);
50 
51  DEBUG("lsm303dlhc reboot...");
52  res = i2c_write_reg(dev->i2c, dev->acc_address,
53  LSM303DLHC_REG_CTRL5_A, LSM303DLHC_REG_CTRL5_A_BOOT);
54  /* Release the bus for other threads. */
55  i2c_release(dev->i2c);
56  DEBUG("[OK]\n");
57 
58  /* configure accelerometer */
59  /* enable all three axis and set sample rate */
60  tmp = (LSM303DLHC_CTRL1_A_XEN
61  | LSM303DLHC_CTRL1_A_YEN
62  | LSM303DLHC_CTRL1_A_ZEN
63  | acc_sample_rate);
64  i2c_acquire(dev->i2c);
65  res += i2c_write_reg(dev->i2c, dev->acc_address,
66  LSM303DLHC_REG_CTRL1_A, tmp);
67  /* update on read, MSB @ low address, scale and high-resolution */
68  tmp = (acc_scale | LSM303DLHC_CTRL4_A_HR);
69  res += i2c_write_reg(dev->i2c, dev->acc_address,
70  LSM303DLHC_REG_CTRL4_A, tmp);
71  /* no interrupt generation */
72  res += i2c_write_reg(dev->i2c, dev->acc_address,
73  LSM303DLHC_REG_CTRL3_A, LSM303DLHC_CTRL3_A_I1_NONE);
74  /* configure acc data ready pin */
75  gpio_init(acc_pin, GPIO_IN);
76 
77  /* configure magnetometer and temperature */
78  /* enable temperature output and set sample rate */
79  tmp = LSM303DLHC_TEMP_EN | mag_sample_rate;
80  res += i2c_write_reg(dev->i2c, dev->mag_address,
81  LSM303DLHC_REG_CRA_M, tmp);
82  /* configure z-axis gain */
83  res += i2c_write_reg(dev->i2c, dev->mag_address,
84  LSM303DLHC_REG_CRB_M, mag_gain);
85  /* set continuous mode */
86  res += i2c_write_reg(dev->i2c, dev->mag_address,
87  LSM303DLHC_REG_MR_M, LSM303DLHC_MAG_MODE_CONTINUOUS);
88  i2c_release(dev->i2c);
89  /* configure mag data ready pin */
90  gpio_init(mag_pin, GPIO_IN);
91 
92  return (res < 7) ? -1 : 0;
93 }

95 int lsm303dlhc_read_acc(lsm303dlhc_t *dev, lsm303dlhc_3d_data_t *data)
96 {
97  int res;
98  uint8_t tmp;
99 
100  i2c_acquire(dev->i2c);
101  i2c_read_reg(dev->i2c, dev->acc_address, LSM303DLHC_REG_STATUS_A, &tmp);
102  DEBUG("lsm303dlhc status: %x\n", tmp);
103  DEBUG("lsm303dlhc: wait for acc values ... ");
104 
105  res = i2c_read_reg(dev->i2c, dev->acc_address,
106  LSM303DLHC_REG_OUT_X_L_A, &tmp);
107  data->x_axis = tmp;
108  res += i2c_read_reg(dev->i2c, dev->acc_address,
109  LSM303DLHC_REG_OUT_X_H_A, &tmp);
110  data->x_axis |= tmp<<8;
111  res += i2c_read_reg(dev->i2c, dev->acc_address,
112  LSM303DLHC_REG_OUT_Y_L_A, &tmp);
113  data->y_axis = tmp;
114  res += i2c_read_reg(dev->i2c, dev->acc_address,
115  LSM303DLHC_REG_OUT_Y_H_A, &tmp);
116  data->y_axis |= tmp<<8;
117  res += i2c_read_reg(dev->i2c, dev->acc_address,
118  LSM303DLHC_REG_OUT_Z_L_A, &tmp);
119  data->z_axis = tmp;
120  res += i2c_read_reg(dev->i2c, dev->acc_address,
121  LSM303DLHC_REG_OUT_Z_H_A, &tmp);
122  data->z_axis |= tmp<<8;
123  i2c_release(dev->i2c);
124  DEBUG("read ... ");
125 
126  data->x_axis = data->x_axis>>4;
127  data->y_axis = data->y_axis>>4;
128  data->z_axis = data->z_axis>>4;
129 
130  if (res < 6) {
131  DEBUG("[!!failed!!]\n");
132  return -1;
133  }
134  DEBUG("[done]\n");
135 
136  return 0;
137 }

139 int lsm303dlhc_read_mag(lsm303dlhc_t *dev, lsm303dlhc_3d_data_t *data)
140 {
141  int res;
142 
143  DEBUG("lsm303dlhc: wait for mag values... ");
144  while (gpio_read(dev->mag_pin) == 0){}
145 
146  DEBUG("read ... ");
147 
148  i2c_acquire(dev->i2c);
149  res = i2c_read_regs(dev->i2c, dev->mag_address,
150  LSM303DLHC_REG_OUT_X_H_M, data, 6);
151  i2c_release(dev->i2c);
152 
153  if (res < 6) {
154  DEBUG("[!!failed!!]\n");
155  return -1;
156  }
157  DEBUG("[done]\n");
158 
159  /* interchange y and z axis and fix endiness */
160  int16_t tmp = data->y_axis;
161  data->x_axis = ((data->x_axis<<8)|((data->x_axis>>8)&0xff));
162  data->y_axis = ((data->z_axis<<8)|((data->z_axis>>8)&0xff));
163  data->z_axis = ((tmp<<8)|((tmp>>8)&0xff));
164 
165  /* compensate z-axis sensitivity */
166  /* gain is currently hardcoded to LSM303DLHC_GAIN_5 */
167  data->z_axis = ((data->z_axis * 400) / 355);
168 
169  return 0;
170 }

173 {

175 
176  i2c_acquire(dev->i2c);

178  i2c_release(dev->i2c);
179

183

211  | LSM303DLHC_CTRL1_A_YEN

218  res += i2c_write_reg(dev->i2c, dev->acc_address, LSM303DLHC_REG_CTRL4_A, tmp);
219  res += i2c_write_reg(dev->i2c, dev->acc_address, LSM303DLHC_REG_CTRL3_A, LSM303DLHC_CTRL3_A_I1_DRDY1);
220  gpio_init(dev->acc_pin, GPIO_IN);

224 
225  res += i2c_write_reg(dev->i2c, dev->mag_address,
