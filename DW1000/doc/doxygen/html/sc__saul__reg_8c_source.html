98 static void write(int argc, char **argv)
99 {
100  int num, dim;
101  saul_reg_t *dev;
102  phydat_t data;
103 
104  if (argc < 4) {
105  printf("usage: %s %s <device id> <value 0> [<value 1> [<value 2]]\n",
106  argv[0], argv[1]);
107  return;
108  }
109  num = atoi(argv[2]);
110  dev = saul_reg_find_nth(num);
111  if (dev == NULL) {
112  puts("error: undefined device given");
113  return;
114  }
115  /* parse value(s) */
116  memset(&data, 0, sizeof(data));
117  dim = ((argc - 3) > (int)PHYDAT_DIM) ? (int)PHYDAT_DIM : (argc - 3);
118  for (int i = 0; i < dim; i++) {
119  data.val[i] = (int16_t)atoi(argv[i + 3]);
120  }
121  /* print values before writing */
122  printf("Writing to device #%i - %s\n", num, dev->name);
123  phydat_dump(&data, dim);
124  /* write values to device */
125  dim = saul_reg_write(dev, &data);
126  if (dim <= 0) {
127  if (dim == -ENOTSUP) {
128  printf("error: device #%i is not writable\n", num);
129  }
130  else {
131  printf("error: failure to write to device #%i\n", num);
132  }
133  return;
134  }
135  printf("data successfully written to device #%i\n", num);
136 }
