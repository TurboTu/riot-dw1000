169 static uint_fast8_t i2c_ctrl_blocking(uint_fast8_t flags)
170 {
171 #ifdef MODULE_XTIMER
172  const unsigned int xtimer_timeout = 3 * (DATA_BITS + ACK_BITS) * US_PER_SEC / speed_hz;
173 #endif
174 
175  mutex_trylock(&i2c_wait_mutex);
176 
177  assert(I2CM_IMR & 1);
178  i2cm_ctrl_write(flags);
179 
180 #ifdef MODULE_XTIMER
181  /* Set a timeout at double the expected time to transmit a byte: */
182  xtimer_t xtimer = {.callback = callback};
183  xtimer_set(&xtimer, xtimer_timeout);
184 #endif
185 
186  mutex_lock(&i2c_wait_mutex);
187 
188 #ifdef MODULE_XTIMER
189  xtimer_remove(&xtimer);
190 #endif
191 
192  if (I2CM_STAT & BUSY) {
193  /* If the controller is still busy, it probably will be forever */
194 #ifdef MODULE_XTIMER
195  DEBUG("Master is still BUSY after %u usec. Resetting.\n", xtimer_timeout);
196 #endif
197  cc2538_i2c_init_master(speed_hz);
198  }
199 
200  WARN_IF(I2CM_STAT & BUSY);
201

216 void cc2538_i2c_init_master(uint32_t speed_hz)
217 {
218  SYS_CTRL_RCGCI2C |= 1;
219  SYS_CTRL_SCGCI2C |= 1;
220  SYS_CTRL_DCGCI2C |= 1;
222  /* Reset I2C peripheral */
223  SYS_CTRL_SRI2C |= 1;
224 
225 #ifdef MODULE_XTIMER
226  xtimer_usleep(50);
227 #else
228  thread_yield();
229 #endif
230 
231  SYS_CTRL_SRI2C &= ~1;
232 
233  /* Clear all pin override flags except PUE (Pull-Up Enable) */
234  IOC_PXX_OVER[I2C_0_SCL_PIN] &= IOC_OVERRIDE_PUE;
235  IOC_PXX_OVER[I2C_0_SDA_PIN] &= IOC_OVERRIDE_PUE;
236 
237  IOC_PXX_SEL[I2C_0_SCL_PIN] = I2C_CMSSCL;
238  IOC_PXX_SEL[I2C_0_SDA_PIN] = I2C_CMSSDA;
239 
240  IOC_I2CMSSCL = I2C_0_SCL_PIN;
241  IOC_I2CMSSDA = I2C_0_SDA_PIN;
242 
243  gpio_hardware_control(I2C_0_SCL_PIN);
244  gpio_hardware_control(I2C_0_SDA_PIN);
245 
246  /* Initialize the I2C master by setting the Master Function Enable bit */
247  I2CM_CR |= MFE;
248 
249  /* Set the SCL clock speed */
250  uint32_t ps = sys_clock_freq();
251  uint32_t denom = 2 * (SCL_LP + SCL_HP) * speed_hz;
252  ps += denom / 2;
253  ps /= denom;
254  I2CM_TPR = ps - 1;
255 
256  /* Enable I2C master interrupts */
257  NVIC_SetPriority(I2C_IRQn, I2C_IRQ_PRIO);
258  NVIC_EnableIRQ(I2C_IRQn);
259 
260  i2cm_ctrl_write(STOP);
261 
262  /* Enable I2C master interrupts */
263  I2CM_IMR = 1;
264 }

354 static int i2c_read_bytes_dumb(uint8_t address, uint8_t *data, int length)
355 {
356  int n = 0;
357  uint_fast8_t stat;
358 
359  switch (length) {
360  case 0:
361  break;
362 
363  case 1:
364  if (i2c_busy()) {
365  break;
366  }
367 
368  I2CM_SA = (address << 1) | RS;
369  stat = i2c_ctrl_blocking(STOP | START | RUN);
370  if (stat & ANY_ERROR) {
371  break;
372  }
373  data[n] = I2CM_DR;
374  n++;
375  break;
376 
377  default:
378  if (i2c_busy()) {
379  break;
380  }
381 
382  I2CM_SA = (address << 1) | RS;
383  stat = i2c_ctrl_blocking(ACK | START | RUN);
384 
385  if (stat & ARBLST) {
386  break;
387  }
388  else if (stat & ANY_ERROR) {
389  i2cm_ctrl_write(STOP);
390  break;
391  }
392 
393  data[n] = I2CM_DR;
394  n++;
395 
396  while (n < length) {
397  stat = i2c_ctrl_blocking((n < length - 1) ? (ACK | RUN) : (STOP | RUN));
398 
399  if (stat & ARBLST) {
400  break;
401  }
402  else if (stat & ANY_ERROR) {
403  i2cm_ctrl_write(STOP);
404  break;
405  }
406 
407  data[n] = I2CM_DR;
408  n++;
409  }
410 
411  break;
412  }
413 
414  return n;
415 }
