36 nhdp_addr_t *nhdp_addr_db_get_address(uint8_t *addr, size_t addr_size, uint8_t addr_type)
37 {
38  nhdp_addr_t *addr_elt;
39 
40  mutex_lock(&mtx_addr_access);
41 
42  LL_FOREACH(nhdp_addr_db_head, addr_elt) {
43  if ((addr_elt->addr_size == addr_size) && (addr_elt->addr_type == addr_type)) {
44  if (memcmp(addr_elt->addr, addr, addr_size) == 0) {
45  /* Found a matching entry */
46  break;
47  }
48  }
49  }
50 
51  if (!addr_elt) {
52  /* No matching entry, create a new one */
53  addr_elt = (nhdp_addr_t *) malloc(sizeof(nhdp_addr_t));
54 
55  if (!addr_elt) {
56  /* Insufficient memory */
57  return NULL;
58  }
59 
60  /* Allocate space for the address */
61  addr_elt->addr = (uint8_t *) malloc(addr_size * sizeof(uint8_t));
62 
63  if (!addr_elt->addr) {
64  /* Insufficient memory */
65  free(addr_elt);
66  return NULL;
67  }
68 
69  memcpy(addr_elt->addr, addr, addr_size);
70  addr_elt->addr_size = addr_size;
71  addr_elt->addr_type = addr_type;
72  addr_elt->usg_count = 0;
73  addr_elt->in_tmp_table = NHDP_ADDR_TMP_NONE;
74  addr_elt->tmp_metric_val = NHDP_METRIC_UNKNOWN;
75  LL_PREPEND(nhdp_addr_db_head, addr_elt);
76  }
77 
78  addr_elt->usg_count++;
79 
80  mutex_unlock(&mtx_addr_access);
81 
82  return addr_elt;
83 }

85 void nhdp_decrement_addr_usage(nhdp_addr_t *addr)
86 {
87  mutex_lock(&mtx_addr_access);
88 
89  /* Decrement usage count and delete address if no longer used */
90  if (addr) {

92  if (addr->usg_count == 0) {
93  /* Free address space if address is no longer used */
94  LL_DELETE(nhdp_addr_db_head, addr);
95  free(addr->addr);
96  free(addr);
97  }
98  }
99  mutex_unlock(&mtx_addr_access);
