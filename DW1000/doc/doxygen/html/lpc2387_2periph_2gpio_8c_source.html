95 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
96  gpio_cb_t cb, void *arg)
97 {
98  (void)mode;
99 
100  DEBUG("gpio_init_int(): pin %u\n", pin);
101  int isr_map_entry;
102 
103  /* check if interrupt is possible for this pin */
104  if ((isr_map_entry = _isr_map_entry(pin)) == -1) {
105  DEBUG("gpio_init_int(): pin %u cannot be used to generate interrupts.\n", pin);
106  return -1;
107  }
108 
109  /* find free isr state entry */
110  int _state_index = _gpio_isr_map[isr_map_entry];
111 
112  if (_state_index == 0xff) {
113  _state_index = bf_get_unset(_gpio_config_bitfield, GPIO_NUM_ISR);
114  _gpio_isr_map[isr_map_entry] = _state_index;
115  DEBUG("gpio_init_int(): pin has state_index=%i isr_map_entry=%i\n", _state_index, isr_map_entry);
116  }
117

119 #ifdef DEVELHELP
120  puts("lpc2387: gpio: warning: no free gpio callback state!");
121 #endif
122  return -1;
123  }
124 
125  /* store callback */
126  _gpio_states[_state_index].cb = cb;
127  _gpio_states[_state_index].arg = arg;
128 
129  extern void GPIO_IRQHandler(void);

132  bf_set(_gpio_falling, _state_index);
133  }

136  }

154 static void _gpio_configure(gpio_t pin, unsigned rising, unsigned falling)
155 {
156  unsigned _pin = pin & 31;
157  unsigned port = pin >> 5;
158 
159  /* set irq settings */
160  volatile unsigned long *en_f;
161  volatile unsigned long *en_r;
162  volatile unsigned long *en_clr;
163 
164  if (!port) {
165  en_f = &IO0_INT_EN_F;
166  en_r = &IO0_INT_EN_R;
167  en_clr = &IO0_INT_CLR;
168  }
169  else {
170  en_f = &IO2_INT_EN_F;
171  en_r = &IO2_INT_EN_R;
172  en_clr = &IO2_INT_CLR;
173  }
174 
175  /* configure irq */
176  unsigned int bit = 0x1 << _pin;
177 
178  unsigned state = irq_disable();
179 
180  *en_clr |= bit; /* clear interrupt */
181 
182  if (falling) {
183  *en_f |= bit; /* enable falling edge */
184  }
185  else {
186  *en_f &= ~bit; /* disable falling edge */
187  }
188 
189  if (rising) {
190  *en_r |= bit; /* enable rising edge */
191  }
192  else {
193  *en_r &= ~bit; /* disable rising edge */
194  }
195 
196  irq_restore(state);
197 }

266 static void test_irq(int port, unsigned long f_mask, unsigned long r_mask)
267 {
268  /* Test each bit of rising and falling masks, if set trigger interrupt
269  * on corresponding device */
270  unsigned bit = 0x1;
271  int n = 0;
272  while (bit) {
273  if ((r_mask & bit) | (f_mask & bit)) {
274  int _state_index = _gpio_isr_map[n + (port<<1)];
275  if (_state_index != 0xff) {
276  _gpio_states[_state_index].cb(_gpio_states[_state_index].arg);
277  }
278  }
279 
280  bit <<= 1;
281  n++;
282  }
283 }
