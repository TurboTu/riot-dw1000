154 static int _sock_ep_to_netconn_pars(const struct _sock_tl_ep *local,
155  const struct _sock_tl_ep *remote,
156  ip_addr_t *local_addr, u16_t *local_port,
157  ip_addr_t *remote_addr, u16_t *remote_port,
158  int *type)
159 {
160  bool res = 0;
161  int family = AF_UNSPEC;
162  uint16_t netif = SOCK_ADDR_ANY_NETIF;
163 
164  if ((local != NULL) && (remote != NULL) &&
165  (remote->netif != SOCK_ADDR_ANY_NETIF) &&
166  (local->netif != SOCK_ADDR_ANY_NETIF) &&
167  (remote->netif != local->netif)) {
168  return -EINVAL;
169  }
170 
171 #if LWIP_IPV6
172  *type &= ~NETCONN_TYPE_IPV6;
173 #else
174  (void)type; /* is read but not set => compiler complains */
175 #endif
176  if (local != NULL) {
177  if (_af_not_supported(local->family)) {
178  return -EAFNOSUPPORT;
179  }
180  if (local->netif != SOCK_ADDR_ANY_NETIF) {
181  netif = local->netif;
182  }
183  family = local->family;
184  _set_port(*local_port, local, *type);
185  }
186  if (remote != NULL) {
187  if (_af_not_supported(remote->family) ||
188  ((local != NULL) && (local->family != remote->family))) {
189  return -EAFNOSUPPORT;
190  }
191  if ((remote->netif != SOCK_ADDR_ANY_NETIF) &&
192  (local != NULL) && (local->netif != SOCK_ADDR_ANY_NETIF)) {
193  netif = remote->netif;
194  }
195  family = remote->family;
196  memcpy(remote_addr, &remote->addr, sizeof(remote->addr));
197 #if LWIP_IPV6 && LWIP_IPV4
198  remote_addr->type = lwip_af_to_ip_addr_type(family);
199 #endif
200  if (ip_addr_isany(remote_addr)) {
201  return -EINVAL;
202  }
203  _set_port(*remote_port, remote, *type);
204  }
205 
206 #if LWIP_IPV6
207  if (family == AF_INET6) {
208  *type |= NETCONN_TYPE_IPV6;
209  }
210 #endif
211 
212  if (netif != SOCK_ADDR_ANY_NETIF) {
213  if (_ep_isany(local)) {
214  const ip_addr_t *tmp = _netif_to_bind_addr(family, netif);
215  if (tmp != NULL) {
216  memcpy(local_addr, tmp, sizeof(ip_addr_t));
217  res = 1;
218  }
219  else {
220  /* netif was not a valid interface */
221  return -EINVAL;
222  }
223  }
224  /* case (local == NULL) is included in _ep_isany() */
225  /* cast to ip_addr_t alright, since type field is never used */
226  else if (_addr_on_netif(family, netif, (ip_addr_t *)&local->addr)) {
227  memcpy(local_addr, &local->addr, sizeof(local->addr));
228  res = 1;
229  }
230  else {
231  return -EINVAL;
232  }
233  }
234  else if (local != NULL) {
235  memcpy(local_addr, &local->addr, sizeof(local->addr));
236  res = 1;
237  }
238 #if LWIP_IPV6 && LWIP_IPV4
239  if (local_addr != NULL) {
240  local_addr->type = lwip_af_to_ip_addr_type(family);
241  }
242 #endif
243 
244  return res;
245 }

264 int lwip_sock_create(struct netconn **conn, const struct _sock_tl_ep *local,
265  const struct _sock_tl_ep *remote, int proto,
266  uint16_t flags, int type)
267 {
268  assert(conn != NULL);
269 #if LWIP_IPV6
270  assert(!(type & NETCONN_TYPE_IPV6));
271 #endif
272 
273  ip_addr_t local_addr, remote_addr;
274  u16_t local_port = 0, remote_port = 0; /* 0 is used by lwIP to
275  * automatically generate
276  * port */
277  /* convert parameters */
278  int bind = _sock_ep_to_netconn_pars(local, remote, &local_addr, &local_port,
279  &remote_addr, &remote_port, &type);
280 
281  /* error occurred during parameter conversion */
282  if (bind < 0) {
283  return bind;
284  }
285  if ((remote != NULL) && ip_addr_isany_val(remote_addr)) {
286  return -EINVAL;
287  }
288  /* if local or remote parameters are given */
289  else if ((local != NULL) || (remote != NULL)) {
290  int res = 0;
291  if ((res = _create(type, proto, flags, conn)) < 0) {
292  return res;
293  }
294  /* if parameters (local->netif, remote->netif, local->addr or
295  * local->port) demand binding */
296  if (bind) {
297  switch (netconn_bind(*conn, &local_addr, local_port)) {
298  case ERR_USE:
299  res = -EADDRINUSE;
300  break;
301  case ERR_VAL:
302  res = -EINVAL;
303  break;
304  default:
305  break;
306  }
307  if (res < 0) {
308  netconn_delete(*conn);
309  return res;
310  }
311  }
312  if (remote != NULL) {
313  switch (netconn_connect(*conn, &remote_addr, remote_port)) {
314  case ERR_USE:
315  res = -EADDRINUSE;
316  break;
317  case ERR_VAL:
318  res = -EINVAL;
319  break;
320  default:
321  break;
322  }
323  if (res < 0) {
324  netconn_delete(*conn);
325  return res;
326  }
327  }
328  }
329  return 0;
330 }

409

446  int proto, const struct _sock_tl_ep *remote, int type)
447 {
448  ip_addr_t remote_addr;
449  struct netconn *tmp;
450  struct netbuf *buf;
451  int res;
452  err_t err;
453  u16_t remote_port = 0;
454 
455 #if LWIP_IPV6
456  assert(!(type & NETCONN_TYPE_IPV6));
457 #endif
458  if (remote != NULL) {
459  if ((res = _sock_ep_to_netconn_pars(NULL, remote, NULL, NULL,
460  &remote_addr, &remote_port,
461  &type)) < 0) {
462  return res;
463  }
464  if (ip_addr_isany_val(remote_addr)) {
465  return -EINVAL;
466  }
467  }
468 
469  buf = netbuf_new();
470  if ((buf == NULL) || (netbuf_alloc(buf, len) == NULL) ||
471  (netbuf_take(buf, data, len) != ERR_OK)) {
472  netbuf_delete(buf);
473  return -ENOMEM;
474  }
475  if (((conn == NULL) || (*conn == NULL)) && (remote != NULL)) {
476  if ((res = _create(type, proto, 0, &tmp)) < 0) {
477  netbuf_delete(buf);
478  return res;
479  }
480  }
481  else if (*conn != NULL) {
482  ip_addr_t addr;
483  u16_t port;
484 
485  if (((remote != NULL) &&
486  (remote->netif != SOCK_ADDR_ANY_NETIF) &&
487  (netconn_getaddr(*conn, &addr, &port, 1) == 0) &&
488  (remote->netif != lwip_sock_bind_addr_to_netif(&addr)))) {
489  return -EINVAL;
490  }
491  tmp = *conn;
492  }
493  else {
494  netbuf_delete(buf);
495  return -ENOTCONN;
496  }
497  if (remote != NULL) {
498  err = netconn_sendto(tmp, buf, &remote_addr, remote_port);
499  }
500  else {
501  err = netconn_send(tmp, buf);
502  }
503  switch (err) {
504  case ERR_OK:
505  res = len;
506  if (conn != NULL) {
507  *conn = tmp;
508  }
509  break;
510  case ERR_BUF:
511  case ERR_MEM:
512  res = -ENOMEM;
513  break;
514  case ERR_RTE:
515  case ERR_IF:
516  res = -EHOSTUNREACH;
517  break;
518  case ERR_VAL:
519  default:
520  res = -EINVAL;
521  break;
522  }
523  netbuf_delete(buf);
