79 static int udp_send(char *addr_str, char *port_str, char *data, unsigned int num,
80  unsigned int delay)
81 {
82  ipv6_addr_t src = IPV6_ADDR_UNSPECIFIED, dst;
83  uint16_t port;
84  uint8_t byte_data[strlen(data) / 2];
85  size_t data_len;
86 
87  /* parse destination address */
88  if (ipv6_addr_from_str(&dst, addr_str) == NULL) {
89  puts("Error: unable to parse destination address");
90  return 1;
91  }
92  /* parse port */
93  port = (uint16_t)atoi(port_str);
94  data_len = hex2ints(byte_data, data);
95  for (unsigned int i = 0; i < num; i++) {
96  if (conn_udp_sendto(byte_data, data_len, &src, sizeof(src), (struct sockaddr *)&dst,
97  sizeof(dst), AF_INET6, port, port) < 0) {
98  puts("could not send");
99  }
100  else {
101  printf("Success: send %u byte to [%s]:%" PRIu16 ")\n",
102  (unsigned)data_len, addr_str, port);
103  }
104  xtimer_usleep(delay);
105  }
106  return 0;
107 }
