213 unsigned int timer_read(tim_t dev)
214 {
215  unsigned int timer_base;
216  unsigned int timer_side = TIMER_A;
217  unsigned long long high_bits, high_bits_dup;
218  unsigned long long low_bits;
219  unsigned long long total;
220  unsigned int scaled_value;
221 
222  if (dev >= TIMER_NUMOF){
223  return -1;
224  }
225 
226  switch(dev){
227 #if TIMER_0_EN
228  case TIMER_0:
229  timer_base = WTIMER0_BASE;
230  break;
231 #endif
232 #if TIMER_1_EN
233  case TIMER_1:
234  timer_base = WTIMER1_BASE;
235  break;
236 #endif
237  default:
238  return -1; /* unreachable */
239  break;
240  }
241 
242  /* handle overflow happening between the 2 register reads */
243  do {
244  high_bits = ((unsigned long long)PRIV_TimerPrescaleSnapshotGet(timer_base, timer_side)) << 32;
245  low_bits = (unsigned long long)ROM_TimerValueGet(timer_base, timer_side);
246  high_bits_dup = ((unsigned long long)PRIV_TimerPrescaleSnapshotGet(timer_base, timer_side)) << 32;
247  } while (high_bits != high_bits_dup);
248 
249  total = high_bits + low_bits;
250  DEBUG("Combined %lx:%lx\n", (unsigned long) (total>>32), (unsigned long) (total & 0xFFFFFFFF));
251 
252  scaled_value = _llvalue_to_scaled_value(total, config[dev].divisor);
253 
254  return scaled_value;
255 }
