23 static int _send(gnrc_netdev2_t *gnrc_netdev2, gnrc_pktsnip_t *pkt)
24 {
25  cc110x_pkt_t cc110x_pkt;
26  netdev2_t *dev = gnrc_netdev2->dev;
27  netdev2_cc110x_t *netdev_cc110x = (netdev2_cc110x_t *) dev;
28  cc110x_t *cc110x = &netdev_cc110x->cc110x;
29 
30  assert(pkt != NULL);
31  assert(dev->driver == &netdev2_cc110x_driver);
32 
33  gnrc_netif_hdr_t *netif_hdr;
34  gnrc_pktsnip_t *payload;
35 
36  payload = pkt->next;
37 
38  if (pkt->type != GNRC_NETTYPE_NETIF) {
39  DEBUG("gnrc_netdev2_cc110x: First header was not generic netif header\n");
40  gnrc_pktbuf_release(pkt);
41  return -EBADMSG;
42  }
43 
44  netif_hdr = (gnrc_netif_hdr_t *) pkt->data;
45 
46  /* set up header */
47  if (netif_hdr->src_l2addr_len == 1) {
48  uint8_t *_src_addr = gnrc_netif_hdr_get_src_addr(netif_hdr);
49  cc110x_pkt.phy_src = *_src_addr;
50  }
51  else {
52  cc110x_pkt.phy_src = cc110x->radio_address;
53  }
54 
55  if (netif_hdr->flags & (GNRC_NETIF_HDR_FLAGS_BROADCAST |
56  GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
57  cc110x_pkt.address = 0;
58  }
59  else {
60  uint8_t *_dst_addr = gnrc_netif_hdr_get_dst_addr(netif_hdr);
61  cc110x_pkt.address = _dst_addr[netif_hdr->dst_l2addr_len-1];
62  }
63 
64  switch (payload->type) {
65 #ifdef MODULE_GNRC_SIXLOWPAN
66  case GNRC_NETTYPE_SIXLOWPAN:
67  cc110x_pkt.flags = 1;
68  break;
69 #endif
70  default:
71  cc110x_pkt.flags = 0;
72  }
73 
74  struct iovec vector;
75  vector.iov_base = (char*)&cc110x_pkt;
76  vector.iov_len = sizeof(cc110x_pkt_t);
77 
78  unsigned payload_len = 0;
79  uint8_t *pos = cc110x_pkt.data;
80 
81  while (payload) {
82  payload_len += payload->size;
83 
84  if (payload_len > CC110X_MAX_DATA_LENGTH) {
85  DEBUG("gnrc_netdev2_cc110x: payload length exceeds maximum"
86  "(%u>%u)\n", payload_len, CC110X_MAX_DATA_LENGTH);
87  gnrc_pktbuf_release(pkt);
88  return -EBADMSG;
89  }
90 
91  memcpy(pos, payload->data, payload->size);
92  pos += payload->size;
93  payload = payload->next;
94  }
95 
96  /* pkt has been copied into iovec, we're done with it. */
97  gnrc_pktbuf_release(pkt);
98 
99  cc110x_pkt.length = (uint8_t) payload_len + CC110X_HEADER_LENGTH;
100 
101  DEBUG("gnrc_netdev2_cc110x: sending packet from %u to %u with payload "
102  "length %u\n",
103  (unsigned)cc110x_pkt.phy_src,
104  (unsigned)cc110x_pkt.address,
105  (unsigned)cc110x_pkt.length);
106 
107  return dev->driver->send(dev, &vector, 1);
108 }

110 static gnrc_pktsnip_t *_recv(gnrc_netdev2_t *gnrc_netdev2)
111 {
112  netdev2_t *dev = gnrc_netdev2->dev;
113  cc110x_t *cc110x = &((netdev2_cc110x_t*) dev)->cc110x;
114 
115  cc110x_pkt_t *cc110x_pkt = &cc110x->pkt_buf.packet;
116 
117  int payload_length = cc110x_pkt->length - CC110X_HEADER_LENGTH;
118 
119  int nettype;
120 
121  int addr_len;
122  switch (cc110x_pkt->flags) {
123 #ifdef MODULE_GNRC_SIXLOWPAN
124  case 1:
125  addr_len = 8;
126  nettype = GNRC_NETTYPE_SIXLOWPAN;
127  break;
128 #endif
129  default:
130  addr_len = 1;
131  nettype = GNRC_NETTYPE_UNDEF;
132  }
133 
134  /* copy packet payload into pktbuf */
135  gnrc_pktsnip_t *pkt = gnrc_pktbuf_add(NULL, cc110x_pkt->data,
136  payload_length, nettype);
137 
138  if(!pkt) {
139  DEBUG("cc110x: _recv: cannot allocate pktsnip.\n");
140  return NULL;
141  }
142 
143 
144  gnrc_pktsnip_t *netif_hdr;
145  netif_hdr = gnrc_pktbuf_add(NULL, NULL,
146  sizeof(gnrc_netif_hdr_t) + 2*addr_len,
147  GNRC_NETTYPE_NETIF);
148 
149  if (netif_hdr == NULL) {
150  DEBUG("gnrc_netdev2_cc110x: no space left in packet buffer\n");
151  gnrc_pktbuf_release(pkt);
152  return NULL;
153  }
154 
155  gnrc_netif_hdr_init(netif_hdr->data, addr_len, addr_len);
156  if (addr_len == 8) {
157  uint64_t src_addr = cc110x_pkt->phy_src;
158  uint64_t dst_addr = cc110x_pkt->address;
159  gnrc_netif_hdr_set_src_addr(netif_hdr->data, (uint8_t*)&src_addr, addr_len);
160  gnrc_netif_hdr_set_dst_addr(netif_hdr->data, (uint8_t*)&dst_addr, addr_len);
161  }
162  else {
163  gnrc_netif_hdr_set_src_addr(netif_hdr->data, (uint8_t*)&cc110x_pkt->phy_src, addr_len);
164  gnrc_netif_hdr_set_dst_addr(netif_hdr->data, (uint8_t*)&cc110x_pkt->address, addr_len);
165  }
166 
167  ((gnrc_netif_hdr_t *)netif_hdr->data)->if_pid = thread_getpid();
168  ((gnrc_netif_hdr_t *)netif_hdr->data)->lqi = cc110x->pkt_buf.lqi;
169  ((gnrc_netif_hdr_t *)netif_hdr->data)->rssi = cc110x->pkt_buf.rssi;
170 
171  DEBUG("gnrc_netdev2_cc110x: received packet from %02x"
172  " of length %u\n",
173  (unsigned)cc110x_pkt->phy_src,
174  (unsigned)cc110x_pkt->length-CC110X_HEADER_LENGTH);
175 #if defined(MODULE_OD) && ENABLE_DEBUG
176  od_hex_dump(cc110x_pkt->data, payload_length, OD_WIDTH_DEFAULT);
177 #endif
178 
179 
180  pkt->next = netif_hdr;
181 
182  return pkt;
183 }
