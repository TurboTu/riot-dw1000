44 int timer_init(tim_t dev, unsigned long freq, timer_cb_t cb, void *arg)
45 {
46  /* at the moment, the timer can only run at 1MHz */
47  if (freq != 1000000ul) {
48  return -1;
49  }
50 
51 /* select the clock generator depending on the main clock source:
52  * GCLK0 (1MHz) if we use the internal 8MHz oscillator
53  * GCLK1 (8MHz) if we use the PLL */
54 #if CLOCK_USE_PLL
55  /* configure GCLK1 (configured to 1MHz) to feed TC3, TC4 and TC5 */;
56  GCLK->CLKCTRL.reg = (uint16_t)((GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK1 | (TC3_GCLK_ID << GCLK_CLKCTRL_ID_Pos)));
57  while (GCLK->STATUS.bit.SYNCBUSY) {}
58  /* TC4 and TC5 share the same channel */
59  GCLK->CLKCTRL.reg = (uint16_t)((GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK1 | (TC4_GCLK_ID << GCLK_CLKCTRL_ID_Pos)));
60 #else
61  /* configure GCLK0 to feed TC3, TC4 and TC5 */;
62  GCLK->CLKCTRL.reg = (uint16_t)((GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK0 | (TC3_GCLK_ID << GCLK_CLKCTRL_ID_Pos)));
63  /* TC4 and TC5 share the same channel */
64  GCLK->CLKCTRL.reg = (uint16_t)((GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK0 | (TC4_GCLK_ID << GCLK_CLKCTRL_ID_Pos)));
65 #endif
66  while (GCLK->STATUS.bit.SYNCBUSY) {}
67 
68  switch (dev) {
69 #if TIMER_0_EN
70  case TIMER_0:
71  if (TIMER_0_DEV.CTRLA.bit.ENABLE) {
72  return 0;
73  }
74  PM->APBCMASK.reg |= PM_APBCMASK_TC3;
75  /* reset timer */
76  TIMER_0_DEV.CTRLA.bit.SWRST = 1;
77  while (TIMER_0_DEV.CTRLA.bit.SWRST) {}
78  /* choosing 16 bit mode */
79  TIMER_0_DEV.CTRLA.bit.MODE = TC_CTRLA_MODE_COUNT16_Val;
80 #if CLOCK_USE_PLL
81  /* sourced by 1MHz with prescaler 1 results in... you know it :-) */
82  TIMER_0_DEV.CTRLA.bit.PRESCALER = TC_CTRLA_PRESCALER_DIV1_Val;
83 #else
84  /* sourced by 8MHz with Presc 8 results in 1MHz clk */
85  TIMER_0_DEV.CTRLA.bit.PRESCALER = TC_CTRLA_PRESCALER_DIV8_Val;
86 #endif
87  /* choose normal frequency operation */
88  TIMER_0_DEV.CTRLA.bit.WAVEGEN = TC_CTRLA_WAVEGEN_NFRQ_Val;
89  break;
90 #endif
91 #if TIMER_1_EN
92  case TIMER_1:
93  if (TIMER_1_DEV.CTRLA.bit.ENABLE) {
94  return 0;
95  }
96  PM->APBCMASK.reg |= PM_APBCMASK_TC4;
97  /* reset timer */
98  TIMER_1_DEV.CTRLA.bit.SWRST = 1;
99 
100  while (TIMER_1_DEV.CTRLA.bit.SWRST) {}
101 
102 
103  TIMER_1_DEV.CTRLA.bit.MODE = TC_CTRLA_MODE_COUNT32_Val;
104 #if CLOCK_USE_PLL
105  /* sourced by 1MHz and prescaler 1 to reach 1MHz */
106  TIMER_1_DEV.CTRLA.bit.PRESCALER = TC_CTRLA_PRESCALER_DIV1_Val;
107 #else
108  /* sourced by 8MHz with Presc 8 results in 1Mhz clk */
109  TIMER_1_DEV.CTRLA.bit.PRESCALER = TC_CTRLA_PRESCALER_DIV8_Val;
110 #endif
111  /* choose normal frequency operation */
112  TIMER_1_DEV.CTRLA.bit.WAVEGEN = TC_CTRLA_WAVEGEN_NFRQ_Val;
113  break;
114 #endif
115  case TIMER_UNDEFINED:
116  default:
117  return -1;
118  }
119 
120  /* save callback */
121  config[dev].cb = cb;
122  config[dev].arg = arg;
123 
124  /* enable interrupts for given timer */
125  _irq_enable(dev);
126 
127  timer_start(dev);
128 
129  return 0;
130 }
