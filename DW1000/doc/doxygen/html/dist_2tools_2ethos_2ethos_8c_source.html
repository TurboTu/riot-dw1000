57 int set_serial_attribs (int fd, int speed, int parity)
58 {
59  struct termios tty;
60  memset (&tty, 0, sizeof tty);
61  if (tcgetattr (fd, &tty) != 0)
62  {
63  perror ("error in tcgetattr");
64  return -1;
65  }
66 
67  cfsetospeed (&tty, speed);
68  cfsetispeed (&tty, speed);
69 
70  tty.c_cflag = (tty.c_cflag & ~CSIZE) | CS8; /* 8-bit chars*/
71  /* disable IGNBRK for mismatched speed
72  * tests; otherwise receive break*/
73  /* as \000 chars*/
74  tty.c_iflag &= ~IGNBRK; /* disable break processing*/
75  tty.c_lflag = 0; /* no signaling chars, no echo,*/
76  /* no canonical processing*/
77  tty.c_oflag = 0; /* no remapping, no delays*/
78  tty.c_cc[VMIN] = 0; /* read doesn't block*/
79  tty.c_cc[VTIME] = TTY_TIMEOUT_MS / 100; /* 0.5 seconds read timeout*/
80  /* in tenths of a second*/
81 
82  tty.c_iflag &= ~(IXON | IXOFF | IXANY); /* shut off xon/xoff ctrl*/
83 
84  tty.c_cflag |= (CLOCAL | CREAD); /* ignore modem controls,*/
85  /* enable reading*/
86  tty.c_cflag &= ~(PARENB | PARODD); /* shut off parity*/
87  tty.c_cflag |= parity;
88  tty.c_cflag &= ~CSTOPB;
89  tty.c_cflag &= ~CRTSCTS;
90  cfmakeraw(&tty);
91 
92  if (tcsetattr (fd, TCSANOW, &tty) != 0)
93  {
94  perror("error from tcsetattr");
95  return -1;
96  }
97  return 0;
98 }

184 static int _serial_handle_byte(serial_t *serial, char c)
185 {
186  switch (serial->state) {
187  case WAIT_FRAMESTART:
188  if (c == LINE_FRAME_DELIMITER) {
189  TRACE("frame start");
190  serial->state = IN_FRAME;
191  serial->frametype = 0;
192  serial->framebytes = 0;
193  }
194  break;
195  case IN_FRAME:
196  if (c == LINE_ESC_CHAR) {
197  TRACE("i esc");
198  serial->state = IN_ESCAPE;
199  }
200  else if (c == LINE_FRAME_DELIMITER) {
201  if (!serial->framebytes) {
202  TRACE("i del unexp");
203  } else {
204  TRACE("i del");
205  serial->state = WAIT_FRAMESTART;
206  return serial->framebytes;
207  }
208  }
209  else {
210  _handle_char(serial, c);
211  }
212  break;
213  case IN_ESCAPE:
214  if (c == (LINE_FRAME_DELIMITER ^ 0x20)) {
215  _handle_char(serial, LINE_FRAME_DELIMITER);
216  }
217  else if (c == (LINE_ESC_CHAR ^ 0x20)) {
218  _handle_char(serial, LINE_ESC_CHAR);
219  }
220  else if (c == (LINE_FRAME_TYPE_TEXT ^ 0x20)) {
221  TRACE("esc txt");
222  serial->frametype = LINE_FRAME_TYPE_TEXT;
223  }
224  else if (c == (LINE_FRAME_TYPE_HELLO ^ 0x20)) {
225  TRACE("esc hello");
226  serial->frametype = LINE_FRAME_TYPE_HELLO;
227  }
228  else if (c == (LINE_FRAME_TYPE_HELLO_REPLY ^ 0x20)) {
229  TRACE("esc hello_reply");
230  serial->frametype = LINE_FRAME_TYPE_HELLO_REPLY;
231  }
232  else if (c == LINE_FRAME_DELIMITER) {
233  TRACE("esc -del");
234  }
235  serial->state = IN_FRAME;
236  break;
237  }
238 
239  return 0;
240 }

242 static void _write_escaped(int fd, char* buf, ssize_t n)
243 {
244  uint8_t out[2];
245  size_t escaped = 0;
246 
247  while(n--) {
248  char c = *buf++;
249  switch(c) {
250  case LINE_FRAME_DELIMITER:
251  out[0] = LINE_ESC_CHAR;
252  out[1] = (LINE_FRAME_DELIMITER ^ 0x20);
253  checked_write(fd, out, 2);
254  escaped++;
255  break;
256  case LINE_ESC_CHAR:
257  out[0] = LINE_ESC_CHAR;
258  out[1] = (LINE_ESC_CHAR ^ 0x20);
259  checked_write(fd, out, 2);
260  escaped++;
261  break;
262  default:
263  checked_write(fd, &c, 1);
264  }
265  }

372 /* Adapted from 'getaddrinfo' manpage example */
373 int _tcp_connect(char *host, char *port)
374 {
375  int sfd = -1;
376  struct addrinfo hints, *result, *rp;
377 
378  memset(&hints, 0, sizeof(struct addrinfo));
379  hints.ai_family = AF_UNSPEC;
380  hints.ai_socktype = SOCK_STREAM;
381 
382  int s = getaddrinfo(host, port, &hints, &result);
383  if (s) {
384  fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(s));
385  return -1;
386  }
387 
388  /* getaddrinfo() returns a list of address structures.
389  Try each address until we successfully connect(2).
390  If socket(2) (or connect(2)) fails, we (close the socket
391  and) try the next address. */
392  for (rp = result; rp != NULL; rp = rp->ai_next) {
393  sfd = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
394  if (sfd == -1)
395  continue;
396 
397  if (connect(sfd, rp->ai_addr, rp->ai_addrlen) != -1)
398  break;
399 
400  close(sfd);
401  }
402 
403  freeaddrinfo(result);
404 
405  if (rp == NULL) {
406  fprintf(stderr, "Could not connect to '%s:%s'\n", host, port);
407  return -1;
408  }
409 
410  return sfd;
411 }

483 int main(int argc, char *argv[])
484 {
485  char inbuf[MTU];
486  char *serial_option = NULL;
487 
488  serial_t serial = {0};
489 
490  if (argc < 3) {
491  usage();
492  return 1;
493  }
494 
495  if (argc >= 4) {
496  serial_option = argv[3];
497  }
498 
499  char ifname[IFNAMSIZ];
500  strncpy(ifname, argv[1], IFNAMSIZ);
501  int tap_fd = tun_alloc(ifname, IFF_TAP | IFF_NO_PI);
502 
503  if (tap_fd < 0) {
504  return 1;
505  }
506 
507 
508  int serial_fd = _open_connection(argv[2], serial_option);
509  if (serial_fd < 0) {
510  fprintf(stderr, "Error opening serial device %s\n", argv[2]);
511  return 1;
512  }
513 
514  fd_set readfds;
515  int max_fd = (serial_fd > tap_fd) ? serial_fd : tap_fd;
516 
517  fprintf(stderr, "----> ethos: sending hello.\n");
518  _send_hello(serial_fd, &serial, LINE_FRAME_TYPE_HELLO);
519 
520  fprintf(stderr, "----> ethos: activating serial pass through.\n");
521  _send_hello(serial_fd, &serial, LINE_FRAME_TYPE_HELLO);
522  while(1) {
523  int activity;
524  FD_ZERO(&readfds);
525  FD_SET(STDIN_FILENO, &readfds);
526  FD_SET(tap_fd, &readfds);
527  FD_SET(serial_fd, &readfds);
528  activity = select( max_fd + 1 , &readfds , NULL , NULL , NULL);
529 
530  if ((activity < 0) && (errno != EINTR))
531  {
532  perror("select error");
533  }
534 
535  if (FD_ISSET(serial_fd, &readfds)) {
536  ssize_t n = read(serial_fd, inbuf, sizeof(inbuf));
537  if (n > 0) {
538  char *ptr = inbuf;
539  while (n--) {
540  size_t res = _serial_handle_byte(&serial, *ptr++);
541  if (res) {
542  switch (serial.frametype) {
543  case LINE_FRAME_TYPE_DATA:
544  checked_write(tap_fd, serial.frame, serial.framebytes);
545  break;
546  case LINE_FRAME_TYPE_TEXT:
547  checked_write(STDOUT_FILENO, serial.frame, serial.framebytes);
548  break;
549  case LINE_FRAME_TYPE_HELLO:
550  case LINE_FRAME_TYPE_HELLO_REPLY:
551  if (serial.framebytes == 6) {
552  memcpy(serial.remote_l2_addr, serial.frame, 6);
553  if (serial.frametype == LINE_FRAME_TYPE_HELLO) {
554  fprintf(stderr, "----> ethos: hello received\n");
555  _send_hello(serial_fd, &serial, LINE_FRAME_TYPE_HELLO_REPLY);
556  } else {
557  fprintf(stderr, "----> ethos: hello reply received\n");
558  }
559  _clear_neighbor_cache(ifname);
560  }
561  break;
562  }
563  memset(&serial, '\0', sizeof(serial));
564  }
565  }
566  }
567  else {
568  fprintf(stderr, "lost serial connection.\n");
569  exit(1);
570  }
571  }
572 
573  if (FD_ISSET(tap_fd, &readfds)) {
574  ssize_t res = read(tap_fd, inbuf, sizeof(inbuf));
575  if (res <= 0) {
576  fprintf(stderr, "error reading from tap device. res=%zi\n", res);
577  continue;
578  }
579  char delim = LINE_FRAME_DELIMITER;
580  checked_write(serial_fd, &delim, 1);
581  _write_escaped(serial_fd, inbuf, res);
582  checked_write(serial_fd, &delim, 1);
583  }
584 
585  if (FD_ISSET(STDIN_FILENO, &readfds)) {
586  ssize_t res = read(STDIN_FILENO, inbuf, sizeof(inbuf));
587  if (res <= 0) {
588  fprintf(stderr, "error reading from stdio. res=%zi\n", res);
589  continue;
590  }
591 
592  if (res) {
593  char delim = LINE_FRAME_DELIMITER;
594  char head[] = { LINE_FRAME_DELIMITER, LINE_ESC_CHAR, (LINE_FRAME_TYPE_TEXT ^ 0x20) };
595  checked_write(serial_fd, head, sizeof(head));
596  _write_escaped(serial_fd, inbuf, res);
597  checked_write(serial_fd, &delim, 1);
598  }
599  }
600  }
601 
602  return 0;
603 }
