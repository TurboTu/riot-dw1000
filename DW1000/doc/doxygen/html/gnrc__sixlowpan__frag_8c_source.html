84 static uint16_t _send_1st_fragment(gnrc_sixlowpan_netif_t *iface, gnrc_pktsnip_t *pkt,
85  size_t payload_len, size_t datagram_size)
86 {
87  gnrc_pktsnip_t *frag;
88  uint16_t local_offset = 0;
89  /* payload_len: actual size of the packet vs
90  * datagram_size: size of the uncompressed IPv6 packet */
91  int payload_diff = (datagram_size - payload_len);
92  /* virtually add payload_diff to flooring to account for offset (must be divisable by 8)
93  * in uncompressed datagram */
94  uint16_t max_frag_size = _floor8(iface->max_frag_size + payload_diff -
95  sizeof(sixlowpan_frag_t)) - payload_diff;
96  sixlowpan_frag_t *hdr;
97  uint8_t *data;
98 
99  DEBUG("6lo frag: determined max_frag_size = %" PRIu16 "\n", max_frag_size);
100 
101  frag = _build_frag_pkt(pkt, payload_len,
102  max_frag_size + sizeof(sixlowpan_frag_t));
103 
104  if (frag == NULL) {
105  return 0;
106  }
107 
108  hdr = frag->next->data;
109  data = (uint8_t *)(hdr + 1);
110 
111  hdr->disp_size = byteorder_htons((uint16_t)datagram_size);
112  hdr->disp_size.u8[0] |= SIXLOWPAN_FRAG_1_DISP;
113  hdr->tag = byteorder_htons(_tag);
114 
115  pkt = pkt->next; /* don't copy netif header */
116 
117  while (pkt != NULL) {
118  size_t clen = _min(max_frag_size - local_offset, pkt->size);
119 
120  memcpy(data + local_offset, pkt->data, clen);
121  local_offset += clen;
122 
123  if (local_offset >= max_frag_size) {
124  break;
125  }
126 
127  pkt = pkt->next;
128  }
129 
130  DEBUG("6lo frag: send first fragment (datagram size: %u, "
131  "datagram tag: %" PRIu16 ", fragment size: %" PRIu16 ")\n",
132  (unsigned int)datagram_size, _tag, local_offset);
133  if (gnrc_netapi_send(iface->pid, frag) < 1) {
134  DEBUG("6lo frag: unable to send first fragment\n");
135  gnrc_pktbuf_release(frag);
136  }
137 
138  return local_offset;
139 }

141 static uint16_t _send_nth_fragment(gnrc_sixlowpan_netif_t *iface, gnrc_pktsnip_t *pkt,
142  size_t payload_len, size_t datagram_size,
143  uint16_t offset)
144 {
145  gnrc_pktsnip_t *frag;
146  /* since dispatches aren't supposed to go into subsequent fragments, we need not account
147  * for payload difference as for the first fragment */
148  uint16_t max_frag_size = _floor8(iface->max_frag_size - sizeof(sixlowpan_frag_n_t));
149  uint16_t local_offset = 0, offset_count = 0;
150  sixlowpan_frag_n_t *hdr;
151  uint8_t *data;
152 
153  DEBUG("6lo frag: determined max_frag_size = %" PRIu16 "\n", max_frag_size);
154 
155  frag = _build_frag_pkt(pkt,
156  payload_len - offset + sizeof(sixlowpan_frag_n_t),
157  max_frag_size + sizeof(sixlowpan_frag_n_t));
158 
159  if (frag == NULL) {
160  return 0;
161  }
162 
163  hdr = frag->next->data;
164  data = (uint8_t *)(hdr + 1);
165 
166  /* XXX: truncation of datagram_size > 4095 may happen here */
167  hdr->disp_size = byteorder_htons((uint16_t)datagram_size);
168  hdr->disp_size.u8[0] |= SIXLOWPAN_FRAG_N_DISP;
169  hdr->tag = byteorder_htons(_tag);
170  /* don't mention payload diff in offset */
171  hdr->offset = (uint8_t)((offset + (datagram_size - payload_len)) >> 3);
172  pkt = pkt->next; /* don't copy netif header */
173 
174  while ((pkt != NULL) && (offset_count != offset)) { /* go to offset */
175  offset_count += (uint16_t)pkt->size;
176 
177  if (offset_count > offset) { /* we overshot */
178  /* => copy rest of partly send packet snip */
179  uint16_t pkt_offset = offset - (offset_count - ((uint16_t)pkt->size));
180  size_t clen = _min(max_frag_size, pkt->size - pkt_offset);
181 
182  memcpy(data, ((uint8_t *)pkt->data) + pkt_offset, clen);
183  local_offset = clen;
184  pkt = pkt->next;
185  break;
186  }
187 
188  pkt = pkt->next;
189  }
190 
191  if (local_offset < max_frag_size) { /* copy other packet snips */
192  while (pkt != NULL) {
193  size_t clen = _min(max_frag_size - local_offset, pkt->size);
194 
195  memcpy(data + local_offset, pkt->data, clen);
196  local_offset += clen;
197 
198  if (local_offset == max_frag_size) {
199  break;
200  }
201 
202  pkt = pkt->next;
203  }
204  }
205 
206  DEBUG("6lo frag: send subsequent fragment (datagram size: %u, "
207  "datagram tag: %" PRIu16 ", offset: %" PRIu8 " (%u bytes), "
208  "fragment size: %" PRIu16 ")\n",
209  (unsigned int)datagram_size, _tag, hdr->offset, hdr->offset << 3,
210  local_offset);
211  if (gnrc_netapi_send(iface->pid, frag) < 1) {
212  DEBUG("6lo frag: unable to send subsequent fragment\n");
213  gnrc_pktbuf_release(frag);
214  }
215 
216  return local_offset;
217 }

219 void gnrc_sixlowpan_frag_send(gnrc_sixlowpan_msg_frag_t *fragment_msg)
220 {
221  gnrc_sixlowpan_netif_t *iface = gnrc_sixlowpan_netif_get(fragment_msg->pid);
222  uint16_t res;
223  /* payload_len: actual size of the packet vs
224  * datagram_size: size of the uncompressed IPv6 packet */
225  size_t payload_len = gnrc_pkt_len(fragment_msg->pkt->next);
226  msg_t msg;
227 
228 #if defined(DEVELHELP) && defined(ENABLE_DEBUG)
229  if (iface == NULL) {
230  DEBUG("6lo frag: iface == NULL, expect segmentation fault.\n");
231  /* remove original packet from packet buffer */
232  gnrc_pktbuf_release(fragment_msg->pkt);
233  /* 6LoWPAN free for next fragmentation */
234  fragment_msg->pkt = NULL;
235  return;
236  }
237 #endif
238 
239  /* Check weater to send the first or an Nth fragment */
240  if (fragment_msg->offset == 0) {
241  /* increment tag for successive, fragmented datagrams */
242  _tag++;
243  if ((res = _send_1st_fragment(iface, fragment_msg->pkt, payload_len, fragment_msg->datagram_size)) == 0) {
244  /* error sending first fragment */
245  DEBUG("6lo frag: error sending 1st fragment\n");
246  gnrc_pktbuf_release(fragment_msg->pkt);
247  fragment_msg->pkt = NULL;
248  return;
249  }
250  fragment_msg->offset += res;
251 
252  /* send message to self*/
253  msg.type = GNRC_SIXLOWPAN_MSG_FRAG_SND;
254  msg.content.ptr = (void *)fragment_msg;
255  msg_send_to_self(&msg);
256  thread_yield();
257  }
258  else {
259  /* (offset + (datagram_size - payload_len) < datagram_size) simplified */
260  if (fragment_msg->offset < payload_len) {
261  if ((res = _send_nth_fragment(iface, fragment_msg->pkt, payload_len, fragment_msg->datagram_size,
262  fragment_msg->offset)) == 0) {
263  /* error sending subsequent fragment */
264  DEBUG("6lo frag: error sending subsequent fragment (offset = %" PRIu16
265  ")\n", fragment_msg->offset);
266  gnrc_pktbuf_release(fragment_msg->pkt);
267  fragment_msg->pkt = NULL;
268  return;
269  }
270  fragment_msg->offset += res;
271 
272  /* send message to self*/
273  msg.type = GNRC_SIXLOWPAN_MSG_FRAG_SND;
274  msg.content.ptr = (void *)fragment_msg;
275  msg_send_to_self(&msg);
276  thread_yield();
277  }
278  else {
279  gnrc_pktbuf_release(fragment_msg->pkt);
280  fragment_msg->pkt = NULL;
281  }
282  }
283 }

285 void gnrc_sixlowpan_frag_handle_pkt(gnrc_pktsnip_t *pkt)
286 {
287  gnrc_netif_hdr_t *hdr = pkt->next->data;
288  sixlowpan_frag_t *frag = pkt->data;
289  uint16_t offset = 0;
290  size_t frag_size;
291 
292  switch (frag->disp_size.u8[0] & SIXLOWPAN_FRAG_DISP_MASK) {
293  case SIXLOWPAN_FRAG_1_DISP:
294  frag_size = (pkt->size - sizeof(sixlowpan_frag_t));
295  break;
296 
297  case SIXLOWPAN_FRAG_N_DISP:
298  offset = (((sixlowpan_frag_n_t *)frag)->offset * 8);
299  frag_size = (pkt->size - sizeof(sixlowpan_frag_n_t));
300  break;
301 
302  default:
303  DEBUG("6lo rbuf: Not a fragment header.\n");
304  gnrc_pktbuf_release(pkt);
305 
306  return;
307  }
308 
309  rbuf_add(hdr, pkt, frag_size, offset);
310 
311  gnrc_pktbuf_release(pkt);
312 }
