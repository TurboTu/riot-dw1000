26 static inline float temperature_to_float(uint16_t temp)
27 {
28  /* Integer part is 8-bit signed */
29  int8_t temp_int = (temp & AT30TSE75X_INTEGER_MASK) >>
30  AT30TSE75X_INTEGER_SHIFT;
31  /* Fractional part is multiple of 0.0625 */
32  uint8_t frac_multiplier = (temp & AT30TSE75X_FRACTIONAL_MASK) >>
33  AT30TSE75X_FRACTIONAL_SHIFT;
34 
35  return temp_int + (frac_multiplier * AT30TSE75X_FRACTIONAL_BASE);
36 }

131 int at30tse75x_set_mode(at30tse75x_t* dev, at30tse75x_mode_t mode)
132 {
133  uint8_t config;
134  if(at30tse75x_get_config(dev, &config) != 0) {
135  return -1;
136  }
137 
138  switch(mode) {
139  case AT30TSE75X_MODE_ONE_SHOT:
140  config |= AT30TSE75X_CONFIG__SHUTDOWN_BIT;
141  /* Don't touch alarm mode */
142  break;
143  case AT30TSE75X_MODE_COMPARATOR:
144  config &= ~(AT30TSE75X_CONFIG__SHUTDOWN_BIT);
145  config &= ~(AT30TSE75X_CONFIG__ALARM_MODE_BIT);
146  break;
147  case AT30TSE75X_MODE_INTERRUPT:
148  config &= ~(AT30TSE75X_CONFIG__SHUTDOWN_BIT);
149  config |= AT30TSE75X_CONFIG__ALARM_MODE_BIT;
150  break;
151  default:
152  return -2;
153  }
154 
155  if(at30tse75x_set_config(dev, config) != 0) {
156  return -1;
157  }
158 
159  return 0;

249 int at30tse75x_get_temperature(at30tse75x_t* dev, float* temperature)
250 {
251  uint16_t tmp;
252  uint8_t config;
253 
254  if(at30tse75x_get_config(dev, &config) != 0) {
255  return -1;
256  }
257 
258  /* If sensor is shutdown trigger One-Shot mode*/
259  if(config & AT30TSE75X_CONFIG__SHUTDOWN_BIT) {
260 
261  config |= AT30TSE75X_CONFIG__OS_BIT;
262  if(at30tse75x_set_config(dev, config) != 0) {
263  return -1;
264  }
265 
266  /* Use resolution to calculate conversion time */
267  uint8_t resolution = (config & AT30TSE75X_CONFIG__RESOLUTION_MASK) >>
268  AT30TSE75X_CONFIG__RESOLUTION_SHIFT;
269 
270  /* Wait until conversion is finished */
271  xtimer_usleep((uint32_t)(25000 << resolution));
272  }
273 
274  /* Read temperature */
275  if(at30tse75x_get_register(dev, AT30TSE75X_REG__TEMPERATURE, &tmp) != 0) {
276  return -1;
277  }
278 
279  /* Convert fixed point to float */
280  *temperature = temperature_to_float(tmp);
281 
282  return 0;
283 }

285 int at30tse75x_init(at30tse75x_t* dev, i2c_t i2c, i2c_speed_t speed, uint8_t addr)
286 {
287  uint8_t config;
288 
289  dev->i2c = i2c;
290 
291  if( (addr < 0x48) || (addr > 0x4f) ) {
292  DEBUG("[at30tse75x] Invalid address\n");

294  }
295  dev->addr = addr;
296 
297  i2c_acquire(dev->i2c);
298  if(i2c_init_master(dev->i2c, speed) != 0) {
299  DEBUG("[at30tse75x] Can't initialize I2C master\n");
300  i2c_release(dev->i2c);
301  return -1;
302  }
303  i2c_release(dev->i2c);
304 
305  /* Reset the device */
306  if(at30tse75x_reset(dev) != 0) {
307  return -1;
308  }
309 
310  /* Poll the device, fail if unavailable */
311  if(at30tse75x_get_config(dev, &config) != 0) {
312  return -1;
313  }
314 
315  DEBUG("[at30tse75x] Config: 0x%x\n", config);
316 
317  return 0;
