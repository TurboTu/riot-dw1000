163 double decode_float_half(unsigned char *halfp)
164 {
165  int half = (halfp[0] << 8) + halfp[1];
166  int exp = (half >> 10) & 0x1f;
167  int mant = half & 0x3ff;
168  double val;
169 
170  if (exp == 0) {
171  val = ldexp(mant, -24);
172  }
173  else if (exp != 31) {
174  val = ldexp(mant + 1024, exp - 25);
175  }
176  else {
177  val = mant == 0 ? INFINITY : NAN;
178  }
179 
180  return (half & 0x8000) ? -val : val;
181 }

186 static uint16_t encode_float_half(float x)
187 {
188  union u {
189  float f;
190  uint32_t i;
191  } u = { .f = x };
192 
193  uint16_t bits = (u.i >> 16) & 0x8000; /* Get the sign */
194  uint16_t m = (u.i >> 12) & 0x07ff; /* Keep one extra bit for rounding */
195  unsigned int e = (u.i >> 23) & 0xff; /* Using int is faster here */
196 
197  /* If zero, or denormal, or exponent underflows too much for a denormal
198  * half, return signed zero. */
199  if (e < 103) {
200  return bits;
201  }
202 
203  /* If NaN, return NaN. If Inf or exponent overflow, return Inf. */
204  if (e > 142) {
205  bits |= 0x7c00u;
206  /* If exponent was 0xff and one mantissa bit was set, it means NaN,
207  * not Inf, so make sure we set one mantissa bit too. */
208  bits |= (e == 255) && (u.i & 0x007fffffu);
209  return bits;
210  }
211 
212  /* If exponent underflows but not too much, return a denormal */
213  if (e < 113) {
214  m |= 0x0800u;
215  /* Extra rounding may overflow and set mantissa to 0 and exponent
216  * to 1, which is OK. */
217  bits |= (m >> (114 - e)) + ((m >> (113 - e)) & 1);
218  return bits;
219  }
220 
221  bits |= ((e - 112) << 10) | (m >> 1);
222  /* Extra rounding. An overflow will set mantissa to 0 and increment
223  * the exponent, which is OK. */
224  bits += m & 1;
225  return bits;
226 }

285 {
286  if (val < CBOR_UINT8_FOLLOWS) {

317 static size_t encode_int(unsigned char major_type, cbor_stream_t *s, uint64_t val)
318 {
319  if (!s) {
320  return 0;
321  }
322 
323  unsigned char additional_info = uint_additional_info(val);
324  unsigned char bytes_follow = uint_bytes_follow(additional_info);
325  CBOR_ENSURE_SIZE(s, bytes_follow + 1);
326  s->data[s->pos++] = major_type | additional_info;
327 
328  for (int i = bytes_follow - 1; i >= 0; --i) {
329  s->data[s->pos++] = (val >> (8 * i)) & 0xff;
330  }
331 
332  return bytes_follow + 1;
333 }

335 static size_t decode_int(const cbor_stream_t *s, size_t offset, uint64_t *val)
336 {
337  if (!s) {
338  return 0;
339  }
340 
341  *val = 0; /* clear val first */
342 
343  CBOR_ENSURE_SIZE_READ(s, offset + 1);
344 
345  unsigned char *in = &s->data[offset];
346  unsigned char additional_info = CBOR_ADDITIONAL_INFO(s, offset);
347  unsigned char bytes_follow = uint_bytes_follow(additional_info);
348 
349  CBOR_ENSURE_SIZE_READ(s, offset + 1 + bytes_follow);
350 
351  switch (bytes_follow) {
352  case 0:
353  *val = (in[0] & CBOR_INFO_MASK);
354  break;
355 
356  case 1:
357  *val = in[1];
358  break;
359 
360  case 2:
361  *val = HTONS(((cast_align_u8_t *)in)->u.u16);
362  break;
363 
364  case 4:
365  *val = HTONL(((cast_align_u8_t *)in)->u.u32);
366  break;
367 
368  default:
369  *val = HTONLL(((cast_align_u8_t *)in)->u.u64);
370  break;
371  }
372 
373  return bytes_follow + 1;
374 }

376 static size_t encode_bytes(unsigned char major_type, cbor_stream_t *s, const char *data,
377  size_t length)
378 {
379  size_t length_field_size = uint_bytes_follow(uint_additional_info(length)) + 1;
380  CBOR_ENSURE_SIZE(s, length_field_size + length);
381 
382  size_t bytes_start = encode_int(major_type, s, (uint64_t) length);
383 
384  if (!bytes_start) {
385  return 0;
386  }
387 
388  memcpy(&(s->data[s->pos]), data, length); /* copy byte string into our cbor struct */
389  s->pos += length;
390  return (bytes_start + length);
391 }

393 static size_t decode_bytes(const cbor_stream_t *s, size_t offset, char *out, size_t length)
394 {
395  CBOR_ENSURE_SIZE_READ(s, offset + 1);
396 
397  if ((CBOR_TYPE(s, offset) != CBOR_BYTES && CBOR_TYPE(s, offset) != CBOR_TEXT) || !out) {
398  return 0;
399  }
400 
401  uint64_t bytes_length;
402  size_t bytes_start = decode_int(s, offset, &bytes_length);
403 
404  if (!bytes_start) {
405  return 0;
406  }
407 
408  if (length + 1 < bytes_length) {
409  return 0;
410  }
411 
412  CBOR_ENSURE_SIZE_READ(s, offset + bytes_start + bytes_length);
413 
414  memcpy(out, &s->data[offset + bytes_start], bytes_length);
415  out[bytes_length] = '\0';
416  return (bytes_start + bytes_length);
417 }

445 size_t cbor_deserialize_int(const cbor_stream_t *stream, size_t offset, int *val)
446 {
447  CBOR_ENSURE_SIZE_READ(stream, offset + 1);
448 
449  if ((CBOR_TYPE(stream, offset) != CBOR_UINT && CBOR_TYPE(stream, offset) != CBOR_NEGINT) || !val) {
450  return 0;
451  }
452 
453  uint64_t buf;
454  size_t read_bytes = decode_int(stream, offset, &buf);
455 
456  if (!read_bytes) {
457  return 0;
458  }
459 
460  if (CBOR_TYPE(stream, offset) == CBOR_UINT) {
461  *val = buf; /* resolve as CBOR_UINT */
462  }
463  else {
464  *val = -1 - buf; /* resolve as CBOR_NEGINT */
465  }
466 
467  return read_bytes;
468 }

915 static size_t cbor_stream_decode_at(cbor_stream_t *stream, size_t offset, int indent)
916 {
917 #define DESERIALIZE_AND_PRINT(type, suffix, format_string) { \
918  type val; \
919  size_t read_bytes = cbor_deserialize_##suffix(stream, offset, &val); \
920  DEBUG("("#type", "format_string")\n", val); \
921  return read_bytes; \
922  }
923 
924  DEBUG("%*s", indent, "");
925 
926  switch (CBOR_TYPE(stream, offset)) {
927  case CBOR_UINT:
928  DESERIALIZE_AND_PRINT(uint64_t, uint64_t, "%" PRIu64)
929  case CBOR_NEGINT:
930  DESERIALIZE_AND_PRINT(int64_t, int64_t, "%" PRId64)
931  case CBOR_BYTES: {
932  char buffer[CBOR_STREAM_PRINT_BUFFERSIZE];
933  size_t read_bytes = cbor_deserialize_byte_string(stream, offset, buffer, sizeof(buffer));
934  DEBUG("(byte string, \"%s\")\n", buffer);
935  return read_bytes;
936  }
937 
938  case CBOR_TEXT: {
939  char buffer[CBOR_STREAM_PRINT_BUFFERSIZE];
940  size_t read_bytes = cbor_deserialize_unicode_string(stream, offset, buffer, sizeof(buffer));
941  DEBUG("(unicode string, \"%s\")\n", buffer);
942  return read_bytes;
943  }
944 
945  case CBOR_ARRAY: {
946  const bool is_indefinite = (stream->data[offset] == (CBOR_ARRAY | CBOR_VAR_FOLLOWS));
947  uint64_t array_length;
948  size_t read_bytes;
949 
950  if (is_indefinite) {
951  offset += read_bytes = cbor_deserialize_array_indefinite(stream, offset);
952  DEBUG("(array, length: [indefinite])\n");
953  }
954  else {
955  offset += read_bytes = decode_int(stream, offset, &array_length);
956  DEBUG("(array, length: %"PRIu64")\n", array_length);
957  }
958 
959  size_t i = 0;
960 
961  while (is_indefinite ? !cbor_at_break(stream, offset) : i < array_length) {
962  size_t inner_read_bytes;
963  offset += inner_read_bytes = cbor_stream_decode_at(stream, offset, indent + 2);
964 
965  if (inner_read_bytes == 0) {
966  DEBUG("Failed to read array item at position %d\n", i);
967  break;
968  }
969 
970  read_bytes += inner_read_bytes;
971  ++i;
972  }
973 
974  read_bytes += cbor_at_break(stream, offset);
975  return read_bytes;
976  }
977 
978  case CBOR_MAP: {
979  const bool is_indefinite = (stream->data[offset] == (CBOR_MAP | CBOR_VAR_FOLLOWS));
980  uint64_t map_length;
981  size_t read_bytes;
982 
983  if (is_indefinite) {
984  offset += read_bytes = cbor_deserialize_map_indefinite(stream, offset);
985  DEBUG("(map, length: [indefinite])\n");
986  }
987  else {
988  offset += read_bytes = decode_int(stream, offset, &map_length);
989  DEBUG("(map, length: %"PRIu64")\n", map_length);
990  }
991 
992  size_t i = 0;
993 
994  while (is_indefinite ? !cbor_at_break(stream, offset) : i < map_length) {
995  size_t key_read_bytes, value_read_bytes;
996  offset += key_read_bytes = cbor_stream_decode_at(stream, offset, indent + 1); /* key */
997  offset += value_read_bytes = cbor_stream_decode_at(stream, offset, indent + 2); /* value */
998 
999  if (key_read_bytes == 0 || value_read_bytes == 0) {
1000  DEBUG("Failed to read key-value pair at position %d\n", i);
1001  break;
1002  }
1003 
1004  read_bytes += key_read_bytes + value_read_bytes;
1005  ++i;
1006  }
1007 
1008  read_bytes += cbor_at_break(stream, offset);
1009  return read_bytes;
1010  }
1011 
1012  case CBOR_TAG: {
1013  unsigned char tag = CBOR_ADDITIONAL_INFO(stream, offset);
1014 
1015  switch (tag) {
1016  /* Non-native builds likely don't have support for ctime (hence disable it there)
1017  * TODO: Better check for availability of ctime functions? */
1018 #ifndef CBOR_NO_CTIME
1019  case CBOR_DATETIME_STRING_FOLLOWS: {
1020  char buf[64];
1021  struct tm timeinfo;
1022  size_t read_bytes = cbor_deserialize_date_time(stream, offset, &timeinfo);
1023  strftime(buf, sizeof(buf), "%c", &timeinfo);
1024  DEBUG("(tag: %u, date/time string: \"%s\")\n", tag, buf);
1025  return read_bytes;
1026  }
1027 
1028  case CBOR_DATETIME_EPOCH_FOLLOWS: {
1029  time_t time;
1030  size_t read_bytes = cbor_deserialize_date_time_epoch(stream, offset, &time);
1031  DEBUG("(tag: %u, date/time epoch: %d)\n", tag, (int)time);
1032  return read_bytes;
1033  }
1034 
1035 #endif /* CBOR_NO_CTIME */
1036 
1037  default:
1038  break;
1039  }
1040  }
1041 
1042  case CBOR_7: {
1043  switch (stream->data[offset]) {
1044  case CBOR_FALSE:
1045  case CBOR_TRUE:
1046  DESERIALIZE_AND_PRINT(bool, bool, "%d")
1047 #ifndef CBOR_NO_FLOAT
1048  case CBOR_FLOAT16:
1049  DESERIALIZE_AND_PRINT(float, float_half, "%f")
1050  case CBOR_FLOAT32:
1051  DESERIALIZE_AND_PRINT(float, float, "%f")
1052  case CBOR_FLOAT64:
1053  DESERIALIZE_AND_PRINT(double, double, "%lf")
1054 #endif /* CBOR_NO_FLOAT */
1055  default:
1056  break;
1057  }
1058  }
1059  }
1060 
1061  /* if we end up here, we weren't able to parse the current byte
1062  * let's just skip this (and the next one as well if required) */
1063  return cbor_stream_decode_skip(stream, offset);
1064 
1065 #undef DESERIALIZE_AND_PRINT
1066 }
