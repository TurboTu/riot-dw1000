63 static uint16_t _calc_csum(gnrc_pktsnip_t *hdr, gnrc_pktsnip_t *pseudo_hdr,
64  gnrc_pktsnip_t *payload)
65 {
66  uint16_t csum = 0;
67  uint16_t len = (uint16_t)hdr->size;
68 
69  /* process the payload */
70  while (payload && payload != hdr && payload != pseudo_hdr) {
71  csum = inet_csum_slice(csum, (uint8_t *)(payload->data), payload->size, len);
72  len += (uint16_t)payload->size;
73  payload = payload->next;
74  }
75  /* process applicable UDP header bytes */
76  csum = inet_csum(csum, (uint8_t *)hdr->data, sizeof(udp_hdr_t));
77 
78  switch (pseudo_hdr->type) {
79 #ifdef MODULE_GNRC_IPV6
80  case GNRC_NETTYPE_IPV6:
81  csum = ipv6_hdr_inet_csum(csum, pseudo_hdr->data, PROTNUM_UDP, len);
82  break;
83 #endif
84  default:
85  (void)len;
86  return 0;
87  }
88  /* return inverted results */
89  if (csum == 0xFFFF) {
90  /* https://tools.ietf.org/html/rfc2460#section-8.1
91  * bullet 4
92  * "if that computation yields a result of zero, it must be changed
93  * to hex FFFF for placement in the UDP header."
94  */
95  return 0xFFFF;
96  } else {
97  return ~csum;
98  }
99 }

101 static void _receive(gnrc_pktsnip_t *pkt)
102 {
103  gnrc_pktsnip_t *udp, *ipv6;
104  udp_hdr_t *hdr;
105  uint32_t port;
106 
107  /* mark UDP header */
108  udp = gnrc_pktbuf_start_write(pkt);
109  if (udp == NULL) {
110  DEBUG("udp: unable to get write access to packet\n");
111  gnrc_pktbuf_release(pkt);
112  return;
113  }
114  pkt = udp;
115 
116  ipv6 = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_IPV6);
117 
118  assert(ipv6 != NULL);
119 
120  if ((pkt->next != NULL) && (pkt->next->type == GNRC_NETTYPE_UDP) &&
121  (pkt->next->size == sizeof(udp_hdr_t))) {
122  /* UDP header was already marked. Take it. */
123  udp = pkt->next;
124  }
125  else {
126  udp = gnrc_pktbuf_mark(pkt, sizeof(udp_hdr_t), GNRC_NETTYPE_UDP);
127  if (udp == NULL) {
128  DEBUG("udp: error marking UDP header, dropping packet\n");
129  gnrc_pktbuf_release(pkt);
130  return;
131  }
132  }
133  /* mark payload as Type: UNDEF */
134  pkt->type = GNRC_NETTYPE_UNDEF;
135  /* get explicit pointer to UDP header */
136  hdr = (udp_hdr_t *)udp->data;
137 
138  /* validate checksum */
139  if (byteorder_ntohs(hdr->checksum) == 0) {
140  /* RFC 2460 Section 8.1
141  * "IPv6 receivers must discard UDP packets containing a zero checksum,
142  * and should log the error."
143  */
144  DEBUG("udp: received packet with zero checksum, dropping it\n");
145  gnrc_pktbuf_release(pkt);
146  return;
147  }
148  if (_calc_csum(udp, ipv6, pkt) != 0xFFFF) {
149  DEBUG("udp: received packet with invalid checksum, dropping it\n");
150  gnrc_pktbuf_release(pkt);
151  return;
152  }
153 
154  /* get port (netreg demux context) */
155  port = (uint32_t)byteorder_ntohs(hdr->dst_port);
156 
157  /* send payload to receivers */
158  if (!gnrc_netapi_dispatch_receive(GNRC_NETTYPE_UDP, port, pkt)) {
159  DEBUG("udp: unable to forward packet as no one is interested in it\n");
160  gnrc_pktbuf_release(pkt);
161  }
162 }

164 static void _send(gnrc_pktsnip_t *pkt)
165 {
166  udp_hdr_t *hdr;
167  gnrc_pktsnip_t *udp_snip, *tmp;
168  gnrc_nettype_t target_type = pkt->type;
169 
170  /* write protect first header */
171  tmp = gnrc_pktbuf_start_write(pkt);
172  if (tmp == NULL) {
173  DEBUG("udp: cannot send packet: unable to allocate packet\n");
174  gnrc_pktbuf_release(pkt);
175  return;
176  }
177  pkt = tmp;
178  udp_snip = tmp->next;
179 
180  /* get and write protect until udp snip */
181  while ((udp_snip != NULL) && (udp_snip->type != GNRC_NETTYPE_UDP)) {
182  udp_snip = gnrc_pktbuf_start_write(udp_snip);
183  if (udp_snip == NULL) {
184  DEBUG("udp: cannot send packet: unable to allocate packet\n");
185  gnrc_pktbuf_release(pkt);
186  return;
187  }
188  tmp->next = udp_snip;
189  tmp = udp_snip;
190  udp_snip = udp_snip->next;
191  }
192 
193  assert(udp_snip != NULL);
194 
195  /* write protect UDP snip */
196  udp_snip = gnrc_pktbuf_start_write(udp_snip);
197  if (udp_snip == NULL) {
198  DEBUG("udp: cannot send packet: unable to allocate packet\n");
199  gnrc_pktbuf_release(pkt);
200  return;
201  }
202  tmp->next = udp_snip;
203  hdr = (udp_hdr_t *)udp_snip->data;
204  /* fill in size field */
205  hdr->length = byteorder_htons(gnrc_pkt_len(udp_snip));
206 
207  /* set to IPv6, if first header is netif header */
208  if (target_type == GNRC_NETTYPE_NETIF) {
209  target_type = pkt->next->type;
210  }
211 
212  /* and forward packet to the network layer */
213  if (!gnrc_netapi_dispatch_send(target_type, GNRC_NETREG_DEMUX_CTX_ALL,
214  pkt)) {
215  DEBUG("udp: cannot send packet: network layer not found\n");
216  gnrc_pktbuf_release(pkt);
217  }
218 }

220 static void *_event_loop(void *arg)
221 {
222  (void)arg;
223  msg_t msg, reply;
224  msg_t msg_queue[GNRC_UDP_MSG_QUEUE_SIZE];
225  gnrc_netreg_entry_t netreg = GNRC_NETREG_ENTRY_INIT_PID(GNRC_NETREG_DEMUX_CTX_ALL,
226  sched_active_pid);
227  /* preset reply message */
228  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
229  reply.content.value = (uint32_t)-ENOTSUP;
230  /* initialize message queue */
231  msg_init_queue(msg_queue, GNRC_UDP_MSG_QUEUE_SIZE);
232  /* register UPD at netreg */
233  gnrc_netreg_register(GNRC_NETTYPE_UDP, &netreg);
234 
235  /* dispatch NETAPI messages */
236  while (1) {
237  msg_receive(&msg);
238  switch (msg.type) {
239  case GNRC_NETAPI_MSG_TYPE_RCV:
240  DEBUG("udp: GNRC_NETAPI_MSG_TYPE_RCV\n");
241  _receive(msg.content.ptr);
242  break;
243  case GNRC_NETAPI_MSG_TYPE_SND:
244  DEBUG("udp: GNRC_NETAPI_MSG_TYPE_SND\n");
245  _send(msg.content.ptr);
246  break;
247  case GNRC_NETAPI_MSG_TYPE_SET:
248  case GNRC_NETAPI_MSG_TYPE_GET:
249  msg_reply(&msg, &reply);
250  break;
251  default:
252  DEBUG("udp: received unidentified message\n");
253  break;
254  }
255  }
256 
257  /* never reached */
258  return NULL;
259 }

261 int gnrc_udp_calc_csum(gnrc_pktsnip_t *hdr, gnrc_pktsnip_t *pseudo_hdr)
262 {
263  uint16_t csum;
264 
265  if ((hdr == NULL) || (pseudo_hdr == NULL)) {
266  return -EFAULT;
267  }
268  if (hdr->type != GNRC_NETTYPE_UDP) {
269  return -EBADMSG;
270  }
271 
272  csum = _calc_csum(hdr, pseudo_hdr, hdr->next);
273  if (csum == 0) {
274  return -ENOENT;
275  }
276  ((udp_hdr_t *)hdr->data)->checksum = byteorder_htons(csum);
277  return 0;
278 }

281  uint16_t dst)
