163 void
164 relay_dhcp_to_server(struct ip *ip, int len)
165 {
166  struct dhcp_light_msg *inm;
167  struct dhcp_msg m;
168  int n;
169  u_int8_t *optptr;
170 
171  inm = (void *)(((u_int8_t *)ip) + 20 + 8); /* Skip over IP&UDP headers. */
172 
173  if (inm->op != BOOTREQUEST) {
174  return;
175  }
176 
177  inm->flags = ntohs(BOOTP_BROADCAST);
178 
179  memcpy(&m, inm, DHCP_BASE_LEN);
180  memset(&m.sname, 0x0, DHCP_HOLE_LEN);
181  memcpy(&m.options, &inm->options, len - 20 - 8 - DHCP_BASE_LEN);
182  n = (len - 20 - 8) + DHCP_HOLE_LEN; /* +HOLE -IP&UDP headers. */
183 
184  /*
185  * Ideally we would like to use the Relay Agent information option
186  * (RFC3046) together with the Link Selection sub-option (RFC3527)
187  * to ensure that addresses are allocated for this
188  * subnet. Unfortunately ISC-DHCPD does not currently implement
189  * RFC3527 and some other mechanism must be used. For this reason
190  * this implementation in addition uses the DHCP option for subnet
191  * selection (RFC3011) which is really not intended to be used by
192  * relays.
193  *
194  * Find DHCP_OPTION_END and add the new option here.
195  */
196  optptr = &m.options[n - DHCP_BASE_LEN - DHCP_HOLE_LEN - 1];
197  {
198  *optptr++ = DHCP_OPTION_SUBNET_SELECTION; /* RFC3011 */
199  *optptr++ = 4;
200  memcpy(optptr, &netaddr, 4);
201  optptr += 4;
202  n += 4 + 2;
203  }
204  {
205  *optptr++ = DHCP_OPTION_AGENT; /* RFC3046 */
206  *optptr++ = 18; /* Sum of all suboptions below! */
207 
208  *optptr++ = RAI_SUBNET_SELECTION; /* RFC3527 */
209  *optptr++ = 4;
210  memcpy(optptr, &netaddr, 4);
211  optptr += 4;
212  *optptr++ = RAI_CIRCUIT_ID;
213  *optptr++ = 4;
214  memcpy(optptr, &circuit_addr, 4);
215  optptr += 4;
216  *optptr++ = RAI_AGENT_ID;
217  *optptr++ = 4;
218  memcpy(optptr, &giaddr, 4);
219  optptr += 4;
220  n += 18 + 2; /* Sum of all suboptions + 2! */
221  }
222  /* And finally put back the END. */
223  *optptr++ = DHCP_OPTION_END;
224 
225  m.hops++;
226  memcpy(m.giaddr, &giaddr, sizeof(m.giaddr));
227 
228  if (n != sendto(dhsock, &m, n, 0x0/*flags*/,
229  (struct sockaddr *)&dhaddr, sizeof(dhaddr))) {
230  err(1, "sendto relay failed");
231  }
232 }

236 void
237 relay_dhcp_to_client(int slipfd)
238 {
239  struct dhcp_msg inm;
240  struct {
241  struct ip ip;
242  struct dhcp_light_msg m;
243  } pkt;
244  int n, optlen, ip_len, udp_len;
245  u_int8_t *p, *t, *end;
246  u_int16_t sum;
247  u_int8_t op, msg_type = 0;
248  struct in_addr yiaddr;
249 
250  memset(&inm.options, 0x0, sizeof(inm.options));
251 
252  n = recv(dhsock, &inm, sizeof(inm), 0x0/*flags*/);
253 
254  if (inm.op != BOOTREPLY) {
255  return;
256  }
257 
258  memcpy(&yiaddr, inm.yiaddr, sizeof(inm.yiaddr));
259  memcpy(&pkt.m, &inm, DHCP_BASE_LEN);
260  pkt.m.hops++;
261  memset(pkt.m.giaddr, 0x0, sizeof(pkt.m.giaddr));
262 
263  /*
264  * Copy options we would like to send to client.
265  */
266  memcpy(pkt.m.options, inm.options, 4); /* Magic cookie */
267 
268  end = &inm.op + n;
269  p = inm.options + 4; /* Magic cookie */
270  t = pkt.m.options + 4; /* Magic cookie */
271 
272  while (p < end) {
273  op = p[0];
274 
275  switch (op) {
276  case DHCP_OPTION_END:
277  goto done;
278 
279  case DHCP_OPTION_MSG_TYPE:
280  msg_type = p[2];
281  /* deliberate fall-through */
282  case DHCP_OPTION_SUBNET_MASK:
283  case DHCP_OPTION_ROUTER:
284  case DHCP_OPTION_LEASE_TIME:
285  case DHCP_OPTION_SERVER_ID: /* Copy these options */
286  memcpy(t, p, p[1] + 2);
287  t += p[1] + 2;
288  p += p[1] + 2;
289  break;
290 
291  case DHCP_OPTION_DNS_SERVER: /* Only copy first server */
292  *t++ = p[0];
293  *t++ = 4;
294  memcpy(t, p + 2, 4);
295  t += 4;
296  p += p[1] + 2;
297  break;
298 
299  default: /* Ignore these options */
300  /* printf("option type %d len %d\n", op, p[1]); */
301  p += p[1] + 2;
302  continue;
303  }
304  }
305 
306 done:
307 
308  if (op == DHCP_OPTION_END) {
309  *t++ = op;
310  *p++;
311  }
312 
313  optlen = t - pkt.m.options;
314  ip_len = 20 + 8 + DHCP_BASE_LEN + optlen;
315  udp_len = 8 + DHCP_BASE_LEN + optlen;
316 
317  pkt.ip.ip_vhl = 0x45; /* IPv4 and hdrlen=5*4 */
318  pkt.ip.ip_tos = 0;
319  pkt.ip.ip_len = htons(ip_len);
320  pkt.ip.ip_id = htons(ip_id++);
321  pkt.ip.ip_off = 0;
322  pkt.ip.ip_ttl = 64;
323  pkt.ip.ip_p = 17; /* proto UDP */
324  pkt.ip.ip_sum = 0;
325  pkt.ip.ip_src = giaddr;
326 
327  if (inm.flags & htons(BOOTP_BROADCAST)) { /* check bcast bit */
328  pkt.ip.ip_dst = 0xffffffff; /* 255.255.255.255 */
329  }
330  else {
331  pkt.ip.ip_dst = yiaddr.s_addr;
332  }
333 
334  pkt.ip.uh_sport = htons(BOOTPS);
335  pkt.ip.uh_dport = htons(BOOTPC);
336  pkt.ip.uh_ulen = htons(udp_len);
337  pkt.ip.uh_sum = 0;
338 
339  pkt.ip.ip_sum = ~htons(ip4sum(0, &pkt.ip, 20));
340  sum = 17 + udp_len;
341  sum = ip4sum(sum, &pkt.ip.ip_src, 8);
342  sum = ip4sum(sum, &pkt.ip.uh_sport, udp_len);
343 
344  if (sum != 0xffff) {
345  pkt.ip.uh_sum = ~htons(sum);
346  }
347  else {
348  pkt.ip.uh_sum = 0xffff;
349  }
350 
351  write_to_serial(slipfd, &pkt, ip_len);
352 
353  if (msg_type == DHCPACK) {
354  printf("DHCPACK %02x:%02x:%02x:%02x:%02x:%02x:%02x:%02x IP %s\n",
355  pkt.m.chaddr[0], pkt.m.chaddr[1], pkt.m.chaddr[2], pkt.m.chaddr[3],
356  pkt.m.chaddr[4], pkt.m.chaddr[5], pkt.m.chaddr[6], pkt.m.chaddr[7],
357  inet_ntoa(yiaddr));
358  /* ssystem("arp -s %s auto pub only", inet_ntoa(yiaddr)); */
359  }
360 }

362 /*
363  * Internet checksum in host byte order.
364  */
365 u_int16_t
366 ip4sum(u_int16_t sum, const void *_p, u_int16_t len)
367 {
368  u_int16_t t;
369  const u_int8_t *p = _p;
370  const u_int8_t *end = p + len;
371 
372  while (p < (end - 1)) {
373  t = (p[0] << 8) + p[1];
374  sum += t;
375 
376  if (sum < t) {
377  sum++;
378  }
379 
380  p += 2;
381  }
382 
383  if (p < end) {
384  t = (p[0] << 8) + 0;
385  sum += t;
386 
387  if (sum < t) {
388  sum++;
389  }
390  }
391 
392  return sum;
393 }

395 int
396 check_ip(const struct ip *ip, unsigned ip_len)
397 {
398  u_int16_t sum, ip_hl;
399 
400  /* Check IP version and length. */
401  if ((ip->ip_vhl & IP_V) != IP_V4) {
402  return -1;
403  }
404 
405  if (ntohs(ip->ip_len) > ip_len) {
406  return -2;
407  }
408 
409  if (ntohs(ip->ip_len) < ip_len) {
410  return -3;
411  }
412 
413  /* Check IP header. */
414  ip_hl = 4 * (ip->ip_vhl & IP_HL);
415  sum = ip4sum(0, ip, ip_hl);
416 
417  if (sum != 0xffff && sum != 0x0) {
418  return -4;
419  }
420 
421  if (ip->ip_p == 6 || ip->ip_p == 17) { /* Check TCP or UDP header. */
422  u_int16_t tcp_len = ip_len - ip_hl;
423 
424  /* Sum pseudoheader. */
425  sum = ip->ip_p + tcp_len; /* proto and len, no carry */
426  sum = ip4sum(sum, &ip->ip_src, 8); /* src and dst */
427 
428  /* Sum TCP/UDP header and data. */
429  sum = ip4sum(sum, (u_int8_t *)ip + ip_hl, tcp_len);
430 
431  /* Failed checksum test? */
432  if (sum != 0xffff && sum != 0x0) {
433  if (ip->ip_p == 6) { /* TCP == 6 */
434  return -5;
435  }
436  else { /* UDP */
437  /* Deal with disabled UDP checksums. */
438  if (ip->uh_sum != 0) {
439  return -6;
440  }
441  }
442  }
443  }
444  else if (ip->ip_p == 1) { /* ICMP */
445  u_int16_t icmp_len = ip_len - ip_hl;
446 
447  sum = ip4sum(0, (u_int8_t *)ip + ip_hl, icmp_len);
448 
449  if (sum != 0xffff && sum != 0x0) {
450  return -7;
451  }
452  }
453 
454  return 0;
455 }

496  * Read from serial, when we have a packet write it to tun. No output
497  * buffering, input buffered by stdio.
498  */
499 void
500 serial_to_tun(FILE *inslip, int outfd)
501 {
502  static union {
503  unsigned char inbuf[2000];
504  struct ip iphdr;
505  } uip;
506  static int inbufptr = 0;
507 
508  int ret;
509  unsigned char c;
510 
511 #ifdef __linux__
512  ret = fread(&c, 1, 1, inslip);
513 
514  if (ret == -1 || ret == 0) {
515  err(1, "serial_to_tun: read");
516  }
517 
518  goto after_fread;
519 #endif
520 
521 read_more:
522 
523  if (inbufptr >= sizeof(uip.inbuf)) {
524  inbufptr = 0;
525  }
526 
527  ret = fread(&c, 1, 1, inslip);
528 #ifdef __linux__
529 after_fread:
530 #endif
531 
532  if (ret == -1) {
533  err(1, "serial_to_tun: read");
534  }
535 
536  if (ret == 0) {
537  clearerr(inslip);
538  return;
539  fprintf(stderr, "serial_to_tun: EOF\n");
540  exit(1);
541  }
542 
543  /* fprintf(stderr, ".");*/
544  switch (c) {
545  case SLIP_END:
546  if (inbufptr > 0) {
547  /*
548  * Sanity checks.
549  */
550 #define DEBUG_LINE_MARKER '\r'
551  int ecode;
552  ecode = check_ip(&uip.iphdr, inbufptr);
553 
554  if (ecode < 0 && inbufptr == 8 && strncmp(uip.inbuf, "=IPA", 4) == 0) {
555  static struct in_addr ipa;
556 
557  inbufptr = 0;
558 
559  if (memcmp(&ipa, &uip.inbuf[4], sizeof(ipa)) == 0) {
560  break;
561  }
562 
563  /* New address. */
564  if (ipa.s_addr != 0) {
565 #ifdef __linux__
566  ssystem("route delete -net %s netmask %s dev %s",
567  inet_ntoa(ipa), "255.255.255.255", tundev);
568 #else
569  ssystem("route delete -net %s -netmask %s -interface %s",
570  inet_ntoa(ipa), "255.255.255.255", tundev);
571 #endif
572  }
573 
574  memcpy(&ipa, &uip.inbuf[4], sizeof(ipa));
575 
576  if (ipa.s_addr != 0) {
577 #ifdef __linux__
578  ssystem("route add -net %s netmask %s dev %s",
579  inet_ntoa(ipa), "255.255.255.255", tundev);
580 #else
581  ssystem("route add -net %s -netmask %s -interface %s",
582  inet_ntoa(ipa), "255.255.255.255", tundev);
583 #endif
584  }
585 
586  break;
587  }
588  else if (ecode < 0) {
589  /*
590  * If sensible ASCII string, print it as debug info!
591  */
592  if (uip.inbuf[0] == DEBUG_LINE_MARKER) {
593  fwrite(uip.inbuf + 1, inbufptr - 1, 1, stderr);
594  }
595  else if (is_sensible_string(uip.inbuf, inbufptr)) {
596  fwrite(uip.inbuf, inbufptr, 1, stderr);
597  }
598  else {
599  fprintf(stderr,
600  "serial_to_tun: drop packet len=%d ecode=%d\n",
601  inbufptr, ecode);
602  }
603 
604  inbufptr = 0;
605  break;
606  }
607 
608  PROGRESS("s");
609 
610  if (dhsock != -1) {
611  struct ip *ip = (void *)uip.inbuf;
612 
613  if (ip->ip_p == 17 && ip->ip_dst == 0xffffffff /* UDP and broadcast */
614  && ip->uh_sport == ntohs(BOOTPC) && ip->uh_dport == ntohs(BOOTPS)) {
615  relay_dhcp_to_server(ip, inbufptr);
616  inbufptr = 0;
617  }
618  }
619 
620  if (write(outfd, uip.inbuf, inbufptr) != inbufptr) {
621  err(1, "serial_to_tun: write");
622  }
623 
624  inbufptr = 0;
625  }
626 
627  break;
628 
629  case SLIP_ESC:
630  if (fread(&c, 1, 1, inslip) != 1) {
631  clearerr(inslip);
632  /* Put ESC back and give up! */
633  ungetc(SLIP_ESC, inslip);
634  return;
635  }
636 
637  switch (c) {
638  case SLIP_ESC_END:
639  c = SLIP_END;
640  break;
641 
642  case SLIP_ESC_ESC:
643  c = SLIP_ESC;
644  break;
645  }
646 
647  /* FALLTHROUGH */
648  default:
649  uip.inbuf[inbufptr++] = c;
650  break;
651  }
652 
653  goto read_more;
654 }
