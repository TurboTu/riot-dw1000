229 static int gnrc_sending(char *addr_str, char *data, size_t data_len )
230 {
231  ipv6_addr_t addr;
232  gnrc_pktsnip_t *payload, *udp, *ip;
233 
234  /* parse destination address */
235  if (ipv6_addr_from_str(&addr, addr_str) == NULL) {
236  puts("Error: unable to parse destination address");
237  return -1;
238  }
239 
240  /* allocate payload */
241  payload = gnrc_pktbuf_add(NULL, data, data_len, GNRC_NETTYPE_UNDEF);
242 
243  if (payload == NULL) {
244  puts("Error: unable to copy data to packet buffer");
245  return -1;
246  }
247 
248  /* allocate UDP header */
249  udp = gnrc_udp_hdr_build(payload, (uint16_t) CLIENT_PORT, (uint16_t) DEFAULT_PORT);
250  if (udp == NULL) {
251  puts("Error: unable to allocate UDP header");
252  gnrc_pktbuf_release(payload);
253  return -1;
254  }
255 
256  /* allocate IPv6 header */
257  ip = gnrc_ipv6_hdr_build(udp, NULL, &addr);
258  if (ip == NULL) {
259  puts("Error: unable to allocate IPv6 header");
260  gnrc_pktbuf_release(udp);
261  return -1;
262  }
263 
264  /*
265  * WARNING: Too fast and the nodes dies in middle of retransmissions.
266  * This issue appears in the FIT-Lab (m3 motes).
267  * In native, is not required.
268  */
269  xtimer_usleep(500000);
270 
271  /* send packet */
272  if (!gnrc_netapi_dispatch_send(GNRC_NETTYPE_UDP, GNRC_NETREG_DEMUX_CTX_ALL, ip)) {
273  puts("Error: unable to locate UDP thread");
274  gnrc_pktbuf_release(ip);
275  return -1;
276  }
277 
278  return 1;
279 }

368 #ifdef DTLS_PSK
369  puts("Client support PSK");
370 #endif

373 #endif
374 
375  DEBUG("DBG-Client: Debug ON");
376  /*
377  * The objective of ctx->App is be able to retrieve
378  * enough information for restablishing a connection.
379  * This is to be used in the send_to_peer and (potentially) the
380  * read_from_peer, to continue the transmision with the next
381  * step of the DTLS flights.
382  * TODO: Take away the DEFAULT_PORT and CLIENT_PORT.
383  */
384  dst->size = sizeof(ipv6_addr_t) + sizeof(unsigned short);

411 static void client_send(char *addr_str, char *data, unsigned int delay)
412 {
413  static int8_t iWatch;
414  static session_t dst;
415  static int connected = 0;
416  msg_t msg;
417 
418  gnrc_netreg_entry_t entry = GNRC_NETREG_ENTRY_INIT_PID(CLIENT_PORT,
419  sched_active_pid);
420  dtls_init();
421 
422  if (gnrc_netreg_register(GNRC_NETTYPE_UDP, &entry)) {
423  puts("Unable to register ports");
424  /*FIXME: Release memory?*/
425  return;
426  }
427 
428  if (strlen(data) > DTLS_MAX_BUF) {
429  puts("Data too long ");
430  return;
431  }
432 
433  init_dtls(&dst, addr_str);
434  if (!dtls_context) {
435  dtls_emerg("cannot create context\n");
436  puts("Client unable to load context!");
437  return;
438  }
439 
440  /* client_payload is global due to the SIGNAL function send_to_peer */
441  client_payload = data;
442  buflen = strlen(client_payload);
443  iWatch = MAX_TIMES_TRY_TO_SEND;
444 
445  /*
446  * dtls_connect is the one who begin all the process.
447  * However, do it too fast, and the node will attend it before having a
448  * valid IPv6 or even a route to the destiny (This could be verified as the
449  * sequence number of the first DTLS Hello message will be greater than
450  * zero).
451  */
452  //connected = dtls_connect(dtls_context, &dst) >= 0;
453 
454  /*
455  * Until all the data is not sent we remains trying to connect to the
456  * server. Plus a small watchdog.
457  */
458  while ((buflen > 0) && (iWatch > 0)) {
459 
460  /*
461  * NOTE: I (rfuentess) personally think this should be until this point
462  * instead before (that is why the previous dtls_connect is by defualt
463  * commented..
464  */
465  if (!connected) {
466  connected = dtls_connect(dtls_context, &dst);
467  }
468  else if (connected < 0) {
469  puts("Client DTLS was unable to establish a channel!\n");
470  /*NOTE: Not sure what to do in this scenario (if can happens)*/
471  }
472  else {
473  /*TODO: must happens always or only when connected?*/
474  try_send(dtls_context, &dst);
475  }
476 
477  /*
478  * WARNING: The delay is KEY HERE! Too fast, and we can kill the
479  * DTLS state machine. Another alternative is change to
480  * blocking states (making the watchdog useless)
481  *
482  * msg_receive(&msg);
483  */
484 
485  xtimer_usleep(delay);
486 
487  if (msg_try_receive(&msg) == 1) {
488  dtls_handle_read(dtls_context, (gnrc_pktsnip_t *)(msg.content.ptr));
489  }
490 
491  iWatch--;
492  } /*END while*/
493 
494  dtls_free_context(dtls_context);
495  /* unregister our UDP listener on this thread */
496  gnrc_netreg_unregister(GNRC_NETTYPE_UDP, &entry);
497  connected = 0; /*Probably this should be removed or global */
498 
499  /* Permanent or not permanent? */
500  DEBUG("DTLS-Client: DTLS session finished\n");
501 }
