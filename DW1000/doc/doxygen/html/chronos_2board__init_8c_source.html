33 void cc430_cpu_init(void)
34 {
35  volatile uint16_t i;
36  volatile unsigned char *ptr;
37 
38  /* disable watchdog */
39  WDTCTL = WDTPW + WDTHOLD;
40 
41  // ---------------------------------------------------------------------
42  // Enable 32kHz ACLK
43  P5SEL |= 0x03; // Select XIN, XOUT on P5.0 and P5.1
44  UCSCTL6 &= ~XT1OFF; // XT1 On, Highest drive strength
45  UCSCTL6 |= XCAP_3; // Internal load cap
46 
47  UCSCTL3 = SELA__XT1CLK; // Select XT1 as FLL reference
48  UCSCTL4 = SELA__XT1CLK | SELS__DCOCLKDIV | SELM__DCOCLKDIV;
49 
50  // ---------------------------------------------------------------------
51  // Configure CPU clock for 12MHz
52  _BIS_SR(SCG0); // Disable the FLL control loop
53  UCSCTL0 = 0x0000; // Set lowest possible DCOx, MODx
54  UCSCTL1 = DCORSEL_5; // Select suitable range
55  UCSCTL2 = FLLD_1 + 0x16E; // Set DCO Multiplier
56  _BIC_SR(SCG0); // Enable the FLL control loop
57 
58  // Worst-case settling time for the DCO when the DCO range bits have been
59  // changed is n x 32 x 32 x f_MCLK / f_FLL_reference. See UCS chapter in 5xx
60  // UG for optimization.
61  // 32 x 32 x 8 MHz / 32,768 Hz = 250000 = MCLK cycles for DCO to settle
62  for (i = 0xFF; i > 0; i--); // Time for flag to set
63 
64  // Loop until XT1 & DCO stabilizes, use do-while to insure that
65  // body is executed at least once
66  do {
67  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + XT1HFOFFG + DCOFFG);
68  SFRIFG1 &= ~OFIFG; // Clear fault flags
69  }
70  while ((SFRIFG1 & OFIFG));
71 
72  // Disable all interrupts
73  __disable_interrupt();
74  // Get write-access to port mapping registers:
75  PMAPPWD = 0x02D52;
76  // Allow reconfiguration during runtime:
77  PMAPCTL = PMAPRECFG;
78 
79  // P2.7 = TA0CCR1A or TA1CCR0A output (buzzer output)
80  ptr = &P2MAP0;
81  *(ptr + 7) = PM_TA1CCR0A;
82  P2OUT &= ~BIT7;
83  P2DIR |= BIT7;
84 
85  // P1.5 = SPI MISO input
86  ptr = &P1MAP0;
87  *(ptr + 5) = PM_UCA0SOMI;
88  // P1.6 = SPI MOSI output
89  *(ptr + 6) = PM_UCA0SIMO;
90  // P1.7 = SPI CLK output
91  *(ptr + 7) = PM_UCA0CLK;
92 
93  // Disable write-access to port mapping registers:
94  PMAPPWD = 0;
95  // Re-enable all interrupts
96  irq_enable();
97 
98 }
