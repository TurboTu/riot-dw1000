40 static void _nc_remove(kernel_pid_t iface, gnrc_ipv6_nc_t *entry)
41 {
42  (void) iface;
43  if (entry == NULL) {
44  return;
45  }
46 
47  DEBUG("ipv6_nc: Remove %s for interface %" PRIkernel_pid "\n",
48  ipv6_addr_to_str(addr_str, &(entry->ipv6_addr), sizeof(addr_str)),
49  iface);
50 
51 #ifdef MODULE_GNRC_NDP_NODE
52  while (entry->pkts != NULL) {
53  gnrc_pktbuf_release(entry->pkts->pkt);
54  entry->pkts->pkt = NULL;
55  gnrc_pktqueue_remove_head(&entry->pkts);
56  }
57 #endif
58 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
59  xtimer_remove(&entry->type_timeout);
60 
61  gnrc_ipv6_netif_t *if_entry = gnrc_ipv6_netif_get(iface);
62 
63  if ((if_entry != NULL) && (if_entry->rtr_adv_msg.content.ptr == entry)) {
64  /* cancel timer set by gnrc_ndp_rtr_sol_handle */
65  xtimer_remove(&if_entry->rtr_adv_timer);
66  }
67 #endif
68 #if defined(MODULE_GNRC_NDP_ROUTER) || defined(MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER)
69  xtimer_remove(&entry->rtr_adv_timer);
70 #endif
71 
72  xtimer_remove(&entry->rtr_timeout);
73  xtimer_remove(&entry->nbr_sol_timer);
74  xtimer_remove(&entry->nbr_adv_timer);
75 
76  ipv6_addr_set_unspecified(&(entry->ipv6_addr));
77  entry->iface = KERNEL_PID_UNDEF;
78  entry->flags = 0;
79 }

102 gnrc_ipv6_nc_t *gnrc_ipv6_nc_add(kernel_pid_t iface, const ipv6_addr_t *ipv6_addr,
103  const void *l2_addr, size_t l2_addr_len, uint8_t flags)
104 {
105  gnrc_ipv6_nc_t *free_entry = NULL;
106 
107  if (ipv6_addr == NULL) {
108  DEBUG("ipv6_nc: address was NULL\n");
109  return NULL;
110  }
111 
112  if ((l2_addr_len > GNRC_IPV6_NC_L2_ADDR_MAX) || ipv6_addr_is_unspecified(ipv6_addr)) {
113  DEBUG("ipv6_nc: invalid parameters\n");
114  return NULL;
115  }
116 
117  for (int i = 0; i < GNRC_IPV6_NC_SIZE; i++) {
118  if (ipv6_addr_equal(&(ncache[i].ipv6_addr), ipv6_addr)) {
119  DEBUG("ipv6_nc: Address %s already registered.\n",
120  ipv6_addr_to_str(addr_str, ipv6_addr, sizeof(addr_str)));
121 
122  if ((l2_addr != NULL) && (l2_addr_len > 0)) {
123  DEBUG("ipv6_nc: Update to L2 address %s",
124  gnrc_netif_addr_to_str(addr_str, sizeof(addr_str),
125  l2_addr, l2_addr_len));
126 
127  memcpy(&(ncache[i].l2_addr), l2_addr, l2_addr_len);
128  ncache[i].l2_addr_len = l2_addr_len;
129  ncache[i].flags = flags;
130  DEBUG(" with flags = 0x%0x\n", flags);
131 
132  }
133  return &ncache[i];
134  }
135 
136  if (ipv6_addr_is_unspecified(&(ncache[i].ipv6_addr)) && !free_entry) {
137  /* found the first free entry */
138  free_entry = &ncache[i];
139  }
140  }
141 
142  if (!free_entry) {
143  /* reached end of NC without finding updateable or free entry */
144  DEBUG("ipv6_nc: neighbor cache full.\n");
145  return NULL;
146  }
147 
148  /* Otherwise, fill free entry with your fresh information */
149  free_entry->iface = iface;
150 
151 #ifdef MODULE_GNRC_NDP_NODE
152  free_entry->pkts = NULL;
153 #endif
154  memcpy(&(free_entry->ipv6_addr), ipv6_addr, sizeof(ipv6_addr_t));
155  DEBUG("ipv6_nc: Register %s for interface %" PRIkernel_pid,
156  ipv6_addr_to_str(addr_str, ipv6_addr, sizeof(addr_str)),
157  iface);
158 
159  if ((l2_addr != NULL) && (l2_addr_len > 0)) {
160  DEBUG(" to L2 address %s",
161  gnrc_netif_addr_to_str(addr_str, sizeof(addr_str),
162  l2_addr, l2_addr_len));
163  memcpy(&(free_entry->l2_addr), l2_addr, l2_addr_len);
164  free_entry->l2_addr_len = l2_addr_len;
165  }
166 
167  free_entry->flags = flags;
168 
169  DEBUG(" with flags = 0x%0x\n", flags);
170 
171  if (gnrc_ipv6_nc_get_state(free_entry) == GNRC_IPV6_NC_STATE_INCOMPLETE) {
172  DEBUG("ipv6_nc: Set remaining probes to %" PRIu8 "\n", (uint8_t) GNRC_NDP_MAX_MC_NBR_SOL_NUMOF);
173  free_entry->probes_remaining = GNRC_NDP_MAX_MC_NBR_SOL_NUMOF;
174  }
175 
176 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
177  free_entry->type_timeout_msg.type = GNRC_SIXLOWPAN_ND_MSG_AR_TIMEOUT;
178  free_entry->type_timeout_msg.content.ptr = free_entry;
179 #endif
180 
181  free_entry->rtr_timeout_msg.type = GNRC_NDP_MSG_RTR_TIMEOUT;
182  free_entry->rtr_timeout_msg.content.ptr = free_entry;
183 
184 #if defined(MODULE_GNRC_NDP_ROUTER) || defined(MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER)
185  free_entry->rtr_adv_msg.type = GNRC_NDP_MSG_RTR_ADV_DELAY;
186  free_entry->rtr_adv_msg.content.ptr = free_entry;
187 #endif
188 
189  free_entry->nbr_sol_msg.content.ptr = free_entry;
190 
191  return free_entry;
192 }

200 gnrc_ipv6_nc_t *gnrc_ipv6_nc_get(kernel_pid_t iface, const ipv6_addr_t *ipv6_addr)
201 {
202  if ((ipv6_addr == NULL) || (ipv6_addr_is_unspecified(ipv6_addr))) {
203  DEBUG("ipv6_nc: address was NULL or ::\n");
204  return NULL;
205  }
206 
207  for (int i = 0; i < GNRC_IPV6_NC_SIZE; i++) {
208  if (((ncache[i].iface == KERNEL_PID_UNDEF) || (iface == KERNEL_PID_UNDEF) ||
209  (iface == ncache[i].iface)) &&
210  ipv6_addr_equal(&(ncache[i].ipv6_addr), ipv6_addr)) {
211  DEBUG("ipv6_nc: Found entry for %s on interface %" PRIkernel_pid
212  " (0 = all interfaces) [%p]\n",
213  ipv6_addr_to_str(addr_str, ipv6_addr, sizeof(addr_str)),
214  iface, (void *)(ncache + i));
215 
216  return ncache + i;
217  }
218  }
219 
220  return NULL;
221 }

223 gnrc_ipv6_nc_t *gnrc_ipv6_nc_get_next(gnrc_ipv6_nc_t *prev)
224 {
225  if (prev == NULL) {
226  prev = ncache;
227  }
228  else {
229  prev++; /* get next entry */
230  }
231 
232  while (prev < (ncache + GNRC_IPV6_NC_SIZE)) { /* while not reached end */
233  if (!ipv6_addr_is_unspecified(&(prev->ipv6_addr))) {
234  return prev;
235  }
236 
237  prev++;
238  }
239 
240  return NULL;

255 gnrc_ipv6_nc_t *gnrc_ipv6_nc_still_reachable(const ipv6_addr_t *ipv6_addr)
256 {
257  gnrc_ipv6_nc_t *entry = gnrc_ipv6_nc_get(KERNEL_PID_UNDEF, ipv6_addr);
258 
259  if (entry == NULL) {
260  DEBUG("ipv6_nc: No entry found for %s\n",
261  ipv6_addr_to_str(addr_str, ipv6_addr, sizeof(addr_str)));
262  return NULL;
263  }
264 
265  if ((gnrc_ipv6_nc_get_state(entry) != GNRC_IPV6_NC_STATE_INCOMPLETE) &&
266  (gnrc_ipv6_nc_get_state(entry) != GNRC_IPV6_NC_STATE_UNMANAGED)) {
267 #if defined(MODULE_GNRC_IPV6_NETIF) && defined(MODULE_XTIMER) && defined(MODULE_GNRC_IPV6)
268  gnrc_ipv6_netif_t *iface = gnrc_ipv6_netif_get(entry->iface);
269 
270  gnrc_ndp_internal_reset_nbr_sol_timer(entry, iface->reach_time,
271  GNRC_NDP_MSG_NC_STATE_TIMEOUT, gnrc_ipv6_pid);
272 #endif
273 
274  DEBUG("ipv6_nc: Marking entry %s as reachable\n",
275  ipv6_addr_to_str(addr_str, ipv6_addr, sizeof(addr_str)));
276  entry->flags &= ~(GNRC_IPV6_NC_STATE_MASK >> GNRC_IPV6_NC_STATE_POS);
277  entry->flags |= (GNRC_IPV6_NC_STATE_REACHABLE >> GNRC_IPV6_NC_STATE_POS);
278  }
279 
280  return entry;
281 }
