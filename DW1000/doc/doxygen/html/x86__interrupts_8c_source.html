171 static unsigned in_intr_handler = 0, old_intr;
172 __attribute__((used)) void x86_int_handler(void)
173 {
174  switch (in_intr_handler++) {
175  case 0:
176  break;
177  case 1:
178  printf("Interrupt 0x%02x (%s) while handling 0x%02x (%s)\n",
179  x86_current_interrupt, exception_name[x86_current_interrupt],
180  old_intr, exception_name[old_intr]);
181  x86_print_registers(&x86_interrupted_ctx, x86_current_interrupt_error_code);
182  puts("Halting.");
183  x86_hlt();
184  default:
185  x86_hlt();
186  }
187  old_intr = x86_current_interrupt;
188 
189  bool old_in_isr = x86_in_isr;
190  x86_in_isr = true;
191 
192  x86_intr_handler_t intr_handler = x86_registered_interrupts[x86_current_interrupt];
193  (intr_handler ? intr_handler : intr_handler_default)(x86_current_interrupt, &x86_interrupted_ctx, x86_current_interrupt_error_code);
194 
195  --in_intr_handler;
196 
197  unsigned long *sp = (void *) x86_interrupted_ctx.sp; /* ip, cs, flags */
198  if (!sched_context_switch_request || !(sp[2] & X86_IF)) {
199  x86_in_isr = old_in_isr;
200  return;
201  }
202 
203  ucontext_t *ctx = (ucontext_t *) sched_active_thread->sp;
204  ctx->uc_context.registers = x86_interrupted_ctx;
205  ctx->uc_stack.ss_sp = x86_interrupt_handler_stack;
206  ctx->uc_stack.ss_size = sizeof x86_interrupt_handler_stack;
207  __asm__ volatile ("pushf; pop %0" : "=g"(ctx->uc_context.flags));
208  ctx->uc_context.ip = (void *) (uintptr_t) &continue_after_intr;
209  ctx->__intr.ip = sp[0];
210  ctx->__intr.flags = sp[2];
211 
212  thread_yield_higher();
213  __builtin_unreachable();
214 }

313 {

334 static void test_int_bp(void)
335 {
336  x86_registered_interrupts[X86_INT_BP] = intr_handler_test_int_bp;
337 
338  unsigned long ax = INTR_TEST_REG_AX;
339  unsigned long dx = INTR_TEST_REG_DX;
340  unsigned long cx = INTR_TEST_REG_CX;
341  unsigned long bx = INTR_TEST_REG_BX;
342  unsigned long si;
343  unsigned long di;
344  unsigned long eflags_before, eflags_after;
345  __asm__ volatile (
346  "mov %8, %%esi\n"
347  "mov %9, %%edi\n"
348  "pushf; pop %6\n"
349  "int3\n"
350  "pushf; pop %7\n"
351  "mov %%esi, %4\n"
352  "mov %%edi, %5\n"
353  : "+a"(ax), "+d"(dx), "+c"(cx), "+b"(bx), "=g"(si), "=g"(di), "=g"(eflags_before), "=g"(eflags_after)
354  : "n"(INTR_TEST_REG_SI), "n"(INTR_TEST_REG_DI)
355  : "esi", "edi"
356  );
357  if (ax != (INTR_TEST_REG_AX ^ -1) ||
358  dx != (INTR_TEST_REG_DX ^ -2) ||
359  cx != (INTR_TEST_REG_CX ^ -3) ||
360  bx != (INTR_TEST_REG_BX ^ -4) ||
361  si != (INTR_TEST_REG_SI ^ -5) ||
362  di != (INTR_TEST_REG_DI ^ -6) ||
363  /* ignore EFLAGS.TF, hurts debugging */
364  ((eflags_before != eflags_after) && ((eflags_before ^ eflags_after) != X86_TF))
365  ) {
366  puts("Interrupt handler test failed (int 3, return code).");
367  x86_hlt();
368  }
369 
370  x86_registered_interrupts[X86_INT_BP] = NULL;
371 }
