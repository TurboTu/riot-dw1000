69 bool __pthread_rwlock_blocked_readingly(const pthread_rwlock_t *rwlock)
70 {
71  if (rwlock->readers < 0) {
72  /* a writer holds the lock */
73  return true;
74  }
75 
76  /* Determine if there is a writer waiting to get this lock who has a higher or the same priority: */
77 
78  if (rwlock->queue.first == NULL) {
79  /* no waiting thread */
80  return false;
81  }
82 
83  priority_queue_node_t *qnode = rwlock->queue.first;
84  if (qnode->priority > sched_active_thread->priority) {
85  /* the waiting thread has a lower priority */
86  return false;
87  }
88 
89  /* if the waiting node is a writer, then we cannot enter the critical section (to prevent starving the writer) */
90  __pthread_rwlock_waiter_node_t *waiting_node = (__pthread_rwlock_waiter_node_t *) qnode->data;
91  return waiting_node->is_writer;
92 }

94 bool __pthread_rwlock_blocked_writingly(const pthread_rwlock_t *rwlock)
95 {
96  /* if any thread holds the lock, then no writer may enter the critical section */
97  return rwlock->readers != 0;

100 static int pthread_rwlock_lock(pthread_rwlock_t *rwlock,
101  bool (*is_blocked)(const pthread_rwlock_t *rwlock),
102  bool is_writer,
103  int incr_when_held,
104  bool allow_spurious)
105 {
106  if (rwlock == NULL) {
107  DEBUG("Thread %" PRIkernel_pid": pthread_rwlock_%s(): is_writer=%u, allow_spurious=%u %s\n",
108  thread_pid, "lock", is_writer, allow_spurious, "rwlock=NULL");
109  return EINVAL;
110  }
111 
112  mutex_lock(&rwlock->mutex);
113  if (!is_blocked(rwlock)) {
114  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): is_writer=%u, allow_spurious=%u %s\n",
115  thread_pid, "lock", is_writer, allow_spurious, "is open");
116  rwlock->readers += incr_when_held;
117  }
118  else {
119  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): is_writer=%u, allow_spurious=%u %s\n",
120  thread_pid, "lock", is_writer, allow_spurious, "is locked");
121 
122  /* queue for the lock */
123  __pthread_rwlock_waiter_node_t waiting_node = {
124  .is_writer = is_writer,
125  .thread = (thread_t *) sched_active_thread,
126  .qnode = {
127  .next = NULL,
128  .data = (uintptr_t) &waiting_node,
129  .priority = sched_active_thread->priority,
130  },
131  .continue_ = false,
132  };
133  priority_queue_add(&rwlock->queue, &waiting_node.qnode);
134 
135  while (1) {
136  /* wait to be unlocked, so this thread can try to acquire the lock again */
137  mutex_unlock_and_sleep(&rwlock->mutex);
138 
139  mutex_lock(&rwlock->mutex);
140  if (waiting_node.continue_) {
141  /* pthread_rwlock_unlock() already set rwlock->readers */
142  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): is_writer=%u, allow_spurious=%u %s\n",
143  thread_pid, "lock", is_writer, allow_spurious, "continued");
144  break;
145  }
146  else if (allow_spurious) {
147  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): is_writer=%u, allow_spurious=%u %s\n",
148  thread_pid, "lock", is_writer, allow_spurious, "is timed out");
149  priority_queue_remove(&rwlock->queue, &waiting_node.qnode);
150  mutex_unlock(&rwlock->mutex);
151  return ETIMEDOUT;
152  }
153  }
154  }
155  mutex_unlock(&rwlock->mutex);
156 
157  return 0;
158 }

160 static int pthread_rwlock_trylock(pthread_rwlock_t *rwlock,
161  bool (*is_blocked)(const pthread_rwlock_t *rwlock),
162  int incr_when_held)
163 {
164  if (rwlock == NULL) {
165  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): rwlock=NULL supplied\n", thread_pid, "trylock");
166  return EINVAL;
167  }
168  else if (mutex_trylock(&rwlock->mutex) == 0) {
169  return EBUSY;
170  }
171  else if (is_blocked(rwlock)) {
172  mutex_unlock(&rwlock->mutex);
173  return EBUSY;
174  }
175 
176  rwlock->readers += incr_when_held;
177 
178  mutex_unlock(&rwlock->mutex);
179  return 0;
180 }

182 static int pthread_rwlock_timedlock(pthread_rwlock_t *rwlock,
183  bool (*is_blocked)(const pthread_rwlock_t *rwlock),
184  bool is_writer,
185  int incr_when_held,
186  const struct timespec *abstime)
187 {
188  timex_t now, then;
189 
190  then.seconds = abstime->tv_sec;
191  then.microseconds = abstime->tv_nsec / 1000u;
192  timex_normalize(&then);
193 
194  xtimer_now_timex(&now);
195 
196  if (timex_cmp(then, now) <= 0) {
197  return ETIMEDOUT;
198  }
199  else {
200  timex_t reltime = timex_sub(then, now);
201 
202  xtimer_t timer;
203  xtimer_set_wakeup64(&timer, timex_uint64(reltime) , sched_active_pid);
204  int result = pthread_rwlock_lock(rwlock, is_blocked, is_writer, incr_when_held, true);
205  if (result != ETIMEDOUT) {
206  xtimer_remove(&timer);
207  }
208 
209  return result;
210  }
211 }

243 int pthread_rwlock_unlock(pthread_rwlock_t *rwlock)
244 {
245  if (rwlock == NULL) {
246  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): rwlock=NULL supplied\n", thread_pid, "unlock");
247  return EINVAL;
248  }
249 
250  mutex_lock(&rwlock->mutex);
251  if (rwlock->readers == 0) {
252  /* the lock is open */
253  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): lock is open\n", thread_pid, "unlock");
254  mutex_unlock(&rwlock->mutex);
255  return EPERM;
256  }
257 
258  if (rwlock->readers > 0) {
259  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): release %s lock\n", thread_pid, "unlock", "read");
260  --rwlock->readers;
261  }
262  else {
263  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): release %s lock\n", thread_pid, "unlock", "write");
264  rwlock->readers = 0;
265  }
266 
267  if (rwlock->readers != 0 || rwlock->queue.first == NULL) {
268  /* this thread was not the last reader, or no one is waiting to aquire the lock */
269  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): no one is waiting\n", thread_pid, "unlock");
270  mutex_unlock(&rwlock->mutex);
271  return 0;
272  }
273 
274  /* wake up the next thread */
275  priority_queue_node_t *qnode = priority_queue_remove_head(&rwlock->queue);
276  __pthread_rwlock_waiter_node_t *waiting_node = (__pthread_rwlock_waiter_node_t *) qnode->data;
277  waiting_node->continue_ = true;
278  uint16_t prio = qnode->priority;
279  sched_set_status(waiting_node->thread, STATUS_PENDING);
280 
281  if (waiting_node->is_writer) {
282  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): continue %s %" PRIkernel_pid "\n",
283  thread_pid, "unlock", "writer", waiting_node->thread->pid);
284  --rwlock->readers;
285  }
286  else {
287  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): continue %s %" PRIkernel_pid "\n",
288  thread_pid, "unlock", "reader", waiting_node->thread->pid);
289  ++rwlock->readers;
290 
291  /* wake up further readers */
292  while (rwlock->queue.first) {
293  waiting_node = (__pthread_rwlock_waiter_node_t *) rwlock->queue.first->data;
294  if (waiting_node->is_writer) {
295  /* Not to be unfair to writers, we don't try to wake up readers that came after the first writer. */
296  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): continuing readers blocked by writer %" PRIkernel_pid "\n",
297  thread_pid, "unlock", waiting_node->thread->pid);
298  break;
299  }
300  waiting_node->continue_ = true;
301  DEBUG("Thread %" PRIkernel_pid ": pthread_rwlock_%s(): continue %s %" PRIkernel_pid "\n",
302  thread_pid, "unlock", "reader", waiting_node->thread->pid);
303 
304  /* wake up this reader */
305  qnode = priority_queue_remove_head(&rwlock->queue);
306  if (qnode->priority < prio) {
307  prio = qnode->priority;
308  }
309  sched_set_status(waiting_node->thread, STATUS_PENDING);
310 
311  ++rwlock->readers;
312  }
313  }
314 
315  mutex_unlock(&rwlock->mutex);
316 
317  /* yield if a woken up thread had a higher priority */
318  sched_switch(prio);
319  return 0;
320 }
