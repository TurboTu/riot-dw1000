29  * Starts a blocking and never-returning loop dispatching CoAP requests.
30  *
31  * When using gnrc, make sure the calling thread has an initialized msg queue.
32  */
33 void microcoap_server_loop(void)
34 {
35 
36  static const sock_udp_ep_t local = { .family = AF_INET6,
37  .port = COAP_SERVER_PORT };
38  sock_udp_ep_t remote;
39 
40  sock_udp_t sock;
41 
42  int rc = sock_udp_create(&sock, &local, NULL, 0);
43 
44  while (1) {
45  DEBUG("Waiting for incoming UDP packet...\n");
46  rc = sock_udp_recv(&sock, (char *)_udp_buf, sizeof(_udp_buf),
47  SOCK_NO_TIMEOUT, &remote);
48  if (rc < 0) {
49  DEBUG("Error in conn_udp_recvfrom(). rc=%u\n", rc);
50  continue;
51  }
52 
53  size_t n = rc;
54 
55  coap_packet_t pkt;
56  DEBUG("Received packet: ");
57  coap_dump(_udp_buf, n, true);
58  DEBUG("\n");
59 
60  /* parse UDP packet to CoAP */
61  if (0 != (rc = coap_parse(&pkt, _udp_buf, n))) {
62  DEBUG("Bad packet rc=%d\n", rc);
63  }
64  else {
65  coap_packet_t rsppkt;
66  DEBUG("content:\n");
67  coap_dumpPacket(&pkt);
68 
69  /* handle CoAP request */
70  coap_handle_req(&scratch_buf, &pkt, &rsppkt);
71 
72  /* build reply */
73  size_t rsplen = sizeof(_udp_buf);
74  if ((rc = coap_build(_udp_buf, &rsplen, &rsppkt)) != 0) {
75  DEBUG("coap_build failed rc=%d\n", rc);
76  }
77  else {
78  DEBUG("Sending packet: ");
79  coap_dump(_udp_buf, rsplen, true);
80  DEBUG("\n");
81  DEBUG("content:\n");
82  coap_dumpPacket(&rsppkt);
83 
84  /* send reply via UDP */
85  rc = sock_udp_send(&sock, _udp_buf, rsplen, &remote);
86  if (rc < 0) {
87  DEBUG("Error sending CoAP reply via udp; %u\n", rc);
88  }
89  }
90  }
91  }
92 }
