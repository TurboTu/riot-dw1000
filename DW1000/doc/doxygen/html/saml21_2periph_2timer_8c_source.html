48 int timer_init(tim_t dev, unsigned long freq, timer_cb_t cb, void *arg)
49 {
50  /* configure GCLK0 to feed TC0 & TC1*/;
51  GCLK->PCHCTRL[TC0_GCLK_ID].reg |= GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN_GCLK0;
52  while (!(GCLK->PCHCTRL[TC0_GCLK_ID].reg & GCLK_PCHCTRL_CHEN)) {}
53 
54  /* select the timer and enable the timer specific peripheral clocks */
55  switch (dev) {
56 #if TIMER_0_EN
57  case TIMER_0:
58  if (TIMER_0_DEV.CTRLA.bit.ENABLE) {
59  return 0;
60  }
61  MCLK->APBCMASK.reg |= MCLK_APBCMASK_TC0;
62  /* reset timer */
63  TIMER_0_DEV.CTRLA.bit.SWRST = 1;
64  while (TIMER_0_DEV.SYNCBUSY.bit.SWRST) {}
65  TIMER_0_DEV.CTRLA.reg |= TC_CTRLA_MODE_COUNT32 | /* choosing 32 bit mode */
66  TC_CTRLA_PRESCALER(4) | /* sourced by 4MHz with Presc 4 results in 1MHz*/
67  TC_CTRLA_PRESCSYNC_RESYNC; /* initial prescaler resync */
68  break;
69 #endif
70  case TIMER_UNDEFINED:
71  default:
72  return -1;
73  }
74 
75  /* save callback */
76  config[dev].cb = cb;
77  config[dev].arg = arg;
78

80  _irq_enable(dev);
