28 static void clk_init(void)
29 {
30  /* enable clocks for the power, sysctrl and gclk modules */
31  PM->APBAMASK.reg = (PM_APBAMASK_PM | PM_APBAMASK_SYSCTRL |
32  PM_APBAMASK_GCLK);
33 
34  /* adjust NVM wait states, see table 42.30 (p. 1070) in the datasheet */
35 #if (CLOCK_CORECLOCK > 24000000)
36  PM->APBAMASK.reg |= PM_AHBMASK_NVMCTRL;
37  NVMCTRL->CTRLB.reg |= NVMCTRL_CTRLB_RWS(1);
38  PM->APBAMASK.reg &= ~PM_AHBMASK_NVMCTRL;
39 #endif
40 
41  /* configure internal 8MHz oscillator to run without prescaler */
42  SYSCTRL->OSC8M.bit.PRESC = 0;
43  SYSCTRL->OSC8M.bit.ONDEMAND = 1;
44  SYSCTRL->OSC8M.bit.RUNSTDBY = 0;
45  SYSCTRL->OSC8M.bit.ENABLE = 1;
46  while (!(SYSCTRL->PCLKSR.reg & SYSCTRL_PCLKSR_OSC8MRDY)) {}
47 
48 #if CLOCK_USE_PLL
49  /* reset the GCLK module so it is in a known state */
50  GCLK->CTRL.reg = GCLK_CTRL_SWRST;
51  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
52 
53  /* setup generic clock 1 to feed DPLL with 1MHz */
54  GCLK->GENDIV.reg = (GCLK_GENDIV_DIV(8) |
55  GCLK_GENDIV_ID(1));
56  GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN |
57  GCLK_GENCTRL_SRC_OSC8M |
58  GCLK_GENCTRL_ID(1));
59  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_GEN(1) |
60  GCLK_CLKCTRL_ID(1) |
61  GCLK_CLKCTRL_CLKEN);
62  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
63 
64  /* enable PLL */
65  SYSCTRL->DPLLRATIO.reg = (SYSCTRL_DPLLRATIO_LDR(CLOCK_PLL_MUL));
66  SYSCTRL->DPLLCTRLB.reg = (SYSCTRL_DPLLCTRLB_REFCLK_GCLK);
67  SYSCTRL->DPLLCTRLA.reg = (SYSCTRL_DPLLCTRLA_ENABLE);
68  while(!(SYSCTRL->DPLLSTATUS.reg &
69  (SYSCTRL_DPLLSTATUS_CLKRDY | SYSCTRL_DPLLSTATUS_LOCK))) {}
70 
71  /* select the PLL as source for clock generator 0 (CPU core clock) */
72  GCLK->GENDIV.reg = (GCLK_GENDIV_DIV(CLOCK_PLL_DIV) |
73  GCLK_GENDIV_ID(0));
74  GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN |
75  GCLK_GENCTRL_SRC_FDPLL |
76  GCLK_GENCTRL_ID(0));
77 #else /* do not use PLL, use internal 8MHz oscillator directly */
78  GCLK->GENDIV.reg = (GCLK_GENDIV_DIV(CLOCK_DIV) |
79  GCLK_GENDIV_ID(0));
80  GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN |
81  GCLK_GENCTRL_SRC_OSC8M |
82  GCLK_GENCTRL_ID(0));
83 #endif
84 
85  /* make sure we synchronize clock generator 0 before we go on */
86  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
87 
88  /* Setup Clock generator 2 with divider 1 (32.768kHz) */
89  GCLK->GENDIV.reg = (GCLK_GENDIV_ID(2) | GCLK_GENDIV_DIV(0));
90  GCLK->GENCTRL.reg = (GCLK_GENCTRL_ID(2) | GCLK_GENCTRL_GENEN |
91  GCLK_GENCTRL_RUNSTDBY |
92  GCLK_GENCTRL_SRC_OSCULP32K);
93 
94  while (GCLK->STATUS.bit.SYNCBUSY) {}
95 
96  /* redirect all peripherals to a disabled clock generator (7) by default */
97  for (int i = 0x3; i <= 0x22; i++) {
98  GCLK->CLKCTRL.reg = ( GCLK_CLKCTRL_ID(i) | GCLK_CLKCTRL_GEN_GCLK7 );
99  while (GCLK->STATUS.bit.SYNCBUSY) {}
100  }
101 }
