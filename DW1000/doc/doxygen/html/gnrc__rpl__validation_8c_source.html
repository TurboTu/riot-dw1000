31  gnrc_rpl_opt_t *opt, uint16_t len)
32 {
33  uint16_t expected_len = 0;
34 
35  while(expected_len < len) {
36  switch(opt->type) {
37  case (GNRC_RPL_OPT_PAD1):
38  expected_len += 1;
39  opt = (gnrc_rpl_opt_t *) (((uint8_t *) opt) + 1);
40  continue;
41 
42  case (GNRC_RPL_OPT_DODAG_CONF):
43  if (msg_type != GNRC_RPL_ICMPV6_CODE_DIO) {
44  DEBUG("RPL: DODAG CONF DIO option not expected\n");
45  return false;
46  }
47

50  opt->length, GNRC_RPL_OPT_DODAG_CONF_LEN);
51  return false;
52  }
53  break;
54 
55  case (GNRC_RPL_OPT_PREFIX_INFO):
56  if (msg_type != GNRC_RPL_ICMPV6_CODE_DIO) {

58  return false;
59  }
60

62  DEBUG("RPL: wrong DIO option (PREFIX INFO) len: %d, expected: %d\n",
63  opt->length, GNRC_RPL_OPT_PREFIX_INFO_LEN);
64  return false;
65  }
66  break;
67 
68  case (GNRC_RPL_OPT_TARGET):
69  if (msg_type != GNRC_RPL_ICMPV6_CODE_DAO) {
70  DEBUG("RPL: RPL TARGET DAO option not expected\n");
71  return false;
72  }
73

79  break;
80 
81  case (GNRC_RPL_OPT_TRANSIT):

83  DEBUG("RPL: RPL TRANSIT INFO DAO option not expected\n");
84  return false;

86

94  opt->length, (GNRC_RPL_OPT_TRANSIT_INFO_LEN + parent_addr));
95  return false;
96  }
97  break;
98 
99  default:
100  break;
101 
102  }
103  expected_len += opt->length + sizeof(gnrc_rpl_opt_t);
104  opt = (gnrc_rpl_opt_t *) (((uint8_t *) (opt + 1)) + opt->length);
105  }
106 
107  if (expected_len == len) {
108  return true;
109  }
110 
111  DEBUG("RPL: wrong options len: %d, expected: %d\n", len, expected_len);
112 
113  return false;
114 }
