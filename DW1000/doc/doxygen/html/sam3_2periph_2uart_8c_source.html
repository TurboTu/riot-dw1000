54  uart_config[uart].rx_port->PIO_PDR = (1 << uart_config[uart].rx_pin);
55  uart_config[uart].rx_port->PIO_ABSR &= ~(1 << uart_config[uart].rx_pin);
56  uart_config[uart].tx_port->PIO_ABSR |= (uart_config[uart].mux <<

58  uart_config[uart].tx_port->PIO_PDR = (1 << uart_config[uart].tx_pin);

60  uart_config[uart].tx_port->PIO_ABSR |= (uart_config[uart].mux <<
61  uart_config[uart].tx_pin);

65  dev->UART_MR = UART_MR_PAR_NO | US_MR_CHRL_8_BIT;
