71 {
72  cc110x_t *cc110x = &((netdev2_cc110x_t*) netdev)->cc110x;

89 {

91

94  assert(value_len == 2);

104  assert(value_len > 1);
105  *((uint16_t *)value) = (uint16_t)cc110x->radio_channel;
106  return 2;

113  *((uint8_t *)value) = CC110X_PACKET_LENGTH;
114  return 1;

124 static int _set(netdev2_t *dev, netopt_t opt, void *value, size_t value_len)
125 {
126  cc110x_t *cc110x = &((netdev2_cc110x_t*) dev)->cc110x;
127 
128  switch (opt) {
129  case NETOPT_CHANNEL:
130  {
131  uint8_t *arg = (uint8_t*)value;
132  uint8_t channel = arg[value_len-1];
133  if ((channel < CC110X_MIN_CHANNR) || (channel > CC110X_MAX_CHANNR)) {
134  return -EINVAL;
135  }
136  if (cc110x_set_channel(cc110x, channel) == -1) {
137  return -EINVAL;
138  }
139  return 1;
140  }
141  case NETOPT_ADDRESS:
142  if (value_len < 1) {
143  return -EINVAL;
144  }
145  if (!cc110x_set_address(cc110x, *(uint8_t*)value)) {
146  return -EINVAL;
147  }
148  return 1;
149 #ifdef MODULE_GNRC_NETIF
150  case NETOPT_PROTO:
151  if (value_len != sizeof(gnrc_nettype_t)) {
152  return -EINVAL;
153  }
154  else {
155  cc110x->proto = (gnrc_nettype_t) value;
156  return sizeof(gnrc_nettype_t);
157  }
158  break;
159 #endif
160  default:
161  return -ENOTSUP;
162  }
163 
164  return 0;
165 }
