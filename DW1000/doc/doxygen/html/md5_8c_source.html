139 static void permute(uint32_t abcd[4], const uint8_t block[64] )
140 {
141  uint8_t s;
142  uint32_t a, b, c, d;
143  uint32_t keep_abcd[4];
144  uint32_t x[16];
145 
146  /* Store the current ABCD values for later re-use */
147  for (int i = 0; i < 4; i++) {
148  keep_abcd[i] = abcd[i];
149  }
150 
151  /* Convert the input block into an array of unsigned longs, taking care
152  * to read the block in Little Endian order (the algorithm assumes this).
153  * The uint32_t values are then handled in host order. */
154  for (int i = 0, j = 0; i < 16; i++) {
155  x[i] = (uint32_t)block[j++];
156  x[i] |= ((uint32_t)block[j++] << 8);
157  x[i] |= ((uint32_t)block[j++] << 16);
158  x[i] |= ((uint32_t)block[j++] << 24);
159  }
160 
161  /* This loop performs the four rounds of permutations.
162  * The rounds are each very similar. The differences are in three areas:
163  * - The function (F, G, H, or I) used to perform bitwise permutations
164  * on the registers,
165  * - The order in which values from X[] are chosen.
166  * - Changes to the number of bits by which the registers are rotated.
167  * This implementation uses a switch statement to deal with some of the
168  * differences between rounds. Other differences are handled by storing
169  * values in arrays and using the round number to select the correct set
170  * of values.
171  *
172  * (My implementation appears to be a poor compromise between speed, size,
173  * and clarity. Ugh. [crh]) */
174  for (int round = 0; round < 4; round++) {
175  for (int i = 0; i < 16; i++) {
176  /* <j> handles the rotation of ABCD */
177  int j = (4 - (i % 4)) & 0x3;
178  /* <s> is the bit shift for this iteration */
179  s = S[round][i % 4];
180 
181  /* Copy the b,c,d values per ABCD rotation. This isn't really
182  * necessary, it just looks clean & will hopefully be optimized
183  * away. */
184  b = abcd[(j + 1) & 0x3];
185  c = abcd[(j + 2) & 0x3];
186  d = abcd[(j + 3) & 0x3];
187 
188  /* The actual perumation function.
189  * This is broken out to minimize the code within the switch(). */
190  switch (round) {
191  case 0: /* round 1 */
192  a = md5F( b, c, d ) + x[i];
193  break;
194  case 1: /* round 2 */
195  a = md5G( b, c, d ) + x[ K[0][i] ];
196  break;
197  case 2: /* round 3 */
198  a = md5H( b, c, d ) + x[ K[1][i] ];
199  break;
200  default: /* round 4 */
201  a = md5I( b, c, d ) + x[ K[2][i] ];
202  break;
203  }
204  a = 0xFFFFFFFF & (abcd[j] + a + T[round][i]);
205  abcd[j] = b + (0xFFFFFFFF & ((a << s) | (a >> (32 - s))));
206  }
207  }
208 
209  /* Use the stored original A, B, C, D values to perform
210  * one last convolution. */
211  for (int i = 0; i < 4; i++) {
212  abcd[i] = (abcd[i] + keep_abcd[i]);
213  }
214 }

221  /* The array ABCD[] contains the four 4-byte "registers" that are

224  * (pg. 4). Note, however, that RFC 1321 provides these values as bytes, not

235 void md5_update(md5_ctx_t *ctx, const void *data, size_t len)
236 {
237  /* Add the new block's length to the total length. */
238  ctx->len += (uint32_t)len;
239 
240  /* Copy the new block's data into the context block.
241  * Call the permute() function whenever the context block is full. */
242  for (size_t i = 0; i < len; i++) {
243  const uint8_t *d = data;
244 
245  ctx->block[ctx->b_used] = d[i];
246  (ctx->b_used)++;
247  if (64 == ctx->b_used) {
248  permute(ctx->abcd, ctx->block);
249  ctx->b_used = 0;
250  }
251  }
252 }

255 void md5_final(md5_ctx_t *ctx, void *digest)
256 {
257  uint32_t l;
258 
259  /* Add the required 0x80 padding initiator byte.
260  * The md5_update() function always permutes and resets the context
261  * block when it gets full, so we know that there must be at least one
262  * free byte in the context block.
263  */
264  ctx->block[ctx->b_used] = 0x80;
265  (ctx->b_used)++;
266 
267  /* Zero out any remaining free bytes in the context block. */
268  for (int i = ctx->b_used; i < 64; i++) {
269  ctx->block[i] = 0;
270  }
271 
272  /* We need 8 bytes to store the length field.
273  * If we don't have 8, call permute() and reset the context block. */
274  if (56 < ctx->b_used) {
275  permute(ctx->abcd, ctx->block);
276  for (int i = 0; i < 64; i++) {
277  ctx->block[i] = 0;
278  }
279  }
280 
281  /* Add the total length and perform the final perumation.
282  * Note: The 60'th byte is read from the *original* <ctx->len> value
283  * and shifted to the correct position. This neatly avoids
284  * any MAXINT numeric overflow issues. */
285  l = ctx->len << 3;
286  for (int i = 0; i < 4; i++) {
287  ctx->block[56 + i] |= GETBYTE(l, i);
288  }
289  ctx->block[60] = ((GETBYTE(ctx->len, 3) & 0xE0) >> 5); /* See Above! */
290  permute(ctx->abcd, ctx->block);
291 
292  /* Now copy the result into the output buffer and we're done */
293  for (int i = 0; i < 4; i++) {
294  uint8_t *d = digest;
295 
296  d[ 0 + i] = GETBYTE(ctx->abcd[0], i);
297  d[ 4 + i] = GETBYTE(ctx->abcd[1], i);
298  d[ 8 + i] = GETBYTE(ctx->abcd[2], i);
299  d[12 + i] = GETBYTE(ctx->abcd[3], i);
300  }
301 }
