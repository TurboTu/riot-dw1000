47 int uart_init(uart_t uart, uint32_t baudrate, uart_rx_cb_t rx_cb, void *arg)
48 {
49  /* initialize basic functionality */
50  int res = init_base(uart, baudrate);
51  if (res != UART_OK) {
52  return res;
53  }
54 
55  /* register callbacks */
56  uart_ctx[uart].rx_cb = rx_cb;
57  uart_ctx[uart].arg = arg;
58  /* configure interrupts and enable RX interrupt */
59  _uart(uart)->INTENSET.reg = SERCOM_USART_INTENSET_RXC;
60  NVIC_EnableIRQ(SERCOM0_IRQn + sercom_id(_uart(uart)));
61  return UART_OK;
62 }

64 static int init_base(uart_t uart, uint32_t baudrate)
65 {
66  uint32_t baud;
67  SercomUsart *dev;
68 
69  if ((unsigned int)uart >= UART_NUMOF) {
70  return UART_NODEV;
71  }
72 
73  /* get the devices base register */
74  dev = _uart(uart);
75  /* calculate baudrate */
76  baud = ((((uint32_t)CLOCK_CORECLOCK * 10) / baudrate) / 16);
77  /* enable sync and async clocks */
78  uart_poweron(uart);
79  /* configure pins */
80  gpio_init(uart_config[uart].rx_pin, GPIO_IN);
81  gpio_init_mux(uart_config[uart].rx_pin, uart_config[uart].mux);
82  gpio_init(uart_config[uart].tx_pin, GPIO_OUT);
83  gpio_init_mux(uart_config[uart].tx_pin, uart_config[uart].mux);
84  /* reset the UART device */
85  dev->CTRLA.reg = SERCOM_USART_CTRLA_SWRST;
86  while (dev->SYNCBUSY.reg & SERCOM_USART_SYNCBUSY_SWRST) {}
87  /* set asynchronous mode w/o parity, LSB first, TX and RX pad as specified
88  * by the board in the periph_conf.h, x16 sampling and use internal clock */
89  dev->CTRLA.reg = (SERCOM_USART_CTRLA_DORD |
90  SERCOM_USART_CTRLA_SAMPR(0x1) |
91  SERCOM_USART_CTRLA_TXPO(uart_config[uart].tx_pad) |
92  SERCOM_USART_CTRLA_RXPO(uart_config[uart].rx_pad) |
93  SERCOM_USART_CTRLA_MODE_USART_INT_CLK);
94  /* set baudrate */
95  dev->BAUD.FRAC.FP = (baud % 10);
96  dev->BAUD.FRAC.BAUD = (baud / 10);
97  /* enable receiver and transmitter, use 1 stop bit */
98  dev->CTRLB.reg = (SERCOM_USART_CTRLB_RXEN | SERCOM_USART_CTRLB_TXEN);
99  while (dev->SYNCBUSY.reg & SERCOM_USART_SYNCBUSY_CTRLB) {}
100  /* finally, enable the device */
101  dev->CTRLA.reg |= SERCOM_USART_CTRLA_ENABLE;
102  return UART_OK;
103 }

113 void uart_poweron(uart_t uart)
114 {
115  PM->APBCMASK.reg |= (PM_APBCMASK_SERCOM0 << sercom_id(_uart(uart)));
116  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN |
117  GCLK_CLKCTRL_GEN_GCLK0 |
118  (SERCOM0_GCLK_ID_CORE + sercom_id(_uart(uart))) <<
119  GCLK_CLKCTRL_ID_Pos);
120  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
121 }

131 static inline void irq_handler(int dev)
132 {
133  SercomUsart *uart = _uart(dev);
134 
135  if (uart->INTFLAG.reg & SERCOM_USART_INTFLAG_RXC) {
136  /* interrupt flag is cleared by reading the data register */
137  uart_ctx[dev].rx_cb(uart_ctx[dev].arg, (uint8_t)(uart->DATA.reg));
138  }
139  else if (uart->INTFLAG.reg & SERCOM_USART_INTFLAG_ERROR) {
140  /* clear error flag */
141  uart->INTFLAG.reg = SERCOM_USART_INTFLAG_ERROR;
142  }
143  cortexm_isr_end();
144 }
