64 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
65 {
66  uint32_t pclksel;
67  uint32_t cpsr;
68 
69  /* only support for mode 0 at the moment */
70  if (mode != SPI_MODE_0) {
71  return SPI_NOMODE;
72  }
73 
74  /* lock bus */
75  mutex_lock(&lock);
76  /* power on */
77  PCONP |= (PCSSP0);
78 
79  /* configure bus clock */
80  lpc2387_pclk_scale(CLOCK_CORECLOCK / 1000, (uint32_t)clk, &pclksel, &cpsr);
81  PCLKSEL1 &= ~(BIT10 | BIT11); /* CCLK to PCLK divider*/
82  PCLKSEL1 |= pclksel << 10;
83  SSP0CPSR = cpsr;
84 
85  /* enable the bus */
86  SSP0CR1 |= BIT1;
87 
88  /* clear RxFIFO */
89  int dummy;
90  while (SPI_RX_AVAIL) { /* while RNE (Receive FIFO Not Empty)...*/
91  dummy = SSP0DR; /* read data*/
92  }
93  (void) dummy; /* to suppress unused-but-set-variable */
94 
95  return SPI_OK;
96 }

106 void spi_transfer_bytes(spi_t bus, spi_cs_t cs, bool cont,
107  const void *out, void *in, size_t len)
108 {
109  uint8_t *out_buf = (uint8_t *)out;
110  uint8_t *in_buf = (uint8_t *)in;
111 
112  assert(out_buf || in_buf);
113 
114  if (cs != SPI_CS_UNDEF) {
115  gpio_clear((gpio_t)cs);
116  }
117 
118  for (size_t i = 0; i < len; i++) {
119  uint8_t tmp = (out_buf) ? out_buf[i] : 0;
120  while (!SPI_TX_EMPTY) {}
121  SSP0DR = tmp;
122  while (SPI_BUSY) {}
123  while (!SPI_RX_AVAIL) {}
124  tmp = (uint8_t)SSP0DR;
125  if (in_buf) {
126  in_buf[i] = tmp;
127  }
128  }
129 
130  if ((!cont) && cs != SPI_CS_UNDEF) {
131  gpio_set((gpio_t)cs);
132  }
133 }
