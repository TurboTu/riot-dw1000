60 static ipv6_addr_t *_add_addr_to_entry(gnrc_ipv6_netif_t *entry, const ipv6_addr_t *addr,
61  uint8_t prefix_len, uint8_t flags)
62 {
63  gnrc_ipv6_netif_addr_t *tmp_addr = NULL;
64 
65  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
66  if (ipv6_addr_equal(&(entry->addrs[i].addr), addr)) {
67  return &(entry->addrs[i].addr);
68  }
69 
70  if (ipv6_addr_is_unspecified(&(entry->addrs[i].addr)) && !tmp_addr) {
71  tmp_addr = &(entry->addrs[i]);
72  }
73  }
74 
75  if (!tmp_addr) {
76  DEBUG("ipv6 netif: couldn't add %s/%" PRIu8 " to interface %" PRIkernel_pid "\n: No space left.\n",
77  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)),
78  prefix_len, entry->pid);
79  return NULL;
80  }
81 
82  memcpy(&(tmp_addr->addr), addr, sizeof(ipv6_addr_t));
83  DEBUG("ipv6 netif: Added %s/%" PRIu8 " to interface %" PRIkernel_pid "\n",
84  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)),
85  prefix_len, entry->pid);
86 
87  tmp_addr->prefix_len = prefix_len;
88  tmp_addr->flags = flags;
89 
90 #ifdef MODULE_GNRC_SIXLOWPAN_ND
91  if (!ipv6_addr_is_multicast(&(tmp_addr->addr)) &&
92  (entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN)) {
93  ipv6_addr_t *router = gnrc_ndp_internal_default_router();
94  if (router != NULL) {
95  mutex_unlock(&entry->mutex); /* function below relocks mutex */
96  gnrc_ndp_internal_send_nbr_sol(entry->pid, &tmp_addr->addr, router, router);
97  mutex_lock(&entry->mutex); /* relock mutex */
98  }
99  /* otherwise there is no default router to register to */
100  }
101 #endif
102 
103  if (ipv6_addr_is_multicast(addr)) {
104  tmp_addr->flags |= GNRC_IPV6_NETIF_ADDR_FLAGS_NON_UNICAST;
105  }
106  else {
107  if (!ipv6_addr_is_link_local(addr)) {
108 #ifdef MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER
109  tmp_addr->valid = UINT32_MAX;
110  tmp_addr->preferred = UINT32_MAX;
111  gnrc_sixlowpan_nd_router_abr_t *abr = gnrc_sixlowpan_nd_router_abr_get();
112  mutex_unlock(&entry->mutex);
113  gnrc_ipv6_netif_set_rtr_adv(entry, true);
114  mutex_lock(&entry->mutex);
115  if (gnrc_sixlowpan_nd_router_abr_add_prf(abr, entry, tmp_addr) < 0) {
116  DEBUG("ipv6_netif: error adding prefix to 6LoWPAN-ND management\n");
117  }
118 #endif
119 #if defined(MODULE_GNRC_NDP_ROUTER) || defined(MODULE_GNRC_SIXLOWPAN_ND_ROUTER)
120  if ((entry->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER) &&
121  (entry->flags & GNRC_IPV6_NETIF_FLAGS_RTR_ADV)) {
122  mutex_unlock(&entry->mutex); /* function below relocks mutex */
123 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
124  if (entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) {
125  gnrc_ndp_internal_send_rtr_adv(entry->pid, NULL, NULL, false);
126  }
127 #endif
128 #ifdef MODULE_GNRC_NDP_ROUTER
129  /* New prefixes MAY allow the router to retransmit up to
130  * GNRC_NDP_MAX_INIT_RTR_ADV_NUMOF unsolicited RA
131  * (see https://tools.ietf.org/html/rfc4861#section-6.2.4) */
132  if (!(entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN)) {
133  entry->rtr_adv_count = GNRC_NDP_MAX_INIT_RTR_ADV_NUMOF;
134  gnrc_ndp_router_retrans_rtr_adv(entry);
135  }
136 #endif
137  mutex_lock(&entry->mutex); /* relock mutex */
138  }
139 #endif
140  }
141  else {
142  tmp_addr->flags |= GNRC_IPV6_NETIF_ADDR_FLAGS_NDP_ON_LINK;
143  }
144 #if defined(MODULE_GNRC_NDP_NODE) || defined(MODULE_GNRC_SIXLOWPAN_ND_ROUTER)
145  /* add solicited-nodes multicast address for new address if interface is not a
146  * 6LoWPAN host interface (see: https://tools.ietf.org/html/rfc6775#section-5.2) */
147  if (!(entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) ||
148  (entry->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER)) {
149  ipv6_addr_t sol_node;
150  ipv6_addr_set_solicited_nodes(&sol_node, addr);
151  _add_addr_to_entry(entry, &sol_node, IPV6_ADDR_BIT_LEN, 0);
152  }
153 #endif
154  /* TODO: send NS with ARO on 6LoWPAN interfaces, but not so many and only for the new
155  * source address. */
156  }
157 
158  tmp_addr->valid_timeout_msg.type = GNRC_NDP_MSG_ADDR_TIMEOUT;
159  tmp_addr->valid_timeout_msg.content.ptr = &tmp_addr->addr;
160 
161  return &(tmp_addr->addr);
162 }

201 void gnrc_ipv6_netif_add(kernel_pid_t pid)
202 {
203  gnrc_ipv6_netif_t *free_entry = NULL;
204 
205  for (int i = 0; i < GNRC_NETIF_NUMOF; i++) {
206  if (ipv6_ifs[i].pid == pid) {
207  /* pid has already been added */
208  return;
209  }
210 
211  else if ((ipv6_ifs[i].pid == KERNEL_PID_UNDEF) && !free_entry) {
212  /* found the first free entry */
213  free_entry = &ipv6_ifs[i];
214  }
215  }
216 
217  if (!free_entry) {
218  DEBUG("ipv6 netif: Could not add %" PRIkernel_pid " to IPv6: No space left.\n", pid);
219  return;
220  }
221 
222  /* Otherwise, fill the free entry */
223 
224  mutex_lock(&free_entry->mutex);
225 
226  DEBUG("ipv6 netif: Add IPv6 interface %" PRIkernel_pid " (i = %d)\n", pid,
227  free_entry - ipv6_ifs);
228  free_entry->pid = pid;
229  free_entry->mtu = GNRC_IPV6_NETIF_DEFAULT_MTU;
230  free_entry->cur_hl = GNRC_IPV6_NETIF_DEFAULT_HL;
231  free_entry->flags = 0;
232 
233  _add_addr_to_entry(free_entry, &ipv6_addr_all_nodes_link_local,
234  IPV6_ADDR_BIT_LEN, 0);
235 
236  mutex_unlock(&free_entry->mutex);
237 
238 #ifdef MODULE_GNRC_NDP
239  gnrc_ndp_netif_add(free_entry);
240 #endif
241 
242  DEBUG(" * pid = %" PRIkernel_pid " ", free_entry->pid);
243  DEBUG("cur_hl = %d ", free_entry->cur_hl);
244  DEBUG("mtu = %d ", free_entry->mtu);
245  DEBUG("flags = %04" PRIx16 "\n", free_entry->flags);

315 static void _remove_addr_from_entry(gnrc_ipv6_netif_t *entry, ipv6_addr_t *addr)
316 {
317  mutex_lock(&entry->mutex);
318 
319  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
320  if (ipv6_addr_equal(&(entry->addrs[i].addr), addr)) {
321  DEBUG("ipv6 netif: Remove %s to interface %" PRIkernel_pid "\n",
322  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)), entry->pid);
323  ipv6_addr_set_unspecified(&(entry->addrs[i].addr));
324  entry->addrs[i].flags = 0;
325 #ifdef MODULE_GNRC_NDP_ROUTER
326  /* Removal of prefixes MAY allow the router to retransmit up to
327  * GNRC_NDP_MAX_INIT_RTR_ADV_NUMOF unsolicited RA
328  * (see https://tools.ietf.org/html/rfc4861#section-6.2.4) */
329  if ((entry->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER) &&
330  (entry->flags & GNRC_IPV6_NETIF_FLAGS_RTR_ADV) &&
331  (!ipv6_addr_is_multicast(addr) &&
332  !ipv6_addr_is_link_local(addr))) {
333  entry->rtr_adv_count = GNRC_NDP_MAX_INIT_RTR_ADV_NUMOF;
334  mutex_unlock(&entry->mutex); /* function below relocks the mutex */
335  gnrc_ndp_router_retrans_rtr_adv(entry);
336  return;
337  }
338 #endif
339 #ifdef MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER
340  gnrc_sixlowpan_nd_router_abr_t *abr = gnrc_sixlowpan_nd_router_abr_get();
341  gnrc_sixlowpan_nd_router_abr_rem_prf(abr, entry, &entry->addrs[i]);
342 #endif
343 
344  mutex_unlock(&entry->mutex);
345  return;
346  }
347  }
348 
349  mutex_unlock(&entry->mutex);
350 }

439 static uint8_t _find_by_prefix_unsafe(ipv6_addr_t **res, gnrc_ipv6_netif_t *iface,
440  const ipv6_addr_t *addr, uint8_t *only)
441 {
442  uint8_t best_match = 0;
443 
444  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
445  uint8_t match;
446 
447  if ((only != NULL) && !(bf_isset(only, i))) {
448  continue;
449  }
450 
451  if (((only != NULL) &&
452  gnrc_ipv6_netif_addr_is_non_unicast(&(iface->addrs[i].addr))) ||
453  ipv6_addr_is_unspecified(&(iface->addrs[i].addr))) {
454  continue;
455  }
456 
457  match = ipv6_addr_match_prefix(&(iface->addrs[i].addr), addr);
458 
459  if ((only == NULL) && !ipv6_addr_is_multicast(addr) &&
460  (match < iface->addrs[i].prefix_len)) {
461  /* match but not of same subnet */
462  continue;
463  }
464 
465  if (match > best_match) {
466  if (res != NULL) {
467  *res = &(iface->addrs[i].addr);
468  }
469 
470  best_match = match;
471  }
472  }
473 
474 #if ENABLE_DEBUG
475  if (*res != NULL) {
476  DEBUG("ipv6 netif: Found %s on interface %" PRIkernel_pid " matching ",
477  ipv6_addr_to_str(addr_str, *res, sizeof(addr_str)),
478  iface->pid);
479  DEBUG("%s by %" PRIu8 " bits (used as source address = %s)\n",
480  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)),
481  best_match,
482  (only != NULL) ? "true" : "false");
483  }
484  else {
485  DEBUG("ipv6 netif: Did not found any address on interface %" PRIkernel_pid
486  " matching %s (used as source address = %s)\n",
487  iface->pid,
488  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)),
489  (only != NULL) ? "true" : "false");
490  }
491 #endif
492 
493  return best_match;
494 }

496 kernel_pid_t gnrc_ipv6_netif_find_by_prefix(ipv6_addr_t **out, const ipv6_addr_t *prefix)
497 {
498  uint8_t best_match = 0;
499  ipv6_addr_t *tmp_res = NULL;
500  kernel_pid_t res = KERNEL_PID_UNDEF;
501 
502  for (int i = 0; i < GNRC_NETIF_NUMOF; i++) {
503  uint8_t match;
504 
505  mutex_lock(&(ipv6_ifs[i].mutex));
506 
507  match = _find_by_prefix_unsafe(&tmp_res, ipv6_ifs + i, prefix, NULL);
508 
509  if (match > best_match) {
510  *out = tmp_res;
511  res = ipv6_ifs[i].pid;
512  best_match = match;
513  }
514 
515  mutex_unlock(&(ipv6_ifs[i].mutex));
516  }
517 
518 #if ENABLE_DEBUG
519  if (res != KERNEL_PID_UNDEF) {
520  DEBUG("ipv6 netif: Found %s on interface %" PRIkernel_pid " globally matching ",
521  ipv6_addr_to_str(addr_str, *out, sizeof(addr_str)),
522  res);
523  DEBUG("%s by %" PRIu8 " bits\n",
524  ipv6_addr_to_str(addr_str, prefix, sizeof(addr_str)),
525  best_match);
526  }
527  else {
528  DEBUG("ipv6 netif: Did not found any address globally matching %s\n",
529  ipv6_addr_to_str(addr_str, prefix, sizeof(addr_str)));
530  }
531 #endif
532 
533  return res;
534 }

553 static int _create_candidate_set(gnrc_ipv6_netif_t *iface, const ipv6_addr_t *dst,
554  uint8_t *candidate_set, bool link_local_only)
555 {
556  int res = -1;
557 
558  DEBUG("gathering candidates\n");
559 
560  /* currently this implementation supports only addresses as source address
561  * candidates assigned to this interface. Thus we assume all addresses to be
562  * on interface @p iface */
563  (void) dst;
564 
565  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
566  gnrc_ipv6_netif_addr_t *iter = &(iface->addrs[i]);
567 
568  DEBUG("Checking address: %s\n",
569  ipv6_addr_to_str(addr_str, &(iter->addr), sizeof(addr_str)));
570 
571  /* "In any case, multicast addresses and the unspecified address MUST NOT
572  * be included in a candidate set."
573  */
574  if (ipv6_addr_is_multicast(&(iter->addr)) ||
575  ipv6_addr_is_unspecified(&(iter->addr))) {
576  continue;
577  }
578 
579  /* Check if we only want link local addresses */
580  if (link_local_only && !ipv6_addr_is_link_local(&(iter->addr))) {
581  continue;
582  }
583 
584  /* "For all multicast and link-local destination addresses, the set of
585  * candidate source addresses MUST only include addresses assigned to
586  * interfaces belonging to the same link as the outgoing interface."
587  *
588  * "For site-local unicast destination addresses, the set of candidate
589  * source addresses MUST only include addresses assigned to interfaces
590  * belonging to the same site as the outgoing interface."
591  * -> we should also be fine, since we're only iterating addresses of
592  * the sending interface

666 static ipv6_addr_t *_source_address_selection(gnrc_ipv6_netif_t *iface, const ipv6_addr_t *dst,
667  uint8_t *candidate_set)
668 {
669  /* create temporary set for assigning "points" to candidates wining in the
670  * corresponding rules.
671  */
672  uint8_t winner_set[GNRC_IPV6_NETIF_ADDR_NUMOF];
673  memset(winner_set, 0, GNRC_IPV6_NETIF_ADDR_NUMOF);
674 
675  uint8_t max_pts = 0;
676 
677  /* _create_candidate_set() assures that `dest` is not unspecified and if
678  * `dst` is loopback rule 1 will fire anyway. */
679  uint8_t dst_scope = _get_scope(dst, true);
680  DEBUG("finding the best match within the source address candidates\n");
681 
682  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
683  gnrc_ipv6_netif_addr_t *iter = &(iface->addrs[i]);
684  DEBUG("Checking address: %s\n",
685  ipv6_addr_to_str(addr_str, &(iter->addr), sizeof(addr_str)));
686  /* entries which are not part of the candidate set can be ignored */
687  if (!(bf_isset(candidate_set, i))) {
688  DEBUG("Not part of the candidate set - skipping\n");
689  continue;
690  }
691 
692  /* Rule 1: if we have an address configured that equals the destination
693  * use this one as source */
694  if (ipv6_addr_equal(&(iter->addr), dst)) {
695  DEBUG("Ease one - rule 1\n");
696  return &(iter->addr);
697  }
698 
699  /* Rule 2: Prefer appropriate scope. */
700  /* both link local */
701  uint8_t candidate_scope = _get_scope(&(iter->addr), false);
702  if (candidate_scope == dst_scope) {
703  DEBUG("winner for rule 2 (same scope) found\n");
704  winner_set[i] += RULE_2A_PTS;
705  if (winner_set[i] > max_pts) {
706  max_pts = RULE_2A_PTS;
707  }
708  }
709  else if (candidate_scope < dst_scope) {
710  DEBUG("winner for rule 2 (smaller scope) found\n");
711  winner_set[i] += RULE_2B_PTS;
712  if (winner_set[i] > max_pts) {
713  max_pts = winner_set[i];
714  }
715  }
716 
717  /* Rule 3: Avoid deprecated addresses. */
718  if (iter->preferred > 0) {
719  DEBUG("winner for rule 3 found\n");
720  winner_set[i] += RULE_3_PTS;
721  if (winner_set[i] > max_pts) {
722  max_pts = winner_set[i];
723  }
724  }
725 
726  /* Rule 4: Prefer home addresses.
727  * Does not apply, gnrc does not support Mobile IP.
728  * TODO: update as soon as gnrc supports Mobile IP
729  */
730 
731  /* Rule 5: Prefer outgoing interface.
732  * RFC 6724 says:
733  * "It is RECOMMENDED that the candidate source addresses be the set of
734  * unicast addresses assigned to the interface that will be used to
735  * send to the destination (the "outgoing" interface). On routers,
736  * the candidate set MAY include unicast addresses assigned to any
737  * interface that forwards packets, subject to the restrictions
738  * described below."
739  * Currently this implementation uses ALWAYS source addresses assigned
740  * to the outgoing interface. Hence, Rule 5 is always fulfilled.
741  */
742 
743  /* Rule 6: Prefer matching label.
744  * Flow labels are currently not supported by gnrc.
745  * TODO: update as soon as gnrc supports flow labels
746  */
747 
748  /* Rule 7: Prefer temporary addresses.
749  * Temporary addresses are currently not supported by gnrc.
750  * TODO: update as soon as gnrc supports temporary addresses
751  */
752  }
753 
754  /* reset candidate set to mark winners */
755  memset(candidate_set, 0, (GNRC_IPV6_NETIF_ADDR_NUMOF + 7) / 8);
756  /* check if we have a clear winner */
757  /* collect candidates with maximum points */
758  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
759  if (winner_set[i] == max_pts) {
760  bf_set(candidate_set, i);
761  }
762  }
763 
764  /* otherwise apply rule 8: Use longest matching prefix. */
765  ipv6_addr_t *res = NULL;
766  _find_by_prefix_unsafe(&res, iface, dst, candidate_set);
767  return res;
768 }

770 ipv6_addr_t *gnrc_ipv6_netif_find_best_src_addr(kernel_pid_t pid, const ipv6_addr_t *dst, bool ll_only)
771 {
772  gnrc_ipv6_netif_t *iface = gnrc_ipv6_netif_get(pid);
773  ipv6_addr_t *best_src = NULL;
774  mutex_lock(&(iface->mutex));
775  BITFIELD(candidate_set, GNRC_IPV6_NETIF_ADDR_NUMOF);
776  memset(candidate_set, 0, sizeof(candidate_set));
777 
778  int first_candidate = _create_candidate_set(iface, dst, candidate_set, ll_only);
779  if (first_candidate >= 0) {
780  best_src = _source_address_selection(iface, dst, candidate_set);
781  if (best_src == NULL) {
782  best_src = &(iface->addrs[first_candidate].addr);
783  }
784  }
785  mutex_unlock(&(iface->mutex));
786 
787  return best_src;
788 }

790 void gnrc_ipv6_netif_init_by_dev(void)
791 {
792  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
793  size_t ifnum = gnrc_netif_get(ifs);
794 #ifdef MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER
795  bool abr_init = false;
796 #endif
797 
798  for (size_t i = 0; i < ifnum; i++) {
799  ipv6_addr_t addr;
800  eui64_t iid;
801  uint16_t tmp;
802  gnrc_ipv6_netif_t *ipv6_if = gnrc_ipv6_netif_get(ifs[i]);
803 
804  if (ipv6_if == NULL) {
805  continue;
806  }
807 
808  mutex_lock(&ipv6_if->mutex);
809 
810 #ifdef MODULE_GNRC_SIXLOWPAN
811  gnrc_nettype_t if_type = GNRC_NETTYPE_UNDEF;
812 
813  if ((gnrc_netapi_get(ifs[i], NETOPT_PROTO, 0, &if_type,
814  sizeof(if_type)) != -ENOTSUP) &&
815  (if_type == GNRC_NETTYPE_SIXLOWPAN)) {
816  uint16_t src_len = 8;
817  uint16_t max_frag_size = UINT16_MAX;
818 
819  DEBUG("ipv6 netif: Set 6LoWPAN flag\n");
820  ipv6_ifs[i].flags |= GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN;
821 
822  /* the router flag must be set early here, because otherwise
823  * _add_addr_to_entry() wouldn't set the solicited node address.
824  * However, addresses have to be configured before calling
825  * gnrc_ipv6_netif_set_router().
826  */
827 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
828  DEBUG("ipv6 netif: Set router flag\n");
829  ipv6_ifs[i].flags |= GNRC_IPV6_NETIF_FLAGS_ROUTER;
830 #endif
831  /* use EUI-64 (8-byte address) for IID generation and for sending
832  * packets */
833  gnrc_netapi_set(ifs[i], NETOPT_SRC_LEN, 0, &src_len,
834  sizeof(src_len)); /* don't care for result */
835 
836  if (gnrc_netapi_get(ifs[i], NETOPT_MAX_PACKET_SIZE,
837  0, &max_frag_size, sizeof(max_frag_size)) < 0) {
838  /* if error we assume it works */
839  DEBUG("ipv6 netif: Can not get max packet size from interface %"
840  PRIkernel_pid "\n", ifs[i]);
841  }
842 
843  gnrc_sixlowpan_netif_add(ifs[i], max_frag_size);
844  }
845 #endif
846 
847  /* set link-local address */
848  if (gnrc_netapi_get(ifs[i], NETOPT_IPV6_IID, 0, &iid,
849  sizeof(eui64_t)) >= 0) {
850  ipv6_addr_set_aiid(&addr, iid.uint8);
851  ipv6_addr_set_link_local_prefix(&addr);
852  _add_addr_to_entry(ipv6_if, &addr, 64, 0);
853 
854  }
855 
856  /* set link MTU */
857  if ((gnrc_netapi_get(ifs[i], NETOPT_MAX_PACKET_SIZE, 0, &tmp,
858  sizeof(uint16_t)) >= 0)) {
859  if (tmp >= IPV6_MIN_MTU) {
860  ipv6_if->mtu = tmp;
861  }
862  /* otherwise leave at GNRC_IPV6_NETIF_DEFAULT_MTU as initialized in
863  * gnrc_ipv6_netif_add() */
864  }
865 
866  if (gnrc_netapi_get(ifs[i], NETOPT_IS_WIRED, 0, NULL, 0) > 0) {
867  ipv6_if->flags |= GNRC_IPV6_NETIF_FLAGS_IS_WIRED;
868  }
869  else {
870  ipv6_if->flags &= ~GNRC_IPV6_NETIF_FLAGS_IS_WIRED;
871  }
872 
873  mutex_unlock(&ipv6_if->mutex);
874 #if (defined(MODULE_GNRC_NDP_ROUTER) || defined(MODULE_GNRC_SIXLOWPAN_ND_ROUTER))
875  gnrc_ipv6_netif_set_router(ipv6_if, true);
876 #endif
877 #ifdef MODULE_GNRC_SIXLOWPAN_ND
878  if (ipv6_if->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) {
879 #ifdef MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER
880  /* first interface wins */
881  if (!abr_init) {
882  gnrc_sixlowpan_nd_router_abr_create(&addr, 0);
883  /* XXX should be set to true if there ever is an hard-coded
884  * prefix set */
885  gnrc_ipv6_netif_set_rtr_adv(ipv6_if, false);
886  abr_init = true;
887  }
888 #endif
889  gnrc_sixlowpan_nd_init(ipv6_if);
890  continue; /* skip gnrc_ndp_host_init() */
891  }
892 #endif
893 #ifdef MODULE_GNRC_NDP_HOST
894  /* start periodic router solicitations */
895  gnrc_ndp_host_init(ipv6_if);
896 #endif
897  }
898 }
