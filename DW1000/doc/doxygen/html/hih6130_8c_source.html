80 static inline int hih6130_get_humidity_temperature_raw(hih6130_t *dev, uint16_t *humidity_raw, uint16_t *temperature_raw)
81 {
82  int status;
83  uint8_t buf[HIH6130_FULL_DATA_LENGTH];
84 
85  i2c_acquire(dev->i2c);
86 
87  if (i2c_read_bytes(dev->i2c, dev->addr, &buf[0], sizeof(buf)) != sizeof(buf)) {
88  i2c_release(dev->i2c);
89  return -1;
90  }
91 
92  i2c_release(dev->i2c);
93 
94  /* data is in big-endian format, with status bits in the first byte. */
95  switch (buf[0] & HIH6130_STATUS_MASK) {
96  case HIH6130_STATUS_OK:
97  status = 0;
98  break;
99  case HIH6130_STATUS_STALE_DATA:
100  status = 1;
101  break;
102  default:
103  return -2;
104  }
105 
106  *humidity_raw = ((buf[0] << 8) | buf[1]) & HIH6130_HUMIDITY_MASK;
107  *temperature_raw = (((buf[2] << 8) | buf[3]) >> HIH6130_TEMPERATURE_SHIFT);
108 
109  return status;
110 }
