33 void gnrc_sixlowpan_netif_add(kernel_pid_t pid, uint16_t max_frag_size)
34 {
35  gnrc_sixlowpan_netif_t *free_entry = NULL;
36 
37  for (int i = 0; i < GNRC_NETIF_NUMOF; i++) {
38  if (sixlow_ifs[i].pid == pid) {
39  return;
40  }
41 
42  if ((sixlow_ifs[i].pid == KERNEL_PID_UNDEF) && !free_entry) {
43  /* found the first free entry */
44  free_entry = &sixlow_ifs[i];
45  }
46  }
47 
48  if (!free_entry) {
49  DEBUG("gnrc_sixlowpan_netif_add: couldn't add interface with PID %d: No space left.\n", pid);
50  return;
51  }
52 
53  free_entry->pid = pid;
54  free_entry->max_frag_size = max_frag_size;
55 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC
56  free_entry->iphc_enabled = true;
57 #endif
58  return;
59 }
