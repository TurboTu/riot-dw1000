33 /* At 1Hz, RTC goes till 63 years (2^5, see 17.8.22 in datasheet)
34 * reference_year is set to 100 (offset) to be in our current time (2000)
35 * Thanks to this, the user will be able to set time in 2000's*/

38 void rtc_init(void)
39 {
40  RtcMode2 *rtcMode2 = &(RTC_DEV);
41 
42  /* Turn on power manager for RTC */
43  PM->APBAMASK.reg |= PM_APBAMASK_RTC;
44 
45  /* RTC uses External 32,768KHz Oscillator (OSC32K isn't accurate enough p1075/1138)*/
46  SYSCTRL->XOSC32K.reg = SYSCTRL_XOSC32K_ONDEMAND |
47  SYSCTRL_XOSC32K_EN32K |
48  SYSCTRL_XOSC32K_XTALEN |
49  SYSCTRL_XOSC32K_STARTUP(6) |
50  SYSCTRL_XOSC32K_ENABLE;
51 
52  /* Setup clock GCLK2 with OSC32K divided by 32 */
53  GCLK->GENDIV.reg = GCLK_GENDIV_ID(2)|GCLK_GENDIV_DIV(4);
54  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
55  GCLK->GENCTRL.reg = (GCLK_GENCTRL_GENEN | GCLK_GENCTRL_SRC_XOSC32K | GCLK_GENCTRL_ID(2) | GCLK_GENCTRL_DIVSEL );
56  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
57  GCLK->CLKCTRL.reg = (uint32_t)((GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK2 | (RTC_GCLK_ID << GCLK_CLKCTRL_ID_Pos)));
58  while (GCLK->STATUS.bit.SYNCBUSY) {}
59 
60  /* DISABLE RTC MASTER */
61  while (rtcMode2->STATUS.reg & RTC_STATUS_SYNCBUSY) {}
62  rtc_poweroff();
63 
64  /* Reset RTC */
65  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
66  rtcMode2->CTRL.reg= RTC_MODE2_CTRL_SWRST;
67  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
68 
69  /* RTC config with RTC_MODE2_CTRL_CLKREP = 0 (24h) */
70  rtcMode2->CTRL.reg = RTC_MODE2_CTRL_PRESCALER_DIV1024|RTC_MODE2_CTRL_MODE_CLOCK;
71  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
72  rtcMode2->INTENSET.reg = RTC_MODE2_INTENSET_OVF;
73  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
74  rtc_poweron();
75 }

77 int rtc_set_time(struct tm *time)
78 {
79  RtcMode2 *rtcMode2 = &(RTC_DEV);
80  if ((time->tm_year < reference_year) || (time->tm_year > reference_year + 63)) {
81  return -1;
82  }
83  else {
84  rtcMode2->CLOCK.reg = RTC_MODE2_CLOCK_YEAR(time->tm_year - reference_year)
85  | RTC_MODE2_CLOCK_MONTH(time->tm_mon + 1)
86  | RTC_MODE2_CLOCK_DAY(time->tm_mday)
87  | RTC_MODE2_CLOCK_HOUR(time->tm_hour)
88  | RTC_MODE2_CLOCK_MINUTE(time->tm_min)
89  | RTC_MODE2_CLOCK_SECOND(time->tm_sec);
90  }
91  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
92  return 0;
93 }

105  time->tm_min = rtcMode2->CLOCK.bit.MINUTE;

107  while (rtcMode2->STATUS.bit.SYNCBUSY) {}

111 int rtc_set_alarm(struct tm *time, rtc_alarm_cb_t cb, void *arg)
112 {
113  RtcMode2 *rtcMode2 = &(RTC_DEV);
114  rtc_clear_alarm();
115  if ((time->tm_year < reference_year) || (time->tm_year > (reference_year + 63))) {
116  return -2;
117  }
118  else {
119  rtcMode2->Mode2Alarm[0].ALARM.reg = RTC_MODE2_ALARM_YEAR(time->tm_year - reference_year)
120  | RTC_MODE2_ALARM_MONTH(time->tm_mon + 1)
121  | RTC_MODE2_ALARM_DAY(time->tm_mday)
122  | RTC_MODE2_ALARM_HOUR(time->tm_hour)
123  | RTC_MODE2_ALARM_MINUTE(time->tm_min)
124  | RTC_MODE2_ALARM_SECOND(time->tm_sec);
125  rtcMode2->Mode2Alarm[0].MASK.reg = RTC_MODE2_MASK_SEL(6);
126  }
127  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
128 
129  /* Setup interrupt */
130  NVIC_EnableIRQ(RTC_IRQn);
131 
132  /* Enable IRQ */
133  rtc_callback.cb = cb;
134  rtc_callback.arg = arg;
135  rtcMode2->INTFLAG.reg = RTC_MODE2_INTFLAG_ALARM0;
136  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
137  rtcMode2->INTENSET.reg = RTC_MODE2_INTENSET_ALARM0;
138  while (rtcMode2->STATUS.bit.SYNCBUSY) {}
139 
140  return 0;
141 }

182 void isr_rtc(void)
183 {
184  RtcMode2 *rtcMode2 = &(RTC_DEV);
185  uint16_t status = rtcMode2->INTFLAG.reg;
186  if ((status & RTC_MODE2_INTFLAG_ALARM0) && (rtc_callback.cb != NULL)) {
187  rtc_callback.cb(rtc_callback.arg);
188  rtcMode2->INTFLAG.reg = RTC_MODE2_INTFLAG_ALARM0;
189  }
190  if (status & RTC_MODE2_INTFLAG_OVF) {
191  /* At 1Hz, RTC goes till 63 years (2^5, see 17.8.22 in datasheet)
192  * Start RTC again with reference_year 64 years more (Be careful with alarm set) */
193  reference_year += 64;
194  rtcMode2->INTFLAG.reg = RTC_MODE2_INTFLAG_OVF;
195  }
196  cortexm_isr_end();
197 }
