124 int nhdp_register_if(kernel_pid_t if_pid, uint8_t *addr, size_t addr_size, uint8_t addr_type,
125  uint16_t max_pl_size, uint16_t hello_int_ms, uint16_t val_time_ms)
126 {
127  nhdp_if_entry_t *if_entry = NULL;
128  nhdp_addr_t *nhdp_addr;
129  msg_t signal_msg;
130 
131  if (nhdp_rcv_pid != KERNEL_PID_UNDEF) {
132  return -2;
133  }
134 
135  for (int i = 0; i < GNRC_NETIF_NUMOF; i++) {
136  if (nhdp_if_table[i].if_pid == KERNEL_PID_UNDEF) {
137  if_entry = &nhdp_if_table[i];
138  break;
139  }
140  }
141 
142  if (!if_entry) {
143  /* Maximum number of registerable interfaces reached */
144  return -2;
145  }
146 
147  uint16_t payload_size = max_pl_size > NHDP_MAX_RFC5444_PACKET_SZ
148  ? NHDP_MAX_RFC5444_PACKET_SZ : max_pl_size;
149  if_entry->wr_target.packet_buffer = (uint8_t *) calloc(payload_size, sizeof(uint8_t));
150 
151  if (!if_entry->wr_target.packet_buffer) {
152  /* Insufficient memory */
153  return -1;
154  }
155 
156  if_entry->wr_target.packet_size = payload_size;
157  if_entry->wr_target.sendPacket = write_packet;
158 
159  /* Get NHDP address entry for the given address */
160  nhdp_addr = nhdp_addr_db_get_address(addr, addr_size, addr_type);
161 
162  if (!nhdp_addr) {
163  /* Insufficient memory */
164  free(if_entry->wr_target.packet_buffer);
165  return -1;
166  }
167 
168  /* Add the interface to the LIB */
169  if (lib_add_if_addr(if_pid, nhdp_addr) != 0) {
170  free(if_entry->wr_target.packet_buffer);
171  nhdp_decrement_addr_usage(nhdp_addr);
172  return -1;
173  }
174 
175  /* Create new IIB for the interface */
176  if (iib_register_if(if_pid) != 0) {
177  /* TODO: Cleanup lib entry */
178  free(if_entry->wr_target.packet_buffer);
179  nhdp_decrement_addr_usage(nhdp_addr);
180  return -1;
181  }
182 
183  /* Set Interface's PID */
184  if_entry->if_pid = if_pid;
185  /* Set HELLO_INTERVAL and H_HOLD_TIME (validity time) */
186  if_entry->hello_interval.seconds = 0;
187  if_entry->hello_interval.microseconds = US_PER_MS * hello_int_ms;
188  if_entry->validity_time.seconds = 0;
189  if_entry->validity_time.microseconds = US_PER_MS * val_time_ms;
190  timex_normalize(&if_entry->hello_interval);
191  timex_normalize(&if_entry->validity_time);
192  /* Reset sequence number */
193  if_entry->seq_no = 0;
194 
195  /* Everything went well */
196  nhdp_decrement_addr_usage(nhdp_addr);
197  nhdp_writer_register_if(&if_entry->wr_target);
198  helper_pid = if_pid;
199 
200  /* Start the receiving thread */
201  nhdp_rcv_pid = thread_create(nhdp_rcv_stack, sizeof(nhdp_rcv_stack), THREAD_PRIORITY_MAIN - 1,
202  THREAD_CREATE_STACKTEST, _nhdp_receiver, NULL, "nhdp_rcv_thread");
203 
204  /* Start sending periodic HELLO */
205  signal_msg.type = HELLO_TIMER;
206  signal_msg.content.ptr = if_entry;
207  /* TODO: msg_send or msg_try_send? */
208  msg_try_send(&signal_msg, nhdp_pid);
209 
210  return 0;
211 }

244 static void *_nhdp_runner(void *arg)
245 {
246  nhdp_if_entry_t *if_entry;
247  msg_t msg_rcvd, msg_queue[NHDP_MSG_QUEUE_SIZE];
248 
249  (void)arg;
250  msg_init_queue(msg_queue, NHDP_MSG_QUEUE_SIZE);
251 
252  while (1) {
253  msg_receive(&msg_rcvd);
254 
255  switch (msg_rcvd.type) {
256  case HELLO_TIMER:
257  mutex_lock(&send_rcv_mutex);
258  if_entry = msg_rcvd.content.ptr;
259 
260  nhdp_writer_send_hello(if_entry);
261 
262  /* TODO: Add jitter */
263 
264  /* Schedule next sending */
265  xtimer_set_msg64(&if_entry->if_timer,
266  timex_uint64(if_entry->hello_interval),
267  &msg_rcvd, thread_getpid());
268  mutex_unlock(&send_rcv_mutex);
269  break;
270 
271 #if (NHDP_METRIC_NEEDS_TIMER)
272  case NHDP_METRIC_TIMER:
273  mutex_lock(&send_rcv_mutex);
274  /* Process necessary metric computations */
275  iib_process_metric_refresh();
276 
277  /* Schedule next sending */
278  metric_msg.type = NHDP_METRIC_TIMER;
279  metric_msg.content.ptr = NULL;
280  xtimer_set_msg64(&metric_timer, timex_uint64(metric_interval),
281  metric_msg, thread_getpid());
282  mutex_unlock(&send_rcv_mutex);
283  break;
284 #endif
285  default:
286  break;
287  }
288  }
289 
290  return 0;
291 }

296 static void *_nhdp_receiver(void *arg __attribute__((unused)))
297 {
298  char nhdp_rcv_buf[NHDP_MAX_RFC5444_PACKET_SZ];
299  msg_t msg_q[NHDP_MSG_QUEUE_SIZE];
300 
301  msg_init_queue(msg_q, NHDP_MSG_QUEUE_SIZE);
302 
303  /* Configure socket address for the manet port 269 */
304  ipv6_addr_t unspec = IPV6_ADDR_UNSPECIFIED;
305 
306  /* Bind UDP socket to socket address */
307  if (conn_udp_create(&conn, &unspec, sizeof(unspec), AF_INET6, MANET_PORT) == -1) {
308  /* Failed creating the connection */
309  return 0;
310  }
311 
312  while (1) {
313  ipv6_addr_t rcv_addr;
314  uint16_t rcv_port;
315  size_t addr_len = sizeof(rcv_addr);
316  int32_t rcv_size = conn_udp_recvfrom(&conn, (void *)nhdp_rcv_buf,
317  NHDP_MAX_RFC5444_PACKET_SZ, &rcv_addr,
318  &addr_len, &rcv_port);
319 
320  if (rcv_size > 0) {
321  /* Packet received, let the reader handle it */
322  mutex_lock(&send_rcv_mutex);
323  nhdp_reader_handle_packet(helper_pid, (void *)nhdp_rcv_buf, rcv_size);
324  mutex_unlock(&send_rcv_mutex);
325  }
326  }
327 
328  conn_udp_close(&conn);
329  return 0;
330 }
