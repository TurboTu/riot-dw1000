102 static void handle_input_line(const shell_command_t *command_list, char *line)
103 {
104  static const char *INCORRECT_QUOTING = "shell: incorrect quoting";
105 
106  /* first we need to calculate the number of arguments */
107  unsigned argc = 0;
108  char *pos = line;
109  int contains_esc_seq = 0;
110  while (1) {
111  if ((unsigned char) *pos > ' ') {
112  /* found an argument */
113  if (*pos == '"' || *pos == '\'') {
114  /* it's a quoted argument */
115  const char quote_char = *pos;
116  do {
117  ++pos;
118  if (!*pos) {
119  puts(INCORRECT_QUOTING);
120  return;
121  }
122  else if (*pos == '\\') {
123  /* skip over the next character */
124  ++contains_esc_seq;
125  ++pos;
126  if (!*pos) {
127  puts(INCORRECT_QUOTING);
128  return;
129  }
130  continue;
131  }
132  } while (*pos != quote_char);
133  if ((unsigned char) pos[1] > ' ') {
134  puts(INCORRECT_QUOTING);
135  return;
136  }
137  }
138  else {
139  /* it's an unquoted argument */
140  do {
141  if (*pos == '\\') {
142  /* skip over the next character */
143  ++contains_esc_seq;
144  ++pos;
145  if (!*pos) {
146  puts(INCORRECT_QUOTING);
147  return;
148  }
149  }
150  ++pos;
151  if (*pos == '"') {
152  puts(INCORRECT_QUOTING);
153  return;
154  }
155  } while ((unsigned char) *pos > ' ');
156  }
157 
158  /* count the number of arguments we got */
159  ++argc;
160  }
161 
162  /* zero out the current position (space or quotation mark) and advance */
163  if (*pos > 0) {
164  *pos = 0;
165  ++pos;
166  }
167  else {
168  break;
169  }
170  }
171  if (!argc) {
172  return;
173  }
174 
175  /* then we fill the argv array */
176  char *argv[argc + 1];
177  argv[argc] = NULL;
178  pos = line;
179  for (unsigned i = 0; i < argc; ++i) {
180  while (!*pos) {
181  ++pos;
182  }
183  if (*pos == '"' || *pos == '\'') {
184  ++pos;
185  }
186  argv[i] = pos;
187  while (*pos) {
188  ++pos;
189  }
190  }
191  for (char **arg = argv; contains_esc_seq && *arg; ++arg) {
192  for (char *c = *arg; *c; ++c) {
193  if (*c != '\\') {
194  continue;
195  }
196  for (char *d = c; *d; ++d) {
197  *d = d[1];
198  }
199  if (--contains_esc_seq == 0) {
200  break;
201  }
202  }
203  }
204 
205  /* then we call the appropriate handler */
206  shell_command_handler_t handler = find_handler(command_list, argv[0]);
207  if (handler != NULL) {
208  handler(argc, argv);
209  }
210  else {
211  if (strcmp("help", argv[0]) == 0) {
212  print_help(command_list);
213  }
214  else {
215  printf("shell: command not found: %s\n", argv[0]);
216  }
217  }
218 }

220 static int readline(char *buf, size_t size)
221 {
222  char *line_buf_ptr = buf;
223 
224  while (1) {
225  if ((line_buf_ptr - buf) >= ((int) size) - 1) {
226  return -1;
227  }
228 
229  int c = getchar();
230  if (c < 0) {
231  return 1;
232  }
233 
234  /* We allow Unix linebreaks (\n), DOS linebreaks (\r\n), and Mac linebreaks (\r). */
235  /* QEMU transmits only a single '\r' == 13 on hitting enter ("-serial stdio"). */
236  /* DOS newlines are handled like hitting enter twice, but empty lines are ignored. */
237  if (c == '\r' || c == '\n') {
238  *line_buf_ptr = '\0';
239 #ifndef SHELL_NO_ECHO
240  _putchar('\r');
241  _putchar('\n');
242 #endif
243 
244  /* return 1 if line is empty, 0 otherwise */
245  return line_buf_ptr == buf;
246  }
247  /* QEMU uses 0x7f (DEL) as backspace, while 0x08 (BS) is for most terminals */
248  else if (c == 0x08 || c == 0x7f) {
249  if (line_buf_ptr == buf) {
250  /* The line is empty. */
251  continue;
252  }
253 
254  *--line_buf_ptr = '\0';
255  /* white-tape the character */
256 #ifndef SHELL_NO_ECHO
257  _putchar('\b');
258  _putchar(' ');
259  _putchar('\b');
260 #endif
261  }
262  else {
263  *line_buf_ptr++ = c;
264 #ifndef SHELL_NO_ECHO
265  _putchar(c);
266 #endif
267  }
268  }
269 }
