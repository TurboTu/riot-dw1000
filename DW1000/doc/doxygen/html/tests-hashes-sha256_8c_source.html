137 static int calc_and_compare_hash(const char *str, const unsigned char *expected)
138 {
139  static unsigned char hash[SHA256_DIGEST_LENGTH];
140  sha256_context_t sha256;
141 
142  sha256_init(&sha256);
143  sha256_update(&sha256, (uint8_t*)str, strlen(str));
144  sha256_final(&sha256, hash);
145 
146  return (memcmp(expected, hash, SHA256_DIGEST_LENGTH) == 0);
147 }
