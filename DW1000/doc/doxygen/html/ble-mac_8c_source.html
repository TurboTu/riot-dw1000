166 int ble_mac_send(uint8_t dest[8], void *data, size_t len)
167 {
168  DEBUG("ble_mac_send(): sending pkt with len %u\n", (unsigned)len);
169 
170 #if defined(MODULE_OD) && ENABLE_DEBUG
171  od_hex_dump(dest, 8, OD_WIDTH_DEFAULT);
172  od_hex_dump(data, len, OD_WIDTH_DEFAULT);
173 #endif
174 
175  int i;
176  ble_ipsp_handle_t *handle;
177  int ret = -1;
178 
179  if ((!dest) || _is_broadcast(dest)) {
180  DEBUG("broadcast\n");
181  for (i = 0; i < BLE_MAC_MAX_INTERFACE_NUM; i++) {
182  if (interfaces[i].handle.cid != 0 && interfaces[i].handle.conn_handle != 0) {
183  ret = _send_to_peer(&interfaces[i].handle, data, len);
184  DEBUG("ret=%i\n", ret);
185  }
186  }
187  }
188  else if ((handle = _find_handle(dest)) != NULL) {
189  DEBUG("unicast\n");
190  ret = _send_to_peer(handle, data, len);
191  }
192  else {
193  DEBUG("ble-mac: no connection found for peer\n");
194  }
195 
196  if (ret == NRF_SUCCESS) {
197  ble_mac_busy_tx = 1;
198  return 0;
199  }
200  else {
201  DEBUG("ble-mac: send error: %i\n", ret);
202  return -1;
203  }
204 }

206 static uint32_t ble_mac_ipsp_evt_handler_irq(ble_ipsp_handle_t *p_handle, ble_ipsp_evt_t *p_evt)
207 {
208  uint32_t retval = NRF_SUCCESS;
209 
210  ble_mac_interface_t *p_instance = NULL;
211 
212  p_instance = ble_mac_interface_lookup(p_handle);
213 
214  if (p_handle) {
215  DEBUG("ble-mac: IPSP event [handle:%d CID 0x%04X]\n", p_handle->conn_handle, p_handle->cid);
216  }
217 
218  switch (p_evt->evt_id) {
219  case BLE_IPSP_EVT_CHANNEL_CONNECTED: {
220  uint8_t peer_addr[8];
221 
222  DEBUG("ble-mac: channel connected\n");
223  ble_eui64_from_eui48(peer_addr, p_evt->evt_param->params.ch_conn_request.peer_addr.addr,
224  p_evt->evt_param->params.ch_conn_request.peer_addr.addr_type ==
225  BLE_GAP_ADDR_TYPE_PUBLIC);
226 
227  p_instance = ble_mac_interface_add(peer_addr, p_handle);
228 
229  if (p_instance != NULL) {
230  DEBUG("ble-mac: added new IPSP interface\n");
231  }
232  else {
233  DEBUG("ble-mac: cannot add new interface. Table is full\n");
234  ble_ipsp_disconnect(p_handle);
235  }
236  break;
237  }
238 
239  case BLE_IPSP_EVT_CHANNEL_DISCONNECTED: {
240  DEBUG("ble-mac: channel disconnected\n");
241  if (p_instance != NULL) {
242  DEBUG("ble-mac: removed IPSP interface\n");
243  ble_mac_interface_delete(p_instance);
244  }
245  break;
246  }
247 
248  case BLE_IPSP_EVT_CHANNEL_DATA_RX: {
249  DEBUG("ble-mac: data received\n");
250  if (p_instance != NULL) {
251  if (ble_mac_busy_rx) {
252  DEBUG("ble-mac: packet dropped as input buffer is busy\n");
253  break;
254  }
255 
256  if (p_evt->evt_param->params.ch_rx.len > BLE_SIXLOWPAN_MTU) {
257  DEBUG("ble-mac: packet buffer is too small!\n");
258  break;
259  }
260 
261  ble_mac_busy_rx = 1;
262 
263  inbuf.len = p_evt->evt_param->params.ch_rx.len;
264  memcpy(inbuf.payload, p_evt->evt_param->params.ch_rx.p_data, inbuf.len);
265  memcpy(inbuf.src, p_instance->peer_addr, 8);
266  sd_ble_gap_rssi_get(p_handle->conn_handle, &inbuf.rssi);
267 
268  _callback(BLE_EVENT_RX_DONE, &inbuf);
269  }
270  else {
271  DEBUG("ble-mac: got data to unknown interface!\n");
272  }
273  break;
274  }
275 
276  case BLE_IPSP_EVT_CHANNEL_DATA_TX_COMPLETE: {
277  DEBUG("ble-mac: data transmitted\n");
278  ble_mac_busy_tx = 0;
279  //_callback(BLE_EVENT_TX_DONE, NULL);
280  break;
281  }
282  }
283 
284  if (sched_context_switch_request) {
285  NVIC_SetPendingIRQ( SWI0_EGU0_IRQn );
286  }
287 
288  return retval;
289 }
