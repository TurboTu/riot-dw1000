60 static int write(i2c_t bus, uint8_t addr, const void *data, int len, int stop)
61 {
62  uint8_t *buf = (uint8_t *)data;
63 
64  assert((bus <= I2C_NUMOF) && (len > 0));
65  DEBUG("i2c: writing %i byte to the bus\n", len);
66 
67  dev(bus)->ADDRESS = (addr & 0x7f);
68 
69  for (int i = 0; i < len; i++) {
70  dev(bus)->TXD = *buf++;
71  dev(bus)->EVENTS_TXDSENT = 0;
72  dev(bus)->TASKS_STARTTX = 1;
73  while (!(dev(bus)->EVENTS_TXDSENT) && !(dev(bus)->EVENTS_ERROR)) {}
74  if (dev(bus)->EVENTS_ERROR) {
75  return error(bus);
76  }
77  }
78 
79  if (stop) {
80  dev(bus)->EVENTS_STOPPED = 0;
81  dev(bus)->TASKS_STOP = 1;
82  while (!(dev(bus)->EVENTS_STOPPED) && !(dev(bus)->EVENTS_ERROR)) {}
83  if (dev(bus)->EVENTS_ERROR) {
84  return error(bus);
85  }
86  }
87 
88  return len;
89 }

92 {

140 int i2c_read_bytes(i2c_t bus, uint8_t address, void *data, int length)
141 {
142  uint8_t *in_buf = (uint8_t *)data;
143 
144  assert((bus <= I2C_NUMOF) && (length > 0));
145  DEBUG("[i2c] reading %i byte from the bus\n", length);
146 
147  /* set the client address */
148  dev(bus)->ADDRESS = (address & 0x7f);
149 
150  /* setup PPI channel as alternative to the broken SHORTS
151  * -> see PAN notice #36: "Shortcuts described in nRF51 Reference Manual are
152  * not functional." */
153  if (length == 1) {
154  NRF_PPI->CH[i2c_config[bus].ppi].TEP = (uint32_t)&dev(bus)->TASKS_STOP;
155  }
156  else {
157  NRF_PPI->CH[i2c_config[bus].ppi].TEP = (uint32_t)&dev(bus)->TASKS_SUSPEND;
158  }
159  NRF_PPI->CHENSET = (1 << i2c_config[bus].ppi);
160 
161  dev(bus)->EVENTS_RXDREADY = 0;
162  dev(bus)->EVENTS_STOPPED = 0;
163  dev(bus)->TASKS_STARTRX = 1;
164 
165  for (int i = (length - 1); i >= 0; i--) {
166  while (!(dev(bus)->EVENTS_RXDREADY) && !(dev(bus)->EVENTS_ERROR)) {}
167  if (dev(bus)->EVENTS_ERROR) {
168  return error(bus);
169  }
170 
171  *in_buf++ = (uint8_t)dev(bus)->RXD;
172 
173  if (i == 1) {
174  NRF_PPI->CH[i2c_config[bus].ppi].TEP = (uint32_t)&dev(bus)->TASKS_STOP;
175  }
176 
177  dev(bus)->EVENTS_RXDREADY = 0;
178  dev(bus)->TASKS_RESUME = 1;
179  }
180 
181  /* wait for the device to finish up */
182  while (dev(bus)->EVENTS_STOPPED == 0) {}
183  NRF_PPI->CHENCLR = (1 << i2c_config[bus].ppi);
184 
185  return length;
186 }
