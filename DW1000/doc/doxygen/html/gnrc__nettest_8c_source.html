51 static gnrc_nettest_res_t _pkt_test(uint16_t cmd_type, kernel_pid_t pid,
52  gnrc_pktsnip_t *in, unsigned int exp_pkts,
53  const kernel_pid_t *exp_senders,
54  const gnrc_pktsnip_t **exp_out)
55 {
56  msg_t msg;
57  gnrc_nettest_res_t res = GNRC_NETTEST_SUCCESS;
58 
59  msg.type = cmd_type;
60  msg.content.ptr = in;
61 
62  msg_send(&msg, pid);
63 
64  if (exp_pkts == 0) {
65  thread_yield();
66  }
67 
68  for (unsigned int i = 0; i < exp_pkts; i++) {
69  gnrc_pktsnip_t *out;
70  const gnrc_pktsnip_t *exp = exp_out[i];
71 
72  if (xtimer_msg_receive_timeout(&msg, GNRC_NETTEST_TIMEOUT) < 0) {
73  return GNRC_NETTEST_TIMED_OUT;
74  }
75 
76  if (msg.type != cmd_type) {
77  return GNRC_NETTEST_WRONG_MSG;
78  }
79 
80  if (msg.sender_pid != exp_senders[i]) {
81  return GNRC_NETTEST_WRONG_SENDER;
82  }
83 
84  out = msg.content.ptr;
85 
86  if (out == NULL) {
87  return GNRC_NETTEST_FAIL;
88  }
89 
90  while (out && exp) {
91  if ((out->users != exp->users) ||
92  (out->size != exp->size) ||
93  (out->type != exp->type) ||
94  (memcmp(out->data, exp->data, out->size) != 0)) {
95  return GNRC_NETTEST_FAIL;
96  }
97 
98  out = out->next;
99  exp = exp->next;
100  }
101 
102  gnrc_pktbuf_release(msg.content.ptr);
103  }
104 
105  return res;
106 }

207 static inline uint32_t _get_set_opt(gnrc_nettest_opt_cb_t cb, uint16_t context,
208  void *data, uint16_t data_len)
209 {
210  int res;
211 
212  mutex_lock(&_mutex);
213  if (cb != NULL) {
214  res = cb(context, data, data_len);
215  }
216  else {
217  res = -ENOTSUP;
218  }
219  mutex_unlock(&_mutex);
220  return (uint32_t)res;
221 }

223 static void *_event_loop(void *arg)
224 {
225  msg_t reply, msg_queue[GNRC_NETTEST_MSG_QUEUE_SIZE];
226 
227  (void)arg;
228  msg_init_queue(msg_queue, GNRC_NETTEST_MSG_QUEUE_SIZE);
229  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
230 
231  while (1) {
232  msg_t msg;
233  gnrc_netapi_opt_t *opt;
234 
235  msg_receive(&msg);
236 
237  switch (msg.type) {
238  case GNRC_NETAPI_MSG_TYPE_GET:
239  opt = msg.content.ptr;
240  reply.content.value = _get_set_opt(_opt_cbs[opt->opt].get,
241  opt->context, opt->data,
242  opt->data_len);
243  break;
244 
245  case GNRC_NETAPI_MSG_TYPE_SET:
246  opt = msg.content.ptr;
247  reply.content.value = _get_set_opt(_opt_cbs[opt->opt].set,
248  opt->context, opt->data,
249  opt->data_len);
250  break;
251  }
252 
253  msg_reply(&msg, &reply);
254  }
255 
256  return NULL;
257 }
