51 uint8_t flashrom_write(uint8_t *dst, const uint8_t *src, size_t size)
52 {
53  (void) size; /* unused */
54 
55  char err;
56  uint8_t sec;
57 
58  sec = iap_get_sector((uint32_t) dst);
59 
60  if (sec == INVALID_ADDRESS) {
61  DEBUG("Invalid address\n");
62  return 0;
63  }
64 
65  /* check sector */
66  if (blank_check_sector(sec, sec) == SECTOR_NOT_BLANK) {
67  DEBUG("Warning: Sector %i not blank\n", sec);
68  }
69 
70  /* prepare sector */
71  err = prepare_sectors(sec, sec);
72  if (err) {
73  DEBUG("\n-- ERROR: PREPARE_SECTOR_FOR_WRITE_OPERATION: %u\n", err);
74  return 0;
75  }
76 
77  /* write flash */
78  unsigned intstate = irq_disable();
79  err = copy_ram_to_flash((uint32_t) dst, (uint32_t) src, 256);
80  irq_restore(intstate);
81 
82  if (err) {
83  DEBUG("ERROR: COPY_RAM_TO_FLASH: %u\n", err);
84  /* set interrupts back and return */
85  irq_restore(intstate);
86  return 0;
87  }
88  /* check result */
89  err = compare((uint32_t) dst, (uint32_t) src, 256);
90 
91  if (err) {
92  DEBUG("ERROR: COMPARE: %i (at position %u)\n", err, iap_result[1]);
93  return 0;
94  }
95 
96  DEBUG("Data successfully written!\n");
97  return 1;
98 }

101 uint8_t flashrom_erase(uint8_t *addr)
102 {
103  uint8_t sec = iap_get_sector((uint32_t) addr);
104  unsigned intstate;
105 
106  if (sec == INVALID_ADDRESS) {
107  DEBUG("Invalid address\n");
108  return 0;
109  }
110 
111  /* check sector */
112  if (!blank_check_sector(sec, sec)) {
113  DEBUG("Sector already blank!\n");
114  return 1;
115  }
116 
117  /* prepare sector */
118  if (prepare_sectors(sec, sec)) {
119  DEBUG("-- ERROR: PREPARE_SECTOR_FOR_WRITE_OPERATION --\n");
120  return 0;
121  }
122 
123  intstate = irq_disable();
124 
125  /* erase sector */
126  if (erase_sectors(sec, sec)) {
127  DEBUG("-- ERROR: ERASE SECTOR --\n");
128  irq_restore(intstate);
129  return 0;
130  }
131 
132  irq_restore(intstate);
133 
134  /* check again */
135  if (blank_check_sector(sec, sec)) {
136  DEBUG("-- ERROR: BLANK_CHECK_SECTOR\n");
137  return 0;
138  }
139 
140  DEBUG("Sector successfully erased.\n");
141  return 1;
142 }

163  *
164  * Description: This command is used to blank check a sector or multiple sectors

166  * same "Start" and "End" sector numbers.
167  * Command: 53

181  *****************************************************************************/

188 /******************************************************************************
189  * Function: copy_ram_to_flash
190  *
191  * Description: This command is used to programm the flash memory. the affected should be
192  * prepared first by calling "Prepare Sector for Write Operation" command. the
193  * affected sectors are automatically protected again once the copy command is
194  * successfully executed. the boot sector cannot be written by this command.
195  * Command: 51
196  * Param0: (DST) Destination Flash adress where data bytes are to be written.
197  * This address should be a 512 byte boundary.
198  * Param1: (SRC) Source RAM adress from which data byre are to be read.
199  * Param2: Number of bytes to be written. Should be 512 | 1024 | 4096 | 8192.
200  * Param3: System Clock Frequency (CCLK) in KHz.
201  *
202  * Parameters: long tmp_adr_dst: Param0
203  * long tmp_adr_src: Param1
204  * long tmp_size: Param2
205  *
206  * Return: Code CMD_SUCCESS |
207  * SRC_ADDR_ERROR (Address not on word boundary) |
208  * DST_ADDR_ERROR (Address not on correct boundary) |
209  * SRC_ADDR_NOT_MAPPED |
210  * DST_ADDR_NOT_MAPPED |
211  * COUNT_ERROR (Byte count is not 512 | 1024 | 4096 | 8192) |
212  * SECTOR_NOT_PREPARED_FOR_WRITE_OPERATION |
213  * BUSY
214  *****************************************************************************/

273  * Description: This command is used to compare the memory contents at two locations. compare result may not
274  * be correct when source or destination address contains any of the first 64 bytes starting

276  * Command Code: 56
277  * Param0(DST): Starting Flash or RAM address from where data bytes are to be

280  * address should be a word boundary.

282  *

284  * long tmp_adr_src
285  * long tmp_size
286  *
287  * Return: Code CMD_SUCCESS |
288  * COMPARE_ERROR |
289  * COUNT_ERROR (Byte count is not multiple of 4) |
290  * ADDR_ERROR |
291  * ADDR_NOT_MAPPED
292  * Result0: Offset of the first mismatch if the Status Code is COMPARE_ERROR.
293  *****************************************************************************/
