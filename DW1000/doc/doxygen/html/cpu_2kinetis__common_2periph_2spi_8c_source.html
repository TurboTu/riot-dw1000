81 void spi_init(spi_t bus)
82 {
83  /* make sure given bus device is valid */
84  assert(bus < SPI_NUMOF);
85 
86  /* initialize the buses lock */
87  mutex_init(&locks[bus]);
88  /* trigger pin initialization */
89  spi_init_pins(bus);
90  /* power on the bus temporarily */
91  poweron(bus);
92 
93  /* make the base configuration: configure as SPI master, set CS inactive
94  * state (for HWCS lines) and clear FIFO counters and disable FIFOs */
95  dev(bus)->MCR = (SPI_MCR_MSTR_MASK | SPI_MCR_PCSIS_MASK |
96  SPI_MCR_CLR_RXF_MASK | SPI_MCR_CLR_TXF_MASK |
97  SPI_MCR_DIS_RXF_MASK | SPI_MCR_DIS_TXF_MASK |
98  SPI_MCR_DOZE_MASK | SPI_MCR_HALT_MASK | SPI_MCR_MDIS_MASK);
99 
100  /* disable all DMA and interrupt requests */
101  dev(bus)->RSER = 0;
102 
103  /* and power off the bus until it is actually used */
104  poweroff(bus);
105 }

114 int spi_init_cs(spi_t bus, spi_cs_t cs)
115 {
116  if (bus >= SPI_NUMOF) {

118  }

121  }
122

126  else {

128  (spi_config[bus].pin_cs[cs] == GPIO_UNDEF)) {
129  return SPI_NOCS;

131  gpio_init_port(spi_config[bus].pin_cs[cs], spi_config[bus].pcr);
132  }
133

137 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
138 {
139  /* lock and power on the bus */
140  mutex_lock(&locks[bus]);
141  poweron(bus);
142 
143  /* enable the device */
144  dev(bus)->MCR &= ~(SPI_MCR_HALT_MASK | SPI_MCR_MDIS_MASK);
145 
146  /* configure clock and mode */
147  dev(bus)->CTAR[0] = (mode | SPI_CTAR_FMSZ(7) | spi_clk_config[clk]);
148 
149  return SPI_OK;

160 void spi_transfer_bytes(spi_t bus, spi_cs_t cs, bool cont,
161  const void *out, void *in, size_t len)
162 {
163  uint8_t *out_buf = (uint8_t *)out;
164  uint8_t *in_buf = (uint8_t *)in;
165  uint32_t flags = SPI_PUSHR_CONT_MASK;
166 
167  assert(out_buf || in_buf);
168 
169  /* handle chip select */
170  if (cs != SPI_CS_UNDEF) {
171  if (cs & SWCS_MASK) {
172  gpio_clear((gpio_t)cs);
173  }
174  else {
175  flags |= (uint32_t)(1 << (cs + SPI_PUSHR_PCS_SHIFT));
176  }
177  }
178 
179  for (size_t i = 0; i < len; i++) {
180  uint8_t tmp = (out_buf) ? out_buf[i] : 0;
181  if ((i == (len - 1)) && (!cont)) {
182  flags &= ~(SPI_PUSHR_CONT_MASK);
183  }
184 
185  while (!(dev(bus)->SR & SPI_SR_TFFF_MASK)) {}
186  dev(bus)->PUSHR = (tmp | flags);
187 
188  while (!(dev(bus)->SR & SPI_SR_RXCTR_MASK)) {}
189  tmp = dev(bus)->POPR;
190  if (in_buf) {
191  in_buf[i] = tmp;
192  }
193  }
194 
195  if ((!cont) && (cs != SPI_CS_UNDEF) && (cs & SWCS_MASK)) {
196  gpio_set((gpio_t)cs);
197  }
198 }
