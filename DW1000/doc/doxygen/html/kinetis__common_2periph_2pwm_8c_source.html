38 uint32_t pwm_init(pwm_t pwm, pwm_mode_t mode, uint32_t freq, uint16_t res)
39 {
40  uint8_t pre = 0;
41 
42  if (pwm >= PWM_NUMOF || ((res * freq) > CLOCK_BUSCLOCK)) {
43  return 0;
44  }
45 
46  /* figure out the clock settings
47  * the resulting frequency is calculated by
48  * ticks := BUS_CLK / 2 ^ pre
49  * where `ticks` is `freq * res`
50  * and `pre` must be between [0, 7].
51  *
52  * The resulting prescaler yields a timer frequency, which is the closest
53  * possible frequency requested. */
54  while ((CLOCK_BUSCLOCK >> pre) > (res * freq)) {
55  ++pre;
56  }
57  /* make sure the calculated prescaler is valid, else return */
58  if (pre > PRESCALER_MAX) {
59  return 0;
60  }
61 
62  /* configure the used timer */
63  pwm_poweron(pwm);
64  /* disable write protect for changing settings */
65  ftm(pwm)->MODE = FTM_MODE_WPDIS_MASK;
66  /* clear any existing configuration */
67  ftm(pwm)->COMBINE = 0;
68  ftm(pwm)->CNTIN = 0;
69  ftm(pwm)->SWOCTRL = 0;
70  /* apply prescaler and set resolution */
71  ftm(pwm)->SC = FTM_SC_PS(pre);
72  ftm(pwm)->MOD = (res - 1);
73 
74  /* set CPWMS bit in the SC register in case of center aligned mode */
75  if (mode == PWM_CENTER) {
76  BITBAND_REG32(ftm(pwm)->SC, FTM_SC_CPWMS_SHIFT) = 1;
77  }
78 
79  /* setup the configured channels */
80  for (int i = 0; i < (int)pwm_config[pwm].chan_numof; i++) {
81  /* configure the used pin */
82  gpio_init_port(pwm_config[pwm].chan[i].pin,
83  PORT_PCR_MUX(pwm_config[pwm].chan[i].af));
84  /* set the given mode */
85  ftm(pwm)->CONTROLS[pwm_config[pwm].chan[i].ftm_chan].CnSC = mode;
86  /* and reset the PWM to 0% duty cycle */
87  ftm(pwm)->CONTROLS[pwm_config[pwm].chan[i].ftm_chan].CnV = 0;
88  }
89 
90  /* and now we start the actual waveform generation */
91  pwm_start(pwm);
92 
93  /* finally we need to return the actual applied PWM frequency */
94  return (CLOCK_BUSCLOCK >> pre) / res;
95 }
