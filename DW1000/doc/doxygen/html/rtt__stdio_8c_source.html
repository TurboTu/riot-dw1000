188 static int rtt_read(char* buf_ptr, uint16_t buf_size) {
189  int16_t num_bytes_rem;
190  uint16_t num_bytes_read;
191  int16_t rd_off;
192  int16_t wr_off;
193 
194  rd_off = rtt_cb.down[0].rd_off;
195  wr_off = rtt_cb.down[0].wr_off;
196  num_bytes_read = 0;
197 
198  /* Read from current read position to wrap-around of buffer, first */
199  if (rd_off > wr_off) {
200  num_bytes_rem = rtt_cb.down[0].buf_size - rd_off;
201  num_bytes_rem = MIN(num_bytes_rem, (int)buf_size);
202  memcpy(buf_ptr, rtt_cb.down[0].buf_ptr + rd_off, num_bytes_rem);
203  num_bytes_read += num_bytes_rem;
204  buf_ptr += num_bytes_rem;
205  buf_size -= num_bytes_rem;
206  rd_off += num_bytes_rem;
207  /* Handle wrap-around of buffer */
208  if (rd_off == rtt_cb.down[0].buf_size) {
209  rd_off = 0;
210  }
211  }
212 
213  /* Read remaining items of buffer */
214  num_bytes_rem = wr_off - rd_off;
215  num_bytes_rem = MIN(num_bytes_rem, (int)buf_size);
216  if (num_bytes_rem > 0) {
217  memcpy(buf_ptr, rtt_cb.down[0].buf_ptr + rd_off, num_bytes_rem);
218  num_bytes_read += num_bytes_rem;
219  rd_off += num_bytes_rem;
220  }
221  if (num_bytes_read) {
222  rtt_cb.down[0].rd_off = rd_off;
223  }
224  return num_bytes_read;
225 }

235 int rtt_write(const char* buf_ptr, unsigned num_bytes) {
236  int num_bytes_to_write;
237  unsigned num_bytes_written;
238  int rd_off;
239 
240  rd_off = rtt_cb.up[0].rd_off;
241  num_bytes_to_write = rd_off - rtt_cb.up[0].wr_off - 1;
242  if (num_bytes_to_write < 0) {
243  num_bytes_to_write += rtt_cb.up[0].buf_size;
244  }
245  /* If the complete data does not fit in the buffer, trim the data */
246  if ((int)num_bytes > num_bytes_to_write) {
247  num_bytes = num_bytes_to_write;
248  }
249 
250  /* Early out if there is nothing to do */
251  if (num_bytes == 0) {
252  return 0;
253  }
254 
255  /* Write data to buffer and handle wrap-around if necessary */
256  num_bytes_written = 0;
257  do {
258  /* May be changed by host (debug probe) in the meantime */
259  rd_off = rtt_cb.up[0].rd_off;
260  num_bytes_to_write = rd_off - rtt_cb.up[0].wr_off - 1;
261  if (num_bytes_to_write < 0) {
262  num_bytes_to_write += rtt_cb.up[0].buf_size;
263  }
264  /* Number of bytes that can be written until buffer wrap-around */
265  num_bytes_to_write = MIN(num_bytes_to_write, (rtt_cb.up[0].buf_size -
266  rtt_cb.up[0].wr_off));
267  num_bytes_to_write = MIN(num_bytes_to_write, (int)num_bytes);
268  memcpy(rtt_cb.up[0].buf_ptr + rtt_cb.up[0].wr_off, buf_ptr, num_bytes_to_write);
269  num_bytes_written += num_bytes_to_write;
270  buf_ptr += num_bytes_to_write;
271  num_bytes -= num_bytes_to_write;
272  rtt_cb.up[0].wr_off += num_bytes_to_write;
273  if (rtt_cb.up[0].wr_off == rtt_cb.up[0].buf_size) {
274  rtt_cb.up[0].wr_off = 0;
275  }
276  } while (num_bytes);
277  return num_bytes_written;
278 }

302 /* The reason we have this strange logic is as follows:
303  If we have an RTT console, we are powered, and so don't care
304  that polling uses a lot of power. If however, we do not
305  actually have an RTT console (because we are deployed on
306  a battery somewhere) then we REALLY don't want to poll
307  especially since we are not expecting to EVER get input. */
308 int uart_stdio_read(char* buffer, int count) {
309  int res = rtt_read(buffer, count);
310  if (res == 0) {
311  if (!stdin_enabled) {
312  mutex_lock(&_rx_mutex);
313  /* We only unlock when rtt_stdio_enable_stdin is called
314  Note that we assume only one caller invoked this function */
315  }
316  xtimer_ticks32_t last_wakeup = xtimer_now();
317  while(1) {
318  xtimer_periodic_wakeup(&last_wakeup, STDIO_POLL_INTERVAL);
319  res = rtt_read(buffer, count);
320  if (res > 0)
321  return res;
322  }
323  }
324  return res;
325 }

327 int uart_stdio_write(const char* buffer, int len) {
328  int written = rtt_write(buffer, len);
329  xtimer_ticks32_t last_wakeup = xtimer_now();
330  while (blocking_stdout && written < len) {
331  xtimer_periodic_wakeup(&last_wakeup, STDIO_POLL_INTERVAL);
332  written += rtt_write(&buffer[written], len-written);
333  }
334  return written;
335 }
