76 char *thread_stack_init(thread_task_func_t task_func, void *arg, void *stack_start, int stack_size)
77 {
78  unsigned short stk = (unsigned short)((uintptr_t) stack_start + stack_size);
79 
80  /* ensure correct stack alignment (on 16-bit boundary) */
81  stk &= 0xfffe;
82  unsigned short *stackptr = (unsigned short *)stk;
83 
84  /* now make SP point on the first AVAILABLE slot in stack */
85  --stackptr;
86 
87  *stackptr = (unsigned short) sched_task_exit;
88  --stackptr;
89 
90  *stackptr = (unsigned short) task_func;
91  --stackptr;
92 
93  /* initial value for SR */
94 
95  *stackptr = GIE;
96  --stackptr;
97 
98  /* set argument to task_func */
99  *stackptr = (unsigned short) arg;
100  --stackptr;
101 
102  /* Space for registers. */
103  for (unsigned int i = 14; i > 4; i--) {
104  *stackptr = i;
105  --stackptr;
106  }
107 
108  return (char *) stackptr;
109 }
