51 nib_entry_t *nib_process_hello(void)
52 {
53  nib_entry_t *nb_match = NULL;
54  nib_entry_t *nib_elt, *nib_tmp;
55  timex_t now;
56  uint8_t matches = 0;
57 
58  mutex_lock(&mtx_nib_access);
59 
60  xtimer_now_timex(&now);
61 
62  LL_FOREACH_SAFE(nib_entry_head, nib_elt, nib_tmp) {
63  nhdp_addr_entry_t *list_elt;
64  LL_FOREACH(nib_elt->address_list_head, list_elt) {
65  if (NHDP_ADDR_TMP_IN_NB_LIST(list_elt->address)) {
66  /* Matching neighbor tuple */
67  matches++;
68 
69  if (matches > 1) {
70  /* Multiple matching nb tuples, delete the previous one */
71  iib_propagate_nb_entry_change(nb_match, nib_elt);
72  rem_nib_entry(nb_match, &now);
73  }
74 
75  nb_match = nib_elt;
76  break;
77  }
78  }
79  }
80 
81  /* Add or update nb tuple */
82  if (matches > 0) {
83  /* We found matching nb tuples, reuse the last one */
84  clear_nb_addresses(nb_match, &now);
85 
86  if (matches > 1) {
87  nb_match->symmetric = 0;
88  }
89 
90  nb_match->address_list_head = nhdp_generate_addr_list_from_tmp(NHDP_ADDR_TMP_NB_LIST);
91 
92  if (!nb_match->address_list_head) {
93  /* Insufficient memory */
94  LL_DELETE(nib_entry_head, nb_match);
95  free(nb_match);
96  nb_match = NULL;
97  }
98  }
99  else {
100  nb_match = add_nib_entry_for_nb_addr_list();
101  }
102 
103  mutex_unlock(&mtx_nib_access);
104  return nb_match;
105 }

107 void nib_fill_wr_addresses(struct rfc5444_writer *wr)
108 {
109  nib_entry_t *nib_elt;
110  nhdp_addr_entry_t *addr_elt;
111  nib_lost_address_entry_t *lost_elt, *lost_tmp;
112  timex_t now;
113 
114  mutex_lock(&mtx_nib_access);
115 
116  xtimer_now_timex(&now);
117 
118  /* Add addresses of symmetric neighbors to HELLO msg */
119  LL_FOREACH(nib_entry_head, nib_elt) {
120  if (nib_elt->symmetric) {
121  LL_FOREACH(nib_elt->address_list_head, addr_elt) {
122  /* Check whether address is not already included with link status symmetric */
123  if (!NHDP_ADDR_TMP_IN_SYM(addr_elt->address)) {
124  nhdp_writer_add_addr(wr, addr_elt->address, RFC5444_ADDRTLV_OTHER_NEIGHB,
125  RFC5444_OTHERNEIGHB_SYMMETRIC,
126  rfc5444_metric_encode(nib_elt->metric_in),
127  rfc5444_metric_encode(nib_elt->metric_out));
128  addr_elt->address->in_tmp_table = NHDP_ADDR_TMP_SYM;
129  }
130  }
131  }
132  }
133 
134  /* Add lost addresses of neighbors to HELLO msg */
135  LL_FOREACH_SAFE(nib_lost_address_entry_head, lost_elt, lost_tmp) {
136  if (timex_cmp(lost_elt->expiration_time, now) != 1) {
137  /* Entry expired, remove it */
138  rem_ln_entry(lost_elt);
139  }
140  else {
141  /* Check if address is not already present in one of the temporary lists */
142  if (!NHDP_ADDR_TMP_IN_ANY(lost_elt->address)) {
143  /* Address is not present in one of the lists, add it */
144  nhdp_writer_add_addr(wr, lost_elt->address, RFC5444_ADDRTLV_OTHER_NEIGHB,
145  RFC5444_OTHERNEIGHB_LOST, NHDP_METRIC_UNKNOWN,
146  NHDP_METRIC_UNKNOWN);
147  }
148  }
149  }
150 
151  mutex_unlock(&mtx_nib_access);
152 }

161 void nib_set_nb_entry_sym(nib_entry_t *nib_entry)
162 {
163  nib_lost_address_entry_t *ln_elt, *ln_tmp;
164  nhdp_addr_entry_t *nb_elt;
165 
166  nib_entry->symmetric = 1;
167  LL_FOREACH(nib_entry->address_list_head, nb_elt) {
168  LL_FOREACH_SAFE(nib_lost_address_entry_head, ln_elt, ln_tmp) {
169  /* Remove all Lost Neighbor Tuples matching an address of the newly sym nb */

172  break;

200 static nib_entry_t *add_nib_entry_for_nb_addr_list(void)
201 {
202  nib_entry_t *new_elem;
203 
204  new_elem = malloc(sizeof(nib_entry_t));
205 
206  if (!new_elem) {
207  /* Insufficient memory */
208  return NULL;
209  }
210 
211  /* Copy neighbor address list to new neighbor tuple */
212  new_elem->address_list_head = nhdp_generate_addr_list_from_tmp(NHDP_ADDR_TMP_NB_LIST);
213 
214  if (!new_elem->address_list_head) {
215  /* Insufficient memory */
216  free(new_elem);
217  return NULL;
218  }
219 
220  new_elem->symmetric = 0;
221  new_elem->metric_in = NHDP_METRIC_UNKNOWN;
222  new_elem->metric_out = NHDP_METRIC_UNKNOWN;
223  LL_PREPEND(nib_entry_head, new_elem);
224 
225  return new_elem;
226 }

242 static void clear_nb_addresses(nib_entry_t *nib_entry, timex_t *now)
243 {
244  nhdp_addr_entry_t *nib_elt, *nib_tmp;
245 
246  LL_FOREACH_SAFE(nib_entry->address_list_head, nib_elt, nib_tmp) {
247  /* Check whether address is still present in the new neighbor address list */
248  if (!NHDP_ADDR_TMP_IN_NB_LIST(nib_elt->address)) {
249  /* Address is not in the newly received address list of the neighbor */
250  /* Add it to the Removed Address List */
251  nib_elt->address->in_tmp_table |= NHDP_ADDR_TMP_REM_LIST;
252  /* Increment usage counter of address in central NHDP address storage */
253  nib_elt->address->usg_count++;
254 
255  if (nib_entry->symmetric) {
256  /* Additionally create a Lost Neighbor Tuple for symmetric neighbors */
257  add_lost_neighbor_address(nib_elt->address, now);
258  }
259  }
260 
261  /* Free the address entry */
262  nhdp_free_addr_entry(nib_elt);
263  }
264  nib_entry->address_list_head = NULL;
265 }

270 static int add_lost_neighbor_address(nhdp_addr_t *lost_addr, timex_t *now)
271 {
272  nib_lost_address_entry_t *elt;
273  timex_t n_hold = timex_from_uint64(((uint64_t)NHDP_N_HOLD_TIME_MS) * US_PER_MS);
274 
275  LL_FOREACH(nib_lost_address_entry_head, elt) {
276  if (elt->address == lost_addr) {
277  /* Existing entry for this address, no need to add a new one */
278  if (timex_cmp(elt->expiration_time, *now) == -1) {
279  /* Entry expired, so just update expiration time */
280  elt->expiration_time = timex_add(*now, n_hold);
281  }
282 
283  return 0;
284  }
285  }
286 
287  /* No existing entry, create a new one */
288  elt = malloc(sizeof(nib_lost_address_entry_t));
289 
290  if (!elt) {
291  /* Insufficient memory */
292  return -1;
293  }
294 
295  /* Increment usage counter of address in central NHDP address storage */
296  lost_addr->usg_count++;
297  elt->address = lost_addr;
298  elt->expiration_time = timex_add(*now, n_hold);
299  LL_PREPEND(nib_lost_address_entry_head, elt);
300 
301  return 0;
302 }
