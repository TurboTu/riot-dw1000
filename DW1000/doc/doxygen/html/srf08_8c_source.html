94 int srf08_get_distances(srf08_t *dev, uint16_t *range_array, int num_echos, srf08_mode_t ranging_mode)
95 {
96  int status;
97  int echo_number = 0;
98  uint8_t range_bytes[sizeof(uint16_t)];
99  uint8_t register_location;
100  char max_reg_no_read = (num_echos * sizeof(range_bytes)) +1;
101 
102  /* Acquire exclusive access to the bus. */
103  i2c_acquire(dev->i2c);
104  /* set ranging mode */
105  status = i2c_write_reg(dev->i2c, dev->addr, SRF08_COMMAND_REG, ranging_mode);
106  /* Release the bus for other threads. */
107  i2c_release(dev->i2c);
108 
109  if (!status) {
110  DEBUG("Write the ranging command to the i2c-interface is failed\n");
111  return -1;
112  }
113 
114  if(max_reg_no_read > SRF08_MAX_REGISTER_NUMBER) {
115  DEBUG("Too many echos requested. Max. is 17\n");
116  return -2;
117  }
118 
119  xtimer_usleep(70000);
120 
121  /* read all echo buffers */
122  for (register_location = 2; register_location < max_reg_no_read;
123  register_location += sizeof(range_bytes)) {
124 
125  /* Acquire exclusive access to the bus. */
126  i2c_acquire(dev->i2c);
127  /* read the echo bytes */
128  status = i2c_read_regs(dev->i2c, dev->addr, register_location, range_bytes, sizeof(range_bytes));
129  /* Release the bus for other threads. */
130  i2c_release(dev->i2c);
131 
132  if (!status) {
133  DEBUG("Read the echo bytes from the i2c-interface is failed\n");
134  return -3;
135  }
136 
137  else {
138  uint16_t distance = (range_bytes[0] << 8) | range_bytes[1];
139  range_array[(register_location - 2) / 2] = distance;
140  echo_number++;
141  }
142  }
143 
144  return echo_number;
145 }
