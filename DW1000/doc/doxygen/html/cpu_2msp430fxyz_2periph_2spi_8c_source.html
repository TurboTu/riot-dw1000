36 void spi_init(spi_t bus)
37 {
38  assert(bus <= SPI_NUMOF);
39 
40 /* we need to differentiate between the legacy SPI device and USCI */
41 #ifndef SPI_USE_USCI
42  /* put SPI device in reset state */
43  SPI_BASE->CTL = USART_CTL_SWRST;
44  SPI_BASE->CTL |= (USART_CTL_CHAR | USART_CTL_SYNC | USART_CTL_MM);
45  SPI_BASE->RCTL = 0;
46  SPI_BASE->MCTL = 0;
47  /* enable SPI mode */
48  SPI_ME |= SPI_ME_BIT;
49 #else
50  /* reset SPI device */
51  SPI_BASE->CTL1 = USCI_SPI_CTL1_SWRST;
52  SPI_BASE->CTL1 |= (USCI_SPI_CTL1_SSEL_SMCLK);
53 #endif
54 
55  /* trigger the pin configuration */
56  spi_init_pins(bus);
57 }

66 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
67 {
68  if (clk == SPI_CLK_10MHZ) {
69  return SPI_NOCLK;
70  }
71 
72  /* lock the bus */
73  mutex_lock(&spi_lock);
74 
75  /* calculate baudrate */
76  uint32_t br = CLOCK_CMCLK / clk;
77  /* make sure the is not smaller then 2 */
78  if (br < 2) {
79  br = 2;
80  }
81  SPI_BASE->BR0 = (uint8_t)br;
82  SPI_BASE->BR1 = (uint8_t)(br >> 8);
83 
84  /* configure bus mode */
85 #ifndef SPI_USE_USCI
86  /* configure mode */
87  SPI_BASE->TCTL = (USART_TCTL_SSEL_SMCLK | USART_TCTL_STC | mode);
88  /* release from software reset */
89  SPI_BASE->CTL &= ~(USART_CTL_SWRST);
90 #else
91  /* configure mode */
92  SPI_BASE->CTL0 = (USCI_SPI_CTL0_UCSYNC | USCI_SPI_CTL0_MST|
93  USCI_SPI_CTL0_MODE_0 | USCI_SPI_CTL0_MSB | mode);
94  /* release from software reset */
95  SPI_BASE->CTL1 &= ~(USCI_SPI_CTL1_SWRST);
96 #endif
97 
98  return SPI_OK;
99 }

114 void spi_transfer_bytes(spi_t bus, spi_cs_t cs, bool cont,
115  const void *out, void *in, size_t len)
116 {
117  uint8_t *out_buf = (uint8_t *)out;
118  uint8_t *in_buf = (uint8_t *)in;
119 
120  assert(out_buf || in_buf);
121 
122  if (cs != SPI_CS_UNDEF) {
123  gpio_clear((gpio_t)cs);
124  }
125 
126  /* if we only send out data, we do this the fast way... */
127  if (!in_buf) {
128  for (size_t i = 0; i < len; i++) {
129  while (!(SPI_IF & SPI_IE_TX_BIT)) {}
130  SPI_BASE->TXBUF = (uint8_t)out_buf[i];
131  }
132  /* finally we need to wait, until all transfers are complete */
133 #ifndef SPI_USE_USCI
134  while (!(SPI_IF & SPI_IE_TX_BIT) || !(SPI_IF & SPI_IE_RX_BIT)) {}
135 #else
136  while (SPI_BASE->STAT & USCI_SPI_STAT_UCBUSY) {}
137 #endif
138  SPI_BASE->RXBUF;
139  }
140  else if (!out_buf) {
141  for (size_t i = 0; i < len; i++) {
142  SPI_BASE->TXBUF = 0;
143  while (!(SPI_IF & SPI_IE_RX_BIT)) {}
144  in_buf[i] = (char)SPI_BASE->RXBUF;
145  }
146  }
147  else {
148  for (size_t i = 0; i < len; i++) {
149  while (!(SPI_IF & SPI_IE_TX_BIT)) {}
150  SPI_BASE->TXBUF = out_buf[i];
151  while (!(SPI_IF & SPI_IE_RX_BIT)) {}
152  in_buf[i] = (char)SPI_BASE->RXBUF;
153  }
154  }
155 
156  if ((!cont) && (cs != SPI_CS_UNDEF)) {
157  gpio_set((gpio_t)cs);
158  }
159 }
