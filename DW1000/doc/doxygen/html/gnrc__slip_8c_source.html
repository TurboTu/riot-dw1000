52 /* UART callbacks */
53 static void _slip_rx_cb(void *arg, uint8_t data)
54 {
55  if (data == (uint8_t)_SLIP_END) {
56  msg_t msg;
57 
58  msg.type = _SLIP_MSG_TYPE;
59  msg.content.value = _SLIP_DEV(arg)->in_bytes;
60 
61  msg_send_int(&msg, _SLIP_DEV(arg)->slip_pid);
62 
63  _SLIP_DEV(arg)->in_bytes = 0;
64  }
65  else if (_SLIP_DEV(arg)->in_esc) {
66  _SLIP_DEV(arg)->in_esc = 0;
67 
68  switch (data) {
69  case ((uint8_t)_SLIP_END_ESC):
70  if (ringbuffer_add_one(&_SLIP_DEV(arg)->in_buf, _SLIP_END) < 0) {
71  _SLIP_DEV(arg)->in_bytes++;
72  }
73 
74  break;
75 
76  case ((uint8_t)_SLIP_ESC_ESC):
77  if (ringbuffer_add_one(&_SLIP_DEV(arg)->in_buf, _SLIP_ESC) < 0) {
78  _SLIP_DEV(arg)->in_bytes++;
79  }
80 
81  break;
82 
83  default:
84  break;
85  }
86  }
87  else if (data == (uint8_t)_SLIP_ESC) {
88  _SLIP_DEV(arg)->in_esc = 1;
89  }
90  else {
91  if (ringbuffer_add_one(&_SLIP_DEV(arg)->in_buf, data) < 0) {
92  _SLIP_DEV(arg)->in_bytes++;
93  }
94  }
95 }

97 /* SLIP receive handler */
98 static void _slip_receive(gnrc_slip_dev_t *dev, size_t bytes)
99 {
100  gnrc_pktsnip_t *pkt, *hdr;
101 
102  hdr = gnrc_netif_hdr_build(NULL, 0, NULL, 0);
103  if (hdr == NULL) {
104  DEBUG("slip: no space left in packet buffer\n");
105  return;
106  }
107 
108  ((gnrc_netif_hdr_t *)(hdr->data))->if_pid = thread_getpid();
109 
110  pkt = gnrc_pktbuf_add(hdr, NULL, bytes, GNRC_NETTYPE_UNDEF);
111 
112  if (pkt == NULL) {
113  DEBUG("slip: no space left in packet buffer\n");
114  gnrc_pktbuf_release(hdr);
115  return;
116  }
117 
118  if (ringbuffer_get(&dev->in_buf, pkt->data, bytes) != bytes) {
119  DEBUG("slip: could not read %u bytes from ringbuffer\n", (unsigned)bytes);
120  gnrc_pktbuf_release(pkt);
121  return;
122  }
123 #if ENABLE_DEBUG && defined(MODULE_OD)
124  else {
125  DEBUG("slip: received data\n");
126  od_hex_dump(pkt->data, bytes, OD_WIDTH_DEFAULT);
127  }
128 #endif
129 
130 #ifdef MODULE_GNRC_IPV6
131  if ((pkt->size >= sizeof(ipv6_hdr_t)) && ipv6_hdr_is(pkt->data)) {
132  pkt->type = GNRC_NETTYPE_IPV6;
133  }
134 #endif
135 
136  if (gnrc_netapi_dispatch_receive(pkt->type, GNRC_NETREG_DEMUX_CTX_ALL, pkt) == 0) {
137  DEBUG("slip: unable to forward packet of type %i\n", pkt->type);
138  gnrc_pktbuf_release(pkt);
139  }
140 }

147 /* SLIP send handler */
148 static void _slip_send(gnrc_slip_dev_t *dev, gnrc_pktsnip_t *pkt)
149 {
150  gnrc_pktsnip_t *ptr;
151 
152  ptr = pkt->next; /* ignore gnrc_netif_hdr_t, we don't need it */
153 
154  while (ptr != NULL) {
155  DEBUG("slip: send pktsnip of length %u over UART_%d\n", (unsigned)ptr->size, dev->uart);
156  char *data = ptr->data;
157 
158  for (size_t i = 0; i < ptr->size; i++) {
159  switch (data[i]) {
160  case _SLIP_END:
161  DEBUG("slip: encountered END byte on send: stuff with ESC\n");
162  _slip_send_char(dev, _SLIP_ESC);
163  _slip_send_char(dev, _SLIP_END_ESC);
164  break;
165 
166  case _SLIP_ESC:
167  DEBUG("slip: encountered ESC byte on send: stuff with ESC\n");
168  _slip_send_char(dev, _SLIP_ESC);
169  _slip_send_char(dev, _SLIP_ESC_ESC);
170  break;
171 
172  default:
173  _slip_send_char(dev, data[i]);
174 
175  break;
176  }
177  }
178 
179  ptr = ptr->next;
180  }
181 
182  _slip_send_char(dev, _SLIP_END);
183 
184  gnrc_pktbuf_release(pkt);
185 }

197 static void *_slip(void *args)
198 {
199  gnrc_slip_dev_t *dev = _SLIP_DEV(args);
200  msg_t msg, reply, msg_q[_SLIP_MSG_QUEUE_SIZE];
201 
202  msg_init_queue(msg_q, _SLIP_MSG_QUEUE_SIZE);
203  dev->slip_pid = thread_getpid();
204  gnrc_netif_add(dev->slip_pid);
205 
206  DEBUG("slip: SLIP runs on UART_%d\n", dev->uart);
207 
208  while (1) {
209  DEBUG("slip: waiting for incoming messages\n");
210  msg_receive(&msg);
211 
212  switch (msg.type) {
213  case _SLIP_MSG_TYPE:
214  DEBUG("slip: incoming message of size %" PRIu32 " from UART_%d in buffer\n",
215  msg.content.value, dev->uart);
216  _slip_receive(dev, (size_t)msg.content.value);
217  break;
218 
219  case GNRC_NETAPI_MSG_TYPE_SND:
220  DEBUG("slip: GNRC_NETAPI_MSG_TYPE_SND received\n");
221  _slip_send(dev, msg.content.ptr);
222  break;
223 
224  case GNRC_NETAPI_MSG_TYPE_GET:
225  DEBUG("slip: GNRC_NETAPI_MSG_TYPE_GET received\n");
226  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
227  reply.content.value = (uint32_t)_slip_get(msg.content.ptr);
228  msg_reply(&msg, &reply);
229  break;
230 
231  case GNRC_NETAPI_MSG_TYPE_SET:
232  DEBUG("slip: GNRC_NETAPI_MSG_TYPE_SET received\n");
233  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
234  reply.content.value = (uint32_t)(-ENOTSUP);
235  DEBUG("slip: I don't support this but have to reply.\n");
236  msg_reply(&msg, &reply);
237  break;
238  }
239  }
240 
241  /* should be never reached */
242  return NULL;
243 }

245 kernel_pid_t gnrc_slip_init(gnrc_slip_dev_t *dev, uart_t uart, uint32_t baudrate,
246  char *stack, size_t stack_size, char priority)
247 {
248  kernel_pid_t pid;
249 
250  /* reset device descriptor fields */
251  dev->uart = uart;
252  dev->in_bytes = 0;
253  dev->in_esc = 0;
254  dev->slip_pid = KERNEL_PID_UNDEF;
255 
256  /* initialize buffers */
257  ringbuffer_init(&dev->in_buf, dev->rx_mem, sizeof(dev->rx_mem));
258 
259  /* initialize UART */
260  DEBUG("slip: initialize UART_%d with baudrate %" PRIu32 "\n", uart,
261  baudrate);
262  if (uart_init(uart, baudrate, _slip_rx_cb, dev) != UART_OK) {
263  DEBUG("slip: error initializing UART_%i with baudrate %" PRIu32 "\n",
264  uart, baudrate);
265  return -ENODEV;
266  }
267 
268  /* start SLIP thread */
269  DEBUG("slip: starting SLIP thread\n");
270  pid = thread_create(stack, stack_size, priority, THREAD_CREATE_STACKTEST,
271  _slip, dev, _SLIP_NAME);
272  if (pid < 0) {
273  DEBUG("slip: unable to create SLIP thread\n");
274  return -EFAULT;
275  }
276  return pid;
277 }
