28 void color_rgb2hsv(color_rgb_t *rgb, color_hsv_t *hsv)
29 {
30  float rd, gd, bd, delta;
31 
32  /* norm RGB colors to the range [0 - 1.0] */
33  rd = (float)rgb->r / 255.0f;
34  gd = (float)rgb->g / 255.0f;
35  bd = (float)rgb->b / 255.0f;
36 
37  /* find value as maximum of the three colors */
38  if (rd >= gd) {
39  hsv->v = (rd >= bd) ? rd : bd;
40  }
41  else {
42  hsv->v = (gd >= bd) ? gd : bd;
43  }
44 
45  /* compute delta from value and minimum of the three RGB colors */
46  if (rd <= gd) {
47  delta = (rd <= bd) ? (hsv->v - rd) : (hsv->v - bd);
48  }
49  else {
50  delta = (gd <= bd) ? (hsv->v - gd) : (hsv->v - bd);
51  }
52 
53  /* find the saturation from value and delta */
54  hsv->s = (hsv->v != 0.0f) ? (delta / hsv->v) : 0x0f;
55 
56  /* compute hue */
57  hsv->h = 0.0f;
58  if (hsv->s != 0.0) {
59  float rc, gc, bc;
60 
61  rc = (hsv->v - rd) / delta;
62  gc = (hsv->v - gd) / delta;
63  bc = (hsv->v - bd) / delta;
64 
65  if (rd == hsv->v) {
66  hsv->h = bc - gc;
67  }
68  else if (gd == hsv->v) {
69  hsv->h = 2.0f + rc - bc;
70  }
71  else {
72  hsv->h = 4.0f + gc - rc;
73  }
74  hsv->h *= 60.0f;
75  if (hsv->h < 0.0f) {
76  hsv->h += 360.0;
77  }
78  }
