38 int servo_init(servo_t *dev, pwm_t pwm, int pwm_channel, unsigned int min, unsigned int max)
39 {
40  int actual_frequency;
41 
42  actual_frequency = pwm_init(pwm, PWM_LEFT, SERVO_FREQUENCY, SERVO_RESOLUTION);
43 
44  DEBUG("servo: requested %d hz, got %d hz\n", SERVO_FREQUENCY, actual_frequency);
45 
46  if (actual_frequency < 0) {
47  /* PWM error */
48  return -1;
49  }
50  dev->device = pwm;
51  dev->channel = pwm_channel;
52  dev->min = min;
53  dev->max = max;
54 
55  /* Compute scaling fractional */
56  /*
57  * The PWM pulse width can be written as:
58  *
59  * t = k / (f * r)
60  *
61  * where t is the pulse high time, k is the value set in the PWM peripheral,
62  * f is the frequency, and r is the resolution of the PWM module.
63  *
64  * define t0 as the desired pulse width:
65  *
66  * t0 = k0 / (f0 * r)
67  *
68  * where f0 is the requested frequency, k0 is the requested number of ticks.
69  * Introducing f1 as the closest achievable frequency and k1 as the set tick
70  * value yields:
71  *
72  * t1 = k1 / (f1 * r)
73  *
74  * setting t1 = t0 and substituting k1 = k0 * s yields:
75  *
76  * k0 / (f0 * r) = k0 * s / (f1 * r)
77  *
78  * solve for s:
79  *
80  * s = f1 / f0
81  *
82  * where s is the optimal scale factor to translate from requested position
83  * to actual hardware ticks.
84  */
85  dev->scale_nom = actual_frequency;
86  dev->scale_den = SERVO_FREQUENCY;
87 
88  return 0;
89 }
