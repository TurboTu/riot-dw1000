147 int gpio_init(gpio_t pin, gpio_mode_t mode)
148 {
149  Pio *port = _port(pin);
150  int pin_num = _pin_num(pin);
151  int port_num = _port_num(pin);
152 
153  /* make sure port is valid and no pull-down is selected*/
154  if (!_port_valid(port) || (mode == GPIO_IN_PD)) {
155  return -1;
156  }
157 
158  /* power on the corresponding port */
159  PMC->PMC_PCER0 = (1 << (port_num + 11));
160 
161  /* disable interrupt and clear context (to be safe) */
162  port->PIO_IDR = (1 << pin_num);
163  _ctx_clear(port_num, pin_num);
164 
165  /* give the PIO module the power over the corresponding pin */
166  port->PIO_PER = (1 << pin_num);
167 
168  /* configure pin direction (in/out) */
169  if (mode & MODE_BIT_IO) {
170  port->PIO_OER = (1 << pin_num);
171  }
172  else {
173  port->PIO_ODR = (1 << pin_num);
174  }
175  /* set pull-up */
176  if (mode & MODE_BIT_PUE) {
177  port->PIO_PUER = (1 << pin_num);
178  }
179  else {
180  port->PIO_PUDR = (1 << pin_num);
181  }
182  /* set multi-driver (open-drain) mode */
183  if (mode & MODE_BIT_ODE) {
184  port->PIO_MDER = (1 << pin_num);
185  }
186  else {
187  port->PIO_MDDR = (1 << pin_num);
188  }
189 
190  return 0;
191 }

193 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
194  gpio_cb_t cb, void *arg)
195 {
196  Pio *port = _port(pin);
197  int pin_num = _pin_num(pin);
198  int port_num = _port_num(pin);
199 
200  /* make sure pin is valid */
201  if (!_port_valid(port)) {
202  return -1;
203  }
204 
205  /* configure pin as input */
206  gpio_init(pin, mode);
207 
208  /* try go grab a free spot in the context array */
209  int ctx_num = _get_free_ctx();
210  if (ctx_num < 0) {
211  return -1;
212  }
213  /* save context */
214  exti_ctx[ctx_num].cb = cb;
215  exti_ctx[ctx_num].arg = arg;
216  _write_map(port_num, pin_num, ctx_num);
217 
218  /* set the active flank */
219  switch (flank) {
220  case GPIO_FALLING:
221  port->PIO_AIMER = (1 << pin_num);
222  port->PIO_ESR = (1 << pin_num);
223  port->PIO_FELLSR =(1 << pin_num);
224  break;
225  case GPIO_RISING:
226  port->PIO_AIMER = (1 << pin_num);
227  port->PIO_ESR = (1 << pin_num);
228  port->PIO_REHLSR = (1 << pin_num);
229  break;
230  case GPIO_BOTH:
231  port->PIO_AIMDR = (1 << pin_num);
232  break;
233  }
234  /* clean interrupt status register */
235  port->PIO_ISR;
236  /* enable the interrupt for the given channel */
237  NVIC_EnableIRQ(port_num + PIOA_IRQn);
238  port->PIO_IER = (1 << pin_num);
239 
240  return 0;
241 }

305 static inline void isr_handler(Pio *port, int port_num)
306 {
307  /* take interrupt flags only from pins which interrupt is enabled */
308  uint32_t status = (port->PIO_ISR & port->PIO_IMR);
309 
310  for (int i = 0; i < 32; i++) {
311  if (status & (1 << i)) {
312  int ctx = _ctx(port_num, i);
313  exti_ctx[ctx].cb(exti_ctx[ctx].arg);
314  }
315  }
316  cortexm_isr_end();
317 }
