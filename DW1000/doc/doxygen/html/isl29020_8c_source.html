30  isl29020_range_t range, isl29020_mode_t mode)
31 {

33  uint8_t tmp;

35  /* initialize device descriptor */
36  dev->i2c = i2c;

40  /* acquire exclusive access to the bus */
41  i2c_acquire(dev->i2c);

48  /* release the bus for other threads */

56 int isl29020_read(isl29020_t *dev)
57 {
58  uint8_t low, high;
59  uint16_t res;
60  int ret;
61 
62  i2c_acquire(dev->i2c);
63  /* read lighting value */
64  ret = i2c_read_reg(dev->i2c, dev->address, ISL29020_REG_LDATA, &low);
65  ret += i2c_read_reg(dev->i2c, dev->address, ISL29020_REG_HDATA, &high);
66  i2c_release(dev->i2c);
67  if (ret < 2) {
68  return -1;
69  }
70  res = (high << 8) | low;
71  DEBUG("ISL29020: Raw value: %i - high: %i, low: %i\n", res, high, low);
72  /* calculate and return the actual lux value */
73  return (int)(dev->lux_fac * res);
74 }
