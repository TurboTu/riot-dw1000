127 char *thread_arch_stack_init(thread_task_func_t task_func,
128  void *arg,
129  void *stack_start,
130  int stack_size)
131 {
132  uint32_t *stk;
133  stk = (uint32_t *)((uintptr_t)stack_start + stack_size);
134 
135  /* adjust to 32 bit boundary by clearing the last two bits in the address */
136  stk = (uint32_t *)(((uint32_t)stk) & ~((uint32_t)0x3));
137 
138  /* stack start marker */
139  stk--;
140  *stk = STACK_MARKER;
141 
142  /* make sure the stack is double word aligned (8 bytes) */
143  /* This is required in order to conform with Procedure Call Standard for the
144  * ARM® Architecture (AAPCS) */
145  /* http://infocenter.arm.com/help/topic/com.arm.doc.ihi0042e/IHI0042E_aapcs.pdf */
146  if (((uint32_t) stk & 0x7) != 0) {
147  /* add a single word padding */
148  --stk;
149  *stk = ~((uint32_t)STACK_MARKER);
150  }
151 
152 #ifdef CPU_ARCH_CORTEX_M4F
153  /* TODO: fix FPU handling for Cortex-M4f */
154  /*
155  stk--;
156  *stk = (unsigned int) 0;
157  */
158 
159  /* S0 - S15 */
160  /*
161  for (int i = 15; i >= 0; i--) {
162  stk--;
163  *stk = i;
164  }
165  */
166 #endif
167 
168  /* ****************************** */
169  /* Automatically popped registers */
170  /* ****************************** */
171 
172  /* The following eight stacked registers are popped by the hardware upon
173  * return from exception. (bx instruction in context_restore) */
174 
175  /* xPSR - initial status register */
176  stk--;
177  *stk = (uint32_t)INITIAL_XPSR;
178  /* pc - initial program counter value := thread entry function */
179  stk--;
180  *stk = (uint32_t)task_func;
181  /* lr - contains the return address when the thread exits */
182  stk--;
183  *stk = (uint32_t)sched_task_exit;
184  /* r12 */
185  stk--;
186  *stk = 0;
187  /* r3 - r1 */
188  for (int i = 3; i >= 1; i--) {
189  stk--;
190  *stk = i;
191  }
192  /* r0 - contains the thread function parameter */
193  stk--;
194  *stk = (uint32_t)arg;
195 
196  /* ************************* */
197  /* Manually popped registers */
198  /* ************************* */
199 
200  /* The following registers are not handled by hardware in return from
201  * exception, but manually by context_restore.
202  * For the Cortex-M0(plus) we write registers R11-R4 in two groups to allow
203  * for more efficient context save/restore code.
204  * For the Cortex-M3 and Cortex-M4 we write them continuously onto the stack
205  * as they can be read/written continuously by stack instructions. */
206 
207 #if defined(CPU_ARCH_CORTEX_M0) || defined(CPU_ARCH_CORTEX_M0PLUS)
208  /* start with r7 - r4 */
209  for (int i = 7; i >= 4; i--) {
210  stk--;
211  *stk = i;
212  }
213  /* and put r11 - r8 on top of them */
214  for (int i = 11; i >= 8; i--) {
215  stk--;
216  *stk = i;
217  }
218 #else
219  /* r11 - r4 */
220  for (int i = 11; i >= 4; i--) {
221  stk--;
222  *stk = i;
223  }
224 #endif
225 
226  /* exception return code - return to task-mode process stack pointer */
227  stk--;
228  *stk = (uint32_t)EXCEPT_RET_TASK_MODE;
229 
230  /* The returned stack pointer will be aligned on a 32 bit boundary not on a
231  * 64 bit boundary because of the odd number of registers above (8+9).
232  * This is not a problem since the initial stack pointer upon process entry
233  * _will_ be 64 bit aligned (because of the cleared bit 9 in the stacked
234  * xPSR and aligned stacking of the hardware-handled registers). */
235 
236  return (char*) stk;
237 }

259 {
260  uint32_t *ptr = &_sstack;
261 
262  while(((*ptr) == STACK_CANARY_WORD) && (ptr < &_estack)) {
263  ++ptr;
264  }

299 void __attribute__((naked)) __attribute__((used)) isr_pendsv(void) {
300  __asm__ volatile (
301  /* PendSV handler entry point */
302  /* save context by pushing unsaved registers to the stack */
303  /* {r0-r3,r12,LR,PC,xPSR} are saved automatically on exception entry */
304  ".thumb_func \n"
305  "mrs r0, psp \n" /* get stack pointer from user mode */
306 #if defined(CPU_ARCH_CORTEX_M0) || defined(CPU_ARCH_CORTEX_M0PLUS)
307  "mov r12, sp \n" /* remember the exception SP */
308  "mov sp, r0 \n" /* set user mode SP as active SP */
309  /* we can not push high registers directly, so we move R11-R8 into
310  * R4-R0, as these are already saved */
311  "mov r0, r8 \n" /* move R11-R8 into R3-R0 */
312  "mov r1, r9 \n"
313  "mov r2, r10 \n"
314  "mov r3, r11 \n"
315  "push {r0-r7} \n" /* now push them onto the stack */
316  "mov r0, lr \n" /* next we save the link register */
317  "push {r0} \n"
318  "mov r0, sp \n" /* switch back to the exception SP */
319  "mov sp, r12 \n"
320 #else
321  "stmdb r0!,{r4-r11} \n" /* save regs */
322  "stmdb r0!,{lr} \n" /* exception return value */
323 #ifdef CPU_ARCH_CORTEX_M4F
324 /* "vstmdb sp!, {s16-s31} \n" */ /* TODO save FPU registers */
325 #endif
326 #endif
327  "ldr r1, =sched_active_thread \n" /* load address of current tcb */
328  "ldr r1, [r1] \n" /* dereference pdc */
329  "str r0, [r1] \n" /* write r0 to pdc->sp */
330  "bl isr_svc \n" /* continue with svc */
331  );
332 }

334 void __attribute__((naked)) __attribute__((used)) isr_svc(void) {
335  __asm__ volatile (
336  /* SVC handler entry point */
337  /* PendSV will continue here as well (via jump) */
338  ".thumb_func \n"
339  /* perform scheduling */
340  "bl sched_run \n"
341  /* restore context and return from exception */
342  ".thumb_func \n"
343  "context_restore: \n"
344 #if defined(CPU_ARCH_CORTEX_M0) || defined(CPU_ARCH_CORTEX_M0PLUS)
345  "mov lr, sp \n" /* save MSR stack pointer for later */
346  "ldr r0, =sched_active_thread \n" /* load address of current TCB */
347  "ldr r0, [r0] \n" /* dereference TCB */
348  "ldr r0, [r0] \n" /* load tcb-sp to R0 */
349  "mov sp, r0 \n" /* make user mode SP active SP */
350  "pop {r0} \n" /* restore LR from stack */
351  "mov r12, r0 \n" /* remember LR by parking it in R12 */
352  "pop {r0-r7} \n" /* get R11-R8 and R7-R4 from stack */
353  "mov r8, r0 \n" /* move R11-R8 to correct registers */
354  "mov r9, r1 \n"
355  "mov r10, r2 \n"
356  "mov r11, r3 \n"
357  /* restore the application mode stack pointer PSP */
358  "mov r0, sp \n" /* restore the user mode SP */
359  "msr psp, r0 \n" /* for this write it to the PSP reg */
360  "mov sp, lr \n" /* and get the parked MSR SP back */
361  /* return from exception mode to application mode */
362  "bx r12 \n" /* return from exception mode */
363 #else
364  "ldr r0, =sched_active_thread \n" /* load address of current TCB */
365  "ldr r0, [r0] \n" /* dereference TCB */
366  "ldr r1, [r0] \n" /* load tcb->sp to register 1 */
367  "ldmia r1!, {r0} \n" /* restore exception return value */
368 #ifdef CPU_ARCH_CORTEX_M4F
369 /* "pop {s16-s31} \n" */ /* TODO load FPU registers */
370 #endif
371  "ldmia r1!, {r4-r11} \n" /* restore other registers */
372  "msr psp, r1 \n" /* restore user mode SP to PSP reg */
373  "bx r0 \n" /* load exception return value to PC,
374  * causes end of exception*/
375 #endif
376  /* {r0-r3,r12,LR,PC,xPSR} are restored automatically on exception return */
377  );
378 }
