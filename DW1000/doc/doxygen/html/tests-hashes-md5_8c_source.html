30 static int calc_and_compare_hash(const char *str, const char *expected)
31 {
32  uint8_t hash[MD5_DIGEST_LENGTH];
33  char tmp[(2 * MD5_DIGEST_LENGTH) + 1];
34 
35  /* calculate hash */
36  md5(hash, (const uint8_t *)str, strlen(str));
37  /* copy hash to string */
38  for (size_t i = 0; i < MD5_DIGEST_LENGTH; i++) {
39  sprintf(&(tmp[i * 2]), "%02x", hash[i]);
40  }
41  tmp[MD5_DIGEST_LENGTH * 2] = '\0';
42  /* compare with result string */
43  return strncmp(tmp, expected, strlen(tmp));
44 }
