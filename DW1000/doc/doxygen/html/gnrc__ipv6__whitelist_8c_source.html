31 int gnrc_ipv6_whitelist_add(const ipv6_addr_t *addr)
32 {
33  for (int i = 0; i < GNRC_IPV6_WHITELIST_SIZE; i++) {
34  if (!bf_isset(gnrc_ipv6_whitelist_set, i)) {
35  bf_set(gnrc_ipv6_whitelist_set, i);
36  memcpy(&gnrc_ipv6_whitelist[i], addr, sizeof(*addr));
37  DEBUG("IPv6 whitelist: whitelisted %s\n",
38  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)));
39  return 0;
40  }
41  }
42  return -1;
43 }
