242 static inline uint8_t GNRC_RPL_COUNTER_INCREMENT(uint8_t counter)
243 {
244  return ((counter > GNRC_RPL_COUNTER_LOWER_REGION) ?
245  ((counter == GNRC_RPL_COUNTER_MAX) ? counter = 0 : ++counter) :
246  ((counter == GNRC_RPL_COUNTER_LOWER_REGION) ? counter = 0 : ++counter));
247 }

254 static inline bool GNRC_RPL_COUNTER_GREATER_THAN_LOCAL(uint8_t A, uint8_t B)
255 {
256  return (((A < B) && (GNRC_RPL_COUNTER_LOWER_REGION + 1 - B + A < GNRC_RPL_COUNTER_SEQ_WINDOW))
257  || ((A > B) && (A - B < GNRC_RPL_COUNTER_SEQ_WINDOW)));
258 }
259 
260 static inline bool GNRC_RPL_COUNTER_GREATER_THAN(uint8_t A, uint8_t B)
261 {
262  return ((A > GNRC_RPL_COUNTER_LOWER_REGION) ? ((B > GNRC_RPL_COUNTER_LOWER_REGION) ?
263  GNRC_RPL_COUNTER_GREATER_THAN_LOCAL(A, B) : 0) :
264  ((B > GNRC_RPL_COUNTER_LOWER_REGION) ? 1 : GNRC_RPL_COUNTER_GREATER_THAN_LOCAL(A, B)));
265 }
