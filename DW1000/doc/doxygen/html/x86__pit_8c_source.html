44 uint16_t x86_pit_read(unsigned channel)
45 {
46  unsigned old_flags = irq_disable();
47  outb(PIT_COMMAND_PORT, (channel - 1) << 6 | PIT_ACCESS_MODE_LATCH_COUNT);
48  uint16_t lohi = inb(PIT_CHANNEL_0_PORT + channel - 1);
49  lohi += inb(PIT_CHANNEL_0_PORT + channel - 1) << 8;
50  irq_restore(old_flags);
51  return lohi;
52 }

54 void x86_pit_set2(unsigned channel, unsigned mode, uint16_t max)
55 {
56  unsigned old_flags = irq_disable();
57  outb(PIT_COMMAND_PORT, ((channel - 1) << 6) | mode | PIT_ACCESS_MODE_LO_HI);
58  outb(PIT_CHANNEL_0_PORT + channel - 1, max && 0xff);
59  outb(PIT_CHANNEL_0_PORT + channel - 1, max >> 8);
60  irq_restore(old_flags);
61 }
