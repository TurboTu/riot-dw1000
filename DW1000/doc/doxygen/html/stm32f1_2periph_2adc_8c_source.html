69 int adc_init(adc_t line)
70 {
71  uint32_t clk_div = 2;
72 
73  /* check if the line is valid */
74  if (line >= ADC_NUMOF) {
75  return -1;
76  }
77 
78  /* lock and power-on the device */
79  prep(line);
80 
81  /* configure the pin */
82  gpio_init_analog(adc_config[line].pin);
83  /* set clock prescaler to get the maximal possible ADC clock value */
84  for (clk_div = 2; clk_div < 8; clk_div += 2) {
85  if ((CLOCK_CORECLOCK / clk_div) <= MAX_ADC_SPEED) {
86  break;
87  }
88  }
89  RCC->CFGR &= ~(RCC_CFGR_ADCPRE);
90  RCC->CFGR |= ((clk_div / 2) - 1) << 14;
91 
92  /* enable the ADC module */
93  dev(line)->CR2 |= ADC_CR2_ADON;
94 
95  /* resets the selected ADC calibration registers */
96  dev(line)->CR2 |= ADC_CR2_RSTCAL;
97  /* check the status of RSTCAL bit */
98  while (dev(line)->CR2 & ADC_CR2_RSTCAL) {}
99 
100  /* enable the selected ADC calibration process */
101  dev(line)->CR2 |= ADC_CR2_CAL;
102  /* wait for the calibration to have finished */
103  while (dev(line)->CR2 & ADC_CR2_CAL) {}
104 
105  /* set all channels to maximum (239.5) cycles for best accuracy */
106  dev(line)->SMPR1 |= 0x00ffffff;
107  dev(line)->SMPR2 |= 0x3fffffff;
108  /* we want to sample one channel */
109  dev(line)->SQR1 = ADC_SQR1_L_0;
110  /* start sampling from software */
111  dev(line)->CR2 |= ADC_CR2_EXTTRIG | ADC_CR2_EXTSEL;
112 
113  /* check if this channel is an internal ADC channel, if so
114  * enable the internal temperature and Vref */
115  if (adc_config[line].chan == 16 || adc_config[line].chan == 17) {
116  /* check if the internal channels are configured to use ADC1 */
117  if (dev(line) != ADC1) {
118  return -3;
119  }
120 
121  dev(line)->CR2 |= ADC_CR2_TSVREFE;
122  }
123 
124  /* free the device again */
125  done(line);
126  return 0;
127 }

129 int adc_sample(adc_t line, adc_res_t res)
130 {
131  int sample;
132 
133  /* check if the linenel is valid */
134  if (line >= ADC_NUMOF) {
135  return -1;
136  }
137 
138  /* check if resolution is applicable */
139  if (res != ADC_RES_12BIT) {
140  return -2;
141  }
142 
143  /* lock and power on the ADC device */
144  prep(line);
145 
146  /* set conversion channel */
147  dev(line)->SQR3 = adc_config[line].chan;
148  /* start conversion and wait for results */
149  dev(line)->CR2 |= ADC_CR2_SWSTART;
150  while (!(dev(line)->SR & ADC_SR_EOC)) {}
151  /* finally read sample and reset the STRT bit in the status register */
152  sample = (int)dev(line)->DR;
153 
154  /* power off and unlock device again */
155  done(line);
156 
157  return sample;
158 }
