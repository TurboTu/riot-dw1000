65 int nrf24l01p_init(nrf24l01p_t *dev, spi_t spi, gpio_t ce, gpio_t cs, gpio_t irq)
66 {
67  int status;
68  char INITIAL_TX_ADDRESS[] = {0xe7, 0xe7, 0xe7, 0xe7, 0xe7,};
69  char INITIAL_RX_ADDRESS[] = {0xe7, 0xe7, 0xe7, 0xe7, 0xe7,};
70 
71  dev->spi = spi;
72  dev->ce = ce;
73  dev->cs = cs;
74  dev->irq = irq;
75  dev->listener = KERNEL_PID_UNDEF;
76 
77  /* Init CE pin */
78  gpio_init(dev->ce, GPIO_OUT);
79 
80  /* Init CS pin */
81  spi_init_cs(dev->spi, dev->cs);
82 
83  /* Init IRQ pin */
84  gpio_init_int(dev->irq, GPIO_IN_PU, GPIO_FALLING, nrf24l01p_rx_cb, dev);
85 
86  /* Test the SPI connection */
87  if (spi_acquire(dev->spi, dev->cs, SPI_MODE, SPI_CLK) != SPI_OK) {
88  DEBUG("error: unable to acquire SPI bus with given params\n");
89  return -1;
90  }
91  spi_release(dev->spi);
92 
93  xtimer_spin(DELAY_AFTER_FUNC_TICKS);
94 
95  /* Flush TX FIFIO */
96  status = nrf24l01p_flush_tx_fifo(dev);
97 
98  if (status < 0) {
99  return status;
100  }
101 
102  /* Flush RX FIFIO */
103  status = nrf24l01p_flush_rx_fifo(dev);
104 
105  if (status < 0) {
106  return status;
107  }
108 
109  /* Setup adress width */
110  status = nrf24l01p_set_address_width(dev, NRF24L01P_AW_5BYTE);
111 
112  if (status < 0) {
113  return status;
114  }
115 
116  /* Setup payload width */
117  status = nrf24l01p_set_payload_width(dev, NRF24L01P_PIPE0, NRF24L01P_MAX_DATA_LENGTH);
118 
119  if (status < 0) {
120  return status;
121  }
122 
123  /* Set RF channel */
124  status = nrf24l01p_set_channel(dev, INITIAL_RF_CHANNEL);
125 
126  if (status < 0) {
127  return status;
128  }
129 
130  /* Set RF power */
131  status = nrf24l01p_set_power(dev, INITIAL_RX_POWER_0dB);
132 
133  if (status < 0) {
134  return status;
135  }
136 
137  /* Set RF datarate */
138  status = nrf24l01p_set_datarate(dev, NRF24L01P_DR_250KBS);
139 
140  if (status < 0) {
141  return status;
142  }
143 
144  /* Set TX Address */
145  status = nrf24l01p_set_tx_address(dev, INITIAL_TX_ADDRESS, INITIAL_ADDRESS_WIDTH);
146 
147  if (status < 0) {
148  return status;
149  }
150 
151  /* Set RX Adress */
152  status = nrf24l01p_set_rx_address(dev, NRF24L01P_PIPE0, INITIAL_RX_ADDRESS, INITIAL_ADDRESS_WIDTH);
153 
154  if (status < 0) {
155  return status;
156  }
157 
158  /* Reset auto ack for all pipes */
159  status = nrf24l01p_disable_all_auto_ack(dev);
160 
161  if (status < 0) {
162  return status;
163  }
164 
165  /* Setup Auto ACK and retransmission */
166  status = nrf24l01p_setup_auto_ack(dev, NRF24L01P_PIPE0, NRF24L01P_RETR_750US, 15);
167 
168  if (status < 0) {
169  return status;
170  }
171 
172  /* Setup CRC */
173  status = nrf24l01p_enable_crc(dev, NRF24L01P_CRC_2BYTE);
174 
175  if (status < 0) {
176  return status;
177  }
178 
179  /* Reset all interrupt flags */
180  status = nrf24l01p_reset_all_interrupts(dev);
181 
182  if (status < 0) {
183  return status;
184  }
185 
186  return nrf24l01p_on(dev);
187 }
