66 int jc42_get_temperature(jc42_t* dev, int16_t* temperature)
67 {
68  struct { signed int x:12;} s;
69  uint16_t tmp;
70 
71  /* Read temperature */
72  if (jc42_get_register(dev, JC42_REG_TEMP, &tmp) != 0) {
73  return JC42_NODEV;
74  }
75  tmp = NTOHS(tmp);
76  /* Convert fixed point to uint16_t */
77  *temperature = ((s.x = tmp)*100)>>4;
78  return JC42_OK;
79 }

81 int jc42_init(jc42_t* dev, jc42_params_t* params)
82 {
83  uint16_t config;
84  dev->i2c = params->i2c;
85  dev->addr = params->addr;
86  i2c_acquire(dev->i2c);
87  if (i2c_init_master(dev->i2c, params->speed) != 0) {
88  DEBUG("[jc42] Problem initializing I2C master\n");
89  i2c_release(dev->i2c);
90  return JC42_NOI2C;

92  i2c_release(dev->i2c);
93 
94  /* Poll the device, fail if unavailable */

96  return JC42_NODEV;
97  }
98 
99  DEBUG("[jc42] Config: 0x%x\n", config);
100 
101  return JC42_OK;
