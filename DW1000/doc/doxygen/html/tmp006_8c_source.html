200 {
201  int status;
202  uint8_t buf[2];
203 
204  if (dev->initialized == false) {
205  return -1;
206  }
207 
208  i2c_acquire(dev->i2c);
209  /* Register bytes are sent MSB first. */
210  status = i2c_read_regs(dev->i2c, dev->addr, TMP006_CONFIG, buf, 2);
211  if (status != 2) {
212  i2c_release(dev->i2c);
213  return -1;

215  i2c_release(dev->i2c);
216 
217  *drdy = buf[1] & (TMP006_CONFIG_DRDY);
218 
219  if (!(*drdy)) {
220  /* conversion in progress */
221  return -1;
222  }
223

226  if (status != 2) {
227  i2c_release(dev->i2c);
228  return -1;
229  }
230  i2c_release(dev->i2c);
231 
232  *rawv = ((uint16_t)buf[0] << 8) | buf[1];
233 
234  i2c_acquire(dev->i2c);
235  status = i2c_read_regs(dev->i2c, dev->addr, TMP006_T_AMBIENT, buf, 2);
236  if (status != 2) {

245 void tmp006_convert(int16_t rawv, int16_t rawt, float *tamb, float *tobj)
246 {
247  /* calculate die temperature */
248  *tamb = (double)rawt / 128.0;
249  /* die temperature in Kelvin */
250  double tdie_k = *tamb + 273.15;
251 
252  /* calculate sensor voltage */
253  double sens_v = (double)rawv * TMP006_CCONST_LSB_SIZE;
254 
255  double tdiff = tdie_k - TMP006_CCONST_TREF;
256  double tdiff_pow2 = pow(tdiff, 2);
257 
258  double s = TMP006_CCONST_S0 * (1 + TMP006_CCONST_A1 * tdiff
259  + TMP006_CCONST_A2 * tdiff_pow2);
260 
261  double v_os = TMP006_CCONST_B0 + TMP006_CCONST_B1 * tdiff
262  + TMP006_CCONST_B2 * tdiff_pow2;
263 
264  double f_obj = (sens_v - v_os) + TMP006_CCONST_C2 * pow((sens_v - v_os), 2);
265 
266  double t = pow(pow(tdie_k, 4) + (f_obj / s), 0.25);
267  /* calculate object temperature in Celsius */
268  *tobj = (t - 273.15);
269 }
