123 /*
124  * returns the corresponding base64 code for the given ascii symbol
125  */
126 static int getcode(char symbol)
127 {
128  if (symbol == '/') {
129  return BASE64_SLASH;
130  }
131 
132  if (symbol == '+') {
133  return BASE64_PLUS;
134  }
135 
136  if (symbol == '=') {
137  /* indicates a padded base64 end */
138  return BASE64_EQUALS;
139  }
140 
141  if (symbol < '0') {
142  /* indicates that the given symbol is not base64 and should be ignored */
143  return BASE64_NOT_DEFINED;
144  }
145 
146  if (symbol <= '9' && symbol >= '0') {
147  return (symbol + (BASE64_NUMBER_UPPER_BOUND - '9'));
148  }
149 
150  if (symbol <= 'Z' && symbol >= 'A') {
151  return (symbol - 'A');
152  }
153 
154  if (symbol <= 'z' && symbol >= 'a') {
155  return (symbol + (BASE64_SMALL_UPPER_BOUND - 'z'));
156  }
157 
158  /* indicates that the given symbol is not base64 and should be ignored */
159  return BASE64_NOT_DEFINED;
160 }

176  *data_out_size = required_size;

178  }
179 
180  if (data_out == NULL) {
181  return BASE64_ERROR_BUFFER_OUT;
182  }
183 
184  int iterate_data_buffer = 0;
185  unsigned char nNum = 0;
186  int nLst = getcode(base64_in[0]) << 2;
187  int code = 0;
188 
189  int mask = 2;
190 
191  for (int i = 1; i < (int)(base64_in_size); i++) {
192  code = getcode(base64_in[i]);
193 
194  if (code == BASE64_NOT_DEFINED || code == BASE64_EQUALS) {
195  continue;
196  }
197 
198  int nm = (0xFF << (2 * mask));
199 
200  nNum = nLst + ((code & (0xFF & nm)) >> (2 * mask));
201  nLst = (code & (0xFF & ~nm)) << (8 - (2 * mask));
202 
203  (mask != 3) ? data_out[iterate_data_buffer++] = nNum : nNum;
204  (mask == 0) ? mask = 3 : mask--;
205  }
206 
207  if (code == BASE64_EQUALS) {
208  /* add the last character to the data_out buffer */
209  data_out[iterate_data_buffer] = nNum;
210  }
211

213  return BASE64_SUCCESS;
