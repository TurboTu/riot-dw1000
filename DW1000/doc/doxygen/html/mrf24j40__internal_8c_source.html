39 void mrf24j40_init(mrf24j40_t *dev)
40 {
41 
42  mrf24j40_hardware_reset(dev);
43 
44 #if ENABLE_DEBUG
45  /* Check if MRF24J40 is available */
46  uint8_t txmcr = mrf24j40_reg_read_short(dev, MRF24J40_REG_TXMCR);
47  if ((txmcr == 0xFF) || (txmcr == 0x00)) {
48  /* Write default value to TXMCR register */
49  mrf24j40_reg_write_short(dev, MRF24J40_REG_TXMCR, MRF24J40_TXMCR_MACMINBE1 |
50  MRF24J40_TXMCR_MACMINBE0 |
51  MRF24J40_TXMCR_CSMABF2);
52  txmcr = mrf24j40_reg_read_short(dev, MRF24J40_REG_TXMCR);
53  if (txmcr != (MRF24J40_TXMCR_MACMINBE1 |
54  MRF24J40_TXMCR_MACMINBE0 |
55  MRF24J40_TXMCR_CSMABF2)) {
56  DEBUG("[mrf24j40] Initialization failure, SPI interface communication failed\n");
57  /* Return to prevents hangup later in the initialization */
58  return;
59  }
60  }
61 #endif
62 
63  /* do a soft reset */
64  mrf24j40_reg_write_short(dev, MRF24J40_REG_SOFTRST, MRF24J40_SOFTRST_RSTPWR |
65  MRF24J40_SOFTRST_RSTBB |
66  MRF24J40_SOFTRST_RSTMAC );
67 
68  /* flush RX FIFO */
69  mrf24j40_reg_write_short(dev, MRF24J40_REG_RXFLUSH, MRF24J40_RXFLUSH_RXFLUSH);
70 
71  /* Here starts init-process as described on MRF24J40 Manual Chap. 3.2 */
72  mrf24j40_reg_write_short(dev, MRF24J40_REG_PACON2, (MRF24J40_PACON2_TXONTS2 |
73  MRF24J40_PACON2_TXONTS1 |
74  MRF24J40_PACON2_FIFOEN));
75  mrf24j40_reg_write_short(dev, MRF24J40_REG_TXSTBL, (MRF24J40_TXSTBL_RFSTBL3 |
76  MRF24J40_TXSTBL_RFSTBL0 |
77  MRF24J40_TXSTBL_MSIFS2 |
78  MRF24J40_TXSTBL_MSIFS0));
79  mrf24j40_reg_write_long(dev, MRF24J40_REG_RFCON1, MRF24J40_RFCON1_VCOOPT1);
80  mrf24j40_reg_write_long(dev, MRF24J40_REG_RFCON2, MRF24J40_RFCON2_PLLEN);
81  mrf24j40_reg_write_long(dev, MRF24J40_REG_RFCON6, (MRF24J40_RFCON6_TXFIL |
82  MRF24J40_RFCON6_20MRECVR));
83  mrf24j40_reg_write_long(dev, MRF24J40_REG_RFCON7, MRF24J40_RFCON7_SLPCLKSEL1 );
84  mrf24j40_reg_write_long(dev, MRF24J40_REG_RFCON8, MRF24J40_RFCON8_RFVCO );
85  mrf24j40_reg_write_long(dev, MRF24J40_REG_SLPCON1, (MRF24J40_SLPCON1_CLKOUTEN |
86  MRF24J40_SLPCON1_SLPCLKDIV0));
87  mrf24j40_reg_write_short(dev, MRF24J40_REG_BBREG2, MRF25J40_BBREG2_CCAMODE1 );
88  mrf24j40_reg_write_short(dev, MRF24J40_REG_CCAEDTH, 0x60);
89  mrf24j40_reg_write_short(dev, MRF24J40_REG_BBREG6, MRF24J40_BBREG6_RSSIMODE2 );
90 
91  /* Enable immediate sleep mode */
92  mrf24j40_reg_write_short(dev, MRF24J40_REG_WAKECON, MRF24J40_WAKECON_IMMWAKE);
93 
94  /* set interrupt pin polarity, rising edge */
95  mrf24j40_reg_write_long(dev, MRF24J40_REG_SLPCON0, MRF24J40_SLPCON0_INTEDGE );
96  /* reset RF state machine */
97  mrf24j40_reset_state_machine(dev);
98 
99  /* clear interrupts */
100  mrf24j40_reg_read_short(dev, MRF24J40_REG_INTSTAT);
101 
102  /* mrf24j40_set_interrupts */
103  mrf24j40_reg_write_short(dev, MRF24J40_REG_INTCON, ~(MRF24J40_INTCON_RXIE | MRF24J40_INTCON_TXNIE));
104 }

106 uint8_t mrf24j40_reg_read_short(mrf24j40_t *dev, const uint8_t addr)
107 {
108  char value;
109 
110  getbus(dev);
111  value = spi_transfer_reg(SPIDEV, CSPIN, MRF24J40_SHORT_ADDR_TRANS |
112  (addr << MRF24J40_ADDR_OFFSET) |
113  MRF24J40_ACCESS_READ, 0);
114  spi_release(SPIDEV);
115  return (uint8_t)value;
116 }

127 uint8_t mrf24j40_reg_read_long(mrf24j40_t *dev, const uint16_t addr)
128 {
129  uint8_t reg1, reg2;
130 
131  reg1 = MRF24J40_LONG_ADDR_TRANS | (addr >> 3);
132  reg2 = (addr << 5) | MRF24J40_ACCESS_READ;
133  char value;
134  getbus(dev);
135  spi_transfer_byte(SPIDEV, CSPIN, true, reg1);
136  spi_transfer_byte(SPIDEV, CSPIN, true, reg2);
137  value = spi_transfer_byte(SPIDEV, CSPIN, false, 0);
138  spi_release(SPIDEV);
139 
140  return (uint8_t)value;
141 }

220 void mrf24j40_update_tasks(mrf24j40_t *dev)
221 {
222  if (dev->irq_flag) {
223  uint8_t newpending = 0;
224  uint8_t instat = 0;
225 
226  dev->irq_flag = 0;
227  instat = mrf24j40_reg_read_short(dev, MRF24J40_REG_INTSTAT);
228  /* check if TX done */
229  if (instat & MRF24J40_INTSTAT_TXNIF) {
230  newpending |= MRF24J40_TASK_TX_DONE | MRF24J40_TASK_TX_READY;
231  /* transmit done, returning to configured idle state */
232  mrf24j40_assert_sleep(dev);
233  }
234  if (instat & MRF24J40_INTSTAT_RXIF) {
235  newpending |= MRF24J40_TASK_RX_READY;
236  }
237  /* check if RX pending */
238  dev->pending |= newpending;
239  }
240 }

243 void mrf24j40_hardware_reset(mrf24j40_t *dev)
244 {
245  /* wake up from sleep in case radio is sleeping */
246  mrf24j40_assert_awake(dev);
247 
248  /* trigger hardware reset */
249  gpio_clear(dev->params.reset_pin);
250  /* Datasheet - Not specified */
251  xtimer_usleep(MRF24J40_RESET_PULSE_WIDTH);
252  gpio_set(dev->params.reset_pin);
253  /* Datasheet - MRF24J40 ~2ms */
254  xtimer_usleep(MRF24J40_RESET_DELAY);
255 }
