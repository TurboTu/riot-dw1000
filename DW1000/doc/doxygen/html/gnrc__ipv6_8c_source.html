109 void gnrc_ipv6_demux(kernel_pid_t iface, gnrc_pktsnip_t *current, gnrc_pktsnip_t *pkt, uint8_t nh)
110 {
111  bool interested = false;
112 
113  current->type = gnrc_nettype_from_protnum(nh);
114 
115  switch (nh) {
116 #ifdef MODULE_GNRC_ICMPV6
117  case PROTNUM_ICMPV6:
118  assert(current == pkt);
119  interested = true;
120  break;
121 #endif
122 #ifdef MODULE_GNRC_IPV6_EXT
123  case PROTNUM_IPV6_EXT_HOPOPT:
124  case PROTNUM_IPV6_EXT_DST:
125  case PROTNUM_IPV6_EXT_RH:
126  case PROTNUM_IPV6_EXT_FRAG:
127  case PROTNUM_IPV6_EXT_AH:
128  case PROTNUM_IPV6_EXT_ESP:
129  case PROTNUM_IPV6_EXT_MOB:
130  interested = true;
131 
132  break;
133 #endif
134  case PROTNUM_IPV6:
135  assert(current == pkt);
136  interested = true;
137 
138  break;
139  default:
140  (void)iface;
141 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC_NHC
142  /* second statement is true for small 6LoWPAN NHC decompressed frames
143  * since in this case it looks like
144  *
145  * * GNRC_NETTYPE_UNDEF <- pkt
146  * v
147  * * GNRC_NETTYPE_UDP <- current
148  * v
149  * * GNRC_NETTYPE_EXT
150  * v
151  * * GNRC_NETTYPE_IPV6
152  */
153  assert((current == pkt) || (current == pkt->next));
154 #else
155  assert(current == pkt);
156 #endif
157  break;
158  }
159 
160  _dispatch_next_header(current, pkt, nh, interested);
161 
162  if (!interested) {
163  return;
164  }
165 
166  switch (nh) {
167 #ifdef MODULE_GNRC_ICMPV6
168  case PROTNUM_ICMPV6:
169  DEBUG("ipv6: handle ICMPv6 packet (nh = %u)\n", nh);
170  gnrc_icmpv6_demux(iface, pkt);
171  return;
172 #endif
173 #ifdef MODULE_GNRC_IPV6_EXT
174  case PROTNUM_IPV6_EXT_HOPOPT:
175  case PROTNUM_IPV6_EXT_DST:
176  case PROTNUM_IPV6_EXT_RH:
177  case PROTNUM_IPV6_EXT_FRAG:
178  case PROTNUM_IPV6_EXT_AH:
179  case PROTNUM_IPV6_EXT_ESP:
180  case PROTNUM_IPV6_EXT_MOB:
181  DEBUG("ipv6: handle extension header (nh = %u)\n", nh);
182 
183  gnrc_ipv6_ext_demux(iface, current, pkt, nh);
184 
185  return;
186 #endif
187  case PROTNUM_IPV6:
188  DEBUG("ipv6: handle encapsulated IPv6 packet (nh = %u)\n", nh);
189  _decapsulate(pkt);
190  return;
191  default:
192  assert(false);
193  break;
194  }

196  assert(false);

211 static void _dispatch_next_header(gnrc_pktsnip_t *current, gnrc_pktsnip_t *pkt,
212  uint8_t nh, bool interested)
213 {
214 #ifdef MODULE_GNRC_IPV6_EXT
215  const bool should_dispatch_current_type = ((current->type != GNRC_NETTYPE_IPV6_EXT) ||
216  (current->next->type == GNRC_NETTYPE_IPV6));
217 #else
218  const bool should_dispatch_current_type = (current->next->type == GNRC_NETTYPE_IPV6);
219 #endif
220 
221  DEBUG("ipv6: forward nh = %u to other threads\n", nh);
222 
223  /* dispatch IPv6 extension header only once */
224  if (should_dispatch_current_type) {
225  bool should_release = (gnrc_netreg_num(GNRC_NETTYPE_IPV6, nh) == 0) &&
226  (!interested);
227 
228  if (!should_release) {
229  gnrc_pktbuf_hold(pkt, 1); /* don't remove from packet buffer in
230  * next dispatch */
231  }
232  if (gnrc_netapi_dispatch_receive(current->type,
233  GNRC_NETREG_DEMUX_CTX_ALL,
234  pkt) == 0) {
235  gnrc_pktbuf_release(pkt);
236  }
237 
238  if (should_release) {
239  return;
240  }
241  }
242  if (interested) {
243  gnrc_pktbuf_hold(pkt, 1); /* don't remove from packet buffer in
244  * next dispatch */
245  }
246  if (gnrc_netapi_dispatch_receive(GNRC_NETTYPE_IPV6, nh, pkt) == 0) {
247  gnrc_pktbuf_release(pkt);
248  }
249 }

255  sched_active_pid);

259 
260  /* register interest in all IPv6 packets */
261  gnrc_netreg_register(GNRC_NETTYPE_IPV6, &me_reg);
262 
263  /* preinitialize ACK */
264  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
265

372 static void _send_to_iface(kernel_pid_t iface, gnrc_pktsnip_t *pkt)
373 {
374  ((gnrc_netif_hdr_t *)pkt->data)->if_pid = iface;
375  gnrc_ipv6_netif_t *if_entry = gnrc_ipv6_netif_get(iface);
376 
377  assert(if_entry != NULL);
378  if (gnrc_pkt_len(pkt->next) > if_entry->mtu) {
379  DEBUG("ipv6: packet too big\n");
380  gnrc_pktbuf_release(pkt);
381  return;
382  }
383 #ifdef MODULE_NETSTATS_IPV6
384  if_entry->stats.tx_success++;
385  if_entry->stats.tx_bytes += gnrc_pkt_len(pkt->next);
386 #endif
387 
388 #ifdef MODULE_GNRC_SIXLOWPAN
389  if (if_entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) {
390  DEBUG("ipv6: send to 6LoWPAN instead\n");
391  if (!gnrc_netapi_dispatch_send(GNRC_NETTYPE_SIXLOWPAN, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
392  DEBUG("ipv6: no 6LoWPAN thread found\n");
393  gnrc_pktbuf_release(pkt);
394  }
395  return;
396  }
397 #endif
398  if (gnrc_netapi_send(iface, pkt) < 1) {
399  DEBUG("ipv6: unable to send packet\n");
400  gnrc_pktbuf_release(pkt);
401  }

404 static gnrc_pktsnip_t *_create_netif_hdr(uint8_t *dst_l2addr,
405  uint16_t dst_l2addr_len,
406  gnrc_pktsnip_t *pkt)
407 {
408  gnrc_pktsnip_t *netif = gnrc_netif_hdr_build(NULL, 0, dst_l2addr, dst_l2addr_len);
409 
410  if (netif == NULL) {
411  DEBUG("ipv6: error on interface header allocation, dropping packet\n");
412  gnrc_pktbuf_release(pkt);
413  return NULL;
414  }
415 
416  if (pkt->type == GNRC_NETTYPE_NETIF) {
417  /* remove old netif header, since checking it for correctness would
418  * cause to much overhead.
419  * netif header might have been allocated by some higher layer either
420  * to set a sending interface or some flags. Interface was already
421  * copied using iface parameter, so we only need to copy the flags
422  * (minus the broadcast/multicast flags) */
423  DEBUG("ipv6: copy old interface header flags\n");
424  gnrc_netif_hdr_t *netif_new = netif->data, *netif_old = pkt->data;
425  netif_new->flags = netif_old->flags & \
426  ~(GNRC_NETIF_HDR_FLAGS_BROADCAST | GNRC_NETIF_HDR_FLAGS_MULTICAST);
427  DEBUG("ipv6: removed old interface header\n");
428  pkt = gnrc_pktbuf_remove_snip(pkt, pkt);
429  }
430 
431  /* add netif to front of the pkt list */
432  LL_PREPEND(pkt, netif);
433 
434  return pkt;
435 }

453 static int _fill_ipv6_hdr(kernel_pid_t iface, gnrc_pktsnip_t *ipv6,
454  gnrc_pktsnip_t *payload)
455 {
456  int res;
457  ipv6_hdr_t *hdr = ipv6->data;
458 
459  hdr->len = byteorder_htons(gnrc_pkt_len(payload));
460  DEBUG("ipv6: set payload length to %u (network byteorder %04" PRIx16 ")\n",
461  (unsigned) gnrc_pkt_len(payload), hdr->len.u16);
462 
463  /* check if e.g. extension header was not already marked */
464  if (hdr->nh == PROTNUM_RESERVED) {
465  hdr->nh = gnrc_nettype_to_protnum(payload->type);
466 
467  /* if still reserved: mark no next header */
468  if (hdr->nh == PROTNUM_RESERVED) {
469  hdr->nh = PROTNUM_IPV6_NONXT;
470  }
471  }
472 
473  DEBUG("ipv6: set next header to %u\n", hdr->nh);
474 
475  if (hdr->hl == 0) {
476  if (iface == KERNEL_PID_UNDEF) {
477  hdr->hl = GNRC_IPV6_NETIF_DEFAULT_HL;
478  }
479  else {
480  hdr->hl = gnrc_ipv6_netif_get(iface)->cur_hl;
481  }
482  }
483 
484  if (ipv6_addr_is_unspecified(&hdr->src)) {
485  if (ipv6_addr_is_loopback(&hdr->dst)) {
486  ipv6_addr_set_loopback(&hdr->src);
487  }
488  else {
489  ipv6_addr_t *src = gnrc_ipv6_netif_find_best_src_addr(iface, &hdr->dst, false);
490 
491  if (src != NULL) {
492  DEBUG("ipv6: set packet source to %s\n",
493  ipv6_addr_to_str(addr_str, src, sizeof(addr_str)));
494  memcpy(&hdr->src, src, sizeof(ipv6_addr_t));
495  }
496  /* Otherwise leave unspecified */
497  }
498  }
499 
500  DEBUG("ipv6: calculate checksum for upper header.\n");
501 
502  if ((res = gnrc_netreg_calc_csum(payload, ipv6)) < 0) {
503  if (res != -ENOENT) { /* if there is no checksum we are okay */
504  DEBUG("ipv6: checksum calculation failed.\n");
505  return res;
506  }
507  }
508 
509  return 0;
510 }

524 static void _send_multicast(kernel_pid_t iface, gnrc_pktsnip_t *pkt,
525  gnrc_pktsnip_t *ipv6, gnrc_pktsnip_t *payload,
526  bool prep_hdr)
527 {
528  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
529  size_t ifnum = 0;
530 
531  if (iface == KERNEL_PID_UNDEF) {
532  /* get list of interfaces */
533  ifnum = gnrc_netif_get(ifs);
534 
535  /* throw away packet if no one is interested */
536  if (ifnum == 0) {
537  DEBUG("ipv6: no interfaces registered, dropping packet\n");
538  gnrc_pktbuf_release(pkt);
539  return;
540  }
541  }
542 
543 
544 #if GNRC_NETIF_NUMOF > 1
545  /* interface not given: send over all interfaces */
546  if (iface == KERNEL_PID_UNDEF) {
547  /* send packet to link layer */
548  gnrc_pktbuf_hold(pkt, ifnum - 1);
549 
550  for (size_t i = 0; i < ifnum; i++) {
551  if (prep_hdr) {
552  /* need to get second write access (duplication) to fill IPv6
553  * header interface-local */
554  gnrc_pktsnip_t *tmp = gnrc_pktbuf_start_write(pkt);
555  gnrc_pktsnip_t *ptr = tmp->next;
556  ipv6 = tmp;
557 
558  if (ipv6 == NULL) {
559  DEBUG("ipv6: unable to get write access to IPv6 header, "
560  "for interface %" PRIkernel_pid "\n", ifs[i]);
561  gnrc_pktbuf_release(pkt);
562  return;
563  }
564 
565  /* multiple interfaces => possibly different source addresses
566  * => different checksums => duplication of payload needed */
567  while (ptr != payload->next) {
568  /* duplicate everything including payload */
569  tmp->next = gnrc_pktbuf_start_write(ptr);
570  if (tmp->next == NULL) {
571  DEBUG("ipv6: unable to get write access to payload, drop it\n");
572  gnrc_pktbuf_release(ipv6);
573  return;
574  }
575  tmp = tmp->next;
576  ptr = ptr->next;
577  }
578 
579  if (_fill_ipv6_hdr(ifs[i], ipv6, tmp) < 0) {
580  /* error on filling up header */
581  gnrc_pktbuf_release(ipv6);
582  return;
583  }
584  }
585 
586  if ((ipv6 = _create_netif_hdr(NULL, 0, ipv6)) == NULL) {
587  return;
588  }
589 
590  _send_multicast_over_iface(ifs[i], ipv6);
591  }
592  }
593  else {
594  if (prep_hdr) {
595  if (_fill_ipv6_hdr(iface, ipv6, payload) < 0) {
596  /* error on filling up header */
597  gnrc_pktbuf_release(pkt);
598  return;
599  }

657 static void _send(gnrc_pktsnip_t *pkt, bool prep_hdr)
658 {
659  kernel_pid_t iface = KERNEL_PID_UNDEF;
660  gnrc_pktsnip_t *ipv6, *payload;
661  ipv6_addr_t *tmp;
662  ipv6_hdr_t *hdr;
663  /* get IPv6 snip and (if present) generic interface header */
664  if (pkt->type == GNRC_NETTYPE_NETIF) {
665  /* If there is already a netif header (routing protocols and
666  * neighbor discovery might add them to preset sending interface) */
667  iface = ((gnrc_netif_hdr_t *)pkt->data)->if_pid;
668  /* seize payload as temporary variable */
669  ipv6 = gnrc_pktbuf_start_write(pkt); /* write protect for later removal
670  * in _send_unicast() */
671  if (ipv6 == NULL) {
672  DEBUG("ipv6: unable to get write access to netif header, dropping packet\n");
673  gnrc_pktbuf_release(pkt);
674  return;
675  }
676  pkt = ipv6; /* Reset pkt from temporary variable */
677 
678  ipv6 = pkt->next;
679  }
680  else {
681  ipv6 = pkt;
682  }
683  /* seize payload as temporary variable */
684  payload = gnrc_pktbuf_start_write(ipv6);
685  if (payload == NULL) {
686  DEBUG("ipv6: unable to get write access to IPv6 header, dropping packet\n");
687  gnrc_pktbuf_release(pkt);
688  return;
689  }
690  if (ipv6 != pkt) { /* in case packet has netif header */
691  pkt->next = payload;/* pkt is already write-protected so we can do that */
692  }
693  else {
694  pkt = payload; /* pkt is the IPv6 header so we just write-protected it */
695  }
696  ipv6 = payload; /* Reset ipv6 from temporary variable */
697 
698  hdr = ipv6->data;
699  payload = ipv6->next;
700 
701  if (ipv6_addr_is_multicast(&hdr->dst)) {
702  _send_multicast(iface, pkt, ipv6, payload, prep_hdr);
703  }
704  else if ((ipv6_addr_is_loopback(&hdr->dst)) || /* dst is loopback address */
705  ((iface == KERNEL_PID_UNDEF) && /* or dst registered to any local interface */
706  ((iface = gnrc_ipv6_netif_find_by_addr(&tmp, &hdr->dst)) != KERNEL_PID_UNDEF)) ||
707  ((iface != KERNEL_PID_UNDEF) && /* or dst registered to given interface */
708  (gnrc_ipv6_netif_find_addr(iface, &hdr->dst) != NULL))) {
709  uint8_t *rcv_data;
710  gnrc_pktsnip_t *ptr = ipv6, *rcv_pkt;
711 
712  if (prep_hdr) {
713  if (_fill_ipv6_hdr(iface, ipv6, payload) < 0) {
714  /* error on filling up header */
715  gnrc_pktbuf_release(pkt);
716  return;
717  }
718  }
719 
720  rcv_pkt = gnrc_pktbuf_add(NULL, NULL, gnrc_pkt_len(ipv6), GNRC_NETTYPE_IPV6);
721 
722  if (rcv_pkt == NULL) {
723  DEBUG("ipv6: error on generating loopback packet\n");
724  gnrc_pktbuf_release(pkt);
725  return;
726  }
727 
728  rcv_data = rcv_pkt->data;
729 
730  /* "reverse" packet (by making it one snip as if received from NIC) */
731  while (ptr != NULL) {
732  memcpy(rcv_data, ptr->data, ptr->size);
733  rcv_data += ptr->size;
734  ptr = ptr->next;
735  }
736 
737  gnrc_pktbuf_release(pkt);
738 
739  DEBUG("ipv6: packet is addressed to myself => loopback\n");
740 
741  if (gnrc_netapi_receive(gnrc_ipv6_pid, rcv_pkt) < 1) {
742  DEBUG("ipv6: unable to deliver packet\n");
743  gnrc_pktbuf_release(rcv_pkt);
744  }
745  }
746  else {
747  uint8_t l2addr_len = GNRC_IPV6_NC_L2_ADDR_MAX;
748  uint8_t l2addr[l2addr_len];
749 
750  iface = _next_hop_l2addr(l2addr, &l2addr_len, iface, &hdr->dst, pkt);
751 
752  if (iface == KERNEL_PID_UNDEF) {
753  DEBUG("ipv6: error determining next hop's link layer address\n");
754  gnrc_pktbuf_release(pkt);
755  return;
756  }
757 
758  if (prep_hdr) {
759  if (_fill_ipv6_hdr(iface, ipv6, payload) < 0) {
760  /* error on filling up header */
761  gnrc_pktbuf_release(pkt);
762  return;
763  }
764  }
765 
766  _send_unicast(iface, l2addr, l2addr_len, pkt);
767  }
768 }

790 static void _receive(gnrc_pktsnip_t *pkt)
791 {
792  kernel_pid_t iface = KERNEL_PID_UNDEF;
793  gnrc_pktsnip_t *ipv6, *netif, *first_ext;
794  ipv6_hdr_t *hdr;
795 
796  assert(pkt != NULL);
797 
798  netif = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_NETIF);
799 
800  if (netif != NULL) {
801  iface = ((gnrc_netif_hdr_t *)netif->data)->if_pid;
802 
803 #ifdef MODULE_NETSTATS_IPV6
804  assert(iface);
805  netstats_t *stats = gnrc_ipv6_netif_get_stats(iface);
806  stats->rx_count++;
807  stats->rx_bytes += (gnrc_pkt_len(pkt) - netif->size);
808 #endif
809  }
810 
811  first_ext = pkt;
812 
813  for (ipv6 = pkt; ipv6 != NULL; ipv6 = ipv6->next) { /* find IPv6 header if already marked */
814  if ((ipv6->type == GNRC_NETTYPE_IPV6) && (ipv6->size == sizeof(ipv6_hdr_t)) &&
815  (ipv6_hdr_is(ipv6->data))) {
816  break;
817  }
818 
819  first_ext = ipv6;
820  }
821 
822  if (ipv6 == NULL) {
823  if (!ipv6_hdr_is(pkt->data)) {
824  DEBUG("ipv6: Received packet was not IPv6, dropping packet\n");
825  gnrc_pktbuf_release(pkt);
826  return;
827  }
828 #ifdef MODULE_GNRC_IPV6_WHITELIST
829  if (!gnrc_ipv6_whitelisted(&((ipv6_hdr_t *)(pkt->data))->src)) {
830  DEBUG("ipv6: Source address not whitelisted, dropping packet\n");
831  gnrc_pktbuf_release(pkt);
832  return;
833  }
834 #endif
835 #ifdef MODULE_GNRC_IPV6_BLACKLIST
836  if (gnrc_ipv6_blacklisted(&((ipv6_hdr_t *)(pkt->data))->src)) {
837  DEBUG("ipv6: Source address blacklisted, dropping packet\n");
838  gnrc_pktbuf_release(pkt);
839  return;
840  }
841 #endif
842  /* seize ipv6 as a temporary variable */
843  ipv6 = gnrc_pktbuf_start_write(pkt);
844 
845  if (ipv6 == NULL) {
846  DEBUG("ipv6: unable to get write access to packet, drop it\n");
847  gnrc_pktbuf_release(pkt);
848  return;
849  }
850 
851  pkt = ipv6; /* reset pkt from temporary variable */
852 
853  ipv6 = gnrc_pktbuf_mark(pkt, sizeof(ipv6_hdr_t), GNRC_NETTYPE_IPV6);
854 
855  first_ext = pkt;
856  pkt->type = GNRC_NETTYPE_UNDEF; /* snip is no longer IPv6 */
857 
858  if (ipv6 == NULL) {
859  DEBUG("ipv6: error marking IPv6 header, dropping packet\n");
860  gnrc_pktbuf_release(pkt);
861  return;
862  }
863  }
864 #ifdef MODULE_GNRC_IPV6_WHITELIST
865  else if (!gnrc_ipv6_whitelisted(&((ipv6_hdr_t *)(ipv6->data))->src)) {
866  /* if ipv6 header already marked*/
867  DEBUG("ipv6: Source address not whitelisted, dropping packet\n");
868  gnrc_pktbuf_release(pkt);
869  return;
870  }
871 #endif
872 #ifdef MODULE_GNRC_IPV6_BLACKLIST
873  else if (gnrc_ipv6_blacklisted(&((ipv6_hdr_t *)(ipv6->data))->src)) {
874  /* if ipv6 header already marked*/
875  DEBUG("ipv6: Source address blacklisted, dropping packet\n");
876  gnrc_pktbuf_release(pkt);
877  return;
878  }
879 #endif
880 
881  /* extract header */
882  hdr = (ipv6_hdr_t *)ipv6->data;
883 
884  /* if available, remove any padding that was added by lower layers
885  * to fulfill their minimum size requirements (e.g. ethernet) */
886  if (byteorder_ntohs(hdr->len) < pkt->size) {
887  gnrc_pktbuf_realloc_data(pkt, byteorder_ntohs(hdr->len));
888  }
889  else if (byteorder_ntohs(hdr->len) >
890  (gnrc_pkt_len_upto(pkt, GNRC_NETTYPE_IPV6) - sizeof(ipv6_hdr_t))) {
891  DEBUG("ipv6: invalid payload length: %d, actual: %d, dropping packet\n",
892  (int) byteorder_ntohs(hdr->len),
893  (int) (gnrc_pkt_len_upto(pkt, GNRC_NETTYPE_IPV6) - sizeof(ipv6_hdr_t)));
894  gnrc_pktbuf_release(pkt);
895  return;
896  }
897 
898  DEBUG("ipv6: Received (src = %s, ",
899  ipv6_addr_to_str(addr_str, &(hdr->src), sizeof(addr_str)));
900  DEBUG("dst = %s, next header = %u, length = %" PRIu16 ")\n",
901  ipv6_addr_to_str(addr_str, &(hdr->dst), sizeof(addr_str)),
902  hdr->nh, byteorder_ntohs(hdr->len));
903 
904  if (_pkt_not_for_me(&iface, hdr)) { /* if packet is not for me */
905  DEBUG("ipv6: packet destination not this host\n");
906 
907 #ifdef MODULE_GNRC_IPV6_ROUTER /* only routers redirect */
908  /* redirect to next hop */
909  DEBUG("ipv6: decrement hop limit to %u\n", (uint8_t) (hdr->hl - 1));
910 
911  /* RFC 4291, section 2.5.6 states: "Routers must not forward any
912  * packets with Link-Local source or destination addresses to other
913  * links."
914  */
915  if ((ipv6_addr_is_link_local(&(hdr->src))) || (ipv6_addr_is_link_local(&(hdr->dst)))) {
916  DEBUG("ipv6: do not forward packets with link-local source or"
917  " destination address\n");
918  gnrc_pktbuf_release(pkt);
919  return;
920  }
921  /* TODO: check if receiving interface is router */
922  else if (--(hdr->hl) > 0) { /* drop packets that *reach* Hop Limit 0 */
923  gnrc_pktsnip_t *reversed_pkt = NULL, *ptr = pkt;
924 
925  DEBUG("ipv6: forward packet to next hop\n");
926 
927  /* pkt might not be writable yet, if header was given above */
928  ipv6 = gnrc_pktbuf_start_write(ipv6);
929  if (ipv6 == NULL) {
930  DEBUG("ipv6: unable to get write access to packet: dropping it\n");
931  gnrc_pktbuf_release(pkt);
932  return;
933  }
934 
935  /* remove L2 headers around IPV6 */
936  netif = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_NETIF);
937  if (netif != NULL) {
938  gnrc_pktbuf_remove_snip(pkt, netif);
939  }
940 
941  /* reverse packet snip list order */
942  while (ptr != NULL) {
943  gnrc_pktsnip_t *next;
944  ptr = gnrc_pktbuf_start_write(ptr); /* duplicate if not already done */
945  if (ptr == NULL) {
946  DEBUG("ipv6: unable to get write access to packet: dropping it\n");
947  gnrc_pktbuf_release(reversed_pkt);
948  gnrc_pktbuf_release(pkt);
949  return;
950  }
951  next = ptr->next;
952  ptr->next = reversed_pkt;
953  reversed_pkt = ptr;
954  ptr = next;
955  }
956  _send(reversed_pkt, false);
957  return;
958  }
959  else {
960  DEBUG("ipv6: hop limit reached 0: drop packet\n");
961  gnrc_pktbuf_release(pkt);
962  return;
963  }
964 
965 #else /* MODULE_GNRC_IPV6_ROUTER */
966  DEBUG("ipv6: dropping packet\n");
967  /* non rounting hosts just drop the packet */
968  gnrc_pktbuf_release(pkt);
969  return;
970 #endif /* MODULE_GNRC_IPV6_ROUTER */
971  }
972 
973  /* IPv6 internal demuxing (ICMPv6, Extension headers etc.) */
974  gnrc_ipv6_demux(iface, first_ext, pkt, hdr->nh);
975 }

978 {

980
