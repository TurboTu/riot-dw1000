185 static int _read(int argc, char **argv)
186 {
187  int blockaddr;
188  int cnt;
189  bool print_as_char = false;
190 
191  if ((argc == 3) || (argc == 4)) {
192  blockaddr = (uint32_t)atoi(argv[1]);
193  cnt = (uint32_t)atoi(argv[2]);
194  if (argc == 4 && (strcmp("-c", argv[3]) == 0)) {
195  print_as_char = true;
196  }
197  }
198  else {
199  printf("usage: %s blockaddr cnt [-c]\n", argv[0]);
200  return -1;
201  }
202 
203  int total_read = 0;
204  while (total_read < cnt) {
205  int chunk_blocks = cnt - total_read;
206  if (chunk_blocks > MAX_BLOCKS_IN_BUFFER) {
207  chunk_blocks = MAX_BLOCKS_IN_BUFFER;
208  }
209  sd_rw_response_t state;
210  int chunks_read = sdcard_spi_read_blocks(card, blockaddr + total_read, buffer,
211  SD_HC_BLOCK_SIZE, chunk_blocks, &state);
212 
213  if (state != SD_RW_OK) {
214  printf("read error %d (block %d/%d)\n", state, total_read + chunks_read, cnt);
215  return -1;
216  }
217 
218  for (int i = 0; i < chunk_blocks * SD_HC_BLOCK_SIZE; i++) {
219 
220  if ((i % SD_HC_BLOCK_SIZE) == 0) {
221  printf("BLOCK %d:\n", blockaddr + total_read + i / SD_HC_BLOCK_SIZE);
222  }
223 
224  if (print_as_char) {
225  if (buffer[i] >= FIRST_PRINTABLE_ASCII_CHAR) {
226  printf("%c", buffer[i]);
227  }
228  else {
229  printf(ASCII_UNPRINTABLE_REPLACEMENT);
230  }
231  }
232  else {
233  printf("%02x ", buffer[i]);
234  }
235 
236  if ((i % BLOCK_PRINT_BYTES_PER_LINE) == (BLOCK_PRINT_BYTES_PER_LINE - 1)) {
237  puts(""); /* line break after BLOCK_PRINT_BYTES_PER_LINE bytes */
238  }
239 
240  if ((i % SD_HC_BLOCK_SIZE) == (SD_HC_BLOCK_SIZE - 1)) {
241  puts(""); /* empty line after each printed block */
242  }
243  }
244  total_read += chunks_read;
245  }
246  return 0;
247 }

249 static int _write(int argc, char **argv)
250 {
251  int bladdr;
252  char *data;
253  int size;
254  bool repeat_data = false;
255 
256  if (argc == 3 || argc == 4) {
257  bladdr = (int)atoi(argv[1]);
258  data = argv[2];
259  size = strlen(argv[2]);
260  printf("will write '%s' (%d chars) at start of block %d\n", data, size, bladdr);
261  if (argc == 4 && (strcmp("-r", argv[3]) == 0)) {
262  repeat_data = true;
263  puts("the rest of the block will be filled with copies of that string");
264  }
265  else {
266  puts("the rest of the block will be filled with zeros");
267  }
268  }
269  else {
270  printf("usage: %s blockaddr string [-r]\n", argv[0]);
271  return -1;
272  }
273 
274  if (size > SD_HC_BLOCK_SIZE) {
275  printf("maximum stringsize to write at once is %d ...aborting\n", SD_HC_BLOCK_SIZE);
276  return -1;
277  }
278 
279  /* copy data to a full-block-sized buffer an fill remaining block space according to -r param*/
280  char buffer[SD_HC_BLOCK_SIZE];
281  for (int i = 0; i < sizeof(buffer); i++) {
282  if (repeat_data || i < size) {
283  buffer[i] = data[i % size];
284  }
285  else {
286  buffer[i] = 0;
287  }
288  }
289 
290  sd_rw_response_t state;
291  int chunks_written = sdcard_spi_write_blocks(card, bladdr, buffer, SD_HC_BLOCK_SIZE, 1, &state);
292 
293  if (state != SD_RW_OK) {
294  printf("write error %d (wrote %d/%d blocks)\n", state, chunks_written, 1);
295  return -1;
296  }
297 
298  printf("write block %d [OK]\n", bladdr);
299  return 0;
300 }

329  return -2;
330  }
331 
332  printf("copy block %d to %d [OK]\n", src_block, dst_block);
