56 static void check_running_cpu_revision(void)
57 {
58  /* Check that the running CPU revision matches the compiled revision */
59  if (SCB->CPUID != K60_EXPECTED_CPUID) {
60  uint32_t CPUID = SCB->CPUID; /* This is only to ease debugging, type
61  * "print /x CPUID" in gdb */
62  uint32_t SILICON_REVISION = (SCB->CPUID & SCB_CPUID_REVISION_Msk) + 1;
63  (void)CPUID; /* prevents compiler warnings about an unused variable. */
64  (void)SILICON_REVISION;
65 
66  /* Running on the wrong CPU, the clock initialization is different
67  * between silicon revision 1.x and 2.x (LSB of CPUID) */
68  /* If you unexpectedly end up on this line when debugging:
69  * Rebuild the code using the correct value for K60_CPU_REV */
70  __asm__ volatile ("bkpt #99\n");
71 
72  while (1);
73  }
74 }

76 void SystemCoreClockUpdate(void)
77 {
78  /* Variable to store output clock frequency of the MCG module */
79  uint32_t MCGOUT_clock;
80 
81  if ((MCG->C1 & MCG_C1_CLKS_MASK) == 0x0u) {
82  /* Output of FLL or PLL is selected */
83  if ((MCG->C6 & MCG_C6_PLLS_MASK) == 0x0u) {
84  /* FLL is selected */
85  if ((MCG->C1 & MCG_C1_IREFS_MASK) == 0x0u) {
86  /* External reference clock is selected */
87 #if K60_CPU_REV == 1
88  /* rev.1 silicon */
89  if ((SIM->SOPT2 & SIM_SOPT2_MCGCLKSEL_MASK) == 0x0u) {
90  /* System oscillator drives MCG clock */
91  MCGOUT_clock = CPU_XTAL_CLK_HZ;
92  }
93  else {
94  /* RTC 32 kHz oscillator drives MCG clock */
95  MCGOUT_clock = CPU_XTAL32k_CLK_HZ;
96  }
97 
98 #else /* K60_CPU_REV */
99 
100  /* rev.2 silicon */
101  if ((MCG->C7 & MCG_C7_OSCSEL_MASK) == 0x0u) {
102  /* System oscillator drives MCG clock */
103  MCGOUT_clock = CPU_XTAL_CLK_HZ;
104  }
105  else {
106  /* RTC 32 kHz oscillator drives MCG clock */
107  MCGOUT_clock = CPU_XTAL32k_CLK_HZ;
108  }
109 
110 #endif /* K60_CPU_REV */
111  uint8_t divider = (uint8_t)(1u << ((MCG->C1 & MCG_C1_FRDIV_MASK) >> MCG_C1_FRDIV_SHIFT));
112  /* Calculate the divided FLL reference clock */
113  MCGOUT_clock /= divider;
114 
115  if ((MCG->C2 & MCG_C2_RANGE0_MASK) != 0x0u) {
116  /* If high range is enabled, additional 32 divider is active */
117  MCGOUT_clock /= 32u;
118  }
119  }
120  else {
121  /* The slow internal reference clock is selected */
122  MCGOUT_clock = CPU_INT_SLOW_CLK_HZ;
123  }
124 
125  /* Select correct multiplier to calculate the MCG output clock */
126  switch (MCG->C4 & (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS_MASK)) {
127  case (0x0u):
128  MCGOUT_clock *= 640u;
129  break;
130 
131  case (MCG_C4_DRST_DRS(0b01)): /* 0x20u */
132  MCGOUT_clock *= 1280u;
133  break;
134 
135  case (MCG_C4_DRST_DRS(0b10)): /* 0x40u */
136  MCGOUT_clock *= 1920u;
137  break;
138 
139  case (MCG_C4_DRST_DRS(0b11)): /* 0x60u */
140  MCGOUT_clock *= 2560u;
141  break;
142 
143  case (MCG_C4_DMX32_MASK): /* 0x80u */
144  MCGOUT_clock *= 732u;
145  break;
146 
147  case (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS(0b01)): /* 0xA0u */
148  MCGOUT_clock *= 1464u;
149  break;
150 
151  case (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS(0b10)): /* 0xC0u */
152  MCGOUT_clock *= 2197u;
153  break;
154 
155  case (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS(0b11)): /* 0xE0u */
156  MCGOUT_clock *= 2929u;
157  break;
158 
159  default:
160  break;
161  }
162  }
163  else {
164  /* PLL is selected */
165  /* Calculate the PLL reference clock */
166  uint8_t divider = (1u + (MCG->C5 & MCG_C5_PRDIV0_MASK));
167  MCGOUT_clock = (uint32_t)(CPU_XTAL_CLK_HZ / divider);
168  /* Calculate the MCG output clock */
169  divider = ((MCG->C6 & MCG_C6_VDIV0_MASK) + 24u);
170  MCGOUT_clock *= divider;
171  }
172  }
173  else if ((MCG->C1 & MCG_C1_CLKS_MASK) == MCG_C1_CLKS(0b01)) { /* 0x40u */
174  /* Internal reference clock is selected */
175  if ((MCG->C2 & MCG_C2_IRCS_MASK) == 0x0u) {
176  /* Slow internal reference clock selected */
177  MCGOUT_clock = CPU_INT_SLOW_CLK_HZ;
178  }
179  else {
180  /* Fast internal reference clock selected */
181 #if K60_CPU_REV == 1
182  /* rev.1 silicon */
183  MCGOUT_clock = CPU_INT_FAST_CLK_HZ;
184 #else /* K60_CPU_REV */
185  /* rev.2 silicon */
186  MCGOUT_clock = CPU_INT_FAST_CLK_HZ /
187  (1 << ((MCG->SC & MCG_SC_FCRDIV_MASK) >> MCG_SC_FCRDIV_SHIFT));
188 #endif /* K60_CPU_REV */
189  }
190  }
191  else if ((MCG->C1 & MCG_C1_CLKS_MASK) == MCG_C1_CLKS(0b10)) { /* 0x80u */
192  /* External reference clock is selected */
193 #if K60_CPU_REV == 1
194  /* rev.1 silicon */
195  if ((SIM->SOPT2 & SIM_SOPT2_MCGCLKSEL_MASK) == 0x0u) {
196  /* System oscillator drives MCG clock */
197  MCGOUT_clock = CPU_XTAL_CLK_HZ;
198  }
199  else {
200  /* RTC 32 kHz oscillator drives MCG clock */
201  MCGOUT_clock = CPU_XTAL32k_CLK_HZ;
202  }
203 
204 #else /* K60_CPU_REV */
205 
206  /* rev.2 silicon */
207  if ((MCG->C7 & MCG_C7_OSCSEL_MASK) == 0x0u) {
208  /* System oscillator drives MCG clock */
209  MCGOUT_clock = CPU_XTAL_CLK_HZ;
210  }
211  else {
212  /* RTC 32 kHz oscillator drives MCG clock */
213  MCGOUT_clock = CPU_XTAL32k_CLK_HZ;
214  }
215 
216 #endif /* K60_CPU_REV */
217  }
218  else {
219  /* Reserved value */
220  return;
221  }
222 
223  /* Core clock and system clock use the same divider setting */
224  SystemCoreClock = SystemSysClock = (MCGOUT_clock / (1u + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV1_MASK)
225  >> SIM_CLKDIV1_OUTDIV1_SHIFT)));
226  SystemBusClock = (MCGOUT_clock / (1u + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV2_MASK) >>
227  SIM_CLKDIV1_OUTDIV2_SHIFT)));
228  SystemFlexBusClock = (MCGOUT_clock / (1u + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV3_MASK) >>
229  SIM_CLKDIV1_OUTDIV3_SHIFT)));
230  SystemFlashClock = (MCGOUT_clock / (1u + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV4_MASK) >>
231  SIM_CLKDIV1_OUTDIV4_SHIFT)));
232 
233  /* Module helper variables */
234  if (SystemBusClock >= 1000000) {
235  /* PIT module clock_delay_usec scale factor */
236  PIT_ticks_per_usec = (SystemBusClock + 500000) / 1000000; /* Rounded to nearest integer */
237  }
238  else {
239  /* less than 1 MHz clock frequency on the PIT module, round up. */
240  PIT_ticks_per_usec = 1;
241  }
242 }
