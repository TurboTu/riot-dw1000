42 gnrc_sixlowpan_ctx_t *gnrc_sixlowpan_ctx_lookup_addr(const ipv6_addr_t *addr)
43 {
44  uint8_t best = 0;
45  gnrc_sixlowpan_ctx_t *res = NULL;
46 
47  mutex_lock(&_ctx_mutex);
48 
49  for (unsigned int id = 0; id < GNRC_SIXLOWPAN_CTX_SIZE; id++) {
50  if (_valid(id)) {
51  uint8_t match = ipv6_addr_match_prefix(&_ctxs[id].prefix, addr);
52 
53  if ((_ctxs[id].prefix_len <= match) && (match > best)) {
54  best = match;
55  res = &(_ctxs[id]);
56  }
57  }
58  }
59 
60  mutex_unlock(&_ctx_mutex);
61 
62 #if ENABLE_DEBUG

64  DEBUG("6lo ctx: found context (%u, %s/%" PRIu8 ") ",

66  ipv6_addr_to_str(ipv6str, &res->prefix, sizeof(ipv6str)),
67  res->prefix_len);

96 gnrc_sixlowpan_ctx_t *gnrc_sixlowpan_ctx_update(uint8_t id, const ipv6_addr_t *prefix,
97  uint8_t prefix_len, uint16_t ltime,
98  bool comp)
99 {
100  if ((id >= GNRC_SIXLOWPAN_CTX_SIZE) || (prefix_len == 0)) {
101  return NULL;
102  }
103 
104  mutex_lock(&_ctx_mutex);
105 
106  _ctxs[id].ltime = ltime;
107 
108  if (ltime == 0) {
109  comp = false;
110  }
111 
112  if (prefix_len > IPV6_ADDR_BIT_LEN) {
113  _ctxs[id].prefix_len = IPV6_ADDR_BIT_LEN;
114  }
115  else {
116  _ctxs[id].prefix_len = prefix_len;
117  }
118 
119  _ctxs[id].flags_id = (comp) ? (GNRC_SIXLOWPAN_CTX_FLAGS_COMP | id) : id;
120 
121  if (!ipv6_addr_equal(&(_ctxs[id].prefix), prefix)) {
122  ipv6_addr_set_unspecified(&(_ctxs[id].prefix));
123  ipv6_addr_init_prefix(&(_ctxs[id].prefix), prefix, _ctxs[id].prefix_len);
124  }
125  DEBUG("6lo ctx: update context (%u, %s/%" PRIu8 "), lifetime: %" PRIu16 " min\n",
126  id, ipv6_addr_to_str(ipv6str, &_ctxs[id].prefix, sizeof(ipv6str)),
127  _ctxs[id].prefix_len, _ctxs[id].ltime);
128  _ctx_inval_times[id] = ltime + _current_minute();
129 
130  mutex_unlock(&_ctx_mutex);
131  return &(_ctxs[id]);
132 }

139 static void _update_lifetime(uint8_t id)
140 {
141  uint32_t now;
142 
143  if (_ctxs[id].ltime == 0) {
144  _ctxs[id].flags_id &= ~GNRC_SIXLOWPAN_CTX_FLAGS_COMP;
145  return;
146  }
147 
148  now = _current_minute();
149 
150  if (now >= _ctx_inval_times[id]) {
151  DEBUG("6lo ctx: context %u was invalidated for compression\n", id);
152  _ctxs[id].ltime = 0;
153  _ctxs[id].flags_id &= ~GNRC_SIXLOWPAN_CTX_FLAGS_COMP;
154  }
155  else {
156  _ctxs[id].ltime = (uint16_t)(_ctx_inval_times[id] - now);
157  }
158 }
