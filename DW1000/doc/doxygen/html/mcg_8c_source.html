141 static void kinetis_mcg_enable_osc(void)
142 {
143  if (KINETIS_MCG_ERC_RANGE == 1) {
144  /* select high frequency range and oscillator clock */
145  MCG->C2 = (uint8_t)(MCG_C2_RANGE0(1));
146  }
147  else if (KINETIS_MCG_ERC_RANGE == 2) {
148  /* select very high frequency range and osciallor clock */
149  MCG->C2 = (uint8_t)(MCG_C2_RANGE0(2));
150  }
151  else {
152  /* select low frequency range and osciallor clock */
153  MCG->C2 = (uint8_t)(MCG_C2_RANGE0(0));
154  }
155 
156  OSC0->CR = (uint8_t)(OSC_CR_ERCLKEN_MASK | OSC_CR_EREFSTEN_MASK
157  | (KINETIS_MCG_OSC_CLC & 0xf));
158 
159  /* Enable Oscillator */
160  if (KINETIS_MCG_ERC_OSCILLATOR) {
161  MCG->C2 |= (uint8_t)(MCG_C2_EREFS0_MASK);
162 
163  /* wait fo OSC initialization */
164  while ((MCG->S & MCG_S_OSCINIT0_MASK) == 0);
165  }
166 }

175 static void kinetis_mcg_set_fei(void)
176 {
177  kinetis_mcg_set_fll_factor(KINETIS_MCG_FLL_FACTOR_FEI);
178  /* enable and select slow internal reference clock */
179  MCG->C1 = (uint8_t)(MCG_C1_CLKS(0) | MCG_C1_IREFS_MASK);
180 
181  /* set to defaults */
182  MCG->C2 = (uint8_t)0;
183 
184  /* source of the FLL reference clock shall be internal reference clock */
185  while ((MCG->S & MCG_S_IREFST_MASK) == 0);
186 
187  /* Wait until output of the FLL is selected */
188  while (MCG->S & (MCG_S_CLKST_MASK));
189 
190  kinetis_mcg_disable_pll();
191  current_mode = KINETIS_MCG_FEI;
192 }

225 static void kinetis_mcg_set_fbi(void)
226 {
227  /* select IRC source */
228  if (KINETIS_MCG_USE_FAST_IRC) {
229  MCG->C2 = (uint8_t)(MCG_C2_IRCS_MASK);
230  }
231  else {
232  MCG->C2 = (uint8_t)(0);
233  }
234 
235  kinetis_mcg_set_fll_factor(KINETIS_MCG_FLL_FACTOR_FEI);
236  /* enable and select slow internal reference clock */
237  MCG->C1 = (uint8_t)(MCG_C1_CLKS(1) | MCG_C1_IREFS_MASK);
238 
239  /* Wait until output of IRC is selected */
240  while ((MCG->S & MCG_S_CLKST_MASK) != MCG_S_CLKST(1));
241

243  while ((MCG->S & MCG_S_IREFST_MASK) == 0);

312 static void kinetis_mcg_set_pbe(void)
313 {
314  /* select external reference clock and divide factor */
315  MCG->C1 = (uint8_t)(MCG_C1_CLKS(2) | MCG_C1_FRDIV(KINETIS_MCG_ERC_FRDIV));
316 
317  /* Wait until ERC is selected */
318  while ((MCG->S & MCG_S_CLKST_MASK) != MCG_S_CLKST(2));
319 
320  /* PLL is not disabled in bypass mode */
321  MCG->C2 &= ~(uint8_t)(MCG_C2_LP_MASK);
322 
323  /* set external reference devider */
324  MCG->C5 = (uint8_t)(MCG_C5_PRDIV0(KINETIS_MCG_PLL_PRDIV));
325 
326  /* set external reference devider */
327  MCG->C6 = (uint8_t)(MCG_C6_VDIV0(KINETIS_MCG_PLL_VDIV0));
328 
329  /* select PLL */
330  MCG->C6 |= (uint8_t)(MCG_C6_PLLS_MASK);
331 
332  /* Wait until the source of the PLLS clock is PLL */
333  while ((MCG->S & MCG_S_PLLST_MASK) == 0);
334 
335  /* Wait until PLL locked */
336  while ((MCG->S & MCG_S_LOCK0_MASK) == 0);
337 
338  current_mode = KINETIS_MCG_PBE;
339 }

359 int kinetis_mcg_set_mode(kinetis_mcg_mode_t mode)
360 {
361  if (mode > KINETIS_MCG_FEI) {
362  return -1;
363  }
364 
365  if (mcg_pm[current_mode] & (1 << mode)) {
366  if (mode == KINETIS_MCG_FEI) {
367  kinetis_mcg_set_fei();
368  }
369 
370  if (mode == KINETIS_MCG_FBI) {
371  kinetis_mcg_set_fbi();
372  }
373 
374  if (mode == KINETIS_MCG_FEE) {
375  kinetis_mcg_set_fee();
376  }
377 
378  if (mode == KINETIS_MCG_FBE) {
379  kinetis_mcg_set_fbe();
380  }
381 
382  if (mode == KINETIS_MCG_BLPI) {
383  kinetis_mcg_set_blpi();
384  }
385 
386  if (mode == KINETIS_MCG_BLPE) {
387  kinetis_mcg_set_blpe();
388  }
389 
390  if (mode == KINETIS_MCG_PBE) {
391  kinetis_mcg_set_pbe();
392  }
393 
394  if (mode == KINETIS_MCG_PEE) {
395  kinetis_mcg_set_pee();
396  }
397 
398  return 0;
399  }

402  case KINETIS_MCG_PEE:
403  /* cppcheck-suppress duplicateExpression */
404  if (!(KINETIS_MCG_USE_ERC || KINETIS_MCG_USE_PLL)) {
405  return -1;
406  }
407 
408  if (current_mode == KINETIS_MCG_FEI) {

410  kinetis_mcg_set_fbe();
411  kinetis_mcg_set_pbe();
412  kinetis_mcg_set_pee();
413  return 0;
414  }

416  if (current_mode == KINETIS_MCG_BLPE) {
417  /* set PBE -> PEE */
418  kinetis_mcg_set_pbe();
419  kinetis_mcg_set_pee();
420  return 0;
421  }
422 
423  break;
424

426  if (!KINETIS_MCG_USE_ERC) {
427  return -1;
428  }
429 
430  if (current_mode == KINETIS_MCG_PEE) {
431  /* set PBE -> BLPE */
432  kinetis_mcg_set_pbe();
433  kinetis_mcg_set_blpe();
434  return 0;
435  }
436 
437  if (current_mode == KINETIS_MCG_FEE) {
438  /* set FBE -> BLPE */
439  kinetis_mcg_set_fbe();
440  kinetis_mcg_set_blpe();
441  return 0;
442  }
443 
444  break;
445 
446  case KINETIS_MCG_BLPI:
447  if (current_mode == KINETIS_MCG_FEE) {
448  /* set FBI -> BLPI */
449  kinetis_mcg_set_fbi();
450  kinetis_mcg_set_blpi();
451  return 0;

462  /* set FBE -> FEE */

464  kinetis_mcg_set_fee();

472  return 0;
