140 int udp_sendto(uint8_t *buf, size_t len, uint8_t *dst, uint16_t dst_port, uhcp_iface_t iface)
141 {

144  dst_addr.sin6_family = AF_INET6;
145  memcpy(&dst_addr.sin6_addr, dst, 16);
146  dst_addr.sin6_port = htons(dst_port);
147  dst_addr.sin6_scope_id = iface;
148 
149  int fd = socket(AF_INET6, SOCK_DGRAM, 0);

151  perror("creating send socket");

153  }
154 
155  ssize_t res;

157  perror("udp_sendto(): sendto()");
158  }
159 
160  close(fd);
161 
162  return res;
163 }
