40 static ubjson_read_callback_result_t _ubjson_read_marker(ubjson_cookie_t *restrict cookie,
41  char *marker)
42 {
43  while (1) {
44  ssize_t bytes_read;
45 
46  *marker = cookie->marker;
47  if (!*marker) {
48  bytes_read = cookie->rw.read(cookie, marker, 1);
49  }
50  else {
51  cookie->marker = 0;
52  bytes_read = 1;
53  }
54 
55  if (bytes_read == 1) {
56  return UBJSON_OKAY;
57  }
58  else if (bytes_read != 0) {
59  return UBJSON_PREMATURELY_ENDED;
60  }
61  }
62 }

64 ssize_t ubjson_get_string(ubjson_cookie_t *restrict cookie, ssize_t length, void *dest_)
65 {
66  ssize_t total = 0;
67  char *dest = dest_;
68  while (total < length) {
69  ssize_t read = cookie->rw.read(cookie, dest, length - total);
70  if (read < 0) {
71  return read;
72  }
73  total += read;
74  dest += read;
75  }
76  return total;
77 }

79 ssize_t ubjson_get_i32(ubjson_cookie_t *restrict cookie, ssize_t content, int32_t *dest)
80 {
81  static const int8_t LENGHTS[] = { 1, 1, 2, 4 };
82 
83  union {
84  int8_t i8;
85  uint8_t u8;
86  network_uint16_t i16;
87  network_uint32_t i32;
88  } value;
89 
90  int type = content;
91 
92  ssize_t result = ubjson_get_string(cookie, LENGHTS[type], &value);
93  if (result > 0) {
94  switch (type) {
95  case UBJSON_INT32_INT8:
96  *dest = value.i8;
97  break;
98  case UBJSON_INT32_UINT8:
99  *dest = value.u8;
100  break;
101  case UBJSON_INT32_INT16:
102  *dest = (int16_t) byteorder_ntohs(value.i16);
103  break;
104  case UBJSON_INT32_INT32:
105  *dest = (int32_t) byteorder_ntohl(value.i32);
106  break;
107  default:
108  return -1;
109  }
110  }
111  return result;
112 }

124 static ubjson_read_callback_result_t _ubjson_read_length(ubjson_cookie_t *restrict cookie,
125  ssize_t *len)
126 {
127  ubjson_type_t type;
128  ssize_t content = 0;
129  ubjson_read_callback_result_t result = ubjson_peek_value(cookie, &type, &content);
130  if (result != UBJSON_OKAY) {
131  return result;
132  }
133 
134  int64_t len64 = -1;
135  ssize_t read;
136  if (type == UBJSON_TYPE_INT32) {
137  int32_t len32;
138  read = ubjson_get_i32(cookie, content, &len32);
139  if (read > 0) {
140  len64 = len32;
141  }
142  }
143  else if (type == UBJSON_TYPE_INT64) {
144  read = ubjson_get_i64(cookie, content, &len64);
145  }
146  else {
147  return UBJSON_INVALID_DATA;
148  }
149 
150  if (read <= 0) {
151  return UBJSON_PREMATURELY_ENDED;
152  }
153  else if ((ssize_t) len64 < 0) {
154  return UBJSON_SIZE_ERROR;
155  }
156  else if (len64 < 0) {
157  return UBJSON_INVALID_DATA;
158  }
159 
160  *len = len64;
161  return UBJSON_OKAY;
162 }

164 static ubjson_read_callback_result_t _ubjson_get_call(ubjson_cookie_t *restrict cookie,
165  char marker,
166  ubjson_type_t *type, ssize_t *content)
167 {
168  *content = -1;
169 
170  switch (marker) {
171  case UBJSON_MARKER_NULL:
172  *type = UBJSON_TYPE_NULL;
173  break;
174 
175  case UBJSON_MARKER_NOOP:
176  *type = UBJSON_TYPE_NOOP;
177  break;
178 
179  case UBJSON_MARKER_TRUE:
180  case UBJSON_MARKER_FALSE:
181  *type = UBJSON_TYPE_BOOL,
182  *content = (marker == UBJSON_MARKER_TRUE);
183  break;
184 
185  case UBJSON_MARKER_INT8:
186  *content = UBJSON_INT32_INT8;
187  *type = UBJSON_TYPE_INT32;
188  break;
189 
190  case UBJSON_MARKER_UINT8:
191  *content = UBJSON_INT32_UINT8;
192  *type = UBJSON_TYPE_INT32;
193  break;
194 
195  case UBJSON_MARKER_INT16:
196  *content = UBJSON_INT32_INT16;
197  *type = UBJSON_TYPE_INT32;
198  break;
199 
200  case UBJSON_MARKER_INT32:
201  *content = UBJSON_INT32_INT32;
202  *type = UBJSON_TYPE_INT32;
203  break;
204 
205  case UBJSON_MARKER_INT64:
206  *type = UBJSON_TYPE_INT64;
207  break;
208 
209  case UBJSON_MARKER_CHAR:
210  *type = UBJSON_TYPE_STRING;
211  break;
212 
213  case UBJSON_MARKER_FLOAT32:
214  *type = UBJSON_TYPE_FLOAT;
215  break;
216 
217  case UBJSON_MARKER_FLOAT64:
218  *type = UBJSON_TYPE_DOUBLE;
219  break;
220 
221  case UBJSON_MARKER_STRING: {
222  ubjson_read_callback_result_t result = _ubjson_read_length(cookie, content);
223  if (result != UBJSON_OKAY) {
224  return result;
225  }
226  *type = UBJSON_TYPE_STRING;
227  break;
228  }
229 
230  case UBJSON_MARKER_ARRAY_START:
231  *type = UBJSON_ENTER_ARRAY;
232  break;
233 
234  case UBJSON_MARKER_OBJECT_START:
235  *type = UBJSON_ENTER_OBJECT;
236  break;
237 
238  /* NOTE: High-precision numbers are not implemented, yet. Implement if needed.
239  *
240  * case UBJSON_MARKER_HP_NUMBER:
241  * ...
242  * break;
243  */
244 
245  case UBJSON_MARKER_ARRAY_END:
246  case UBJSON_MARKER_OBJECT_END:
247  case UBJSON_MARKER_COUNT:
248  case UBJSON_MARKER_TYPE:
249  default:
250  return UBJSON_INVALID_DATA;
251  }
252 
253  return UBJSON_OKAY;
254 }

261 static ubjson_read_callback_result_t _ubjson_read_struct(ubjson_cookie_t *restrict cookie,
262  _ubjson_read_struct_continue get_continue)
263 {
264  ubjson_read_callback_result_t result;
265  ssize_t count = -1;
266  char marker, type_marker = 0;
267 
268  READ_MARKER();
269 
270  if (marker == UBJSON_MARKER_TYPE) {
271  READ_MARKER();
272  if (marker == 0) {
273  return UBJSON_INVALID_DATA;
274  }
275  type_marker = marker;
276  READ_MARKER();
277  }
278 
279  if (marker == UBJSON_MARKER_COUNT) {
280  result = _ubjson_read_length(cookie, &count);
281  if (result != UBJSON_OKAY) {
282  return result;
283  }
284  READ_MARKER();
285  }
286 
287  cookie->marker = marker;
288 
289  if ((type_marker != 0) && (count < 0)) {
290  /* If a type is specified, a count must be specified as well.
291  * Otherwise a ']' could either be data (e.g. the character ']'),
292  * or be meant to close the array.
293  */
294  return UBJSON_INVALID_DATA;
295  }
296 
297  for (ssize_t index = 0; (count < 0) || (index < count); ++index) {
298  ubjson_type_t type1;
299  ssize_t content1;
300 
301  READ_MARKER();
302  if (!get_continue(cookie, marker, &result, count, index, &type1, &content1)
303  || (result != UBJSON_OKAY)) {
304  break;
305  }
306 
307  result = cookie->callback.read(cookie, type1, content1,
308  UBJSON_ABSENT, (unsigned char) type_marker);
309  if (result != UBJSON_OKAY) {
310  break;
311  }
312  }
313 
314  return result;
315 }
