32 static gnrc_pktsnip_t *_recv(gnrc_netdev2_t *gnrc_netdev2)
33 {
34  netdev2_t *dev = gnrc_netdev2->dev;
35  int bytes_expected = dev->driver->recv(dev, NULL, 0, NULL);
36  gnrc_pktsnip_t *pkt = NULL;
37 
38  if (bytes_expected > 0) {
39  pkt = gnrc_pktbuf_add(NULL, NULL,
40  bytes_expected,
41  GNRC_NETTYPE_UNDEF);
42 
43  if(!pkt) {
44  DEBUG("_recv_ethernet_packet: cannot allocate pktsnip.\n");
45 
46  /* drop the packet */
47  dev->driver->recv(dev, NULL, bytes_expected, NULL);
48 
49  goto out;
50  }
51 
52  int nread = dev->driver->recv(dev, pkt->data, bytes_expected, NULL);
53  if(nread <= 0) {
54  DEBUG("_recv_ethernet_packet: read error.\n");
55  goto safe_out;
56  }
57 
58  if (nread < bytes_expected) {
59  /* we've got less then the expected packet size,
60  * so free the unused space.*/
61 
62  DEBUG("_recv_ethernet_packet: reallocating.\n");
63  gnrc_pktbuf_realloc_data(pkt, nread);
64  }
65 
66  /* mark ethernet header */
67  gnrc_pktsnip_t *eth_hdr = gnrc_pktbuf_mark(pkt, sizeof(ethernet_hdr_t), GNRC_NETTYPE_UNDEF);
68  if (!eth_hdr) {
69  DEBUG("gnrc_netdev2_eth: no space left in packet buffer\n");
70  goto safe_out;
71  }
72 
73  ethernet_hdr_t *hdr = (ethernet_hdr_t *)eth_hdr->data;
74 
75  /* set payload type from ethertype */
76  pkt->type = gnrc_nettype_from_ethertype(byteorder_ntohs(hdr->type));
77 
78  /* create netif header */
79  gnrc_pktsnip_t *netif_hdr;
80  netif_hdr = gnrc_pktbuf_add(NULL, NULL,
81  sizeof(gnrc_netif_hdr_t) + (2 * ETHERNET_ADDR_LEN),
82  GNRC_NETTYPE_NETIF);
83 
84  if (netif_hdr == NULL) {
85  DEBUG("gnrc_netdev2_eth: no space left in packet buffer\n");
86  pkt = eth_hdr;
87  goto safe_out;
88  }
89 
90  gnrc_netif_hdr_init(netif_hdr->data, ETHERNET_ADDR_LEN, ETHERNET_ADDR_LEN);
91  gnrc_netif_hdr_set_src_addr(netif_hdr->data, hdr->src, ETHERNET_ADDR_LEN);
92  gnrc_netif_hdr_set_dst_addr(netif_hdr->data, hdr->dst, ETHERNET_ADDR_LEN);
93  ((gnrc_netif_hdr_t *)netif_hdr->data)->if_pid = thread_getpid();
94 
95  DEBUG("gnrc_netdev2_eth: received packet from %02x:%02x:%02x:%02x:%02x:%02x "
96  "of length %d\n",
97  hdr->src[0], hdr->src[1], hdr->src[2], hdr->src[3], hdr->src[4],
98  hdr->src[5], nread);
99 #if defined(MODULE_OD) && ENABLE_DEBUG
100  od_hex_dump(hdr, nread, OD_WIDTH_DEFAULT);
101 #endif
102 
103  gnrc_pktbuf_remove_snip(pkt, eth_hdr);
104  LL_APPEND(pkt, netif_hdr);
105  }
106 
107 out:
108  return pkt;
109 
110 safe_out:
111  gnrc_pktbuf_release(pkt);
112  return NULL;
113 }

120 static inline void _addr_set_multicast(uint8_t *dst, gnrc_pktsnip_t *payload)
121 {
122  switch (payload->type) {
123 #ifdef MODULE_GNRC_IPV6
124  case GNRC_NETTYPE_IPV6:
125  /* https://tools.ietf.org/html/rfc2464#section-7 */
126  dst[0] = 0x33;
127  dst[1] = 0x33;
128  ipv6_hdr_t *ipv6 = payload->data;
129  memcpy(dst + 2, ipv6->dst.u8 + 12, 4);
130  break;
131 #endif
132  default:
133  _addr_set_broadcast(dst);
134  break;
135  }
136 }

138 static int _send(gnrc_netdev2_t *gnrc_netdev2, gnrc_pktsnip_t *pkt)
139 {
140  ethernet_hdr_t hdr;
141  gnrc_netif_hdr_t *netif_hdr;
142  gnrc_pktsnip_t *payload;
143  int res;
144 
145  netdev2_t *dev = gnrc_netdev2->dev;
146 
147  if (pkt == NULL) {
148  DEBUG("gnrc_netdev2_eth: pkt was NULL\n");
149  return -EINVAL;
150  }
151 
152  payload = pkt->next;
153 
154  if (pkt->type != GNRC_NETTYPE_NETIF) {
155  DEBUG("gnrc_netdev2_eth: First header was not generic netif header\n");
156  return -EBADMSG;
157  }
158 
159  if (payload) {
160  hdr.type = byteorder_htons(gnrc_nettype_to_ethertype(payload->type));
161  }
162  else {
163  hdr.type = byteorder_htons(ETHERTYPE_UNKNOWN);
164  }
165 
166  netif_hdr = pkt->data;
167 
168  /* set ethernet header */
169  if (netif_hdr->src_l2addr_len == ETHERNET_ADDR_LEN) {
170  memcpy(hdr.dst, gnrc_netif_hdr_get_src_addr(netif_hdr),
171  netif_hdr->src_l2addr_len);
172  }
173  else {
174  dev->driver->get(dev, NETOPT_ADDRESS, hdr.src, ETHERNET_ADDR_LEN);
175  }
176 
177  if (netif_hdr->flags & GNRC_NETIF_HDR_FLAGS_BROADCAST) {
178  _addr_set_broadcast(hdr.dst);
179  }
180  else if (netif_hdr->flags & GNRC_NETIF_HDR_FLAGS_MULTICAST) {
181  if (payload == NULL) {
182  DEBUG("gnrc_netdev2_eth: empty multicast packets over Ethernet "\
183  "are not yet supported\n");
184  return -ENOTSUP;
185  }
186  _addr_set_multicast(hdr.dst, payload);
187  }
188  else if (netif_hdr->dst_l2addr_len == ETHERNET_ADDR_LEN) {
189  memcpy(hdr.dst, gnrc_netif_hdr_get_dst_addr(netif_hdr),
190  ETHERNET_ADDR_LEN);
191  }
192  else {
193  DEBUG("gnrc_netdev2_eth: destination address had unexpected format\n");
194  return -EBADMSG;
195  }
196 
197  DEBUG("gnrc_netdev2_eth: send to %02x:%02x:%02x:%02x:%02x:%02x\n",
198  hdr.dst[0], hdr.dst[1], hdr.dst[2],
199  hdr.dst[3], hdr.dst[4], hdr.dst[5]);
200 
201  size_t n;
202  payload = gnrc_pktbuf_get_iovec(pkt, &n); /* use payload as temporary
203  * variable */
204  res = -ENOBUFS;
205  if (payload != NULL) {
206  pkt = payload; /* reassign for later release; vec_snip is prepended to pkt */
207  struct iovec *vector = (struct iovec *)pkt->data;
208  vector[0].iov_base = (char*)&hdr;
209  vector[0].iov_len = sizeof(ethernet_hdr_t);
210 #ifdef MODULE_NETSTATS_L2
211  if ((netif_hdr->flags & GNRC_NETIF_HDR_FLAGS_BROADCAST) ||
212  (netif_hdr->flags & GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
213  gnrc_netdev2->dev->stats.tx_mcast_count++;
214  }
215  else {
216  gnrc_netdev2->dev->stats.tx_unicast_count++;
217  }
218 #endif
219  res = dev->driver->send(dev, vector, n);
220  }
221 
222  gnrc_pktbuf_release(pkt);
223 
224  return res;
225 }
