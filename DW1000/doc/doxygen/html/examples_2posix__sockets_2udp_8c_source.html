41 static void *_server_thread(void *args)
42 {
43  struct sockaddr_in6 server_addr;
44  uint16_t port;
45  msg_init_queue(server_msg_queue, SERVER_MSG_QUEUE_SIZE);
46  server_socket = socket(AF_INET6, SOCK_DGRAM, IPPROTO_UDP);
47  /* parse port */
48  port = (uint16_t)atoi((char *)args);
49  if (port == 0) {
50  puts("Error: invalid port specified");
51  return NULL;
52  }
53  server_addr.sin6_family = AF_INET6;
54  memset(&server_addr.sin6_addr, 0, sizeof(server_addr.sin6_addr));
55  server_addr.sin6_port = htons(port);
56  if (server_socket < 0) {
57  puts("error initializing socket");
58  server_socket = 0;
59  return NULL;
60  }
61  if (bind(server_socket, (struct sockaddr *)&server_addr, sizeof(server_addr)) < 0) {
62  server_socket = -1;
63  puts("error binding socket");
64  return NULL;
65  }
66  printf("Success: started UDP server on port %" PRIu16 "\n", port);
67  while (1) {
68  int res;
69  struct sockaddr_in6 src;
70  socklen_t src_len = sizeof(struct sockaddr_in6);
71  if ((res = recvfrom(server_socket, server_buffer, sizeof(server_buffer), 0,
72  (struct sockaddr *)&src, &src_len)) < 0) {
73  puts("Error on receive");
74  }
75  else if (res == 0) {
76  puts("Peer did shut down");
77  }
78  else {
79  printf("Received data: ");
80  puts(server_buffer);
81  }
82  }
83  return NULL;
84 }

86 static int udp_send(char *addr_str, char *port_str, char *data, unsigned int num,
87  unsigned int delay)
88 {
89  struct sockaddr_in6 src, dst;
90  size_t data_len = strlen(data);
91  uint16_t port;
92  int s;
93  src.sin6_family = AF_INET6;
94  dst.sin6_family = AF_INET6;
95  memset(&src.sin6_addr, 0, sizeof(src.sin6_addr));
96  /* parse destination address */
97  if (inet_pton(AF_INET6, addr_str, &dst.sin6_addr) != 1) {
98  puts("Error: unable to parse destination address");
99  return 1;
100  }
101  /* parse port */
102  port = (uint16_t)atoi(port_str);
103  dst.sin6_port = htons(port);
104  src.sin6_port = htons(port);
105  s = socket(AF_INET6, SOCK_DGRAM, IPPROTO_UDP);
106  if (s < 0) {
107  puts("error initializing socket");
108  return 1;
109  }
110  for (unsigned int i = 0; i < num; i++) {
111  if (sendto(s, data, data_len, 0, (struct sockaddr *)&dst, sizeof(dst)) < 0) {
112  puts("could not send");
113  }
114  else {
115  printf("Success: send %u byte to %s:%u\n", (unsigned)data_len, addr_str, port);
116  }
117 
118  usleep(delay);
119  }
120  close(s);
121  return 0;
122 }
