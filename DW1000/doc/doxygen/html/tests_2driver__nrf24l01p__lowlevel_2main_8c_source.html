114 void *nrf24l01p_rx_handler(void *arg)
115 {
116  msg_t msg_q[1];
117  msg_init_queue(msg_q, 1);
118  unsigned int pid = thread_getpid();
119  char rx_buf[NRF24L01P_MAX_DATA_LENGTH];
120 
121  puts("Registering nrf24l01p_rx_handler thread...");
122  nrf24l01p_register(&nrf24l01p_0, &pid);
123 
124  msg_t m;
125 
126  while (msg_receive(&m)) {
127  printf("nrf24l01p_rx_handler got a message: ");
128 
129  switch (m.type) {
130  case RCV_PKT_NRF24L01P:
131  puts("Received packet.");
132 
133  /* CE low */
134  nrf24l01p_stop(m.content.ptr);
135 
136  /* read payload */
137  nrf24l01p_read_payload(m.content.ptr, rx_buf, NRF24L01P_MAX_DATA_LENGTH);
138 
139  /* flush rx fifo */
140  nrf24l01p_flush_rx_fifo(m.content.ptr);
141 
142  /* CE high */
143  nrf24l01p_start(m.content.ptr);
144 
145  /* print rx buffer */
146  for (int i = 0; i < NRF24L01P_MAX_DATA_LENGTH; i++) {
147  printf("%i ", rx_buf[i]);
148  }
149 
150  puts("");
151 
152  break;
153 
154  default:
155  puts("stray message.");
156  break;
157  }
158  }
159 
160  puts("nrf24l01p_rx_handler: this should not have happened!");
161 
162  return NULL;
163 }

201 int cmd_send(int argc, char **argv)
202 {
203  (void) argc;
204  (void) argv;
205 
206  puts("Send");
207 
208  int status = 0;
209  char tx_buf[NRF24L01P_MAX_DATA_LENGTH];
210 
211  /* fill TX buffer with numbers 32..1 */
212  for (int i = 0; i < sizeof(tx_buf); i++) {
213  tx_buf[i] = NRF24L01P_MAX_DATA_LENGTH - i;
214  }
215  /* power on the device */
216  if (nrf24l01p_on(&nrf24l01p_0) < 0) {
217  puts("Error in nrf24l01p_on");
218  return 1;
219  }
220  /* setup device as transmitter */
221  if (nrf24l01p_set_txmode(&nrf24l01p_0) < 0) {
222  puts("Error in nrf24l01p_set_txmode");
223  return 1;
224  }
225  /* load data to transmit into device */
226  if (nrf24l01p_preload(&nrf24l01p_0, tx_buf, NRF24L01P_MAX_DATA_LENGTH) < 0) {
227  puts("Error in nrf24l01p_preload");
228  return 1;
229  }
230  /* trigger transmitting */
231  nrf24l01p_transmit(&nrf24l01p_0);
232  /* wait while data is pysically transmitted */
233  xtimer_usleep(DELAY_DATA_ON_AIR);
234  /* get status of the transceiver */
235  status = nrf24l01p_get_status(&nrf24l01p_0);
236  if (status < 0) {
237  puts("Error in nrf24l01p_get_status");
238  }
239  if (status & TX_DS) {
240  puts("Sent Packet");
241  }
242  /* setup device as receiver */
243  if (nrf24l01p_set_rxmode(&nrf24l01p_0) < 0) {
244  puts("Error in nrf24l01p_set_rxmode");
245  return 1;
246  }
247 
248  return 0;
249 }
