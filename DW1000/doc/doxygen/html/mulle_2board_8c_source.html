59 void board_init(void)
60 {
61  int status;
62 
63  /* initialize the boards LEDs */
64  gpio_init(LED0_PIN, GPIO_OUT);
65  gpio_init(LED1_PIN, GPIO_OUT);
66  gpio_init(LED2_PIN, GPIO_OUT);
67 
68  /* Initialize power control pins */
69  power_pins_init();
70 
71  /* Turn on Vperiph for peripherals */
72  /*
73  * By turning on Vperiph first, and before waiting for the clocks to
74  * stabilize, we will have used enough time to have let the FRAM start up
75  * properly when we want to access it later without having to add any extra
76  * delays.
77  */
78  gpio_set(MULLE_POWER_VPERIPH);
79 
80  /* Turn on AVDD for reading voltages */
81  gpio_set(MULLE_POWER_AVDD);
82 
83  /* Initialize RTC oscillator as early as possible since we are using it as a
84  * base clock for the FLL.
85  * It takes a while to stabilize the oscillator, therefore we do this as
86  * soon as possible during boot in order to let it stabilize while other
87  * stuff is initializing. */
88  /* If the clock is not stable then the UART will have the wrong baud rate
89  * for debug prints as well */
90  rtt_init();
91 
92  /* Set up clocks */
93  set_safe_clock_dividers();
94 
95  set_fll_source();
96 
97  kinetis_mcg_set_mode(KINETIS_MCG_FEE);
98 
99  /* At this point we need to wait for 1 ms until the clock is stable.
100  * Since the clock is not yet stable we can only guess how long we must
101  * wait. I have tried to make this as short as possible but still being able
102  * to read the initialization messages written on the UART.
103  * (If the clock is not stable all UART output is garbled until it has
104  * stabilized) */
105  for (int i = 0; i < 100000; ++i) {
106  __asm__ volatile("nop\n");
107  }
108 
109  /* Update SystemCoreClock global var */
110  SystemCoreClockUpdate();
111 
112  /* initialize the CPU */
113  cpu_init();
114 
115  /* NVRAM requires xtimer for timing */
116  xtimer_init();
117 
118  /* Initialize NVRAM */
119  status = mulle_nvram_init();
120  if (status == 0) {
121  /* Increment boot counter */
122  increase_boot_count();
123  }
124 }

136 static inline void set_safe_clock_dividers(void)
137 {
138  /*
139  * We want to achieve the following clocks:
140  * Core/system: <100MHz
141  * Bus: <50MHz
142  * FlexBus: <50MHz
143  * Flash: <25MHz
144  *
145  * using dividers 1-2-2-4 will obey the above limits when using a 96MHz FLL source.
146  */
147  SIM->CLKDIV1 = (
148  SIM_CLKDIV1_OUTDIV1(CONFIG_CLOCK_K60_SYS_DIV) | /* Core/System clock divider */
149  SIM_CLKDIV1_OUTDIV2(CONFIG_CLOCK_K60_BUS_DIV) | /* Bus clock divider */
150  SIM_CLKDIV1_OUTDIV3(CONFIG_CLOCK_K60_FB_DIV) | /* FlexBus divider, not used in Mulle */
151  SIM_CLKDIV1_OUTDIV4(CONFIG_CLOCK_K60_FLASH_DIV)); /* Flash clock divider */
152 
153 }

155 static inline void set_fll_source(void)
156 {
157  /* Select FLL as source (as opposed to PLL) */
158  SIM->SOPT2 &= ~(SIM_SOPT2_PLLFLLSEL_MASK);
159  /* Use external 32kHz RTC clock as source for OSC32K */
160 #if K60_CPU_REV == 1
161  SIM->SOPT1 |= SIM_SOPT1_OSC32KSEL_MASK;
162 #elif K60_CPU_REV == 2
163  SIM->SOPT1 = (SIM->SOPT1 & ~(SIM_SOPT1_OSC32KSEL_MASK)) | SIM_SOPT1_OSC32KSEL(2);
164 #else
165 #error Unknown K60 CPU revision
166 #endif
167 
168  /* Select RTC 32kHz clock as reference clock for the FLL */
169 #if K60_CPU_REV == 1
170  /* Rev 1 parts */
171  SIM->SOPT2 |= SIM_SOPT2_MCGCLKSEL_MASK;
172 #elif K60_CPU_REV == 2
173  /* Rev 2 parts */
174  MCG->C7 = (MCG_C7_OSCSEL_MASK);
175 #else
176 #error Unknown K60 CPU revision
177 #endif
178 }

180 static int mulle_nvram_init(void)
181 {
182  union {
183  uint32_t u32;
184  uint8_t u8[sizeof(uint32_t)];
185  } rec;
186  rec.u32 = 0;
187 
188  if (nvram_spi_init(mulle_nvram, &nvram_spi_params, MULLE_NVRAM_CAPACITY) != 0) {
189  return -2;
190  }
191 
192  if (mulle_nvram->read(mulle_nvram, &rec.u8[0], MULLE_NVRAM_MAGIC, sizeof(rec.u32)) != sizeof(rec.u32)) {
193  return -3;
194  }
195 
196  if (rec.u32 != MULLE_NVRAM_MAGIC_EXPECTED) {
197  int i;
198  union {
199  uint64_t u64;
200  uint8_t u8[sizeof(uint64_t)];
201  } zero;
202  zero.u64 = 0;
203  for (i = 0; i < MULLE_NVRAM_CAPACITY; i += sizeof(zero)) {
204  if (mulle_nvram->write(mulle_nvram, &zero.u8[0], i, sizeof(zero.u64)) != sizeof(zero.u64)) {
205  return -4;
206  }
207  }
208  rec.u32 = MULLE_NVRAM_MAGIC_EXPECTED;
209  if (mulle_nvram->write(mulle_nvram, &rec.u8[0], MULLE_NVRAM_MAGIC, sizeof(rec.u32)) != sizeof(rec.u32)) {
210  return -5;
211  }
212  }
213  return 0;
214 }

216 static void increase_boot_count(void)
217 {
218  union {
219  uint32_t u32;
220  uint8_t u8[sizeof(uint32_t)];
221  } rec;
222  rec.u32 = 0;
223  if (mulle_nvram->read(mulle_nvram, &rec.u8[0], MULLE_NVRAM_BOOT_COUNT, sizeof(rec.u32)) != sizeof(rec.u32)) {
224  return;
225  }
226  ++rec.u32;
227  mulle_nvram->write(mulle_nvram, &rec.u8[0], MULLE_NVRAM_BOOT_COUNT, sizeof(rec.u32));
228 }
