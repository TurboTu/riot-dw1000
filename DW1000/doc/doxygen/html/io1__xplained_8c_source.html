44 int io1_xplained_init(io1_xplained_t *dev, uint8_t addr)
45 {
46  /* Initialize I2C interface */
47  if (at30tse75x_init(&dev->temp,
48  I2C_DEV(0),
49  I2C_SPEED_NORMAL, (TEMPERATURE_BASE_ADDR | addr)) < 0) {
50  DEBUG("[Error] Cannot initialize temperature sensor.\n");
51  return -1;
52  }
53 
54  /* Use maximum resolution */
55  at30tse75x_set_resolution(&dev->temp, AT30TSE75X_RESOLUTION_12BIT);
56 
57  if (gpio_init(IO1_LED_PIN, GPIO_OUT) < 0) {
58  DEBUG("[Error] GPIO LED not enabled\n");
59  return -1;
60  }
61 
62  if (gpio_init(IO1_GPIO1_PIN, GPIO_OUT) < 0) {
63  DEBUG("[Error] GPIO1 not enabled\n");
64  return -1;
65  }
66 
67  if (gpio_init(IO1_GPIO2_PIN, GPIO_OUT) < 0) {
68  DEBUG("[Error] GPIO2 not enabled\n");
69  return -1;
70  }
71 
72  DEBUG("[Info] IO1 Xplained extension initialized!\n");
73 
74  return 0;
75 }

77 void io1_xplained_auto_init(void)
78 {
79  for (unsigned i = 0; i < IO1_XPLAINED_NUMOF; i++) {
80  if (io1_xplained_init(&io1_xplained_devs[i],
81  io1_xplained_params[i].addr) < 0) {
82  LOG_ERROR("Unable to initialize IO1 Xplained #%i\n", i);
83  }
84 #ifdef MODULE_SAUL_REG
85  io1_xplained_saul_reg[i][0].dev = &io1_xplained_devs[i];
86  saul_reg_add(&io1_xplained_saul_reg[i][0]);
87 #endif
88 #ifdef MODULE_SAUL_GPIO
89  for (unsigned j = 1; j < 4; j++) {
90  io1_xplained_saul_reg[i][j].dev = &(io1_xplained_saul_gpios[j-1]);
91  saul_reg_add(&(io1_xplained_saul_reg[i][j]));
92  }
93 #endif
94  }
