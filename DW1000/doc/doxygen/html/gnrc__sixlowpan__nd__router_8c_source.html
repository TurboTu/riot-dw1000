141 bool gnrc_sixlowpan_nd_router_abr_older(sixlowpan_nd_opt_abr_t *abr_opt)
142 {
143  gnrc_sixlowpan_nd_router_abr_t *abr;
144  uint32_t version;
145 
146  if (abr_opt->len != SIXLOWPAN_ND_OPT_ABR_LEN) {
147  /* invalid option received */
148  return true;
149  }
150 
151  if (_is_me(&abr_opt->braddr)) {
152  return false;
153  }
154 
155  abr = _get_abr(&abr_opt->braddr);
156 
157  if (abr == NULL) {
158  return false;
159  }
160 
161  version = (uint32_t)byteorder_ntohs(abr_opt->vlow);
162  version |= ((uint32_t)byteorder_ntohs(abr_opt->vhigh)) << 16;
163 
164  return (version < abr->version);

182  prefix->prefix = NULL;

189 void gnrc_sixlowpan_nd_opt_abr_handle(kernel_pid_t iface, ndp_rtr_adv_t *rtr_adv, int sicmpv6_size,
190  sixlowpan_nd_opt_abr_t *abr_opt)
191 {
192  uint16_t opt_offset = 0;
193  uint8_t *buf = (uint8_t *)(rtr_adv + 1);
194  gnrc_sixlowpan_nd_router_abr_t *abr;
195  uint32_t t = 0;
196 
197  if (_is_me(&abr_opt->braddr)) {
198  return;
199  }
200  /* validity and version was checked in previously called
201  * gnrc_sixlowpan_nd_router_abr_older() */
202  abr = _get_abr(&abr_opt->braddr);
203 
204  if (abr == NULL) {
205  return;
206  }
207 
208  abr->ltime = byteorder_ntohs(abr_opt->ltime);
209 
210  if (abr->ltime == 0) {
211  abr->ltime = GNRC_SIXLOWPAN_ND_BORDER_ROUTER_DEFAULT_LTIME;
212  return;
213  }
214 
215  sicmpv6_size -= sizeof(ndp_rtr_adv_t);
216 
217  while (sicmpv6_size > 0) {
218  ndp_opt_t *opt = (ndp_opt_t *)(buf + opt_offset);
219 
220  switch (opt->type) {
221  case NDP_OPT_PI:
222  _add_prefix(iface, abr, (ndp_opt_pi_t *)opt);
223 
224  case NDP_OPT_6CTX:
225  _add_ctx(abr, (sixlowpan_nd_opt_6ctx_t *)opt);
226 
227  default:
228  break;
229  }
230 
231  opt_offset += (opt->len * 8);
232  sicmpv6_size -= (opt->len * 8);
233  }
234 
235  abr->version = (uint32_t)byteorder_ntohs(abr_opt->vlow);
236  abr->version |= ((uint32_t)byteorder_ntohs(abr_opt->vhigh)) << 16;
237  abr->addr.u64[0] = abr_opt->braddr.u64[0];
238  abr->addr.u64[1] = abr_opt->braddr.u64[1];
239  memset(abr->ctxs, 0, sizeof(abr->ctxs));
240  abr->prfs = NULL;
241 
242  t = abr->ltime * 60 * US_PER_SEC;
243 
244  xtimer_remove(&abr->ltimer);
245  abr->ltimer_msg.type = GNRC_SIXLOWPAN_ND_MSG_ABR_TIMEOUT;
246  abr->ltimer_msg.content.ptr = abr;
247  xtimer_set_msg(&abr->ltimer, t, &abr->ltimer_msg, gnrc_ipv6_pid);
248 }
