35 static int _comp_uncomp(const uint8_t *buf, size_t len)
36 {
37  memset(_buf, 0, sizeof(_buf));
38  memset(_buf_res, 0, sizeof(_buf_res));
39 
40  heatshrink_encoder_reset(&_encoder);
41  heatshrink_decoder_reset(&_decoder);
42 
43  size_t n = len;
44  uint8_t *inpos = (uint8_t*)buf;
45  uint8_t *outpos = _buf;
46 
47  /* compress */
48  while(1) {
49  size_t n_sunk = 0;
50  if (n) {
51  heatshrink_encoder_sink(&_encoder, inpos, n, &n_sunk);
52  if (n_sunk) {
53  inpos += n_sunk;
54  n -= n_sunk;
55  }
56  size_t written = 0;
57  heatshrink_encoder_poll(&_encoder, outpos, (_buf + sizeof(_buf) - outpos), &written);
58  outpos += written;
59  }
60  else {
61  while (heatshrink_encoder_finish(&_encoder) == HSER_FINISH_MORE) {
62  size_t written = 0;
63  heatshrink_encoder_poll(&_encoder, outpos, (_buf + sizeof(_buf) - outpos), &written);
64  outpos += written;
65  }
66  break;
67  }
68  }
69 
70  /* decompress */
71  n = (outpos - _buf);
72  inpos = (uint8_t*) _buf;
73  outpos = _buf_res;
74 
75  while(1) {

77  if (n) {

79  if (n_sunk) {

85  outpos += written;

91  outpos += written;
