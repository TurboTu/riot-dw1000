78 static int _tftp_client_data_cb(uint32_t offset, void *data, size_t data_len)
79 {
80  char *c = (char *) data;
81 
82  /* if we are reading return the part of the data that is being requested */
83  if (_tftp_action == TFTP_WRITE) {
84  /* calculate the length of the data block to transfer */
85  if (offset + data_len > sizeof(_tftp_client_hello)) {
86  data_len -= (offset + data_len) - sizeof(_tftp_client_hello);
87  }
88 
89  /* copy the block to the output buffer */
90  memcpy(data, _tftp_client_hello + offset, data_len);
91  }
92  else {
93  /* we received a data block which we output to the console */
94  printf("\n -- CLIENT DATA --\n%.*s\n -- CLIENT DATA --\n", (int)data_len, c);
95  }
96
