47 static uint16_t read(gpio_t pin, int bits)
48 {
49  uint16_t res = 0;
50 
51  for (int i = 0; i < bits; i++) {
52  uint32_t start, end;
53  res <<= 1;
54  /* measure the length between the next rising and falling flanks (the
55  * time the pin is high - smoke up :-) */
56  while (!gpio_read(pin));
57  start = xtimer_now_usec();
58  while (gpio_read(pin));
59  end = xtimer_now_usec();
60  /* if the high phase was more than 40us, we got a 1 */
61  if ((end - start) > PULSE_WIDTH_THRESHOLD) {
62  res |= 0x0001;
63  }
64  }
65  return res;
66 }

100 int dht_read(dht_t *dev, int16_t *temp, int16_t *hum)
101 {
102  uint8_t csum, sum;
103  uint16_t raw_hum, raw_temp;
104 
105  /* send init signal to device */
106  gpio_clear(dev->pin);
107  xtimer_usleep(20 * US_PER_MS);
108  gpio_set(dev->pin);
109  xtimer_usleep(40);
110 
111  /* sync on device */
112  gpio_init(dev->pin, dev->in_mode);
113  while (!gpio_read(dev->pin)) ;
114  while (gpio_read(dev->pin)) ;
115 
116  /*
117  * data is read in sequentially, highest bit first:
118  * 40 .. 24 23 .. 8 7 .. 0
119  * [humidity][temperature][checksum]
120  */
121 
122  /* read the humidity, temperature, and checksum bits */
123  raw_hum = read(dev->pin, 16);
124  raw_temp = read(dev->pin, 16);
125  csum = (uint8_t)read(dev->pin, 8);
126 
127  /* set pin high again - so we can trigger the next reading by pulling it low
128  * again */
129  gpio_init(dev->pin, GPIO_OUT);
130  gpio_set(dev->pin);
131 
132  /* validate the checksum */
133  sum = (raw_temp >> 8) + (raw_temp & 0xff) + (raw_hum >> 8) + (raw_hum & 0xff);
134  if ((sum != csum) || (csum == 0)) {
135  DEBUG("error: checksum invalid\n");
136  return -1;
137  }
138 
139  /* parse the RAW values */
140  DEBUG("RAW values: temp: %7i hum: %7i\n", (int)raw_temp, (int)raw_hum);
141  switch (dev->type) {
142  case DHT11:
143  *temp = (int16_t)((raw_temp >> 8) * 10);
144  *hum = (int16_t)((raw_hum >> 8) * 10);
145  break;
146  case DHT22:
147  *hum = (int16_t)raw_hum;
148  /* if the high-bit is set, the value is negative */
149  if (raw_temp & 0x8000) {
150  *temp = (int16_t)((raw_temp & ~0x8000) * -1);
151  }
152  else {
153  *temp = (int16_t)raw_temp;
154  }
155  break;
156  default:
157  return -2;
158  }
159 
160  return 0;
161 }
