79 static void pic_interrupt_entry(uint8_t intr_num, struct x86_pushad *orig_ctx, unsigned long error_code)
80 {
81  (void) error_code;
82  (void) orig_ctx;
83 
84  uint8_t irq_num = intr_num - PIC_MASTER_INTERRUPT_BASE;
85  if (spurious_irq(irq_num)) {
86  return;
87  }
88 
89  x86_irq_handler_t handler = x86_registered_irqs[irq_num];
90  if (handler) {
91  handler(irq_num);
92  }
93 
94  outb(PIC_MASTER + PIC_COMMAND, PIC_EOI);
95  if (irq_num > 7) {
96  outb(PIC_SLAVE + PIC_COMMAND, PIC_EOI);
97  }
98 }

100 static void pic_remap(void)
101 {
102  /* tell both PICs we want to initialize them */
103  outb(PIC_MASTER + PIC_COMMAND, PIC_ICW1_INIT + PIC_ICW1_ICW4);
104  io_wait();
105  outb(PIC_SLAVE + PIC_COMMAND, PIC_ICW1_INIT + PIC_ICW1_ICW4);
106  io_wait();
107 
108  /* which ISR should be called if an IRQ occurs */
109  outb(PIC_MASTER + PIC_DATA, PIC_MASTER_INTERRUPT_BASE);
110  io_wait();
111  outb(PIC_SLAVE + PIC_DATA, PIC_SLAVE_INTERRUPT_BASE);
112  io_wait();
113 
114  /* IRQ of the master the slave talks to */
115  outb(PIC_MASTER + PIC_DATA, 1 << PIC_NUM_SLAVE);
116  io_wait();
117  outb(PIC_SLAVE + PIC_DATA, PIC_NUM_SLAVE);
118  io_wait();
119 
120  /* use PC mode */
121  outb(PIC_MASTER + PIC_DATA, PIC_ICW4_8086);
122  io_wait();
123  outb(PIC_SLAVE + PIC_DATA, PIC_ICW4_8086);
124 }

133 void x86_pic_set_enabled_irqs(uint16_t mask)
134 {
135  unsigned old_status = irq_disable();
136 
137  mask |= PIC_MASK_SLAVE;
138  mask &= ~PIC_MASK_FPU;
139  outb(PIC_MASTER + PIC_IMR, ~(uint8_t) mask);
140  io_wait();
141  outb(PIC_SLAVE + PIC_IMR, ~(uint8_t) (mask >> 8));

146 void x86_pic_enable_irq(unsigned num)
147 {
148  unsigned old_status = irq_disable();
149 
150  uint16_t port;
151  if (num < 8) {
152  port = PIC_MASTER;
153  }
154  else {
155  num -= 8;
156  port = PIC_SLAVE;
157  }
158  uint8_t cur = inb(port + PIC_IMR);
159  outb(port + PIC_IMR, cur & ~(1 << num));
160 
161  irq_restore(old_status);
162 }
