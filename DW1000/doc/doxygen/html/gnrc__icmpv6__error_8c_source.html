38  gnrc_pktsnip_t *orig_pkt, uint32_t value)

40  gnrc_pktsnip_t *pkt = gnrc_icmpv6_build(NULL, type, code, _fit(orig_pkt));

42  /* copy as much of the originating packet into error message as fits the message's size */
43  if (pkt != NULL) {
44  size_t offset = ICMPV6_ERROR_SZ;
45  uint8_t *data = pkt->data;
46  ICMPV6_ERROR_SET_VALUE(data, value);

86  if (pkt != NULL) {

88  uint8_t *data = pkt->data;
89  uint32_t ptr_offset = 0U;
90  bool found_offset = false;
91 
92  while (orig_pkt != NULL) {
93  /* copy as long as it fits into packet */

101  ptr_offset += (uint32_t)(((uint8_t *)ptr) - ((uint8_t *)orig_pkt->data));

111  break;
