45 static inline void modem_clock_init(void)
46 {
47  /* Use the CLK_OUT of the modem as the clock source. */
48 
49  /* Enable GPIO clock gates */
50  KW2XDRF_PORT_CLKEN();
51  KW2XDRF_CLK_CTRL_CLKEN();
52 
53  /* Modem RST_B is connected to PTB19 and can be used to reset the modem. */
54  KW2XDRF_PORT_DEV->PCR[KW2XDRF_RST_PIN] = PORT_PCR_MUX(1);
55  BITBAND_REG32(KW2XDRF_GPIO->PDDR, KW2XDRF_RST_PIN) = 1;
56  KW2XDRF_GPIO->PCOR = (1 << KW2XDRF_RST_PIN);
57 
58  /* Modem GPIO5 is connected to PTC0 and can be used to select CLK_OUT frequency, */
59  /* set PTC0 high for CLK_OUT=32.787kHz and low for CLK_OUT=4MHz. */
60  KW2XDRF_CLK_CTRL_PORT_DEV->PCR[KW2XDRF_CLK_CTRL_PIN] = PORT_PCR_MUX(1);
61  BITBAND_REG32(KW2XDRF_CLK_CTRL_GPIO->PDDR, KW2XDRF_CLK_CTRL_PIN) = 1;
62  KW2XDRF_CLK_CTRL_GPIO->PCOR = (1 << KW2XDRF_CLK_CTRL_PIN);
63 
64  /* Modem IRQ_B is connected to PTB3, modem interrupt request to the MCU. */
65  KW2XDRF_PORT_DEV->PCR[KW2XDRF_IRQ_PIN] = PORT_PCR_MUX(1);
66  BITBAND_REG32(KW2XDRF_GPIO->PDDR, KW2XDRF_IRQ_PIN) = 0;
67 
68  /* release the reset */
69  KW2XDRF_GPIO->PSOR = (1 << KW2XDRF_RST_PIN);
70 
71  /* wait for modem IRQ_B interrupt request */
72  while (KW2XDRF_GPIO->PDIR & (1 << KW2XDRF_IRQ_PIN));
73 }
