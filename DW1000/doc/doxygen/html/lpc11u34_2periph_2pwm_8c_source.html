36 uint32_t pwm_init(pwm_t pwm, pwm_mode_t mode, uint32_t freq, uint16_t res)
37 {
38  assert((pwm < PWM_NUMOF) && (mode != PWM_CENTER));
39 
40  /* make sure the given frequency settings are applicable */
41  if (CLOCK_CORECLOCK < (res * freq)) {
42  return 0;
43  }
44 
45  /* setup pins */
46  for (unsigned i = 0; i < PWM_CHAN_NUMOF; i++) {
47  if (pwm_config[pwm].pins[i]) {
48  *(pwm_config[pwm].pins[i]) = pwm_config[pwm].af;
49  }
50  }
51 
52  /* power on and configure the timer */
53  pwm_poweron(pwm);
54  /* enable the timer and keep it in reset state */
55  dev(pwm)->TCR = BIT0 | BIT1;
56  /* set prescaler */
57  dev(pwm)->PR = (CLOCK_CORECLOCK / (res * freq));
58  /* reset timer on MR3 */
59  dev(pwm)->MCR = BIT10;
60 
61  /* set PWM period */
62  dev(pwm)->MR0 = res;
63  dev(pwm)->MR1 = res;
64  dev(pwm)->MR2 = res;
65  dev(pwm)->MR3 = (res - 1);
66 
67  /* set mode for channels 0, 1, and 2 */
68  dev(pwm)->EMR = (((mode + 1) << 4) | ((mode + 1) << 6) | ((mode + 1) << 8));
69 
70  /* enable channels 0, 1, and 2 */
71  dev(pwm)->PWMC = (BIT0 | BIT1 | BIT2);
72 
73  return freq;
74 }
