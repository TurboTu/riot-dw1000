42 char *ipv6_addr_to_str(char *result, const ipv6_addr_t *addr, uint8_t result_len)
43 {
44  char tmp[IPV6_ADDR_MAX_STR_LEN], *tp;
45  struct {
46  int16_t base, len;
47  } best = { -1, 0}, cur = { -1, 0};
48 
49  if ((result == NULL) || (addr == NULL)) {
50  return NULL;
51  }
52 
53  /*
54  * Preprocess:
55  * Find the longest run of 0x0000's in address for :: shorthanding.
56  */
57  for (uint8_t i = 0; i < IPV6_ADDR_WORD_LEN; i++) {
58  if (addr->u16[i].u16 == 0) {
59  if (cur.base == -1) {
60  cur.base = i;
61  cur.len = 1;
62  }
63  else {
64  cur.len++;
65  }
66  }
67  else {
68  if (cur.base != -1) {
69  if (best.base == -1 || cur.len > best.len) {
70  best = cur;
71  }
72 
73  cur.base = -1;
74  }
75  }
76  }
77 
78  if (cur.base != -1) {
79  if (best.base == -1 || cur.len > best.len) {
80  best = cur;
81  }
82  }
83 
84  if (best.base != -1 && best.len < 2) {
85  best.base = -1;
86  }
87 
88  /*
89  * Format the result.
90  */
91  tp = tmp;
92 
93  for (int i = 0; i < ((int)IPV6_ADDR_WORD_LEN);) {
94  /* Are we inside the best run of 0x00's? */
95  if (i == best.base) {
96  *tp++ = ':';
97  i += best.len;
98  continue;
99  }
100 
101  /* Are we following an initial run of 0x00s or any real hex? */
102  if (i != 0) {
103  *tp++ = ':';
104  }
105 
106 #ifdef MODULE_IPV4_ADDR
107  /* Is this address an encapsulated IPv4? */
108  if (i == 6 && best.base == 0 &&
109  (best.len == 6 || (best.len == 5 && addr->u16[5].u16 == 0xffff))) {
110  if (!ipv4_addr_to_str(tp, (const ipv4_addr_t *)&addr->u32[3],
111  sizeof(tmp) - (tp - tmp))) {
112  return (NULL);
113  }
114 
115  tp += strlen(tp);
116  break;
117  }
118 #endif
119 
120  if ((addr->u16[i].u8[0] & 0xf0) != 0x00) {
121  *tp++ = HEX_L[addr->u16[i].u8[0] >> 4];
122  *tp++ = HEX_L[addr->u16[i].u8[0] & 0x0f];
123  *tp++ = HEX_L[addr->u16[i].u8[1] >> 4];
124  }
125  else if ((addr->u16[i].u8[0] & 0x0f) != 0x00) {
126  *tp++ = HEX_L[addr->u16[i].u8[0] & 0x0f];
127  *tp++ = HEX_L[addr->u16[i].u8[1] >> 4];
128  }
129  else if ((addr->u16[i].u8[1] & 0xf0) != 0x00) {
130  *tp++ = HEX_L[addr->u16[i].u8[1] >> 4];
131  }
132 
133  *tp++ = HEX_L[addr->u16[i].u8[1] & 0xf];
134 
135  i++;
136  }
137 
138  /* Was it a trailing run of 0x00's? */
139  if (best.base != -1 && (best.base + best.len) == IPV6_ADDR_WORD_LEN) {
140  *tp++ = ':';
141  }
142 
143  *tp++ = '\0';
144 
145  /*
146  * Check for overflow, copy, and we're done.
147  */
148  if ((size_t)(tp - tmp) > result_len) {
149  return NULL;
150  }
151 
152  strcpy(result, tmp);
153  return result;
154 }
