110 static int calc_and_compare_hash(const char *str, const char *expected,
111  long int repeatcount)
112 {
113  sha1_context ctx;
114 
115  uint8_t hash[SHA1_DIGEST_LENGTH];
116  char tmp[(3 * SHA1_DIGEST_LENGTH) + 1];
117 
118  /* calculate hash */
119  sha1_init(&ctx);
120  for (long int i = 0; i < repeatcount; ++i) {
121  sha1_update(&ctx, (unsigned char*) str, strlen(str));
122  }
123  sha1_final(&ctx, hash);
124  /* copy hash to string */
125  for (size_t i = 0; i < SHA1_DIGEST_LENGTH; i++) {
126  sprintf(&(tmp[i * 3]), "%02X ", (unsigned) hash[i]);
127  }
128  tmp[SHA1_DIGEST_LENGTH* 2] = '\0';
129  /* compare with result string */
130  return strncmp(tmp, expected, strlen((char*) tmp));
131 }
