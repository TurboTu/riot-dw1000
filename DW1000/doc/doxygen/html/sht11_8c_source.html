93 static uint8_t write_byte(uint8_t value)
94 {
95  uint8_t i;
96  uint8_t ack;
97 
98  SHT11_DATA_OUT;
99 
100  /* send value bit by bit to sht11 */
101  for (i = 0; i < 8; i++) {
102  if (value & BIT7) {
103  SHT11_DATA_HIGH;
104  xtimer_usleep(SHT11_DATA_WAIT);
105  }
106  else {
107  SHT11_DATA_LOW;
108  xtimer_usleep(SHT11_DATA_WAIT);
109  }
110 
111  /* trigger clock signal */
112  clk_signal();
113 
114  /* shift value to write next bit */
115  value = value << 1;
116  }
117 
118  /* wait for ack */
119  SHT11_DATA_IN;
120  xtimer_usleep(SHT11_CLK_WAIT);
121  ack = SHT11_DATA;
122 
123  clk_signal();
124 
125  return ack;
126 }

128 static uint8_t read_byte(uint8_t ack)
129 {
130  uint8_t i;
131  uint8_t value = 0;
132 
133  SHT11_DATA_IN;
134  xtimer_usleep(SHT11_DATA_WAIT);
135 
136  /* read value bit by bit */
137  for (i = 0; i < 8; i++) {
138  value = value << 1;
139  SHT11_SCK_HIGH;
140  xtimer_usleep(SHT11_CLK_WAIT);
141 
142  if (SHT11_DATA) {
143  /* increase data by one when DATA is high */
144  value++;
145  }
146 
147  SHT11_SCK_LOW;
148  xtimer_usleep(SHT11_CLK_WAIT);
149  }
150 
151  /* send ack if necessary */
152  SHT11_DATA_OUT;
153 
154  if (ack) {
155  SHT11_DATA_LOW;
156  xtimer_usleep(SHT11_DATA_WAIT);
157  }
158  else {
159  SHT11_DATA_HIGH;
160  xtimer_usleep(SHT11_DATA_WAIT);
161  }
162 
163  clk_signal();
164 
165  /* release data line */
166  SHT11_DATA_IN;
167 
168  return value;
169 }

225 static uint8_t measure(uint8_t *p_value, uint8_t *p_checksum, uint8_t mode)
226 {
227  uint8_t error = 0;
228  uint8_t ack = 1;
229  uint16_t i;
230 
231  transmission_start();
232  error = write_byte(mode);
233 
234  xtimer_usleep(1000);
235 
236  /* wait untile sensor has finished measurement or timeout */
237  for (i = 0; (i < SHT11_MEASURE_TIMEOUT) && (!error); i++) {
238  ack = SHT11_DATA;
239 
240  if (!ack) {
241  break;
242  }
243 
244  xtimer_usleep(1000);
245  }
246 
247  error += ack;
248 
249  /* read MSB */
250  *(p_value + 1) = read_byte(SHT11_ACK);
251  /* read LSB */
252  *(p_value) = read_byte(SHT11_ACK);
253  /* read checksum */
254  *p_checksum = read_byte(SHT11_NO_ACK);
255 
256  return (!error);
257 }

287 uint8_t sht11_read_sensor(sht11_val_t *value, sht11_mode_t mode)
288 {
289  uint8_t error = 0;
290  uint8_t checksum;
291  uint16_t humi_int, temp_int;
292 
293  /* Temperature arithmetic where S0(T) is read value
294  * T = D1 + D2 * S0(T) */
295  const float D1 = -39.6;
296  const float D2 = 0.01;
297 
298  /* Arithmetic for linear humdity where S0(RH) is read value
299  * HL = C1 + C2 * S0(RH) + C3 * SO(RH)^2 */
300  const float C1 = -4.0;
301  const float C2 = +0.0405;
302  const float C3 = -0.0000028;
303 
304  /* Arithmetic for temperature compesated relative humdity
305  * HT = (T-25) * ( T1 + T2 * SO(RH) ) + HL */
306  const float T1 = +0.01;
307  const float T2 = +0.00008;
308 
309  /* check for valid buffer */
310  if (value == NULL) {
311  return 0;
312  }
313 
314  value->temperature = 0;
315  value->relhum = 0;
316  value->relhum_temp = 0;
317 
318  mutex_lock(&sht11_mutex);
319  connection_reset();
320 
321  /* measure humidity */
322  if (mode & HUMIDITY) {
323  error += (!measure((uint8_t *) &humi_int, &checksum, SHT11_MEASURE_HUMI));
324  }
325 
326  /* measure temperature */
327  if (mode & TEMPERATURE) {
328  error += (!measure((uint8_t *) &temp_int, &checksum, SHT11_MEASURE_TEMP));
329  }
330 
331  /* break on error */
332  if (error != 0) {
333  connection_reset();
334  mutex_unlock(&sht11_mutex);
335  return 0;
336  }
337 
338  if (mode & TEMPERATURE) {
339  value->temperature = D1 + (D2 * ((float) temp_int)) + sht11_temperature_offset;
340  }
341 
342  if (mode & HUMIDITY) {
343  value->relhum = C1 + (C2 * ((float) humi_int)) + (C3 * ((float) humi_int) * ((float) humi_int));
344 
345  if (mode & TEMPERATURE) {
346  value->relhum_temp = (value->temperature - 25) * (T1 + (T2 * (float) humi_int)) + value->relhum;
347  }
348  }
349 
350  mutex_unlock(&sht11_mutex);
351  return 1;
352 }
