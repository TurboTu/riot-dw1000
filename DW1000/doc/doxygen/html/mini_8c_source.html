128 static void shift_next(uint8_t *cur, const uint8_t *next, uint32_t delay)
129 {
130  for (int i = 0; i < COLS; i++) {
131  for (int r = 0; r < ROWS; r++) {
132  for (int c = 0; c < (COLS - 1); c++) {
133  cur[(r * COLS) + c] = cur[(r * COLS) + c + 1];
134  }
135  cur[(r * COLS) + COLS - 1] = next[(r * COLS) + i];
136  }
137  mini_matrix_set_raw((uint8_t *)cur);
138  xtimer_usleep(delay);
139  }

142 static void refresh(void *arg, int channel)
143 {
144  (void)arg;
145  (void)channel;
146 
147  /* set next refresh */
148  timer_set(TIMER_DEV(2), 0, REFRESH);
149 
150  /* disable current row */
151  gpio_clear(rows[cur_row]);
152  /* goto next row */
153  cur_row = ((++cur_row) < ROWS_HW) ? cur_row : 0;
154  /* setup columns */
155  int base = (COLS_HW * cur_row);
156  for (int i = 0; i < COLS_HW; i++) {
157  gpio_write(cols[i], !(framebuf[base + i]));
158  }
159  /* and finally enable the new row */
160  gpio_set(rows[cur_row]);
161 }

163 void mini_matrix_init(void)
164 {
165  /* initialize rows */
166  for (int i = 0; i < ROWS_HW; i++) {
167  gpio_init(rows[i], GPIO_OUT);
168  gpio_clear(rows[i]);
169  }
170  /* initialize columns */
171  for (int i = 0; i < COLS_HW; i++) {
172  gpio_init(cols[i], GPIO_OUT);
173  gpio_set(cols[i]);
174  }
175  /* and finally initialize and start the refresh timer */
176  timer_init(TIMER_DEV(2), 1000000, refresh, NULL);
177  timer_set(TIMER_DEV(2), 0, REFRESH);
