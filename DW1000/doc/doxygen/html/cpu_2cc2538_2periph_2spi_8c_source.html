100 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
101 {
102  /* lock the bus */
103  mutex_lock(&locks[bus]);
104  /* power on device */
105  poweron(bus);
106  /* configure SCR clock field, data-width and mode */
107  dev(bus)->CR0 = 0;
108  dev(bus)->CPSR = (spi_clk_config[clk].cpsr);
109  dev(bus)->CR0 = ((spi_clk_config[clk].scr << 8) | mode | SSI_CR0_DSS(8));
110  /* enable SSI device */
111  dev(bus)->CR1 = SSI_CR1_SSE;
112 
113  return SPI_OK;
114 }
