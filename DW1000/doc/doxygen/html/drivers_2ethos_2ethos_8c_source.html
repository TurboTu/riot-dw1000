87 static void _handle_char(ethos_t *dev, char c)
88 {
89  switch (dev->frametype) {
90  case ETHOS_FRAME_TYPE_DATA:
91  case ETHOS_FRAME_TYPE_HELLO:
92  case ETHOS_FRAME_TYPE_HELLO_REPLY:
93  if (tsrb_add_one(&dev->inbuf, c) == 0) {
94  dev->framesize++;
95  } else {
96  //puts("lost frame");
97  dev->inbuf.reads = 0;
98  dev->inbuf.writes = 0;
99  _reset_state(dev);
100  }
101  break;
102 #ifdef USE_ETHOS_FOR_STDIO
103  case ETHOS_FRAME_TYPE_TEXT:
104  dev->framesize++;
105  isrpipe_write_one(&uart_stdio_isrpipe, c);
106 #endif
107  }
108 }

110 static void _end_of_frame(ethos_t *dev)
111 {
112  switch(dev->frametype) {
113  case ETHOS_FRAME_TYPE_DATA:
114  if (dev->framesize) {
115  dev->last_framesize = dev->framesize;
116  dev->netdev.event_callback((netdev2_t*) dev, NETDEV2_EVENT_ISR);
117  }
118  break;
119  case ETHOS_FRAME_TYPE_HELLO:
120  ethos_send_frame(dev, dev->mac_addr, 6, ETHOS_FRAME_TYPE_HELLO_REPLY);
121  /* fall through */
122  case ETHOS_FRAME_TYPE_HELLO_REPLY:
123  if (dev->framesize == 6) {
124  tsrb_get(&dev->inbuf, (char*)dev->remote_mac_addr, 6);
125  }
126  break;
127  }
128 
129  _reset_state(dev);
130 }

132 static void ethos_isr(void *arg, uint8_t c)
133 {
134  ethos_t *dev = (ethos_t *) arg;
135 
136  switch (dev->state) {
137  case WAIT_FRAMESTART:
138  if (c == ETHOS_FRAME_DELIMITER) {
139  _reset_state(dev);
140  dev->state = IN_FRAME;
141  }
142  break;
143  case IN_FRAME:
144  if (c == ETHOS_ESC_CHAR) {
145  dev->state = IN_ESCAPE;
146  }
147  else if (c == ETHOS_FRAME_DELIMITER) {
148  if (dev->framesize) {
149  _end_of_frame(dev);
150  }
151  }
152  else {
153  _handle_char(dev, c);
154  }
155  break;
156  case IN_ESCAPE:
157  switch (c) {
158  case (ETHOS_FRAME_DELIMITER ^ 0x20):
159  _handle_char(dev, ETHOS_FRAME_DELIMITER);
160  break;
161  case (ETHOS_ESC_CHAR ^ 0x20):
162  _handle_char(dev, ETHOS_ESC_CHAR);
163  break;
164  case (ETHOS_FRAME_TYPE_TEXT ^ 0x20):
165  dev->frametype = ETHOS_FRAME_TYPE_TEXT;
166  break;
167  case (ETHOS_FRAME_TYPE_HELLO ^ 0x20):
168  dev->frametype = ETHOS_FRAME_TYPE_HELLO;
169  break;
170  case (ETHOS_FRAME_TYPE_HELLO_REPLY ^ 0x20):
171  dev->frametype = ETHOS_FRAME_TYPE_HELLO_REPLY;
172  break;
173  }
174  dev->state = IN_FRAME;
175  break;
176  }
177 }

224 void ethos_send_frame(ethos_t *dev, const uint8_t *data, size_t len, unsigned frame_type)
225 {
226  uint8_t frame_delim = ETHOS_FRAME_DELIMITER;
227 
228  if (!irq_is_in()) {
229  mutex_lock(&dev->out_mutex);
230  }
231  else {
232  /* Send frame delimiter. This cancels the current frame,
233  * but enables in-ISR writes. */
234  uart_write(dev->uart, &frame_delim, 1);
235  }
236 
237  /* send frame delimiter */
238  uart_write(dev->uart, &frame_delim, 1);
239 
240  /* set frame type */
241  if (frame_type) {
242  uint8_t out[2] = { ETHOS_ESC_CHAR, (frame_type ^ 0x20) };
243  uart_write(dev->uart, out, 2);
244  }
245 
246  /* send frame content */
247  while(len--) {
248  _write_escaped(dev->uart, *(uint8_t*)data++);
249  }
250 
251  /* end of frame */
252  uart_write(dev->uart, &frame_delim, 1);
253 
254  if (!irq_is_in()) {
255  mutex_unlock(&dev->out_mutex);
256  }
257 }

259 static int _send(netdev2_t *netdev, const struct iovec *vector, unsigned count)
260 {
261  ethos_t * dev = (ethos_t *) netdev;
262  (void)dev;
263 
264  /* count total packet length */
265  size_t pktlen = iovec_count_total(vector, count);
266 
267  /* lock line in order to prevent multiple writes */
268  mutex_lock(&dev->out_mutex);
269 
270  /* send start-frame-delimiter */
271  uint8_t frame_delim = ETHOS_FRAME_DELIMITER;
272  uart_write(dev->uart, &frame_delim, 1);
273 
274  /* send iovec */
275  while(count--) {
276  size_t n = vector->iov_len;
277  uint8_t *ptr = vector->iov_base;
278  while(n--) {
279  _write_escaped(dev->uart, *ptr++);
280  }
281  vector++;
282  }
283 
284  uart_write(dev->uart, &frame_delim, 1);
285 
286  mutex_unlock(&dev->out_mutex);
287 
288  return pktlen;
289 }
