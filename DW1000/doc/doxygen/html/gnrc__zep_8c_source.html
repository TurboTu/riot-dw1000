90 kernel_pid_t gnrc_zep_init(gnrc_zep_t *dev, uint16_t src_port, ipv6_addr_t *dst,
91  uint16_t dst_port)
92 {
93 #if CPUID_LEN
94  uint8_t cpuid[CPUID_LEN];
95  uint32_t hash1, hash2;
96 #endif
97 
98  if (_pid != KERNEL_PID_UNDEF) {
99  DEBUG("zep: ZEP thread already running at pid=%" PRIkernel_pid "\n", _pid);
100  return -EEXIST;
101  }
102 
103  if (dev == NULL) {
104  DEBUG("zep: dev was NULL\n");
105  return -ENODEV;
106  }
107 
108  if ((dst == NULL) || (ipv6_addr_is_unspecified(dst))) {
109  DEBUG("zep: dst (%s) was NULL or unspecified\n", dst);
110  return -ENOTSUP;
111  }
112 
113  if (gnrc_netreg_lookup(GNRC_NETTYPE_UDP, src_port)) {
114  DEBUG("zep: port (%" PRIu16 ") already registered\n", src_port);
115  return -EADDRINUSE;
116  }
117 
118  dev->driver = (gnrc_netdev_driver_t *)&_zep_driver;
119  dev->chan = GNRC_ZEP_DEFAULT_CHANNEL;
120  dev->pan = byteorder_btols(byteorder_htons(GNRC_ZEP_DEFAULT_PANID));
121  dev->flags = GNRC_ZEP_FLAGS_USE_SRC_PAN;
122 #if CPUID_LEN
123  /* initialize dev->addr and dev->eui64 from cpuid if available */
124  cpuid_get(cpuid);
125 
126  hash1 = djb2_hash(cpuid, CPUID_LEN / 2);
127  dev->addr.u16 = (uint16_t)((hash1 >> 16) ^ (hash1 & 0xffff));
128 
129  if (CPUID_LEN % 2) {
130  hash2 = djb2_hash(cpuid + (CPUID_LEN / 2), (CPUID_LEN / 2) - 1);
131  }
132  else {
133  hash2 = djb2_hash(cpuid + (CPUID_LEN / 2), CPUID_LEN / 2);
134  }
135 
136  dev->eui64.u32[0] = hash1;
137  dev->eui64.u32[1] = hash2;
138 
139  dev->eui64.u8[7] &= 0xfe; /* set to unicast */
140  dev->eui64.u8[7] |= 0x02; /* set to locally administered */
141 #else
142  dev->addr = GNRC_ZEP_DEFAULT_ADDR_SHORT;
143  dev->eui64 = GNRC_ZEP_DEFAULT_ADDR_LONG;
144 #endif
145  DEBUG("zep: initialized radio parameters: chan: %" PRIu8 ", pan: 0x%04" PRIx16
146  "addr: 0x%04" PRIx16 ", eui64: %016" PRIx64 "\n", dev->chan,
147  dev->pan, byteorder_ltobs(dev->addr).u16,
148  byteorder_ltobll(dev->eui64).u64);
149 
150 #ifdef MODULE_GNRC_SIXLOWPAN
151  dev->proto = GNRC_NETTYPE_SIXLOWPAN;
152 #else
153  dev->proto = GNRC_NETTYPE_UNDEF;
154 #endif
155 
156  dev->seq = random_uint32();
157  dev->src_port = src_port;
158  dev->dst.u64[0] = dst->u64[0];
159  dev->dst.u64[1] = dst->u64[1];
160  dev->dst_port = dst_port;
161  dev->version = 2;
162  dev->lqi_mode = 1;
163 
164  _pid = thread_create(_rx_stack, GNRC_ZEP_STACK_SIZE, GNRC_ZEP_PRIO,
165  THREAD_CREATE_STACKTEST, _event_loop, dev, "zep_app");
166 
167  DEBUG("zep: started thread with PID %" PRIkernel_pid "\n", _pid);
168 
169  return _pid;
170 }

199 static int _send(gnrc_netdev_t *netdev, gnrc_pktsnip_t *pkt)
200 {
201  if ((netdev == NULL) || (netdev->driver != &_zep_driver)) {
202  DEBUG("zep: wrong device on sending\n");
203  gnrc_pktbuf_release(pkt);
204  return -ENODEV;
205  }
206 
207  gnrc_zep_t *dev = (gnrc_zep_t *)netdev;
208  gnrc_pktsnip_t *ptr, *new_pkt, *hdr;
209  gnrc_zep_hdr_t *zep;
210  size_t payload_len = gnrc_pkt_len(pkt->next), hdr_len, mhr_offset;
211  uint8_t mhr[IEEE802154_MAX_HDR_LEN], *data;
212  uint16_t fcs = 0;
213 
214  /* create 802.15.4 header */
215  hdr_len = _make_data_frame_hdr(dev, mhr, (gnrc_netif_hdr_t *)pkt->data);
216 
217  if (hdr_len == 0) {
218  DEBUG("zep: error on frame creation\n");
219  gnrc_pktbuf_release(pkt);
220  return -ENOMSG;
221  }
222 
223  new_pkt = _zep_hdr_build(dev, hdr_len + payload_len + IEEE802154_FCS_LEN, false);
224 
225  if (new_pkt == NULL) {
226  DEBUG("zep: could not allocate ZEP header in pktbuf\n");
227  gnrc_pktbuf_release(pkt);
228  return -ENOBUFS;
229  }
230 
231  zep = new_pkt->data;
232 
233  hdr = gnrc_udp_hdr_build(new_pkt, dev->src_port, dev->dst_port);
234 
235  if (hdr == NULL) {
236  DEBUG("zep: could not allocate UDP header in pktbuf\n");
237  gnrc_pktbuf_release(pkt);
238  gnrc_pktbuf_release(new_pkt);
239  return -ENOBUFS;
240  }
241 
242  new_pkt = hdr;
243 
244  hdr = gnrc_ipv6_hdr_build(new_pkt, NULL, &(dev->dst));
245 
246  if (hdr == NULL) {
247  DEBUG("zep: could not allocate IPv6 header in pktbuf\n");
248  gnrc_pktbuf_release(pkt);
249  gnrc_pktbuf_release(new_pkt);
250  return -ENOBUFS;
251  }
252 
253  new_pkt = hdr;
254 
255  mhr_offset = _zep_hdr_fill(dev, zep, payload_len + hdr_len + IEEE802154_FCS_LEN);
256 
257  if (mhr_offset == 0) {
258  DEBUG("zep: error filling ZEP header\n");
259  gnrc_pktbuf_release(pkt);
260  gnrc_pktbuf_release(new_pkt);
261  return -EINVAL;
262  }
263 
264  memcpy(((uint8_t *)zep) + mhr_offset, mhr, hdr_len);
265 
266  fcs = _calc_fcs(fcs, ((uint8_t *)zep) + mhr_offset, hdr_len);
267  data = ((uint8_t *)zep) + mhr_offset + hdr_len;
268  ptr = pkt->next;
269 
270  while (ptr != NULL) {
271  fcs = _calc_fcs(fcs, ptr->data, ptr->size);
272  memcpy(data, ptr->data, ptr->size);
273  data += ptr->size;
274  ptr = ptr->next;
275  }
276 
277  gnrc_pktbuf_release(pkt);
278 
279  DEBUG("zep: set frame FCS to 0x%04 " PRIx16 "\n", fcs);
280  _set_uint16_ptr((uint16_t *)data, byteorder_btols(byteorder_htons(fcs)).u16);
281 
282  if (!gnrc_netapi_dispatch_send(GNRC_NETTYPE_UDP, GNRC_NETREG_DEMUX_CTX_ALL, new_pkt)) {
283  DEBUG("zep: no UDP handler found: dropping packet\n");
284  gnrc_pktbuf_release(new_pkt);
285  return -ENOENT;
286  }
287 
288  return payload_len + hdr_len + IEEE802154_FCS_LEN;
289 }

321 static int _get(gnrc_netdev_t *netdev, netopt_t opt, void *value, size_t max_len)
322 {
323  gnrc_zep_t *dev = (gnrc_zep_t *)netdev;
324 
325  if (dev == NULL) {
326  return -ENODEV;
327  }
328 
329  switch (opt) {
330  case NETOPT_CHANNEL:
331  if (max_len < sizeof(uint16_t)) {
332  return -EOVERFLOW;
333  }
334 
335  _set_uint16_ptr(value, (uint16_t)dev->chan);
336  return sizeof(uint16_t);
337 
338  case NETOPT_ADDRESS:
339  if (max_len < sizeof(uint16_t)) {
340  return -EOVERFLOW;
341  }
342 
343  _set_uint16_ptr(value, byteorder_ltobs(dev->addr).u16);
344  return sizeof(uint16_t);
345 
346  case NETOPT_ADDRESS_LONG:
347  if (max_len < sizeof(uint64_t)) {
348  return -EOVERFLOW;
349  }
350 
351  _set_uint64_ptr(value, byteorder_ltobll(dev->eui64).u64);
352  return sizeof(uint64_t);
353 
354  case NETOPT_ADDR_LEN:
355  if (max_len < sizeof(uint16_t)) {
356  return -EOVERFLOW;
357  }
358 
359  if (dev->flags & GNRC_ZEP_FLAGS_DST_ADDR_LONG) {
360  _set_uint16_ptr(value, IEEE802154_LONG_ADDRESS_LEN);
361  }
362  else {
363  _set_uint16_ptr(value, IEEE802154_SHORT_ADDRESS_LEN);
364  }
365 
366  return sizeof(uint16_t);
367 
368  case NETOPT_SRC_LEN:
369  if (max_len < sizeof(uint16_t)) {
370  return -EOVERFLOW;
371  }
372 
373  if (dev->flags & GNRC_ZEP_FLAGS_SRC_ADDR_LONG) {
374  _set_uint16_ptr(value, IEEE802154_LONG_ADDRESS_LEN);
375  }
376  else {
377  _set_uint16_ptr(value, IEEE802154_SHORT_ADDRESS_LEN);
378  }
379 
380  return sizeof(uint16_t);
381 
382  case NETOPT_PROTO:
383  if (max_len < sizeof(gnrc_nettype_t)) {
384  return -EOVERFLOW;
385  }
386 
387  *((gnrc_nettype_t *)value) = dev->proto;
388  return sizeof(gnrc_nettype_t);
389 
390  case NETOPT_NID:
391  if (max_len < sizeof(uint16_t)) {
392  return -EOVERFLOW;
393  }
394 
395  _set_uint16_ptr(value, byteorder_ltobs(dev->pan).u16);
396  return sizeof(uint16_t);
397 
398  case NETOPT_IPV6_IID:
399  if (max_len < sizeof(eui64_t)) {
400  return -EOVERFLOW;
401  }
402  if (dev->flags & GNRC_ZEP_FLAGS_SRC_ADDR_LONG) {
403  uint64_t addr = byteorder_ltobll(dev->eui64).u64;
404  ieee802154_get_iid(value, (uint8_t *)&addr, IEEE802154_LONG_ADDRESS_LEN);
405  }
406  else {
407  uint16_t addr = byteorder_ltobs(dev->addr).u16;
408  ieee802154_get_iid(value, (uint8_t *)&addr, IEEE802154_SHORT_ADDRESS_LEN);
409  }
410  return sizeof(eui64_t);
411 
412  case NETOPT_MAX_PACKET_SIZE:
413  if (max_len < sizeof(uint16_t)) {
414  return -EOVERFLOW;
415  }
416 
417  _set_uint16_ptr(value, GNRC_ZEP_MAX_PKT_LENGTH);
418  return sizeof(uint16_t);
419 
420  case NETOPT_AUTOACK:
421  if (max_len < sizeof(netopt_enable_t)) {
422  return -EOVERFLOW;
423  }
424 
425  _set_flag_ptr(value, dev->flags, GNRC_ZEP_FLAGS_AUTOACK);
426  return sizeof(uint16_t);
427 
428  default:
429  return -ENOTSUP;
430  }
431 }

433 static int _set(gnrc_netdev_t *netdev, netopt_t opt, void *value, size_t value_len)
434 {
435  gnrc_zep_t *dev = (gnrc_zep_t *)netdev;
436 
437  if (dev == NULL) {
438  return -ENODEV;
439  }
440 
441  switch (opt) {
442  case NETOPT_CHANNEL:
443  if (value_len < sizeof(uint16_t)) {
444  return -EOVERFLOW;
445  }
446 
447  if ((*_get_uint16_ptr(value) < GNRC_ZEP_MIN_CHANNEL) ||
448  (*_get_uint16_ptr(value)) > GNRC_ZEP_MAX_CHANNEL) {
449  return -ENOTSUP;
450  }
451 
452  dev->chan = *_get_uint16_ptr(value);
453  return sizeof(uint16_t);
454 
455  case NETOPT_ADDRESS:
456  if (value_len < sizeof(be_uint16_t)) {
457  return -EOVERFLOW;
458  }
459  else {
460  be_uint16_t *val = value;
461 
462  dev->addr = byteorder_btols(*val);
463  return sizeof(be_uint16_t);
464  }
465 
466  case NETOPT_ADDRESS_LONG:
467  if (value_len < sizeof(be_uint64_t)) {
468  return -EOVERFLOW;
469  }
470  else {
471  be_uint64_t *val = value;
472 
473  dev->eui64 = byteorder_btolll(*val);
474  return sizeof(be_uint64_t);
475  }
476 
477  case NETOPT_ADDR_LEN:
478  if (value_len < sizeof(uint16_t)) {
479  return -EOVERFLOW;
480  }
481 
482  switch (*_get_uint16_ptr(value)) {
483  case IEEE802154_SHORT_ADDRESS_LEN:
484  dev->flags &= ~GNRC_ZEP_FLAGS_DST_ADDR_LONG;
485  break;
486 
487  case IEEE802154_LONG_ADDRESS_LEN:
488  dev->flags |= GNRC_ZEP_FLAGS_DST_ADDR_LONG;
489  break;
490 
491  default:
492  return -ENOTSUP;
493  }
494 
495  return sizeof(uint16_t);
496 
497  case NETOPT_SRC_LEN:
498  if (value_len < sizeof(uint16_t)) {
499  return -EOVERFLOW;
500  }
501 
502  switch (*_get_uint16_ptr(value)) {
503  case IEEE802154_SHORT_ADDRESS_LEN:
504  dev->flags &= ~GNRC_ZEP_FLAGS_SRC_ADDR_LONG;
505  break;
506 
507  case IEEE802154_LONG_ADDRESS_LEN:
508  dev->flags |= GNRC_ZEP_FLAGS_SRC_ADDR_LONG;
509  break;
510 
511  default:
512  return -ENOTSUP;
513  }
514 
515  return sizeof(uint16_t);
516 
517  case NETOPT_NID:
518  if (value_len < sizeof(be_uint16_t)) {
519  return -EOVERFLOW;
520  }
521  else {
522  be_uint16_t *val = value;
523 
524  dev->pan = byteorder_btols(*val);
525  return sizeof(be_uint16_t);
526  }
527 
528  case NETOPT_AUTOACK:
529  if (value_len < sizeof(netopt_enable_t)) {
530  return -EOVERFLOW;
531  }
532 
533  _set_flag_ptr(value, dev->flags, GNRC_ZEP_FLAGS_AUTOACK);
534  return sizeof(uint16_t);
535 
536  default:
537  return -ENOTSUP;
538  }
539 }

555 void *_event_loop(void *args)
556 {
557  msg_t msg, ack, msg_q[GNRC_ZEP_MSG_QUEUE_SIZE];
558  gnrc_netdev_t *dev = (gnrc_netdev_t *)args;
559  gnrc_netapi_opt_t *opt;
560  gnrc_netreg_entry_t my_reg = GNRC_NETREG_ENTRY_INIT_PID(((gnrc_zep_t *)args)->src_port,
561  sched_active_pid);
562 
563  msg_init_queue(msg_q, GNRC_ZEP_MSG_QUEUE_SIZE);
564 
565  gnrc_netreg_register(GNRC_NETTYPE_UDP, &my_reg);
566 
567  while (1) {
568  msg_receive(&msg);
569 
570  switch (msg.type) {
571  case GNRC_NETAPI_MSG_TYPE_RCV:
572  DEBUG("zep: GNRC_NETAPI_MSG_TYPE_RCV\n");
573  ringbuffer_add(&_rx_buf, (void*)&msg.content.ptr,
574  sizeof(gnrc_pktsnip_t *));
575  ack.type = GNRC_NETDEV_MSG_TYPE_EVENT;
576  ack.content.value = _EVENT_RX_STARTED;
577  msg_send_int(&ack, dev->mac_pid);
578  break;
579 
580  case GNRC_NETAPI_MSG_TYPE_SND:
581  DEBUG("zep: GNRC_NETAPI_MSG_TYPE_SND\n");
582  _send(dev, msg.content.ptr);
583  break;
584 
585  case GNRC_NETAPI_MSG_TYPE_GET:
586  DEBUG("zep: GNRC_NETAPI_MSG_TYPE_GET\n");
587  opt = msg.content.ptr;
588  ack.type = GNRC_NETAPI_MSG_TYPE_ACK;
589  ack.content.value = _get(dev, opt->opt, opt->data, opt->data_len);
590  msg_reply(&msg, &ack);
591  break;
592 
593  case GNRC_NETAPI_MSG_TYPE_SET:
594  DEBUG("zep: GNRC_NETAPI_MSG_TYPE_SET\n");
595  opt = msg.content.ptr;
596  ack.type = GNRC_NETAPI_MSG_TYPE_ACK;
597  ack.content.value = _set(dev, opt->opt, opt->data, opt->data_len);
598  msg_reply(&msg, &ack);
599  break;
600 
601  default:
602  DEBUG("udp: received unidentified message 0x%04" PRIx16 "\n",
603  msg.type);
604  break;
605  }
606  }
607 
608  return NULL;
609 }

611 static gnrc_pktsnip_t *_zep_hdr_build(gnrc_zep_t *dev, size_t size, bool ack)
612 {
613  gnrc_pktsnip_t *zep;
614 
615  switch (dev->version) {
616  case 1:
617  DEBUG("zep: Build ZEPv1 data header in pktbuf\n");
618  zep = gnrc_pktbuf_add(NULL, NULL, sizeof(gnrc_zep_v1_hdr_t) + size,
619  GNRC_NETTYPE_UNDEF);
620  break;
621 
622  case 2:
623  if (ack) {
624  DEBUG("zep: Build ZEPv2 ACK header in pktbuf\n");
625  zep = gnrc_pktbuf_add(NULL, NULL, sizeof(gnrc_zep_v2_ack_hdr_t) + size,
626  GNRC_NETTYPE_UNDEF);
627  }
628  else {
629  DEBUG("zep: Build ZEPv2 data header in pktbuf\n");
630  zep = gnrc_pktbuf_add(NULL, NULL, sizeof(gnrc_zep_v2_data_hdr_t) + size,
631  GNRC_NETTYPE_UNDEF);
632  }
633 
634  break;
635 
636  default:
637  DEBUG("zep: malconfigured version: %" PRIu8 "\n", dev->version);
638  return NULL;

861 /* TODO: Generalize and move all below to ieee802154 */
862 static size_t _make_data_frame_hdr(gnrc_zep_t *dev, uint8_t *buf,
863  gnrc_netif_hdr_t *hdr)
864 {
865  int pos = 0;
866 
867  /* we are building a data frame here */
868  buf[0] = IEEE802154_FCF_TYPE_DATA;
869  buf[1] = 0x88; /* use short src and dst addresses as starting point */
870 
871  /* if AUTOACK is enabled, then we also expect ACKs for this packet */
872  if (dev->flags & GNRC_ZEP_FLAGS_AUTOACK) {
873  buf[0] |= IEEE802154_FCF_ACK_REQ;
874  }
875 
876  /* fill in destination PAN ID */
877  pos = 3;
878  buf[pos++] = dev->pan.u8[0];
879  buf[pos++] = dev->pan.u8[1];
880 
881  /* fill in destination address */
882  if (hdr->flags &
883  (GNRC_NETIF_HDR_FLAGS_BROADCAST | GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
884  buf[pos++] = 0xff;
885  buf[pos++] = 0xff;
886  }
887  else if (hdr->dst_l2addr_len == IEEE802154_SHORT_ADDRESS_LEN) {
888  uint8_t *dst_addr = gnrc_netif_hdr_get_dst_addr(hdr);
889  buf[pos++] = dst_addr[1];
890  buf[pos++] = dst_addr[0];
891  }
892  else if (hdr->dst_l2addr_len == IEEE802154_LONG_ADDRESS_LEN) {
893  buf[1] |= 0x04;
894  uint8_t *dst_addr = gnrc_netif_hdr_get_dst_addr(hdr);
895 
896  for (int i = (IEEE802154_LONG_ADDRESS_LEN - 1); i >= 0; i--) {
897  buf[pos++] = dst_addr[i];
898  }
899  }
900  else {
901  /* unsupported address length */
902  return 0;
903  }
904 
905  /* fill in source PAN ID (if applicable) */
906  if (dev->flags & GNRC_ZEP_FLAGS_USE_SRC_PAN) {
907  buf[pos++] = dev->pan.u8[0];
908  buf[pos++] = dev->pan.u8[1];
909  }
910  else {
911  buf[0] |= IEEE802154_FCF_PAN_COMP;
912  }
913 
914  /* fill in source address */
915  if (dev->flags & GNRC_ZEP_FLAGS_SRC_ADDR_LONG) {
916  buf[1] |= 0x40;
917  memcpy(&(buf[pos]), &dev->eui64, IEEE802154_LONG_ADDRESS_LEN);
918  pos += IEEE802154_LONG_ADDRESS_LEN;
919  }
920  else {
921  memcpy(&(buf[pos]), &dev->addr, IEEE802154_SHORT_ADDRESS_LEN);
922  pos += IEEE802154_SHORT_ADDRESS_LEN;
923  }
924 
925  /* set sequence number */
926  buf[2] = dev->seq++;
927  /* return actual header length */
928  return pos;
929 }
