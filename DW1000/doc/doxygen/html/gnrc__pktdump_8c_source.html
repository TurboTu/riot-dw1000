100 static void _dump(gnrc_pktsnip_t *pkt)
101 {
102  int snips = 0;
103  int size = 0;
104  gnrc_pktsnip_t *snip = pkt;
105 
106  while (snip != NULL) {
107  printf("~~ SNIP %2i - size: %3u byte, type: ", snips,
108  (unsigned int)snip->size);
109  _dump_snip(snip);
110  ++snips;
111  size += snip->size;
112  snip = snip->next;
113  }
114 
115  printf("~~ PKT - %2i snips, total size: %3i byte\n", snips, size);
116  gnrc_pktbuf_release(pkt);
117 }

119 static void *_eventloop(void *arg)
120 {
121  (void)arg;
122  msg_t msg, reply;
123  msg_t msg_queue[GNRC_PKTDUMP_MSG_QUEUE_SIZE];
124 
125  /* setup the message queue */
126  msg_init_queue(msg_queue, GNRC_PKTDUMP_MSG_QUEUE_SIZE);
127 
128  reply.content.value = (uint32_t)(-ENOTSUP);
129  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
130 
131  while (1) {
132  msg_receive(&msg);
133 
134  switch (msg.type) {
135  case GNRC_NETAPI_MSG_TYPE_RCV:
136  puts("PKTDUMP: data received:");
137  _dump(msg.content.ptr);
138  break;
139  case GNRC_NETAPI_MSG_TYPE_SND:
140  puts("PKTDUMP: data to send:");
141  _dump(msg.content.ptr);
142  break;
143  case GNRC_NETAPI_MSG_TYPE_GET:
144  case GNRC_NETAPI_MSG_TYPE_SET:
145  msg_reply(&msg, &reply);
146  break;
147  default:
148  puts("PKTDUMP: received something unexpected");
149  break;
150  }
151  }
152 
153  /* never reached */
154  return NULL;
155 }
