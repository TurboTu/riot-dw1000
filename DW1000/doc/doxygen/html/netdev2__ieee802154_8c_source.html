67  case NETOPT_SRC_LEN:

135 int netdev2_ieee802154_set(netdev2_ieee802154_t *dev, netopt_t opt, void *value,
136  size_t len)
137 {
138  int res = -ENOTSUP;
139 
140  switch (opt) {
141  case NETOPT_CHANNEL:
142  {
143  assert(len == sizeof(uint16_t));
144  uint16_t chan = *((uint16_t *)value);
145  /* real validity needs to be checked by device, since sub-GHz and
146  * 2.4 GHz band radios have different legal values. Here we only
147  * check that it fits in an 8-bit variabl*/
148  assert(chan <= UINT8_MAX);
149  dev->chan = chan;
150  res = sizeof(uint16_t);
151  break;
152  }
153  case NETOPT_ADDRESS:
154  assert(len <= sizeof(dev->short_addr));
155  memset(dev->short_addr, 0, sizeof(dev->short_addr));
156  memcpy(dev->short_addr, value, len);
157  res = sizeof(dev->short_addr);
158  break;
159  case NETOPT_ADDRESS_LONG:
160  assert(len <= sizeof(dev->long_addr));
161  memset(dev->long_addr, 0, sizeof(dev->long_addr));
162  memcpy(dev->long_addr, value, len);
163  res = sizeof(dev->long_addr);
164  break;
165  case NETOPT_ADDR_LEN:
166  case NETOPT_SRC_LEN:
167  assert(len == sizeof(uint16_t));
168  switch ((*(uint16_t *)value)) {
169  case IEEE802154_SHORT_ADDRESS_LEN:
170  dev->flags &= ~NETDEV2_IEEE802154_SRC_MODE_LONG;
171  break;
172  case IEEE802154_LONG_ADDRESS_LEN:
173  dev->flags |= NETDEV2_IEEE802154_SRC_MODE_LONG;
174  break;
175  default:
176  res = -EAFNOSUPPORT;
177  break;
178  }
179  res = sizeof(uint16_t);
180  break;
181  case NETOPT_NID:
182  assert(len == sizeof(dev->pan));
183  dev->pan = *((uint16_t *)value);
184  res = sizeof(dev->pan);
185  break;
186  case NETOPT_ACK_REQ:
187  if ((*(bool *)value)) {
188  dev->flags |= NETDEV2_IEEE802154_ACK_REQ;
189  }
190  else {
191  dev->flags &= ~NETDEV2_IEEE802154_ACK_REQ;
192  }
193  res = sizeof(uint16_t);
194  break;
195  case NETOPT_RAWMODE:
196  if ((*(bool *)value)) {
197  dev->flags |= NETDEV2_IEEE802154_RAW;
198  }
199  else {
200  dev->flags &= ~NETDEV2_IEEE802154_RAW;
201  }
202  res = sizeof(uint16_t);
203  break;
204 #ifdef MODULE_GNRC
205  case NETOPT_PROTO:
206  assert(len == sizeof(gnrc_nettype_t));
207  dev->proto = *((gnrc_nettype_t *)value);
208  res = sizeof(gnrc_nettype_t);
209  break;
210 #endif
211  default:
212  break;
213  }
214  return res;
215 }
