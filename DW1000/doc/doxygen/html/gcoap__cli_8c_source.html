45 /*
46  * Response callback.
47  */
48 static void _resp_handler(unsigned req_state, coap_pkt_t* pdu)
49 {
50  if (req_state == GCOAP_MEMO_TIMEOUT) {
51  printf("gcoap: timeout for msg ID %02u\n", coap_get_id(pdu));
52  return;
53  }
54  else if (req_state == GCOAP_MEMO_ERR) {
55  printf("gcoap: error in response\n");
56  return;
57  }
58 
59  char *class_str = (coap_get_code_class(pdu) == COAP_CLASS_SUCCESS)
60  ? "Success" : "Error";
61  printf("gcoap: response %s, code %1u.%02u", class_str,
62  coap_get_code_class(pdu),
63  coap_get_code_detail(pdu));
64  if (pdu->payload_len) {
65  if (pdu->content_type == COAP_FORMAT_TEXT
66  || pdu->content_type == COAP_FORMAT_LINK
67  || coap_get_code_class(pdu) == COAP_CLASS_CLIENT_FAILURE
68  || coap_get_code_class(pdu) == COAP_CLASS_SERVER_FAILURE) {
69  /* Expecting diagnostic payload in failure cases */
70  printf(", %u bytes\n%.*s\n", pdu->payload_len, pdu->payload_len,
71  (char *)pdu->payload);
72  }
73  else {
74  printf(", %u bytes\n", pdu->payload_len);
75  od_hex_dump(pdu->payload, pdu->payload_len, OD_WIDTH_DEFAULT);
76  }
77  }
78  else {
79  printf(", empty payload\n");
80  }
81 }

96 static size_t _send(uint8_t *buf, size_t len, char *addr_str, char *port_str)
97 {
98  ipv6_addr_t addr;
99  size_t bytes_sent;
100  sock_udp_ep_t remote;
101 
102  remote.family = AF_INET6;
103  remote.netif = SOCK_ADDR_ANY_NETIF;
104 
105  /* parse destination address */
106  if (ipv6_addr_from_str(&addr, addr_str) == NULL) {
107  puts("gcoap_cli: unable to parse destination address");
108  return 0;
109  }
110  memcpy(&remote.addr.ipv6[0], &addr.u8[0], sizeof(addr.u8));
111 
112  /* parse port */
113  remote.port = (uint16_t)atoi(port_str);
114  if (remote.port == 0) {
115  puts("gcoap_cli: unable to parse destination port");
116  return 0;
117  }
118 
119  bytes_sent = gcoap_req_send2(buf, len, &remote, _resp_handler);
120  if (bytes_sent > 0) {
121  req_count++;
122  }
123  return bytes_sent;
124 }

126 int gcoap_cli_cmd(int argc, char **argv)
127 {
128  /* Ordered like the RFC method code numbers, but off by 1. GET is code 0. */
129  char *method_codes[] = {"get", "post", "put"};
130  uint8_t buf[GCOAP_PDU_BUF_SIZE];
131  coap_pkt_t pdu;
132  size_t len;
133 
134  if (argc == 1) {
135  /* show help for main commands */
136  goto end;
137  }
138 
139  for (size_t i = 0; i < sizeof(method_codes) / sizeof(char*); i++) {
140  if (strcmp(argv[1], method_codes[i]) == 0) {
141  if (argc == 5 || argc == 6) {
142  if (argc == 6) {
143  gcoap_req_init(&pdu, &buf[0], GCOAP_PDU_BUF_SIZE, i+1, argv[4]);
144  memcpy(pdu.payload, argv[5], strlen(argv[5]));
145  len = gcoap_finish(&pdu, strlen(argv[5]), COAP_FORMAT_TEXT);
146  }
147  else {
148  len = gcoap_request(&pdu, &buf[0], GCOAP_PDU_BUF_SIZE, i+1,
149  argv[4]);
150  }
151  printf("gcoap_cli: sending msg ID %u, %u bytes\n", coap_get_id(&pdu),
152  (unsigned) len);
153  if (!_send(&buf[0], len, argv[2], argv[3])) {
154  puts("gcoap_cli: msg send failed");
155  }
156  return 0;
157  }
158  else {
159  printf("usage: %s <get|post|put> <addr> <port> <path> [data]\n",
160  argv[0]);
161  return 1;
162  }
163  }
164  }
165 
166  if (strcmp(argv[1], "info") == 0) {
167  if (argc == 2) {
168  uint8_t open_reqs;
169  gcoap_op_state(&open_reqs);
170 
171  printf("CoAP server is listening on port %u\n", GCOAP_PORT);
172  printf(" CLI requests sent: %u\n", req_count);
173  printf("CoAP open requests: %u\n", open_reqs);
174  return 0;
175  }
176  }
177 
178  end:
179  printf("usage: %s <get|post|put|info>\n", argv[0]);
180  return 1;
181 }
