38  l3g4200d_mode_t mode, l3g4200d_scale_t scale)

42  /* write device descriptor */
43  dev->i2c = i2c;
44  dev->addr = address;
45  dev->int1 = int1_pin;
46  dev->int2 = int2_pin;

48  /* set scale */
49  switch (scale) {
50  case L3G4200D_SCALE_250DPS:
51  dev->scale = 250;
52  break;
53  case L3G4200D_SCALE_500DPS:
54  dev->scale = 500;
55  break;
56  case L3G4200D_SCALE_2000DPS:
57  dev->scale = 2000;
58  break;
59  default:
60  dev->scale = 500;
61  break;

63 
64  /* acquire exclusive access to the bus. */
65  i2c_acquire(dev->i2c);
66  /* initialize the I2C bus */
67  if (i2c_init_master(i2c, I2C_SPEED) < 0) {
68  /* Release the bus for other threads. */

70  return -1;

72  /* configure CTRL_REG1 */
73  tmp = ((mode & 0xf) << L3G4200D_CTRL1_MODE_POS) | L3G4200D_CTRL1_ALLON;
74  if (i2c_write_reg(dev->i2c, dev->addr, L3G4200D_REG_CTRL1, tmp) != 1) {
75  i2c_release(dev->i2c);
76  return -1;
77  }
78  tmp = ((scale & 0x3) << L3G4200D_CTRL4_FS_POS) | L3G4200D_CTRL4_BDU;
79  if (i2c_write_reg(dev->i2c, dev->addr, L3G4200D_REG_CTRL4, tmp) != 1) {

87 int l3g4200d_read(l3g4200d_t *dev, l3g4200d_data_t *data)
88 {
89  uint8_t tmp[6];
90  int16_t res;
91 
92  i2c_acquire(dev->i2c);
93  /* get acceleration in x direction */
94  i2c_read_regs(dev->i2c, dev->addr, L3G4200D_REG_OUT_X_L | L3G4200D_AUTOINC, tmp, 6);
95  i2c_release(dev->i2c);
96 
97  /* parse and normalize data into result vector */
98  res = (tmp[1] << 8) | tmp[0];
99  data->acc_x = (int16_t)((dev->scale * res) / MAX_VAL);
100  res = (tmp[3] << 8) | tmp[2];
101  data->acc_y = (int16_t)((dev->scale * res) / MAX_VAL);
102  res = (tmp[5] << 8) | tmp[4];
103  data->acc_z = (int16_t)((dev->scale * res) / MAX_VAL);
104  return 0;
105 }
