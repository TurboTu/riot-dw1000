109  }
110 
111  /* parse the given SPI device */
112  dev = atoi(argv[1]);
113  if (dev < 0 || dev >= SPI_NUMOF) {
114  puts("error: invalid SPI device specified");
115  return 1;
116  }
117  spiconf.dev = SPI_DEV(dev);
118 
119  /* parse the SPI mode */
120  mode = atoi(argv[2]);
121  switch (mode) {
122  case 0: spiconf.mode = SPI_MODE_0; break;
123  case 1: spiconf.mode = SPI_MODE_1; break;
124  case 2: spiconf.mode = SPI_MODE_2; break;
125  case 3: spiconf.mode = SPI_MODE_3; break;
126  default:
127  puts("error: invalid SPI mode specified");
128  return 1;
129  }
130 
131  /* parse the targeted clock speed */
132  clk = atoi(argv[3]);
133  switch (clk) {
134  case 0: spiconf.clk = SPI_CLK_100KHZ; break;
135  case 1: spiconf.clk = SPI_CLK_400KHZ; break;
136  case 2: spiconf.clk = SPI_CLK_1MHZ; break;
137  case 3: spiconf.clk = SPI_CLK_5MHZ; break;
138  case 4: spiconf.clk = SPI_CLK_10MHZ; break;
139  default:
140  puts("error: invalid bus speed specified");
141  return 1;
142  }
143 
144  /* parse chip select port and pin */
145  port = atoi(argv[4]);
146  pin = atoi(argv[5]);
147  if (pin < 0 || port < -1) {
148  puts("error: invalid CS port/pin combination specified");
149  }
150  if (port == -1) { /* hardware chip select line */
151  spiconf.cs = SPI_HWCS(pin);
152  }
153  else {
154  spiconf.cs = (spi_cs_t)GPIO_PIN(port, pin);
155  }
156 
157  /* test setup */
158  tmp = spi_init_cs(spiconf.dev, spiconf.cs);
159  if (tmp != SPI_OK) {
160  puts("error: unable to initialize the given chip select line");
161  return 1;
162  }
163  tmp = spi_acquire(spiconf.dev, spiconf.cs, spiconf.mode, spiconf.clk);
164  if (tmp == SPI_NOMODE) {
165  puts("error: given SPI mode is not supported");
166  return 1;
167  }
168  else if (tmp == SPI_NOCLK) {
169  puts("error: targeted clock speed is not supported");
170  return 1;
171  }
172  else if (tmp != SPI_OK) {
173  puts("error: unable to acquire bus with given parameters");
174  return 1;
175  }
176  spi_release(spiconf.dev);
177 
178  printf("SPI_DEV(%i) initialized: mode: %i, clk: %i, cs_port: %i, cs_pin: %i\n",
179  dev, mode, clk, port, pin);
180 
181  return 0;
182 }

184 int cmd_transfer(int argc, char **argv)
185 {
186  size_t len;
187 
188  if (argc < 2) {
189  printf("usage: %s <data>\n", argv[0]);
190  return 1;
191  }
192 
193  if (spiconf.dev == SPI_UNDEF) {
194  puts("error: SPI is not initialized, please initialize bus first");
195  return 1;
196  }
197 
198  /* get bus access */
199  if (spi_acquire(spiconf.dev, spiconf.cs,
200  spiconf.mode, spiconf.clk) != SPI_OK) {
201  puts("error: unable to acquire the SPI bus");
202  return 1;
203  }
204 
205  /* transfer data */
206  len = strlen(argv[1]);
207  memset(buf, 0, sizeof(buf));
208  spi_transfer_bytes(spiconf.dev, spiconf.cs, false, argv[1], buf, len);
209 
210  /* release the bus */
211  spi_release(spiconf.dev);
212 
213  /* print results */
214  print_bytes("Sent bytes", (uint8_t *)argv[1], len);
215  print_bytes("Received bytes", buf, len);
216 
217  return 0;
218 }
