257 
258  putchar('\n');

260  gnrc_rpl_dodag_t *dodag = NULL;
261  char addr_str[IPV6_ADDR_MAX_STR_LEN];
262  int8_t cleanup;
263  uint64_t tc, ti, xnow = xtimer_now_usec64();
264 
265  for (uint8_t i = 0; i < GNRC_RPL_INSTANCES_NUMOF; ++i) {
266  if (gnrc_rpl_instances[i].state == 0) {
267  continue;
268  }
269 
270  dodag = &gnrc_rpl_instances[i].dodag;
271 
272  printf("instance [%d | Iface: %" PRIkernel_pid " | mop: %d | ocp: %d | mhri: %d | mri %d]\n",
273  gnrc_rpl_instances[i].id, dodag->iface,
274  gnrc_rpl_instances[i].mop, gnrc_rpl_instances[i].of->ocp,
275  gnrc_rpl_instances[i].min_hop_rank_inc, gnrc_rpl_instances[i].max_rank_inc);
276 
277  tc = (((uint64_t) dodag->trickle.msg_callback_timer.long_target << 32)
278  | dodag->trickle.msg_callback_timer.target) - xnow;
279  tc = (int64_t) tc < 0 ? 0 : tc / US_PER_SEC;
280 
281  ti = (((uint64_t) dodag->trickle.msg_interval_timer.long_target << 32)
282  | dodag->trickle.msg_interval_timer.target) - xnow;
283  ti = (int64_t) ti < 0 ? 0 : ti / US_PER_SEC;
284 
285  cleanup = dodag->instance->cleanup < 0 ? 0 : dodag->instance->cleanup;
286 
287  printf("\tdodag [%s | R: %d | OP: %s | PIO: %s | CL: %ds | "
288  "TR(I=[%d,%d], k=%d, c=%d, TC=%" PRIu32 "s, TI=%" PRIu32 "s)]\n",
289  ipv6_addr_to_str(addr_str, &dodag->dodag_id, sizeof(addr_str)),
290  dodag->my_rank, (dodag->node_status == GNRC_RPL_LEAF_NODE ? "Leaf" : "Router"),
291  ((dodag->dio_opts & GNRC_RPL_REQ_DIO_OPT_PREFIX_INFO) ? "on" : "off"),
292  (int) cleanup, (1 << dodag->dio_min), dodag->dio_interval_doubl, dodag->trickle.k,
293  dodag->trickle.c, (uint32_t) (tc & 0xFFFFFFFF), (uint32_t) (ti & 0xFFFFFFFF));
294 
295 #ifdef MODULE_GNRC_RPL_P2P
296  if (dodag->instance->mop == GNRC_RPL_P2P_MOP) {
297  gnrc_rpl_p2p_ext_t *p2p_ext = gnrc_rpl_p2p_ext_get(dodag);
298  printf("\tP2P-Ext [%s | HBH: %s | R: %s | L: %us]\n",
299  ipv6_addr_to_str(addr_str, &p2p_ext->target, sizeof(addr_str)),
300  p2p_ext->hop_by_hop ? "True" : "False",
301  p2p_ext->reply ? "True" : "False", p2p_ext->lifetime_sec);
302  }
303 #endif
304 
305  gnrc_rpl_parent_t *parent;
306  LL_FOREACH(gnrc_rpl_instances[i].dodag.parents, parent) {
307  printf("\t\tparent [addr: %s | rank: %d | lifetime: %" PRIu32 "s]\n",
308  ipv6_addr_to_str(addr_str, &parent->addr, sizeof(addr_str)),
309  parent->rank, ((int32_t) (parent->lifetime - (((uint32_t) xnow / US_PER_SEC))))
310  < 0 ? 0 : (parent->lifetime - ((uint32_t) xnow / US_PER_SEC)));
311  }
312  }
313  return 0;
314 }
