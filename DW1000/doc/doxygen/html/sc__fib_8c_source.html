80 static void _fib_add(const char *dest, const char *next, kernel_pid_t pid, uint32_t lifetime)
81 {
82  uint32_t prefix = 0;
83  /* Get the prefix length */
84  size_t i = 0;
85  for (i = strlen(dest); i > 0; --i) {
86  if (dest[i] == '/') {
87  prefix = atoi(&dest[i+1]);
88  break;
89  }
90  if (dest[i] == ':' || dest[i] == '.') {
91  i = strlen(dest);
92  break;
93  }
94  }
95 
96  size_t dst_size = (i+1);
97  unsigned char dst_arr[dst_size];
98  memset(dst_arr, 0, dst_size);
99  memcpy(dst_arr, dest, i);
100  unsigned char *dst = &dst_arr[0];
101  uint32_t dst_flags = 0;
102 
103  unsigned char *nxt = (unsigned char *)next;
104  size_t nxt_size = (strlen(next));
105  uint32_t nxt_flags = 0;
106 
107  /* determine destination address */
108  if (inet_pton(AF_INET6, (char*)dst, tmp_ipv6_dst)) {
109  dst = tmp_ipv6_dst;
110  dst_size = IN6ADDRSZ;
111  }
112  else if (inet_pton(AF_INET, dest, tmp_ipv4_dst)) {
113  dst = tmp_ipv4_dst;
114  dst_size = INADDRSZ;
115  }
116 
117  /* determine next-hop address */
118  if (inet_pton(AF_INET6, next, tmp_ipv6_nxt)) {
119  nxt = tmp_ipv6_nxt;
120  nxt_size = IN6ADDRSZ;
121  }
122  else if (inet_pton(AF_INET, next, tmp_ipv4_nxt)) {
123  nxt = tmp_ipv4_nxt;
124  nxt_size = INADDRSZ;
125  }
126 
127  dst_flags |= (prefix << FIB_FLAG_NET_PREFIX_SHIFT);
128  fib_add_entry(&gnrc_ipv6_fib_table, pid, dst, dst_size, dst_flags, nxt,
129  nxt_size, nxt_flags, lifetime);
130 }
