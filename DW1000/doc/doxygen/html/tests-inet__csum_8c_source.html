124 static void test_inet_csum__two_app_snips(void)
125 {
126  /* CoAP header with Uri-Path and Content-Format options; odd length */
127  uint8_t data_hdr[] = {
128  0x50, 0x02, 0x00, 0x01, 0xb4, 0x74, 0x65, 0x73,
129  0x74, 0x10, 0xff,
130  };
131  /* Single character payload, 'a' */
132  uint8_t data_pyld[] = {
133  0x61,
134  };
135  uint16_t hdr_sum, pyld_sum, hdr_expected = 0xdcfc;
136 
137  /* result unnormalized:
138  * initial sum (0) is arbitrary, and incoming length (0) must be even;
139  * we expect last byte is shifted left for this odd-sized header */
140  hdr_sum = inet_csum_slice(0, data_hdr, sizeof(data_hdr), 0);
141  TEST_ASSERT_EQUAL_INT(hdr_expected, hdr_sum);
142 
143  /* Since header was odd length, we expect the single byte in the payload
144  * snip is not shifted left */
145  pyld_sum = inet_csum_slice(hdr_expected, data_pyld, sizeof(data_pyld), sizeof(data_hdr));
146  TEST_ASSERT_EQUAL_INT(hdr_expected + 0x61, pyld_sum);
147 }
