37 int ccm_compute_cbc_mac(cipher_t* cipher, uint8_t iv[16],
38  uint8_t* input, size_t length, uint8_t* mac)
39 {
40  uint8_t offset, block_size, mac_enc[16] = {0};
41 
42  block_size = cipher_get_block_size(cipher);
43  memmove(mac, iv, 16);
44  offset = 0;
45  do {
46  uint8_t block_size_input = (length - offset > block_size) ?
47  block_size : length - offset;
48 
49  /* CBC-Mode: XOR plaintext with ciphertext of (n-1)-th block */
50  for (int i = 0; i < block_size_input; ++i) {
51  mac[i] ^= input[offset + i];
52  }
53 
54  if (cipher_encrypt(cipher, mac, mac_enc) != 1) {
55  return CIPHER_ERR_ENC_FAILED;
56  }
57 
58  memcpy(mac, mac_enc, block_size);
59  offset += block_size_input;
60  } while (offset < length);
61 
62  return offset;
63 }

66 int ccm_create_mac_iv(cipher_t* cipher, uint8_t auth_data_len, uint8_t M,
67  uint8_t L, uint8_t* nonce, uint8_t nonce_len,
68  size_t plaintext_len, uint8_t X1[16])
69 {
70  uint8_t M_, L_;
71 
72  /* ensure everything is set to zero */
73  memset(X1, 0, 16);
74 
75  /* set flags in B[0] - bit format:
76  7 6 5..3 2..0
77  Reserved Adata M_ L_ */
78  M_ = (M - 2) / 2;
79  L_ = L - 1;
80  X1[0] = 64 * (auth_data_len > 0) + 8 * M_ + L_;
81 
82  /* copy nonce to B[1..15-L] */
83  memcpy(&X1[1], nonce, min(nonce_len, 15 - L));
84 
85  /* write plaintext_len to B[15..16-L] */
86  for (uint8_t i = 15; i > 16 - L; --i) {
87  X1[i] = plaintext_len & 0xff;
88  plaintext_len >>= 8;
89  }
90 
91  /* if there is still data, plaintext_len was too big */
92  if (plaintext_len > 0) {
93  return CIPHER_ERR_INVALID_LENGTH;
94  }
95 
96  if (cipher_encrypt(cipher, X1, X1) != 1) {
97  return CIPHER_ERR_ENC_FAILED;
98  }
99  return 0;
100 }

132 int cipher_encrypt_ccm(cipher_t* cipher, uint8_t* auth_data, uint32_t auth_data_len,
133  uint8_t mac_length, uint8_t length_encoding,
134  uint8_t* nonce, size_t nonce_len,
135  uint8_t* input, size_t input_len,
136  uint8_t* output)
137 {
138  int len = -1;
139  uint32_t length_max;
140  uint8_t nonce_counter[16] = {0}, mac_iv[16] = {0}, mac[16] = {0},
141  stream_block[16] = {0}, zero_block[16] = {0}, block_size;
142 
143  if (mac_length % 2 != 0 || mac_length < 4 || mac_length > 16) {
144  return CCM_ERR_INVALID_MAC_LENGTH;
145  }
146 
147  length_max = 2 << (8 * length_encoding);
148  if (length_encoding < 2 || length_encoding > 8 ||
149  input_len - auth_data_len > length_max) {
150  return CCM_ERR_INVALID_LENGTH_ENCODING;
151  }
152 
153  /* Create B0, encrypt it (X1) and use it as mac_iv */
154  block_size = cipher_get_block_size(cipher);
155  if (ccm_create_mac_iv(cipher, auth_data_len, mac_length, length_encoding,
156  nonce, nonce_len, input_len, mac_iv) < 0) {
157  return CCM_ERR_INVALID_DATA_LENGTH;
158  }
159 
160  /* MAC calulation (T) with additional data and plaintext */
161  ccm_compute_adata_mac(cipher, auth_data, auth_data_len, mac_iv);
162  len = ccm_compute_cbc_mac(cipher, mac_iv, input, input_len, mac);
163  if (len < 0) {
164  return len;
165  }
166 
167  /* Compute first stream block */
168  nonce_counter[0] = length_encoding - 1;
169  memcpy(&nonce_counter[1], nonce,
170  min(nonce_len, (size_t) 15 - length_encoding));
171  len = cipher_encrypt_ctr(cipher, nonce_counter, block_size,
172  zero_block, block_size, stream_block);
173  if (len < 0) {
174  return len;
175  }
176 
177  /* Encrypt message in counter mode */
178  crypto_block_inc_ctr(nonce_counter, block_size - nonce_len);
179  len = cipher_encrypt_ctr(cipher, nonce_counter, nonce_len, input,
180  input_len, output);
181  if (len < 0) {
182  return len;
183  }
184 
185  /* auth value: mac ^ first stream block */
186  for (uint8_t i = 0; i < mac_length; ++i) {
187  output[len + i] = mac[i] ^ stream_block[i];
188  }
189 
190  return len + mac_length;
191 }
