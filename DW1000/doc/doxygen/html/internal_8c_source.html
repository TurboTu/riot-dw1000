121 bool gnrc_mac_queue_tx_packet(gnrc_mac_tx_t* tx, uint32_t priority, gnrc_pktsnip_t* pkt)
122 {
123  assert(tx != NULL);
124  assert(pkt != NULL);
125 
126 #if GNRC_MAC_NEIGHBOR_COUNT == 0
127 
128  gnrc_priority_pktqueue_node_t* node;
129  node = _alloc_pktqueue_node(tx->_queue_nodes, GNRC_MAC_TX_QUEUE_SIZE);
130 
131  if (node) {
132  gnrc_priority_pktqueue_node_init(node, priority, pkt);
133  gnrc_priority_pktqueue_push(&tx->queue, node);
134  return true;
135  }
136 
137  DEBUG("[gnrc_mac-int] Can't push to TX queue, no entries left\n");
138  return false;
139 
140 #else
141 
142  gnrc_mac_tx_neighbor_t* neighbor;
143  int neighbor_id;
144 
145  /* Check whether the packet it for broadcast */
146  if (gnrc_netif_hdr_get_flag(pkt)&GNRC_NETIF_HDR_FLAGS_BROADCAST) {
147  /* Broadcast queue is neighbor 0 by definition */
148  neighbor_id = 0;
149  neighbor = &tx->neighbors[neighbor_id];
150 
151  } else {
152  uint8_t* addr;
153  int addr_len;
154  bool neighbor_known = true;
155 
156  /* Get destination address of packet */
157  addr_len = gnrc_netif_hdr_get_dstaddr(pkt, &addr);
158  if (addr_len <= 0) {
159  DEBUG("[gnrc_mac-int] Packet has no destination address\n");
160  return false;
161  }
162 
163  /* Search for existing queue for destination */
164  neighbor_id = _gnrc_mac_find_neighbor(tx, addr, addr_len);
165 
166  /* neighbor node doesn't have a queue yet */
167  if (neighbor_id < 0) {
168  neighbor_known = false;
169 
170  /* Try to allocate neighbor entry */
171  neighbor_id = _gnrc_mac_alloc_neighbor(tx);
172 
173  /* No neighbor entries left */
174  if (neighbor_id < 0) {
175  DEBUG("[gnrc_mac-int] No neighbor entries left, maybe increase "
176  "GNRC_MAC_NEIGHBOR_COUNT for better performance\n");
177 
178  /* Try to free an unused queue */
179  neighbor_id = _gnrc_mac_free_neighbor(tx);
180 
181  /* All queues are in use, so reject */
182  if (neighbor_id < 0) {
183  DEBUG("[gnrc_mac-int] Couldn't allocate tx queue for packet\n");
184  return false;
185  }
186  }
187  }
188 
189  neighbor = &tx->neighbors[neighbor_id];
190 
191  if (!neighbor_known) {
192  _gnrc_mac_init_neighbor(neighbor, addr, addr_len);
193  }
194  }
195 
196  gnrc_priority_pktqueue_node_t* node;
197  node = _alloc_pktqueue_node(tx->_queue_nodes, GNRC_MAC_TX_QUEUE_SIZE);
198  if (node) {
199  gnrc_priority_pktqueue_node_init(node, priority, pkt);
200  gnrc_priority_pktqueue_push(&neighbor->queue, node);
201  DEBUG("[gnrc_mac-int] Queuing pkt to neighbor #%d\n", neighbor_id);
202  return true;
203  }
204 
205  DEBUG("[gnrc_mac-int] Can't push to neighbor #%d's queue, no entries left\n",
206  neighbor_id);
207  return false;
208 
209 #endif /* GNRC_MAC_NEIGHBOR_COUNT == 0 */
210 }
