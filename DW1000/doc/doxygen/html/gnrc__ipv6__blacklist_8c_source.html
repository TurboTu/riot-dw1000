32 int gnrc_ipv6_blacklist_add(const ipv6_addr_t *addr)
33 {
34  for (int i = 0; i < GNRC_IPV6_BLACKLIST_SIZE; i++) {
35  if (!bf_isset(gnrc_ipv6_blacklist_set, i)) {
36  bf_set(gnrc_ipv6_blacklist_set, i);
37  memcpy(&gnrc_ipv6_blacklist[i], addr, sizeof(*addr));
38  DEBUG("IPv6 blacklist: blacklisted %s\n",
39  ipv6_addr_to_str(addr_str, addr, sizeof(addr_str)));
40  return 0;
41  }
42  }
43  return -1;
44 }
