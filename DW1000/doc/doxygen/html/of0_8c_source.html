53 uint16_t calc_rank(gnrc_rpl_parent_t *parent, uint16_t base_rank)
54 {
55  if (base_rank == 0) {
56  if (parent == NULL) {
57  return GNRC_RPL_INFINITE_RANK;
58  }
59 
60  base_rank = parent->rank;
61  }
62 
63  uint16_t add;
64 
65  if (parent != NULL) {
66  add = parent->dodag->instance->min_hop_rank_inc;
67  }
68  else {
69  add = GNRC_RPL_DEFAULT_MIN_HOP_RANK_INCREASE;
70  }
71 
72  if ((base_rank + add) < base_rank) {
73  return GNRC_RPL_INFINITE_RANK;
74  }
75 
76  return base_rank + add;
77 }
