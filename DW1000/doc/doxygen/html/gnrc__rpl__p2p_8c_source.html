49 void gnrc_rpl_p2p_update(void)
50 {
51  gnrc_rpl_p2p_ext_t *p2p_ext;
52  for (uint8_t i = 0; i < GNRC_RPL_P2P_EXTS_NUMOF; ++i) {
53  p2p_ext = &gnrc_rpl_p2p_exts[i];
54  if ((p2p_ext->state) && (p2p_ext->lifetime_sec > 0)) {
55  p2p_ext->lifetime_sec -= GNRC_RPL_LIFETIME_UPDATE_STEP;
56  if (p2p_ext->lifetime_sec <= 0) {
57  gnrc_rpl_dodag_remove_all_parents(p2p_ext->dodag);
58  p2p_ext->dodag->instance->cleanup = GNRC_RPL_CLEANUP_TIME;
59  continue;
60  }
61  p2p_ext->dro_delay -= GNRC_RPL_LIFETIME_UPDATE_STEP;
62  if (p2p_ext->reply && (p2p_ext->dro_delay < 0) && (p2p_ext->for_me)) {
63  gnrc_rpl_p2p_send_DRO(NULL, p2p_ext);
64  }
65  }
66  }

127  gnrc_pktbuf_release(pkt);

131  pkt = opt_snip;

133 
134  if ((opt_snip = gnrc_pktbuf_add(pkt, NULL, sizeof(gnrc_rpl_p2p_opt_rdo_t),
135  GNRC_NETTYPE_UNDEF)) == NULL) {
136  DEBUG("RPL: BUILD RDO - no space left in packet buffer\n");
137  gnrc_pktbuf_release(pkt);
138  return NULL;
139  }
140  rdo = opt_snip->data;
141  rdo->type = GNRC_RPL_P2P_OPT_RDO;
142  rdo->length = GNRC_RPL_P2P_RDO_LEN +
143  p2p_ext->addr_numof * (sizeof(ipv6_addr_t) - p2p_ext->compr);
144  rdo->compr_flags = (p2p_ext->reply << GNRC_RPL_P2P_RDO_FLAGS_REPLY) |
145  (p2p_ext->hop_by_hop << GNRC_RPL_P2P_RDO_FLAGS_HBH)
146  | ((p2p_ext->routes_numof & 0x3) << GNRC_RPL_P2P_RDO_FLAGS_ADDRNUM)
147  | (p2p_ext->compr & GNRC_RPL_P2P_RDO_FLAGS_COMPR);
148  rdo->lmn = ((p2p_ext->lifetime_enc & 0x3) << GNRC_RPL_P2P_RDO_FLAGS_LIFETIME) |
149  (p2p_ext->maxrank & GNRC_RPL_P2P_RDO_FLAGS_MAXRANK);
150  rdo->target = p2p_ext->target;
151  return opt_snip;

154 void gnrc_rpl_p2p_rdo_parse(gnrc_rpl_p2p_opt_rdo_t *rdo, gnrc_rpl_p2p_ext_t *p2p_ext)
155 {
156  DEBUG("RPL: Route Discovery DIO option parsed\n");
157 
158  uint8_t addr_num = (rdo->length - GNRC_RPL_P2P_RDO_LEN)
159  / (sizeof(ipv6_addr_t) - p2p_ext->compr);
160  if (addr_num >= GNRC_RPL_P2P_ADDR_VEC_NUMOF) {
161  DEBUG("RPL: cannot parse RDO - too many hops\n");
162  return;
163  }
164 
165  p2p_ext->for_me = (gnrc_ipv6_netif_find_by_addr(NULL, &rdo->target) != KERNEL_PID_UNDEF);
166 
167  p2p_ext->reply = (rdo->compr_flags & (1 << GNRC_RPL_P2P_RDO_FLAGS_REPLY))
168  >> GNRC_RPL_P2P_RDO_FLAGS_REPLY;
169  p2p_ext->hop_by_hop = (rdo->compr_flags & (1 << GNRC_RPL_P2P_RDO_FLAGS_HBH))
170  >> GNRC_RPL_P2P_RDO_FLAGS_HBH;
171  p2p_ext->routes_numof = (rdo->compr_flags & (0x3 << GNRC_RPL_P2P_RDO_FLAGS_ADDRNUM))
172  >> GNRC_RPL_P2P_RDO_FLAGS_ADDRNUM;
173  p2p_ext->compr = rdo->compr_flags & GNRC_RPL_P2P_RDO_FLAGS_COMPR;
174  p2p_ext->lifetime_enc = (rdo->lmn & (0x3 << GNRC_RPL_P2P_RDO_FLAGS_LIFETIME))
175  >> GNRC_RPL_P2P_RDO_FLAGS_LIFETIME;
176  if (p2p_ext->lifetime_sec == INT8_MIN) {
177  p2p_ext->lifetime_sec = gnrc_rpl_p2p_lifetime_lookup[p2p_ext->lifetime_enc];
178  }
179  p2p_ext->maxrank = rdo->lmn & GNRC_RPL_P2P_RDO_FLAGS_MAXRANK;
180  p2p_ext->target = rdo->target;
181 
182  memset(&p2p_ext->addr_vec, 0, sizeof(ipv6_addr_t) * GNRC_RPL_P2P_ADDR_VEC_NUMOF);
183  p2p_ext->addr_numof = 0;
184  uint8_t *tmp = (uint8_t *) (rdo + 1);
185  uint8_t *addr = NULL;
186  uint8_t addr_len = sizeof(ipv6_addr_t) - p2p_ext->compr;
187  uint8_t i = 0;
188  for (i = 0; i < addr_num; i++) {
189  addr = ((uint8_t *) &p2p_ext->addr_vec[i]) + p2p_ext->compr;
190  memcpy(addr, tmp, addr_len);
191  tmp += addr_len;
192  p2p_ext->addr_numof++;
193  }
194 
195  if (!p2p_ext->for_me) {
196  ipv6_addr_t *me = NULL;
197  if(gnrc_ipv6_netif_find_by_prefix(&me, &p2p_ext->dodag->dodag_id) == KERNEL_PID_UNDEF) {
198  DEBUG("RPL: no address configured\n");
199  return;
200  }
201  addr = ((uint8_t *) &p2p_ext->addr_vec[i]) + p2p_ext->compr;
202  memcpy(addr, ((uint8_t *) me) + p2p_ext->compr, addr_len);
203  p2p_ext->addr_numof++;
204  }
205  else if (p2p_ext->reply) {
206  p2p_ext->stop = true;
207  p2p_ext->dro_ack = true;
208  p2p_ext->dro_delay = GNRC_RPL_P2P_DRO_DELAY;
209  }
210 }

212 static gnrc_pktsnip_t *_build_initial_DRO(gnrc_rpl_p2p_ext_t *p2p_ext)
213 {
214  gnrc_pktsnip_t *pkt = NULL, *opt_snip = NULL;
215  gnrc_rpl_p2p_dro_t *dro = NULL;
216  gnrc_rpl_p2p_opt_rdo_t *rdo = NULL;
217  size_t addr_len = (sizeof(ipv6_addr_t) - p2p_ext->compr);
218  uint8_t addr_size = p2p_ext->addr_numof * addr_len;
219 
220  for (uint8_t i = p2p_ext->addr_numof; i > 0; i--) {
221  if ((opt_snip = gnrc_pktbuf_add(pkt, NULL, addr_len, GNRC_NETTYPE_UNDEF)) == NULL) {
222  DEBUG("RPL-P2P: cannot add addresses to RDO - no space left in packet buffer\n");
223  gnrc_pktbuf_release(pkt);
224  return NULL;
225  }
226  memcpy(opt_snip->data, &p2p_ext->addr_vec[i-1], addr_len);
227  pkt = opt_snip;
228  }
229 
230  if ((opt_snip = gnrc_pktbuf_add(pkt, NULL, sizeof(gnrc_rpl_p2p_opt_rdo_t),
231  GNRC_NETTYPE_UNDEF)) == NULL) {
232  DEBUG("RPL-P2P: cannot allocate DRO - no space left in packet buffer\n");
233  gnrc_pktbuf_release(pkt);
234  return NULL;
235  }
236 
237  rdo = opt_snip->data;
238  rdo->type = GNRC_RPL_P2P_OPT_RDO;
239  rdo->length = GNRC_RPL_P2P_RDO_LEN + addr_size;
240  rdo->compr_flags = (p2p_ext->hop_by_hop << GNRC_RPL_P2P_RDO_FLAGS_HBH) |
241  (p2p_ext->compr & GNRC_RPL_P2P_RDO_FLAGS_COMPR);
242  /* rdo->length does not include the first two bytes, thus we have to add them manually */
243  rdo->lmn = (((rdo->length + 2 - sizeof(*rdo)) / addr_len) & GNRC_RPL_P2P_RDO_FLAGS_NEXT_HOP);
244  rdo->target = p2p_ext->target;
245  pkt = opt_snip;
246 
247  if ((opt_snip = gnrc_pktbuf_add(pkt, NULL, sizeof(gnrc_rpl_p2p_dro_t),
248  GNRC_NETTYPE_UNDEF)) == NULL) {
249  DEBUG("RPL-P2P: cannot allocate RDO - no space left in packet buffer\n");
250  gnrc_pktbuf_release(pkt);
251  return NULL;
252  }
253  dro = opt_snip->data;
254  dro->instance_id = p2p_ext->dodag->instance->id;
255  dro->version_number = 0;
256  dro->flags_rev = byteorder_htons((((p2p_ext->stop << 1) | (p2p_ext->dro_ack << 0))
257  << GNRC_RPL_P2P_DRO_FLAGS_ACK) |
258  ((p2p_ext->dro_seq & 0x3) << GNRC_RPL_P2P_DRO_FLAGS_SEQ));
259  dro->dodag_id = p2p_ext->dodag->dodag_id;
260  pkt = opt_snip;
261 
262  if ((opt_snip = gnrc_icmpv6_build(pkt, ICMPV6_RPL_CTRL, GNRC_RPL_P2P_ICMPV6_CODE_DRO,
263  sizeof(icmpv6_hdr_t))) == NULL) {
264  DEBUG("RPL-P2P: cannot allocate ICMPv6 - no space left in packet buffer\n");
265  gnrc_pktbuf_release(pkt);
266  return NULL;
267  }
268 
269  pkt = opt_snip;
270 
271  return pkt;
272 }

290 void gnrc_rpl_p2p_recv_DRO(gnrc_pktsnip_t *pkt, ipv6_addr_t *src)
291 {
292  gnrc_pktsnip_t *icmpv6_snip = gnrc_pktbuf_mark(pkt, sizeof(icmpv6_hdr_t), GNRC_NETTYPE_ICMPV6);
293  gnrc_pktsnip_t *dro_snip = gnrc_pktbuf_mark(pkt, sizeof(gnrc_rpl_p2p_dro_t),
294  GNRC_NETTYPE_UNDEF);
295  gnrc_pktsnip_t *rdo_snip = gnrc_pktbuf_mark(pkt, sizeof(gnrc_rpl_p2p_opt_rdo_t),
296  GNRC_NETTYPE_UNDEF);
297  gnrc_pktsnip_t *addr_snip = pkt;
298  gnrc_rpl_instance_t *inst;
299  gnrc_rpl_dodag_t *dodag;
300  gnrc_rpl_p2p_ext_t *p2p_ext;
301  gnrc_rpl_p2p_dro_t *dro;
302  gnrc_rpl_p2p_opt_rdo_t *rdo;
303  uint8_t *addr_vec;
304  uint16_t flags;
305  size_t addr_len;
306 
307  if (!rdo_snip || !dro_snip) {
308  DEBUG("RPL-P2P: Error - No DRO or RDO received\n");
309  gnrc_pktbuf_release(pkt);
310  return;
311  }
312 
313  dro = dro_snip->data;
314 
315  if ((inst = gnrc_rpl_instance_get(dro->instance_id)) == NULL) {
316  DEBUG("RPL-P2P: Error - Instance (%d) does not exist\n", dro->instance_id);
317  return;
318  }
319 
320  dodag = &inst->dodag;
321 
322  if ((p2p_ext = gnrc_rpl_p2p_ext_get(dodag)) == NULL) {
323  DEBUG("RPL-P2P: Error - No P2P-RPL DODAG extension found\n");
324  return;
325  }
326 
327  if (p2p_ext->for_me) {
328  DEBUG("RPL-P2P: Ignore DRO\n");
329  return;
330  }
331 
332  flags = byteorder_ntohs(dro->flags_rev);
333  p2p_ext->stop = (flags & (1 << GNRC_RPL_P2P_DRO_FLAGS_STOP)) >> GNRC_RPL_P2P_DRO_FLAGS_STOP;
334  p2p_ext->dro_ack = (flags & (1 << GNRC_RPL_P2P_DRO_FLAGS_ACK)) >> GNRC_RPL_P2P_DRO_FLAGS_ACK;
335  p2p_ext->dro_seq = (flags & (0x3 << GNRC_RPL_P2P_DRO_FLAGS_SEQ)) >> GNRC_RPL_P2P_DRO_FLAGS_SEQ;
336 
337  addr_len = sizeof(ipv6_addr_t) - p2p_ext->compr;
338  ipv6_addr_t addr = p2p_ext->dodag->dodag_id;
339  ipv6_addr_t *me = NULL;
340  addr_vec = addr_snip->data;
341 
342  rdo = rdo_snip->data;
343 
344  if (rdo->lmn > 0) {
345  rdo->lmn--;
346  memcpy(&addr.u8[p2p_ext->compr], &addr_vec[addr_len * rdo->lmn], addr_len);
347  }
348 
349  if (gnrc_ipv6_netif_find_by_addr(&me, &addr) == dodag->iface) {
350  fib_add_entry(&gnrc_ipv6_fib_table, dodag->iface, p2p_ext->target.u8,
351  sizeof(ipv6_addr_t), 0x0, src->u8,
352  sizeof(ipv6_addr_t), FIB_FLAG_RPL_ROUTE,
353  p2p_ext->dodag->default_lifetime *
354  p2p_ext->dodag->lifetime_unit * MS_PER_SEC);
355 
356  if (p2p_ext->dodag->node_status != GNRC_RPL_ROOT_NODE) {
357  if ((rdo_snip = gnrc_pktbuf_start_write(rdo_snip)) == NULL) {
358  DEBUG("RPL-P2P: Error - Cannot allocate new RDO\n");
359  return;
360  }
361 
362  addr_snip->next = NULL;
363  rdo_snip->next = addr_snip;
364  dro_snip->next = rdo_snip;
365 
366  icmpv6_snip = gnrc_icmpv6_build(dro_snip, ICMPV6_RPL_CTRL, GNRC_RPL_P2P_ICMPV6_CODE_DRO,
367  sizeof(icmpv6_hdr_t));
368  if (icmpv6_snip == NULL) {
369  DEBUG("RPL-P2P: cannot allocate ICMPv6 - no space left in packet buffer\n");
370  gnrc_pktbuf_release(pkt);
371  return;
372  }
373 
374  gnrc_rpl_send(icmpv6_snip, p2p_ext->dodag->iface, NULL, NULL, &p2p_ext->dodag->dodag_id);
375  return;
376  }
377  }
378 
379  gnrc_pktbuf_release(pkt);
380 
381  return;
382 }
