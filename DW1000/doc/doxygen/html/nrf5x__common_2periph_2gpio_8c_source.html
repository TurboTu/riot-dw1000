56 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
57  gpio_cb_t cb, void *arg)
58 {
59  /* disable external interrupt in case one is active */
60  NRF_GPIOTE->INTENSET &= ~(GPIOTE_INTENSET_IN0_Msk);
61  /* save callback */
62  exti_chan.cb = cb;
63  exti_chan.arg = arg;
64  /* configure pin as input */
65  gpio_init(pin, mode);
66  /* set interrupt priority and enable global GPIOTE interrupt */
67  NVIC_EnableIRQ(GPIOTE_IRQn);
68  /* configure the GPIOTE channel: set even mode, pin and active flank */
69  NRF_GPIOTE->CONFIG[0] = (GPIOTE_CONFIG_MODE_Event |
70  (pin << GPIOTE_CONFIG_PSEL_Pos) |
71  (flank << GPIOTE_CONFIG_POLARITY_Pos));
72  /* enable external interrupt */
73  NRF_GPIOTE->INTENSET |= GPIOTE_INTENSET_IN0_Msk;
