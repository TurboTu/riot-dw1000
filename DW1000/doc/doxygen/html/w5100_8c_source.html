129 static int init(netdev2_t *netdev)
130 {
131  w5100_t *dev = (w5100_t *)netdev;
132  uint8_t tmp;
133  uint8_t hwaddr[ETHERNET_ADDR_LEN];
134 
135  /* get access to the SPI bus for the duration of this function */
136  spi_acquire(dev->p.spi, dev->p.cs, SPI_CONF, dev->p.clk);
137 
138  /* test the SPI connection by reading the value of the RMSR register */
139  tmp = rreg(dev, REG_TMSR);
140  if (tmp != RMSR_DEFAULT_VALUE) {
141  spi_release(dev->p.spi);
142  LOG_ERROR("[w5100] error: no SPI connection\n");
143  return W5100_ERR_BUS;
144  }
145 
146  /* reset the device */
147  wreg(dev, REG_MODE, MODE_RESET);
148  while (rreg(dev, REG_MODE) & MODE_RESET) {};
149 
150  /* initialize the device, start with writing the MAC address */
151  uuid_get(hwaddr, ETHERNET_ADDR_LEN);
152  hwaddr[0] &= ~0x03; /* no group address and not globally unique */
153  wchunk(dev, REG_SHAR0, hwaddr, ETHERNET_ADDR_LEN);
154 
155  /* configure all memory to be used by socket 0 */
156  wreg(dev, REG_RMSR, RMSR_8KB_TO_S0);
157  wreg(dev, REG_TMSR, TMSR_8KB_TO_S0);
158 
159  /* configure interrupt pin to trigger on socket 0 events */
160  wreg(dev, REG_IMR, IMR_S0_INT);
161 
162  /* next we configure socket 0 to work in MACRAW mode */
163  wreg(dev, S0_MR, MR_MACRAW);
164  wreg(dev, S0_CR, CR_OPEN);
165 
166  /* set the source IP address to something random to prevent the device to do
167  * stupid thing (e.g. answering ICMP echo requests on its own) */
168  wreg(dev, REG_SIPR0, 0x01);
169  wreg(dev, REG_SIPR1, 0x01);
170  wreg(dev, REG_SIPR2, 0x01);
171  wreg(dev, REG_SIPR3, 0x01);
172 
173  /* start receiving packets */
174  wreg(dev, S0_CR, CR_RECV);
175 
176  /* release the SPI bus again */
177  spi_release(dev->p.spi);
178 
179  return 0;
180 }

182 static uint16_t tx_upload(w5100_t *dev, uint16_t start, void *data, size_t len)
183 {
184  if ((start + len) >= (S0_TX_BASE + S0_MEMSIZE)) {
185  size_t limit = ((S0_TX_BASE + S0_MEMSIZE) - start);
186  wchunk(dev, start, data, limit);
187  wchunk(dev, S0_TX_BASE, &((uint8_t *)data)[limit], len - limit);
188  return (S0_TX_BASE + limit);
189  }
190  else {
191  wchunk(dev, start, data, len);
192  waddr(dev, S0_TX_WR0, S0_TX_WR1, start + len);
193  return (start + len);
194  }
195 }

197 static int send(netdev2_t *netdev, const struct iovec *vector, unsigned count)
198 {
199  w5100_t *dev = (w5100_t *)netdev;
200  int sum = 0;
201 
202  /* get access to the SPI bus for the duration of this function */
203  spi_acquire(dev->p.spi, dev->p.cs, SPI_CONF, dev->p.clk);
204 
205  uint16_t pos = raddr(dev, S0_TX_WR0, S0_TX_WR1);
206 
207  /* the register is only set correctly after the first send pkt, so we need
208  * this fix here */
209  if (pos == 0) {
210  pos = S0_TX_BASE;
211  }
212 
213  for (unsigned i = 0; i < count; i++) {
214  pos = tx_upload(dev, pos, vector[i].iov_base, vector[i].iov_len);
215  sum += vector[i].iov_len;
216  }
217 
218  waddr(dev, S0_TX_WR0, S0_TX_WR1, pos);
219 
220  /* trigger the sending process */
221  wreg(dev, S0_CR, CR_SEND_MAC);
222  while (!(rreg(dev, S0_IR) & IR_SEND_OK)) {};
223  wreg(dev, S0_IR, IR_SEND_OK);
224 
225  DEBUG("[w5100] send: transferred %i byte (at 0x%04x)\n", sum, (int)pos);
226 
227  /* release the SPI bus again */
228  spi_release(dev->p.spi);
229 
230  return sum;
231 }

233 static int recv(netdev2_t *netdev, void *buf, size_t len, void *info)
234 {
235  (void)info;
236  w5100_t *dev = (w5100_t *)netdev;
237  uint8_t *in_buf = (uint8_t *)buf;
238  int n = 0;
239 
240  /* get access to the SPI bus for the duration of this function */
241  spi_acquire(dev->p.spi, dev->p.cs, SPI_CONF, dev->p.clk);
242 
243  uint16_t num = raddr(dev, S0_RX_RSR0, S0_RX_RSR1);
244 
245  if (num > 0) {
246  /* find the size of the next packet in the RX buffer */
247  uint16_t rp = raddr(dev, S0_RX_RD0, S0_RX_RD1);
248  uint16_t psize = raddr(dev, (S0_RX_BASE + (rp & S0_MASK)),
249  (S0_RX_BASE + ((rp + 1) & S0_MASK)));
250  n = psize - 2;
251 
252  DEBUG("[w5100] recv: got packet of %i byte (at 0x%04x)\n", n, (int)rp);
253 
254  /* read the actual data into the given buffer if wanted */
255  if (in_buf != NULL) {
256  uint16_t pos = rp + 2;
257  len = (n <= len) ? n : len;
258  for (int i = 0; i < (int)len; i++) {
259  in_buf[i] = rreg(dev, (S0_RX_BASE + ((pos++) & S0_MASK)));
260  }
261 
262  DEBUG("[w5100] recv: read %i byte from device (at 0x%04x)\n",
263  n, (int)rp);
264 
265  /* set the new read pointer address */
266  waddr(dev, S0_RX_RD0, S0_RX_RD1, rp += psize);
267  wreg(dev, S0_CR, CR_RECV);
268 
269  /* if RX buffer now empty, clear RECV interrupt flag */
270  if ((num - psize) == 0) {
271  wreg(dev, S0_IR, IR_RECV);
272  }
273  }
274  }
275 
276  /* release the SPI bus again */
277  spi_release(dev->p.spi);
278 
279  return n;
280 }

282 static void isr(netdev2_t *netdev)
283 {
284  uint8_t ir;
285  w5100_t *dev = (w5100_t *)netdev;
286 
287  /* we only react on RX events, and if we see one, we read from the RX buffer
288  * until it is empty */
289  spi_acquire(dev->p.spi, dev->p.cs, SPI_CONF, dev->p.clk);
290  ir = rreg(dev, S0_IR);
291  spi_release(dev->p.spi);
292  while (ir & IR_RECV) {
293  DEBUG("[w5100] netdev2 RX complete\n");
294  netdev->event_callback(netdev, NETDEV2_EVENT_RX_COMPLETE);
295  }
296 }

298 static int get(netdev2_t *netdev, netopt_t opt, void *value, size_t max_len)
299 {
300  w5100_t *dev = (w5100_t *)netdev;
301  int res = 0;
302 
303  switch (opt) {
304  case NETOPT_ADDRESS:
305  assert(max_len >= ETHERNET_ADDR_LEN);
306  spi_acquire(dev->p.spi, dev->p.cs, SPI_CONF, dev->p.clk);
307  rchunk(dev, REG_SHAR0, value, ETHERNET_ADDR_LEN);
308  spi_release(dev->p.spi);
309  res = ETHERNET_ADDR_LEN;
310  break;
311  default:
312  res = netdev2_eth_get(netdev, opt, value, max_len);
313  break;
314  }
315 
316  return res;
317 }
