37 #define ONE_FOURTH ((int)1 << (INT_BITS - 2))
38 #define ONE_HALF (ONE_FOURTH * 2.0)
39 #define ONE (ONE_FOURTH * 4.0)

41 /*
42  * Convert float to (unsigned) quad. We do most of our work in double,
43  * out of sheer paranoia.
44  *
45  * Not sure what to do with negative numbers---for now, anything out
46  * of range becomes UQUAD_MAX.
47  *
48  * N.B.: must use new ANSI syntax (sorry).
49  */
50 u_quad_t __fixunssfdi(float f)
51 {
52  double x, toppart;
53  union uu t;
54 
55  if (f < 0) {
56  return (UQUAD_MAX); /* ??? should be 0? ERANGE??? */
57  }
58 
59 #ifdef notdef /* this falls afoul of a GCC bug */
60  if (f >= UQUAD_MAX) {
61  return UQUAD_MAX;
62  }
63 #else /* so we wire in 2^64-1 instead */
64  if (f >= 18446744073709551615.0) { /* XXX */
65  return (UQUAD_MAX);
66  }
67 
68 #endif
69 
70  x = f;
71  /*
72  * Get the upper part of the result. Note that the divide
73  * may round up; we want to avoid this if possible, so we
74  * subtract `1/2' first.
75  */
76  toppart = (x - ONE_HALF) / ONE;
77  /*
78  * Now build a u_quad_t out of the top part. The difference
79  * between x and this is the bottom part (this may introduce
80  * a few fuzzy bits, but what the heck). With any luck this
81  * difference will be nonnegative: x should wind up in the
82  * range [0..UINT_MAX]. For paranoia, we assume [INT_MIN..
83  * 2*UINT_MAX] instead.
84  */
85  t.ul[H] = (unsigned int) toppart;
86  t.ul[L] = 0;
87  x -= (double) t.uq;
88 
89  if (x < 0) {
90  t.ul[H]--;
91  x += UINT_MAX;
92  }
93 
94  if (x > UINT_MAX) {
95  t.ul[H]++;
96  x -= UINT_MAX;
97  }
98 
99  t.ul[L] = (u_int)x;
100  return t.uq;
101 }
