137 static int _ipv6_nc_add(int argc, char **argv)
138 {
139  char *ipv6_addr_str;
140  char *l2_addr_str = NULL;
141  kernel_pid_t iface = KERNEL_PID_UNDEF;
142  ipv6_addr_t ipv6_addr;
143  uint8_t l2_addr[MAX_L2_ADDR_LEN];
144  size_t l2_addr_len = 0;
145 
146  switch (argc) {
147  case 1:
148  iface = _get_iface();
149  ipv6_addr_str = argv[0]; /* store for output */
150  l2_addr_len = 0;
151  break;
152  case 2:
153  ipv6_addr_str = argv[0]; /* store for output */
154  if (ipv6_addr_from_str(&ipv6_addr, ipv6_addr_str) == NULL) {
155  iface = atoi(argv[0]);
156  ipv6_addr_str = argv[1];
157  l2_addr_len = 0;
158  }
159  else {
160  iface = _get_iface();
161  l2_addr_str = argv[1];
162  }
163  break;
164  default:
165  iface = atoi(argv[0]);
166  ipv6_addr_str = argv[1];
167  l2_addr_str = argv[2];
168  break;
169  }
170 
171  if (!_is_iface(iface)) {
172  puts("error: invalid interface given.");
173  return 1;
174  }
175 
176  if (ipv6_addr_from_str(&ipv6_addr, ipv6_addr_str) == NULL) {
177  puts("error: unable to parse IPv6 address.");
178  return 1;
179  }
180 
181  if ((l2_addr_str != NULL) && (l2_addr_len = gnrc_netif_addr_from_str(l2_addr, sizeof(l2_addr),
182  l2_addr_str)) == 0) {
183  puts("error: unable to parse link-layer address.");
184  return 1;
185  }
186 
187  if (gnrc_ipv6_nc_add(iface, &ipv6_addr, l2_addr, l2_addr_len, 0) == NULL) {
188  puts("error: unable to add address to neighbor cache.");
189  return 1;
190  }
191 
192  printf("success: added IPv6 address %s to neighbor cache\n", ipv6_addr_str);
193  return 0;
194 }

211 static int _ipv6_nc_reset(void)
212 {
213  gnrc_ipv6_nc_t *tmp = NULL;
214 
215  for (gnrc_ipv6_nc_t *entry = gnrc_ipv6_nc_get_next(NULL);
216  entry != NULL;
217  tmp = entry, entry = gnrc_ipv6_nc_get_next(entry)) {
218  if (tmp) {
219  gnrc_ipv6_nc_remove(tmp->iface, &tmp->ipv6_addr);
220  }
221  }
222 
223  /* remove last entry */
224  if (tmp) {
225  gnrc_ipv6_nc_remove(tmp->iface, &tmp->ipv6_addr);
226  }
227 
228  printf("success: reset neighbor cache\n");
229 
230  return 0;
231 }

233 int _ipv6_nc_manage(int argc, char **argv)
234 {
235  if ((argc == 1) || (strcmp("list", argv[1]) == 0)) {
236  return _ipv6_nc_list();
237  }
238 
239  if (argc > 1) {
240  if ((argc > 2) && (strcmp("add", argv[1]) == 0)) {
241  return _ipv6_nc_add(argc - 2, &argv[2]);
242  }
243  if ((argc > 2) && (strcmp("del", argv[1]) == 0)) {
244  return _ipv6_nc_del(argv[2]);
245  }
246  if (strcmp("reset", argv[1]) == 0) {
247  return _ipv6_nc_reset();
248  }
249  }
250 
251  printf("usage: %s [list]\n"
252  " or: %s add [<iface pid>] <ipv6_addr> <l2_addr>\n"
253  " * <iface pid> is optional if only one interface exists.\n"
254  " or: %s del <ipv6_addr>\n"
255  " or: %s reset\n", argv[0], argv[0], argv[0], argv[0]);
256  return 1;
257 }
