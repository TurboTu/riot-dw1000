42 static void _r(uint32_t *d, uint32_t *a, const uint32_t *b, unsigned c)
43 {
44  *a += *b;
45  uint32_t tmp = *a ^ *d;
46  *d = (tmp << c) | (tmp >> (32 - c));
47 }

49 static void _doubleround(void *output_, const uint32_t input[16], uint8_t rounds)
50 {
51  uint32_t *output = (uint32_t *) output_;
52  memcpy(output, input, 64);
53 
54  rounds *= 4;
55  for (unsigned i = 0; i < rounds; ++i) {
56  uint32_t *a = &output[((i + ((i & 4) ? 0 : 0)) & 3) + (4 * 0)];
57  uint32_t *b = &output[((i + ((i & 4) ? 1 : 0)) & 3) + (4 * 1)];
58  uint32_t *c = &output[((i + ((i & 4) ? 2 : 0)) & 3) + (4 * 2)];
59  uint32_t *d = &output[((i + ((i & 4) ? 3 : 0)) & 3) + (4 * 3)];
60 
61  _r(d, a, b, 16);
62  _r(b, c, d, 12);
63  _r(d, a, b, 8);
64  _r(b, c, d, 7);
65  }
66 
67  for (unsigned i = 0; i < 16; ++i) {
68  output[i] += input[i];
69  }
70 }
