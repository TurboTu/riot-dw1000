39 void rtc_init(void)
40 {
41  /* Turn on power manager for RTC */
42  /* Already done in cpu_init() */
43  /* MCLK->APBAMASK.reg |= MCLK_APBAMASK_RTC; */
44 
45  /* DISABLE RTC MASTER */
46  rtc_poweroff();
47 
48 
49 #if EXTERNAL_OSC32_SOURCE
50 
51  /* RTC uses External 32,768KHz Oscillator */
52  OSC32KCTRL->XOSC32K.reg = OSC32KCTRL_XOSC32K_XTALEN
53  | OSC32KCTRL_XOSC32K_EN1K
54  | OSC32KCTRL_XOSC32K_RUNSTDBY
55  | OSC32KCTRL_OSC32K_ENABLE;
56 
57  /* Wait XOSC32K Ready */
58  while (OSC32KCTRL->STATUS.bit.XOSC32KRDY==0);
59 
60  /* RTC source clock is external oscillator at 1kHz */
61  OSC32KCTRL->RTCCTRL.reg = OSC32KCTRL_RTCCTRL_RTCSEL_XOSC1K;
62 
63 #endif /* EXTERNAL_OSC32_SOURCE */
64 
65 #if INTERNAL_OSC32_SOURCE
66  uint32_t * pCalibrationArea;
67  uint32_t osc32kcal;
68 
69  /* Read OSC32KCAL, calibration data for OSC32 !!! */
70  pCalibrationArea = (uint32_t*) NVMCTRL_OTP5;
71  osc32kcal = ( (*pCalibrationArea) & 0x1FC0 ) >> 6;
72 
73  /* RTC use Low Power Internal Oscillator at 1kHz */
74  OSC32KCTRL->OSC32K.reg = OSC32KCTRL_OSC32K_RUNSTDBY
75  | OSC32KCTRL_OSC32K_EN1K
76  | OSC32KCTRL_OSC32K_CALIB(osc32kcal)
77  | OSC32KCTRL_OSC32K_ENABLE;
78 
79  /* Wait OSC32K Ready */
80  while (OSC32KCTRL->STATUS.bit.OSC32KRDY==0);
81 
82  /* RTC uses internal 32,768KHz Oscillator */
83  OSC32KCTRL->RTCCTRL.reg = OSC32KCTRL_RTCCTRL_RTCSEL_OSC1K;
84 
85 
86 #endif /* INTERNAL_OSC32_SOURCE */
87 
88 #if ULTRA_LOW_POWER_INTERNAL_OSC_SOURCE
89 
90  /* RTC uses Ultra Low Power internal 32,768KHz Oscillator */
91  OSC32KCTRL->RTCCTRL.reg = OSC32KCTRL_RTCCTRL_RTCSEL_ULP1K;
92 
93 #endif /* ULTRA_LOW_POWER_INTERNAL_OSC_SOURCE */
94 
95  /* Software Reset the RTC */
96  RTC->MODE2.CTRLA.bit.SWRST = 1;
97  /* Wait end of reset */
98  while (RTC->MODE2.CTRLA.bit.SWRST);
99 
100  /* RTC config with RTC_MODE2_CTRL_CLKREP = 0 (24h) */
101  RTC->MODE2.CTRLA.reg = RTC_MODE2_CTRLA_PRESCALER_DIV1024 | /* CLK_RTC_CNT = 1KHz / 1024 -> 1Hz */
102 #if (SAML21XXXB) || (SAMR30)
103  RTC_MODE2_CTRLA_CLOCKSYNC | /* Clock Read Synchronization Enable */
104 #endif
105  RTC_MODE2_CTRLA_MODE_CLOCK; /* Mode 2: Clock/Calendar */
106 
107  /* Clear interrupt flags */
108  RTC->MODE2.INTFLAG.reg |= RTC_MODE2_INTFLAG_OVF;
109  RTC->MODE2.INTFLAG.reg |= RTC_MODE2_INTFLAG_ALARM0;
110 
111  rtc_poweron();
112 }

114 int rtc_set_time(struct tm *time)
115 {
116  if ((time->tm_year < reference_year) || (time->tm_year > reference_year + 63)) {
117  return -1;
118  }
119  else {
120  while (RTC->MODE2.SYNCBUSY.bit.CLOCK);
121  RTC->MODE2.CLOCK.reg = RTC_MODE2_CLOCK_YEAR(time->tm_year - reference_year)
122  | RTC_MODE2_CLOCK_MONTH(time->tm_mon + 1)
123  | RTC_MODE2_CLOCK_DAY(time->tm_mday)
124  | RTC_MODE2_CLOCK_HOUR(time->tm_hour)
125  | RTC_MODE2_CLOCK_MINUTE(time->tm_min)
126  | RTC_MODE2_CLOCK_SECOND(time->tm_sec);
127  while (RTC->MODE2.SYNCBUSY.bit.CLOCK);
128  }
129  return 0;
130 }

151 int rtc_set_alarm(struct tm *time, rtc_alarm_cb_t cb, void *arg)
152 {
153  rtc_clear_alarm();
154  if ((time->tm_year < reference_year) || (time->tm_year > (reference_year + 63))) {
155  return -2;
156  }
157  else {
158  RTC->MODE2.Mode2Alarm[0].ALARM.reg = RTC_MODE2_ALARM_YEAR(time->tm_year - reference_year)
159  | RTC_MODE2_ALARM_MONTH(time->tm_mon + 1)
160  | RTC_MODE2_ALARM_DAY(time->tm_mday)
161  | RTC_MODE2_ALARM_HOUR(time->tm_hour)
162  | RTC_MODE2_ALARM_MINUTE(time->tm_min)
163  | RTC_MODE2_ALARM_SECOND(time->tm_sec);
164  RTC->MODE2.Mode2Alarm[0].MASK.reg = RTC_MODE2_MASK_SEL(6);
165  while (RTC->MODE2.SYNCBUSY.bit.ALARM0);
166  }
167 
168  /* Setup interrupt */
169  NVIC_EnableIRQ(RTC_IRQn);
170 
171  /* Enable IRQ */
172  rtc_callback.cb = cb;
173  rtc_callback.arg = arg;
174  RTC->MODE2.INTFLAG.reg |= RTC_MODE2_INTFLAG_ALARM0;
175  RTC->MODE2.INTENSET.bit.ALARM0 = 1;
176 
177  return 0;
178 }

220 void isr_rtc(void)
221 {
222  if (RTC->MODE2.INTFLAG.bit.ALARM0) {
223  rtc_callback.cb(rtc_callback.arg);
224  /* clear flag */
225  RTC->MODE2.INTFLAG.reg |= RTC_MODE2_INTFLAG_ALARM0;
226  }
227  if (RTC->MODE2.INTFLAG.bit.OVF) {
228  /* clear flag */
229  RTC->MODE2.INTFLAG.reg |= RTC_MODE2_INTFLAG_OVF;
230  /* At 1Hz, RTC goes till 63 years (2^5, see 17.8.22 in datasheet)
231  * Start RTC again with reference_year 64 years more (Be careful with alarm set) */
232  reference_year += 64;
233  }
234  cortexm_isr_end();
235 }
