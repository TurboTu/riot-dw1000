76 ssize_t ubjson_write_i32(ubjson_cookie_t *restrict cookie, int32_t value)
77 {
78  ssize_t result = 0;
79  if ((INT8_MIN <= value) && (value <= INT8_MAX)) {
80  WRITE_MARKER(UBJSON_MARKER_INT8);
81  WRITE_MARKER((uint8_t) value);
82  }
83  else if ((0 <= value) && (value <= UINT8_MAX)) {
84  WRITE_MARKER(UBJSON_MARKER_UINT8);
85  WRITE_MARKER((uint8_t) value);
86  }
87  else if ((INT16_MIN <= value) && (value <= UINT16_MAX)) {
88  WRITE_MARKER(UBJSON_MARKER_INT16);
89  network_uint16_t buf = byteorder_htons((uint16_t) value);
90  WRITE_BUF(&buf, sizeof(buf));
91  }
92  else {
93  WRITE_MARKER(UBJSON_MARKER_INT32);
94  network_uint32_t buf = byteorder_htonl((uint32_t) value);
95  WRITE_BUF(&buf, sizeof(buf));
96  }
97  return result;
98 }
