46 {
47  lib_entry_t *lib_elt;
48  nhdp_addr_entry_t *addr_elt;
49  int result = -1;
50 
51  mutex_lock(&mtx_lib_access);
52 
53  /* Check whether the given interface is already registered */
54  LL_FOREACH(lib_entry_head, lib_elt) {
55  if (lib_elt->if_pid == if_pid) {
56  LL_FOREACH(lib_entry_head->if_addr_list_head, addr_elt) {

101 {
102  lib_entry_t *lib_elt;
103  nhdp_addr_entry_t *add_tmp;
104 
105  mutex_lock(&mtx_lib_access);
106 
107  /* First fill the list for LOCAL_IF = THIS_IF */
108  LL_FOREACH(lib_entry_head, lib_elt) {
109  if (lib_elt->if_pid == if_pid) {
110  LL_FOREACH(lib_elt->if_addr_list_head, add_tmp) {
111  nhdp_writer_add_addr(wr, add_tmp->address,
112  RFC5444_ADDRTLV_LOCAL_IF, RFC5444_LOCALIF_THIS_IF,
113  NHDP_METRIC_UNKNOWN, NHDP_METRIC_UNKNOWN);
114  add_tmp->address->in_tmp_table = NHDP_ADDR_TMP_ANY;
115  }
116  break;
117  }
118  }
119 
120  /* Second fill the list for LOCAL_IF = OTHER_IF */
121  LL_FOREACH(lib_entry_head, lib_elt) {
122  if (lib_elt->if_pid != if_pid) {
123  LL_FOREACH(lib_elt->if_addr_list_head, add_tmp) {
124  /* Check if this address is not already included in a list */
125  if (!NHDP_ADDR_TMP_IN_ANY(add_tmp->address)) {
126  /* Address can be added */
127  nhdp_writer_add_addr(wr, add_tmp->address,
128  RFC5444_ADDRTLV_LOCAL_IF, RFC5444_LOCALIF_OTHER_IF,
129  NHDP_METRIC_UNKNOWN, NHDP_METRIC_UNKNOWN);
130  add_tmp->address->in_tmp_table = NHDP_ADDR_TMP_ANY;
131  }
132  }
133  }
134  }
135 
136  mutex_unlock(&mtx_lib_access);

168 static int create_if_entry(kernel_pid_t if_pid, nhdp_addr_t *addr)
169 {
170  lib_entry_t *new_entry;
171 
172  new_entry = (lib_entry_t *) malloc(sizeof(lib_entry_t));
173 
174  if (!new_entry) {
175  /* Insufficient memory */
176  return -1;
177  }
178 
179  new_entry->if_addr_list_head = NULL;
180  new_entry->if_pid = if_pid;
181 
182  if (add_address_to_if(new_entry, addr)) {
183  /* Insufficient memory */
184  free(new_entry);
185  return -1;
186  }
187 
188  LL_PREPEND(lib_entry_head, new_entry);
189 
190  return 0;
191 }

196 static int add_address_to_if(lib_entry_t *if_entry, nhdp_addr_t *addr)
197 {
198  nhdp_addr_entry_t *new_entry = (nhdp_addr_entry_t *) malloc(sizeof(nhdp_addr_entry_t));
199 
200  if (!new_entry) {
201  /* Insufficient memory */
202  return -1;
203  }
204 
205  /* Increment usage counter of address in central NHDP address storage */
206  addr->usg_count++;
207  new_entry->address = addr;
208  LL_PREPEND(if_entry->if_addr_list_head, new_entry);
