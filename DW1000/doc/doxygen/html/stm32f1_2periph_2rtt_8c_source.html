154 void rtt_poweron(void)
155 {
156  periph_clk_en(APB1, (RCC_APB1ENR_BKPEN|RCC_APB1ENR_PWREN)); /* enable BKP and PWR, Clock */
157  /* RTC clock source configuration */
158  PWR->CR |= PWR_CR_DBP; /* Allow access to BKP Domain */
159  RCC->BDCR |= RCC_BDCR_LSEON; /* Enable LSE OSC */
160  while(!(RCC->BDCR & RCC_BDCR_LSERDY)) {} /* Wait till LSE is ready */
161  RCC->BDCR |= RCC_BDCR_RTCSEL_LSE; /* Select the RTC Clock Source */
162  RCC->BDCR |= RCC_BDCR_RTCEN; /* enable RTC */
163 }

166 {

168  RCC->BDCR &= ~RCC_BDCR_RTCEN; /* disable RTC */
169  periph_clk_dis(APB1, (RCC_APB1ENR_BKPEN|RCC_APB1ENR_PWREN)); /* disable BKP and PWR, Clock */
170 }

172 inline void _rtt_enter_config_mode(void)
173 {
174  /* Loop until RTOFF flag is set */
175  while (!(RTT_DEV->CRL & RTT_FLAG_RTOFF)) {}
176  /* enter configuration mode */
