94 int gnrc_netapi_dispatch(gnrc_nettype_t type, uint32_t demux_ctx,
95  uint16_t cmd, gnrc_pktsnip_t *pkt)
96 {
97  int numof = gnrc_netreg_num(type, demux_ctx);
98 
99  if (numof != 0) {
100  gnrc_netreg_entry_t *sendto = gnrc_netreg_lookup(type, demux_ctx);
101 
102  gnrc_pktbuf_hold(pkt, numof - 1);
103 
104  while (sendto) {
105 #if defined(MODULE_GNRC_NETAPI_MBOX) || defined(MODULE_GNRC_NETAPI_CALLBACKS)
106  int release = 0;
107  switch (sendto->type) {
108  case GNRC_NETREG_TYPE_DEFAULT:
109  if (_snd_rcv(sendto->target.pid, cmd, pkt) < 1) {
110  /* unable to dispatch packet */
111  release = 1;
112  }
113  break;
114 #ifdef MODULE_GNRC_NETAPI_MBOX
115  case GNRC_NETREG_TYPE_MBOX:
116  if (_snd_rcv_mbox(sendto->target.mbox, cmd, pkt) < 1) {
117  /* unable to dispatch packet */
118  release = 1;
119  }
120  break;
121 #endif
122 #ifdef MODULE_GNRC_NETAPI_CALLBACKS
123  case GNRC_NETREG_TYPE_CB:
124  sendto->target.cbd->cb(cmd, pkt, sendto->target.cbd->ctx);
125  break;
126 #endif
127  default:
128  /* unknown dispatch type */
129  release = 1;
130  break;
131  }
132  if (release) {
133  gnrc_pktbuf_release(pkt);
134  }
135 #else
136  if (_snd_rcv(sendto->target.pid, cmd, pkt) < 1) {
137  /* unable to dispatch packet */
138  gnrc_pktbuf_release(pkt);
139  }
140 #endif
141  sendto = gnrc_netreg_getnext(sendto);
142  }
143  }
144 
145  return numof;
146 }
