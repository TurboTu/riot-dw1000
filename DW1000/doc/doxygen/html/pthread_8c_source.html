121 int pthread_create(pthread_t *newthread, const pthread_attr_t *attr, void *(*start_routine)(void *), void *arg)
122 {
123  pthread_thread_t *pt = calloc(1, sizeof(pthread_thread_t));
124 
125  kernel_pid_t pthread_pid = insert(pt);
126  if (pthread_pid == KERNEL_PID_UNDEF) {
127  free(pt);
128  return -1;
129  }
130  *newthread = pthread_pid;
131 
132  pt->status = attr && attr->detached ? PTS_DETACHED : PTS_RUNNING;
133  pt->start_routine = start_routine;
134  pt->arg = arg;
135 
136  bool autofree = attr == NULL || attr->ss_sp == NULL || attr->ss_size == 0;
137  size_t stack_size = attr && attr->ss_size > 0 ? attr->ss_size : PTHREAD_STACKSIZE;
138  void *stack = autofree ? malloc(stack_size) : attr->ss_sp;
139  pt->stack = autofree ? stack : NULL;
140 
141  if (autofree && pthread_reaper_pid != KERNEL_PID_UNDEF) {
142  mutex_lock(&pthread_mutex);
143  if (pthread_reaper_pid != KERNEL_PID_UNDEF) {
144  /* volatile pid to overcome problems with double checking */
145  volatile kernel_pid_t pid = thread_create(pthread_reaper_stack,
146  PTHREAD_REAPER_STACKSIZE,
147  0,
148  THREAD_CREATE_STACKTEST,
149  pthread_reaper,
150  NULL,
151  "pthread-reaper");
152  pthread_reaper_pid = pid;
153  }
154  mutex_unlock(&pthread_mutex);
155  }
156 
157  pt->thread_pid = thread_create(stack,
158  stack_size,
159  THREAD_PRIORITY_MAIN,
160  THREAD_CREATE_WOUT_YIELD |
161  THREAD_CREATE_STACKTEST,
162  pthread_start_routine,
163  pt,
164  "pthread");
165  if (pt->thread_pid == KERNEL_PID_UNDEF) {
166  free(pt->stack);
167  free(pt);
168  pthread_sched_threads[pthread_pid-1] = NULL;
169  return -1;
170  }
171 
172  sched_switch(THREAD_PRIORITY_MAIN);
173 
174  return 0;
175 }

177 void pthread_exit(void *retval)
178 {
179  pthread_t self_id = pthread_self();
180 
181  if (self_id == 0) {
182  DEBUG("ERROR called pthread_self() returned 0 in \"%s\"!\n", __func__);
183  }
184  else {
185  pthread_thread_t *self = pthread_sched_threads[self_id - 1];
186 
187  while (self->cleanup_top) {
188  __pthread_cleanup_datum_t *ct = self->cleanup_top;
189  self->cleanup_top = ct->__next;
190 
191  ct->__routine(ct->__arg);
192  }
193 
194  /* Prevent linking in pthread_tls.o if no TSS functions were used. */
195  extern void __pthread_keys_exit(int self_id) __attribute__((weak));
196  if (__pthread_keys_exit) {
197  __pthread_keys_exit(self_id);
198  }
199 
200  self->thread_pid = KERNEL_PID_UNDEF;
201  DEBUG("pthread_exit(%p), self == %p\n", retval, (void *) self);
202  if (self->status != PTS_DETACHED) {
203  self->returnval = retval;
204  self->status = PTS_ZOMBIE;
205 
206  if (self->joining_thread) {
207  /* our thread got an other thread waiting for us */
208  thread_wakeup(self->joining_thread);
209  }
210  }
211 
212  irq_disable();
213  if (self->stack) {
214  msg_t m;
215  m.content.ptr = self->stack;
216  msg_send_int(&m, pthread_reaper_pid);
217  }
218  }
219 
220  sched_task_exit();
221 }

223 int pthread_join(pthread_t th, void **thread_return)
224 {
225  if (th < 1 || th > MAXTHREADS) {
226  DEBUG("passed pthread_t th (%d) exceeds bounds of pthread_sched_threads[] in \"%s\"!\n", th, __func__);
227  return -3;
228  }
229 
230  pthread_thread_t *other = pthread_sched_threads[th-1];
231  if (!other) {
232  return -1;
233  }
234 
235  switch (other->status) {
236  case (PTS_RUNNING):
237  other->joining_thread = sched_active_pid;
238  /* go blocked, I'm waking up if other thread exits */
239  thread_sleep();
240  /* no break */
241  case (PTS_ZOMBIE):
242  if (thread_return) {
243  *thread_return = other->returnval;
244  }
245  free(other);
246  /* we only need to free the pthread layer struct,
247  native thread stack is freed by other */
248  pthread_sched_threads[th-1] = NULL;
249  return 0;
250  case (PTS_DETACHED):
251  return -1;
252  }
253 
254  return -2;
255 }
