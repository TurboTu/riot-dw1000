39 void hwrng_read(uint8_t *buf, unsigned int num)
40 {

43  unsigned int count = 0;
44 
45  /* enable RNG reset state */
46  periph_clk_en(AHB2, RCC_AHB2ENR_RNGEN);
47  /* enable the RNG */
48  RNG->CR |= RNG_CR_RNGEN;
49 
50  while (count < num) {
51  /* wait for random data to be ready to read */
52  while (!(RNG->SR & RNG_SR_DRDY));
53  /* read next 4 bytes */
54  tmp = RNG->DR;
55  /* copy data into result vector */
56  for (int i = 0; i < 4 && count < num; i++) {
