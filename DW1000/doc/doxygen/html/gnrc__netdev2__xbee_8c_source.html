31 static gnrc_pktsnip_t *xbee_adpt_recv(gnrc_netdev2_t *gnrc_netdev2)
32 {
33  netdev2_t *dev = gnrc_netdev2->dev;
34  int pktlen;
35  int xhdr_len;
36  gnrc_pktsnip_t *payload;
37  gnrc_pktsnip_t *netif_snip;
38  gnrc_pktsnip_t *xhdr_snip;
39  gnrc_netif_hdr_t *netif;
40  xbee_l2hdr_t l2hdr;
41 
42  assert(dev);
43 
44  /* see how much data there is to process */
45  pktlen = dev->driver->recv(dev, NULL, 0, NULL);
46  if (pktlen <= 0) {
47  DEBUG("[xbee-gnrc] recv: no data available to process\n");
48  return NULL;
49  }
50 
51  /* allocate space for the packet in the pktbuf */
52  payload = gnrc_pktbuf_add(NULL, NULL, pktlen, XBEE_DEFAULT_PROTOCOL);
53  if (payload == NULL) {
54  DEBUG("[xbee-gnrc] recv: unable to allocate space in the pktbuf\n");
55  /* tell the driver to drop the packet */
56  dev->driver->recv(dev, NULL, 1, NULL);
57  return NULL;
58  }
59 
60  /* copy the complete including the XBee header into the packet buffer */
61  dev->driver->recv(dev, payload->data, pktlen, NULL);
62 
63  /* try to parse the L2 header data from the XBee header */
64  xhdr_len = xbee_parse_hdr((xbee_t *)dev, (uint8_t *)payload->data, &l2hdr);
65  if (xhdr_len < 0) {
66  DEBUG("[xbee-gnrc] recv: unable to parse XBee header\n");
67  gnrc_pktbuf_release(payload);
68  return NULL;
69  }
70 
71  /* crop the XBee header from the payload */
72  xhdr_snip = gnrc_pktbuf_mark(payload, xhdr_len, GNRC_NETTYPE_UNDEF);
73  if (xhdr_snip == NULL) {
74  DEBUG("[xbee-gnrc] recv: unable to mark XBee header snip\n");
75  gnrc_pktbuf_release(payload);
76  return NULL;
77  }
78  gnrc_pktbuf_remove_snip(payload, xhdr_snip);
79 
80  /* create a netif hdr from the obtained data */
81  netif_snip = gnrc_netif_hdr_build(l2hdr.src_addr, l2hdr.addr_len,
82  l2hdr.dst_addr, l2hdr.addr_len);
83  if (netif_snip == NULL) {
84  DEBUG("[xbee-gnrc] recv: unable to allocate netif header\n");
85  gnrc_pktbuf_release(payload);
86  return NULL;
87  }
88  netif = (gnrc_netif_hdr_t *)netif_snip->data;
89  netif->if_pid = gnrc_netdev2->pid;
90  netif->rssi = l2hdr.rssi;
91  if (l2hdr.bcast) {
92  netif->flags = GNRC_NETIF_HDR_FLAGS_BROADCAST;
93  }
94 
95  DEBUG("[xbee-gnrc] recv: successfully parsed packet\n");
96 
97  /* and append the netif header */
98  LL_APPEND(payload, netif_snip);
99 
100  return payload;
101 }

103 static int xbee_adpt_send(gnrc_netdev2_t *dev, gnrc_pktsnip_t *pkt)
104 {
105  int res;
106  size_t size;
107  size_t count;
108  gnrc_pktsnip_t *vec;
109  gnrc_netif_hdr_t *hdr;
110  uint8_t xhdr[XBEE_MAX_TXHDR_LENGTH];
111 
112  /* check device descriptor and packet */
113  assert(dev && pkt);
114 
115  /* get the payload size and the dst address details */
116  size = gnrc_pkt_len(pkt->next);
117  DEBUG("[xbee-gnrc] send: payload of packet is %i\n", (int)size);
118  hdr = (gnrc_netif_hdr_t *)pkt->data;
119  if (hdr->flags & BCAST) {
120  uint16_t addr = 0xffff;
121  res = xbee_build_hdr((xbee_t *)dev, xhdr, size, &addr, 2);
122  DEBUG("[xbee-gnrc] send: preparing to send broadcast\n");
123  }
124  else {
125  uint8_t *addr = gnrc_netif_hdr_get_dst_addr(hdr);
126  res = xbee_build_hdr((xbee_t *)dev, xhdr, size, addr, hdr->dst_l2addr_len);
127  if (res < 0) {
128  if (res == -EOVERFLOW) {
129  DEBUG("[xbee-gnrc] send: payload length exceeds max limit\n");
130  }
131  else if (res == -ENOMSG) {
132  DEBUG("[xbee-gnrc] send: invalid destination l2 address\n");
133  }
134  return res;
135  }
136  if (hdr->dst_l2addr_len == IEEE802154_SHORT_ADDRESS_LEN) {
137  DEBUG("[xbee-gnrc] send: preparing to send unicast to %02x:%02x\n",
138  (int)addr[0], (int)addr[1]);
139  }
140  else {
141  DEBUG("[xbee-gnrc] send: preparing to send unicast to "
142  "%02x:%02x:%02x:%02x:%02x:%02x:%02x:%02x\n",
143  (int)addr[0], (int)addr[1], (int)addr[2], (int)addr[3],
144  (int)addr[4], (int)addr[5], (int)addr[6], (int)addr[7]);
145  }
146  }
147 
148  /* now let's extract the iovector and send out the stuff */
149  vec = gnrc_pktbuf_get_iovec(pkt, &count);
150  if (vec != NULL) {
151  pkt = vec;
152  struct iovec *vector = (struct iovec *)pkt->data;
153  vector[0].iov_base = xhdr;
154  vector[0].iov_len = res;
155 #ifdef MODULE_NETSTATS_L2
156  if (hdr->flags & BCAST) {
157  dev->dev->stats.tx_mcast_count++;
158  }
159  else {
160  dev->dev->stats.tx_unicast_count++;
161  }
162 #endif
163  DEBUG("[xbee-gnrc] send: triggering the drivers send function\n");
164  res = dev->dev->driver->send(dev->dev, vector, count);
165  }
166  else {
167  DEBUG("[xbee-gnrc] send: unable to create iovec\n");
168  }
169 
170  gnrc_pktbuf_release(pkt);
171 
172  return res;
173 }
