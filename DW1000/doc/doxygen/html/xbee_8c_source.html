124 static void _api_at_cmd(xbee_t *dev, uint8_t *cmd, uint8_t size, resp_t *resp)
125 {
126  DEBUG("[xbee] AT_CMD: %s\n", cmd);
127 
128  /* acquire TX lock */
129  mutex_lock(&(dev->tx_lock));
130  /* construct API frame */
131  dev->cmd_buf[0] = API_START_DELIMITER;
132  dev->cmd_buf[1] = (size + 2) >> 8;
133  dev->cmd_buf[2] = (size + 2) & 0xff;
134  dev->cmd_buf[3] = API_ID_AT;
135  dev->cmd_buf[4] = 1; /* use fixed frame id */
136  memcpy(dev->cmd_buf + 5, cmd, size);
137  dev->cmd_buf[size + 5] = _cksum(3, dev->cmd_buf, size + 5);
138 
139  /* reset the response data counter */
140  dev->resp_count = 0;
141  /* start send data */
142  uart_write(dev->p.uart, dev->cmd_buf, size + 6);
143 
144  xtimer_ticks64_t sent_time = xtimer_now64();
145 
146  xtimer_t resp_timer;
147 
148  resp_timer.callback = isr_resp_timeout;
149  resp_timer.arg = dev;
150 
151  xtimer_set(&resp_timer, RESP_TIMEOUT_USEC);
152 
153  /* wait for results */
154  while ((dev->resp_limit != dev->resp_count) &&
155  (xtimer_less(
156  xtimer_diff32_64(xtimer_now64(), sent_time),
157  xtimer_ticks_from_usec(RESP_TIMEOUT_USEC)))) {
158  mutex_lock(&(dev->resp_lock));
159  }
160 
161  xtimer_remove(&resp_timer);
162 
163  if (dev->resp_limit != dev->resp_count) {
164  DEBUG("[xbee] api_at_cmd: response timeout\n");
165  resp->status = 255;
166  mutex_unlock(&(dev->tx_lock));
167 
168  return;
169  }
170 
171  /* populate response data structure */
172  resp->status = dev->resp_buf[3];
173  resp->data_len = dev->resp_limit - 5;
174  if (resp->data_len > 0) {
175  memcpy(resp->data, &(dev->resp_buf[4]), resp->data_len);
176  }
177  mutex_unlock(&(dev->tx_lock));
178 }

180 /*
181  * Interrupt callbacks
182  */
183 static void _rx_cb(void *arg, uint8_t c)
184 {
185  xbee_t *dev = (xbee_t *)arg;
186 
187  switch (dev->int_state) {
188  case XBEE_INT_STATE_IDLE:
189  /* check for beginning of new data frame */
190  if (c == API_START_DELIMITER) {
191  dev->int_state = XBEE_INT_STATE_SIZE1;
192  }
193  break;
194  case XBEE_INT_STATE_SIZE1:
195  dev->int_size = ((uint16_t)c) << 8;
196  dev->int_state = XBEE_INT_STATE_SIZE2;
197  break;
198  case XBEE_INT_STATE_SIZE2:
199  dev->int_size += c;
200  dev->int_state = XBEE_INT_STATE_TYPE;
201  break;
202  case XBEE_INT_STATE_TYPE:
203  if (c == API_ID_RX_SHORT_ADDR || c == API_ID_RX_LONG_ADDR) {
204  /* in case old data was not processed, ignore incoming data */
205  if (dev->rx_count != 0) {
206  dev->int_state = XBEE_INT_STATE_IDLE;
207  return;
208  }
209  dev->rx_limit = dev->int_size + 1;
210  dev->rx_buf[dev->rx_count++] = c;
211  dev->int_state = XBEE_INT_STATE_RX;
212  }
213  else if (c == API_ID_AT_RESP) {
214  dev->resp_limit = dev->int_size;
215  dev->int_state = XBEE_INT_STATE_RESP;
216  }
217  else {
218  dev->int_state = XBEE_INT_STATE_IDLE;
219  }
220  break;
221  case XBEE_INT_STATE_RESP:
222  dev->resp_buf[dev->resp_count++] = c;
223  if (dev->resp_count == dev->resp_limit) {
224  /* here we ignore the checksum to prevent deadlocks */
225  mutex_unlock(&(dev->resp_lock));
226  dev->int_state = XBEE_INT_STATE_IDLE;
227  }
228  break;
229  case XBEE_INT_STATE_RX:
230  dev->rx_buf[dev->rx_count++] = c;
231  if (dev->rx_count == dev->rx_limit) {
232  /* packet is complete */
233  if (dev->event_callback) {
234  dev->event_callback((netdev2_t *)dev, NETDEV2_EVENT_ISR);
235  }
236  dev->int_state = XBEE_INT_STATE_IDLE;
237  }
238  break;
239  default:
240  /* this should never be the case */
241  break;
242  }
243 }

248 static int _get_addr_long(xbee_t *dev, uint8_t *val, size_t len)
249 {
250  uint8_t cmd[2];
251  resp_t resp;
252 
253  if (len < IEEE802154_LONG_ADDRESS_LEN) {
254  return -EOVERFLOW;
255  }
256 
257  /* read 4 high byte - AT command: SH*/
258  cmd[0] = 'S';
259  cmd[1] = 'H';
260  _api_at_cmd(dev, cmd, 2, &resp);
261  if (resp.status == 0) {
262  memcpy(val, resp.data, 4);
263  }
264  else {
265  return -ECANCELED;
266  }
267  /* read next 4 byte - AT command: SL */
268  cmd[1] = 'L';
269  _api_at_cmd(dev, cmd, 2, &resp);
270  if (resp.status == 0) {
271  memcpy(val + 4, resp.data, 4);
272  return IEEE802154_LONG_ADDRESS_LEN;
273  }
274  return -ECANCELED;
275 }

321 {
322  if (len != sizeof(uint16_t)) {
323  return -EOVERFLOW;
324  }
325 
326  switch (*val) {
327  case IEEE802154_LONG_ADDRESS_LEN:
328  dev->addr_flags |= XBEE_ADDR_FLAGS_LONG;
329 
330  /* disable short address */
331  uint8_t disabled_addr[] = { 0xFF, 0xFF };
332 
333  _set_short_addr(dev, disabled_addr);
334 
335  break;
336  case IEEE802154_SHORT_ADDRESS_LEN:

341 
342  break;
343  default:
344  return -EINVAL;
345  }

497 int xbee_build_hdr(xbee_t *dev, uint8_t *xhdr, size_t payload_len,
498  void *dst_addr, size_t addr_len)
499 {
500  /* make sure payload fits into a packet */
501  if (payload_len > XBEE_MAX_PAYLOAD_LENGTH) {
502  return -EOVERFLOW;
503  }
504 
505  /* set start delimiter, configure address and set options. Also make sure,
506  * that the link layer address is of known length */
507  xhdr[0] = API_START_DELIMITER;
508  xhdr[4] = dev->tx_fid++;
509  if (addr_len == IEEE802154_SHORT_ADDRESS_LEN) {
510  xhdr[3] = API_ID_TX_SHORT_ADDR;
511  xhdr[7] = dev->options;
512  }
513  else if (addr_len == IEEE802154_LONG_ADDRESS_LEN) {
514  xhdr[3] = API_ID_TX_LONG_ADDR;
515  xhdr[13] = dev->options;
516  }
517  else {
518  return -ENOMSG;
519  }
520 
521  /* finally configure the packet size and copy the actual dst address */
522  uint16_t size = (uint16_t)(payload_len + addr_len + 3);
523  xhdr[1] = (uint8_t)(size >> 8);
524  xhdr[2] = (uint8_t)(size & 0xff);
525  memcpy(&xhdr[5], dst_addr, addr_len);
526 
527  return (int)(addr_len + 6);
528 }

530 int xbee_parse_hdr(xbee_t *dev, const uint8_t *xhdr, xbee_l2hdr_t *l2hdr)
531 {
532  uint8_t alen;
533 
534  assert(xhdr && l2hdr);
535 
536  /* get the address length */
537  if (xhdr[0] == API_ID_RX_SHORT_ADDR) {
538  alen = IEEE802154_SHORT_ADDRESS_LEN;
539  }
540  else if (xhdr[0] == API_ID_RX_LONG_ADDR) {
541  alen = IEEE802154_LONG_ADDRESS_LEN;
542  }
543  else {
544  return -ENOMSG;
545  }
546 
547  /* copy the actual SRC address and the RSSI value */
548  memcpy(l2hdr->src_addr, &xhdr[1], alen);
549  l2hdr->rssi = xhdr[1 + alen];
550 
551  /* copy the destination address */
552  l2hdr->bcast = (xhdr[2 + alen] & OPT_BCAST_ADDR) ? 1 : 0;
553  if (l2hdr->bcast) {
554  memset(l2hdr->dst_addr, 0xff, alen);
555  }
556  else {
557  if (alen == IEEE802154_SHORT_ADDRESS_LEN) {
558  memcpy(l2hdr->dst_addr, dev->addr_short, alen);
559  }
560  else {
561  memcpy(l2hdr->dst_addr, dev->addr_long.uint8, alen);
562  }
563  }
564  l2hdr->addr_len = alen;
565 
566  return (int)(alen + 3);
567 }

569 int xbee_init(netdev2_t *dev)
570 {
571  uint8_t tmp[2];
572  xbee_t *xbee = (xbee_t *)dev;
573 
574  /* set default options */
575  xbee->addr_flags = XBEE_ADDR_FLAGS_LONG;
576  xbee->options = 0;
577  /* initialize buffers and locks*/
578  mutex_init(&(xbee->tx_lock));
579  mutex_init(&(xbee->resp_lock));
580  xbee->resp_limit = 1; /* needs to be greater then 0 initially */
581  xbee->rx_count = 0;
582  /* initialize UART and GPIO pins */
583  if (uart_init(xbee->p.uart, xbee->p.br, _rx_cb, xbee) != UART_OK) {
584  DEBUG("[xbee] init: error initializing UART\n");
585  return -ENXIO;
586  }
587  /* if reset pin is connected, do a hardware reset */
588  if (xbee->p.pin_reset != GPIO_UNDEF) {
589  gpio_clear(xbee->p.pin_reset);
590  xtimer_usleep(RESET_DELAY);
591  gpio_set(xbee->p.pin_reset);
592  }
593  /* put the XBee device into command mode */
594  xtimer_usleep(ENTER_CMD_MODE_DELAY);
595  _at_cmd(xbee, "+++");
596  xtimer_usleep(ENTER_CMD_MODE_DELAY);
597  /* disable non IEEE802.15.4 extensions */
598  _at_cmd(xbee, "ATMM2\r");
599  /* put XBee module in "API mode without escaped characters" */
600  _at_cmd(xbee, "ATAP1\r");
601  /* apply AT commands */
602  _at_cmd(xbee, "ATAC\r");
603  /* exit command mode */
604  _at_cmd(xbee, "ATCN\r");
605 
606  /* load long address (we can not set it, its read only for Xbee devices) */
607  if (_get_addr_long(xbee, xbee->addr_long.uint8, IEEE802154_LONG_ADDRESS_LEN) < 0) {
608  DEBUG("[xbee] init: error getting address\n");
609  return -EIO;
610  }
611  if (_set_addr(xbee, &((xbee->addr_long).uint8[6]), IEEE802154_SHORT_ADDRESS_LEN) < 0) {
612  DEBUG("[xbee] init: error setting short address\n");
613  return -EIO;
614  }
615  /* set default channel */
616  tmp[1] = 0;
617  tmp[0] = XBEE_DEFAULT_CHANNEL;
618  if (_set_channel(xbee, tmp, 2) < 0) {
619  DEBUG("[xbee] init: error setting channel\n");
620  return -EIO;
621  }
622  /* set default PAN ID */
623  tmp[1] = (uint8_t)(XBEE_DEFAULT_PANID >> 8);
624  tmp[0] = (uint8_t)(XBEE_DEFAULT_PANID & 0xff);
625  if (_set_panid(xbee, tmp, 2) < 0) {
626  DEBUG("[xbee] init: error setting PAN ID\n");
627  return -EIO;
628  }
629 
630  DEBUG("[xbee] init: Initialization successful\n");
631  return 0;
632 }

634 static int xbee_send(netdev2_t *dev, const struct iovec *vector, unsigned count)
635 {
636  xbee_t *xbee = (xbee_t *)dev;
637  size_t size;
638  uint8_t csum;
639 
640  assert(xbee && vector && (count > 0));
641 
642  /* calculate the checksum and the packet size */
643  size = vector[0].iov_len;
644  csum = _cksum(3, (uint8_t *)vector[0].iov_base, size);
645  for (unsigned i = 1; i < count; i++) {
646  size += vector[i].iov_len;
647  for (size_t p = 0; p < vector[i].iov_len; p++) {
648  csum -= ((uint8_t *)vector[i].iov_base)[p];
649  }
650  }
651 
652  /* make sure the data fits into a packet */
653  if (size >= XBEE_MAX_PKT_LENGTH) {
654  DEBUG("[xbee] send: data to send is too large for TX buffer\n");
655  return -1;
656  }
657 
658  /* send the actual data packet */
659  DEBUG("[xbee] send: now sending out %i byte\n", (int)size);
660  mutex_lock(&(xbee->tx_lock));
661  for (unsigned i = 0; i < count; i++) {
662  uart_write(xbee->p.uart, vector[i].iov_base, vector[i].iov_len);
663  }
664  uart_write(xbee->p.uart, &csum, 1);
665  mutex_unlock(&(xbee->tx_lock));
666 
667  /* return number of payload byte */
668  return (int)size;
669 }

671 static int xbee_recv(netdev2_t *dev, void *buf, size_t len, void *info)
672 {
673  (void)info;
674  size_t size;
675  xbee_t *xbee = (xbee_t *)dev;
676 
677  assert(xbee);
678 
679  /* make sure we have new data waiting */
680  if (xbee->rx_count != xbee->rx_limit) {
681  DEBUG("[xbee] recv: no data available for reading\n");
682  return 0;
683  }
684 
685  /* data available, so we read it (or it's size) */
686  size = (size_t)(xbee->rx_limit - 1);
687  if (buf == NULL) {
688  if (len > 0) {
689  DEBUG("[xbee] recv: reading size and dropping: %i\n", size);
690  xbee->rx_count = 0;
691  }
692  else {
693  DEBUG("[xbee] recv: reading size without dropping: %i\n", size);
694  }
695  }
696  else {
697  size = (size > len) ? len : size;
698  DEBUG("[xbee] recv: consuming packet: reading %i byte\n", size);
699  memcpy(buf, xbee->rx_buf, size);
700  xbee->rx_count = 0;
701  }
702 
703  return (int)size;
704 }

706 static void xbee_isr(netdev2_t *netdev)
707 {
708  xbee_t *dev = (xbee_t *)netdev;

723 static int xbee_get(netdev2_t *ndev, netopt_t opt, void *value, size_t max_len)
724 {
725  xbee_t *dev = (xbee_t *)ndev;
726  assert(dev);
727 
728  switch (opt) {
729  case NETOPT_ADDRESS:
730  assert(max_len >= IEEE802154_SHORT_ADDRESS_LEN);
731  memcpy(value, dev->addr_short, IEEE802154_SHORT_ADDRESS_LEN);
732  return IEEE802154_SHORT_ADDRESS_LEN;
733  case NETOPT_ADDRESS_LONG:
734  assert(max_len >= IEEE802154_LONG_ADDRESS_LEN);
735  memcpy(value, dev->addr_long.uint8, IEEE802154_LONG_ADDRESS_LEN);
736  return IEEE802154_LONG_ADDRESS_LEN;
737  case NETOPT_ADDR_LEN:
738  case NETOPT_SRC_LEN:
739  if (max_len < sizeof(uint16_t)) {
740  return -EOVERFLOW;
741  }
742  if (dev->addr_flags & XBEE_ADDR_FLAGS_LONG) {
743  *((uint16_t *)value) = IEEE802154_LONG_ADDRESS_LEN;
744  }
745  else {
746  *((uint16_t *)value) = IEEE802154_SHORT_ADDRESS_LEN;
747  }
748  return sizeof(uint16_t);
749  case NETOPT_IPV6_IID:
750  if (max_len < sizeof(eui64_t)) {
751  return -EOVERFLOW;
752  }
753  if (dev->addr_flags & XBEE_ADDR_FLAGS_LONG) {
754  ieee802154_get_iid(value, dev->addr_long.uint8,
755  IEEE802154_LONG_ADDRESS_LEN);
756  }
757  else {
758  ieee802154_get_iid(value, dev->addr_short,
759  IEEE802154_SHORT_ADDRESS_LEN);
760  }
761  return sizeof(eui64_t);
762  case NETOPT_CHANNEL:
763  return _get_channel(dev, (uint8_t *)value, max_len);
764  case NETOPT_MAX_PACKET_SIZE:
765  if (max_len < sizeof(uint16_t)) {
766  return -EOVERFLOW;
767  }
768  *((uint16_t *)value) = XBEE_MAX_PAYLOAD_LENGTH;
769  return sizeof(uint16_t);
770  case NETOPT_NID:
771  return _get_panid(dev, (uint8_t *)value, max_len);
772  default:
773  return -ENOTSUP;
774  }
775 }
