47 int uart_init(uart_t uart, uint32_t baudrate, uart_rx_cb_t rx_cb, void *arg)
48 {
49  uint16_t mantissa;
50  uint8_t fraction;
51  uint32_t clk;
52 
53  assert(uart < UART_NUMOF);
54 
55  /* save ISR context */
56  isr_ctx[uart].rx_cb = rx_cb;
57  isr_ctx[uart].arg = arg;
58 
59  /* configure RX and TX pin */
60  gpio_init(uart_config[uart].rx_pin, GPIO_IN);
61  gpio_init(uart_config[uart].tx_pin, GPIO_OUT);
62  /* set TX pin high to avoid garbage during further initialization */
63  gpio_set(uart_config[uart].tx_pin);
64 #ifdef CPU_FAM_STM32F1
65  gpio_init_af(uart_config[uart].tx_pin, GPIO_AF_OUT_PP);
66 #else
67  gpio_init_af(uart_config[uart].tx_pin, uart_config[uart].tx_af);
68  gpio_init_af(uart_config[uart].rx_pin, uart_config[uart].rx_af);
69 #endif
70 
71  /* enable the clock */
72  periph_clk_en(uart_config[uart].bus, uart_config[uart].rcc_mask);
73 
74  /* reset UART configuration -> defaults to 8N1 mode */
75  dev(uart)->CR1 = 0;
76  dev(uart)->CR2 = 0;
77  dev(uart)->CR3 = 0;
78 
79  /* calculate and apply baudrate */
80  clk = periph_apb_clk(uart_config[uart].bus) / baudrate;
81  mantissa = (uint16_t)(clk / 16);
82  fraction = (uint8_t)(clk - (mantissa * 16));
83  dev(uart)->BRR = ((mantissa & 0x0fff) << 4) | (fraction & 0x0f);
84 
85  /* enable RX interrupt if applicable */
86  if (rx_cb) {
87  NVIC_EnableIRQ(uart_config[uart].irqn);
88  dev(uart)->CR1 = (USART_CR1_UE | USART_CR1_TE | RXENABLE);
89  }
90  else {
91  dev(uart)->CR1 = (USART_CR1_UE | USART_CR1_TE);
92  }
93 
94  return UART_OK;
95 }

97 void uart_write(uart_t uart, const uint8_t *data, size_t len)
98 {
99  assert(uart < UART_NUMOF);
100 
101  for (size_t i = 0; i < len; i++) {
102 #if defined(CPU_FAM_STM32F0) || defined(CPU_FAM_STM32F3)
103  while (!(dev(uart)->ISR & USART_ISR_TXE)) {}
104  dev(uart)->TDR = data[i];
105 #else
106  while (!(dev(uart)->SR & USART_SR_TXE)) {}
107  dev(uart)->DR = data[i];
108 #endif
109  }
110 
111  /* make sure the function is synchronous by waiting for the transfer to
112  * finish */
113 #if defined(CPU_FAM_STM32F0) || defined(CPU_FAM_STM32F3)
114  while (!(dev(uart)->ISR & USART_ISR_TC)) {}
115 #else
116  while (!(dev(uart)->SR & USART_SR_TC)) {}
117 #endif
118 }

132 static inline void irq_handler(uart_t uart)
133 {
134 #if defined(CPU_FAM_STM32F0) || defined(CPU_FAM_STM32F3)
135 
136  uint32_t status = dev(uart)->ISR;
137 
138  if (status & USART_ISR_RXNE) {
139  isr_ctx[uart].rx_cb(isr_ctx[uart].arg, (uint8_t)dev(uart)->RDR);
140  }
141  if (status & USART_ISR_ORE) {
142  dev(uart)->ICR |= USART_ICR_ORECF; /* simply clear flag on overrun */
143  }
144 
145 #else
146 
147  uint32_t status = dev(uart)->SR;
148 
149  if (status & USART_SR_RXNE) {
150  isr_ctx[uart].rx_cb(isr_ctx[uart].arg, (uint8_t)dev(uart)->DR);
151  }
152  if (status & USART_SR_ORE) {
153  /* ORE is cleared by reading SR and DR sequentially */
154  dev(uart)->DR;
155  }
156 
157 #endif
158 
159  cortexm_isr_end();
160 }
