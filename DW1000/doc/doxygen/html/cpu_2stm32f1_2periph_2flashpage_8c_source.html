28 void flashpage_write(int page, void *data)
29 {
30  assert(page < FLASHPAGE_NUMOF);
31 
32  uint16_t *page_addr = flashpage_addr(page);
33  uint16_t *data_addr = (uint16_t *)data;
34  uint32_t hsi_state = (RCC->CR & RCC_CR_HSION);
35 
36  /* the internal RC oscillator (HSI) must be enabled */
37  RCC->CR |= (RCC_CR_HSION);
38  while (!(RCC->CR & RCC_CR_HSIRDY)) {}
39 
40  /* unlock the flash module */
41  DEBUG("[flashpage] unlocking the flash module\n");
42  if (FLASH->CR & FLASH_CR_LOCK) {
43  FLASH->KEYR = FLASH_KEY1;
44  FLASH->KEYR = FLASH_KEY2;
45  }
46 
47  /* ERASE sequence */
48  /* make sure no flash operation is ongoing */
49  DEBUG("[flashpage] erase: waiting for any operation to finish\n");
50  while (FLASH->SR & FLASH_SR_BSY) {}
51  /* set page erase bit and program page address */
52  DEBUG("[flashpage] erase: setting the erase bit and page address\n");
53  FLASH->CR |= FLASH_CR_PER;
54  FLASH->AR = (uint32_t)flashpage_addr(page);
55  DEBUG("address to erase: %p\n", flashpage_addr(page));
56  /* trigger the page erase and wait for it to be finished */
57  DEBUG("[flashpage] erase: trigger the page erase\n");
58  FLASH->CR |= FLASH_CR_STRT;
59  DEBUG("[flashpage] erase: wait as long as device is busy\n");
60  while (FLASH->SR & FLASH_SR_BSY) {}
61  /* reset PER bit */
62  DEBUG("[flashpage] erase: resetting the page erase bit\n");
63  FLASH->CR &= ~(FLASH_CR_PER);
64 
65  /* WRITE sequence */
66  if (data != NULL) {
67  DEBUG("[flashpage] write: now writing the data\n");
68  /* set PG bit and program page to flash */
69  FLASH->CR |= FLASH_CR_PG;
70  for (unsigned i = 0; i < (FLASHPAGE_SIZE / 2); i++) {
71  *page_addr++ = data_addr[i];
72  while (FLASH->SR & FLASH_SR_BSY) {}
73  }
74  /* clear program bit again */
75  FLASH->CR &= ~(FLASH_CR_PG);
76  DEBUG("[flashpage] write: done writing data\n");
77  }
78 
79  /* finally, lock the flash module again */
80  DEBUG("flashpage] now locking the flash module again\n");
81  FLASH->CR |= FLASH_CR_LOCK;
82 
83  /* restore the HSI state */
84  if (hsi_state) {
85  RCC->CR &= ~(RCC_CR_HSION);
86  while (RCC->CR & RCC_CR_HSIRDY) {}
87  }
88 }
