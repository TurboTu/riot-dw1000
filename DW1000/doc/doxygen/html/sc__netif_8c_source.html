122  (unsigned) stats->rx_bytes,

512 static int _netif_set_u16(kernel_pid_t dev, netopt_t opt, char *u16_str)
513 {
514  unsigned int res;
515  bool hex = false;
516 
517  if (_is_number(u16_str)) {
518  if ((res = strtoul(u16_str, NULL, 10)) == ULONG_MAX) {
519  puts("error: unable to parse value.\n"
520  "Must be a 16-bit unsigned integer (dec or hex)\n");
521  return 1;
522  }
523  }
524  else {
525  if ((res = strtoul(u16_str, NULL, 16)) == ULONG_MAX) {
526  puts("error: unable to parse value.\n"
527  "Must be a 16-bit unsigned integer (dec or hex)\n");
528  return 1;
529  }
530 
531  hex = true;
532  }
533 
534  if (res > 0xffff) {
535  puts("error: unable to parse value.\n"
536  "Must be a 16-bit unsigned integer (dec or hex)\n");
537  return 1;
538  }
539 
540  if (gnrc_netapi_set(dev, opt, 0, (uint16_t *)&res, sizeof(uint16_t)) < 0) {
541  printf("error: unable to set ");
542  _print_netopt(opt);
543  puts("");
544  return 1;
545  }
546 
547  printf("success: set ");
548  _print_netopt(opt);
549  printf(" on interface %" PRIkernel_pid " to ", dev);
550 
551  if (hex) {
552  printf("0x%04x\n", res);
553  }
554  else {
555  printf("%u\n", res);
556  }
557 
558  return 0;
559 }
