61 {

63  ipv6_addr_t *next_hop_ip = NULL, *prefix = NULL;
64  bool dst_link_local = ipv6_addr_is_link_local(dst);
65 
66 #ifdef MODULE_FIB
67  ipv6_addr_t next_hop_actual; /* FIB copies address into this variable */
68  /* don't look-up link local addresses in FIB */
69  if (!dst_link_local) {
70  size_t next_hop_size = sizeof(ipv6_addr_t);
71  uint32_t next_hop_flags = 0;
72 
73  if ((next_hop_ip == NULL) &&
74  (fib_get_next_hop(&gnrc_ipv6_fib_table, &iface, next_hop_actual.u8,
75  &next_hop_size, &next_hop_flags, (uint8_t *)dst,
76  sizeof(ipv6_addr_t), 0) >= 0) &&
77  (next_hop_size == sizeof(ipv6_addr_t))) {
78  next_hop_ip = &next_hop_actual;
79  }
80  }
81 #endif
82 
83  if (next_hop_ip == NULL) { /* no route to host */
84  if (!dst_link_local) {
85  if (iface == KERNEL_PID_UNDEF) {
86  /* gnrc_ipv6_netif_t doubles as prefix list */
87  iface = gnrc_ipv6_netif_find_by_prefix(&prefix, dst);
88  }
89  else {
90  /* gnrc_ipv6_netif_t doubles as prefix list */
91  prefix = gnrc_ipv6_netif_match_prefix(iface, dst);
92  }
93  }
94 
95  if (dst_link_local || ((prefix != NULL) &&
96  (gnrc_ipv6_netif_addr_get(prefix)->flags & /* prefix is on-link */
97  GNRC_IPV6_NETIF_ADDR_FLAGS_NDP_ON_LINK))) {
98  next_hop_ip = dst;
99  }
100  }
101 
102  /* dst has not an on-link prefix */
103  if (next_hop_ip == NULL) {
104  next_hop_ip = gnrc_ndp_internal_default_router();
105  }
106 
107  if (next_hop_ip == NULL) {
108  next_hop_ip = dst; /* Just look if it's in the neighbor cache
109  * (aka on-link but not registered in prefix list as such) */
110  }
111 
112  /* start address resolution */
113  nc_entry = gnrc_ipv6_nc_get(iface, next_hop_ip);
114 
115  if ((nc_entry != NULL) && gnrc_ipv6_nc_is_reachable(nc_entry)) {
116  DEBUG("ndp node: found reachable neighbor (%s => ",
117  ipv6_addr_to_str(addr_str, &nc_entry->ipv6_addr, sizeof(addr_str)));
118  DEBUG("%s)\n",
119  gnrc_netif_addr_to_str(addr_str, sizeof(addr_str),
120  nc_entry->l2_addr, nc_entry->l2_addr_len));
121 
122  if (gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_STALE) {
123  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_DELAY);
124  }
125  return gnrc_ipv6_nc_get_l2_addr(l2addr, l2addr_len, nc_entry);
126  }
127  else if (nc_entry == NULL) {
128  gnrc_pktqueue_t *pkt_node;
129  ipv6_addr_t dst_sol;
130 
131  nc_entry = gnrc_ipv6_nc_add(iface, next_hop_ip, NULL, 0,
132  GNRC_IPV6_NC_STATE_INCOMPLETE << GNRC_IPV6_NC_STATE_POS);
133 
134  if (nc_entry == NULL) {
135  DEBUG("ndp node: could not create neighbor cache entry\n");
136  return KERNEL_PID_UNDEF;
137  }
138 
139  pkt_node = _alloc_pkt_node(pkt);
140 
141  if (pkt_node == NULL) {
142  DEBUG("ndp node: could not add packet to packet queue\n");
143  }
144  else {
145  /* prevent packet from being released by IPv6 */
146  gnrc_pktbuf_hold(pkt_node->pkt, 1);
147  gnrc_pktqueue_add(&nc_entry->pkts, pkt_node);
148  }
149 
150  /* address resolution */
151  ipv6_addr_set_solicited_nodes(&dst_sol, next_hop_ip);
152 
153  if (iface == KERNEL_PID_UNDEF) {
154  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
155  size_t ifnum = gnrc_netif_get(ifs);
156 
157  for (size_t i = 0; i < ifnum; i++) {
158  gnrc_ndp_internal_send_nbr_sol(ifs[i], NULL, next_hop_ip, &dst_sol);
159  }
160 
161  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, GNRC_NDP_RETRANS_TIMER,
162  GNRC_NDP_MSG_NBR_SOL_RETRANS, gnrc_ipv6_pid);
163  }
164  else {
165  gnrc_ipv6_netif_t *ipv6_iface = gnrc_ipv6_netif_get(iface);
166 
167  gnrc_ndp_internal_send_nbr_sol(iface, NULL, next_hop_ip, &dst_sol);
168 
169  mutex_lock(&ipv6_iface->mutex);
170  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, ipv6_iface->retrans_timer,
171  GNRC_NDP_MSG_NBR_SOL_RETRANS, gnrc_ipv6_pid);
172  mutex_unlock(&ipv6_iface->mutex);
173  }
174  }
175 
176  return KERNEL_PID_UNDEF;
177 }
