55 {
56  uint8_t reg;

62

64  return -1;
65  }
66 
67  i2c_acquire(dev->i2c);

84  return -4;

88  reg = MPL3115A2_PT_DATA_CFG_TDEFE

91 
92  i2c_acquire(dev->i2c);
93  if (i2c_write_regs(dev->i2c, dev->addr, MPL3115A2_PT_DATA_CFG, &reg, 1) != 1) {

185 int mpl3115a2_read_pressure(mpl3115a2_t *dev, uint32_t *pres, uint8_t *status)
186 {
187  uint8_t buf[4];
188 
189  if (dev->initialized == false) {
190  return -1;
191  }
192 
193  i2c_acquire(dev->i2c);
194  if (i2c_read_regs(dev->i2c, dev->addr, MPL3115A2_STATUS, buf, 4) != 4) {
195  i2c_release(dev->i2c);
196  return -1;
197  }
198  i2c_release(dev->i2c);
199 
200  *status = buf[0];
201 
202  *pres = ((uint32_t)buf[1] << 16) | ((uint32_t)buf[2] << 8) | buf[3];
203  *pres = *pres / 64;
204 
205  return 0;
206 }
