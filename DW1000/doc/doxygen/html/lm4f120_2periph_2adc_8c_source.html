114  ROM_ADCSequenceStepConfigure(ADC0_BASE, SEQ, 0, line | ADC_CTL_IE | ADC_CTL_END);
115  /* set resolution */
116  ROM_ADCResolutionSet(ADC0_BASE, (unsigned long)res);
117 
118  /* start conversion and wait for results */
119  ROM_ADCSequenceEnable(ADC0_BASE, SEQ);
120  ROM_ADCIntClear(ADC0_BASE, SEQ);
121  ROM_ADCProcessorTrigger(ADC0_BASE, SEQ);
122  while (!ROM_ADCIntStatus(ADC0_BASE, SEQ, false)) {}
123 
124  /* get results */
125  ROM_ADCSequenceDataGet(ADC0_BASE, SEQ, (unsigned long *) value);
126 
127  /* disable device again */
128  ROM_ADCSequenceDisable(ADC0_BASE, SEQ);
129  done();
