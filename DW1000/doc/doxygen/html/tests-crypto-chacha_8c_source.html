98  const uint8_t key[32], const uint8_t iv[8],
99  const uint32_t after_init[16],
100  const uint8_t block0[64], const uint8_t block1[64])

106  TEST_ASSERT_EQUAL_INT(0, memcmp(ctx.state, after_init, 64));
107 
108  chacha_keystream_bytes(&ctx, block);
109  TEST_ASSERT_EQUAL_INT(0, memcmp(block, block0, 64));

111  chacha_keystream_bytes(&ctx, block);

113 }
