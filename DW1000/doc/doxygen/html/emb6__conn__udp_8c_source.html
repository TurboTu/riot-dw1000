65  int family, uint16_t port)

119 int conn_udp_recvfrom(conn_udp_t *conn, void *data, size_t max_len, void *addr,
120  size_t *addr_len, uint16_t *port)
121 {
122  int res = -EIO;
123  msg_t msg;
124 
125  if (conn->sock.input_callback == NULL) {
126  return -ENOTSOCK;
127  }
128  mutex_lock(&conn->mutex);
129  if (conn->waiting_thread != KERNEL_PID_UNDEF) {
130  mutex_unlock(&conn->mutex);
131  return -EALREADY;
132  }
133  conn->waiting_thread = sched_active_pid;
134  mutex_unlock(&conn->mutex);
135  msg_receive(&msg);
136  if (msg.type == _MSG_TYPE_CLOSE) {
137  conn->waiting_thread = KERNEL_PID_UNDEF;
138  return -EINTR;
139  }
140  else if (msg.type == _MSG_TYPE_RCV) {
141  mutex_lock(&conn->mutex);
142  if (msg.content.ptr == conn) {
143  if (max_len < conn->recv_info.datalen) {
144  conn->waiting_thread = KERNEL_PID_UNDEF;
145  mutex_unlock(&conn->mutex);
146  return -ENOBUFS;
147  }
148  memcpy(data, conn->recv_info.data, conn->recv_info.datalen);
149  memcpy(addr, conn->recv_info.src, sizeof(ipv6_addr_t));
150  *addr_len = sizeof(ipv6_addr_t);
151  *port = conn->recv_info.src_port;
152  res = (int)conn->recv_info.datalen;
153  }
154  conn->waiting_thread = KERNEL_PID_UNDEF;
155  mutex_unlock(&conn->mutex);
156  }
157  return res;
158 }

160 int conn_udp_sendto(const void *data, size_t len, const void *src, size_t src_len,
161  const void *dst, size_t dst_len, int family, uint16_t sport,
162  uint16_t dport)
163 {
164  int res;
165  _send_cmd_t send_cmd;
166 
167  if (!send_registered) {
168  if (evproc_regCallback(EVENT_TYPE_CONN_SEND, _output_callback) != E_SUCCESS) {
169  return -EIO;
170  }
171  else {
172  send_registered = true;
173  }
174  }
175  mutex_init(&send_cmd.mutex);
176  if ((len > (UIP_BUFSIZE - (UIP_LLH_LEN + UIP_IPUDPH_LEN))) ||
177  (len > UINT16_MAX)) {
178  return -EMSGSIZE;
179  }
180  if ((dst_len > sizeof(ipv6_addr_t)) || (family != AF_INET6)) {
181  return -EAFNOSUPPORT;
182  }
183  mutex_lock(&send_cmd.mutex);
184  send_cmd.data = data;
185  send_cmd.data_len = (uint16_t)len;
186  if ((res = _reg_and_bind(&send_cmd.sock, NULL, NULL, sport)) < 0) {
187  mutex_unlock(&send_cmd.mutex);
188  return res;
189  }
190  udp_socket_connect(&send_cmd.sock, (uip_ipaddr_t *)dst, dport); /* can't fail at this point */
191  /* change to emb6 thread context */
192  if (evproc_putEvent(E_EVPROC_TAIL, EVENT_TYPE_CONN_SEND, &send_cmd) != E_SUCCESS) {
193  udp_socket_close(&send_cmd.sock);
194  mutex_unlock(&send_cmd.mutex);
195  return -EIO;
196  }
197  /* block thread until data was send */
198  mutex_lock(&send_cmd.mutex);
199  udp_socket_close(&send_cmd.sock);
200  mutex_unlock(&send_cmd.mutex);
201 
202  return send_cmd.res;
203 }

205 static void _input_callback(struct udp_socket *c, void *ptr,
206  const uip_ipaddr_t *src_addr, uint16_t src_port,
207  const uip_ipaddr_t *dst_addr, uint16_t dst_port,

214  mutex_lock(&conn->mutex);
215  if (conn->waiting_thread != KERNEL_PID_UNDEF) {
216  msg_t msg;

219  conn->recv_info.data = data;

221  msg.type = _MSG_TYPE_RCV;
222  msg.content.ptr = conn;

224  msg_send(&msg, conn->waiting_thread);
225  }

227  mutex_unlock(&conn->mutex);

231 static void _output_callback(c_event_t c_event, p_data_t p_data)
232 {
233  if ((c_event != EVENT_TYPE_CONN_SEND) || (p_data == NULL)) {
234  return;
235  }
236 
237  _send_cmd_t *send_cmd = (_send_cmd_t *)p_data;
238 
239  if ((send_cmd->res = udp_socket_send(&send_cmd->sock, send_cmd->data, send_cmd->data_len)) < 0) {
240  send_cmd->res = -EHOSTUNREACH;
241  }
242  mutex_unlock(&send_cmd->mutex);
243 }
