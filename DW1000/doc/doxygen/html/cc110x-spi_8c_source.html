50 void cc110x_cs(cc110x_t *dev)
51 {
52  volatile int retry_count = 0;
53  /* Switch MISO/GDO1 to GPIO input mode */
54 #ifndef GPIO_READS_SPI_PINS
55  gpio_init(dev->params.gdo1, GPIO_IN);
56 #endif
57  /* CS to low */
58  gpio_clear(dev->params.cs);
59  /* Wait for SO to go low (voltage regulator
60  * has stabilized and the crystal is running) */
61  while (gpio_read(dev->params.gdo1)) {
62  /* Wait ~500us and try again */
63  xtimer_usleep(CS_SO_WAIT_TIME);
64 
65  if (gpio_read(dev->params.gdo1)) {
66  retry_count++;
67 
68  if (retry_count > CC110X_GDO1_LOW_RETRY) {
69  puts("[CC110X spi] fatal error\n");
70  break;
71  }
72 
73  gpio_set(dev->params.cs);
74  gpio_clear(dev->params.cs);
75  }
76  }
77  /* Switch MISO/GDO1 to spi mode */
78 #ifndef GPIO_READS_SPI_PINS
79  spi_init_pins(dev->params.spi);
80 #endif
81 }

96 void cc110x_readburst_reg(cc110x_t *dev, uint8_t addr, char *buffer, uint8_t count)
97 {
98  int i = 0;
99  unsigned int cpsr;
100  lock(dev);
101  cpsr = irq_disable();
102  cc110x_cs(dev);
103  spi_transfer_byte(dev->params.spi, SPI_CS_UNDEF, false,
104  (addr | CC110X_READ_BURST));
105  while (i < count) {
106  buffer[i] = (char)spi_transfer_byte(dev->params.spi, SPI_CS_UNDEF,
107  false, CC110X_NOBYTE);
108  i++;
109  }
110  gpio_set(dev->params.cs);
111  irq_restore(cpsr);

157 uint8_t cc110x_get_reg_robust(cc110x_t *dev, uint8_t addr)
158 {
159  uint8_t res1, res2;
160  unsigned int cpsr;
161  lock(dev);
162  cpsr = irq_disable();
163  cc110x_cs(dev);
164  do {
165  res1 = spi_transfer_reg(dev->params.spi, SPI_CS_UNDEF,
166  (addr | CC110X_READ_BURST), CC110X_NOBYTE);
167  res2 = spi_transfer_reg(dev->params.spi, SPI_CS_UNDEF,
168  (addr | CC110X_READ_BURST), CC110X_NOBYTE);
169  } while (res1 != res2);
170  gpio_set(dev->params.cs);
171  irq_restore(cpsr);
172  spi_release(dev->params.spi);
