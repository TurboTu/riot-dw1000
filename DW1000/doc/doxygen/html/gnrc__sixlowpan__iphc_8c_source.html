92 static inline bool _context_overlaps_iid(gnrc_sixlowpan_ctx_t *ctx,
93  ipv6_addr_t *addr,
94  eui64_t *iid)
95 {
96  uint8_t byte_mask[] = {0xff, 0x7f, 0x3f, 0x1f, 0x0f, 0x07, 0x03, 0x01};
97 
98  if ((ctx == NULL) || (ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_COMP)) {
99  return false;
100  }
101 
102  return ((ctx->prefix_len == 128) || /* Full-length prefix overlaps IID in any case */
103  ((ctx->prefix_len > 64) && /* otherwise, if bigger than 64-bit */
104  /* compare bytes until prefix length with IID */
105  (memcmp(&(addr->u8[(ctx->prefix_len / 8) + 1]),
106  &(iid->uint8[(ctx->prefix_len / 8) - 7]),
107  sizeof(network_uint64_t) - ((ctx->prefix_len / 8) - 7)) == 0) &&
108  /* compare bits at prefix length with IID */
109  (addr->u8[(ctx->prefix_len / 8)] & byte_mask[ctx->prefix_len % 8]) ==
110  (iid->uint8[(ctx->prefix_len / 8) - 8] & byte_mask[ctx->prefix_len % 8])));
111 }

114 inline static size_t iphc_nhc_udp_decode(gnrc_pktsnip_t *pkt, gnrc_pktsnip_t **dec_hdr,
115  size_t datagram_size, size_t offset)
116 {
117  uint8_t *payload = pkt->data;
118  gnrc_pktsnip_t *ipv6 = *dec_hdr;
119  ipv6_hdr_t *ipv6_hdr = ipv6->data;
120 #ifdef MODULE_GNRC_UDP
121  const gnrc_nettype_t snip_type = GNRC_NETTYPE_UDP;
122 #else
123  const gnrc_nettype_t snip_type = GNRC_NETTYPE_UNDEF;
124 #endif
125  gnrc_pktsnip_t *udp = NULL;
126  uint8_t udp_nhc = payload[offset++];
127  uint8_t tmp;
128  udp_hdr_t *udp_hdr;
129 
130  if (datagram_size == 0) { /* received packet is not fragmented */
131  udp = gnrc_pktbuf_add(NULL, NULL, sizeof(udp_hdr_t),
132  snip_type);
133  if (udp == NULL) {
134  DEBUG("6lo: error on IPHC NHC UDP decoding\n");
135  return 0;
136  }
137  udp_hdr = udp->data;
138  }
139  else { /* received packet is fragmented */
140  /* reassembly is in-place => don't allocate new packet snip */
141  /* TODO: account for extension headers */
142  udp_hdr = (udp_hdr_t *)(ipv6_hdr + 1);
143  }
144  network_uint16_t *src_port = &(udp_hdr->src_port);
145  network_uint16_t *dst_port = &(udp_hdr->dst_port);
146 
147  switch (udp_nhc & NHC_UDP_PP_MASK) {
148 
149  case NHC_UDP_SD_INLINE:
150  DEBUG("6lo iphc nhc: SD_INLINE\n");
151  src_port->u8[0] = payload[offset++];
152  src_port->u8[1] = payload[offset++];
153  dst_port->u8[0] = payload[offset++];
154  dst_port->u8[1] = payload[offset++];
155  break;
156 
157  case NHC_UDP_S_INLINE:
158  DEBUG("6lo iphc nhc: S_INLINE\n");
159  src_port->u8[0] = payload[offset++];
160  src_port->u8[1] = payload[offset++];
161  *dst_port = byteorder_htons(payload[offset++] + NHC_UDP_8BIT_PORT);
162  break;
163 
164  case NHC_UDP_D_INLINE:
165  DEBUG("6lo iphc nhc: D_INLINE\n");
166  *src_port = byteorder_htons(payload[offset++] + NHC_UDP_8BIT_PORT);
167  dst_port->u8[0] = payload[offset++];
168  dst_port->u8[1] = payload[offset++];
169  break;
170 
171  case NHC_UDP_SD_ELIDED:
172  DEBUG("6lo iphc nhc: SD_ELIDED\n");
173  tmp = payload[offset++];
174  *src_port = byteorder_htons((tmp >> 4) + NHC_UDP_4BIT_PORT);
175  *dst_port = byteorder_htons((tmp & 0xf) + NHC_UDP_4BIT_PORT);
176  break;
177 
178  default:
179  break;
180  }
181 
182  if ((udp_nhc & NHC_UDP_C_ELIDED) != 0) {
183  DEBUG("6lo iphc nhc: unsupported elided checksum\n");
184  gnrc_pktbuf_release(udp);
185  return 0;
186  }
187  else {
188  udp_hdr->checksum.u8[0] = payload[offset++];
189  udp_hdr->checksum.u8[1] = payload[offset++];
190  }
191 
192  /* TODO subtract extension header length */
193  if (udp != NULL) {
194  udp_hdr->length = byteorder_htons(pkt->size - offset + sizeof(udp_hdr_t));
195  }
196  else {
197  udp_hdr->length = byteorder_htons(datagram_size - sizeof(ipv6_hdr_t));
198  }
199  ipv6_hdr->nh = PROTNUM_UDP;
200  ipv6_hdr->len = udp_hdr->length;
201 
202  if (udp != NULL) { /* prepend udp header in case of packet not being fragmented */
203  udp->next = ipv6;
204  *dec_hdr = udp;
205  }
206 
207  return offset;
208 }

211 size_t gnrc_sixlowpan_iphc_decode(gnrc_pktsnip_t **dec_hdr, gnrc_pktsnip_t *pkt,
212  size_t datagram_size, size_t offset,
213  size_t *nh_len)
214 {
215  gnrc_pktsnip_t *ipv6;
216  gnrc_netif_hdr_t *netif_hdr = pkt->next->data;
217  ipv6_hdr_t *ipv6_hdr;
218  uint8_t *iphc_hdr = pkt->data;
219  size_t payload_offset = SIXLOWPAN_IPHC_HDR_LEN;
220  gnrc_sixlowpan_ctx_t *ctx = NULL;
221 
222  assert(dec_hdr != NULL);
223  ipv6 = *dec_hdr;
224  assert(ipv6 != NULL);
225  assert(ipv6->size >= sizeof(ipv6_hdr_t));
226 
227  ipv6_hdr = ipv6->data;
228  iphc_hdr += offset;
229 
230  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_CID_EXT) {
231  payload_offset++;
232  }
233 
234  ipv6_hdr_set_version(ipv6_hdr);
235 
236  switch (iphc_hdr[IPHC1_IDX] & SIXLOWPAN_IPHC1_TF) {
237  case IPHC_TF_ECN_DSCP_FL:
238  ipv6_hdr_set_tc(ipv6_hdr, iphc_hdr[payload_offset++]);
239  ipv6_hdr->v_tc_fl.u8[1] |= iphc_hdr[payload_offset++] & 0x0f;
240  ipv6_hdr->v_tc_fl.u8[2] |= iphc_hdr[payload_offset++];
241  ipv6_hdr->v_tc_fl.u8[3] |= iphc_hdr[payload_offset++];
242  break;
243 
244  case IPHC_TF_ECN_FL:
245  ipv6_hdr_set_tc_ecn(ipv6_hdr, iphc_hdr[payload_offset] >> 6);
246  ipv6_hdr_set_tc_dscp(ipv6_hdr, 0);
247  ipv6_hdr->v_tc_fl.u8[1] |= iphc_hdr[payload_offset++] & 0x0f;
248  ipv6_hdr->v_tc_fl.u8[2] |= iphc_hdr[payload_offset++];
249  ipv6_hdr->v_tc_fl.u8[3] |= iphc_hdr[payload_offset++];
250  break;
251 
252  case IPHC_TF_ECN_DSCP:
253  ipv6_hdr_set_tc(ipv6_hdr, iphc_hdr[payload_offset++]);
254  ipv6_hdr_set_fl(ipv6_hdr, 0);
255  break;
256 
257  case IPHC_TF_ECN_ELIDE:
258  ipv6_hdr_set_tc(ipv6_hdr, 0);
259  ipv6_hdr_set_fl(ipv6_hdr, 0);
260  break;
261  }
262 
263  if (!(iphc_hdr[IPHC1_IDX] & SIXLOWPAN_IPHC1_NH)) {
264  ipv6_hdr->nh = iphc_hdr[payload_offset++];
265  }
266 
267  switch (iphc_hdr[IPHC1_IDX] & SIXLOWPAN_IPHC1_HL) {
268  case IPHC_HL_INLINE:
269  ipv6_hdr->hl = iphc_hdr[payload_offset++];
270  break;
271 
272  case IPHC_HL_1:
273  ipv6_hdr->hl = 1;
274  break;
275 
276  case IPHC_HL_64:
277  ipv6_hdr->hl = 64;
278  break;
279 
280  case IPHC_HL_255:
281  ipv6_hdr->hl = 255;
282  break;
283  }
284 
285  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_SAC) {
286  uint8_t sci = 0;
287 
288  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_CID_EXT) {
289  sci = iphc_hdr[CID_EXT_IDX] >> 4;
290  }
291 
292  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_SAM) {
293  ctx = gnrc_sixlowpan_ctx_lookup_id(sci);
294 
295  if (ctx == NULL) {
296  DEBUG("6lo iphc: could not find source context\n");
297  return 0;
298  }
299  }
300  }
301 
302  switch (iphc_hdr[IPHC2_IDX] & (SIXLOWPAN_IPHC2_SAC | SIXLOWPAN_IPHC2_SAM)) {
303 
304  case IPHC_SAC_SAM_FULL:
305  /* take full 128 from inline */
306  memcpy(&(ipv6_hdr->src), iphc_hdr + payload_offset, 16);
307  payload_offset += 16;
308  break;
309 
310  case IPHC_SAC_SAM_64:
311  ipv6_addr_set_link_local_prefix(&ipv6_hdr->src);
312  memcpy(ipv6_hdr->src.u8 + 8, iphc_hdr + payload_offset, 8);
313  payload_offset += 8;
314  break;
315 
316  case IPHC_SAC_SAM_16:
317  ipv6_addr_set_link_local_prefix(&ipv6_hdr->src);
318  ipv6_hdr->src.u32[2] = byteorder_htonl(0x000000ff);
319  ipv6_hdr->src.u16[6] = byteorder_htons(0xfe00);
320  memcpy(ipv6_hdr->src.u8 + 14, iphc_hdr + payload_offset, 2);
321  payload_offset += 2;
322  break;
323 
324  case IPHC_SAC_SAM_L2:
325  ieee802154_get_iid((eui64_t *)(&ipv6_hdr->src.u64[1]),
326  gnrc_netif_hdr_get_src_addr(netif_hdr),
327  netif_hdr->src_l2addr_len);
328  ipv6_addr_set_link_local_prefix(&ipv6_hdr->src);
329  break;
330 
331  case IPHC_SAC_SAM_UNSPEC:
332  ipv6_addr_set_unspecified(&ipv6_hdr->src);
333  break;
334 
335  case IPHC_SAC_SAM_CTX_64:
336  assert(ctx != NULL);
337  memcpy(ipv6_hdr->src.u8 + 8, iphc_hdr + payload_offset, 8);
338  ipv6_addr_init_prefix(&ipv6_hdr->src, &ctx->prefix,
339  ctx->prefix_len);
340  payload_offset += 8;
341  break;
342 
343  case IPHC_SAC_SAM_CTX_16:
344  assert(ctx != NULL);
345  ipv6_hdr->src.u32[2] = byteorder_htonl(0x000000ff);
346  ipv6_hdr->src.u16[6] = byteorder_htons(0xfe00);
347  memcpy(ipv6_hdr->src.u8 + 14, iphc_hdr + payload_offset, 2);
348  ipv6_addr_init_prefix(&ipv6_hdr->src, &ctx->prefix,
349  ctx->prefix_len);
350  payload_offset += 2;
351  break;
352 
353  case IPHC_SAC_SAM_CTX_L2:
354  assert(ctx != NULL);
355  ieee802154_get_iid((eui64_t *)(&ipv6_hdr->src.u64[1]),
356  gnrc_netif_hdr_get_src_addr(netif_hdr),
357  netif_hdr->src_l2addr_len);
358  ipv6_addr_init_prefix(&ipv6_hdr->src, &ctx->prefix,
359  ctx->prefix_len);
360  break;
361  }
362 
363  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_DAC) {
364  uint8_t dci = 0;
365 
366  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_CID_EXT) {
367  dci = iphc_hdr[CID_EXT_IDX] & 0x0f;
368  }
369 
370  if (iphc_hdr[IPHC2_IDX] & SIXLOWPAN_IPHC2_DAM) {
371  ctx = gnrc_sixlowpan_ctx_lookup_id(dci);
372 
373  if (ctx == NULL) {
374  DEBUG("6lo iphc: could not find destination context\n");
375  return 0;
376  }
377  }
378  }
379 
380  switch (iphc_hdr[IPHC2_IDX] & (SIXLOWPAN_IPHC2_M | SIXLOWPAN_IPHC2_DAC |
381  SIXLOWPAN_IPHC2_DAM)) {
382  case IPHC_M_DAC_DAM_U_FULL:
383  case IPHC_M_DAC_DAM_M_FULL:
384  memcpy(&(ipv6_hdr->dst.u8), iphc_hdr + payload_offset, 16);
385  payload_offset += 16;
386  break;
387 
388  case IPHC_M_DAC_DAM_U_64:
389  ipv6_addr_set_link_local_prefix(&ipv6_hdr->dst);
390  memcpy(ipv6_hdr->dst.u8 + 8, iphc_hdr + payload_offset, 8);
391  payload_offset += 8;
392  break;
393 
394  case IPHC_M_DAC_DAM_U_16:
395  ipv6_addr_set_link_local_prefix(&ipv6_hdr->dst);
396  ipv6_hdr->dst.u32[2] = byteorder_htonl(0x000000ff);
397  ipv6_hdr->dst.u16[6] = byteorder_htons(0xfe00);
398  memcpy(ipv6_hdr->dst.u8 + 14, iphc_hdr + payload_offset, 2);
399  payload_offset += 2;
400  break;
401 
402  case IPHC_M_DAC_DAM_U_L2:
403  ieee802154_get_iid((eui64_t *)(&ipv6_hdr->dst.u64[1]),
404  gnrc_netif_hdr_get_dst_addr(netif_hdr),
405  netif_hdr->dst_l2addr_len);
406  ipv6_addr_set_link_local_prefix(&ipv6_hdr->dst);
407  break;
408 
409  case IPHC_M_DAC_DAM_U_CTX_64:
410  memcpy(ipv6_hdr->dst.u8 + 8, iphc_hdr + payload_offset, 8);
411  ipv6_addr_init_prefix(&ipv6_hdr->dst, &ctx->prefix,
412  ctx->prefix_len);
413  payload_offset += 8;
414  break;
415 
416  case IPHC_M_DAC_DAM_U_CTX_16:
417  ipv6_hdr->dst.u32[2] = byteorder_htonl(0x000000ff);
418  ipv6_hdr->dst.u16[6] = byteorder_htons(0xfe00);
419  memcpy(ipv6_hdr->dst.u8 + 14, iphc_hdr + payload_offset, 2);
420  ipv6_addr_init_prefix(&ipv6_hdr->dst, &ctx->prefix,
421  ctx->prefix_len);
422  payload_offset += 2;
423  break;
424 
425  case IPHC_M_DAC_DAM_U_CTX_L2:
426  ieee802154_get_iid((eui64_t *)(&ipv6_hdr->dst.u64[1]),
427  gnrc_netif_hdr_get_dst_addr(netif_hdr),
428  netif_hdr->dst_l2addr_len);
429  ipv6_addr_init_prefix(&ipv6_hdr->dst, &ctx->prefix,
430  ctx->prefix_len);
431  break;
432 
433  case IPHC_M_DAC_DAM_M_48:
434  /* ffXX::00XX:XXXX:XXXX */
435  ipv6_addr_set_unspecified(&ipv6_hdr->dst);
436  ipv6_hdr->dst.u8[0] = 0xff;
437  ipv6_hdr->dst.u8[1] = iphc_hdr[payload_offset++];
438  memcpy(ipv6_hdr->dst.u8 + 11, iphc_hdr + payload_offset, 5);
439  payload_offset += 5;
440  break;
441 
442  case IPHC_M_DAC_DAM_M_32:
443  /* ffXX::00XX:XXXX */
444  ipv6_addr_set_unspecified(&ipv6_hdr->dst);
445  ipv6_hdr->dst.u8[0] = 0xff;
446  ipv6_hdr->dst.u8[1] = iphc_hdr[payload_offset++];
447  memcpy(ipv6_hdr->dst.u8 + 13, iphc_hdr + payload_offset, 3);
448  payload_offset += 3;
449  break;
450 
451  case IPHC_M_DAC_DAM_M_8:
452  /* ff02::XX: */
453  ipv6_addr_set_unspecified(&ipv6_hdr->dst);
454  ipv6_hdr->dst.u8[0] = 0xff;
455  ipv6_hdr->dst.u8[1] = 0x02;
456  ipv6_hdr->dst.u8[15] = iphc_hdr[payload_offset++];
457  break;
458 
459  case IPHC_M_DAC_DAM_M_UC_PREFIX:
460  do {
461  uint8_t orig_ctx_len = ctx->prefix_len;
462 
463  ipv6_addr_set_unspecified(&ipv6_hdr->dst);
464 
465  if (ctx->prefix_len > 64) {
466  ctx->prefix_len = 64;
467  }
468 
469  ipv6_hdr->dst.u8[0] = 0xff;
470  ipv6_hdr->dst.u8[1] = iphc_hdr[payload_offset++];
471  ipv6_hdr->dst.u8[2] = iphc_hdr[payload_offset++];
472  ipv6_hdr->dst.u8[3] = ctx->prefix_len;
473  ipv6_addr_init_prefix((ipv6_addr_t *)ipv6_hdr->dst.u8 + 4,
474  &ctx->prefix, ctx->prefix_len);
475  memcpy(ipv6_hdr->dst.u8 + 12, iphc_hdr + payload_offset + 2, 4);
476 
477  payload_offset += 4;
478  ctx->prefix_len = orig_ctx_len;
479  } while (0); /* ANSI-C compatible block creation for orig_ctx_len allocation */
480  break;
481 
482  default:
483  DEBUG("6lo iphc: unspecified or reserved M, DAC, DAM combination\n");
484  return 0;
485 
486  }
487 
488  /* set IPv6 header payload length field to the length of whatever is left
489  * after removing the 6LoWPAN header */
490  if (datagram_size == 0) {
491  ipv6_hdr->len = byteorder_htons((uint16_t)(pkt->size - payload_offset));
492  }
493  else {
494  ipv6_hdr->len = byteorder_htons((uint16_t)(datagram_size - sizeof(ipv6_hdr_t)));
495  }
496 
497 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC_NHC
498  if (iphc_hdr[IPHC1_IDX] & SIXLOWPAN_IPHC1_NH) {
499  switch (iphc_hdr[payload_offset] & NHC_ID_MASK) {
500  case NHC_UDP_ID:
501  payload_offset = iphc_nhc_udp_decode(pkt, dec_hdr, datagram_size,
502  payload_offset + offset);
503 
504  if (payload_offset != 0) {
505  payload_offset -= offset;
506  }
507 
508  *nh_len += sizeof(udp_hdr_t);
509  break;
510 
511  default:
512  break;
513  }
514  }
515 #else
516  (void)nh_len;
517 #endif
518 
519  return payload_offset;
520 }

523 inline static size_t iphc_nhc_udp_encode(gnrc_pktsnip_t *udp, ipv6_hdr_t *ipv6_hdr)
524 {
525  udp_hdr_t *udp_hdr = udp->data;
526  network_uint16_t *src_port = &(udp_hdr->src_port);
527  network_uint16_t *dst_port = &(udp_hdr->dst_port);
528  uint8_t *udp_data = udp->data;
529  size_t nhc_len = 0;
530 
531  /* TODO: Add support for elided checksum. */
532 
533  /* Compressing UDP ports, follow the same sequence as the linux kernel (nhc_udp module). */
534  if (((byteorder_ntohs(*src_port) & NHC_UDP_4BIT_MASK) == NHC_UDP_4BIT_PORT) &&
535  ((byteorder_ntohs(*dst_port) & NHC_UDP_4BIT_MASK) == NHC_UDP_4BIT_PORT)) {
536  DEBUG("6lo iphc nhc: elide src and dst\n");
537  ipv6_hdr->nh = NHC_UDP_SD_ELIDED;
538  udp_data[nhc_len++] = byteorder_ntohs(*dst_port) - NHC_UDP_4BIT_PORT +
539  ((byteorder_ntohs(*src_port) - NHC_UDP_4BIT_PORT) << 4);
540  udp_data[nhc_len++] = udp_hdr->checksum.u8[0];
541  udp_data[nhc_len++] = udp_hdr->checksum.u8[1];
542  }
543  else if ((byteorder_ntohs(*dst_port) & NHC_UDP_8BIT_MASK) == NHC_UDP_8BIT_PORT) {
544  DEBUG("6lo iphc nhc: elide dst\n");
545  ipv6_hdr->nh = NHC_UDP_S_INLINE;
546  nhc_len += 2; /* keep src_port */
547  udp_data[nhc_len++] = byteorder_ntohs(*dst_port) - NHC_UDP_8BIT_PORT;
548  udp_data[nhc_len++] = udp_hdr->checksum.u8[0];
549  udp_data[nhc_len++] = udp_hdr->checksum.u8[1];
550  }
551  else if ((byteorder_ntohs(*src_port) & NHC_UDP_8BIT_MASK) == NHC_UDP_8BIT_PORT) {
552  DEBUG("6lo iphc nhc: elide src\n");
553  ipv6_hdr->nh = NHC_UDP_D_INLINE;
554  udp_data[nhc_len++] = byteorder_ntohs(*src_port) - NHC_UDP_8BIT_PORT;
555  udp_data[nhc_len++] = udp_hdr->dst_port.u8[0];
556  udp_data[nhc_len++] = udp_hdr->dst_port.u8[1];
557  udp_data[nhc_len++] = udp_hdr->checksum.u8[0];
558  udp_data[nhc_len++] = udp_hdr->checksum.u8[1];
559  }
560  else {
561  DEBUG("6lo iphc nhc: src and dst inline\n");
562  ipv6_hdr->nh = NHC_UDP_SD_INLINE;
563  nhc_len = sizeof(udp_hdr_t) - 4; /* skip src + dst and elide length */
564  udp_data[nhc_len++] = udp_hdr->checksum.u8[0];
565  udp_data[nhc_len++] = udp_hdr->checksum.u8[1];
566  }
567 
568  /* Set UDP header ID (rfc6282#section-5). */
569  ipv6_hdr->nh |= NHC_UDP_ID;
570 
571  /* In case payload is in this snip (e.g. a forwarded packet):
572  * move data to right place */
573  size_t diff = sizeof(udp_hdr_t) - nhc_len;
574  for (size_t i = nhc_len; i < (udp->size - diff); i++) {
575  udp_data[i] = udp_data[i + diff];
576  }
577  /* NOTE: gnrc_pktbuf_realloc_data overflow if (udp->size - diff) < 4 */
578  gnrc_pktbuf_realloc_data(udp, (udp->size - diff));
579 
580  return nhc_len;
581 }
582 #endif

584 bool gnrc_sixlowpan_iphc_encode(gnrc_pktsnip_t *pkt)
585 {
586  gnrc_netif_hdr_t *netif_hdr = pkt->data;
587  ipv6_hdr_t *ipv6_hdr = pkt->next->data;
588  uint8_t *iphc_hdr;
589  uint16_t inline_pos = SIXLOWPAN_IPHC_HDR_LEN;
590  bool addr_comp = false, nhc_comp = false;
591  gnrc_sixlowpan_ctx_t *src_ctx = NULL, *dst_ctx = NULL;
592  gnrc_pktsnip_t *dispatch = gnrc_pktbuf_add(NULL, NULL, pkt->next->size,
593  GNRC_NETTYPE_SIXLOWPAN);
594 
595  if (dispatch == NULL) {
596  DEBUG("6lo iphc: error allocating dispatch space\n");
597  return false;
598  }
599 
600  iphc_hdr = dispatch->data;
601 
602  /* set initial dispatch value*/
603  iphc_hdr[IPHC1_IDX] = SIXLOWPAN_IPHC1_DISP;
604  iphc_hdr[IPHC2_IDX] = 0;
605 
606  /* check for available contexts */
607  if (!ipv6_addr_is_unspecified(&(ipv6_hdr->src))) {
608  src_ctx = gnrc_sixlowpan_ctx_lookup_addr(&(ipv6_hdr->src));
609  /* do not use source context for compression if */
610  /* GNRC_SIXLOWPAN_CTX_FLAGS_COMP is not set */
611  if (src_ctx && !(src_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_COMP)) {
612  src_ctx = NULL;
613  }
614  }
615 
616  if (!ipv6_addr_is_multicast(&ipv6_hdr->dst)) {
617  dst_ctx = gnrc_sixlowpan_ctx_lookup_addr(&(ipv6_hdr->dst));
618  /* do not use destination context for compression if */
619  /* GNRC_SIXLOWPAN_CTX_FLAGS_COMP is not set */
620  if (dst_ctx && !(dst_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_COMP)) {
621  dst_ctx = NULL;
622  }
623  }
624 
625  /* if contexts available and both != 0 */
626  /* since this moves inline_pos we have to do this ahead*/
627  if (((src_ctx != NULL) &&
628  ((src_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK) != 0)) ||
629  ((dst_ctx != NULL) &&
630  ((dst_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK) != 0))) {
631  /* add context identifier extension */
632  iphc_hdr[IPHC2_IDX] |= SIXLOWPAN_IPHC2_CID_EXT;
633  iphc_hdr[CID_EXT_IDX] = 0;
634 
635  /* move position to behind CID extension */
636  inline_pos += SIXLOWPAN_IPHC_CID_EXT_LEN;
637  }
638 
639  /* compress flow label and traffic class */
640  if (ipv6_hdr_get_fl(ipv6_hdr) == 0) {
641  if (ipv6_hdr_get_tc(ipv6_hdr) == 0) {
642  /* elide both traffic class and flow label */
643  iphc_hdr[IPHC1_IDX] |= IPHC_TF_ECN_ELIDE;
644  }
645  else {
646  /* elide flow label, traffic class (ECN + DSCP) inline (1 byte) */
647  iphc_hdr[IPHC1_IDX] |= IPHC_TF_ECN_DSCP;
648  iphc_hdr[inline_pos++] = ipv6_hdr_get_tc(ipv6_hdr);
649  }
650  }
651  else {
652  if (ipv6_hdr_get_tc_dscp(ipv6_hdr) == 0) {
653  /* elide DSCP, ECN + 2-bit pad + flow label inline (3 byte) */
654  iphc_hdr[IPHC1_IDX] |= IPHC_TF_ECN_FL;
655  iphc_hdr[inline_pos++] = (uint8_t)((ipv6_hdr_get_tc_ecn(ipv6_hdr) << 6) |
656  ((ipv6_hdr_get_fl(ipv6_hdr) & 0x000f0000) >> 16));
657  }
658  else {
659  /* ECN + DSCP + 4-bit pad + flow label (4 bytes) */
660  iphc_hdr[IPHC1_IDX] |= IPHC_TF_ECN_DSCP_FL;
661  iphc_hdr[inline_pos++] = ipv6_hdr_get_tc(ipv6_hdr);
662  iphc_hdr[inline_pos++] = (uint8_t)((ipv6_hdr_get_fl(ipv6_hdr) & 0x000f0000) >> 16);
663  }
664 
665  /* copy remaining byteos of flow label */
666  iphc_hdr[inline_pos++] = (uint8_t)((ipv6_hdr_get_fl(ipv6_hdr) & 0x0000ff00) >> 8);
667  iphc_hdr[inline_pos++] = (uint8_t)((ipv6_hdr_get_fl(ipv6_hdr) & 0x000000ff) >> 8);
668  }
669 
670  /* compress next header */
671  switch (ipv6_hdr->nh) {
672 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC_NHC
673  case PROTNUM_UDP:
674  iphc_nhc_udp_encode(pkt->next->next, ipv6_hdr);
675  iphc_hdr[IPHC1_IDX] |= SIXLOWPAN_IPHC1_NH;
676  nhc_comp = true;
677  break;
678 #endif
679 
680  default:
681  iphc_hdr[inline_pos++] = ipv6_hdr->nh;
682  break;
683  }
684 
685  /* compress hop limit */
686  switch (ipv6_hdr->hl) {
687  case 1:
688  iphc_hdr[IPHC1_IDX] |= IPHC_HL_1;
689  break;
690 
691  case 64:
692  iphc_hdr[IPHC1_IDX] |= IPHC_HL_64;
693  break;
694 
695  case 255:
696  iphc_hdr[IPHC1_IDX] |= IPHC_HL_255;
697  break;
698 
699  default:
700  iphc_hdr[IPHC1_IDX] |= IPHC_HL_INLINE;
701  iphc_hdr[inline_pos++] = ipv6_hdr->hl;
702  break;
703  }
704 
705  if (ipv6_addr_is_unspecified(&(ipv6_hdr->src))) {
706  iphc_hdr[IPHC2_IDX] |= IPHC_SAC_SAM_UNSPEC;
707  }
708  else {
709  if (src_ctx != NULL) {
710  /* stateful source address compression */
711  iphc_hdr[IPHC2_IDX] |= SIXLOWPAN_IPHC2_SAC;
712 
713  if (((src_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK) != 0)) {
714  iphc_hdr[CID_EXT_IDX] |= ((src_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK) << 4);
715  }
716  }
717 
718  if ((src_ctx != NULL) || ipv6_addr_is_link_local(&(ipv6_hdr->src))) {
719  eui64_t iid;
720  iid.uint64.u64 = 0;
721 
722  if ((netif_hdr->src_l2addr_len == 2) ||
723  (netif_hdr->src_l2addr_len == 4) ||
724  (netif_hdr->src_l2addr_len == 8)) {
725  /* prefer to create IID from netif header if available */
726  ieee802154_get_iid(&iid, gnrc_netif_hdr_get_src_addr(netif_hdr),
727  netif_hdr->src_l2addr_len);
728  }
729  else {
730  /* but take from driver otherwise */
731  gnrc_netapi_get(netif_hdr->if_pid, NETOPT_IPV6_IID, 0, &iid,
732  sizeof(eui64_t));
733  }
734 
735  if ((ipv6_hdr->src.u64[1].u64 == iid.uint64.u64) ||
736  _context_overlaps_iid(src_ctx, &ipv6_hdr->src, &iid)) {
737  /* 0 bits. The address is derived from link-layer address */
738  iphc_hdr[IPHC2_IDX] |= IPHC_SAC_SAM_L2;
739  addr_comp = true;
740  }
741  else if ((byteorder_ntohl(ipv6_hdr->src.u32[2]) == 0x000000ff) &&
742  (byteorder_ntohs(ipv6_hdr->src.u16[6]) == 0xfe00)) {
743  /* 16 bits. The address is derived using 16 bits carried inline */
744  iphc_hdr[IPHC2_IDX] |= IPHC_SAC_SAM_16;
745  memcpy(iphc_hdr + inline_pos, ipv6_hdr->src.u16 + 7, 2);
746  inline_pos += 2;
747  addr_comp = true;
748  }
749  else {
750  /* 64 bits. The address is derived using 64 bits carried inline */
751  iphc_hdr[IPHC2_IDX] |= IPHC_SAC_SAM_64;
752  memcpy(iphc_hdr + inline_pos, ipv6_hdr->src.u64 + 1, 8);
753  inline_pos += 8;
754  addr_comp = true;
755  }
756  }
757 
758  if (!addr_comp) {
759  /* full address is carried inline */
760  iphc_hdr[IPHC2_IDX] |= IPHC_SAC_SAM_FULL;
761  memcpy(iphc_hdr + inline_pos, &ipv6_hdr->src, 16);
762  inline_pos += 16;
763  }
764  }
765 
766  addr_comp = false;
767 
768  /* M: Multicast compression */
769  if (ipv6_addr_is_multicast(&(ipv6_hdr->dst))) {
770  iphc_hdr[IPHC2_IDX] |= SIXLOWPAN_IPHC2_M;
771 
772  /* if multicast address is of format ffXX::XXXX:XXXX:XXXX */
773  if ((ipv6_hdr->dst.u16[1].u16 == 0) &&
774  (ipv6_hdr->dst.u32[1].u32 == 0) &&
775  (ipv6_hdr->dst.u16[4].u16 == 0)) {
776  /* if multicast address is of format ff02::XX */
777  if ((ipv6_hdr->dst.u8[1] == 0x02) &&
778  (ipv6_hdr->dst.u32[2].u32 == 0) &&
779  (ipv6_hdr->dst.u16[6].u16 == 0) &&
780  (ipv6_hdr->dst.u8[14] == 0)) {
781  /* 8 bits. The address is derived using 8 bits carried inline */
782  iphc_hdr[IPHC2_IDX] |= IPHC_M_DAC_DAM_M_8;
783  iphc_hdr[inline_pos++] = ipv6_hdr->dst.u8[15];
784  addr_comp = true;
785  }
786  /* if multicast address is of format ffXX::XX:XXXX */
787  else if ((ipv6_hdr->dst.u16[5].u16 == 0) &&
788  (ipv6_hdr->dst.u8[12] == 0)) {
789  /* 32 bits. The address is derived using 32 bits carried inline */
790  iphc_hdr[IPHC2_IDX] |= IPHC_M_DAC_DAM_M_32;
791  iphc_hdr[inline_pos++] = ipv6_hdr->dst.u8[1];
792  memcpy(iphc_hdr + inline_pos, ipv6_hdr->dst.u8 + 13, 3);
793  inline_pos += 3;
794  addr_comp = true;
795  }
796  /* if multicast address is of format ffXX::XX:XXXX:XXXX */
797  else if (ipv6_hdr->dst.u8[10] == 0) {
798  /* 48 bits. The address is derived using 48 bits carried inline */
799  iphc_hdr[IPHC2_IDX] |= IPHC_M_DAC_DAM_M_48;
800  iphc_hdr[inline_pos++] = ipv6_hdr->dst.u8[1];
801  memcpy(iphc_hdr + inline_pos, ipv6_hdr->dst.u8 + 11, 5);
802  inline_pos += 5;
803  addr_comp = true;
804  }
805  }
806  /* try unicast prefix based compression */
807  else {
808  gnrc_sixlowpan_ctx_t *ctx;
809  ipv6_addr_t unicast_prefix;
810  unicast_prefix.u16[0] = ipv6_hdr->dst.u16[2];
811  unicast_prefix.u16[1] = ipv6_hdr->dst.u16[3];
812  unicast_prefix.u16[2] = ipv6_hdr->dst.u16[4];
813  unicast_prefix.u16[3] = ipv6_hdr->dst.u16[5];
814 
815  ctx = gnrc_sixlowpan_ctx_lookup_addr(&unicast_prefix);
816 
817  if ((ctx != NULL) && (ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_COMP) &&
818  (ctx->prefix_len == ipv6_hdr->dst.u8[3])) {
819  /* Unicast prefix based IPv6 multicast address
820  * (https://tools.ietf.org/html/rfc3306) with given context
821  * for unicast prefix -> context based compression */
822  iphc_hdr[IPHC2_IDX] |= SIXLOWPAN_IPHC2_DAC;
823  if ((ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK) != 0) {
824  iphc_hdr[CID_EXT_IDX] |= (ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK);
825  }
826  iphc_hdr[inline_pos++] = ipv6_hdr->dst.u8[1];
827  iphc_hdr[inline_pos++] = ipv6_hdr->dst.u8[2];
828  memcpy(iphc_hdr + inline_pos, ipv6_hdr->dst.u16 + 6, 4);
829  inline_pos += 4;
830  addr_comp = true;
831  }
832  }
833  }
834  else if (((dst_ctx != NULL) ||
835  ipv6_addr_is_link_local(&ipv6_hdr->dst)) && (netif_hdr->dst_l2addr_len > 0)) {
836  eui64_t iid;
837 
838  if (dst_ctx != NULL) {
839  /* stateful destination address compression */
840  iphc_hdr[IPHC2_IDX] |= SIXLOWPAN_IPHC2_DAC;
841 
842  if (((dst_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK) != 0)) {
843  iphc_hdr[CID_EXT_IDX] |= (dst_ctx->flags_id & GNRC_SIXLOWPAN_CTX_FLAGS_CID_MASK);
844  }
845  }
846 
847  ieee802154_get_iid(&iid, gnrc_netif_hdr_get_dst_addr(netif_hdr),
848  netif_hdr->dst_l2addr_len);
849 
850  if ((ipv6_hdr->dst.u64[1].u64 == iid.uint64.u64) ||
851  _context_overlaps_iid(dst_ctx, &(ipv6_hdr->dst), &iid)) {
852  /* 0 bits. The address is derived using the link-layer address */
853  iphc_hdr[IPHC2_IDX] |= IPHC_M_DAC_DAM_U_L2;
854  addr_comp = true;
855  }
856  else if ((byteorder_ntohl(ipv6_hdr->dst.u32[2]) == 0x000000ff) &&
857  (byteorder_ntohs(ipv6_hdr->dst.u16[6]) == 0xfe00)) {
858  /* 16 bits. The address is derived using 16 bits carried inline */
859  iphc_hdr[IPHC2_IDX] |= IPHC_M_DAC_DAM_U_16;
860  memcpy(&(iphc_hdr[inline_pos]), &(ipv6_hdr->dst.u16[7]), 2);
861  inline_pos += 2;
862  addr_comp = true;
863  }
864  else {
865  /* 64 bits. The address is derived using 64 bits carried inline */
866  iphc_hdr[IPHC2_IDX] |= IPHC_M_DAC_DAM_U_64;
867  memcpy(&(iphc_hdr[inline_pos]), &(ipv6_hdr->dst.u8[8]), 8);
868  inline_pos += 8;
869  addr_comp = true;
870  }
871  }
872 
873  if (!addr_comp) {
874  /* full destination address is carried inline */
875  iphc_hdr[IPHC2_IDX] |= IPHC_SAC_SAM_FULL;
876  memcpy(iphc_hdr + inline_pos, &ipv6_hdr->dst, 16);
877  inline_pos += 16;
878  }
879 
880  if (nhc_comp) {
881  iphc_hdr[inline_pos++] = ipv6_hdr->nh;
882  }
883 
884  /* shrink dispatch allocation to final size */
885  /* NOTE: Since this only shrinks the data nothing bad SHOULD happen ;-) */
886  gnrc_pktbuf_realloc_data(dispatch, (size_t)inline_pos);
887 
888  /* remove IPv6 header */
889  pkt = gnrc_pktbuf_remove_snip(pkt, pkt->next);
890 
891  /* insert dispatch into packet */
892  dispatch->next = pkt->next;
893  pkt->next = dispatch;
894 
895  return true;
896 }
