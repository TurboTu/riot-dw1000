40 static gnrc_pktsnip_t *_make_netif_hdr(uint8_t *mhr)
41 {
42  gnrc_pktsnip_t *snip;
43  uint8_t src[IEEE802154_LONG_ADDRESS_LEN], dst[IEEE802154_LONG_ADDRESS_LEN];
44  int src_len, dst_len;
45  le_uint16_t _pan_tmp; /* TODO: hand-up PAN IDs to GNRC? */
46 
47  dst_len = ieee802154_get_dst(mhr, dst, &_pan_tmp);
48  src_len = ieee802154_get_src(mhr, src, &_pan_tmp);
49  if ((dst_len < 0) || (src_len < 0)) {
50  DEBUG("_make_netif_hdr: unable to get addresses\n");
51  return NULL;
52  }
53  /* allocate space for header */
54  snip = gnrc_netif_hdr_build(src, (size_t)src_len, dst, (size_t)dst_len);
55  if (snip == NULL) {
56  DEBUG("_make_netif_hdr: no space left in packet buffer\n");
57  return NULL;
58  }
59  /* set broadcast flag for broadcast destination */
60  if ((dst_len == 2) && (dst[0] == 0xff) && (dst[1] == 0xff)) {
61  gnrc_netif_hdr_t *hdr = snip->data;
62  hdr->flags |= GNRC_NETIF_HDR_FLAGS_BROADCAST;
63  }
64  return snip;
65 }

67 static gnrc_pktsnip_t *_recv(gnrc_netdev2_t *gnrc_netdev2)
68 {
69  netdev2_t *netdev = gnrc_netdev2->dev;
70  netdev2_ieee802154_rx_info_t rx_info;
71  netdev2_ieee802154_t *state = (netdev2_ieee802154_t *)gnrc_netdev2->dev;
72  gnrc_pktsnip_t *pkt = NULL;
73  int bytes_expected = netdev->driver->recv(netdev, NULL, 0, NULL);
74 
75  if (bytes_expected > 0) {
76  int nread;
77 
78  pkt = gnrc_pktbuf_add(NULL, NULL, bytes_expected, GNRC_NETTYPE_UNDEF);
79  if (pkt == NULL) {
80  DEBUG("_recv_ieee802154: cannot allocate pktsnip.\n");
81  return NULL;
82  }
83  nread = netdev->driver->recv(netdev, pkt->data, bytes_expected, &rx_info);
84  if (nread <= 0) {
85  gnrc_pktbuf_release(pkt);
86  return NULL;
87  }
88  if (!(state->flags & NETDEV2_IEEE802154_RAW)) {
89  gnrc_pktsnip_t *ieee802154_hdr, *netif_hdr;
90  gnrc_netif_hdr_t *hdr;
91 #if ENABLE_DEBUG
92  char src_str[GNRC_NETIF_HDR_L2ADDR_PRINT_LEN];
93 #endif
94  size_t mhr_len = ieee802154_get_frame_hdr_len(pkt->data);
95 
96  if (mhr_len == 0) {
97  DEBUG("_recv_ieee802154: illegally formatted frame received\n");
98  gnrc_pktbuf_release(pkt);
99  return NULL;
100  }
101  nread -= mhr_len;
102  /* mark IEEE 802.15.4 header */
103  ieee802154_hdr = gnrc_pktbuf_mark(pkt, mhr_len, GNRC_NETTYPE_UNDEF);
104  if (ieee802154_hdr == NULL) {
105  DEBUG("_recv_ieee802154: no space left in packet buffer\n");
106  gnrc_pktbuf_release(pkt);
107  return NULL;
108  }
109  netif_hdr = _make_netif_hdr(ieee802154_hdr->data);
110  if (netif_hdr == NULL) {
111  DEBUG("_recv_ieee802154: no space left in packet buffer\n");
112  gnrc_pktbuf_release(pkt);
113  return NULL;
114  }
115  hdr = netif_hdr->data;
116  hdr->lqi = rx_info.lqi;
117  hdr->rssi = rx_info.rssi;
118  hdr->if_pid = thread_getpid();
119  pkt->type = state->proto;
120 #if ENABLE_DEBUG
121  DEBUG("_recv_ieee802154: received packet from %s of length %u\n",
122  gnrc_netif_addr_to_str(src_str, sizeof(src_str),
123  gnrc_netif_hdr_get_src_addr(hdr),
124  hdr->src_l2addr_len),
125  nread);
126 #if defined(MODULE_OD)
127  od_hex_dump(pkt->data, nread, OD_WIDTH_DEFAULT);
128 #endif
129 #endif
130  gnrc_pktbuf_remove_snip(pkt, ieee802154_hdr);
131  LL_APPEND(pkt, netif_hdr);
132  }
133 
134  DEBUG("_recv_ieee802154: reallocating.\n");
135  gnrc_pktbuf_realloc_data(pkt, nread);
136  }
137 
138  return pkt;
139 }

141 static int _send(gnrc_netdev2_t *gnrc_netdev2, gnrc_pktsnip_t *pkt)
142 {
143  netdev2_t *netdev = gnrc_netdev2->dev;
144  netdev2_ieee802154_t *state = (netdev2_ieee802154_t *)gnrc_netdev2->dev;
145  gnrc_netif_hdr_t *netif_hdr;
146  gnrc_pktsnip_t *vec_snip;
147  const uint8_t *src, *dst = NULL;
148  int res = 0;
149  size_t n, src_len, dst_len;
150  uint8_t mhr[IEEE802154_MAX_HDR_LEN];
151  uint8_t flags = (uint8_t)(state->flags & NETDEV2_IEEE802154_SEND_MASK);
152  le_uint16_t dev_pan = byteorder_btols(byteorder_htons(state->pan));
153 
154  flags |= IEEE802154_FCF_TYPE_DATA;
155  if (pkt == NULL) {
156  DEBUG("_send_ieee802154: pkt was NULL\n");
157  return -EINVAL;
158  }
159  if (pkt->type != GNRC_NETTYPE_NETIF) {
160  DEBUG("_send_ieee802154: first header is not generic netif header\n");
161  return -EBADMSG;
162  }
163  netif_hdr = pkt->data;
164  /* prepare destination address */
165  if (netif_hdr->flags & /* If any of these flags is set so this is correct */
166  (GNRC_NETIF_HDR_FLAGS_BROADCAST | GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
167  dst = ieee802154_addr_bcast;
168  dst_len = IEEE802154_ADDR_BCAST_LEN;
169  }
170  else {
171  dst = gnrc_netif_hdr_get_dst_addr(netif_hdr);
172  dst_len = netif_hdr->dst_l2addr_len;
173  }
174  src_len = netif_hdr->src_l2addr_len;
175  if (src_len > 0) {
176  src = gnrc_netif_hdr_get_src_addr(netif_hdr);
177  }
178  else if (state->flags & NETDEV2_IEEE802154_SRC_MODE_LONG) {
179  src_len = IEEE802154_LONG_ADDRESS_LEN;
180  src = state->long_addr;
181  }
182  else {
183  src_len = IEEE802154_SHORT_ADDRESS_LEN;
184  src = state->short_addr;
185  }
186  /* fill MAC header, seq should be set by device */
187  if ((res = ieee802154_set_frame_hdr(mhr, src, src_len,
188  dst, dst_len, dev_pan,
189  dev_pan, flags, state->seq++)) == 0) {
190  DEBUG("_send_ieee802154: Error preperaring frame\n");
191  return -EINVAL;
192  }
193  /* prepare packet for sending */
194  vec_snip = gnrc_pktbuf_get_iovec(pkt, &n);
195  if (vec_snip != NULL) {
196  struct iovec *vector;
197 
198  pkt = vec_snip; /* reassign for later release; vec_snip is prepended to pkt */
199  vector = (struct iovec *)pkt->data;
200  vector[0].iov_base = mhr;
201  vector[0].iov_len = (size_t)res;
202 #ifdef MODULE_NETSTATS_L2
203  if (netif_hdr->flags &
204  (GNRC_NETIF_HDR_FLAGS_BROADCAST | GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
205  gnrc_netdev2->dev->stats.tx_mcast_count++;
206  }
207  else {
208  gnrc_netdev2->dev->stats.tx_unicast_count++;
209  }
210 #endif
211  res = netdev->driver->send(netdev, vector, n);
212  }
213  else {
214  return -ENOBUFS;
215  }
216  /* release old data */
217  gnrc_pktbuf_release(pkt);
218  return res;
219 }
