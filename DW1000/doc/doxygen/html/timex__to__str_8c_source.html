21 static unsigned int u32_to_str(char *str, uint32_t u32)
22 {
23  int len;
24  int i = 0;
25  if (u32 == 0) {
26  str[i] = '0';
27  return 1;
28  }
29  for (; u32 > 0; u32 /= RADIX) {
30  char digit = (char)(u32 % RADIX);
31  str[i++] = '0' + digit;
32  }
33  len = i;
34  i--; /* go back to last character */
35  for (int j = 0; j < i; j++, i--) {
36  char tmp = str[i];
37  str[i] = str[j];
38  str[j] = tmp;
39  }
40  return len;
41 }
