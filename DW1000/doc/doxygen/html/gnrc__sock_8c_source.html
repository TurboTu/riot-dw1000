54 ssize_t gnrc_sock_recv(gnrc_sock_reg_t *reg, gnrc_pktsnip_t **pkt_out,
55  uint32_t timeout, sock_ip_ep_t *remote)
56 {
57  gnrc_pktsnip_t *pkt, *ip, *netif;
58  msg_t msg;
59 
60 #ifdef MODULE_XTIMER
61  xtimer_t timeout_timer;
62 
63  if ((timeout != SOCK_NO_TIMEOUT) && (timeout != 0)) {
64  timeout_timer.callback = _callback_put;
65  timeout_timer.arg = reg;
66  xtimer_set(&timeout_timer, timeout);
67  }
68 #endif
69  if (timeout != 0) {
70  mbox_get(&reg->mbox, &msg);
71  }
72  else {
73  if (!mbox_try_get(&reg->mbox, &msg)) {
74  return -EAGAIN;
75  }
76  }
77 #ifdef MODULE_XTIMER
78  xtimer_remove(&timeout_timer);
79 #endif
80  switch (msg.type) {
81  case GNRC_NETAPI_MSG_TYPE_RCV:
82  pkt = msg.content.ptr;
83  break;
84 #ifdef MODULE_XTIMER
85  case _TIMEOUT_MSG_TYPE:
86  if (msg.content.value == _TIMEOUT_MAGIC) {
87  return -ETIMEDOUT;
88  }
89 #endif
90  default:
91  return -EINTR;
92  }
93  /* TODO: discern NETTYPE from remote->family (set in caller), when IPv4
94  * was implemented */
95  ipv6_hdr_t *ipv6_hdr;
96  ip = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_IPV6);
97  assert((ip != NULL) && (ip->size >= 40));
98  ipv6_hdr = ip->data;
99  memcpy(&remote->addr, &ipv6_hdr->src, sizeof(ipv6_addr_t));
100  remote->family = AF_INET6;
101  netif = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_NETIF);
102  if (netif == NULL) {
103  remote->netif = SOCK_ADDR_ANY_NETIF;
104  }
105  else {
106  gnrc_netif_hdr_t *netif_hdr = netif->data;
107  /* TODO: use API in #5511 */
108  remote->netif = (uint16_t)netif_hdr->if_pid;
109  }
110  *pkt_out = pkt; /* set out parameter */
111  return 0;
112 }

114 ssize_t gnrc_sock_send(gnrc_pktsnip_t *payload, sock_ip_ep_t *local,
115  const sock_ip_ep_t *remote, uint8_t nh)
116 {
117  gnrc_pktsnip_t *pkt;
118  kernel_pid_t iface = KERNEL_PID_UNDEF;
119  gnrc_nettype_t type;
120  size_t payload_len = gnrc_pkt_len(payload);
121 
122  if (local->family != remote->family) {
123  gnrc_pktbuf_release(payload);
124  return -EAFNOSUPPORT;
125  }
126  switch (local->family) {
127 #ifdef SOCK_HAS_IPV6
128  case AF_INET6: {
129  ipv6_hdr_t *hdr;
130  pkt = gnrc_ipv6_hdr_build(payload, (ipv6_addr_t *)&local->addr.ipv6,
131  (ipv6_addr_t *)&remote->addr.ipv6);
132  if (pkt == NULL) {
133  return -ENOMEM;
134  }
135  if (payload->type == GNRC_NETTYPE_UNDEF) {
136  payload->type = GNRC_NETTYPE_IPV6;
137  type = GNRC_NETTYPE_IPV6;
138  }
139  else {
140  type = payload->type;
141  }
142  hdr = pkt->data;
143  hdr->nh = nh;
144  break;
145  }
146 #endif
147  default:
148  (void)nh;
149  gnrc_pktbuf_release(payload);
150  return -EAFNOSUPPORT;
151  }
152  if (local->netif != SOCK_ADDR_ANY_NETIF) {
153  /* TODO: use API in #5511 */
154  iface = (kernel_pid_t)local->netif;
155  }
156  else if (remote->netif != SOCK_ADDR_ANY_NETIF) {
157  /* TODO: use API in #5511 */
158  iface = (kernel_pid_t)remote->netif;
159  }
160  if (iface != KERNEL_PID_UNDEF) {
161  gnrc_pktsnip_t *netif = gnrc_netif_hdr_build(NULL, 0, NULL, 0);
162  gnrc_netif_hdr_t *netif_hdr;
163 
164  if (netif == NULL) {
165  gnrc_pktbuf_release(pkt);
166  return -ENOMEM;
167  }
168  netif_hdr = netif->data;
169  netif_hdr->if_pid = iface;
170  LL_PREPEND(pkt, netif);
171  }
172 #ifdef MODULE_GNRC_NETERR
173  gnrc_neterr_reg(pkt); /* no error should occur since pkt was created here */
174 #endif
175  if (!gnrc_netapi_dispatch_send(type, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
176  /* this should not happen, but just in case */
177  gnrc_pktbuf_release(pkt);
178  return -EBADMSG;
179  }
180 #ifdef MODULE_GNRC_NETERR
181  msg_t err_report;
182  err_report.type = 0;
183 
184  while (err_report.type != GNRC_NETERR_MSG_TYPE) {
185  msg_try_receive(err_report);
186  if (err_report.type != GNRC_NETERR_MSG_TYPE) {
187  msg_try_send(err_report, sched_active_pid);
188  }
189  }
190  if (err_report.content.value != GNRC_NETERR_SUCCESS) {
191  return (int)(-err_report.content.value);
192  }
193 #endif
194  return payload_len;
195 }
