64 static void test_encrypt_op(uint8_t* key, uint8_t key_len, uint8_t ctr[16],
65  uint8_t* input, uint8_t input_len, uint8_t* output,
66  uint8_t output_len)
67 {

69  int len, err, cmp;
70  uint8_t data[64];

73  TEST_ASSERT_EQUAL_INT(1, err);
74 
75  len = cipher_encrypt_ctr(&cipher, ctr, 0, input, input_len, data);

77 
78  TEST_ASSERT_EQUAL_INT(output_len, len);
