63 err_t lwip_netdev2_init(struct netif *netif)
64 {
65  LWIP_ASSERT("netif != NULL", (netif != NULL));
66  LWIP_ASSERT("netif->state != NULL", (netif->state != NULL));
67  netdev2_t *netdev;
68  uint16_t dev_type;
69  err_t res = ERR_OK;
70 
71  /* start multiplexing thread (only one needed) */
72  if (_pid <= KERNEL_PID_UNDEF) {
73  _pid = thread_create(_stack, LWIP_NETDEV2_STACKSIZE, LWIP_NETDEV2_PRIO,
74  THREAD_CREATE_STACKTEST, _event_loop, netif,
75  LWIP_NETDEV2_NAME);
76  if (_pid <= 0) {
77  return ERR_IF;
78  }
79  }
80 
81  /* initialize netdev and netif */
82  netdev = (netdev2_t *)netif->state;
83  netdev->driver->init(netdev);
84  netdev->event_callback = _event_cb;
85  if (netdev->driver->get(netdev, NETOPT_DEVICE_TYPE, &dev_type,
86  sizeof(dev_type)) < 0) {
87  return ERR_IF;
88  }
89 #if LWIP_NETIF_HOSTNAME
90  netif->hostname = "riot";
91 #endif /* LWIP_NETIF_HOSTNAME */
92 
93  /* XXX: for now assume its Ethernet, since netdev2 is implemented only by ethernet drivers */
94  switch (dev_type) {
95 #ifdef MODULE_NETDEV2_ETH
96  case NETDEV2_TYPE_ETHERNET:
97  netif->name[0] = ETHERNET_IFNAME1;
98  netif->name[1] = ETHERNET_IFNAME2;
99  netif->hwaddr_len = (u8_t)netdev->driver->get(netdev, NETOPT_ADDRESS, netif->hwaddr,
100  sizeof(netif->hwaddr));
101  if (netif->hwaddr_len > sizeof(netif->hwaddr)) {
102  return ERR_IF;
103  }
104  /* TODO: get from driver (currently not in netdev2_eth) */
105  netif->mtu = ETHERNET_DATA_LEN;
106  netif->linkoutput = _eth_link_output;
107 #if LWIP_IPV4
108  netif->output = etharp_output;
109 #endif
110 #if LWIP_IPV6
111  netif->output_ip6 = ethip6_output;
112  netif_create_ip6_linklocal_address(netif, 1); /* 1: hwaddr is 48-bit MAC addr */
113 #endif
114  netif->flags |= NETIF_FLAG_BROADCAST;
115  netif->flags |= NETIF_FLAG_ETHARP;
116  netif->flags |= NETIF_FLAG_ETHERNET;
117  break;
118 #endif
119 #ifdef MODULE_LWIP_SIXLOWPAN
120  case NETDEV2_TYPE_IEEE802154:
121  {
122  u16_t val;
123  ipv6_addr_t *addr;
124  if (netdev->driver->get(netdev, NETOPT_NID, &val,
125  sizeof(val)) < 0) {
126  return ERR_IF;
127  }
128  lowpan6_set_pan_id(val);
129  netif->hwaddr_len = (u8_t)netdev->driver->get(netdev, NETOPT_ADDRESS_LONG,
130  netif->hwaddr, sizeof(netif->hwaddr));
131  if (netif->hwaddr_len > sizeof(netif->hwaddr)) {
132  return ERR_IF;
133  }
134  netif->linkoutput = _ieee802154_link_output;
135  res = lowpan6_if_init(netif);
136  if (res != ERR_OK) {
137  return res;
138  }
139  /* assure usage of long address as source address */
140  val = netif->hwaddr_len;
141  if (netdev->driver->set(netdev, NETOPT_SRC_LEN, &val, sizeof(val)) < 0) {
142  return ERR_IF;
143  }
144  /* netif_create_ip6_linklocal_address() does weird byte-swapping
145  * with full IIDs, so let's do it ourselves */
146  addr = (ipv6_addr_t *)&(netif->ip6_addr[0]);
147  if (netdev->driver->get(netdev, NETOPT_IPV6_IID, &addr->u8[8], sizeof(eui64_t)) < 0) {
148  return ERR_IF;
149  }
150  ipv6_addr_set_link_local_prefix(addr);
151  /* Set address state. */
152 #if LWIP_IPV6_DUP_DETECT_ATTEMPTS
153  /* Will perform duplicate address detection (DAD). */
154  netif->ip6_addr_state[0] = IP6_ADDR_TENTATIVE;
155 #else
156  /* Consider address valid. */
157  netif->ip6_addr_state[0] = IP6_ADDR_PREFERRED;
158 #endif /* LWIP_IPV6_AUTOCONFIG */
159  break;
160  }
161 #endif
162  default:
163  return ERR_IF; /* device type not supported yet */
164  }
165  netif->flags |= NETIF_FLAG_UP;
166  netif->flags |= NETIF_FLAG_LINK_UP;
167  netif->flags |= NETIF_FLAG_IGMP;
168  netif->flags |= NETIF_FLAG_MLD6;
169  netdev->context = netif;
170 #if LWIP_IPV6_AUTOCONFIG
171  netif->ip6_autoconfig_enabled = 1;
172 #endif
173 
174  return res;
175 }

177 #ifdef MODULE_NETDEV2_ETH
178 static err_t _eth_link_output(struct netif *netif, struct pbuf *p)
179 {
180  netdev2_t *netdev = (netdev2_t *)netif->state;
181  struct pbuf *q;
182  unsigned int count = 0;
183 
184 #if ETH_PAD_SIZE
185  pbuf_header(p, -ETH_PAD_SIZE); /* drop the padding word */
186 #endif
187  LL_COUNT(p, q, count);
188  struct iovec pkt[count];
189  for (q = p, count = 0; q != NULL; q = q->next, count++) {
190  pkt[count].iov_base = q->payload;
191  pkt[count].iov_len = (size_t)q->len;
192  }
193 #if ETH_PAD_SIZE
194  pbuf_header(p, ETH_PAD_SIZE); /* reclaim the padding word */
195 #endif
196  return (netdev->driver->send(netdev, pkt, count) > 0) ? ERR_OK : ERR_BUF;
197 }
198 #endif

200 #ifdef MODULE_LWIP_SIXLOWPAN
201 static err_t _ieee802154_link_output(struct netif *netif, struct pbuf *p)
202 {
203  LWIP_ASSERT("p->next == NULL", p->next == NULL);
204  netdev2_t *netdev = (netdev2_t *)netif->state;
205  struct iovec pkt = {
206  .iov_base = p->payload,
207  .iov_len = (p->len - IEEE802154_FCS_LEN), /* FCS is written by driver */
208  };
209 
210  return (netdev->driver->send(netdev, &pkt, 1) > 0) ? ERR_OK : ERR_BUF;
211 }
212 #endif

214 static struct pbuf *_get_recv_pkt(netdev2_t *dev)
215 {
216  int len = dev->driver->recv(dev, _tmp_buf, sizeof(_tmp_buf), NULL);
217 
218  assert(((unsigned)len) <= UINT16_MAX);
219  struct pbuf *p = pbuf_alloc(PBUF_RAW, (u16_t)len, PBUF_POOL);
220 
221  if (p == NULL) {
222  DEBUG("lwip_netdev2: can not allocate in pbuf\n");
223  return NULL;
224  }
225  pbuf_take(p, _tmp_buf, len);
226  return p;
227 }

229 static void _event_cb(netdev2_t *dev, netdev2_event_t event)
230 {
231  if (event == NETDEV2_EVENT_ISR) {
232  assert(_pid != KERNEL_PID_UNDEF);
233  msg_t msg;
234 
235  msg.type = LWIP_NETDEV2_MSG_TYPE_EVENT;
236  msg.content.ptr = dev;
237 
238  if (msg_send(&msg, _pid) <= 0) {
239  DEBUG("lwip_netdev2: possibly lost interrupt.\n");
240  }
241  }
242  else {
243  struct netif *netif = dev->context;
244  switch (event) {
245  case NETDEV2_EVENT_RX_COMPLETE: {
246  struct pbuf *p = _get_recv_pkt(dev);
247  if (p == NULL) {
248  DEBUG("lwip_netdev2: error receiving packet\n");
249  return;
250  }
251  if (netif->input(p, netif) != ERR_OK) {
252  DEBUG("lwip_netdev2: error inputing packet\n");
253  return;
254  }
255  }
256  break;
257  default:
258  break;
259  }
260  }
261 }

263 static void *_event_loop(void *arg)
264 {
265  (void)arg;
266  msg_init_queue(_queue, LWIP_NETDEV2_QUEUE_LEN);
267  while (1) {
268  msg_t msg;
269  msg_receive(&msg);
270  if (msg.type == LWIP_NETDEV2_MSG_TYPE_EVENT) {
271  netdev2_t *dev = msg.content.ptr;
272  dev->driver->isr(dev);
273  }
274  }
275  return NULL;
276 }
