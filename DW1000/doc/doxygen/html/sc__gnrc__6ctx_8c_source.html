77 int _gnrc_6ctx_add(char *cmd_str, char *ctx_str, char *prefix_str, char *ltime_str)
78 {
79  ipv6_addr_t prefix;
80  char *addr_str, *prefix_len_str, *save_ptr;
81  unsigned prefix_len;
82  unsigned ltime;
83  unsigned ctx = (unsigned)atoi(ctx_str);
84  if (ctx >= GNRC_SIXLOWPAN_CTX_SIZE) {
85  _usage(cmd_str);
86  return 1;
87  }
88  addr_str = strtok_r(prefix_str, "/", &save_ptr);
89  if (addr_str == NULL) {
90  _usage(cmd_str);
91  return 1;
92  }
93  prefix_len_str = strtok_r(NULL, "/", &save_ptr);
94  if (addr_str == NULL) {
95  _usage(cmd_str);
96  return 1;
97  }
98  prefix_len = (unsigned)atoi(prefix_len_str);
99  if ((prefix_len - 1U) > 128U) {
100  puts("ERROR: prefix_len < 1 || prefix_len > 128");
101  return 1;
102  }
103  ipv6_addr_from_str(&prefix, addr_str);
104  if (ipv6_addr_is_unspecified(&prefix)) {
105  puts("ERROR: prefix may not be ::");
106  return 1;
107  }
108  if (abr == NULL) {
109  abr = gnrc_sixlowpan_nd_router_abr_get();
110  }
111  if (gnrc_sixlowpan_nd_router_abr_add_ctx(abr, ctx) < 0) {
112  puts("ERROR: can not add context");
113  return 1;
114  }
115  ltime = (unsigned)atoi(ltime_str);
116  if (gnrc_sixlowpan_ctx_update((uint8_t)ctx, &prefix, (uint8_t)prefix_len, ltime,
117  true) == NULL) {
118  puts("ERROR: can not add context");
119  return 1;
120  }
121  /* advertise new/updated context */
122  _adv_ctx();
123  return 0;
124 }

126 int _gnrc_6ctx_del(char *cmd_str, char *ctx_str)
127 {
128  gnrc_sixlowpan_ctx_t *ctx;
129  unsigned cid = (unsigned)atoi(ctx_str);
130  if (cid >= GNRC_SIXLOWPAN_CTX_SIZE) {
131  _usage(cmd_str);
132  return 1;
133  }
134  else if (del_timer[cid].callback == NULL) {
135  ctx = gnrc_sixlowpan_ctx_lookup_id(cid);
136  if (ctx != NULL) {
137  ctx->flags_id &= ~GNRC_SIXLOWPAN_CTX_FLAGS_COMP;
138  ctx->ltime = 0;
139  del_timer[cid].callback = _del_cb;
140  del_timer[cid].arg = ctx;
141  xtimer_set(&del_timer[cid], GNRC_SIXLOWPAN_ND_RTR_MIN_CTX_DELAY * US_PER_SEC);
142  }
143  }
144  else {
145  printf("Context %u already marked for removal\n", cid);
146  return 1;
147  }
148  /* advertise updated context */
149  _adv_ctx();
150  return 0;
151 }

170  return 1;
