59 ipv6_addr_t *gnrc_ndp_internal_default_router(void)
60 {
61  gnrc_ipv6_nc_t *router = gnrc_ipv6_nc_get_next_router(NULL);
62 
63  /* first look if there is any reachable router */
64  while (router != NULL) {
65  if ((gnrc_ipv6_nc_get_state(router) != GNRC_IPV6_NC_STATE_INCOMPLETE) &&
66  (gnrc_ipv6_nc_get_state(router) != GNRC_IPV6_NC_STATE_UNREACHABLE)) {
67  _last_router = NULL;
68 
69  return &router->ipv6_addr;
70  }
71 
72  router = gnrc_ipv6_nc_get_next_router(router);
73  }
74 
75  /* else take the first one, but keep round-robin in further selections */
76  router = gnrc_ipv6_nc_get_next_router(_last_router);
77 
78  if (router == NULL) { /* end of router list or there is none => wrap around */
79  router = gnrc_ipv6_nc_get_next_router(router);
80 
81  if (router == NULL) { /* still nothing found => no router in list */
82  return NULL;
83  }
84  }
85 
86  _last_router = router;
87 
88  return &router->ipv6_addr;
89 }

91 void gnrc_ndp_internal_set_state(gnrc_ipv6_nc_t *nc_entry, uint8_t state)
92 {
93  gnrc_ipv6_netif_t *ipv6_iface;
94  uint32_t t = GNRC_NDP_FIRST_PROBE_DELAY * US_PER_SEC;
95 
96  nc_entry->flags &= ~GNRC_IPV6_NC_STATE_MASK;
97  nc_entry->flags |= state;
98 
99  DEBUG("ndp internal: set %s state to ",
100  ipv6_addr_to_str(addr_str, &nc_entry->ipv6_addr, sizeof(addr_str)));
101 
102  switch (state) {
103  case GNRC_IPV6_NC_STATE_REACHABLE:
104  ipv6_iface = gnrc_ipv6_netif_get(nc_entry->iface);
105  DEBUG("REACHABLE (reachable time = %" PRIu32 " us)\n", ipv6_iface->reach_time);
106  t = ipv6_iface->reach_time;
107 
108  /* we intentionally fall through here to set the desired timeout t */
109  case GNRC_IPV6_NC_STATE_DELAY:
110 #if ENABLE_DEBUG
111  if (state == GNRC_IPV6_NC_STATE_DELAY) {
112  DEBUG("DELAY (probe with unicast NS in %u seconds)\n",
113  GNRC_NDP_FIRST_PROBE_DELAY);
114  }
115 #endif
116  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, t, GNRC_NDP_MSG_NC_STATE_TIMEOUT,
117  gnrc_ipv6_pid);
118  break;
119 
120  case GNRC_IPV6_NC_STATE_PROBE:
121  ipv6_iface = gnrc_ipv6_netif_get(nc_entry->iface);
122 
123  nc_entry->probes_remaining = GNRC_NDP_MAX_UC_NBR_SOL_NUMOF;
124  DEBUG("PROBE (probe with %" PRIu8 " unicast NS every %" PRIu32 " us)\n",
125  nc_entry->probes_remaining, ipv6_iface->retrans_timer);
126 
127  gnrc_ndp_internal_send_nbr_sol(nc_entry->iface, NULL, &nc_entry->ipv6_addr,
128  &nc_entry->ipv6_addr);
129 
130  mutex_lock(&ipv6_iface->mutex);
131  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, ipv6_iface->retrans_timer,
132  GNRC_NDP_MSG_NBR_SOL_RETRANS, gnrc_ipv6_pid);
133  mutex_unlock(&ipv6_iface->mutex);
134  break;
135 
136 #ifdef ENABLE_DEBUG
137  case GNRC_IPV6_NC_STATE_STALE:
138  DEBUG("STALE (go into DELAY on next packet)\n");
139  break;
140 #endif
141 
142  default:
143  DEBUG("errorneous or unknown\n");
144  break;
145  }

148 void gnrc_ndp_internal_send_nbr_adv(kernel_pid_t iface, ipv6_addr_t *tgt, ipv6_addr_t *dst,
149  bool supply_tl2a, gnrc_pktsnip_t *ext_opts)
150 {
151  gnrc_pktsnip_t *hdr, *pkt = ext_opts;
152  uint8_t adv_flags = 0;
153 
154  DEBUG("ndp internal: send neighbor advertisement (iface: %" PRIkernel_pid ", tgt: %s, ",
155  iface, ipv6_addr_to_str(addr_str, tgt, sizeof(addr_str)));
156  DEBUG("dst: %s, supply_tl2a: %d)\n",
157  ipv6_addr_to_str(addr_str, dst, sizeof(addr_str)), supply_tl2a);
158 
159  if ((gnrc_ipv6_netif_get(iface)->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER) &&
160  (gnrc_ipv6_netif_get(iface)->flags & GNRC_IPV6_NETIF_FLAGS_RTR_ADV)) {
161  adv_flags |= NDP_NBR_ADV_FLAGS_R;
162  }
163 
164  if (ipv6_addr_is_unspecified(dst)) {
165  ipv6_addr_set_all_nodes_multicast(dst, IPV6_ADDR_MCAST_SCP_LINK_LOCAL);
166  }
167  else {
168  adv_flags |= NDP_NBR_ADV_FLAGS_S;
169  }
170 
171  if (supply_tl2a) {
172  uint8_t l2src[8];
173  size_t l2src_len;
174  /* we previously checked if we are the target, so we can take our L2src */
175  l2src_len = _get_l2src(iface, l2src, sizeof(l2src));
176 
177  if (l2src_len > 0) {
178  /* add target address link-layer address option */
179  pkt = gnrc_ndp_opt_tl2a_build(l2src, l2src_len, pkt);
180 
181  if (pkt == NULL) {
182  DEBUG("ndp internal: error allocating Target Link-layer address option.\n");
183  gnrc_pktbuf_release(ext_opts);
184  return;
185  }
186  }
187  }
188 
189  /* TODO: also check if the node provides proxy servies for tgt */
190  if ((pkt != NULL) && !gnrc_ipv6_netif_addr_is_non_unicast(tgt)) {
191  /* TL2A is not supplied and tgt is not anycast */
192  adv_flags |= NDP_NBR_ADV_FLAGS_O;
193  }
194 
195  hdr = gnrc_ndp_nbr_adv_build(adv_flags, tgt, pkt);
196 
197  if (hdr == NULL) {
198  DEBUG("ndp internal: error allocating Neighbor advertisement.\n");
199  gnrc_pktbuf_release(pkt);
200  return;
201  }
202  pkt = hdr;
203  hdr = _build_headers(iface, pkt, dst, NULL);
204  if (hdr == NULL) {
205  DEBUG("ndp internal: error adding lower-layer headers.\n");
206  gnrc_pktbuf_release(pkt);
207  return;
208  }
209  if (gnrc_ipv6_netif_addr_is_non_unicast(tgt)) {
210  /* avoid collision for anycast addresses
211  * (see https://tools.ietf.org/html/rfc4861#section-7.2.7) */
212  uint32_t delay = random_uint32_range(0, GNRC_NDP_MAX_AC_TGT_DELAY * US_PER_SEC);
213  gnrc_ipv6_nc_t *nc_entry = gnrc_ipv6_nc_get(iface, dst);
214  DEBUG("ndp internal: delay neighbor advertisement for %" PRIu32 " sec.",
215  (delay / US_PER_SEC));
216 
217  /* nc_entry must be set so no need to check it */
218  assert(nc_entry);
219 
220  _send_delayed(&nc_entry->nbr_adv_timer, &nc_entry->nbr_adv_msg, delay, hdr);
221  }
222  else if (gnrc_netapi_send(gnrc_ipv6_pid, hdr) < 1) {
223  DEBUG("ndp internal: unable to send neighbor advertisement\n");
224  gnrc_pktbuf_release(hdr);
225  }
226 }

422 void gnrc_ndp_internal_send_rtr_adv(kernel_pid_t iface, ipv6_addr_t *src, ipv6_addr_t *dst,
423  bool fin)
424 {
425  gnrc_pktsnip_t *hdr = NULL, *pkt = NULL;
426  gnrc_ipv6_netif_t *ipv6_iface = gnrc_ipv6_netif_get(iface);
427  uint32_t reach_time = 0, retrans_timer = 0;
428  uint16_t adv_ltime = 0;
429  uint8_t cur_hl = 0;
430 
431  if (dst == NULL) {
432  /* isn't changed afterwards so discarding const should be fine */
433  dst = (ipv6_addr_t *)&ipv6_addr_all_nodes_link_local;
434  }
435  DEBUG("ndp internal: send router advertisement (iface: %" PRIkernel_pid ", dst: %s%s\n",
436  iface, ipv6_addr_to_str(addr_str, dst, sizeof(addr_str)), fin ? ", final" : "");
437  mutex_lock(&ipv6_iface->mutex);
438 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
439  if (!(ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN)) {
440 #endif
441  if (!_add_pios(&hdr, ipv6_iface, pkt)) {
442  /* pkt already released in _add_pios */
443  mutex_unlock(&ipv6_iface->mutex);
444  return;
445  }
446  pkt = hdr;
447 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
448  }
449  else {
450  gnrc_sixlowpan_nd_router_abr_t *abr = gnrc_sixlowpan_nd_router_abr_get();
451  if (abr != NULL) {
452  gnrc_sixlowpan_nd_router_prf_t *prf = abr->prfs;
453  /* add prefixes from border router */
454  while (prf) {
455  bool processed_before = false;
456  /* skip if prefix does not belong to iface */
457  if (prf->iface != ipv6_iface) {
458  prf = prf->next;
459  continue;
460  }
461  /* skip if prefix has been processed already */
462  for (gnrc_sixlowpan_nd_router_prf_t *tmp = abr->prfs; tmp != prf; tmp = tmp->next) {
463  if ((processed_before =
464  _check_prefixes(prf->prefix, tmp->prefix))) {
465  break;
466  }
467  }
468 
469  if (processed_before) {
470  prf = prf->next;
471  continue;
472  }
473 
474  if (_pio_from_iface_addr(&hdr, ipv6_iface, prf->prefix, pkt)) {
475  if (hdr != NULL) {
476  pkt = hdr;
477  }
478  else {
479  DEBUG("ndp rtr: error allocating PIO\n");
480  gnrc_pktbuf_release(pkt);
481  mutex_unlock(&ipv6_iface->mutex);
482  return;
483  }
484  }
485  prf = prf->next;
486  }
487  for (unsigned int i = 0; i < GNRC_SIXLOWPAN_CTX_SIZE; i++) {
488  gnrc_sixlowpan_ctx_t *ctx;
489  if (!bf_isset(abr->ctxs, i)) {
490  continue;
491  }
492  ctx = gnrc_sixlowpan_ctx_lookup_id(i);
493  hdr = gnrc_sixlowpan_nd_opt_6ctx_build(ctx->prefix_len, ctx->flags_id, ctx->ltime,
494  &ctx->prefix, pkt);
495  if (hdr == NULL) {
496  DEBUG("ndp rtr: error allocating 6CO\n");
497  mutex_unlock(&ipv6_iface->mutex);
498  gnrc_pktbuf_release(pkt);
499  return;
500  }
501  pkt = hdr;
502  }
503  hdr = gnrc_sixlowpan_nd_opt_abr_build(abr->version, abr->ltime, &abr->addr, pkt);
504  if (hdr == NULL) {
505  DEBUG("ndp internal: error allocating ABRO.\n");
506  mutex_unlock(&ipv6_iface->mutex);
507  gnrc_pktbuf_release(pkt);
508  return;
509  }
510  pkt = hdr;
511  }
512  }
513 #endif /* MODULE_GNRC_SIXLOWPAN_ND_ROUTER */
514  if (ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_ADV_MTU) {
515  if ((hdr = gnrc_ndp_opt_mtu_build(ipv6_iface->mtu, pkt)) == NULL) {
516  DEBUG("ndp rtr: no space left in packet buffer\n");
517  mutex_unlock(&ipv6_iface->mutex);
518  gnrc_pktbuf_release(pkt);
519  return;
520  }
521  pkt = hdr;
522  }
523  if (src == NULL) {
524  mutex_unlock(&ipv6_iface->mutex);
525  /* get address from source selection algorithm.
526  * Only link local addresses may be used (RFC 4861 section 4.1) */
527  src = gnrc_ipv6_netif_find_best_src_addr(iface, dst, true);
528  mutex_lock(&ipv6_iface->mutex);
529  }
530  /* add SL2A for source address */
531  if (src != NULL) {
532  DEBUG(" - SL2A\n");
533  uint8_t l2src[8];
534  size_t l2src_len;
535  /* optimization note: MAY also be omitted to facilitate in-bound load balancing over
536  * replicated interfaces.
537  * source: https://tools.ietf.org/html/rfc4861#section-6.2.3 */
538  l2src_len = _get_l2src(iface, l2src, sizeof(l2src));
539  if (l2src_len > 0) {
540  /* add source address link-layer address option */
541  hdr = gnrc_ndp_opt_sl2a_build(l2src, l2src_len, pkt);
542 
543  if (hdr == NULL) {
544  DEBUG("ndp internal: error allocating Source Link-layer address option.\n");
545  mutex_unlock(&ipv6_iface->mutex);
546  gnrc_pktbuf_release(pkt);
547  return;
548  }
