47 int cc110x_setup(cc110x_t *dev, const cc110x_params_t *params)
48 {
49  DEBUG("%s:%s:%u\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
50 
51 #ifdef MODULE_CC110X_HOOKS
52  cc110x_hooks_init();
53 #endif
54 
55  dev->params = *params;
56 
57  /* Configure chip-select */
58  spi_init_cs(dev->params.spi, dev->params.cs);
59 
60  /* Configure GDO1 */
61  gpio_init(dev->params.gdo1, GPIO_IN);
62 
63 #ifndef CC110X_DONT_RESET
64  /* reset device*/
65  _power_up_reset(dev);
66 #endif
67 
68  /* set default state */
69  dev->radio_state = RADIO_IDLE;
70 
71  /* Write configuration to configuration registers */
72  cc110x_writeburst_reg(dev, 0x00, cc110x_default_conf, cc110x_default_conf_size);
73 
74  /* Write PATABLE (power settings) */
75  cc110x_writeburst_reg(dev, CC110X_PATABLE, CC110X_DEFAULT_PATABLE, 8);
76 
77  /* set base frequency */
78  cc110x_set_base_freq_raw(dev, CC110X_DEFAULT_FREQ);
79 
80  /* Set default channel number */
81  cc110x_set_channel(dev, CC110X_DEFAULT_CHANNEL);
82 
83  /* set default node id */
84  uint8_t addr;
85  uuid_get(&addr, 1);
86  cc110x_set_address(dev, addr);
87 
88  LOG_INFO("cc110x: initialized with address=%u and channel=%i\n",
89  (unsigned)dev->radio_address,
90  dev->radio_channel);
91 
92  return 0;
93 }

95 uint8_t cc110x_set_address(cc110x_t *dev, uint8_t address)
96 {
97  DEBUG("%s:%s:%u setting address %u\n", RIOT_FILE_RELATIVE, __func__,
98  __LINE__, (unsigned)address);
99  if (!(address < MIN_UID) || (address > MAX_UID)) {
100  if (dev->radio_state != RADIO_UNKNOWN) {
101  cc110x_write_register(dev, CC110X_ADDR, address);
102  dev->radio_address = address;
103  return address;
104  }
105  }
106 
107  return 0;
108 }

221 void cc110x_write_register(cc110x_t *dev, uint8_t r, uint8_t value)
222 {
223  /* Save old radio state */
224  uint8_t old_state = dev->radio_state;
225 
226  /* Wake up from RX (no effect if in other mode) */
227  cc110x_wakeup_from_rx(dev);
228  cc110x_write_reg(dev, r, value);
229 
230  /* Have to put radio back to RX if old radio state
231  * was RX, otherwise no action is necessary */
232  if (old_state == RADIO_RX) {
233  cc110x_switch_to_rx(dev);
234  }
235 }

237 int cc110x_rd_set_mode(cc110x_t *dev, int mode)
238 {
239  DEBUG("%s:%s:%u\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
240 
241  int result;
242 
243  /* Get current radio mode */
244  if ((dev->radio_state == RADIO_UNKNOWN) || (dev->radio_state == RADIO_PWD)) {
245  result = RADIO_MODE_OFF;
246  }
247  else {
248  result = RADIO_MODE_ON;
249  }
250 
251  switch(mode) {
252  case RADIO_MODE_ON:
253  LOG_DEBUG("cc110x: switching to RX mode\n");
254  cc110x_setup_rx_mode(dev); /* Set chip to desired mode */
255  break;
256 
257  case RADIO_MODE_OFF:
258  gpio_irq_disable(dev->params.gdo2); /* Disable interrupts */
259  cc110x_switch_to_pwd(dev); /* Set chip to power down mode */
260  break;
261 
262  case RADIO_MODE_GET:
263  /* do nothing, just return current mode */
264  default:
265  /* do nothing */
266  break;
267  }
268 
269  /* Return previous mode */
270  return result;
271 }
