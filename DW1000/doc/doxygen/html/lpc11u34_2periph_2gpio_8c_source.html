66 int gpio_init(gpio_t pin, gpio_mode_t mode)
67 {
68  /* reset any eventually set interrupt configuration */
69  for (int i = 0; i < ISR_NUMOF; i++) {
70  if (isrmap[i] == pin) {
71  NVIC_DisableIRQ(FLEX_INT0_IRQn + i);
72  isrmap[i] = GPIO_UNDEF;
73  }
74  }

77  LPC_GPIO->DIR[port(pin)] |= ((mode & MODE_DIR_MASK) << num(pin));
78  IOCON->PIO[ionum(pin)] = (mode & MODE_IOCON_MASK);
79 
80  return 0;

83 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
84  gpio_cb_t cb, void *arg)
85 {
86  /* make sure we have an interrupt channel available */
87  int i = 0;
88  while ((i < ISR_NUMOF) && (isrmap[i] != GPIO_UNDEF) && (isrmap[i] != pin)) {
89  i++;
90  }
91  if (i == ISR_NUMOF) {
92  return -1;
93  }
94 
95  /* do basic pin configuration */
96  if (gpio_init(pin, mode) != 0) {
97  return -1;
98  }
99 
100  /* enable power for GPIO pin interrupt interface */
101  LPC_SYSCON->SYSAHBCLKCTRL |= SYSCTL_PINTMASK;
102 
103  /* save ISR context */
104  isrctx[i].cb = cb;
105  isrctx[i].arg = arg;
106  isrmap[i] = pin;
107 
108  /* Register pin as flex interrupt */
109  LPC_SYSCON->PINTSEL[i] = ionum(pin);
110 
111  /* set active flank configuration */
112  LPC_GPIO_PIN_INT->ISEL = 0;
113  switch (flank) {
114  case GPIO_RISING:
115  LPC_GPIO_PIN_INT->IENR |= (1 << i);
116  LPC_GPIO_PIN_INT->IENF &= ~(1 << i);
117  break;
118  case GPIO_FALLING:
119  LPC_GPIO_PIN_INT->IENR &= ~(1 << i);
120  LPC_GPIO_PIN_INT->IENF |= (1 << i);
121  break;
122  case GPIO_BOTH:
123  LPC_GPIO_PIN_INT->IENR |= (1 << i);
124  LPC_GPIO_PIN_INT->IENF |= (1 << i);
125  default:
126  return -1;
127  }
128 
129  /* clear any pending requests and enable the pin's interrupt */
130  LPC_GPIO_PIN_INT->IST = (1 << i);
131  NVIC_EnableIRQ(FLEX_INT0_IRQn + i);
132 
133  return 0;
134 }
