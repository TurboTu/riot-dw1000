165 static enum rfc5444_result
166 _nhdp_blocktlv_address_cb(struct rfc5444_reader_tlvblock_context *cont)
167 {
168  uint8_t tmp_result;
169 
170  /* Get NHDP address for the current netaddr */
171  nhdp_addr_t *current_addr = get_nhdp_db_addr(&cont->addr._addr[0], cont->addr._prefix_len);
172 
173  if (!current_addr) {
174  /* Insufficient memory */
175  return RFC5444_DROP_MESSAGE;
176  }
177 
178  /* Check validity of address tlvs */
179  if (check_addr_validity(current_addr) != RFC5444_OKAY) {
180  nhdp_decrement_addr_usage(current_addr);
181  return RFC5444_DROP_MESSAGE;
182  }
183 
184  /* Handle address and add it to proper temporary list */
185  if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_LOCAL_IF].tlv) {
186  switch (*_nhdp_addr_tlvs[RFC5444_ADDRTLV_LOCAL_IF].tlv->single_value) {
187  case RFC5444_LOCALIF_THIS_IF:
188  current_addr->in_tmp_table = NHDP_ADDR_TMP_SEND_LIST;
189  break;
190 
191  case RFC5444_LOCALIF_OTHER_IF:
192  current_addr->in_tmp_table = NHDP_ADDR_TMP_NB_LIST;
193  break;
194 
195  default:
196  /* Wrong value, drop message */
197  nhdp_decrement_addr_usage(current_addr);
198  return RFC5444_DROP_MESSAGE;
199  }
200  }
201  else if ((tmp_result = lib_is_reg_addr(if_pid, current_addr))) {
202  /* The address is one of our local addresses (do not add it for processing) */
203  if ((!sym) && (tmp_result == 1) && _nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv) {
204  /* If address is a local address of the receiving interface, check */
205  /* whether we can derive a status for this link (symmetry or lost) */
206  switch (*_nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv->single_value) {
207  case RFC5444_LINKSTATUS_SYMMETRIC:
208  /* Fall - through */
209 
210  case RFC5444_LINKSTATUS_HEARD:
211  sym = 1;
212  break;
213 
214  case RFC5444_LINKSTATUS_LOST:
215  lost = 1;
216  break;
217 
218  default:
219  /* Wrong value, drop message */
220  nhdp_decrement_addr_usage(current_addr);
221  return RFC5444_DROP_MESSAGE;
222  }
223  }
224 
225  if (lt_metric_val == NHDP_METRIC_UNKNOWN
226  && _nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_METRIC].tlv != NULL) {
227  /* Determine our outgoing link metric value to the originator interface */
228  uint16_t metric_enc = *((uint16_t*)_nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_METRIC]
229  .tlv->single_value);
230  if (metric_enc & NHDP_KD_LM_INC) {
231  /* Incoming metric value at the neighbor if is outgoing value for our if */
232  lt_metric_val = rfc5444_metric_decode(metric_enc);
233  }
234  }
235 
236  /* Address is one of our own addresses, ignore it */
237  nhdp_decrement_addr_usage(current_addr);
238  }
239  else if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv) {
240  switch (*_nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv->single_value) {
241  case RFC5444_LINKSTATUS_SYMMETRIC:
242  add_temp_metric_value(current_addr);
243  current_addr->in_tmp_table = NHDP_ADDR_TMP_TH_SYM_LIST;
244  break;
245 
246  case RFC5444_LINKSTATUS_HEARD:
247  /* Fall-through */
248 
249  case RFC5444_LINKSTATUS_LOST:
250  if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv
251  && *_nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv->single_value
252  == RFC5444_OTHERNEIGHB_SYMMETRIC) {
253  /* Symmetric has higher priority */
254  add_temp_metric_value(current_addr);
255  current_addr->in_tmp_table = NHDP_ADDR_TMP_TH_SYM_LIST;
256  }
257  else {
258  current_addr->in_tmp_table = NHDP_ADDR_TMP_TH_REM_LIST;
259  }
260 
261  break;
262 
263  default:
264  /* Wrong value, drop message */
265  nhdp_decrement_addr_usage(current_addr);
266  return RFC5444_DROP_MESSAGE;
267  }
268  }
269  else if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv) {
270  switch (*_nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv->single_value) {
271  case RFC5444_OTHERNEIGHB_SYMMETRIC:
272  add_temp_metric_value(current_addr);
273  current_addr->in_tmp_table = NHDP_ADDR_TMP_TH_SYM_LIST;
274  break;
275 
276  case RFC5444_OTHERNEIGHB_LOST:
277  current_addr->in_tmp_table = NHDP_ADDR_TMP_TH_REM_LIST;
278  break;
279 
280  default:
281  /* Wrong value, drop message */
282  nhdp_decrement_addr_usage(current_addr);
283  return RFC5444_DROP_MESSAGE;
284  }
285  }
286  else {
287  /* Addresses without expected TLV are ignored */
288  nhdp_decrement_addr_usage(current_addr);
289  return RFC5444_DROP_ADDRESS;
290  }
291 
292  return RFC5444_OKAY;
293 }

300 _nhdp_blocktlv_msg_cb(struct rfc5444_reader_tlvblock_context *cont)
301 {

310 
311  /* Interval time is not mandatory as message tlv */

313  int_time = rfc5444_timetlv_decode(
314  *_nhdp_msg_tlvs[RFC5444_MSGTLV_INTERVAL_TIME].tlv->single_value);
315  }
316 
317  return RFC5444_OKAY;

350 static enum rfc5444_result check_msg_validity(struct rfc5444_reader_tlvblock_context *cont)
351 {
352  if (cont->has_hoplimit && cont->hoplimit != 1) {
353  /* Hop Limit other than 1 */
354  return RFC5444_DROP_MESSAGE;
355  }
356 
357  if (cont->has_hopcount && cont->hopcount != 0) {
358  /* Hop Count other than zero */
359  return RFC5444_DROP_MESSAGE;
360  }
361 
362  if (!(_nhdp_msg_tlvs[RFC5444_MSGTLV_VALIDITY_TIME].tlv)) {
363  /* No validity time tlv */
364  return RFC5444_DROP_MESSAGE;
365  }
366  else if (_nhdp_msg_tlvs[RFC5444_MSGTLV_VALIDITY_TIME].tlv->next_entry) {
367  /* Multiple validity time tlvs */
368  return RFC5444_DROP_MESSAGE;
369  }
370 
371  if (_nhdp_msg_tlvs[RFC5444_MSGTLV_INTERVAL_TIME].tlv
372  && _nhdp_msg_tlvs[RFC5444_MSGTLV_INTERVAL_TIME].tlv->next_entry) {
373  /* Multiple interval time tlvs */
374  return RFC5444_DROP_MESSAGE;
375  }
376 
377  return RFC5444_OKAY;
378 }

383 static enum rfc5444_result check_addr_validity(nhdp_addr_t *addr)
384 {
385  if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_LOCAL_IF].tlv) {
386  if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv
387  || _nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv) {
388  /* Conflicting tlv types for the address */
389  return RFC5444_DROP_MESSAGE;
390  }
391  else if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_LOCAL_IF].tlv->next_entry) {
392  /* Multiple tlvs of the same type are not allowed */
393  return RFC5444_DROP_MESSAGE;
394  }
395  else if (lib_is_reg_addr(if_pid, addr)) {
396  /* Address of one of neighbor's IFs equals one of ours */
397  return RFC5444_DROP_MESSAGE;
398  }
399  }
400  else if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv
401  && _nhdp_addr_tlvs[RFC5444_ADDRTLV_LINK_STATUS].tlv->next_entry) {
402  /* Multiple tlvs of the same type are not allowed */
403  return RFC5444_DROP_MESSAGE;
404  }
405  else if (_nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv
406  && _nhdp_addr_tlvs[RFC5444_ADDRTLV_OTHER_NEIGHB].tlv->next_entry) {
407  /* Multiple tlvs of the same type are not allowed */
408  return RFC5444_DROP_MESSAGE;
409  }
410 
411  return RFC5444_OKAY;
412 }
