189 static void fpu_used_interrupt(uint8_t intr_num, struct x86_pushad *orig_ctx, unsigned long error_code)
190 {
191  static volatile struct x86_fxsave fpu_data;
192 
193  (void) intr_num;
194  (void) orig_ctx;
195  (void) error_code;
196 
197  __asm__ volatile ("clts"); /* clear task switch flag */
198 
199  if (fpu_owner == sched_active_pid) {
200  return;
201  }
202 
203  if (fpu_owner != KERNEL_PID_UNDEF) {
204  ucontext_t *ctx_owner = (ucontext_t *) sched_threads[fpu_owner]->sp;
205  __asm__ volatile ("fxsave (%0)" :: "r"(&fpu_data));
206  ctx_owner->__fxsave = fpu_data;
207  }
208 
209  ucontext_t *ctx_active = (ucontext_t *) sched_active_thread->sp;
210  fpu_data = ctx_active->__fxsave;
211  __asm__ volatile ("fxrstor (%0)" :: "r"(&fpu_data));
212 
213  fpu_owner = sched_active_pid;
214 }
