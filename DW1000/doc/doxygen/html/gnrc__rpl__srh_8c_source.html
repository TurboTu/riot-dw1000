33  return EXT_RH_CODE_OK;
34  }
35 
36  uint8_t n = (((rh->len * 8) - GNRC_RPL_SRH_PADDING(rh->pad_resv) -
37  (16 - GNRC_RPL_SRH_COMPRE(rh->compr))) /
38  (16 - GNRC_RPL_SRH_COMPRI(rh->compr))) + 1;
39  ipv6_addr_t addr = ipv6->dst, tmp;
40  uint8_t i, pref_elided, tmp_pref_elided, addr_len, compri_addr_len, tmp_addr_len, found_pos = 0;
41  uint8_t *addr_vec = (uint8_t *) (rh + 1);
42  bool found = false;
43 
44  DEBUG("RPL SRH: %u addresses in the routing header\n", (unsigned) n);
45 
46  if (rh->seg_left > n) {
47  DEBUG("RPL SRH: number of segments left > number of addresses - discard\n");
48  /* TODO ICMP Parameter Problem - Code 0 */
49  return EXT_RH_CODE_ERROR;
50  }
51 
52  rh->seg_left--;
53  i = n - rh->seg_left;
54  pref_elided = rh->seg_left ? GNRC_RPL_SRH_COMPRI(rh->compr) : GNRC_RPL_SRH_COMPRE(rh->compr);
55  compri_addr_len = sizeof(ipv6_addr_t) - GNRC_RPL_SRH_COMPRI(rh->compr);
56  addr_len = sizeof(ipv6_addr_t) - pref_elided;
57  memcpy(&addr.u8[pref_elided], &addr_vec[(i - 1) * compri_addr_len], addr_len);
58 
59  if (ipv6_addr_is_multicast(&ipv6->dst) || ipv6_addr_is_multicast(&addr)) {
60  DEBUG("RPL SRH: found a multicast address - discard\n");
61  /* TODO discard the packet */
62  return EXT_RH_CODE_ERROR;
63  }
64 
65  /* check if multiple addresses of my interface exist */
66  tmp_pref_elided = GNRC_RPL_SRH_COMPRI(rh->compr);
67  tmp_addr_len = sizeof(ipv6_addr_t) - tmp_pref_elided;
68  tmp = ipv6->dst;
69  for (uint8_t k = 0; k < n; k++) {
70  if (k == n - 1) {
71  tmp_pref_elided = GNRC_RPL_SRH_COMPRE(rh->compr);
72  tmp_addr_len = sizeof(ipv6_addr_t) - tmp_pref_elided;
73  }

75  if (gnrc_ipv6_netif_find_by_addr(NULL, &tmp) != KERNEL_PID_UNDEF) {
76  if (found && ((k - found_pos) > 1)) {
77  DEBUG("RPL SRH: found multiple addresses that belong to me - discard\n");
78  /* TODO send an ICMP Parameter Problem (Code 0) and discard the packet */
79  return EXT_RH_CODE_ERROR;
80  }
81  found_pos = k;
82  found = true;
83  }
84  }
85 
86  memcpy(&addr_vec[(i - 1) * compri_addr_len], &ipv6->dst.u8[pref_elided], addr_len);
87

92
