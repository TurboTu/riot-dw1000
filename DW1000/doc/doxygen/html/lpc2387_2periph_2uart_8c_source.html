37 int uart_init(uart_t dev, uint32_t baudrate, uart_rx_cb_t rx_cb, void *arg)
38 {
39  (void) baudrate;
40  /* for now, we only support one UART device and only the RX interrupt */
41  if (dev != 0) {
42  return UART_NODEV;
43  }
44 
45  /* save interrupt context */
46  _rx_cb = rx_cb;
47  _cb_arg = arg;
48 
49  /* power on the UART device */
50  PCONP |= PCUART0;
51  /* UART0 clock divider is CCLK/8 */
52  PCLKSEL0 |= BIT6 + BIT7;
53  /* configure to 8N1 */
54  U0LCR = 0x83;
55 
56  /* Baudrate calculation:
57  * BR = PCLK (9 MHz) / (16 x 256 x DLM + DLL) x (1/(DIVADDVAL/MULVAL)) */
58  /* TODO: UART Baudrate calculation using the baudrate parameter */
59  U0FDR = 0x92; /* DIVADDVAL = 0010 = 2, MULVAL = 1001 = 9 */
60  U0DLM = 0x00;
61  U0DLL = 0x04;
62 
63  U0LCR = 0x03; /* DLAB = 0 */
64  U0FCR = 0x07; /* Enable and reset TX and RX FIFO */
65 
66  /* install and enable the IRQ handler */
67  install_irq(UART0_INT, UART0_IRQHandler, 6);
68  U0IER |= BIT0; /* enable only RX irq */
69  return UART_OK;
70 }

81 void UART0_IRQHandler(void)
82 {
83  switch (U0IIR & UIIR_ID_MASK) {
84  case UIIR_CTI_INT: /* Character Timeout Indicator */
85  case UIIR_RDA_INT: /* Receive Data Available */
86  do {
87  uint8_t c = (uint8_t)U0RBR;
88  _rx_cb(_cb_arg, c);
89  } while (U0LSR & ULSR_RDR);
90  break;
91

93  U0LSR;

96  }
