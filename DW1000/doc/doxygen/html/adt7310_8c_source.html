92 static int adt7310_read_reg(const adt7310_t *dev, const uint8_t addr, const uint8_t len,
93  uint8_t *buf)
94 {
95  int status = 0;
96  uint8_t command = ADT7310_CMD_READ | (addr << ADT7310_CMD_ADDR_SHIFT);
97  /* Acquire exclusive access to the bus. */
98  spi_acquire(dev->spi, dev->cs, SPI_MODE_0, dev->clk);
99  /* Perform the transaction */
100  spi_transfer_regs(dev->spi, dev->cs, command, NULL, buf, (size_t)len);
101  /* Release the bus for other threads. */

133 int adt7310_init(adt7310_t *dev, spi_t spi, spi_clk_t clk, gpio_t cs)
134 {
135  int status;
136  uint8_t reg = 0;
137  /* write device descriptor */
138  dev->spi = spi;
139  dev->clk = clk;
140  dev->cs = cs;
141  dev->initialized = false;
142  dev->high_res = false;
143 
144  /* CS */
145  spi_init_cs(dev->spi, dev->cs);
146 
147 #if ENABLE_DEBUG
148  for (int i = 0; i < 8; ++i) {
149  uint16_t dbg_reg = 0;
150  status = adt7310_read_reg(dev, i, sizeof(dbg_reg), (uint8_t *)&dbg_reg);
151  if (status != 0) {
152  printf("Error reading address 0x%02x", i);
153  }
154  dbg_reg = HTONS(dbg_reg);
155  printf("%02x: %04" PRIx16 "\n", i, dbg_reg);
156  }
157 #endif
158 
159  /* Read ID register from device */
160  status = adt7310_read_reg(dev, ADT7310_REG_ID, ADT7310_REG_SIZE_ID, &reg);
161  if (status != 0) {
162  /* SPI bus error */
163  return -1;
164  }
165  if ((reg & ADT7310_REG_ID_MASK_MANUFACTURER_ID) != ADT7310_EXPECTED_MANUF_ID) {
166  /* Wrong part ID */
167  return -2;
168  }
169 
170  /* Set a configuration, go to shut down mode to save power until the sensor is needed. */
171  if (adt7310_set_config(dev, ADT7310_MODE_SHUTDOWN) != 0) {
172  /* communication error */
173  return -3;
174  }
175 
176  dev->initialized = true;
177  return 0;
178 }

188 int16_t adt7310_read_raw(adt7310_t *dev)
189 {
190  int status;
191  int16_t raw;
192 
193  /* Read the temperature value register */
194  status = adt7310_read_reg(dev, ADT7310_REG_VALUE, ADT7310_REG_SIZE_VALUE, (uint8_t*)&raw);
195  if (status < 0) {
196  /* communication error */
197  return INT16_MIN;
198  }
199  /* The temperature value is sent big endian (network byte order) */
200  raw = (int16_t)NTOHS((uint16_t)raw);
201  return raw;
202 }

204 int32_t adt7310_read(adt7310_t *dev)
205 {
206  int16_t raw = adt7310_read_raw(dev);
207  if (raw == INT16_MIN) {
208  return INT32_MIN;
209  }
210  if (!dev->high_res) {
211  /* filter out the flag bits */
212  raw &= ADT7310_REG_VALUE_MASK_13BIT;
213  }
214  return ((((int32_t)raw) * 1000) >> ADT7310_VALUE_FRAC_BITS);
215 }
