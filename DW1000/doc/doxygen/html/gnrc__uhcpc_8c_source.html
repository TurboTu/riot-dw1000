26 static void set_interface_roles(void)
27 {
28  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
29  size_t numof = gnrc_netif_get(ifs);
30 
31  for (size_t i = 0; i < numof && i < GNRC_NETIF_NUMOF; i++) {
32  kernel_pid_t dev = ifs[i];
33  int is_wired = gnrc_netapi_get(dev, NETOPT_IS_WIRED, 0, NULL, 0);
34  if ((!gnrc_border_interface) && (is_wired == 1)) {
35  ipv6_addr_t addr, defroute;
36  gnrc_border_interface = dev;
37 
38  ipv6_addr_from_str(&addr, "fe80::2");
39  gnrc_ipv6_netif_add_addr(dev, &addr, 64,
40  GNRC_IPV6_NETIF_ADDR_FLAGS_UNICAST);
41 
42  ipv6_addr_from_str(&defroute, "::");
43  ipv6_addr_from_str(&addr, "fe80::1");
44  fib_add_entry(&gnrc_ipv6_fib_table, dev, defroute.u8, 16,
45  0x00, addr.u8, 16, 0,
46  (uint32_t)FIB_LIFETIME_NO_EXPIRE);
47  }
48  else if ((!gnrc_wireless_interface) && (is_wired != 1)) {
49  gnrc_wireless_interface = dev;
50  }
51 
52  if (gnrc_border_interface && gnrc_wireless_interface) {
53  break;
54  }
55  }
56 
57  LOG_INFO("gnrc_uhcpc: Using %u as border interface and %u as wireless interface.\n", gnrc_border_interface, gnrc_wireless_interface);
58 }

62 void uhcp_handle_prefix(uint8_t *prefix, uint8_t prefix_len, uint16_t lifetime, uint8_t *src, uhcp_iface_t iface)
63 {
64  (void)prefix_len;
65  (void)lifetime;
66  (void)src;
67 
68  eui64_t iid;
69  if (!gnrc_wireless_interface) {
70  LOG_WARNING("gnrc_uhcpc: uhcp_handle_prefix(): received prefix, but don't know any wireless interface\n");
71  return;
72  }
73 
74  if ((kernel_pid_t)iface != gnrc_border_interface) {
75  LOG_WARNING("gnrc_uhcpc: uhcp_handle_prefix(): received prefix from unexpected interface\n");
76  return;
77  }
78 
79  if (gnrc_netapi_get(gnrc_wireless_interface, NETOPT_IPV6_IID, 0, &iid,
80  sizeof(eui64_t)) >= 0) {
81  ipv6_addr_set_aiid((ipv6_addr_t*)prefix, iid.uint8);
82  }
83  else {
84  LOG_WARNING("gnrc_uhcpc: uhcp_handle_prefix(): cannot get IID of wireless interface\n");
85  return;
86  }
87 
88  if (ipv6_addr_equal(&_prefix, (ipv6_addr_t*)prefix)) {
89  LOG_WARNING("gnrc_uhcpc: uhcp_handle_prefix(): got same prefix again\n");
90  return;
91  }
92 
93  gnrc_ipv6_netif_add_addr(gnrc_wireless_interface, (ipv6_addr_t*)prefix, 64,
94  GNRC_IPV6_NETIF_ADDR_FLAGS_UNICAST |
95  GNRC_IPV6_NETIF_ADDR_FLAGS_NDP_AUTO);
96 
97  gnrc_ipv6_netif_remove_addr(gnrc_wireless_interface, &_prefix);
98  print_str("gnrc_uhcpc: uhcp_handle_prefix(): configured new prefix ");
99  ipv6_addr_print((ipv6_addr_t*)prefix);
100  puts("/64");
101 
102  if (!ipv6_addr_is_unspecified(&_prefix)) {
103  gnrc_ipv6_netif_remove_addr(gnrc_wireless_interface, &_prefix);
104  print_str("gnrc_uhcpc: uhcp_handle_prefix(): removed old prefix ");
105  ipv6_addr_print(&_prefix);
106  puts("/64");
107  }
108 
109  memcpy(&_prefix, prefix, 16);
110 }

126 void auto_init_gnrc_uhcpc(void)
127 {
128  set_interface_roles();
129 
130  /* only start client if more than one interface is given */
131  if (! (gnrc_border_interface && gnrc_wireless_interface)) {
132  LOG_WARNING("gnrc_uhcpc: only one interface found, skipping setup.\n");
133  return;
134  }
135 
136  /* initiate uhcp client */
137  thread_create(_uhcp_client_stack, sizeof(_uhcp_client_stack),
138  THREAD_PRIORITY_MAIN - 1, THREAD_CREATE_STACKTEST,
139  uhcp_client_thread, NULL, "uhcp");
