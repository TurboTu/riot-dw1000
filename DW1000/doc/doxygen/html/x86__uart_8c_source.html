41 void x86_early_init_uart(void)
42 {
43  outb(UART_PORT + IER, 0x00); /* Disable all interrupts */
44 
45  outb(UART_PORT + LCR, LCR_DLAB);
46  outb(UART_PORT + DLL, BAUD_LL); /* Set divisor (lo byte) */
47  outb(UART_PORT + DLH, BAUD_HL); /* (hi byte) */
48  outb(UART_PORT + LCR, LCR_WORD_BITS_8 | LCR_PAR_NONE | LCR_STOP_BITS_1);
49  outb(UART_PORT + FCR, FCR_ENABLE | FCR_CLR_RECV | FCR_CLR_SEND | FCR_TRIGGER_16);
50  outb(UART_PORT + MCR, MCR_DSR | MCR_RTS | MCR_AUX2);
51 }

65 ssize_t x86_uart_write(const char *buf, size_t len)
66 {
67  if (!UART_PORT) {
68  return -1;
69  }
70  (void) buf;
71 
72  size_t written = 0;
73  while (written < len) {
74  while (!is_output_empty()) {
75  __asm__ volatile ("pause");
76  }
77  outb(UART_PORT + THR, buf[written]);
78  ++written;
79  }
80  return written;
81 }
