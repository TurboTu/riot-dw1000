52 uint16_t bh1750fvi_sample(bh1750fvi_t *dev)
53 {
54  uint32_t tmp;
55  uint8_t raw[2];
56 
57  /* power on the device and send single H-mode measurement command */
58  DEBUG("[bh1750fvi] sample: triggering a conversion\n");
59  i2c_acquire(dev->i2c);
60  i2c_write_byte(dev->i2c, dev->addr, OP_POWER_ON);
61  i2c_write_byte(dev->i2c, dev->addr, OP_SINGLE_HRES1);
62  i2c_release(dev->i2c);
63 
64  /* wait for measurement to complete */
65  xtimer_usleep(DELAY_HMODE);
66 
67  /* read the results */
68  DEBUG("[bh1750fvi] sample: reading the results\n");
69  i2c_acquire(dev->i2c);
70  i2c_read_bytes(dev->i2c, dev->addr, raw, 2);
71  i2c_release(dev->i2c);
72 
73  /* and finally we calculate the actual LUX value */
74  tmp = (raw[0] << 24) | (raw[1] << 16);
