58 static void cpu_clock_init(void)
59 {
60  const uint32_t CLOCK_CTRL_VALUE =
61  OSC_PD
62  | (1 << IO_DIV_SHIFT)
63  | (1 << SYS_DIV_SHIFT)
64 #if !SYS_CTRL_OSC32K_USE_XTAL
65  | OSC32K
66 #endif
67  ;
68 
69 #if SYS_CTRL_OSC32K_USE_XTAL
70  /* Set the XOSC32K_Q pads to analog for the external crystal: */
71  gpio_software_control(GPIO_PD6);
72  gpio_dir_input(GPIO_PD6);
73  IOC_PXX_OVER[GPIO_PD6] = IOC_OVERRIDE_ANA;
74 
75  gpio_software_control(GPIO_PD7);
76  gpio_dir_input(GPIO_PD7);
77  IOC_PXX_OVER[GPIO_PD7] = IOC_OVERRIDE_ANA;
78 #endif
79 
80  /* Configure the clock settings: */
81  SYS_CTRL->cc2538_sys_ctrl_clk_ctrl.CLOCK_CTRL = CLOCK_CTRL_VALUE;
82 
83  /* Wait for the new clock settings to take effect: */
84  while ((SYS_CTRL->cc2538_sys_ctrl_clk_sta.CLOCK_STA ^ CLOCK_CTRL_VALUE) & CLOCK_STA_MASK);
