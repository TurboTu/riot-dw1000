112 uint8_t at86rf2xx_get_status(const at86rf2xx_t *dev)
113 {
114  /* if sleeping immediately return state */
115  if(dev->state == AT86RF2XX_STATE_SLEEP)
116  return dev->state;
117 
118  return at86rf2xx_reg_read(dev, AT86RF2XX_REG__TRX_STATUS)
119  & AT86RF2XX_TRX_STATUS_MASK__TRX_STATUS;

122 void at86rf2xx_assert_awake(at86rf2xx_t *dev)
123 {
124  if(at86rf2xx_get_status(dev) == AT86RF2XX_STATE_SLEEP) {
125 
126  /* wake up and wait for transition to TRX_OFF */
127  gpio_clear(dev->params.sleep_pin);
128  xtimer_usleep(AT86RF2XX_WAKEUP_DELAY);
129 
130  /* update state */
131  dev->state = at86rf2xx_reg_read(dev, AT86RF2XX_REG__TRX_STATUS)
132  & AT86RF2XX_TRX_STATUS_MASK__TRX_STATUS;
133  }
134 }

148 void at86rf2xx_configure_phy(at86rf2xx_t *dev)
149 {
150  uint8_t state = at86rf2xx_get_status(dev);
151 
152  /* we must be in TRX_OFF before changing the PHY configuration */
153  at86rf2xx_set_state(dev, AT86RF2XX_STATE_TRX_OFF);
154 
155 #ifdef MODULE_AT86RF212B
156  /* The TX power register must be updated after changing the channel if
157  * moving between bands. */
158  int16_t txpower = at86rf2xx_get_txpower(dev);
159 
160  uint8_t trx_ctrl2 = at86rf2xx_reg_read(dev, AT86RF2XX_REG__TRX_CTRL_2);
161  uint8_t rf_ctrl0 = at86rf2xx_reg_read(dev, AT86RF2XX_REG__RF_CTRL_0);
162 
163  /* Clear previous configuration for PHY mode */
164  trx_ctrl2 &= ~(AT86RF2XX_TRX_CTRL_2_MASK__FREQ_MODE);
165  /* Clear previous configuration for GC_TX_OFFS */
166  rf_ctrl0 &= ~AT86RF2XX_RF_CTRL_0_MASK__GC_TX_OFFS;
167 
168  if (dev->netdev.chan != 0) {
169  /* Set sub mode bit on 915 MHz as recommended by the data sheet */
170  trx_ctrl2 |= AT86RF2XX_TRX_CTRL_2_MASK__SUB_MODE;
171  }
172 
173  if (dev->page == 0) {
174  /* BPSK coding */
175  /* Data sheet recommends using a +2 dB setting for BPSK */
176  rf_ctrl0 |= AT86RF2XX_RF_CTRL_0_GC_TX_OFFS__2DB;
177  }
178  else if (dev->page == 2) {
179  /* O-QPSK coding */
180  trx_ctrl2 |= AT86RF2XX_TRX_CTRL_2_MASK__BPSK_OQPSK;
181  /* Data sheet recommends using a +1 dB setting for O-QPSK */
182  rf_ctrl0 |= AT86RF2XX_RF_CTRL_0_GC_TX_OFFS__1DB;
183  }
184 
185  at86rf2xx_reg_write(dev, AT86RF2XX_REG__TRX_CTRL_2, trx_ctrl2);
186  at86rf2xx_reg_write(dev, AT86RF2XX_REG__RF_CTRL_0, rf_ctrl0);
187 #endif
188 
189  uint8_t phy_cc_cca = at86rf2xx_reg_read(dev, AT86RF2XX_REG__PHY_CC_CCA);
190  /* Clear previous configuration for channel number */
191  phy_cc_cca &= ~(AT86RF2XX_PHY_CC_CCA_MASK__CHANNEL);
192 
193  /* Update the channel register */
194  phy_cc_cca |= (dev->netdev.chan & AT86RF2XX_PHY_CC_CCA_MASK__CHANNEL);
195  at86rf2xx_reg_write(dev, AT86RF2XX_REG__PHY_CC_CCA, phy_cc_cca);
196 
197 #ifdef MODULE_AT86RF212B
198  /* Update the TX power register to achieve the same power (in dBm) */
199  at86rf2xx_set_txpower(dev, txpower);
200 #endif
201 
202  /* Return to the state we had before reconfiguring */
203  at86rf2xx_set_state(dev, state);
204 }

206 #if defined(MODULE_AT86RF233) || defined(MODULE_AT86RF231)
207 void at86rf2xx_get_random(at86rf2xx_t *dev, uint8_t *data, const size_t len)
208 {
209  for (size_t byteCount = 0; byteCount < len; ++byteCount) {
210  uint8_t rnd = 0;
211  for (uint8_t i = 0; i < 4; ++i) {
212  /* bit 5 and 6 of the AT86RF2XX_REG__PHY_RSSI register contain the RND_VALUE */
213  uint8_t regVal = at86rf2xx_reg_read(dev, AT86RF2XX_REG__PHY_RSSI)
214  & AT86RF2XX_PHY_RSSI_MASK__RND_VALUE;
215  /* shift the two random bits first to the right and then to the correct position of the return byte */
216  regVal = regVal >> 5;
217  regVal = regVal << 2 * i;
218  rnd |= regVal;
219  }
220  data[byteCount] = rnd;
221  }
222 }
223 #endif
