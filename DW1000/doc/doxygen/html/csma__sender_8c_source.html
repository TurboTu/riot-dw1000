57 static inline uint32_t choose_backoff_period(int be,
58  const csma_sender_conf_t *conf)
59 {
60  if (be < conf->min_be) {
61  be = conf->min_be;
62  }
63  if (be > conf->max_be) {
64  be = conf->max_be;
65  }
66  uint32_t max_backoff = ((1 << be) - 1) * CSMA_SENDER_BACKOFF_PERIOD_UNIT;

70  period = CSMA_SENDER_BACKOFF_PERIOD_UNIT;

92 {
93  netopt_enable_t hwfeat;
94 
95  /* perform a CCA */
96  DEBUG("csma: Checking radio medium availability...\n");
97  int res = device->driver->get(device,
98  NETOPT_IS_CHANNEL_CLR,
99  (void *) &hwfeat,
100  sizeof(netopt_enable_t));
101  if (res < 0) {
102  /* normally impossible: we got a big internal problem! */
103  DEBUG("csma: !!! DEVICE DRIVER FAILURE! TRANSMISSION ABORTED!\n");
104  return -ECANCELED;
105  }
106 
107  /* if medium is clear, send the packet and return */
108  if (hwfeat == NETOPT_ENABLE) {
109  DEBUG("csma: Radio medium available: sending packet.\n");
110  return device->driver->send(device, vector, count);
111  }
112 
113  /* if we arrive here, medium was not available for transmission */
114  DEBUG("csma: Radio medium busy.\n");
115  return -EBUSY;

120 int csma_sender_csma_ca_send(netdev2_t *dev, struct iovec *vector,
121  unsigned count, const csma_sender_conf_t *conf)
122 {
123  netopt_enable_t hwfeat;
124 
125  assert(dev);
126  /* choose default configuration if none is given */
127  if (conf == NULL) {
128  conf = &CSMA_SENDER_CONF_DEFAULT;
129  }
130  /* Does the transceiver do automatic CSMA/CA when sending? */
131  int res = dev->driver->get(dev,
132  NETOPT_CSMA,
133  (void *) &hwfeat,
134  sizeof(netopt_enable_t));
135  bool ok = false;
136 
137  switch (res) {
138  case -ENODEV:
139  /* invalid device pointer given */
140  return -ENODEV;
141  case -ENOTSUP:
142  /* device doesn't make auto-CSMA/CA */
143  break;
144  case -EOVERFLOW: /* (normally impossible...*/
145  case -ECANCELED:
146  DEBUG("csma: !!! DEVICE DRIVER FAILURE! TRANSMISSION ABORTED!\n");
147  /* internal driver error! */
148  return -ECANCELED;
149  default:
150  ok = (hwfeat == NETOPT_ENABLE);
151  }
152 
153  if (ok) {
154  /* device does CSMA/CA all by itself: let it do its job */
155  DEBUG("csma: Network device does hardware CSMA/CA\n");
156  return dev->driver->send(dev, vector, count);
157  }
158 
159  /* if we arrive here, then we must perform the CSMA/CA procedure
160  ourselves by software */
161  random_init(_xtimer_now());
162  DEBUG("csma: Starting software CSMA/CA....\n");
163 
164  int nb = 0, be = conf->min_be;
165 
166  while (nb <= conf->max_be) {
167  /* delay for an adequate random backoff period */
168  uint32_t bp = choose_backoff_period(be, conf);
169  xtimer_usleep(bp);
170 
171  /* try to send after a CCA */
172  res = send_if_cca(dev, vector, count);
173  if (res >= 0) {
174  /* TX done */
175  return res;
176  }
177  else if (res != -EBUSY) {
178  /* something has gone wrong, return the error code */
179  return res;
180  }
181 
182  /* medium is busy: increment CSMA counters */
183  DEBUG("csma: Radio medium busy.\n");
184  be++;
185  if (be > conf->max_be) {
186  be = conf->max_be;
187  }
188  nb++;
189  /* ... and try again if we have no exceeded the retry limit */
190  }
191 
192  /* if we arrive here, medium was never available for transmission */
193  DEBUG("csma: Software CSMA/CA failure: medium never available.\n");
194  return -EBUSY;
195 }
