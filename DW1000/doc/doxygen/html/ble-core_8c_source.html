78 ble_stack_init(void)
79 {
80  uint32_t err_code;
81 
82  // Enable BLE stack.
83  ble_enable_params_t ble_enable_params;
84  memset(&ble_enable_params, 0, sizeof(ble_enable_params));
85  ble_enable_params.gatts_enable_params.attr_tab_size =
86  BLE_GATTS_ATTR_TAB_SIZE_DEFAULT;
87  ble_enable_params.gatts_enable_params.service_changed =
88  IS_SRVC_CHANGED_CHARACT_PRESENT;
89  err_code = sd_ble_enable(&ble_enable_params);
90  APP_ERROR_CHECK(err_code);
91 
92  // Register with the SoftDevice handler module for BLE events.
93  err_code = softdevice_ble_evt_handler_set(ble_evt_dispatch);
94  APP_ERROR_CHECK(err_code);
95 
96  // Setup address
97  ble_gap_addr_t ble_addr;
98  err_code = sd_ble_gap_address_get(&ble_addr);
99  APP_ERROR_CHECK(err_code);
100 
101  ble_addr.addr[5] = 0x00;
102  ble_addr.addr_type = BLE_GAP_ADDR_TYPE_PUBLIC;
103 
104  err_code = sd_ble_gap_address_set(BLE_GAP_ADDR_CYCLE_MODE_NONE, &ble_addr);
105  APP_ERROR_CHECK(err_code);
106 }

128 void
129 ble_advertising_init(const char *name)
130 {
131  uint32_t err_code;
132  ble_advdata_t advdata;
133  uint8_t flags = BLE_GAP_ADV_FLAG_BR_EDR_NOT_SUPPORTED;
134  ble_gap_conn_sec_mode_t sec_mode;
135 
136  BLE_GAP_CONN_SEC_MODE_SET_OPEN(&sec_mode);
137 
138  err_code = sd_ble_gap_device_name_set(&sec_mode, (const uint8_t *)name,
139  strlen(name));
140  APP_ERROR_CHECK(err_code);
141 
142  ble_uuid_t adv_uuids[] = {{BLE_UUID_IPSP_SERVICE, BLE_UUID_TYPE_BLE}};
143 
144  // Build and set advertising data.
145  memset(&advdata, 0, sizeof(advdata));
146 
147  advdata.name_type = BLE_ADVDATA_FULL_NAME;
148  advdata.flags = flags;
149  advdata.uuids_complete.uuid_cnt = sizeof(adv_uuids) / sizeof(adv_uuids[0]);
150  advdata.uuids_complete.p_uuids = adv_uuids;
151 
152  err_code = ble_advdata_set(&advdata, NULL);
153  APP_ERROR_CHECK(err_code);
154 
155  // Initialize advertising parameters (used when starting advertising).
156  memset(&m_adv_params, 0, sizeof(m_adv_params));
157 
158  m_adv_params.type = BLE_GAP_ADV_TYPE_ADV_IND;
159  m_adv_params.p_peer_addr = NULL; // Undirected advertisement.
160  m_adv_params.fp = BLE_GAP_ADV_FP_ANY;
161  m_adv_params.interval = APP_ADV_ADV_INTERVAL;
162  m_adv_params.timeout = APP_ADV_TIMEOUT;
163 }

200 static void
201 on_ble_evt(ble_evt_t *p_ble_evt)
202 {
203  switch(p_ble_evt->header.evt_id) {
204  case BLE_GAP_EVT_CONNECTED:
205  DEBUG("ble-core: connected [handle:%d, peer: ", p_ble_evt->evt.gap_evt.conn_handle);
206  ble_gap_addr_print(&(p_ble_evt->evt.gap_evt.params.connected.peer_addr));
207  DEBUG("]\n");
208  sd_ble_gap_rssi_start(p_ble_evt->evt.gap_evt.conn_handle,
209  BLE_GAP_RSSI_THRESHOLD_INVALID,
210  0);
211  break;
212 
213  case BLE_GAP_EVT_DISCONNECTED:
214  DEBUG("ble-core: disconnected [handle:%d]\n", p_ble_evt->evt.gap_evt.conn_handle);
215  ble_advertising_start();
216  break;
217  default:
218  break;
219  }
220 }
