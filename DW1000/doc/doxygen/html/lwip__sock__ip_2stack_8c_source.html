143 static int _netdev_send(netdev2_t *dev, const struct iovec *vector, int count)
144 {
145  msg_t done = { .type = _SEND_DONE };
146  unsigned offset = 0;
147 
148  (void)dev;
149  mutex_lock(&_netdev_buffer_mutex);
150  for (int i = 0; i < count; i++) {
151  memcpy(&_netdev_buffer[offset], vector[i].iov_base, vector[i].iov_len);
152  offset += vector[i].iov_len;
153  if (offset > sizeof(_netdev_buffer)) {
154  mutex_unlock(&_netdev_buffer_mutex);
155  return -ENOBUFS;
156  }
157  }
158  mutex_unlock(&_netdev_buffer_mutex);
159  done.content.value = (uint32_t)offset - sizeof(ethernet_hdr_t);
160  msg_send(&done, _check_pid);
161  return offset;
162 }

315 bool _check_4packet(uint32_t src, uint32_t dst, uint8_t proto,
316  void *data, size_t data_len, uint16_t netif)
317 {
318 #if LWIP_IPV4
319  msg_t msg;
320 
321  (void)netif;
322  while (data_len != (msg.content.value - sizeof(struct ip_hdr))) {
323  msg_receive(&msg);
324  }
325  mutex_lock(&_netdev_buffer_mutex);
326  ethernet_hdr_t *eth_hdr = (ethernet_hdr_t *)_netdev_buffer;
327  struct ip_hdr *ip_hdr = (struct ip_hdr *)(eth_hdr + 1);
328  uint8_t *payload = (uint8_t *)(ip_hdr + 1);
329  uint16_t payload_len = HTONS(IPH_LEN(ip_hdr)) - sizeof(struct ip_hdr);
330  const bool ip_correct = ((src == 0) || (src = ip_hdr->src.addr)) &&
331  (dst = ip_hdr->dest.addr) &&
332  (IPH_PROTO(ip_hdr) == proto);
333  const bool payload_correct = (data_len == payload_len) &&
334  (memcmp(data, payload, data_len) == 0);
335  mutex_unlock(&_netdev_buffer_mutex);
336  return ip_correct && payload_correct;
