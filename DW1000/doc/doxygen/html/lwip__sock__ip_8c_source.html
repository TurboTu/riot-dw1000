107 static int _parse_iphdr(const struct netbuf *buf, void *data, size_t max_len,
108  sock_ip_ep_t *remote)
109 {
110  uint8_t *data_ptr = buf->p->payload;
111  size_t data_len = buf->p->len;
112 
113  assert(buf->p->next == NULL); /* TODO this might not be generally the case
114  * check later with larger payloads */
115  switch (data_ptr[0] >> 4) {
116 #if LWIP_IPV4
117  case 4:
118  if ((data_len - sizeof(struct ip_hdr)) > max_len) {
119  return -ENOBUFS;
120  }
121  if (remote != NULL) {
122  struct ip_hdr *iphdr = (struct ip_hdr *)data_ptr;
123 
124  assert(buf->p->len > sizeof(struct ip_hdr));
125  remote->family = AF_INET;
126  memcpy(&remote->addr, &iphdr->src, sizeof(ip4_addr_t));
127  remote->netif = _ip4_addr_to_netif(&iphdr->dest);
128  }
129  data_ptr += sizeof(struct ip_hdr);
130  data_len -= sizeof(struct ip_hdr);
131  break;
132 #endif
133 #if LWIP_IPV6
134  case 6:
135  if ((data_len - sizeof(struct ip6_hdr)) > max_len) {
136  return -ENOBUFS;
137  }
138  if (remote != NULL) {
139  struct ip6_hdr *iphdr = (struct ip6_hdr *)data_ptr;
140 
141  assert(buf->p->len > sizeof(struct ip6_hdr));
142  remote->family = AF_INET6;
143  memcpy(&remote->addr, &iphdr->src, sizeof(ip6_addr_t));
144  remote->netif = _ip6_addr_to_netif(&iphdr->dest);
145  }
146  data_ptr += sizeof(struct ip6_hdr);
147  data_len -= sizeof(struct ip6_hdr);
148  break;
149 #endif
150  default:
151  return -EPROTO;
152  }
153  memcpy(data, data_ptr, data_len);
154  return (ssize_t)data_len;
155 }
