36 int hdc1000_init(hdc1000_t *dev, const hdc1000_params_t *params)
37 {
38  uint8_t reg[2];
39  uint16_t tmp;
40 
41  /* write device descriptor */
42  memcpy(&dev->p, params, sizeof(hdc1000_params_t));
43 
44  /* initialize the I2C bus */
45  i2c_acquire(dev->p.i2c);
46  if (i2c_init_master(dev->p.i2c, I2C_SPEED) < 0) {
47  i2c_release(dev->p.i2c);
48  return HDC1000_NOBUS;
49  }
50 
51  /* try if we can interact with the device by reading its manufacturer ID */
52  if (i2c_read_regs(dev->p.i2c, dev->p.addr,
53  HDC1000_MANUFACTURER_ID, reg, 2) != 2) {
54  i2c_release(dev->p.i2c);
55  return HDC1000_NOBUS;
56  }
57 
58  tmp = ((uint16_t)reg[0] << 8) | reg[1];
59  if (tmp != HDC1000_MID_VALUE) {
60  i2c_release(dev->p.i2c);
61  return HDC1000_NODEV;
62  }
63 
64  /* set resolution for both sensors and sequence mode */
65  tmp = (HDC1000_SEQ_MOD | dev->p.res);
66  reg[0] = (tmp >> 8);
67  reg[1] = tmp;
68 
69  if (i2c_write_regs(dev->p.i2c, dev->p.addr, HDC1000_CONFIG, reg, 2) != 2) {
70  i2c_release(dev->p.i2c);
71  return HDC1000_NOBUS;
72  }
73  i2c_release(dev->p.i2c);
74 
75  /* all set */
76  return HDC1000_OK;
77 }

79 void hdc1000_trigger_conversion(hdc1000_t *dev)
80 {
81  assert(dev);
82 
83  i2c_acquire(dev->p.i2c);
84 
85  /* Trigger the measurements by executing a write access
86  * to the address 0x00 (HDC1000_TEMPERATURE).
87  * Conversion Time is 6.50ms for each value for 14 bit resolution.
88  */
89  i2c_write_byte(dev->p.i2c, dev->p.addr, HDC1000_TEMPERATURE);
90 
91  i2c_release(dev->p.i2c);
92 }

94 void hdc1000_get_results(hdc1000_t *dev, int16_t *temp, int16_t *hum)
95 {
96  assert(dev);
97 
98  uint8_t buf[4];
99  uint16_t traw, hraw;
100 
101  /* first we read the RAW results from the device */
102  i2c_acquire(dev->p.i2c);
103  i2c_read_bytes(dev->p.i2c, dev->p.addr, buf, 4);
104  i2c_release(dev->p.i2c);
105 
106  /* and finally we convert the values to their physical representation */
107  if (temp) {
108  traw = ((uint16_t)buf[0] << 8) | buf[1];
109  *temp = (int16_t)((((int32_t)traw * 16500) >> 16) - 4000);
110  }
111  if (hum) {
112  hraw = ((uint16_t)buf[2] << 8) | buf[3];
113  *hum = (int16_t)(((int32_t)hraw * 10000) >> 16);
114  }
115 }
