66 void spi_init(spi_t bus)
67 {
68  /* make sure given bus is good */
69  assert(bus < SPI_NUMOF);
70 
71  /* initialize the device lock */
72  mutex_init(&locks[bus]);
73 
74  /* configure pins and their muxes */
75  spi_init_pins(bus);
76 
77  /* wake up device */
78  poweron(bus);
79 
80  /* reset all device configuration */
81  dev(bus)->CTRLA.reg |= SERCOM_SPI_CTRLA_SWRST;
82  while ((dev(bus)->CTRLA.reg & SERCOM_SPI_CTRLA_SWRST) ||
83  (dev(bus)->SYNCBUSY.reg & SERCOM_SPI_SYNCBUSY_SWRST));
84 
85  /* configure base clock: using GLK GEN 0 */
86 #if defined(CPU_FAM_SAMD21)
87  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN_GCLK0 |
88  (SERCOM0_GCLK_ID_CORE + sercom_id(dev(bus))));
89  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
90 #elif defined(CPU_FAM_SAML21)
91  GCLK->PCHCTRL[SERCOM0_GCLK_ID_CORE + sercom_id(dev(bus))].reg =
92  (GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN_GCLK0);
93 #endif
94 
95  /* enable receiver and configure character size to 8-bit
96  * no synchronization needed, as SERCOM device is not enabled */
97  dev(bus)->CTRLB.reg = (SERCOM_SPI_CTRLB_CHSIZE(0) | SERCOM_SPI_CTRLB_RXEN);
98 
99  /* put device back to sleep */
100  poweroff(bus);
101 }

113 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
114 {
115  /* get exclusive access to the device */
116  mutex_lock(&locks[bus]);
117  /* power on the device */
118  poweron(bus);
119 
120  /* configure bus clock, in synchronous mode its calculated from
121  * BAUD.reg = (f_ref / (2 * f_bus) - 1)
122  * with f_ref := CLOCK_CORECLOCK as defined by the board */
123  dev(bus)->BAUD.reg = (uint8_t)(((uint32_t)CLOCK_CORECLOCK) / (2 * clk) - 1);
124 
125  /* configure device to be master and set mode and pads,
126  *
127  * NOTE: we could configure the pads already during spi_init, but for
128  * efficiency reason we do that here, so we can do all in one single write
129  * to the CTRLA register */
130  dev(bus)->CTRLA.reg = (SERCOM_SPI_CTRLA_MODE(0x3) | /* 0x3 -> master */
131  SERCOM_SPI_CTRLA_DOPO(spi_config[bus].mosi_pad) |
132  SERCOM_SPI_CTRLA_DIPO(spi_config[bus].miso_pad) |
133  (mode << SERCOM_SPI_CTRLA_CPOL_Pos));
134  /* also no synchronization needed here, as CTRLA is write-synchronized */
135 
136  /* finally enable the device */
137  dev(bus)->CTRLA.reg |= SERCOM_SPI_CTRLA_ENABLE;
138  while (dev(bus)->SYNCBUSY.reg & SERCOM_SPI_SYNCBUSY_ENABLE) {}
139 
140  return SPI_OK;
141 }

143 void spi_release(spi_t bus)
144 {
145  /* disable device and put it back to sleep */
146  dev(bus)->CTRLA.reg &= ~(SERCOM_SPI_CTRLA_ENABLE);
147  while (dev(bus)->SYNCBUSY.reg & SERCOM_SPI_SYNCBUSY_ENABLE) {}
148  poweroff(bus);
149  /* release access to the device */
150  mutex_unlock(&locks[bus]);
151 }

153 void spi_transfer_bytes(spi_t bus, spi_cs_t cs, bool cont,
154  const void *out, void *in, size_t len)
155 {
156  uint8_t *out_buf = (uint8_t *)out;
157  uint8_t *in_buf = (uint8_t *)in;
158 
159  assert(out || in);
160 
161  if (cs != SPI_CS_UNDEF) {
162  gpio_clear((gpio_t)cs);
163  }
164 
165  for (int i = 0; i < (int)len; i++) {
166  uint8_t tmp = (out_buf) ? out_buf[i] : 0;
167  while (!(dev(bus)->INTFLAG.reg & SERCOM_SPI_INTFLAG_DRE)) {}
168  dev(bus)->DATA.reg = tmp;
169  while (!(dev(bus)->INTFLAG.reg & SERCOM_SPI_INTFLAG_RXC)) {}
170  tmp = (uint8_t)dev(bus)->DATA.reg;
171  if (in_buf) {
172  in_buf[i] = tmp;
173  }
174  }
175 
176  if ((!cont) && (cs != SPI_CS_UNDEF)) {
177  gpio_set((gpio_t)cs);
178  }
179 }
