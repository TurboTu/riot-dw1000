118 static int _transition_to(gnrc_tcp_tcb_t* tcb, gnrc_tcp_fsm_state_t state, bool *notify_owner)
119 {
120  gnrc_tcp_tcb_t *iter = NULL;
121  uint8_t found = 0;
122 
123  switch (state) {
124  case GNRC_TCP_FSM_STATE_CLOSED:
125  /* Free Packets in Retransmit queue */
126  _clear_retransmit(tcb);
127 
128  /* Remove from Connection from active connections */
129  mutex_lock(&_list_gnrc_tcp_tcb_lock);
130  LL_FOREACH(_list_gnrc_tcp_tcb_head, iter) {
131  if (iter == tcb) {
132  found = 1;
133  }
134  }
135  if (found) {
136  LL_DELETE(_list_gnrc_tcp_tcb_head, iter);
137  }
138  mutex_unlock(&_list_gnrc_tcp_tcb_lock);
139 
140  /* Free potencially allocated Receive Buffer */
141  _rcvbuf_release_buffer(tcb);
142  *notify_owner = true;
143  break;
144 
145  case GNRC_TCP_FSM_STATE_LISTEN:
146  /* Clear Adress Info */
147  switch (tcb->address_family) {
148 #ifdef MODULE_GNRC_IPV6
149  case AF_INET6:
150  if (tcb->status & GNRC_TCP_STATUS_ALLOW_ANY_ADDR) {
151  ipv6_addr_set_unspecified((ipv6_addr_t *) tcb->local_addr);
152  }
153  ipv6_addr_set_unspecified((ipv6_addr_t *) tcb->peer_addr);
154  break;
155 #endif
156  default:
157  DEBUG("gnrc_tcp_fsm.c : _transition_to() : Undefined Addresses\n");
158  break;
159  }
160  tcb->peer_port = GNRC_TCP_PORT_UNSPEC;
161 
162  /* Allocate rcv Buffer */
163  if (_rcvbuf_get_buffer(tcb) == -ENOMEM) {
164  return -ENOMEM;
165  }
166 
167  /* Add to Connection to active connections (if not already active) */
168  mutex_lock(&_list_gnrc_tcp_tcb_lock);
169  LL_FOREACH(_list_gnrc_tcp_tcb_head, iter) {
170  if (iter == tcb) {
171  found = 1;
172  }
173  }
174  if (!found) {
175  LL_APPEND(_list_gnrc_tcp_tcb_head, tcb);
176  }
177  mutex_unlock(&_list_gnrc_tcp_tcb_lock);
178  break;
179 
180  case GNRC_TCP_FSM_STATE_SYN_SENT:
181  /* Allocate rcv Buffer */
182  if (_rcvbuf_get_buffer(tcb) == -ENOMEM) {
183  return -ENOMEM;
184  }
185 
186  /* Add to Connections to active connection (if not already active) */
187  mutex_lock(&_list_gnrc_tcp_tcb_lock);
188  LL_FOREACH(_list_gnrc_tcp_tcb_head, iter) {
189  if (iter == tcb) {
190  found = 1;
191  }
192  }
193  /* If not already active: Apped tcb but check portnumber first */
194  if (!found) {
195  /* Check if Port Number is not in use */
196  if (tcb->local_port != GNRC_TCP_PORT_UNSPEC ) {
197 
198  /* If Portnumber is used: return error and release buffer */
199  if (_is_local_port_in_use(tcb->local_port)) {
200  mutex_unlock(&_list_gnrc_tcp_tcb_lock);
201  _rcvbuf_release_buffer(tcb);
202  return -EADDRINUSE;
203  }
204  }
205  /* Pick Random Port */
206  else {
207  tcb->local_port = _get_random_local_port();
208  }
209  LL_APPEND(_list_gnrc_tcp_tcb_head, tcb);
210  }
211  mutex_unlock(&_list_gnrc_tcp_tcb_lock);
212  break;
213 
214  case GNRC_TCP_FSM_STATE_ESTABLISHED:
215  *notify_owner = true;
216  break;
217 
218  case GNRC_TCP_FSM_STATE_CLOSE_WAIT:
219  *notify_owner = true;
220  break;
221 
222  case GNRC_TCP_FSM_STATE_TIME_WAIT:
223  _restart_timewait_timer(tcb);
224  break;
225 
226  default:
227  break;
228  }
229  tcb->state = state;
230  return 0;
231 }

243 static int _fsm_call_open(gnrc_tcp_tcb_t* tcb, bool *notify_owner)
244 {
245  gnrc_pktsnip_t *out_pkt = NULL; /* Outgoing packet */
246  uint16_t seq_con = 0; /* Sequence number consumption (out_pkt) */
247  int ret = 0; /* Return value */
248 
249  DEBUG("gnrc_tcp_fsm.c : _fsm_call_open()\n");
250  tcb->rcv_wnd = GNRC_TCP_DEFAULT_WINDOW;
251 
252  if (tcb->status & GNRC_TCP_STATUS_PASSIVE) {
253  /* Passive Open, T: CLOSED -> LISTEN */
254  if (_transition_to(tcb, GNRC_TCP_FSM_STATE_LISTEN, notify_owner) == -ENOMEM){
255  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
256  return -ENOMEM;
257  }
258  }
259  else {
260  /* Active Open, init tcb values, send SYN, T: CLOSED -> SYN_SENT */
261  tcb->iss = random_uint32();
262  tcb->snd_nxt = tcb->iss;
263  tcb->snd_una = tcb->iss;
264 
265  /* Translate to SYN_SENT */
266  ret = _transition_to(tcb, GNRC_TCP_FSM_STATE_SYN_SENT, notify_owner);
267  if ( ret < 0) {
268  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
269  return ret;
270  }
271 
272  /* Send SYN */
273  _pkt_build(tcb, &out_pkt, &seq_con, MSK_SYN, tcb->iss, 0, NULL, 0);
274  _pkt_setup_retransmit(tcb, out_pkt, false);
275  _pkt_send(tcb, out_pkt, seq_con, false);
276  }
277  return ret;
278 }

289 static int _fsm_call_send(gnrc_tcp_tcb_t* tcb, void *buf, size_t nByte)
290 {
291  gnrc_pktsnip_t *out_pkt = NULL; /* Outgoing packet */
292  uint16_t seq_con = 0; /* Sequence number consumption (out_pkt) */
293 
294  DEBUG("gnrc_tcp_fsm.c : _fsm_call_send()\n");
295  size_t payload = (tcb->snd_una + tcb->snd_wnd) - tcb->snd_nxt;
296 
297  /* We are allowed to send further bytes if window is open */
298  if (payload > 0 && tcb->snd_wnd > 0 && tcb->pkt_retransmit == NULL) {
299  /* Calculate segment size */
300  payload = (payload < GNRC_TCP_MSS ? payload : GNRC_TCP_MSS);
301  payload = (payload < tcb->mss) ? payload : tcb->mss;
302  payload = (payload < nByte) ? payload : nByte;
303 
304  /* Calculate payload size for this segment */
305  _pkt_build(tcb, &out_pkt, &seq_con, MSK_ACK, tcb->snd_nxt, tcb->rcv_nxt, buf, payload);
306  _pkt_setup_retransmit(tcb, out_pkt, false);
307  _pkt_send(tcb, out_pkt, seq_con, false);
308  return payload;
309  }
310  return 0;
311 }

322 static int _fsm_call_recv(gnrc_tcp_tcb_t* tcb, void *buf, size_t nByte)
323 {
324  gnrc_pktsnip_t *out_pkt = NULL; /* Outgoing packet */
325  uint16_t seq_con = 0; /* Sequence number consumption (out_pkt) */
326 
327  DEBUG("gnrc_tcp_fsm.c : _fsm_call_recv()\n");
328  if (ringbuffer_empty(&tcb->rcv_buf)) {
329  return 0;
330  }
331 
332  /* Read up to the requesed amount of data */
333  size_t rcvd = ringbuffer_get(&(tcb->rcv_buf), buf, nByte);
334 
335  /* If the buffer can store more than the GNRC_TCP_MSS: open Window to available buffersize */
336  if (ringbuffer_get_free(&tcb->rcv_buf) >= GNRC_TCP_MSS) {
337  tcb->rcv_wnd = ringbuffer_get_free(&(tcb->rcv_buf));
338 
339  /* Send ACK to update window on reopening */

341  _pkt_send(tcb, out_pkt, seq_con, false);
342  }
343  return rcvd;
344 }

354 static int _fsm_call_close(gnrc_tcp_tcb_t* tcb, bool *notify_owner)
355 {
356  gnrc_pktsnip_t *out_pkt = NULL; /* Outgoing packet */
357  uint16_t seq_con = 0; /* Sequence number consumption (out_pkt) */
358 
359  DEBUG("gnrc_tcp_fsm.c : _fsm_call_close()\n");
360  if (tcb->state == GNRC_TCP_FSM_STATE_SYN_RCVD
361  || tcb->state == GNRC_TCP_FSM_STATE_ESTABLISHED
362  || tcb->state == GNRC_TCP_FSM_STATE_CLOSE_WAIT
363  ) {
364  /* Send FIN packet */
365  _pkt_build(tcb, &out_pkt, &seq_con, MSK_FIN_ACK, tcb->snd_nxt, tcb->rcv_nxt, NULL, 0);
366  _pkt_setup_retransmit(tcb, out_pkt, false);
367  _pkt_send(tcb, out_pkt, seq_con, false);
368  }
369  switch (tcb->state) {
370  case GNRC_TCP_FSM_STATE_LISTEN:
371  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
372  break;
373 
374  case GNRC_TCP_FSM_STATE_SYN_RCVD:
375  case GNRC_TCP_FSM_STATE_ESTABLISHED:
376  _transition_to(tcb, GNRC_TCP_FSM_STATE_FIN_WAIT_1, notify_owner);
377  break;
378 
379  case GNRC_TCP_FSM_STATE_CLOSE_WAIT:
380  _transition_to(tcb, GNRC_TCP_FSM_STATE_LAST_ACK, notify_owner);
381  break;
382 
383  default:
384  break;
385  }
386  return 0;
387 }

411 static int _fsm_rcvd_pkt(gnrc_tcp_tcb_t* tcb, gnrc_pktsnip_t *in_pkt, bool *notify_owner)
412 {
413  gnrc_pktsnip_t *out_pkt = NULL; /* Outgoing packet */
414  uint16_t seq_con = 0; /* Sequence number consumption (out_pkt) */
415  gnrc_pktsnip_t *snp = NULL; /* Temporary Packet Snip */
416  gnrc_tcp_tcb_t *lst = NULL; /* Temporary tcb pointer */
417  uint16_t ctl = 0; /* Received control bits */
418  uint32_t seg_seq = 0; /* Received sequence number */
419  uint32_t seg_ack = 0; /* Received acknowledgment number */
420  uint32_t seg_len = 0; /* Segment length */
421  uint32_t pay_len = 0; /* Payload length */
422  uint32_t seg_wnd = 0; /* Segment window */
423 
424  DEBUG("gnrc_tcp_fsm.c : _fsm_rcvd_pkt()\n");
425  /* Search TCP header. */
426  LL_SEARCH_SCALAR(in_pkt, snp, type, GNRC_NETTYPE_TCP);
427  tcp_hdr_t *tcp_hdr = (tcp_hdr_t *) snp->data;
428 
429  /* Verify packet options, return if they were faulty */
430  if (_option_parse(tcb, tcp_hdr) < 0) {
431  return 0;
432  }
433 
434  /* Extract header values */
435  ctl = byteorder_ntohs(tcp_hdr->off_ctl);
436  seg_seq = byteorder_ntohl(tcp_hdr->seq_num);
437  seg_ack = byteorder_ntohl(tcp_hdr->ack_num);
438  seg_wnd = byteorder_ntohs(tcp_hdr->window);
439 
440  /* Extract IPv6-Header */
441 #ifdef MODULE_GNRC_IPV6
442  LL_SEARCH_SCALAR(in_pkt, snp, type, GNRC_NETTYPE_IPV6);
443  if (snp == NULL) {
444  DEBUG("gnrc_tcp_fsm.c : _fsm_rcvd_pkt() : incomming packet had no ip header\n");
445  return 0;
446  }
447  void *ip = snp->data;
448 #endif
449 
450  /* Handle state LISTEN */
451  if (tcb->state == GNRC_TCP_FSM_STATE_LISTEN) {
452  /* 1) Check RST: if set, return */
453  if (ctl & MSK_RST) {
454  return 0;
455  }
456  /* 2) Check ACK: if set, send reset with seq_no = ack_no, return */
457  if (ctl & MSK_ACK) {
458  _pkt_build_reset_from_pkt(&out_pkt, in_pkt);
459  _pkt_send(tcb, out_pkt, 0, false);
460  return 0;
461  }
462  /* 3) Check SYN: Setup incoming connection*/
463  if (ctl & MSK_SYN) {
464  uint16_t src = byteorder_ntohs(tcp_hdr->src_port);
465  uint16_t dst = byteorder_ntohs(tcp_hdr->dst_port);
466 
467  /* Check if SYN Request is handled by another connection */
468  lst = _list_gnrc_tcp_tcb_head;
469  while (lst) {
470  /* Compare Portnumbers and Network Layer Adresses */
471  /* Note: Packets without ip-header were discarded earlier */
472  if (lst->local_port == dst && lst->peer_port == src) {
473 #ifdef MODULE_GNRC_IPV6
474  if (snp->type == GNRC_NETTYPE_IPV6 && lst->address_family == AF_INET6) {
475  ipv6_addr_t *dst_addr = &((ipv6_hdr_t *)ip)->dst;
476  ipv6_addr_t *src_addr = &((ipv6_hdr_t *)ip)->src;
477 
478  if (ipv6_addr_equal((ipv6_addr_t *)lst->local_addr, dst_addr)
479  && ipv6_addr_equal((ipv6_addr_t *)lst->peer_addr, src_addr)
480  ) {
481  break;
482  }
483  }
484 #endif
485  }
486  lst = lst->next;
487  }
488  /* Return if connection is already handled (port and addresses match) */
489  if (lst != NULL) {
490  DEBUG("gnrc_tcp_fsm.c : _fsm_rcvd_pkt() : Connection already handled\n");
491  return 0;
492  }
493 
494  /* SYN Request is valid, fill connection struct with connection information */
495  /* Note: Packets without ipv6-header were discarded earlier */
496 #ifdef MODULE_GNRC_IPV6
497  if (snp->type == GNRC_NETTYPE_IPV6 && tcb->address_family == AF_INET6) {
498  memcpy(tcb->local_addr, &((ipv6_hdr_t *)ip)->dst, sizeof(ipv6_addr_t));
499  memcpy(tcb->peer_addr, &((ipv6_hdr_t *)ip)->src, sizeof(ipv6_addr_t));
500  }
501 #else
502  DEBUG("gnrc_tcp_fsm.c : _fsm_rcvd_pkt() : Received Address was not stored\n");
503  return 0;
504 #endif
505 
506  tcb->local_port = dst;
507  tcb->peer_port = src;
508  tcb->irs = byteorder_ntohl(tcp_hdr->seq_num);
509  tcb->rcv_nxt = tcb->irs + 1;
510  tcb->iss = random_uint32();
511  tcb->snd_una = tcb->iss;
512  tcb->snd_nxt = tcb->iss;
513  tcb->snd_wnd = seg_wnd;
514 
515  /* Send SYN+ACK: seq_no = iss, ack_no = rcv_nxt, T: LISTEN -> SYN_RCVD */
516  _pkt_build(tcb, &out_pkt, &seq_con, MSK_SYN_ACK, tcb->iss, tcb->rcv_nxt, NULL, 0);
517  _pkt_setup_retransmit(tcb, out_pkt, false);
518  _pkt_send(tcb, out_pkt, seq_con, false);
519  _transition_to(tcb, GNRC_TCP_FSM_STATE_SYN_RCVD, notify_owner);
520  }
521  return 0;
522  }
523  /* Handle state SYN_SENT */
524  else if (tcb->state == GNRC_TCP_FSM_STATE_SYN_SENT) {
525  /* 1) Check ACK */
526  if (ctl & MSK_ACK) {
527  /* If ACK is not acceptable ...*/
528  if (seg_ack <= tcb->iss || seg_ack > tcb->snd_nxt) {
529  /* ... send Reset if RST is not set else return */
530  if ((ctl & MSK_RST) != MSK_RST) {
531  _pkt_build(tcb, &out_pkt, &seq_con, MSK_RST, seg_ack, 0, NULL, 0);
532  _pkt_send(tcb, out_pkt, seq_con, false);
533  }
534  return 0;
535  }
536  }
537  /* 2) Check RST: If RST set ... */
538  if (ctl & MSK_RST) {
539  /* ... and ACK: Translate to CLOSED, if not return */
540  if (ctl & MSK_ACK) {
541  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
542  }
543  return 0;
544  }
545  /* 3) Check SYN: Set TCB values accordingly */
546  if (ctl & MSK_SYN) {
547  tcb->rcv_nxt = seg_seq + 1;
548  tcb->irs = seg_seq;
549  if (ctl & MSK_ACK) {
550  tcb->snd_una = seg_ack;
551  _pkt_acknowledge(tcb, seg_ack);
552  }
553  /* Set the local address accordingly */
554  /* Note: Packets without ipv6-header were discarded earlier */
555 #ifdef MODULE_GNRC_IPV6
556  if (snp->type == GNRC_NETTYPE_IPV6 && tcb->address_family == AF_INET6) {
557  memcpy(tcb->local_addr, &((ipv6_hdr_t *)ip)->dst, sizeof(ipv6_addr_t));
558  }
559 #else
560  DEBUG("gnrc_tcp_fsm.c : _fsm_rcvd_pkt() : Received Address was not stored\n");
561  return 0;
562 #endif
563 
564  /* SYN has been ACKed, reply pure ACK, T: SYN_SENT -> ESTABLISHED */
565  if (tcb->snd_una > tcb->iss) {
566  _pkt_build(tcb, &out_pkt, &seq_con, MSK_ACK, tcb->snd_nxt, tcb->rcv_nxt,
567  NULL, 0);
568  _pkt_send(tcb, out_pkt, seq_con, false);
569  _transition_to(tcb, GNRC_TCP_FSM_STATE_ESTABLISHED, notify_owner);
570  }
571  /* Simultaneous SYN received send SYN+ACK, T: SYN_SENT -> SYN_RCVD */
572  else {
573  _pkt_build(tcb, &out_pkt, &seq_con, MSK_SYN_ACK, tcb->iss, tcb->rcv_nxt,
574  NULL, 0);
575  _pkt_setup_retransmit(tcb, out_pkt, false);
576  _pkt_send(tcb, out_pkt, seq_con, false);
577  _transition_to(tcb, GNRC_TCP_FSM_STATE_SYN_RCVD, notify_owner);
578  }
579  tcb->snd_wnd = seg_wnd;
580  tcb->snd_wl1 = seg_seq;
581  tcb->snd_wl2 = seg_ack;
582  }
583  return 0;
584  }
585  /* Handle other states */
586  else {
587  seg_len = _pkt_get_seg_len(in_pkt);
588  pay_len = _pkt_get_pay_len(in_pkt);
589  /* 1) Verify Sequence Number ... */
590  if (!_pkt_chk_seq_num(tcb, seg_seq, pay_len)) {
591  /* ... if invalid, and RST not set, reply with pure ACK, return */
592  if ((ctl & MSK_RST) != MSK_RST) {
593  _pkt_build(tcb, &out_pkt, &seq_con, MSK_ACK, tcb->snd_nxt, tcb->rcv_nxt,
594  NULL, 0);
595  _pkt_send(tcb, out_pkt, seq_con, false);
596  }
597  return 0;
598  }
599  /* 2) Check RST: If RST is set ... */
600  if (ctl & MSK_RST) {
601  /* .. and State is SYN_RCVD and passive Open: SYN_RCVD -> LISTEN */
602  if (tcb->state == GNRC_TCP_FSM_STATE_SYN_RCVD
603  && (tcb->status & GNRC_TCP_STATUS_PASSIVE)
604  ) {
605  if (_transition_to(tcb, GNRC_TCP_FSM_STATE_LISTEN, notify_owner) == -ENOMEM) {
606  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
607  return -ENOMEM;
608  }
609  }
610  else {
611  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
612  }
613  return 0;
614  }
615  /* 3) Check SYN: If SYN is set ... */
616  if (ctl & MSK_SYN) {
617  /* ... send RST, seq_no = snd_nxt, ack_no = rcv_nxt */
618  _pkt_build(tcb, &out_pkt, &seq_con, MSK_RST, tcb->snd_nxt, tcb->rcv_nxt, NULL, 0);
619  _pkt_send(tcb, out_pkt, seq_con, false);
620  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
621  return 0;
622  }
623  /* 4) Check ACK */
624  if (!(ctl & MSK_ACK)) {
625  return 0;
626  }
627  else {
628  if (tcb->state == GNRC_TCP_FSM_STATE_SYN_RCVD) {
629  if (LSS_32_BIT(tcb->snd_una, seg_ack) && LEQ_32_BIT(seg_ack, tcb->snd_nxt)) {
630  tcb->snd_wnd = seg_wnd;
631  tcb->snd_wl1 = seg_seq;
632  tcb->snd_wl2 = seg_ack;
633  _transition_to(tcb, GNRC_TCP_FSM_STATE_ESTABLISHED, notify_owner);
634  }
635  else {
636  _pkt_build(tcb, &out_pkt, &seq_con, MSK_RST, seg_ack, 0, NULL, 0);
637  _pkt_send(tcb, out_pkt, seq_con, false);
638  }
639  }
640  /* Acknowledgment processing */
641  if (tcb->state == GNRC_TCP_FSM_STATE_ESTABLISHED
642  || tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_1
643  || tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_2
644  || tcb->state == GNRC_TCP_FSM_STATE_CLOSE_WAIT
645  || tcb->state == GNRC_TCP_FSM_STATE_CLOSING
646  || tcb->state == GNRC_TCP_FSM_STATE_LAST_ACK
647  ) {
648  /* Sent data has been acknowledged */
649  if (LSS_32_BIT(tcb->snd_una, seg_ack) && LEQ_32_BIT(seg_ack, tcb->snd_nxt)) {
650  tcb->snd_una = seg_ack;
651  _pkt_acknowledge(tcb, seg_ack);
652  }
653  /* ACK received for something not yet sent: Reply with pure ACK */
654  else if (LSS_32_BIT(tcb->snd_nxt, seg_ack)) {
655  _pkt_build(tcb, &out_pkt, &seq_con, MSK_ACK, tcb->snd_nxt, tcb->rcv_nxt,
656  NULL, 0);
657  _pkt_send(tcb, out_pkt, seq_con, false);
658  return 0;
659  }
660  /* Update Window */
661  if (LEQ_32_BIT(tcb->snd_una, seg_ack) && LEQ_32_BIT(seg_ack, tcb->snd_nxt)) {
662  if (LSS_32_BIT(tcb->snd_wl1, seg_seq) || (tcb->snd_wl1 == seg_seq
663  && LEQ_32_BIT(tcb->snd_wl2, seg_ack))
664  ) {
665  tcb->snd_wnd = seg_wnd;
666  tcb->snd_wl1 = seg_seq;
667  tcb->snd_wl2 = seg_ack;
668 
669  /* Signal User after Window Update */
670  *notify_owner = true;
671  }
672  }
673  /* Additional processing */
674  /* Check additionaly if previous our sent FIN has been acknowledged */
675  if (tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_1) {
676  if (tcb->pkt_retransmit == NULL) {
677  _transition_to(tcb, GNRC_TCP_FSM_STATE_FIN_WAIT_2, notify_owner);
678  }
679  }
680  /* If retransmission queue is empty, acknowledge close operation */
681  if (tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_2) {
682  if (tcb->pkt_retransmit == NULL) {
683  /* Optional: Unblock user close operation */
684  }
685  }
686  /* If our FIN has been acknowledged: Translate to TIME_WAIT */
687  if (tcb->state == GNRC_TCP_FSM_STATE_CLOSING) {
688  if (tcb->pkt_retransmit == NULL) {
689  _transition_to(tcb, GNRC_TCP_FSM_STATE_TIME_WAIT, notify_owner);
690  }
691  }
692  /* If our FIN has been acknowledged: last ACK received, close connection */
693  if (tcb->state == GNRC_TCP_FSM_STATE_LAST_ACK) {
694  if (tcb->pkt_retransmit == NULL) {
695  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSED, notify_owner);
696  return 0;
697  }
698  }
699  }
700  }
701  /* 5) Check URG */
702  /* NOTE: Add Urgent Pointer Processing here ... */
703  /* 6) Process Payload, if existing */
704  if (pay_len > 0) {
705  /* Check if State is valid */
706  if (tcb->state == GNRC_TCP_FSM_STATE_ESTABLISHED
707  || tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_1
708  || tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_2
709  ) {
710  /* Search for begin of payload "chain" */
711  LL_SEARCH_SCALAR(in_pkt, snp, type, GNRC_NETTYPE_UNDEF);
712 
713  /* Add only Data that is expected, to be received */
714  if (tcb->rcv_nxt == seg_seq) {
715  /* Copy contents in to buffer */
716  while (snp && snp->type == GNRC_NETTYPE_UNDEF) {
717  tcb->rcv_nxt += ringbuffer_add(&(tcb->rcv_buf), snp->data, snp->size);
718  snp = snp->next;
719  }
720  /* Shrink Receive Window */
721  tcb->rcv_wnd = ringbuffer_get_free(&(tcb->rcv_buf));
722  /* Notify Owner because new data is available */
723  *notify_owner = true;
724  }
725  /* Send pure ACK, if FIN doesn't this already */
726  /* NOTE: this is the place to add piggybagging in the future */
727  if (!(ctl & MSK_FIN)) {
728  _pkt_build(tcb, &out_pkt, &seq_con, MSK_ACK, tcb->snd_nxt, tcb->rcv_nxt,
729  NULL, 0);
730  _pkt_send(tcb, out_pkt, seq_con, false);
731  }
732  }
733  }
734  /* 7) Check FIN */
735  if (ctl & MSK_FIN) {
736  if (tcb->state == GNRC_TCP_FSM_STATE_CLOSED
737  || tcb->state == GNRC_TCP_FSM_STATE_LISTEN
738  || tcb->state == GNRC_TCP_FSM_STATE_SYN_SENT
739  ) {
740  return 0;
741  }
742  /* Advance rcv_nxt over FIN bit. */
743  tcb->rcv_nxt = seg_seq + seg_len;
744  _pkt_build(tcb, &out_pkt, &seq_con, MSK_ACK, tcb->snd_nxt, tcb->rcv_nxt, NULL, 0);
745  _pkt_send(tcb, out_pkt, seq_con, false);
746 
747  if (tcb->state == GNRC_TCP_FSM_STATE_SYN_RCVD
748  || tcb->state == GNRC_TCP_FSM_STATE_ESTABLISHED
749  ) {
750  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSE_WAIT, notify_owner);
751  }
752  else if (tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_1) {
753  if (tcb->pkt_retransmit == NULL) {
754  _transition_to(tcb, GNRC_TCP_FSM_STATE_TIME_WAIT, notify_owner);
755  }
756  else {
757  _transition_to(tcb, GNRC_TCP_FSM_STATE_CLOSING, notify_owner);
758  }
759  }
760  else if (tcb->state == GNRC_TCP_FSM_STATE_FIN_WAIT_2) {
761  _transition_to(tcb, GNRC_TCP_FSM_STATE_TIME_WAIT, notify_owner);
762  }
763  else if (tcb->state == GNRC_TCP_FSM_STATE_TIME_WAIT) {
764  _restart_timewait_timer(tcb);
765  }
766  }
767  }
768  return 0;
769 }
