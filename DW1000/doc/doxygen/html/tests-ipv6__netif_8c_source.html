192 {
193  /* Tests for possible duplicates as described in http://github.com/RIOT-OS/RIOT/issues/2965 */
194  ipv6_addr_t *entry_1;
195  ipv6_addr_t *entry_2;
196 
197  ipv6_addr_t default_addr = DEFAULT_TEST_IPV6_ADDR;
198  ipv6_addr_t sol_addr;
199 
200  test_ipv6_netif_add__success(); /* adds DEFAULT_TEST_NETIF as interface */
201 
202  /* add addresses to the interface */
203  TEST_ASSERT_NOT_NULL((entry_1 = gnrc_ipv6_netif_add_addr(DEFAULT_TEST_NETIF, &default_addr,
204  DEFAULT_TEST_PREFIX_LEN, 0)));
205 
206  /* remove default_addr, but not the corresponding solicited-node addr that
207  * came with it */
208  gnrc_ipv6_netif_remove_addr(DEFAULT_TEST_NETIF, &default_addr);
209 
210  /* create and re-add corresponding solicited node address and check that it
211  * hasn't taken the old place of default_addr*/
212  ipv6_addr_set_solicited_nodes(&sol_addr, &default_addr);
213  TEST_ASSERT_NOT_NULL((entry_2 = gnrc_ipv6_netif_add_addr(DEFAULT_TEST_NETIF, &sol_addr,
214  DEFAULT_TEST_PREFIX_LEN, 0)));
215 
216  TEST_ASSERT(entry_1 != entry_2);
217 }
