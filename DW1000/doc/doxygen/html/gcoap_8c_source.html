63 /* Event/Message loop for gcoap _pid thread. */
64 static void *_event_loop(void *arg)
65 {
66  msg_t msg_rcvd, msg_queue[GCOAP_MSG_QUEUE_SIZE];
67  (void)arg;
68 
69  msg_init_queue(msg_queue, GCOAP_MSG_QUEUE_SIZE);
70 
71  sock_udp_ep_t local;
72  memset(&local, 0, sizeof(sock_udp_ep_t));
73  local.family = AF_INET6;
74  local.netif = SOCK_ADDR_ANY_NETIF;
75  local.port = GCOAP_PORT;
76 
77  int res = sock_udp_create(&_sock, &local, NULL, 0);
78  if (res < 0) {
79  DEBUG("gcoap: cannot create sock: %d\n", res);
80  return 0;
81  }
82 
83  while(1) {
84  res = msg_try_receive(&msg_rcvd);
85 
86  if (res > 0) {
87  switch (msg_rcvd.type) {
88  case GCOAP_MSG_TYPE_TIMEOUT:
89  _expire_request((gcoap_request_memo_t *)msg_rcvd.content.ptr);
90  break;
91  case GCOAP_MSG_TYPE_INTR:
92  /* next _listen() timeout will account for open requests */
93  break;
94  default:
95  break;
96  }
97  }
98 
99  _listen(&_sock);
100  }
101 
102  return 0;
103 }

105 /* Listen for an incoming CoAP message. */
106 static void _listen(sock_udp_t *sock)
107 {
108  coap_pkt_t pdu;
109  uint8_t buf[GCOAP_PDU_BUF_SIZE];
110  sock_udp_ep_t remote;
111  gcoap_request_memo_t *memo = NULL;
112  uint8_t open_reqs;
113 
114  gcoap_op_state(&open_reqs);
115 
116  ssize_t res = sock_udp_recv(sock, buf, sizeof(buf),
117  open_reqs > 0 ? GCOAP_RECV_TIMEOUT : SOCK_NO_TIMEOUT,
118  &remote);
119  if (res <= 0) {
120 #if ENABLE_DEBUG
121  if (res < 0 && res != -ETIMEDOUT) {
122  DEBUG("gcoap: udp recv failure: %d\n", res);
123  }
124 #endif
125  return;
126  }
127 
128  res = coap_parse(&pdu, buf, res);
129  if (res < 0) {
130  DEBUG("gcoap: parse failure: %d\n", res);
131  /* If a response, can't clear memo, but it will timeout later. */
132  return;
133  }
134 
135  /* incoming request */
136  if (coap_get_code_class(&pdu) == COAP_CLASS_REQ) {
137  size_t pdu_len = _handle_req(&pdu, buf, sizeof(buf));
138  if (pdu_len > 0) {
139  sock_udp_send(sock, buf, pdu_len, &remote);
140  }
141  }
142  /* incoming response */
143  else {
144  _find_req_memo(&memo, &pdu, buf, sizeof(buf));
145  if (memo) {
146  xtimer_remove(&memo->response_timer);
147  memo->resp_handler(memo->state, &pdu);
148  memo->state = GCOAP_MEMO_UNUSED;
149  }
150  }
151 }

153 /*
154  * Main request handler: generates response PDU in the provided buffer.
155  *
156  * Caller must finish the PDU and send it.
157  */
158 static size_t _handle_req(coap_pkt_t *pdu, uint8_t *buf, size_t len)
159 {
160  unsigned method_flag = coap_method2flag(coap_get_code_detail(pdu));
161 
162  /* Find path for CoAP msg among listener resources and execute callback. */
163  gcoap_listener_t *listener = _coap_state.listeners;
164  while (listener) {
165  coap_resource_t *resource = listener->resources;
166  for (size_t i = 0; i < listener->resources_len; i++) {
167  if (i) {
168  resource++;
169  }
170  if (! (resource->methods & method_flag)) {
171  continue;
172  }
173 
174  int res = strcmp((char *)&pdu->url[0], resource->path);
175  if (res > 0) {
176  continue;
177  }
178  else if (res < 0) {
179  /* resources expected in alphabetical order */
180  break;
181  }
182  else {
183  ssize_t pdu_len = resource->handler(pdu, buf, len);
184  if (pdu_len < 0) {
185  pdu_len = gcoap_response(pdu, buf, len,
186  COAP_CODE_INTERNAL_SERVER_ERROR);
187  }
188  return pdu_len;
189  }
190  }
191  listener = listener->next;
192  }
193  /* resource not found */
194  return gcoap_response(pdu, buf, len, COAP_CODE_PATH_NOT_FOUND);
195 }

197 /*
198  * Finishes handling a PDU -- write options and reposition payload.
199  *
200  * Returns the size of the PDU within the buffer, or < 0 on error.
201  */
202 static ssize_t _finish_pdu(coap_pkt_t *pdu, uint8_t *buf, size_t len)
203 {
204  ssize_t hdr_len = _write_options(pdu, buf, len);
205  DEBUG("gcoap: header length: %u\n", hdr_len);
206 
207  if (hdr_len > 0) {
208  /* move payload over unused space after options */
209  if (pdu->payload_len) {
210  memmove(buf + hdr_len, pdu->payload, pdu->payload_len);
211  }
212 
213  return hdr_len + pdu->payload_len;
214  }
215  else {
216  return -1; /* generic failure code */
217  }
218 }

220 /*
221  * Finds the memo for an outstanding request within the _coap_state.open_reqs
222  * array. Matches on token.
223  *
224  * src_pdu Source for the match token
225  */
226 static void _find_req_memo(gcoap_request_memo_t **memo_ptr, coap_pkt_t *src_pdu,
227  uint8_t *buf, size_t len)
228 {
229  gcoap_request_memo_t *memo;
230  coap_pkt_t memo_pdu = { .token = NULL };
231  (void) buf;
232  (void) len;
233 
234  for (int i = 0; i < GCOAP_REQ_WAITING_MAX; i++) {
235  if (_coap_state.open_reqs[i].state == GCOAP_MEMO_UNUSED)
236  continue;
237 
238  /* setup memo PDU from memo header */
239  memo = &_coap_state.open_reqs[i];
240  coap_hdr_t *memo_hdr = (coap_hdr_t *) &memo->hdr_buf[0];
241  memo_pdu.hdr = memo_hdr;
242  if (coap_get_token_len(&memo_pdu)) {
243  memo_pdu.token = &memo_hdr->data[0];
244  }
245  /* match on token */
246  if (coap_get_token_len(src_pdu) == coap_get_token_len(&memo_pdu)) {
247  uint8_t *src_byte = src_pdu->token;
248  uint8_t *memo_byte = memo_pdu.token;
249  size_t j;
250  for (j = 0; j < coap_get_token_len(src_pdu); j++) {
251  if (*src_byte++ != *memo_byte++) {
252  break; /* token mismatch */
253  }
254  }
255  if (j == coap_get_token_len(src_pdu)) {
256  *memo_ptr = memo;
257  }
258  }
259  }
260 }

263 static void _expire_request(gcoap_request_memo_t *memo)
264 {
265  coap_pkt_t req;
266 
267  DEBUG("coap: received timeout message\n");
268  if (memo->state == GCOAP_MEMO_WAIT) {
269  memo->state = GCOAP_MEMO_TIMEOUT;
270  /* Pass response to handler */
271  if (memo->resp_handler) {

273  memo->resp_handler(memo->state, &req);
274  }
275  memo->state = GCOAP_MEMO_UNUSED;
276  }
277  else {
278  /* Response already handled; timeout must have fired while response */
279  /* was in queue. */

283 /*
284  * Handler for /.well-known/core. Lists registered handlers, except for
285  * /.well-known/core itself.
286  */
287 static ssize_t _well_known_core_handler(coap_pkt_t* pdu, uint8_t *buf, size_t len)
288 {
289  /* write header */
290  gcoap_resp_init(pdu, buf, len, COAP_CODE_CONTENT);
291 
292  /* skip the first listener, gcoap itself */
293  gcoap_listener_t *listener = _coap_state.listeners->next;
294 
295  /* write payload */
296  uint8_t *bufpos = pdu->payload;
297 
298  while (listener) {
299  coap_resource_t *resource = listener->resources;
300  for (size_t i = 0; i < listener->resources_len; i++) {
301  /* Don't overwrite buffer if paths are too long. */
302  if (bufpos + strlen(resource->path) + 3 > buf + len) {
303  break;
304  }
305  if (i) {
306  *bufpos++ = ',';
307  resource++;
308  }
309  *bufpos++ = '<';
310  unsigned url_len = strlen(resource->path);
311  memcpy(bufpos, resource->path, url_len);
312  bufpos += url_len;
313  *bufpos++ = '>';
314  }
315  listener = listener->next;
316  }
317 
318  /* response content */
319  return gcoap_finish(pdu, bufpos - pdu->payload, COAP_FORMAT_LINK);
320 }

322 /*
323  * Creates CoAP options and sets payload marker, if any.
324  *
325  * Returns length of header + options, or -EINVAL on illegal path.
326  */
327 static ssize_t _write_options(coap_pkt_t *pdu, uint8_t *buf, size_t len)
328 {
329  uint8_t last_optnum = 0;
330  (void)len;
331 
332  uint8_t *bufpos = buf + coap_get_total_hdr_len(pdu); /* position for write */
333 
334  /* Uri-Path for request */
335  if (coap_get_code_class(pdu) == COAP_CLASS_REQ) {
336  size_t url_len = strlen((char *)pdu->url);
337  if (url_len) {
338  if (pdu->url[0] != '/') {
339  return -EINVAL;
340  }
341  bufpos += coap_put_option_url(bufpos, last_optnum, (char *)&pdu->url[0]);
342  last_optnum = COAP_OPT_URI_PATH;
343  }
344  }
345 
346  /* Content-Format */
347  if (pdu->content_type != COAP_FORMAT_NONE) {
348  bufpos += coap_put_option_ct(bufpos, last_optnum, pdu->content_type);
349  /* uncomment when add an option after Content-Format */
350  /* last_optnum = COAP_OPT_CONTENT_FORMAT; */
351  }
352 
353  /* write payload marker */
354  if (pdu->payload_len) {
355  *bufpos++ = GCOAP_PAYLOAD_MARKER;
356  }
357  return bufpos - buf;
358 }

391 int gcoap_req_init(coap_pkt_t *pdu, uint8_t *buf, size_t len, unsigned code,
392  char *path) {
393  uint8_t token[GCOAP_TOKENLEN];
394  ssize_t hdrlen;
395  (void)len;
396 
397  pdu->hdr = (coap_hdr_t *)buf;
398  memset(pdu->url, 0, NANOCOAP_URL_MAX);
399 
400  /* generate token */
401  for (size_t i = 0; i < GCOAP_TOKENLEN; i += 4) {
402  uint32_t rand = random_uint32();
403  memcpy(&token[i],
404  &rand,
405  (GCOAP_TOKENLEN - i >= 4) ? 4 : GCOAP_TOKENLEN - i);
406  }
407  hdrlen = coap_build_hdr(pdu->hdr, COAP_TYPE_NON, &token[0], GCOAP_TOKENLEN,
408  code,
409  ++_coap_state.last_message_id);
410 
411  if (hdrlen > 0) {
412  /* Reserve some space between the header and payload to write options later */
413  pdu->payload = buf + coap_get_total_hdr_len(pdu) + strlen(path)
414  + GCOAP_REQ_OPTIONS_BUF;
415  /* Payload length really zero at this point, but we set this to the available
416  * length in the buffer. Allows us to reconstruct buffer length later. */
417  pdu->payload_len = len - (pdu->payload - buf);
418  pdu->content_type = COAP_FORMAT_NONE;
419 
420  memcpy(&pdu->url[0], path, strlen(path));
421  return 0;
422  }
423  else {
424  /* reason for negative hdrlen is not defined, so we also are vague */
425  return -1;
426  }
427 }

453 size_t gcoap_req_send2(uint8_t *buf, size_t len, sock_udp_ep_t *remote,
454  gcoap_resp_handler_t resp_handler)
455 {
456  gcoap_request_memo_t *memo = NULL;
457  assert(remote != NULL);
458  assert(resp_handler != NULL);
459 
460  /* Find empty slot in list of open requests. */
461  for (int i = 0; i < GCOAP_REQ_WAITING_MAX; i++) {
462  if (_coap_state.open_reqs[i].state == GCOAP_MEMO_UNUSED) {
463  memo = &_coap_state.open_reqs[i];
464  memo->state = GCOAP_MEMO_WAIT;
465  break;
466  }
467  }
468  if (memo) {
469  memcpy(&memo->hdr_buf[0], buf, GCOAP_HEADER_MAXLEN);
470  memo->resp_handler = resp_handler;
471 
472  size_t res = sock_udp_send(&_sock, buf, len, remote);
473 
474  if (res && (GCOAP_NON_TIMEOUT > 0)) {
475  /* interrupt sock listening (to set a listen timeout) */
476  msg_t mbox_msg;
477  mbox_msg.type = GCOAP_MSG_TYPE_INTR;
478  mbox_msg.content.value = 0;
479  if (mbox_try_put(&_sock.reg.mbox, &mbox_msg)) {
480  /* start response wait timer */
481  memo->timeout_msg.type = GCOAP_MSG_TYPE_TIMEOUT;
482  memo->timeout_msg.content.ptr = (char *)memo;
483  xtimer_set_msg(&memo->response_timer, GCOAP_NON_TIMEOUT,
484  &memo->timeout_msg, _pid);
485  }
486  else {
487  memo->state = GCOAP_MEMO_UNUSED;
488  DEBUG("gcoap: can't wake up mbox; no timeout for msg\n");
489  }
490  }
491  else if (!res) {
492  memo->state = GCOAP_MEMO_UNUSED;
493  DEBUG("gcoap: sock send failed: %d\n", res);
494  }
495  return res;
496  } else {
497  DEBUG("gcoap: dropping request; no space for response tracking\n");
498  return 0;
499  }
500 }
