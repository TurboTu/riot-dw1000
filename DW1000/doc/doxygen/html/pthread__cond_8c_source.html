95 int pthread_cond_wait(pthread_cond_t *cond, mutex_t *mutex)
96 {
97  priority_queue_node_t n;
98  n.priority = sched_active_thread->priority;
99  n.data = sched_active_pid;
100  n.next = NULL;
101 
102  /* the signaling thread may not hold the mutex, the queue is not thread safe */
103  unsigned old_state = irq_disable();
104  priority_queue_add(&(cond->queue), &n);
105  irq_restore(old_state);
106 
107  mutex_unlock_and_sleep(mutex);
108 
109  if (n.data != -1u) {
110  /* on signaling n.data is set to -1u */
111  /* if it isn't set, then the wakeup is either spurious or a timer wakeup */
112  old_state = irq_disable();
113  priority_queue_remove(&(cond->queue), &n);
114  irq_restore(old_state);
115  }
116 
117  mutex_lock(mutex);
118  return 0;
119 }

138 int pthread_cond_signal(pthread_cond_t *cond)
139 {
140  unsigned old_state = irq_disable();
141 
142  priority_queue_node_t *head = priority_queue_remove_head(&(cond->queue));
143  int other_prio = -1;
144  if (head != NULL) {
145  thread_t *other_thread = (thread_t *) sched_threads[head->data];
146  if (other_thread) {
147  other_prio = other_thread->priority;
148  sched_set_status(other_thread, STATUS_PENDING);
149  }
150  head->data = -1u;
151  }
152 
153  irq_restore(old_state);
154 
155  if (other_prio >= 0) {
156  sched_switch(other_prio);
157  }
158 
159  return 0;
