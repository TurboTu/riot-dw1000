62 static int _get(netdev2_t *netdev, netopt_t opt, void *value, size_t max_len)
63 {
64  cc2538_rf_t *dev = (cc2538_rf_t *)netdev;
65 
66  if (dev == NULL) {
67  return -ENODEV;
68  }
69 
70  switch (opt) {
71  case NETOPT_AUTOACK:
72  if (RFCORE->XREG_FRMCTRL0bits.AUTOACK) {
73  *((netopt_enable_t *)value) = NETOPT_ENABLE;
74  }
75  else {
76  *((netopt_enable_t *)value) = NETOPT_DISABLE;
77  }
78  return sizeof(netopt_enable_t);
79 
80  case NETOPT_CHANNEL_PAGE:
81  if (max_len < sizeof(uint16_t)) {
82  return -EOVERFLOW;
83  }
84  /* This tranceiver only supports page 0 */
85  *((uint16_t *)value) = 0;
86  return sizeof(uint16_t);
87 
88  case NETOPT_DEVICE_TYPE:
89  if (max_len < sizeof(uint16_t)) {
90  return -EOVERFLOW;
91  }
92  *((uint16_t *) value) = NETDEV2_TYPE_IEEE802154;
93  return sizeof(uint16_t);
94 
95  case NETOPT_IS_CHANNEL_CLR:
96  if (cc2538_channel_clear()) {
97  *((netopt_enable_t *)value) = NETOPT_ENABLE;
98  }
99  else {
100  *((netopt_enable_t *)value) = NETOPT_DISABLE;
101  }
102  return sizeof(netopt_enable_t);
103 
104  case NETOPT_IS_WIRED:
105  return -ENOTSUP;
106 
107  case NETOPT_MAX_PACKET_SIZE:
108  if (max_len < sizeof(int16_t)) {
109  return -EOVERFLOW;
110  }
111  *((uint16_t *)value) = CC2538_RF_MAX_DATA_LEN - _MAX_MHR_OVERHEAD;
112  return sizeof(uint16_t);
113 
114  case NETOPT_PROMISCUOUSMODE:
115  if (cc2538_get_monitor()) {
116  *((netopt_enable_t *)value) = NETOPT_ENABLE;
117  }
118  else {
119  *((netopt_enable_t *)value) = NETOPT_DISABLE;
120  }
121  return sizeof(netopt_enable_t);
122 
123  case NETOPT_STATE:
124  if (max_len < sizeof(netopt_state_t)) {
125  return -EOVERFLOW;
126  }
127  *((netopt_state_t *)value) = dev->state;
128  return sizeof(netopt_state_t);
129 
130  case NETOPT_TX_POWER:
131  if (max_len < sizeof(int16_t)) {
132  return -EOVERFLOW;
133  }
134  *((uint16_t *)value) = cc2538_get_tx_power();
135  return sizeof(uint16_t);
136 
137  default:
138  break;
139  }
140 
141  int res;
142 
143  if (((res = netdev2_ieee802154_get((netdev2_ieee802154_t *)netdev, opt, value,
144  max_len)) >= 0) || (res != -ENOTSUP)) {
145  return res;
146  }
147 
148  return -ENOTSUP;
149 }

151 static int _set(netdev2_t *netdev, netopt_t opt, void *value, size_t value_len)
152 {
153  cc2538_rf_t *dev = (cc2538_rf_t *)netdev;
154  int res = -ENOTSUP;
155 
156  if (dev == NULL) {
157  return -ENODEV;
158  }
159 
160  switch (opt) {
161  case NETOPT_ADDRESS:
162  if (value_len > sizeof(uint16_t)) {
163  res = -EOVERFLOW;
164  }
165  else {
166  cc2538_set_addr_short(*((uint16_t*)value));
167  }
168  break;
169 
170  case NETOPT_ADDRESS_LONG:
171  if (value_len > sizeof(uint64_t)) {
172  res = -EOVERFLOW;
173  }
174  else {
175  cc2538_set_addr_long(*((uint64_t*)value));
176  }
177  break;
178 
179  case NETOPT_AUTOACK:
180  RFCORE->XREG_FRMCTRL0bits.AUTOACK = ((bool *)value)[0];
181  res = sizeof(netopt_enable_t);
182  break;
183 
184  case NETOPT_CHANNEL:
185  if (value_len != sizeof(uint16_t)) {
186  res = -EINVAL;
187  }
188  else {
189  uint8_t chan = ((uint8_t *)value)[0];
190  if (chan < IEEE802154_CHANNEL_MIN ||
191  chan > IEEE802154_CHANNEL_MAX) {
192  res = -EINVAL;
193  }
194  else {
195  cc2538_set_chan(chan);
196  }
197  }
198  break;
199 
200  case NETOPT_CHANNEL_PAGE:
201  /* This tranceiver only supports page 0 */
202  if (value_len != sizeof(uint16_t) ||
203  *((uint16_t *)value) != 0 ) {
204  res = -EINVAL;
205  }
206  else {
207  res = sizeof(uint16_t);
208  }
209  break;
210 
211  case NETOPT_IS_WIRED:
212  return -ENOTSUP;
213 
214  case NETOPT_NID:
215  if (value_len > sizeof(uint16_t)) {
216  res = -EOVERFLOW;
217  }
218  else {
219  cc2538_set_pan(*((uint16_t *)value));
220  }
221  break;
222 
223  case NETOPT_PROMISCUOUSMODE:
224  cc2538_set_monitor(((bool *)value)[0]);
225  res = sizeof(netopt_enable_t);
226  break;
227 
228  case NETOPT_STATE:
229  if (value_len > sizeof(netopt_state_t)) {
230  return -EOVERFLOW;
231  }
232  cc2538_set_state(dev, *((netopt_state_t *)value));
233  res = sizeof(netopt_state_t);
234  break;
235 
236  case NETOPT_TX_POWER:
237  if (value_len > sizeof(int16_t)) {
238  return -EOVERFLOW;
239  }
240  cc2538_set_tx_power(*((int16_t *)value));
241  res = sizeof(uint16_t);
242  break;
243 
244  default:
245  break;
246  }
247 
248  if (res == -ENOTSUP) {
249  res = netdev2_ieee802154_set((netdev2_ieee802154_t *)netdev, opt,
250  value, value_len);
251  }
252 
253  return res;
254 }

256 static int _send(netdev2_t *netdev, const struct iovec *vector, unsigned count)
257 {
258  int pkt_len = 0;
259 
260  /* Flush TX FIFO once no transmission in progress */
261  RFCORE_WAIT_UNTIL(RFCORE->XREG_FSMSTAT1bits.TX_ACTIVE == 0);
262  RFCORE_SFR_RFST = ISFLUSHTX;
263 
264  /* The first byte of the TX FIFO must be the packet length,
265  but we don't know what it is yet. Write a null byte to the
266  start of the FIFO, so we can come back and update it later */
267  rfcore_write_byte(0);
268 
269  for (unsigned i = 0; i < count; i++) {
270  pkt_len += vector[i].iov_len;
271 
272  if (pkt_len > CC2538_RF_MAX_DATA_LEN) {
273  DEBUG("cc2538_rf: packet too large (%u > %u)\n",
274  pkt_len, CC2538_RF_MAX_DATA_LEN);
275  return -EOVERFLOW;
276  }
277 
278  rfcore_write_fifo(vector[i].iov_base, vector[i].iov_len);
279  }
280 
281 #ifdef MODULE_NETSTATS_L2
282  netdev->stats.tx_bytes += pkt_len;
283 #endif
284 
285  /* Set first byte of TX FIFO to the packet length */
286  rfcore_poke_tx_fifo(0, pkt_len + CC2538_AUTOCRC_LEN);
287 
288  RFCORE_SFR_RFST = ISTXON;
289 
290  /* Wait for transmission to complete */
291  RFCORE_WAIT_UNTIL(RFCORE->XREG_FSMSTAT1bits.TX_ACTIVE == 0);
292 
293  return pkt_len;
294 }

296 static int _recv(netdev2_t *netdev, void *buf, size_t len, void *info)
297 {
298  size_t pkt_len;
299 
300  if (buf == NULL) {
301  /* GNRC wants to know how much data we've got for it */
302  pkt_len = rfcore_read_byte();
303 
304  /* Make sure pkt_len is sane */
305  if (pkt_len > CC2538_RF_MAX_DATA_LEN) {
306  RFCORE_SFR_RFST = ISFLUSHRX;
307  return -EOVERFLOW;
308  }
309 
310  /* CRC check */
311  if (!(rfcore_peek_rx_fifo(pkt_len) & 0x80)) {
312  /* CRC failed; discard packet */
313  RFCORE_SFR_RFST = ISFLUSHRX;
314  return -ENODATA;
315  }
316 
317  if (len > 0) {
318  /* GNRC wants us to drop the packet */
319  RFCORE_SFR_RFST = ISFLUSHRX;
320  }
321 
322  return pkt_len - IEEE802154_FCS_LEN;
323  }
324  else {
325  /* GNRC is expecting len bytes of data */
326  pkt_len = len;
327  }
328 
329 #ifdef MODULE_NETSTATS_L2
330  netdev->stats.rx_count++;
331  netdev->stats.rx_bytes += pkt_len;
332 #endif
333  rfcore_read_fifo(buf, pkt_len);
334 
335  if (info != NULL && RFCORE->XREG_RSSISTATbits.RSSI_VALID) {
336  uint8_t corr_val;
337  int8_t rssi_val;
338  netdev2_ieee802154_rx_info_t *radio_info = info;
339  rssi_val = rfcore_read_byte() + CC2538_RSSI_OFFSET;
340 
341  RFCORE_ASSERT(rssi_val > CC2538_RF_SENSITIVITY);
342 
343  /* The number of dB above maximum sensitivity detected for the
344  * received packet */
345  radio_info->rssi = -CC2538_RF_SENSITIVITY + rssi_val;
346 
347  corr_val = rfcore_read_byte() & CC2538_CORR_VAL_MASK;
348 
349  if (corr_val < CC2538_CORR_VAL_MIN) {
350  corr_val = CC2538_CORR_VAL_MIN;
351  }
352  else if (corr_val > CC2538_CORR_VAL_MAX) {
353  corr_val = CC2538_CORR_VAL_MAX;
354  }
355 
356  /* Interpolate the correlation value between 0 - 255
357  * to provide an LQI value */
358  radio_info->lqi = 255 * (corr_val - CC2538_CORR_VAL_MIN) /
359  (CC2538_CORR_VAL_MAX - CC2538_CORR_VAL_MIN);
360  }
361 
362  RFCORE_SFR_RFST = ISFLUSHRX;
363 
364  return pkt_len;
365 }

381

384  &pan, sizeof(pan));
385  netdev2_ieee802154_set((netdev2_ieee802154_t *)netdev, NETOPT_CHANNEL,
386  &chan, sizeof(chan));
387  netdev2_ieee802154_set((netdev2_ieee802154_t *)netdev, NETOPT_ADDRESS,
388  &addr_short, sizeof(addr_short));
389  netdev2_ieee802154_set((netdev2_ieee802154_t *)netdev, NETOPT_ADDRESS_LONG,
390  &addr_long, sizeof(addr_long));
391 
392  cc2538_set_state(dev, NETOPT_STATE_IDLE);
393 
394  /* set default protocol */
395 #ifdef MODULE_GNRC_SIXLOWPAN
396  dev->netdev.proto = GNRC_NETTYPE_SIXLOWPAN;
397 #elif MODULE_GNRC
398  dev->netdev.proto = GNRC_NETTYPE_UNDEF;
399 #endif
400 #ifdef MODULE_NETSTATS_L2
401  memset(&netdev->stats, 0, sizeof(netstats_t));
402 #endif
403 
404  return 0;
