51 {
52  for (unsigned int i = 0; i < L3G4200D_NUM; i++) {
53  const l3g4200d_params_t *p = &l3g4200d_params[i];
54 
55  LOG_DEBUG("[auto_init_saul] initializing l3g4200d #%u\n", i);
56 
57  int res = l3g4200d_init(&l3g4200d_devs[i], p->i2c, p->addr,
58  p->int1_pin, p->int2_pin, p->mode, p->scale);
59  if (res < 0) {
60  LOG_ERROR("[auto_init_saul] error initializing l3g4200d #%u\n", i);
61  }
62  else {
63  saul_entries[i].dev = &(l3g4200d_devs[i]);
