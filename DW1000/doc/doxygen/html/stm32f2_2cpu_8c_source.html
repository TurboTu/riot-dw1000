50 static void clk_init(void)
51 {
52  /* Reset the RCC clock configuration to the default reset state(for debug purpose) */
53  /* Set HSION bit */
54  RCC->CR |= 0x00000001U;
55  /* Reset SW, HPRE, PPRE1, PPRE2, ADCPRE and MCO bits */
56  RCC->CFGR = 0x00000000U;
57  /* Reset HSEON, CSSON and PLLON bits */
58  RCC->CR &= 0xFEF6FFFFU;
59  /* Reset PLLCFGR register */
60  RCC->PLLCFGR = 0x24003010U;
61  /* Reset HSEBYP bit */
62  RCC->CR &= 0xFFFBFFFFU;
63  /* Disable all interrupts and clear pending bits */
64  RCC->CIR = 0x00000000U;
65 
66  /* SYSCLK, HCLK, PCLK2 and PCLK1 configuration */
67  /* Enable the high speed clock source */
68  RCC->CR |= RCC_CR_SOURCE;
69  /* Wait till hish speed clock source is ready,
70  * NOTE: the MCU will stay here forever if no HSE clock is connected */
71  while ((RCC->CR & RCC_CR_SOURCE_RDY) == 0);
72  /* Configure Flash prefetch, Instruction cache, Data cache and wait state */
73  FLASH->ACR = FLASH_ACR_PRFTEN | FLASH_ACR_ICEN | FLASH_ACR_DCEN;
74  /* Flash 2 wait state */
75  FLASH->ACR &= ~((uint32_t)FLASH_ACR_LATENCY);
76  FLASH->ACR |= (uint32_t)CLOCK_FLASH_LATENCY;
77  /* HCLK = SYSCLK */
78  RCC->CFGR |= (uint32_t)CLOCK_AHB_DIV;
79  /* PCLK2 = HCLK */
80  RCC->CFGR |= (uint32_t)CLOCK_APB2_DIV;
81  /* PCLK1 = HCLK */
82  RCC->CFGR |= (uint32_t)CLOCK_APB1_DIV;
83 
84  /* reset PLL config register */
85  RCC->PLLCFGR &= ~((uint32_t)(RCC_PLLCFGR_PLLSRC | RCC_PLLCFGR_PLLN | RCC_PLLCFGR_PLLM | RCC_PLLCFGR_PLLP | RCC_PLLCFGR_PLLQ));
86  /* set HSE as source for the PLL */
87  RCC->PLLCFGR |= RCC_PLL_SOURCE;
88  /* set division factor for main PLL input clock */
89  RCC->PLLCFGR |= (CLOCK_PLL_M & 0x3F);
90  /* set main PLL multiplication factor for VCO */
91  RCC->PLLCFGR |= (CLOCK_PLL_N & 0x1FF) << 6;
92  /* set main PLL division factor for main system clock */
93  RCC->PLLCFGR |= (((CLOCK_PLL_P & 0x03) >> 1) - 1) << 16;
94  /* set main PLL division factor for USB OTG FS, SDIO and RNG clocks */
95  RCC->PLLCFGR |= (CLOCK_PLL_Q & 0x0F) << 24;
96 
97 #ifdef ENABLE_PLLI2S_MCO2
98  /* reset PLL I2S config register */
99  RCC->PLLI2SCFGR = 0x00000000U;
100  /* set PLL I2S division factor */
101  RCC->PLLI2SCFGR |= (CLOCK_PLL_I2S_R & 0x07) << 28;
102  /* set PLL I2S multiplication factor */
103  RCC->PLLI2SCFGR |= (CLOCK_PLL_I2S_N & 0x1FF) << 6;
104 
105  /* MCO2 output is PLLI2S */
106  RCC->CFGR |= (uint32_t) RCC_CFGR_MCO2_0;
107  RCC->CFGR &= ~(uint32_t) RCC_CFGR_MCO2_1;
108  /* MCO2 prescaler div by 5 */
109  RCC->CFGR |= (uint32_t) ((CLOCK_MC02_PRE + 4 - 2) & 0x7) << 27;
110  /* enable PLL I2S clock */
111  RCC->CR |= RCC_CR_PLLI2SON;
112  /* wait till PLL I2S clock is ready */
113  while ((RCC->CR & RCC_CR_PLLI2SRDY) == 0) {}
114 #endif
115 
116  /* Enable PLL */
117  RCC->CR |= RCC_CR_PLLON;
118  /* Wait till PLL is ready */
119  while ((RCC->CR & RCC_CR_PLLRDY) == 0);
120  /* Select PLL as system clock source */
121  RCC->CFGR &= ~((uint32_t)(RCC_CFGR_SW));
122  RCC->CFGR |= (uint32_t)RCC_CFGR_SW_PLL;
123  /* Wait till PLL is used as system clock source */
124  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL);
125 }
