102 int kinetis_adc_calibrate(ADC_Type *dev)
103 {
104  uint16_t cal;
105 
106  dev->SC3 |= ADC_SC3_CAL_MASK;
107 
108  while (dev->SC3 & ADC_SC3_CAL_MASK) {} /* wait for calibration to finish */
109 
110  while (!(dev->SC1[0] & ADC_SC1_COCO_MASK)) {}
111 
112  if (dev->SC3 & ADC_SC3_CALF_MASK) {
113  /* calibration failed for some reason, possibly SC2[ADTRG] is 1 ? */
114  return -1;
115  }
116 
117  /* From the reference manual */
118  /* 1. Initialize or clear a 16-bit variable in RAM. */
119  /* 2. Add the plus-side calibration results CLP0, CLP1, CLP2, CLP3, CLP4,
120  * and CLPS to the variable. */
121  cal = dev->CLP0 + dev->CLP1 + dev->CLP2 + dev->CLP3 + dev->CLP4 + dev->CLPS;
122  /* 3. Divide the variable by two. */
123  cal /= 2;
124  /* 4. Set the MSB of the variable. */
125  cal |= (1 << 15);
126  /* (5. The previous two steps can be achieved by setting the carry bit,
127  * rotating to the right through the carry bit on the high byte and again on
128  * the low byte.)
129  * We ignore the above optimization suggestion, we most likely only perform
130  * this calibration on startup and it will only save nanoseconds. */
131  /* 6. Store the value in the plus-side gain calibration register PG. */
132  dev->PG = cal;
133  /* 7. Repeat the procedure for the minus-side gain calibration value. */
134  cal = dev->CLM0 + dev->CLM1 + dev->CLM2 + dev->CLM3 + dev->CLM4 + dev->CLMS;
135  cal /= 2;
136  cal |= (1 << 15);
137  dev->MG = cal;
138 
139  return 0;
140 }

142 int adc_init(adc_t line)
143 {
144  /* make sure the given line is valid */
145  if (line >= ADC_NUMOF) {
146  return -1;
147  }
148 
149  /* prepare the device: lock and power on */
150  prep(line);
151 
152  /* configure the connected pin mux */
153  if (adc_config[line].pin != GPIO_UNDEF) {
154  gpio_init_port(adc_config[line].pin, GPIO_AF_ANALOG);
155  }
156 
157  /* The ADC requires at least 2 MHz module clock for full accuracy, and less
158  * than 12 MHz */
159  /* For the calibration it is important that the ADC clock is <= 4 MHz */
160  uint32_t adiv;
161  if (CLOCK_BUSCLOCK > (ADC_MAX_CLK << 3)) {
162  adiv = ADC_CFG1_ADIV(3) | ADC_CFG1_ADICLK(1);
163  }
164  else {
165  unsigned int i = 0;
166  while ((i < 3) && (CLOCK_BUSCLOCK > (ADC_MAX_CLK << i))) {
167  ++i;
168  }
169  adiv = ADC_CFG1_ADIV(i);
170  }
171 
172  /* set configuration register 1: clocking and precision */
173  /* Set long sample time */
174  dev(line)->CFG1 = ADC_CFG1_ADLSMP_MASK | adiv;
175  /* select ADxxb channels, longest sample time (20 extra ADC cycles) */
176  dev(line)->CFG2 = ADC_CFG2_MUXSEL_MASK | ADC_CFG2_ADLSTS(0);
177  /* select software trigger, external ref pins */
178  dev(line)->SC2 = ADC_SC2_REFSEL(0);
179  /* select hardware average over 32 samples */
180  dev(line)->SC3 = ADC_SC3_AVGE_MASK | ADC_SC3_AVGS(3);
181  /* set an (arbitrary) input channel, single-ended mode */
182  dev(line)->SC1[0] = ADC_SC1_ADCH(0);
183 
184  /* perform calibration routine */
185  int res = kinetis_adc_calibrate(dev(line));
186  done(line);
187  return res;
188 }

190 int adc_sample(adc_t line, adc_res_t res)
191 {
192  int sample;
193 
194  /* check if resolution is applicable */
195  if (res > 0xf0) {
196  return -1;
197  }
198 
199  /* lock and power on the device */
200  prep(line);
201 
202  /* set resolution */
203  dev(line)->CFG1 &= ~(ADC_CFG1_MODE_MASK);
204  dev(line)->CFG1 |= (res);
205  /* select the channel that is sampled */
206  dev(line)->SC1[0] = adc_config[line].chan;
207  /* wait until conversion is complete */
208  while (!(dev(line)->SC1[0] & ADC_SC1_COCO_MASK)) {}
209  /* read and return result */
210  sample = (int)dev(line)->R[0];
211 
212  /* power off and unlock the device */
213  done(line);
214 
215  return sample;
216 }
