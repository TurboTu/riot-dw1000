54 int mag3110_init(mag3110_t *dev, i2c_t i2c, uint8_t address, uint8_t dros)
55 {
56  uint8_t reg;
57 
58  /* write device descriptor */
59  dev->i2c = i2c;
60  dev->addr = address;
61  dev->initialized = false;
62 
63  if (dros > MAG3110_DROS_0008_128) {
64  return -1;
65  }
66 
67  i2c_acquire(dev->i2c);
68  /* initialize the I2C bus */
69  if (i2c_init_master(i2c, I2C_SPEED) < 0) {
70  i2c_release(dev->i2c);
71  return -2;
72  }
73  i2c_release(dev->i2c);
74 
75  if (mag3110_test(dev)) {
76  return -3;
77  }
78 
79  /* enable automatic magnetic sensor reset */
80  reg = MAG3110_CTRL_REG2_AUTO_MRST_EN;
81 
82  i2c_acquire(dev->i2c);
83  if (i2c_write_regs(dev->i2c, dev->addr, MAG3110_CTRL_REG2, &reg, 1) != 1) {
84  i2c_release(dev->i2c);
85  return -4;
86  }
87 
88  reg = MAG3110_CTRL_REG1_DROS(dros);
89 
90  if (i2c_write_regs(dev->i2c, dev->addr, MAG3110_CTRL_REG1, &reg, 1) != 1) {
91  i2c_release(dev->i2c);
92  return -4;
93  }
94  i2c_release(dev->i2c);
95 
96  dev->initialized = true;
97 
98  return 0;
99 }

186 int mag3110_read(mag3110_t *dev, int16_t *x, int16_t *y, int16_t *z, uint8_t *status)
187 {
188  uint8_t buf[7];
189 
190  if (dev->initialized == false) {
191  return -1;
192  }
193 
194  i2c_acquire(dev->i2c);
195  if (i2c_read_regs(dev->i2c, dev->addr, MAG3110_DR_STATUS, buf, 7) != 7) {
196  i2c_release(dev->i2c);
197  return -1;
198  }
199  i2c_release(dev->i2c);
200 
201  *status = buf[0];
202  *x = ((int16_t)buf[1] << 8) | buf[2];
203  *y = ((int16_t)buf[3] << 8) | buf[4];
204  *z = ((int16_t)buf[5] << 8) | buf[6];
205 
206  return 0;
207 }
