232 void do_term_input(gpointer data, int fd, GdkInputCondition cond)
233 {
234  char buf[256];
235  int num, flags;
236 
237  flags = fcntl(term_fd, F_GETFL);
238  fcntl(term_fd, F_SETFL, flags | O_NONBLOCK);
239  num = read(term_fd, buf, sizeof(buf));
240  fcntl(term_fd, F_SETFL, flags);
241 
242  if (num > 0) {
243  if (download_in_progress) {
244  download_rx_term(buf, num);
245  }
246  else {
247  write_serial_port(buf, num);
248  }
249  }
250 }
