39 int timer_init(tim_t dev, unsigned long freq, timer_cb_t cb, void *arg)
40 {
41  TIMER_TypeDef *pre, *tim;
42 
43  /* test if given timer device is valid */
44  if (dev >= TIMER_NUMOF) {
45  return -1;
46  }
47 
48  /* save callback */
49  isr_ctx[dev].cb = cb;
50  isr_ctx[dev].arg = arg;
51 
52  /* get timers */
53  pre = timer_config[dev].prescaler;
54  tim = timer_config[dev].timer;
55 
56  /* power on timers (if not already powered on) */
57  CMU->HFPERCLKEN0 |= (0x3 << timer_config[dev].pre_cmu);
58  /* stop both (in case they are running) */
59  pre->CMD = TIMER_CMD_STOP;
60  tim->CMD = TIMER_CMD_STOP;
61  /* configure the pre-scale timer to drive the actual timer. For this we
62  * configure it up-counting, driven by the HFPER clock and we set the TOP
63  * register depending on the specified timer speed value */
64  pre->CTRL = 0;
65  pre->TOP = ((CLOCK_HFPERCLK / freq) - 1);
66  pre->CNT = 0;
67  pre->IEN = 0;
68 
69  /* configure the actual timer to up-counting mode and to be fed by the
70  * pre-scale timer */
71  tim->CTRL = TIMER_CTRL_CLKSEL_TIMEROUF;
72  tim->TOP = 0xffff;
73  tim->CNT = 0;
74  /* clear all CC interrupt flags and enable their interrupts */
75  tim->IFC = (TIMER_IFC_CC0 | TIMER_IFC_CC1 | TIMER_IFC_CC2);
76  tim->IEN = (TIMER_IEN_CC0 | TIMER_IEN_CC1 | TIMER_IEN_CC2);
77  /* activate global timer interrupt */
78  NVIC_EnableIRQ(timer_config[dev].irqn);
79  /* start both timers */
80  tim->CMD = TIMER_CMD_START;
81  pre->CMD = TIMER_CMD_START;
82  return 0;
83 }
