187  uint8_t *addr, size_t *addr_size_in_bits)
188 {
189  mutex_lock(&mtx_access);
190 
191  int ret = -ENOENT;
192 
193  /* If we have distinct sizes, the addresses are probably not comperable */
194  if ((size_t)(entry->address_size<<3) != *addr_size_in_bits) {
195  mutex_unlock(&mtx_access);
196  return ret;
197  }
198 
199  /* compare up to fist distinct byte, pretty clumsy method for now */
200  int idx = -1;
201  bool test_all_zeros = true;
202  for (size_t i = 0; i < entry->address_size; i++) {
203  if ((idx == -1) && (entry->address[i] != addr[i])) {
204  idx = i;
205  }
206  if (test_all_zeros) {
207  test_all_zeros = (entry->address[i] == 0);
208  }
209  if ((idx != -1) && !test_all_zeros) {
210  break;
211  }
212  }
213 
214  /* if the address is all 0 its a default route address */
215  if (test_all_zeros) {
216  *addr_size_in_bits = 0;
217  mutex_unlock(&mtx_access);
218  return UNIVERSAL_ADDRESS_IS_ALL_ZERO_ADDRESS;
219  }
220 
221  /* if we have no distinct bytes the addresses are equal */
222  if (idx == -1) {
223  mutex_unlock(&mtx_access);
224  return UNIVERSAL_ADDRESS_EQUAL;
225  }
226 
227  /* count equal bits */
228  uint8_t xor = entry->address[idx]^addr[idx];
229  int8_t j = 7;
230  for ( ; j > 0; --j) {
231  if ((xor >> j) & 0x01) {
232  break;
233  }
234  }
235 
236  /* get the total number of matching bits */
237  *addr_size_in_bits = (idx << 3) + j;
238  ret = UNIVERSAL_ADDRESS_MATCHING_PREFIX;

240  mutex_unlock(&mtx_access);
241  return ret;

251  mutex_unlock(&mtx_access);

254 
255  /* Get the index of the first trailing `0` */
256  int i = 0;
257  for (i = entry->address_size-1; i >= 0; --i) {
258  if (prefix[i] != 0) {
259  break;
260  }
261  }
262

264  /* if the bytes-1 equals we check the bits of the lowest byte */
265  uint8_t bitmask = 0x00;
266  /* get a bitmask for the trailing 0b */
267  for (uint8_t j = 0; j < 8; ++j) {

269  bitmask = 0xff << j;
270  break;
271  }
272  }
