33 /* We generate some context switching and IPC traffic by using multiple threads
34  * and generate some xtimer load by scheduling several messages to be called at
35  * different times. TEST_HZ is the frequency of messages being sent from the
36  * main thread to the worker, all other message frequencies are derived from
37  * TEST_HZ.
38  * TEST_MSG_RX_USLEEP is a tiny sleep inside the message reception thread to
39  * cause extra context switches.
40  */

86 /* This thread will print the drift to stdout once per second */
87 void *worker_thread(void *arg)
88 {
89  (void) arg;
90  uint32_t loop_counter = 0;
91  uint32_t start = 0;
92  uint32_t last = 0;
93 
94  printf("Starting thread %" PRIkernel_pid "\n", thread_getpid());
95 
96  while (1) {
97  msg_t m;
98  msg_receive(&m);
99  xtimer_ticks32_t ticks = xtimer_now();
100  uint32_t now = xtimer_usec_from_ticks(ticks);
101  if (start == 0) {
102  start = now;
103  last = start;
104  ++loop_counter;
105  continue;
106  }
107 
108  uint32_t us, sec;
109  us = now % US_PER_SEC;
110  sec = now / US_PER_SEC;
111  if ((loop_counter % TEST_HZ) == 0) {
112  uint32_t expected = start + loop_counter * TEST_INTERVAL;
113  int32_t drift = now - expected;
114  expected = last + TEST_HZ * TEST_INTERVAL;
115  int32_t jitter = now - expected;
116  printf("now=%" PRIu32 ".%06" PRIu32 " (0x%08" PRIx32 " ticks), ",
117  sec, us, ticks.ticks32);
118  printf("drift=%" PRId32 " us, jitter=%" PRId32 " us\n", drift, jitter);
119  last = now;
120  }
121  ++loop_counter;
122  }
123 }
