56 static uint8_t get_prescaler(unsigned int target, int *scale)
57 {
58  if (target == 0) {
59  return 0xff;
60  }
61 
62  if (target >= 512) {
63  *scale = 1024;
64  return TCC_CTRLA_PRESCALER_DIV1024_Val;
65  }
66  if (target >= 128) {
67  *scale = 256;
68  return TCC_CTRLA_PRESCALER_DIV256_Val;
69  }
70  if (target >= 32) {
71  *scale = 64;
72  return TCC_CTRLA_PRESCALER_DIV64_Val;
73  }
74  if (target >= 12) {
75  *scale = 16;
76  return TCC_CTRLA_PRESCALER_DIV16_Val;
77  }
78  if (target >= 6) {
79  *scale = 8;
80  return TCC_CTRLA_PRESCALER_DIV8_Val;
81  }
82  if (target >= 3) {
83  *scale = 4;
84  return TCC_CTRLA_PRESCALER_DIV4_Val;
85  }
86  *scale = target;
87  return target - 1;
88 }

90 uint32_t pwm_init(pwm_t dev, pwm_mode_t mode, uint32_t freq, uint16_t res)
91 {
92  uint8_t prescaler;
93  int scale = 1;
94  uint32_t f_real;
95 
96  if ((unsigned int)dev >= PWM_NUMOF) {
97  return 0;
98  }
99 
100  /* calculate the closest possible clock presacler */
101  prescaler = get_prescaler(CLOCK_CORECLOCK / (freq * res), &scale);
102  if (prescaler == 0xff) {
103  return 0;
104  }
105  f_real = (CLOCK_CORECLOCK / (scale * res));
106 
107  /* configure the used pins */
108  for (int i = 0; i < PWM_MAX_CHANNELS; i++) {
109  if (pwm_config[dev].chan[i].pin != GPIO_UNDEF) {
110  gpio_init(pwm_config[dev].chan[i].pin, GPIO_OUT);
111  gpio_init_mux(pwm_config[dev].chan[i].pin, pwm_config[dev].chan[i].mux);
112  }
113  }
114 
115  /* power on the device */
116  pwm_poweron(dev);
117 
118  /* reset TCC module */
119  _tcc(dev)->CTRLA.reg = TCC_CTRLA_SWRST;
120  while (_tcc(dev)->SYNCBUSY.reg & TCC_SYNCBUSY_SWRST) {}
121  /* set PWM mode */
122  switch (mode) {
123  case PWM_LEFT:
124  _tcc(dev)->CTRLBCLR.reg = TCC_CTRLBCLR_DIR; /* count up */
125  break;
126  case PWM_RIGHT:
127  _tcc(dev)->CTRLBSET.reg = TCC_CTRLBSET_DIR; /* count down */
128  break;
129  case PWM_CENTER: /* currently not supported */
130  default:
131  return 0;
132  }
133  while (_tcc(dev)->SYNCBUSY.reg & TCC_SYNCBUSY_CTRLB) {}
134 
135  /* configure the TCC device */
136  _tcc(dev)->CTRLA.reg = (TCC_CTRLA_PRESCSYNC_GCLK_Val
137  | TCC_CTRLA_PRESCALER(prescaler));
138  /* select the waveform generation mode -> normal PWM */
139  _tcc(dev)->WAVE.reg = (TCC_WAVE_WAVEGEN_NPWM);
140  while (_tcc(dev)->SYNCBUSY.reg & TCC_SYNCBUSY_WAVE) {}
141  /* set the selected period */
142  _tcc(dev)->PER.reg = (res - 1);
143  while (_tcc(dev)->SYNCBUSY.reg & TCC_SYNCBUSY_PER) {}
144  /* start PWM operation */
145  pwm_start(dev);
146  /* return the actual frequency the PWM is running at */
147  return f_real;
148 }

155 void pwm_set(pwm_t dev, uint8_t channel, uint16_t value)
156 {
157  if ((channel >= PWM_MAX_CHANNELS) ||
158  (pwm_config[dev].chan[channel].pin == GPIO_UNDEF)) {
159  return;
160  }
161  _tcc(dev)->CC[_chan(dev, channel)].reg = value;
162  while (_tcc(dev)->SYNCBUSY.reg & (TCC_SYNCBUSY_CC0 << _chan(dev, channel))) {}
163 }

175 void pwm_poweron(pwm_t dev)
176 {
177  int num = _num(dev);
178  if (num < 0) {
179  return;
180  }
181  PM->APBCMASK.reg |= (PM_APBCMASK_TCC0 << num);
182  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN |
183  GCLK_CLKCTRL_GEN_GCLK0 |
184  GCLK_CLKCTRL_ID(_clk_id(dev)));
185  while (GCLK->STATUS.bit.SYNCBUSY) {}
186 }

189 {

192  return;
193  }
194  PM->APBCMASK.reg &= ~(PM_APBCMASK_TCC0 << num);
195  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_GEN_GCLK7 |
196  GCLK_CLKCTRL_ID(_clk_id(dev)));
197  while (GCLK->STATUS.bit.SYNCBUSY) {}
