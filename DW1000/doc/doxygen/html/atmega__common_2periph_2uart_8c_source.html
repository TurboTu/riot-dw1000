94 static void _set_brr(uart_t uart, uint32_t baudrate)
95 {

113 int uart_init(uart_t uart, uint32_t baudrate, uart_rx_cb_t rx_cb, void *arg)
114 {
115  /* make sure the given device is valid */
116  if (uart >= UART_NUMOF) {
117  return UART_NODEV;
118  }
119 
120  /* register interrupt context */
121  isr_ctx[uart].rx_cb = rx_cb;
122  isr_ctx[uart].arg = arg;
123 
124  /* disable and reset UART */
125  dev[uart]->CSRB = 0;
126  dev[uart]->CSRA = 0;
127 
128  /* configure UART to 8N1 mode */
129  dev[uart]->CSRC = (1 << UCSZ00) | (1 << UCSZ01);
130  /* set clock divider */
131  _set_brr(uart, baudrate);
132  /* enable RX and TX and the RX interrupt */
133  dev[uart]->CSRB = ((1 << RXCIE0) | (1 << RXEN0) | (1 << TXEN0));
134 
135  return UART_OK;
136 }
