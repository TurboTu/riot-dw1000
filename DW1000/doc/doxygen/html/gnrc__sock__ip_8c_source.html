29 int sock_ip_create(sock_ip_t *sock, const sock_ip_ep_t *local,
30  const sock_ip_ep_t *remote, uint8_t proto, uint16_t flags)
31 {
32  assert(sock);
33  if ((local != NULL) && (remote != NULL) &&
34  (local->netif != SOCK_ADDR_ANY_NETIF) &&
35  (remote->netif != SOCK_ADDR_ANY_NETIF) &&
36  (local->netif != remote->netif)) {
37  return -EINVAL;
38  }
39  memset(&sock->local, 0, sizeof(sock_ip_ep_t));
40  if (local != NULL) {
41  if (gnrc_af_not_supported(local->family)) {
42  return -EAFNOSUPPORT;
43  }
44  memcpy(&sock->local, local, sizeof(sock_ip_ep_t));
45  }
46  memset(&sock->remote, 0, sizeof(sock_ip_ep_t));
47  if (remote != NULL) {
48  if (gnrc_af_not_supported(remote->family)) {
49  return -EAFNOSUPPORT;
50  }
51  if (gnrc_ep_addr_any(remote)) {
52  return -EINVAL;
53  }
54  memcpy(&sock->remote, remote, sizeof(sock_ip_ep_t));
55  }
56  gnrc_sock_create(&sock->reg, GNRC_NETTYPE_IPV6,
57  proto);
58  sock->flags = flags;
59  return 0;
60 }

88 ssize_t sock_ip_recv(sock_ip_t *sock, void *data, size_t max_len,
89  uint32_t timeout, sock_ip_ep_t *remote)
90 {
91  gnrc_pktsnip_t *pkt;
92  sock_ip_ep_t tmp;
93  int res;
94 
95  assert((sock != NULL) && (data != NULL) && (max_len > 0));
96  if (sock->local.family == 0) {
97  return -EADDRNOTAVAIL;
98  }
99  tmp.family = sock->local.family;
100  res = gnrc_sock_recv((gnrc_sock_reg_t *)sock, &pkt, timeout, &tmp);
101  if (res < 0) {
102  return res;
103  }
104  if (pkt->size > max_len) {
105  gnrc_pktbuf_release(pkt);
106  return -ENOBUFS;
107  }
108  if (remote != NULL) {
109  /* return remote to possibly block if wrong remote */
110  memcpy(remote, &tmp, sizeof(tmp));
111  }
112  if ((sock->remote.family != AF_UNSPEC) && /* check remote end-point if set */
113  /* We only have IPv6 for now, so just comparing the whole end point
114  * should suffice */
115  ((memcmp(&sock->remote.addr, &ipv6_addr_unspecified,
116  sizeof(ipv6_addr_t)) != 0) &&
117  (memcmp(&sock->remote.addr, &tmp.addr, sizeof(ipv6_addr_t)) != 0))) {
118  gnrc_pktbuf_release(pkt);
119  return -EPROTO;
120  }
121  memcpy(data, pkt->data, pkt->size);
122  gnrc_pktbuf_release(pkt);
123  return (int)pkt->size;
124 }

126 ssize_t sock_ip_send(sock_ip_t *sock, const void *data, size_t len,
127  uint8_t proto, const sock_ip_ep_t *remote)
128 {
129  int res;
130  gnrc_pktsnip_t *pkt;
131  sock_ip_ep_t local;
132  sock_ip_ep_t rem;
133 
134  assert((sock != NULL) || (remote != NULL));
135  assert((len == 0) || (data != NULL)); /* (len != 0) => (data != NULL) */
136  if ((remote != NULL) && (sock != NULL) &&
137  (sock->local.netif != SOCK_ADDR_ANY_NETIF) &&
138  (remote->netif != SOCK_ADDR_ANY_NETIF) &&
139  (sock->local.netif != remote->netif)) {
140  return -EINVAL;
141  }
142  if ((remote == NULL) &&
143  /* sock can't be NULL as per assertion above */
144  (sock->remote.family == AF_UNSPEC)) {
145  return -ENOTCONN;
146  }
147  else if ((remote != NULL) && (gnrc_ep_addr_any(remote))) {
148  return -EINVAL;
149  }
150  /* compiler evaluates lazily so this isn't a redundundant check and cppcheck
151  * is being weird here anyways */
152  /* cppcheck-suppress nullPointerRedundantCheck */
153  /* cppcheck-suppress nullPointer */
154  if ((sock == NULL) || (sock->local.family == AF_UNSPEC)) {
155  /* no sock or sock currently unbound */
156  memset(&local, 0, sizeof(local));
157  }
158  else {
159  if (sock != NULL) {
160  proto = (uint8_t)sock->reg.entry.demux_ctx;
161  }
162  memcpy(&local, &sock->local, sizeof(local));
163  }
164  if (remote == NULL) {
165  /* sock can't be NULL at this point */
166  memcpy(&rem, &sock->remote, sizeof(rem));
167  }
168  else {
169  memcpy(&rem, remote, sizeof(rem));
170  }
171  if ((remote != NULL) && (remote->family == AF_UNSPEC) &&
172  (sock->remote.family != AF_UNSPEC)) {
173  /* remote was set on create so take its family */
174  rem.family = sock->remote.family;
175  }
176  else if ((remote != NULL) && gnrc_af_not_supported(remote->family)) {
177  return -EAFNOSUPPORT;
178  }
179  else if ((local.family == AF_UNSPEC) && (rem.family != AF_UNSPEC)) {
180  /* local was set to 0 above */
181  local.family = rem.family;
182  }
183  else if ((local.family != AF_UNSPEC) && (rem.family == AF_UNSPEC)) {
184  /* local was given on create, but remote family wasn't given by user and
185  * there was no remote given on create, take from local */
186  rem.family = local.family;
187  }
188  pkt = gnrc_pktbuf_add(NULL, (void *)data, len, GNRC_NETTYPE_UNDEF);
189  if (pkt == NULL) {
190  return -ENOMEM;
191  }
192  res = gnrc_sock_send(pkt, &local, &rem, proto);
193  if (res <= 0) {
194  return res;
195  }
196  return res;
197 }
