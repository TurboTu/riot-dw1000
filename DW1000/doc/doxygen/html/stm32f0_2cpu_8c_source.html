79 static void clock_init(void)
80 {
81  /* reset clock configuration register */
82  RCC->CFGR = 0;
83  RCC->CFGR2 = 0;
84 
85  /* disable HSE, CSS and PLL */
86  RCC->CR &= ~(RCC_CR_HSEON | RCC_CR_HSEBYP | RCC_CR_CSSON | RCC_CR_PLLON);
87 
88  /* disable all clock interrupts */
89  RCC->CIR = 0;
90 
91  /* enable the high speed clock source */
92  RCC->CR |= CLOCK_CR_SOURCE;
93 
94  /* wait for the high speed clock to be ready */
95  while (!(RCC->CR & CLOCK_CR_SOURCE_RDY)) {}
96 
97  /* setup the peripheral bus prescalers */
98 
99  /* set HCLK = SYSCLK, so no clock division here */
100  RCC->CFGR |= RCC_CFGR_HPRE_DIV1;
101  /* set PCLK = HCLK, so its not divided */
102  RCC->CFGR |= RCC_CFGR_PPRE_DIV1;
103 
104  /* configure the PLL */
105 
106  /* reset PLL configuration bits */
107  RCC->CFGR &= ~(RCC_CFGR_PLLSRC | RCC_CFGR_PLLXTPRE | RCC_CFGR_PLLMUL);
108  /* set PLL configuration */
109  RCC->CFGR |= CLOCK_PLL_SOURCE | ((((CLOCK_PLL_MUL * CLOCK_PLL_MUL_MUL) - 2) & 0xf) << 18);
110 
111  /* enable PLL again */
112  RCC->CR |= RCC_CR_PLLON;
113  /* wait until PLL is stable */
114  while(!(RCC->CR & RCC_CR_PLLRDY)) {}
115 
116  /* configure flash latency */
117 
118  /* enable pre-fetch buffer and set flash latency to 1 cycle*/
119  FLASH->ACR = FLASH_ACR_PRFTBE | FLASH_ACR_LATENCY;
120 
121  /* configure the sysclock and the peripheral clocks */
122 
123  /* set sysclock to be driven by the PLL clock */
124  RCC->CFGR &= ~RCC_CFGR_SW;
125  RCC->CFGR |= RCC_CFGR_SW_PLL;
126 
127  /* wait for sysclock to be stable */
128  while (!(RCC->CFGR & RCC_CFGR_SWS_PLL)) {}
129 
130 #if CLOCK_DISABLE_HSI
131  /* disable the HSI if we use the HSE */
132  RCC->CR &= ~(RCC_CR_HSION);
