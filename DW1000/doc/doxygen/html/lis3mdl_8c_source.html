43  static inline int16_t _twos_complement(int16_t value)
44 {
45  if (value & MASK_INT16_MSB) {
46  value = ~(value & MASK_INT16_NMSB) + 1;
47  return ~(value & MASK_INT16_NMSB);
48  }
49  else {
50  return value;
51  }

54 int lis3mdl_init(lis3mdl_t *dev,
55  i2c_t i2c,
56  uint8_t address,
57  lis3mdl_xy_mode_t xy_mode,
58  lis3mdl_z_mode_t z_mode,
59  lis3mdl_odr_t odr,
60  lis3mdl_scale_t scale,
61  lis3mdl_op_t op_mode) {
62  uint8_t tmp;
63 
64  dev->i2c = i2c;
65  dev->addr = address;
66 
67  i2c_acquire(dev->i2c);
68 
69  if (i2c_init_master(i2c, I2C_SPEED_NORMAL) < 0) {
70  DEBUG("LIS3MDL: Master initialization failed\n");
71  return -1;
72  }
73 
74  i2c_read_reg(dev->i2c, dev->addr, LIS3DML_WHO_AM_I_REG, &tmp);
75  if (tmp != LIS3MDL_CHIP_ID) {
76  DEBUG("LIS3MDL: Identification failed\n");
77  return -1;
78  }
79 
80  tmp = ( LIS3MDL_MASK_REG1_TEMP_EN /* enable temperature sensor */
81  | xy_mode /* set x-, y-axis operative mode */
82  | odr); /* set output data rate */
83  i2c_write_reg(dev->i2c, dev->addr, LIS3MDL_CTRL_REG1, tmp);
84 
85  /* set Full-scale configuration */
86  i2c_write_reg(dev->i2c, dev->addr, LIS3MDL_CTRL_REG2, scale);
87 
88  /* set continuous-conversion mode */
89  i2c_write_reg(dev->i2c, dev->addr, LIS3MDL_CTRL_REG3, op_mode);
90 
91  /* set z-axis operative mode */
92  i2c_write_reg(dev->i2c, dev->addr, LIS3MDL_CTRL_REG4, z_mode);
93 
94  i2c_release(dev->i2c);
95 
96  return 0;
97 }

99 void lis3mdl_read_mag(lis3mdl_t *dev, lis3mdl_3d_data_t *data)
100 {
101  uint8_t tmp[2] = {0, 0};
102 
103  i2c_acquire(dev->i2c);
104 
105  i2c_read_regs(dev->i2c, dev->addr, LIS3MDL_OUT_X_L_REG, &tmp[0], 2);
106  data->x_axis = (tmp[1] << 8) | tmp[0];
107 
108  i2c_read_regs(dev->i2c, dev->addr, LIS3MDL_OUT_Y_L_REG, &tmp[0], 2);
109  data->y_axis = (tmp[1] << 8) | tmp[0];
110 
111  i2c_read_regs(dev->i2c, dev->addr, LIS3MDL_OUT_Z_L_REG, &tmp[0], 2);
112  data->z_axis = (tmp[1] << 8) | tmp[0];
113 
114  data->x_axis = _twos_complement(data->x_axis);
115  data->y_axis = _twos_complement(data->y_axis);
116  data->z_axis = _twos_complement(data->z_axis);
117 
118  /* Divide the raw data by 1000 to geht [G] := Gauss */
119  data->x_axis /= GAUSS_DIVIDER;
120  data->y_axis /= GAUSS_DIVIDER;
121  data->z_axis /= GAUSS_DIVIDER;
122 
123  i2c_release(dev->i2c);
124 }

126 void lis3mdl_read_temp(lis3mdl_t *dev, int16_t *value)
127 {
128  i2c_acquire(dev->i2c);
129  i2c_read_regs(dev->i2c, dev->addr, LIS3MDL_TEMP_OUT_L_REG, (uint8_t*)value, 2);
130  i2c_release(dev->i2c);
131 
132  *value = _twos_complement(*value);
133 
134  *value = (TEMP_OFFSET + (*value / TEMP_DIVIDER));
135 }
