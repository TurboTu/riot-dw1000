68 static int init_base(uart_t uart, uint32_t baudrate)
69 {
70  switch (uart) {
71 #if UART_0_EN
72  case UART_0:
73  /* this implementation only supports 115200 baud */
74  if (baudrate != 115200) {
75  return UART_NOBAUD;
76  }
77 
78  /* power on UART device and select peripheral clock */
79  UART_0_CLKEN();
80  UART_0_CLKSEL();
81  /* set mode to 8N1 and enable access to divisor latch */
82  UART_0_DEV->LCR = ((0x3 << 0) | (1 << 7));
83  /* set baud rate registers (fixed for now) */
84  UART_0_DEV->DLM = 0;
85  UART_0_DEV->DLL = 13;
86  /* enable FIFOs */
87  UART_0_DEV->FCR = 1;
88  /* select and configure the pin for RX */
89  UART_0_RX_PINSEL &= ~(0x3 << (UART_0_RX_PIN * 2));
90  UART_0_RX_PINSEL |= (UART_0_AF << (UART_0_RX_PIN * 2));
91  UART_0_RX_PINMODE &= ~(0x3 << (UART_0_RX_PIN * 2));
92  UART_0_RX_PINMODE |= (0x2 << (UART_0_RX_PIN * 2));
93  /* select and configure the pin for TX */
94  UART_0_TX_PINSEL &= ~(0x3 << (UART_0_TX_PIN * 2));
95  UART_0_TX_PINSEL |= (UART_0_AF << (UART_0_TX_PIN * 2));
96  UART_0_TX_PINMODE &= ~(0x3 << (UART_0_TX_PIN * 2));
97  UART_0_TX_PINMODE |= (0x2 << (UART_0_TX_PIN * 2));
98  /* disable access to divisor latch */
99  UART_0_DEV->LCR &= ~(1 << 7);
100  break;
101 #endif
102 #if UART_1_EN

140 void uart_write(uart_t uart, const uint8_t *data, size_t len)
141 {
142  LPC_UART_TypeDef *dev;
143 
144  switch (uart) {
145 #if UART_0_EN
146  case UART_0:
147  dev = (LPC_UART_TypeDef *)UART_0_DEV;
148  break;
149 #endif
150 #if UART_1_EN
151  case UART_1:
152  dev = (LPC_UART_TypeDef *)UART_1_DEV;
153  break;
154 #endif
155  default:
156  return;
157  }
158 
159  for (size_t i = 0; i < len; i++) {
160  while (!(dev->LSR & (1 << 5))); /* wait for THRE bit to be set */
161  dev->THR = data[i];
162  }
