68 void _xtimer_periodic_wakeup(uint32_t *last_wakeup, uint32_t period) {
69  xtimer_t timer;
70  mutex_t mutex = MUTEX_INIT;
71 
72  timer.callback = _callback_unlock_mutex;
73  timer.arg = (void*) &mutex;
74 
75  uint32_t target = (*last_wakeup) + period;
76  uint32_t now = _xtimer_now();
77  /* make sure we're not setting a value in the past */
78  if (now < (*last_wakeup)) {
79  /* base timer overflowed between last_wakeup and now */
80  if (!((now < target) && (target < (*last_wakeup)))) {
81  /* target time has already passed */
82  goto out;
83  }
84  }
85  else {
86  /* base timer did not overflow */
87  if ((((*last_wakeup) <= target) && (target <= now))) {
88  /* target time has already passed */
89  goto out;
90  }
91  }
92 
93  /*
94  * For large offsets, set an absolute target time.
95  * As that might cause an underflow, for small offsets, set a relative
96  * target time.
97  * For very small offsets, spin.
98  */
99  /*
100  * Note: last_wakeup _must never_ specify a time in the future after
101  * _xtimer_periodic_sleep returns.
102  * If this happens, last_wakeup may specify a time in the future when the
103  * next call to _xtimer_periodic_sleep is made, which in turn will trigger
104  * the overflow logic above and make the next timer fire too early, causing
105  * last_wakeup to point even further into the future, leading to a chain
106  * reaction.
107  *
108  * tl;dr Don't return too early!
109  */
110  uint32_t offset = target - now;
111  DEBUG("xps, now: %9" PRIu32 ", tgt: %9" PRIu32 ", off: %9" PRIu32 "\n", now, target, offset);
112  if (offset < XTIMER_PERIODIC_SPIN) {
113  _xtimer_spin(offset);
114  }
115  else {
116  if (offset < XTIMER_PERIODIC_RELATIVE) {
117  /* NB: This will overshoot the target by the amount of time it took
118  * to get here from the beginning of xtimer_periodic_wakeup()
119  *
120  * Since interrupts are normally enabled inside this function, this time may
121  * be undeterministic. */
122  target = _xtimer_now() + offset;
123  }
124  mutex_lock(&mutex);
125  DEBUG("xps, abs: %" PRIu32 "\n", target);
126  _xtimer_set_absolute(&timer, target);
127  mutex_lock(&mutex);
128  }
129 out:
130  *last_wakeup = target;
131 }

231 static void _mutex_timeout(void *arg)
232 {
233  mutex_thread_t *mt = (mutex_thread_t *)arg;
234 
235  mt->timeout = 1;
236  list_node_t *node = list_remove(&mt->mutex->queue,
237  (list_node_t *)&mt->thread->rq_entry);
238  if ((node != NULL) && (mt->mutex->queue.next == NULL)) {
239  mt->mutex->queue.next = MUTEX_LOCKED;
