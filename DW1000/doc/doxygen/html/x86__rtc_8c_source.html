95 static void rtc_irq_handler(uint8_t irq_num)
96 {
97  (void) irq_num; /* == PIC_NUM_RTC */
98 
99  uint8_t c = x86_cmos_read(RTC_REG_C);
100  DEBUG("RTC: c = 0x%02x, IRQ=%u, A=%u, P=%u, U=%u\n", c, (c & RTC_REG_C_IRQ) ? 1 : 0,
101  (c & RTC_REG_C_IRQ_ALARM) ? 1 : 0,
102  (c & RTC_REG_C_IRQ_PERIODIC) ? 1 : 0,
103  (c & RTC_REG_C_IRQ_UPDATE) ? 1 : 0);
104  if (!(c & RTC_REG_C_IRQ)) {
105  return;
106  }
107  if (c & RTC_REG_C_IRQ_ALARM) {
108  alarm_callback(c);
109  }
110  if (c & RTC_REG_C_IRQ_PERIODIC) {
111  periodic_callback(c);
112  }
113  if (c & RTC_REG_C_IRQ_UPDATE) {
114  update_callback(c);
115  }
116 }

118 void x86_init_rtc(void)
119 {
120  uint8_t d = x86_cmos_read(RTC_REG_D);
121  valid = (d & RTC_REG_D_VALID) != 0;
122  if (!valid) {
123  puts("Warning: RTC does not work.");
124  return;
125  }
126 
127  x86_cmos_write(RTC_REG_B, x86_cmos_read(RTC_REG_B) & ~RTC_REG_B_INT_MASK);
128  rtc_irq_handler(0);
129  x86_pic_set_handler(PIC_NUM_RTC, &rtc_irq_handler);
130  x86_pic_enable_irq(PIC_NUM_RTC);
131 
132  x86_rtc_data_t now;
133  x86_rtc_read(&now);
134  printf("RTC initialized [%02hhu:%02hhu:%02hhu, %04u-%02hhu-%02hhu]\n",
135  now.hour, now.minute, now.second,
136  now.century * 100 + now.year, now.month, now.day);
137 
138  if (x86_cmos_read(RTC_REG_POST) & (RTC_REG_POST_POWER_LOSS | RTC_REG_POST_TIME_INVALID)) {
139  puts("Warning: RTC time is invalid (power loss?)");
140  }
141 }

158 bool x86_rtc_read(x86_rtc_data_t *dest)
159 {
160  if (!valid) {
161  return false;
162  }
163 
164  unsigned old_status = irq_disable();
165 
166  while (is_update_in_progress()) {
167  __asm__ volatile ("pause");
168  }
169 
170  uint8_t b = x86_cmos_read(RTC_REG_B);
171  do {
172  dest->second = x86_cmos_read(RTC_REG_SECOND);
173  dest->minute = x86_cmos_read(RTC_REG_MINUTE);
174  dest->hour = x86_cmos_read(RTC_REG_HOUR);
175  dest->day = x86_cmos_read(RTC_REG_DAY);
176  dest->month = x86_cmos_read(RTC_REG_MONTH);
177  dest->year = x86_cmos_read(RTC_REG_YEAR);
178  dest->century = bcd2binary(x86_cmos_read(RTC_REG_CENTURY));
179  } while (dest->second != x86_cmos_read(RTC_REG_SECOND));
180  if (dest->century == 0) {
181  dest->century = 20; // safe guess
182  }
183 
184  if (!(b & RTC_REG_B_BIN)) {
185  dest->second = bcd2binary(dest->second);
186  dest->minute = bcd2binary(dest->minute);
187  dest->hour = ((dest->hour & 0x0F) + (((dest->hour & 0x70) / 16) * 10)) | (dest->hour & 0x80);
188  dest->day = bcd2binary(dest->day);
189  dest->month = bcd2binary(dest->month);
190  dest->year = bcd2binary(dest->year);
191  }
192  if (!(b & RTC_REG_B_24H) && (dest->hour & 0x80)) {
193  dest->hour = ((dest->hour & 0x7F) + 12) % 24;
194  }
195 
196  irq_restore(old_status);
197  return true;
198 }

200 bool x86_rtc_set_alarm(const x86_rtc_data_t *when, uint32_t msg_content, kernel_pid_t target_pid, bool allow_replace)
201 {
202  if (!valid) {
203  return false;
204  }
205 
206  unsigned old_status = irq_disable();
207  bool result;
208  if (target_pid == KERNEL_PID_UNDEF) {
209  result = true;
210  alarm_pid = KERNEL_PID_UNDEF;
211 
212  uint8_t b = x86_cmos_read(RTC_REG_B);
213  x86_cmos_write(RTC_REG_B, b & ~RTC_REG_B_INT_ALARM);
214  }
215  else {
216  result = allow_replace || alarm_pid == KERNEL_PID_UNDEF;
217  if (result) {
218  alarm_msg_content = msg_content;
219  alarm_pid = target_pid;
220 
221  uint8_t b = x86_cmos_read(RTC_REG_B);
222  if (b & RTC_REG_B_BIN) {
223  x86_cmos_write(RTC_REG_ALARM_SECOND, when->second);
224  x86_cmos_write(RTC_REG_ALARM_MINUTE, when->minute);
225  x86_cmos_write(RTC_REG_ALARM_HOUR, when->hour);
226  }
227  else {
228  x86_cmos_write(RTC_REG_ALARM_SECOND, binary2bcd(when->second));
229  x86_cmos_write(RTC_REG_ALARM_MINUTE, binary2bcd(when->minute));
230  x86_cmos_write(RTC_REG_ALARM_HOUR, binary2bcd(when->hour));
231  }
232  x86_cmos_write(RTC_REG_B, b | RTC_REG_B_INT_ALARM);
233  }
234  }
235  rtc_irq_handler(0);
236  irq_restore(old_status);
237  return result;
238 }
