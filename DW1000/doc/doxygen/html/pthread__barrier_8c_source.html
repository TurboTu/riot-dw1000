51 int pthread_barrier_wait(pthread_barrier_t *barrier)
52 {
53  /* Idea: the count is decreased by every thread that waits on the barrier.
54  * If the value is bigger than zero afterwards, then the thread has to wait
55  * to be woken up. Once the value reaches zero, everyone gets woken up. */
56 
57  mutex_lock(&barrier->mutex);
58  DEBUG("%s: hit a synchronization barrier. pid=%" PRIkernel_pid"\n",
59  sched_active_thread->name, sched_active_pid);
60 
61  int switch_prio = -1;
62 
63  if (--barrier->count > 0) {
64  /* need to wait for further threads */
65 
66  DEBUG("%s: waiting for %u threads. pid=%" PRIkernel_pid "\n",
67  sched_active_thread->name, barrier->count, sched_active_pid);
68 
69  pthread_barrier_waiting_node_t node;
70  node.pid = sched_active_pid;
71  node.next = barrier->next;
72  node.cont = 0;
73 
74  barrier->next = &node;
75  mutex_unlock(&barrier->mutex);
76 
77  while (1) {
78  /* The mutex is reacquired before checking if we should continue,
79  * so that the waiting thread don't accidentially run before the
80  * wake up loop has ended. Otherwise the thread could run into the
81  * the barrier again before `barrier->count` was reset. */
82  mutex_lock(&barrier->mutex);
83  if (node.cont) {
84  break;
85  }
86  mutex_unlock_and_sleep(&barrier->mutex);
87  }
88  }
89  else {
90  /* all threads have arrived, wake everybody up */
91 
92  DEBUG("%s: waking every other thread up. pid=%" PRIkernel_pid "\n",
93  sched_active_thread->name, sched_active_pid);
94 
95  int count = 1; /* Count number of woken up threads.
96  * The first thread is the current thread. */
97  pthread_barrier_waiting_node_t *next;
98  for (next = barrier->next; next; next = next->next) {
99  ++count;
100  next->cont = 1;
101 
102  thread_t *other = (thread_t *) sched_threads[next->pid];
103  switch_prio = priority_min(switch_prio, other->priority);
104  sched_set_status(other, STATUS_PENDING);
105  }
106  barrier->next = NULL;
107  barrier->count = count;
108  }
109 
110  mutex_unlock(&barrier->mutex);
111 
112  if (switch_prio != -1) {
113  sched_switch(switch_prio);
114  }
115 
116  return 0;
117 }
