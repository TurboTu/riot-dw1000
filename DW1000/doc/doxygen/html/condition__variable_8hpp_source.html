105 cv_status condition_variable::wait_for(unique_lock<mutex>& lock,
106  const std::chrono::duration
107  <Rep, Period>& timeout_duration) {

109  using std::chrono::duration;
110  if (timeout_duration <= timeout_duration.zero()) {
111  return cv_status::timeout;
112  }
113  timex_t timeout, before, after;
114  auto s = duration_cast<seconds>(timeout_duration);
115  timeout.seconds = s.count();
116  timeout.microseconds
117  = (duration_cast<microseconds>(timeout_duration - s)).count();
118  xtimer_now_timex(&before);
119  xtimer_t timer;
120  xtimer_set_wakeup(&timer, timex_uint64(timeout), sched_active_pid);
121  wait(lock);
122  xtimer_now_timex(&after);
123  xtimer_remove(&timer);
124  auto passed = timex_sub(after, before);
125  auto cmp = timex_cmp(passed, timeout);
126  return cmp < 1 ? cv_status::no_timeout : cv_status::timeout;
127 }
