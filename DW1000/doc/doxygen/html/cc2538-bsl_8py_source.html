120  Read a firmware file and store its data ready for device programming.
121 
122  This class will try to guess the file type if python-magic is available.
123

126 
127  In all other cases, the file will be treated as a raw binary file.
128 
129  In both cases, the file's contents are stored in bytes for subsequent
130  usage to program a device or to perform a crc check.
131 
132  Parameters:
133  path -- A str with the path to the firmware file.
134 
135  Attributes:
136  bytes: A bytearray with firmware contents ready to send to the
137  device
138  """

140  firmware_is_hex = False

214  def invoke_bootloader(self, dtr_active_high=False, inverted=False):
215  # Use the DTR and RTS lines to control bootloader and the !RESET pin.
216  # This can automatically invoke the bootloader without the user
217  # having to toggle any pins.
218  #
219  # If inverted is False (default):
220  # DTR: connected to the bootloader pin
221  # RTS: connected to !RESET
222  # If inverted is True, pin connections are the other way round
223  if inverted:
224  set_bootloader_pin = self.sp.setRTS
225  set_reset_pin = self.sp.setDTR
226  else:
227  set_bootloader_pin = self.sp.setDTR
228  set_reset_pin = self.sp.setRTS
229 
230  set_bootloader_pin(1 if not dtr_active_high else 0)
231  set_reset_pin(0)
232  set_reset_pin(1)
233  set_reset_pin(0)
234  # Make sure the pin is still asserted when the chip
235  # comes out of reset. This fixes an issue where
236  # there wasn't enough delay here on Mac.
237  time.sleep(0.002)
238  set_bootloader_pin(0 if not dtr_active_high else 1)
239 
240  # Some boards have a co-processor that detects this sequence here and
241  # then drives the main chip's BSL enable and !RESET pins. Depending on
242  # board design and co-processor behaviour, the !RESET pin may get
243  # asserted after we have finished the sequence here. In this case, we
244  # need a small delay so as to avoid trying to talk to main chip before
245  # it has actually entered its bootloader mode.
246  #
247  # See contiki-os/contiki#1533
248  time.sleep(0.1)

253  def _wait_for_ack(self, info="", timeout=1):
254  stop = time.time() + timeout
255  got = bytearray(2)
256  while got[-2] != 00 or got[-1] not in (CommandInterface.ACK_BYTE,
257  CommandInterface.NACK_BYTE):
258  got += self._read(1)
259  if time.time() > stop:
260  raise CmdException("Timeout waiting for ACK/NACK after '%s'"
261  % (info,))
262 
263  # Our bytearray's length is: 2 initial bytes + 2 bytes for the ACK/NACK
264  # plus a possible N-4 additional (buffered) bytes
265  mdebug(10, "Got %d additional bytes before ACK/NACK" % (len(got) - 4,))
266 
267  # wait for ask
268  ask = got[-1]
269 
270  if ask == CommandInterface.ACK_BYTE:
271  # ACK
272  return 1
273  elif ask == CommandInterface.NACK_BYTE:
274  # NACK
275  mdebug(10, "Target replied with a NACK during %s" % info)
276  return 0
277 
278  # Unknown response
279  mdebug(10, "Unrecognised response 0x%x to %s" % (ask, info))
280  return 0
