133 unsigned _native_rng_read_hq(uint8_t *buf, unsigned num)
134 {
135  DEBUG("_native_rng_read_hq\n");

137 
138  while (num > 0) {
139  _native_syscall_enter();

141  _native_syscall_leave();

144  err(EXIT_FAILURE, "_native_rng_read_hq: read");
145  }
146 
147  num -= r;

149  }
