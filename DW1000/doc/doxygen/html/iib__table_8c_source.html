100 iib_link_set_entry_t *iib_process_hello(kernel_pid_t if_pid, nib_entry_t *nb_elt,
101  uint64_t validity_time, uint8_t is_sym_nb,
102  uint8_t is_lost)
103 {
104  iib_base_entry_t *base_elt;
105  iib_link_set_entry_t *ls_entry = NULL;
106  timex_t now;
107 
108  mutex_lock(&mtx_iib_access);
109 
110  /* Remove link tuple addresses that are included in the Removed Addr List */
111  cleanup_link_sets();
112 
113  LL_FOREACH(iib_base_entry_head, base_elt) {
114  /* Find the link set and two hop set for the interface */
115  if (base_elt->if_pid == if_pid) {
116  break;
117  }
118  }
119 
120  if (base_elt) {
121  xtimer_now_timex(&now);
122 
123  /* Create a new link tuple for the neighbor that originated the hello */
124  ls_entry = update_link_set(base_elt, nb_elt, &now, validity_time, is_sym_nb, is_lost);
125 
126  /* Create new two hop tuples for signaled symmetric neighbors */
127  if (ls_entry) {
128  update_two_hop_set(base_elt, ls_entry, &now, validity_time);
129  }
130  }
131 
132  mutex_unlock(&mtx_iib_access);
133 
134  return ls_entry;
135 }

137 void iib_fill_wr_addresses(kernel_pid_t if_pid, struct rfc5444_writer *wr)
138 {
139  iib_base_entry_t *base_elt;
140  iib_link_set_entry_t *ls_elt;
141  nhdp_addr_entry_t *addr_elt;
142  timex_t now;
143 
144  mutex_lock(&mtx_iib_access);
145 
146  xtimer_now_timex(&now);
147 
148  /* Before adding addresses first update the status of all link tuples */
149  iib_update_lt_status(&now);
150 
151  /* Add all addresses of Link Tuples of the given interface's Link Set to the current HELLO */
152  LL_FOREACH(iib_base_entry_head, base_elt) {
153  if (base_elt->if_pid == if_pid) {
154  LL_FOREACH(base_elt->link_set_head, ls_elt) {
155  if (ls_elt->last_status != IIB_LT_STATUS_PENDING) {
156  /* Exclude addresses from tuples with L_STATUS = PENDING */
157  LL_FOREACH(ls_elt->address_list_head, addr_elt) {
158  if (!NHDP_ADDR_TMP_IN_ANY(addr_elt->address)) {
159  /* Add address to the writers next packet */
160  switch (ls_elt->last_status) {
161  case IIB_LT_STATUS_SYM:
162  nhdp_writer_add_addr(wr, addr_elt->address,
163  RFC5444_ADDRTLV_LINK_STATUS,
164  RFC5444_LINKSTATUS_SYMMETRIC,
165  rfc5444_metric_encode(ls_elt->metric_in),
166  rfc5444_metric_encode(ls_elt->metric_out));
167  addr_elt->address->in_tmp_table = NHDP_ADDR_TMP_SYM;
168  break;
169 
170  case IIB_LT_STATUS_HEARD:
171  nhdp_writer_add_addr(wr, addr_elt->address,
172  RFC5444_ADDRTLV_LINK_STATUS,
173  RFC5444_LINKSTATUS_HEARD,
174  rfc5444_metric_encode(ls_elt->metric_in),
175  rfc5444_metric_encode(ls_elt->metric_out));
176  addr_elt->address->in_tmp_table = NHDP_ADDR_TMP_ANY;
177  break;
178 
179  case IIB_LT_STATUS_UNKNOWN:
180  /* Fall through */
181 
182  case IIB_LT_STATUS_LOST:
183  nhdp_writer_add_addr(wr, addr_elt->address,
184  RFC5444_ADDRTLV_LINK_STATUS,
185  RFC5444_LINKSTATUS_LOST,
186  rfc5444_metric_encode(ls_elt->metric_in),
187  rfc5444_metric_encode(ls_elt->metric_out));
188  addr_elt->address->in_tmp_table = NHDP_ADDR_TMP_ANY;
189  break;
190 
191  case IIB_LT_STATUS_PENDING:
192  /* Pending link tuples are not included */
193  break;
194 
195  default:
196  /* Should not happen */
197  DEBUG("%s:%d in %s: [WARNING] Unknown link tuple status\n",
198  RIOT_FILE_RELATIVE, __LINE__, DEBUG_FUNC);
199  break;
200  }
201  }
202  }
203  }
204  }
205  /* IF's link set found */
206  break;
207  }
208  }
209 
210  mutex_unlock(&mtx_iib_access);
211 }

269 void iib_process_metric_pckt(iib_link_set_entry_t *ls_entry, uint32_t metric_out, uint16_t seq_no)
270 {
271 #if (NHDP_METRIC == NHDP_LMT_HOP_COUNT)
272  /* Nothing to do here */
273  (void)ls_entry;
274  (void)metric_out;
275  (void)seq_no;
276 #elif (NHDP_METRIC == NHDP_LMT_DAT)
277  /* Metric packet processing */
278  if (ls_entry->last_seq_no == 0) {
279  ls_entry->dat_received[0] = 1;
280  ls_entry->dat_total[0] = 1;
281  }
282  /* Don't add values to the queue for duplicate packets */
283  else if (seq_no != ls_entry->last_seq_no) {
284  uint16_t seq_diff;
285  if (seq_no < ls_entry->last_seq_no) {
286  seq_diff = (uint16_t) ((((uint32_t) seq_no) + 0xFFFF) - ls_entry->last_seq_no);
287  }
288  else {
289  seq_diff = seq_no - ls_entry->last_seq_no;
290  }
291  ls_entry->dat_total[0] += (seq_diff > NHDP_SEQNO_RESTART_DETECT) ? 1 : seq_diff;
292  ls_entry->dat_received[0]++;
293  }
294 
295  ls_entry->last_seq_no = seq_no;
296  ls_entry->lost_hellos = 0;
297 
298  if (ls_entry->hello_interval != 0) {
299  timex_t now, i_time;
300  xtimer_now_timex(&now);
301  i_time = timex_from_uint64(rfc5444_timetlv_decode(ls_entry->hello_interval)
302  * US_PER_MS * DAT_HELLO_TIMEOUT_FACTOR);
303  ls_entry->dat_time = timex_add(now, i_time);
304  }
305 
306  /* Refresh metric value for link tuple and corresponding neighbor tuple */
307  if (ls_entry->nb_elt) {
308  if ((metric_out <= ls_entry->nb_elt->metric_out) ||
309  (ls_entry->nb_elt->metric_out == NHDP_METRIC_UNKNOWN)) {
310  /* Better value, use it also for your neighbor */
311  ls_entry->nb_elt->metric_out = metric_out;
312  }
313  else if (ls_entry->metric_out == ls_entry->nb_elt->metric_out){
314  /* The corresponding neighbor tuples metric needs to be updated */
315  iib_base_entry_t *base_elt;
316  iib_link_set_entry_t *ls_elt;
317  ls_entry->nb_elt->metric_out = metric_out;
318  LL_FOREACH(iib_base_entry_head, base_elt) {
319  LL_FOREACH(base_elt->link_set_head, ls_elt) {
320  if ((ls_elt->nb_elt == ls_entry->nb_elt) && (ls_elt != ls_entry)) {
321  if (ls_elt->metric_out < ls_entry->nb_elt->metric_out) {
322  /* Smaller DAT value is better */
323  ls_entry->nb_elt->metric_out = ls_elt->metric_out;
324  }
325  break;
326  }
327  }
328  }
329  }
330  }
331  ls_entry->metric_out = metric_out;
332 #else
333  /* NHDP_METRIC is not set properly */
334  (void)ls_entry;
335  (void)metric_out;
336  (void)seq_no;
337  DEBUG("%s:%d in %s: [WARNING] Unknown NHDP_METRIC setting\n",
338  RIOT_FILE_RELATIVE, __LINE__, DEBUG_FUNC);
339 #endif
340 }

363 static void cleanup_link_sets(void)
364 {
365  iib_base_entry_t *base_elt;
366 
367  /* Loop through all link sets */
368  LL_FOREACH(iib_base_entry_head, base_elt) {
369  /* Loop through all link tuples of the link set */
370  iib_link_set_entry_t *ls_elt, *ls_tmp;
371  LL_FOREACH_SAFE(base_elt->link_set_head, ls_elt, ls_tmp) {
372  /* Loop through all addresses of the link tuples */
373  nhdp_addr_entry_t *lt_elt, *lt_tmp;
374  LL_FOREACH_SAFE(ls_elt->address_list_head, lt_elt, lt_tmp) {
375  if (NHDP_ADDR_TMP_IN_REM_LIST(lt_elt->address)) {
376  /* Remove link tuple address if included in the Removed Addr List */
377  LL_DELETE(ls_elt->address_list_head, lt_elt);
378  nhdp_free_addr_entry(lt_elt);
379  }
380  }
381 
382  /* Remove link tuples with empty address list */
383  if (!ls_elt->address_list_head) {
384  if (ls_elt->last_status == IIB_LT_STATUS_SYM) {
385  /* Remove all two hop entries for the corresponding link tuple */
386  iib_two_hop_set_entry_t *th_elt, *th_tmp;
387  LL_FOREACH_SAFE(base_elt->two_hop_set_head, th_elt, th_tmp) {
388  if (th_elt->ls_elt == ls_elt) {
389  rem_two_hop_entry(base_elt, th_elt);
390  }
391  }
392  }
393 
394  rem_link_set_entry(base_elt, ls_elt);
395  }
396  }
397  }
398 }

403 static iib_link_set_entry_t *update_link_set(iib_base_entry_t *base_entry, nib_entry_t *nb_elt,
404  timex_t *now, uint64_t val_time,
405  uint8_t sym, uint8_t lost)
406 {
407  iib_link_set_entry_t *ls_elt, *ls_tmp;
408  iib_link_set_entry_t *matching_lt = NULL;
409  nhdp_addr_entry_t *lt_elt;
410  timex_t v_time, l_hold;
411  uint8_t matches = 0;
412 
413  /* Loop through every link tuple of the interface to update the link set */
414  LL_FOREACH_SAFE(base_entry->link_set_head, ls_elt, ls_tmp) {
415  /* Loop through all addresses of the link tuple */
416  LL_FOREACH(ls_elt->address_list_head, lt_elt) {
417  if (NHDP_ADDR_TMP_IN_SEND_LIST(lt_elt->address)) {
418  /* If link tuple address matches a sending addr we found a fitting tuple */
419  matches++;
420 
421  if (matches > 1) {
422  /* Multiple matching link tuples, delete the previous one */
423  if (matching_lt->last_status == IIB_LT_STATUS_SYM) {
424  update_nb_tuple_symmetry(base_entry, matching_lt, now);
425  }
426 
427  rem_link_set_entry(base_entry, matching_lt);
428  }
429 
430  matching_lt = ls_elt;
431  break;
432  }
433  }
434  }
435 
436  if (matches > 1) {
437  /* Multiple matching link tuples, reset the last one for reuse */
438  if (matching_lt->last_status == IIB_LT_STATUS_SYM) {
439  update_nb_tuple_symmetry(base_entry, matching_lt, now);
440  }
441 
442  reset_link_set_entry(matching_lt, now, val_time);
443  }
444  else if (matches == 1) {
445  /* A single matching link tuple, only release the address list */
446  release_link_tuple_addresses(matching_lt);
447  }
448  else {
449  /* No single matching link tuple existant, create a new one */
450  matching_lt = add_default_link_set_entry(base_entry, now, val_time);
451 
452  if (!matching_lt) {
453  /* Insufficient memory */
454  return NULL;
455  }
456  }
457 
458  v_time = timex_from_uint64(val_time * US_PER_MS);
459  l_hold = timex_from_uint64(((uint64_t)NHDP_L_HOLD_TIME_MS) * US_PER_MS);
460 
461  /* Set Sending Address List as this tuples address list */
462  matching_lt->address_list_head = nhdp_generate_addr_list_from_tmp(NHDP_ADDR_TMP_SEND_LIST);
463 
464  if (!matching_lt->address_list_head) {
465  /* Insufficient memory */
466  rem_link_set_entry(base_entry, matching_lt);
467  return NULL;
468  }
469 
470  matching_lt->nb_elt = nb_elt;
471 
472  /* Set values dependent on link status */
473  if (sym) {
474  if (matching_lt->last_status != IIB_LT_STATUS_SYM) {
475  /* Set corresponding neighbor tuple to symmetric (Section 13.1 of RFC 6130) */
476  if (matching_lt->nb_elt) {
477  nib_set_nb_entry_sym(matching_lt->nb_elt);
478  }
479  }
480 
481  matching_lt->sym_time = timex_add(*now, v_time);
482  matching_lt->last_status = IIB_LT_STATUS_SYM;
483  }
484  else if (lost) {
485  matching_lt->sym_time.microseconds = 0;
486  matching_lt->sym_time.seconds = 0;
487 
488  if (matching_lt->last_status == IIB_LT_STATUS_SYM) {
489  update_nb_tuple_symmetry(base_entry, matching_lt, now);
490  }
491 
492  if (get_tuple_status(matching_lt, now) == IIB_LT_STATUS_HEARD) {
493  matching_lt->last_status = IIB_LT_STATUS_HEARD;
494  matching_lt->exp_time = timex_add(*now, l_hold);
495  }
496  else {
497  matching_lt->last_status = IIB_LT_STATUS_UNKNOWN;
498  }
499  }
500 
501  /* Set time values */
502  matching_lt->heard_time = get_max_timex(timex_add(*now, v_time), matching_lt->sym_time);
503 
504  if (matching_lt->pending) {
505  /* L_status is PENDING */
506  matching_lt->exp_time = get_max_timex(matching_lt->exp_time, matching_lt->heard_time);
507  }
508  else if (!matching_lt->lost) {
509  if ((timex_cmp(matching_lt->sym_time, *now) == 1)
510  || (timex_cmp(matching_lt->heard_time, *now) == 1)) {
511  /* L_status is HEARD or SYMMETRIC */
512  matching_lt->exp_time = get_max_timex(matching_lt->exp_time,
513  timex_add(matching_lt->heard_time, l_hold));
514  }
515  }
516 
517  return matching_lt;
518 }

525 static void wr_update_ls_status(iib_base_entry_t *base_entry,
526  iib_link_set_entry_t *ls_elt, timex_t *now)
527 {
528  if (timex_cmp(ls_elt->exp_time, *now) != 1) {
529  /* Entry expired and has to be removed */
530  if (ls_elt->last_status == IIB_LT_STATUS_SYM) {
531  update_nb_tuple_symmetry(base_entry, ls_elt, now);
532  }
533 
534  rem_not_heard_nb_tuple(ls_elt, now);
535  rem_link_set_entry(base_entry, ls_elt);
536  }
537  else if ((ls_elt->last_status == IIB_LT_STATUS_SYM)
538  && (timex_cmp(ls_elt->sym_time, *now) != 1)) {
539  /* Status changed from SYMMETRIC to HEARD */
540  update_nb_tuple_symmetry(base_entry, ls_elt, now);
541  ls_elt->last_status = IIB_LT_STATUS_HEARD;
542 
543  if (timex_cmp(ls_elt->heard_time, *now) != 1) {
544  /* New status is LOST (equals IIB_LT_STATUS_UNKNOWN) */
545  rem_not_heard_nb_tuple(ls_elt, now);
546  ls_elt->nb_elt = NULL;
547  ls_elt->last_status = IIB_LT_STATUS_UNKNOWN;
548  }
549  }
550  else if ((ls_elt->last_status == IIB_LT_STATUS_HEARD)
551  && (timex_cmp(ls_elt->heard_time, *now) != 1)) {
552  /* Status changed from HEARD to LOST (equals IIB_LT_STATUS_UNKNOWN) */
553  rem_not_heard_nb_tuple(ls_elt, now);
554  ls_elt->nb_elt = NULL;
555  ls_elt->last_status = IIB_LT_STATUS_UNKNOWN;
556  }
557 }

584 static void reset_link_set_entry(iib_link_set_entry_t *ls_entry, timex_t *now, uint64_t val_time)
585 {
586  timex_t v_time = timex_from_uint64(val_time * US_PER_MS);

634 static int update_two_hop_set(iib_base_entry_t *base_entry, iib_link_set_entry_t *ls_entry,
635  timex_t *now, uint64_t val_time)
636 {
637  /* Check whether a corresponding link tuple was created */
638  if (ls_entry == NULL) {
639  return -1;
640  }
641 
642  /* If the link to the neighbor is still symmetric */
643  if (get_tuple_status(ls_entry, now) == IIB_LT_STATUS_SYM) {
644  iib_two_hop_set_entry_t *ths_elt, *ths_tmp;
645  nhdp_addr_t *addr_elt;
646 
647  /* Loop through all the two hop tuples of the two hop set */
648  LL_FOREACH_SAFE(base_entry->two_hop_set_head, ths_elt, ths_tmp) {
649  if (timex_cmp(ths_elt->exp_time, *now) != 1) {
650  /* Entry is expired, remove it */
651  rem_two_hop_entry(base_entry, ths_elt);
652  }
653  else if (ths_elt->ls_elt == ls_entry) {
654  if (ths_elt->th_nb_addr->in_tmp_table &
655  (NHDP_ADDR_TMP_TH_REM_LIST | NHDP_ADDR_TMP_TH_SYM_LIST)) {
656  rem_two_hop_entry(base_entry, ths_elt);
657  }
658  }
659  }
660 
661  /* Add a new entry for every signaled symmetric neighbor address */
662  LL_FOREACH(nhdp_get_addr_db_head(), addr_elt) {
663  if (NHDP_ADDR_TMP_IN_TH_SYM_LIST(addr_elt)) {
664  if (add_two_hop_entry(base_entry, ls_entry, addr_elt, now, val_time)) {
665  /* No more memory available, return error */
666  return -1;
667  }
668  }
669  }
670  }
671 
672  return 0;
673 }

854 static void dat_metric_refresh(void)
855 {
856  iib_base_entry_t *base_elt;
857  iib_link_set_entry_t *ls_elt;
858  uint32_t metric_temp;
859  double sum_total, sum_rcvd, loss;
860 
861  LL_FOREACH(iib_base_entry_head, base_elt) {
862  LL_FOREACH(base_elt->link_set_head, ls_elt) {
863  sum_rcvd = queue_sum(ls_elt->dat_received);
864  sum_total = queue_sum(ls_elt->dat_total);
865  metric_temp = ls_elt->metric_in;
866 
867  if ((ls_elt->hello_interval != 0) && (ls_elt->lost_hellos > 0)) {
868  /* Compute lost time proportion */
869  loss = (((double)ls_elt->hello_interval) * ((double)ls_elt->lost_hellos))
870  / DAT_MEMORY_LENGTH;
871  if (loss >= 1.0) {
872  sum_rcvd = 0.0;
873  }
874  else {
875  sum_rcvd *= (1.0 - loss);
876  }
877  }
878 
879  if (sum_rcvd < 1.0) {
880  ls_elt->metric_in = NHDP_METRIC_MAXIMUM;
881  }
882  else {
883  loss = sum_total / sum_rcvd;
884  if (loss > DAT_MAXIMUM_LOSS) {
885  loss = DAT_MAXIMUM_LOSS;
886  }
887  ls_elt->metric_in = (const_dat * loss) / (ls_elt->rx_bitrate / DAT_MINIMUM_BITRATE);
888  if (ls_elt->metric_in > NHDP_METRIC_MAXIMUM) {
889  ls_elt->metric_in = NHDP_METRIC_MAXIMUM;
890  }
891  }
892 
893  if (ls_elt->nb_elt) {
894  if (ls_elt->metric_in <= ls_elt->nb_elt->metric_in ||
895  (ls_elt->nb_elt->metric_in == NHDP_METRIC_UNKNOWN)) {
896  /* Better value, use it also for your neighbor */
897  ls_elt->nb_elt->metric_in = ls_elt->metric_in;
898  }
899  else if (metric_temp == ls_elt->nb_elt->metric_in){
900  /* The corresponding neighbor tuples metric needs to be updated */
901  iib_base_entry_t *base_entry;
902  iib_link_set_entry_t *ls_entry;
903  ls_elt->nb_elt->metric_in = ls_elt->metric_in;
904  LL_FOREACH(iib_base_entry_head, base_entry) {
905  LL_FOREACH(base_entry->link_set_head, ls_entry) {
906  if ((ls_elt->nb_elt == ls_entry->nb_elt) && (ls_elt != ls_entry)) {
907  if (ls_entry->metric_in < ls_elt->nb_elt->metric_in) {
908  /* Smaller DAT value is better */
909  ls_elt->nb_elt->metric_in = ls_entry->metric_in;
910  }
911  break;
912  }
913  }
914  }
915  }
916  }
917 
918  queue_rem(ls_elt->dat_received);
919  queue_rem(ls_elt->dat_total);
920  }
921  }
922 }
