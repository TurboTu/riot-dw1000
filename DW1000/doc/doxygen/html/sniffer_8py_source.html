77 def generate_pcap(port, out):
78  # count incoming packets
79  count = 0
80  # output overall PCAP header
81  out.write(pack('<LHHLLLL', MAGIC, MAJOR, MINOR, ZONE, SIG, SNAPLEN,
82  NETWORK))
83  sys.stderr.write("RX: %i\r" % count)
84  while True:
85  line = port.readline().rstrip()
86 
87  pkt_header = re.match(r">? *rftest-rx --- len 0x(\w\w).*",
88  line.decode())
89  if pkt_header:
90  now = time()
91  sec = int(now)
92  usec = int((now - sec) * 1000000)
93  length = int(pkt_header.group(1), 16)
94  out.write(pack('<LLLL', sec, usec, length, length))
95  out.flush()
96  count += 1
97  sys.stderr.write("RX: %i\r" % count)
98  continue
99 
100  pkt_data = re.match(r"(0x\w\w )+", line.decode())
101  if pkt_data:
102  for part in line.decode().split(' '):
103  byte = re.match(r"0x(\w\w)", part)
104  if byte:
105  out.write(pack('<B', int(byte.group(1), 16)))
106  out.flush()
