50 static void to_bytes(float_t f, uint8_t *buff)
51 {
52  int32_t i = (int32_t)(f * 2);
53  buff[0] = (uint8_t)((i >> LM75A_BIT1) & 0xFF);//Most signif. byte
54  buff[1] = (uint8_t)((i & LM75A_BIT1) << LM75A_BIT7);//Least signif. byte
55 }

67 //Read: Sensor --> MC
68 volatile static float_t to_float(uint8_t reg_addr, uint16_t reg_value)
69 {
70  uint16_t sign = reg_value & LM75A_SIGN_BIT_MASK;
71  float_t f_temp = 0.0;
72  float_t factor = 0.0;
73 
74  switch (reg_addr) {
75  case LM75A_OVER_TEMP_REG:
76  case LM75A_THYST_REG:
77  factor = 0.5;
78  break;
79 
80  case LM75A_TEMPERATURE_REG:
81  factor = 0.125;
82  }
83 
84  if (sign) { //the number is negative
85  f_temp = (get_2s_complement(reg_value) & LM75A_DATA_BITS_MASK)
86  * -factor;
87  }
88  else { //the number is positive
89  f_temp = reg_value * factor;
90  }
91  return f_temp;
92 }

94 static void set_register(uint8_t i2c_interface, uint8_t reg_addr, float_t value)
95 {
96  bool status = false;
97  uint8_t tx_buff[2];
98 
99  switch (reg_addr) {
100  case LM75A_OVER_TEMP_REG:
101  case LM75A_THYST_REG:
102  to_bytes(value, tx_buff);
103  status = i2c_write(i2c_interface, LM75A_ADDR, reg_addr, tx_buff, 2);
104  break;
105 
106  case LM75A_CONFIG_REG:
107  tx_buff[0] = (uint8_t) value;
108  status = i2c_write(i2c_interface, LM75A_ADDR, reg_addr, tx_buff, 1);
109  }
110 
111  if (!status) {
112  puts(
113  "[lm75a_tempSensorI2C/lm75A_setRegister]: Slave is not ready !!\n");
114  }
115 }

127 static uint16_t lm75A_get_register_value(uint8_t i2c_interface,
128  uint8_t reg_addr,
129  uint8_t reg_size)
130 {
131  uint8_t rx_buff[reg_size];
132 
133  i2c_clear_buffer(rx_buff, sizeof(rx_buff));
134  if ((reg_size > 0) && (reg_size < 3)) {
135  bool status = i2c_read(i2c_interface, LM75A_ADDR, reg_addr, rx_buff, sizeof(rx_buff));
136  if (!status) { //Slave is not ready
137  puts(
138  "[lm75a_tempSensorI2C/lm75A_getConfigReg]: Slave is not\
139  ready !\n");
140 
141  if (reg_size < 2) {
142  return UCHAR_MAX;
143  }
144  else {
145  return UINT16_MAX;
146  }
147  }
148  else { //Slave acknowledged
149  if (reg_size < 2) {
150  return rx_buff[0];
151  }
152  else {
153  return to_uint16(rx_buff);
154  }
155  }
156  }
157  else {
158  puts("the register size must be less than 2");
159  return UINT16_MAX;
160  }
161 }

204 void lm75A_set_operation_mode(uint8_t op_mode)
205 {
206  uint8_t config_reg = lm75A_get_config_reg();
207 
208  switch (op_mode) {
209  case LM75A_NORMAL_OPERATION_MODE:
210  config_reg &= ~(1 << LM75A_BIT0);
211  break;
212 
213  case LM75A_SHUTDOWN_MODE:
214  config_reg |= (1 << LM75A_BIT0);
215  break;
216 
217  case LM75A_COMPARATOR_MODE:
218  config_reg &= ~(1 << LM75A_BIT1);
219  break;
220 
221  case LM75A_INTERRUPT_MODE:
222  config_reg |= (1 << LM75A_BIT1);
223  break;
224 
225  default:
226  config_reg &= ~(1 << LM75A_BIT0);
227  }
228  set_register(LM75A_I2C_INTERFACE, LM75A_CONFIG_REG, config_reg);
229 }
