59 /************************************************************************************************
60  * @fn adc12_single_conversion
61  * @brief Init ADC12. Do single conversion. Turn off ADC12.
62  * @param none
63  * @return none
64  ************************************************************************************************/
65 uint16_t adc12_single_conversion(uint16_t ref, uint16_t sht, uint16_t channel)
66 {
67  /* Initialize the shared reference module */
68  REFCTL0 |= REFMSTR + ref + REFON; /* Enable internal reference (1.5V or 2.5V) */
69 
70  /* Initialize ADC12_A */
71  ADC12CTL0 = sht + ADC12ON; /* Set sample time */
72  ADC12CTL1 = ADC12SHP; /* Enable sample timer */
73  ADC12MCTL0 = ADC12SREF_1 + channel; /* ADC input channel */
74  ADC12IE = 0x001; /* ADC_IFG upon conv result-ADCMEMO */
75  irq_enable();
76 
77  /* Wait 66us to allow internal reference to settle */
78  xtimer_usleep(66);
79 
80  /* Start ADC12 */
81  ADC12CTL0 |= ADC12ENC;
82 
83  /* Clear data ready flag */
84  adc12_data_ready = 0;
85 
86  /* Sampling and conversion start */
87  ADC12CTL0 |= ADC12SC;
88 
89  /* Wait until ADC12 has finished */
90  xtimer_usleep(150);
91 
92  while (!adc12_data_ready);
93 
94  /* Shut down ADC12 */
95  ADC12CTL0 &= ~(ADC12ENC | ADC12SC | sht);
96  ADC12CTL0 &= ~ADC12ON;
97 
98  /* Shut down reference voltage */
99  REFCTL0 &= ~(REFMSTR + ref + REFON);
100 
101  ADC12IE = 0;
102 
103  /* Return ADC result */
104  return adc12_result;
105 }
