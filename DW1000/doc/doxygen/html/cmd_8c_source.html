180 static inline int _dehex(char c, int default_)
181 {
182  if ('0' <= c && c <= '9') {
183  return c - '0';
184  }
185  else if ('A' <= c && c <= 'F') {
186  return c - 'A' + 10;
187  }
188  else if ('a' <= c && c <= 'f') {
189  return c - 'a' + 10;
190  }
191  else {
192  return default_;

196 static int _parse_addr(uint8_t *out, size_t out_len, const char *in)
197 {
198  const char *end_str = in;
199  uint8_t *out_end = out;
200  size_t count = 0;
201  int assert_cell = 1;
202 
203  if (!in || !*in) {
204  return 0;
205  }
206  while (end_str[1]) {
207  ++end_str;
208  }
209 
210  while (end_str >= in) {
211  int a = 0, b = _dehex(*end_str--, -1);
212  if (b < 0) {
213  if (assert_cell) {
214  return 0;
215  }
216  else {
217  assert_cell = 1;
218  continue;
219  }
220  }
221  assert_cell = 0;
222 
223  if (end_str >= in) {
224  a = _dehex(*end_str--, 0);
225  }
226 
227  if (++count > out_len) {
228  return 0;
229  }
230  *out_end++ = (a << 4) | b;
231  }
232  if (assert_cell) {
233  return 0;
234  }
235  /* out is reversed */
236 
237  while (out < --out_end) {
238  uint8_t tmp = *out_end;
239  *out_end = *out;
240  *out++ = tmp;
241  }
242 
243  return count;
244 }

246 static int send(int iface, le_uint16_t dst_pan, uint8_t *dst, size_t dst_len,
247  char *data)
248 {
249  int res;
250  netdev2_ieee802154_t *dev;
251  const size_t count = 2; /* mhr + payload */
252  struct iovec vector[count];
253  uint8_t *src;
254  size_t src_len;
255  uint8_t mhr[IEEE802154_MAX_HDR_LEN];
256  uint8_t flags;
257  le_uint16_t src_pan;
258 
259  if (((unsigned)iface) > (AT86RF2XX_NUM - 1)) {
260  printf("txtsnd: %d is not an interface\n", iface);
261  return 1;
262  }
263 
264  dev = (netdev2_ieee802154_t *)&devs[iface];
265  flags = (uint8_t)(dev->flags & NETDEV2_IEEE802154_SEND_MASK);
266  flags |= IEEE802154_FCF_TYPE_DATA;
267  vector[1].iov_base = data;
268  vector[1].iov_len = strlen(data);
269  src_pan = byteorder_btols(byteorder_htons(dev->pan));
270  if (dst_pan.u16 == 0) {
271  dst_pan = src_pan;
272  }
273  if (dev->flags & NETDEV2_IEEE802154_SRC_MODE_LONG) {
274  src_len = 8;
275  src = dev->long_addr;
276  }
277  else {
278  src_len = 2;
279  src = dev->short_addr;
280  }
281  /* fill MAC header, seq should be set by device */
282  if ((res = ieee802154_set_frame_hdr(mhr, src, src_len,
283  dst, dst_len,
284  src_pan, dst_pan,
285  flags, dev->seq++)) < 0) {
286  puts("txtsnd: Error preperaring frame");
287  return 1;
288  }
289  vector[0].iov_base = mhr;
290  vector[0].iov_len = (size_t)res;
291  res = dev->netdev.driver->send((netdev2_t *)dev, vector, count);
292  if (res < 0) {
293  puts("txtsnd: Error on sending");
294  return 1;
295  }
296  else {
297  printf("txtsnd: send %u bytes to ", (unsigned)vector[1].iov_len);
298  print_addr(dst, dst_len);
299  printf(" (PAN: ");
300  print_addr((uint8_t *)&dst_pan, sizeof(dst_pan));
301  puts(")");
302  }
303  return 0;
304 }
