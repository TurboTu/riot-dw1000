90 int timer_init(tim_t tim, unsigned long freq, timer_cb_t cb, void *arg)
91 {
92  DEBUG("timer.c: freq = %ld\n", freq);
93  uint8_t pre = 0;
94 
95  /* make sure given device is valid */
96  if (tim >= TIMER_NUMOF) {
97  return -1;
98  }
99 
100  /* figure out if freq is applicable */
101  for (; pre < PRESCALE_NUMOF; pre++) {
102  if ((CLOCK_CORECLOCK >> prescalers[pre]) == freq) {
103  break;
104  }
105  }
106  if (pre == PRESCALE_NUMOF) {
107  DEBUG("timer.c: prescaling failed!\n");
108  return -1;
109  }
110 
111  /* stop and reset timer */
112  ctx[tim].dev->CRA = 0;
113  ctx[tim].dev->CRB = 0;
114  ctx[tim].dev->CRC = 0;
115  ctx[tim].dev->CNT = 0;
116 
117  /* save interrupt context and timer mode */
118  ctx[tim].cb = cb;
119  ctx[tim].arg = arg;
120  ctx[tim].mode = (pre + 1);
121 
122  /* enable timer with calculated prescaler */
123  ctx[tim].dev->CRB = (pre + 1);
124  DEBUG("timer.c: prescaler set at %d\n", pre + 1);
125 
126  return 0;
127 }
