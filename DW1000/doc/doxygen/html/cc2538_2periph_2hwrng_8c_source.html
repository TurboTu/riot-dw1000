26 void hwrng_init(void)
27 {
28  uint16_t seed = 0;
29  int i;
30 
31  /* Make sure the RNG is on */
32  SOC_ADC->cc2538_adc_adccon1.ADCCON1bits.RCTRL = 0;
33 
34  /* Enable clock for the RF Core */
35  SYS_CTRL_RCGCRFC = 1;
36 
37  /* Wait for the clock ungating to take effect */
38  while (SYS_CTRL_RCGCRFC != 1);
39 
40  /* Infinite RX - FRMCTRL0[3:2] = 10. This will mess with radio operation */
41  RFCORE_XREG_FRMCTRL0 = 0x00000008;
42 
43  /* Turn RF on */
44  RFCORE_SFR_RFST = ISRXON;
45 
46  /*
47  * Wait until "the chip has been in RX long enough for the transients to
48  * have died out. A convenient way to do this is to wait for the RSSI-valid
49  * signal to go high."
50  */
51  while (!RFCORE->XREG_RSSISTATbits.RSSI_VALID);
52 
53  /*
54  * Form the seed by concatenating bits from IF_ADC in the RF receive path.
55  * Keep sampling until we have read at least 16 bits AND the seed is valid
56  *
57  * Invalid seeds are 0 and 0x8003 and should not be used.
58  */
59  for (i = 0; (i < 8) || (seed == 0) || (seed == 0x8003); i++) {
60  seed <<= 2;
61  seed ^= RFCORE->XREG_RFRND;
62  }
63 
64  /* Seed the high byte first: */
65  SOC_ADC_RNDL = (seed >> 8) & 0xff;
66  SOC_ADC_RNDL = seed & 0xff;
67 
68  /* Turn RF off: */
69  RFCORE_SFR_RFST = ISRFOFF;
70 }
