114 static void clk_init(void)
115 {
116  /* Reset the RCC clock configuration to the default reset state(for debug purpose) */
117  /* Set HSION bit */
118  RCC->CR |= (uint32_t)0x00000001;
119  /* Reset SW, HPRE, PPRE1, PPRE2, ADCPRE and MCO bits */
120  RCC->CFGR &= (uint32_t)0xF0FF0000;
121  /* Reset HSEON, CSSON and PLLON bits */
122  RCC->CR &= (uint32_t)0xFEF6FFFF;
123  /* Reset HSEBYP bit */
124  RCC->CR &= (uint32_t)0xFFFBFFFF;
125  /* Reset PLLSRC, PLLXTPRE, PLLMUL and USBPRE/OTGFSPRE bits */
126  RCC->CFGR &= (uint32_t)0xFF80FFFF;
127  /* Disable all interrupts and clear pending bits */
128  RCC->CIR = (uint32_t)0x009F0000;
129 
130  /* SYSCLK, HCLK, PCLK2 and PCLK1 configuration */
131  /* Enable high speed clock source */
132  RCC->CR |= ((uint32_t)CLOCK_CR_SOURCE);
133  /* Wait till the high speed clock source is ready
134  * NOTE: the MCU will stay here forever if you use an external clock source and it's not connected */
135  while ((RCC->CR & CLOCK_CR_SOURCE_RDY) == 0) {}
136  /* Enable prefetch buffer and set flash wait state */
137  FLASH->ACR = (uint32_t)(FLASH_ACR_PRFTBE | FLASH_LATENCY);
138  /* HCLK = SYSCLK */
139  RCC->CFGR |= (uint32_t)CLOCK_AHB_DIV;
140  /* PCLK2 = HCLK */
141  RCC->CFGR |= (uint32_t)CLOCK_APB2_DIV;
142  /* PCLK1 = HCLK */
143  RCC->CFGR |= (uint32_t)CLOCK_APB1_DIV;
144 
145 #if (CLOCK_USE_PLL == 1)
146  /* PLL configuration: PLLCLK = CLOCK_SOURCE / PLL_DIV * PLL_MUL */
147  RCC->CFGR |= (uint32_t)(CLOCK_PLL_SOURCE | CLOCK_PLL_DIVMSK | ((CLOCK_PLL_MUL - 2) << 18));
148  /* Enable PLL */
149  RCC->CR |= RCC_CR_PLLON;
150  /* Wait till PLL is ready */
151  while ((RCC->CR & RCC_CR_PLLRDY) == 0) {}
152 #endif
153 
154  /* Select the system clock source */
155  RCC->CFGR &= ~((uint32_t)(RCC_CFGR_SW));
156  RCC->CFGR |= (uint32_t)CLOCK_CFGR_SW;
157  /* Wait till selected system clock source is ready */
158  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS) != CLOCK_CFGR_SW_RDY) {}
159 
160 #if CLOCK_DISABLE_HSI
161  RCC->CR &= ~(RCC_CR_HSION);
162  while ((RCC->CR & RCC_CR_HSIRDY) != 0) {}
163 #endif
164 }
