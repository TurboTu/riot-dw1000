37 static inline uint16_t _calc_csum(gnrc_pktsnip_t *hdr,
38  gnrc_pktsnip_t *pseudo_hdr,
39  gnrc_pktsnip_t *payload)
40 {
41  uint16_t csum = 0;
42  uint16_t len = (uint16_t)hdr->size;
43 
44  while (payload && (payload != hdr)) {
45  csum = inet_csum_slice(csum, payload->data, payload->size, len);
46  len += (uint16_t)payload->size;
47  payload = payload->next;
48  }
49 
50  csum = inet_csum(csum, hdr->data, hdr->size);
51  csum = ipv6_hdr_inet_csum(csum, pseudo_hdr->data, PROTNUM_ICMPV6, len);
52 
53  return ~csum;
54 }

56 void gnrc_icmpv6_demux(kernel_pid_t iface, gnrc_pktsnip_t *pkt)
57 {
58  gnrc_pktsnip_t *icmpv6, *ipv6;
59  icmpv6_hdr_t *hdr;
60 
61  icmpv6 = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_ICMPV6);
62 
63  assert(icmpv6 != NULL);
64 
65  /* there can be extension headers between IPv6 and ICMPv6 header so we have
66  * to search it */
67  ipv6 = gnrc_pktsnip_search_type(icmpv6, GNRC_NETTYPE_IPV6);
68 
69  assert(ipv6 != NULL);
70 
71  if (icmpv6->size < sizeof(icmpv6_hdr_t)) {
72  DEBUG("icmpv6: packet too short.\n");
73  return;
74  }
75 
76  /* Note: size will be checked again in gnrc_icmpv6_echo_req_handle,
77  gnrc_ndp_rtr_sol_handle, and others */
78 
79  hdr = (icmpv6_hdr_t *)icmpv6->data;
80 
81  if (_calc_csum(icmpv6, ipv6, pkt)) {
82  DEBUG("icmpv6: wrong checksum.\n");
83  /* don't release: IPv6 does this */
84  return;
85  }
86 
87  switch (hdr->type) {
88  /* TODO: handle ICMPv6 errors */
89 #ifdef MODULE_GNRC_ICMPV6_ECHO
90  case ICMPV6_ECHO_REQ:
91  DEBUG("icmpv6: handle echo request.\n");
92  gnrc_icmpv6_echo_req_handle(iface, (ipv6_hdr_t *)ipv6->data,
93  (icmpv6_echo_t *)hdr, icmpv6->size);
94  break;
95 #endif
96 
97 #if (defined(MODULE_GNRC_NDP_ROUTER) || defined(MODULE_GNRC_SIXLOWPAN_ND_ROUTER))
98  case ICMPV6_RTR_SOL:
99  DEBUG("icmpv6: router solicitation received\n");
100  gnrc_ndp_rtr_sol_handle(iface, pkt, ipv6->data, (ndp_rtr_sol_t *)hdr,
101  icmpv6->size);
102  break;
103 #endif
104 
105 #ifdef MODULE_GNRC_NDP
106  case ICMPV6_RTR_ADV:
107  DEBUG("icmpv6: router advertisement received\n");
108  gnrc_ndp_rtr_adv_handle(iface, pkt, ipv6->data, (ndp_rtr_adv_t *)hdr,
109  icmpv6->size);
110  break;
111 
112  case ICMPV6_NBR_SOL:
113  DEBUG("icmpv6: neighbor solicitation received\n");
114  gnrc_ndp_nbr_sol_handle(iface, pkt, ipv6->data, (ndp_nbr_sol_t *)hdr,
115  icmpv6->size);
116  break;
117 
118  case ICMPV6_NBR_ADV:
119  DEBUG("icmpv6: neighbor advertisement received\n");
120  gnrc_ndp_nbr_adv_handle(iface, pkt, ipv6->data, (ndp_nbr_adv_t *)hdr,
121  icmpv6->size);
122  break;
123 #endif
124 
125  case ICMPV6_REDIRECT:
126  DEBUG("icmpv6: redirect message received\n");
127  /* TODO */
128  break;
129 
130  default:
131  DEBUG("icmpv6: unknown type field %u\n", hdr->type);
132  break;
133  }
134 
135  /* ICMPv6-all will be send in gnrc_ipv6.c so only dispatch of subtypes is
136  * needed */
137  if (!gnrc_netapi_dispatch_receive(GNRC_NETTYPE_ICMPV6, hdr->type, pkt)) {
138  DEBUG("icmpv6: no one interested in type %d\n", hdr->type);
139  gnrc_pktbuf_release(pkt);
140  }
141 }

166 int gnrc_icmpv6_calc_csum(gnrc_pktsnip_t *hdr, gnrc_pktsnip_t *pseudo_hdr)
167 {
168  uint32_t csum = 0;
169 
170  if (hdr == NULL) {
171  return -EFAULT;
172  }
173  if (hdr->type != GNRC_NETTYPE_ICMPV6) {
174  return -EBADMSG;
175  }
176 
177  csum = _calc_csum(hdr, pseudo_hdr, hdr->next);
178 
179  if (csum == 0) {
180  return -ENOENT;
181  }
182 
183  ((icmpv6_hdr_t *)hdr->data)->csum = byteorder_htons(csum);
184 
185  return 0;
186 }
