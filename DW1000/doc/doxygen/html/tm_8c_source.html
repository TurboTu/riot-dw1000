52 void tm_get_wyday(int year, int mon, int mday, int *wday, int *yday)
53 {
54  int is_leap_year = tm_is_leap_year(year);
55  *yday = TM_MON_DAYS_ACCU[mon] + mday + (mon <= TM_MON_FEB ? 0 : is_leap_year) - 1;
56  int jan1 = tm_doomsday(year) - 2 - is_leap_year;

68  return 0;

72  return 0;
