163 static inline uint64_t X86_CR_ATTR msr_read(uint32_t msr)
164 {
165  uint32_t eax, edx;
166  __asm__ volatile (
167  "rdmsr"
168  : "=a"(eax), "=d"(edx)
169  : "c"(msr)
170  );
171  return (((uint64_t) edx) << 32) | eax;
172 }
