31 void pm_set(unsigned mode)
32 {
33 /* I just copied it from stm32f1/2/4, but I suppose it would work for the
34  * others... /KS */
35 #if defined(CPU_FAM_STM32F1) || defined(CPU_FAM_STM32F2) || defined(CPU_FAM_STM32F4)
36  switch (mode) {
37  case 0:
38  /* Set PDDS to enter standby mode on deepsleep and clear flags */
39  PWR->CR |= (PWR_CR_PDDS | PWR_CR_CWUF | PWR_CR_CSBF);
40  /* Enable WKUP pin to use for wakeup from standby mode */
41  PWR->CSR |= PWR_CSR_EWUP;
42  /* Set SLEEPDEEP bit of system control block */
43  SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
44  break;
45  case 1: /* STM Stop mode */
46  /* Clear PDDS and LPDS bits to enter stop mode on */
47  /* deepsleep with voltage regulator on */
48  PWR->CR &= ~(PWR_CR_PDDS | PWR_CR_LPDS);
49  /* Set SLEEPDEEP bit of system control block */
50  SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
51  break;
52  case 2: /* STM Sleep mode */
53  /* Reset SLEEPDEEP bit of system control block */
54  SCB->SCR &= ~(SCB_SCR_SLEEPDEEP_Msk);
55  break;
56  }
57 #endif
58 
59  /* Executes a device DSB (Data Synchronization Barrier) */
60  __DSB();
61  /* Enter standby mode */
62  __WFI();
63 }
