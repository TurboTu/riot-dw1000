35 void videoram_putc(char c)
36 {
37  static unsigned pos = 0;
38  if (c == '\n') {
39  pos += 80;
40  }
41  else if (c == '\r') {
42  pos = (pos / 80) * 80;
43  }
44  else {
45  VIDEORAM[2 * pos + 0] = c;
46  VIDEORAM[2 * pos + 1] = (char) 0x9E; /* light blue bg, light yellow fg */
47  ++pos;
48  }
49  if (pos >= 25 * 80) {
50  pos = 0;
51  }
52 }
