90 int _ccnl_content(int argc, char **argv)
91 {
92  char *body = (char*) _default_content;
93  int arg_len = strlen(_default_content) + 1;
94  int offs = CCNL_MAX_PACKET_SIZE;
95  if (argc < 2) {
96  _content_usage(argv[0]);
97  return -1;
98  }
99 
100  if (argc > 2) {
101  char buf[BUF_SIZE];
102  memset(buf, ' ', BUF_SIZE);
103  char *buf_ptr = buf;
104  for (int i = 2; (i < argc) && (buf_ptr < (buf + BUF_SIZE)); i++) {
105  arg_len = strlen(argv[i]);
106  if ((buf_ptr + arg_len) > (buf + BUF_SIZE)) {
107  arg_len = (buf + BUF_SIZE) - buf_ptr;
108  }
109  strncpy(buf_ptr, argv[i], arg_len);
110  buf_ptr += arg_len + 1;
111  }
112  *buf_ptr = '\0';
113  body = buf;
114  arg_len = strlen(body);
115  }
116 
117  int suite = CCNL_SUITE_NDNTLV;
118 
119  struct ccnl_prefix_s *prefix = ccnl_URItoPrefix(argv[1], suite, NULL, NULL);
120 
121  arg_len = ccnl_ndntlv_prependContent(prefix, (unsigned char*) body, arg_len, NULL, NULL, &offs, _out);
122 
123  unsigned char *olddata;
124  unsigned char *data = olddata = _out + offs;
125 
126  int len;
127  unsigned typ;
128 
129  if (ccnl_ndntlv_dehead(&data, &arg_len, (int*) &typ, &len) ||
130  typ != NDN_TLV_Data) {
131  return -1;
132  }
133 
134  struct ccnl_content_s *c = 0;
135  struct ccnl_pkt_s *pk = ccnl_ndntlv_bytes2pkt(typ, olddata, &data, &arg_len);
136  c = ccnl_content_new(&ccnl_relay, &pk);
137  ccnl_content_add2cache(&ccnl_relay, c);
138  c->flags |= CCNL_CONTENT_FLAGS_STATIC;
139 
140  return 0;
141 }

143 static struct ccnl_face_s *_intern_face_get(char *addr_str)
144 {
145  /* initialize address with 0xFF for broadcast */
146  uint8_t relay_addr[MAX_ADDR_LEN];
147  memset(relay_addr, UINT8_MAX, MAX_ADDR_LEN);
148  size_t addr_len = gnrc_netif_addr_from_str(relay_addr, sizeof(relay_addr), addr_str);
149 
150  if (addr_len == 0) {
151  printf("Error: %s is not a valid link layer address\n", addr_str);
152  return NULL;
153  }
154 
155  sockunion sun;
156  sun.sa.sa_family = AF_PACKET;
157  memcpy(&(sun.linklayer.sll_addr), relay_addr, addr_len);
158  sun.linklayer.sll_halen = addr_len;
159  sun.linklayer.sll_protocol = htons(ETHERTYPE_NDN);
160 
161  /* TODO: set correct interface instead of always 0 */
162  struct ccnl_face_s *fibface = ccnl_get_face_or_create(&ccnl_relay, 0, &sun.sa, sizeof(sun.linklayer));
163 
164  return fibface;

197 int _ccnl_interest(int argc, char **argv)
198 {
199  if (argc < 2) {
200  _interest_usage(argv[0]);
201  return -1;
202  }
203 
204  if (argc > 2) {
205  if (_intern_fib_add(argv[1], argv[2]) < 0) {
206  _interest_usage(argv[0]);
207  return -1;
208  }
209  }
210 
211  memset(_int_buf, '\0', BUF_SIZE);
212  memset(_cont_buf, '\0', BUF_SIZE);
213  for (int cnt = 0; cnt < CCNL_INTEREST_RETRIES; cnt++) {
214  gnrc_netreg_entry_t _ne =
215  GNRC_NETREG_ENTRY_INIT_PID(GNRC_NETREG_DEMUX_CTX_ALL,
216  sched_active_pid);
217  /* register for content chunks */
218  gnrc_netreg_register(GNRC_NETTYPE_CCN_CHUNK, &_ne);
219 
220  struct ccnl_prefix_s *prefix = ccnl_URItoPrefix(argv[1], CCNL_SUITE_NDNTLV, NULL, 0);
221  ccnl_send_interest(prefix, _int_buf, BUF_SIZE);
222  if (ccnl_wait_for_chunk(_cont_buf, BUF_SIZE, 0) > 0) {
223  gnrc_netreg_unregister(GNRC_NETTYPE_CCN_CHUNK, &_ne);
224  printf("Content received: %s\n", _cont_buf);
225  return 0;
226  }
227  ccnl_free(prefix);
228  gnrc_netreg_unregister(GNRC_NETTYPE_CCN_CHUNK, &_ne);
229  }
230  printf("Timeout! No content received in response to the Interest for %s.\n", argv[1]);
231 
232  return -1;
233 }

249 int _ccnl_fib(int argc, char **argv)
250 {
251  if (argc < 2) {
252  ccnl_fib_show(&ccnl_relay);
253  }
254  else if ((argc == 3) && (strncmp(argv[1], "del", 3) == 0)) {
255  int suite = CCNL_SUITE_NDNTLV;
256  if (strchr(argv[2], '/')) {
257  struct ccnl_prefix_s *prefix = ccnl_URItoPrefix(argv[2], suite, NULL, 0);
258  if (!prefix) {
259  puts("Error: prefix could not be created!");
260  return -1;
261  }
262  int res = ccnl_fib_rem_entry(&ccnl_relay, prefix, NULL);
263  free_prefix(prefix);
264  return res;
265  }
266  else {
267  struct ccnl_face_s *face = _intern_face_get(argv[2]);
268  if (face == NULL) {
269  printf("There is no face for address %s\n", argv[1]);
270  return -1;
271  }
272  int res = ccnl_fib_rem_entry(&ccnl_relay, NULL, face);
273  return res;
274  }
275  }
276  else if ((argc == 4) && (strncmp(argv[1], "add", 3) == 0)) {
277  if (_intern_fib_add(argv[2], argv[3]) < 0) {
278  _ccnl_fib_usage(argv[0]);
279  return -1;
280  }
281  }
282  else {
283  _ccnl_fib_usage(argv[0]);
284  return -1;
285  }
286  return 0;
