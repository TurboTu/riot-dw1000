95 char *thread_arch_stack_init(thread_task_func_t task_func, void *arg,
96  void *stack_start, int stack_size)
97 {
98  uint16_t tmp_adress;
99  uint8_t *stk;
100 
101  /* AVR uses 16 Bit or two 8 Bit registers for storing pointers*/
102  stk = (uint8_t *)((uintptr_t)stack_start + stack_size);
103 
104  /* put marker on stack */
105  stk--;
106  *stk = (uint8_t) 0xAF;
107  stk--;
108  *stk = (uint8_t) 0xFE;
109 
110  /* save sched_task_exit */
111  stk--;
112  tmp_adress = (uint16_t) sched_task_exit;
113  *stk = (uint8_t)(tmp_adress & (uint16_t) 0x00ff);
114  stk--;
115  tmp_adress >>= 8;
116  *stk = (uint8_t)(tmp_adress & (uint16_t) 0x00ff);
117 
118 #if FLASHEND > 0x1ffff
119  /* Devices with more than 128kb FLASH use a 17 bit PC, we set whole the top byte forcibly to 0 */
120  stk--;
121  *stk = (uint8_t) 0x00;
122 #endif
123 
124  /* save address to task_func in place of the program counter */
125  stk--;
126  tmp_adress = (uint16_t) task_func;
127  *stk = (uint8_t)(tmp_adress & (uint16_t) 0x00ff);
128  stk--;
129  tmp_adress >>= 8;
130  *stk = (uint8_t)(tmp_adress & (uint16_t) 0x00ff);
131 
132 #if FLASHEND > 0x1ffff
133  /* Devices with more than 128kb FLASH use a 17 bit PC, we set whole the top byte forcibly to 0 */
134  stk--;
135  *stk = (uint8_t) 0x00;
136 #endif
137 
138 
139  /* r0 */
140  stk--;
141  *stk = (uint8_t) 0x00;
142 
143  /* status register (with interrupts enabled) */
144  stk--;
145  *stk = (uint8_t) 0x80;
146 
147 #if defined(EIND)
148  stk--;
149  *stk = (uint8_t) 0x00;
150 #endif
151 #if defined(RAMPZ)
152  stk--;
153  *stk = (uint8_t) 0x00;
154 #endif
155 
156  /* r1 - has always to be 0 */
157  stk--;
158  *stk = (uint8_t) 0x00;
159  /*
160  * Space for registers r2 -r23
161  *
162  * use loop for better readability, the compiler unrolls anyways
163  */
164 
165  int i;
166 
167  for (i = 2; i <= 23 ; i++) {
168  stk--;
169  *stk = (uint8_t) 0;
170  }
171 
172  /*
173  * In accordance with the AVR calling conventions *arg has to be inside
174  * r24 and r25
175  * */
176  stk--;
177  tmp_adress = (uint16_t) arg;
178  *stk = (uint8_t)(tmp_adress & (uint16_t) 0x00ff);
179  stk--;
180  tmp_adress >>= 8;
181  *stk = (uint8_t)(tmp_adress & (uint16_t) 0x00ff);
182 
183  /*
184  * Space for registers r26-r31
185  */
186  for (i = 26; i <= 31; i++) {
187  stk--;
188  *stk = (uint8_t) i;
189  }
190 
191  stk--;
192  return (char *) stk;
193 }
