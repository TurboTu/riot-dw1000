201 size_t fmt_s16_dfp(char *out, int16_t val, unsigned fp_digits)
202 {
203  int16_t absolute, divider;
204  size_t pos = 0;
205  size_t div_len, len;
206  unsigned e;
207  char tmp[4];
208 
209  if (fp_digits > 4) {
210  return 0;
211  }
212  if (fp_digits == 0) {
213  return fmt_s16_dec(out, val);
214  }
215  if (val < 0) {
216  if (out) {
217  out[pos++] = '-';
218  }
219  val *= -1;
220  }
221 
222  e = pwr(10, fp_digits);
223  absolute = (val / (int)e);
224  divider = val - (absolute * e);
225 
226  pos += fmt_s16_dec(&out[pos], absolute);
227 
228  if (!out) {
229  return pos + 1 + fp_digits; /* abs len + decimal point + divider */
230  }
231 
232  out[pos++] = '.';
233  len = pos + fp_digits;
234  div_len = fmt_s16_dec(tmp, divider);
235 
236  while (pos < (len - div_len)) {
237  out[pos++] = '0';
238  }
239  for (size_t i = 0; i < div_len; i++) {
240  out[pos++] = tmp[i];
241  }
242 
243  return pos;
244 }
