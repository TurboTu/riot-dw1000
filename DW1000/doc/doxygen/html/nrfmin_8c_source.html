134 static void go_idle(void)
135 {
136  /* set device into basic disabled state */
137  NRF_RADIO->EVENTS_DISABLED = 0;
138  NRF_RADIO->TASKS_DISABLE = 1;
139  while (NRF_RADIO->EVENTS_DISABLED == 0) {}

142  state = STATE_IDLE;

152 static void goto_target_state(void)
153 {
154  go_idle();
155 
156  if ((target_state == STATE_RX) && (rx_buf.pkt.hdr.len == 0)) {
157  /* set receive buffer and our own address */
158  rx_lock = 1;
159  NRF_RADIO->PACKETPTR = (uint32_t)(&rx_buf);
160  NRF_RADIO->BASE0 = (CONF_ADDR_BASE | my_addr);
161  /* goto RX mode */
162  NRF_RADIO->TASKS_RXEN = 1;
163  state = STATE_RX;
164  }
165 
166  if (target_state == STATE_OFF) {
167  NRF_RADIO->POWER = 0;
168  state = STATE_OFF;
169  }

301 void isr_radio(void)
302 {
303  if (NRF_RADIO->EVENTS_END == 1) {
304  NRF_RADIO->EVENTS_END = 0;
305  /* did we just send or receive something? */
306  if (state == STATE_RX) {
307  /* drop packet on invalid CRC */
308  if ((NRF_RADIO->CRCSTATUS != 1) || !(nrfmin_dev.event_callback)) {
309  rx_buf.pkt.hdr.len = 0;
310  NRF_RADIO->TASKS_START = 1;
311  return;
312  }
313  rx_lock = 0;
314  nrfmin_dev.event_callback(&nrfmin_dev, NETDEV2_EVENT_ISR);
315  }
316  else if (state == STATE_TX) {
317  goto_target_state();
318  }
319  }
320 
321  cortexm_isr_end();
322 }

324 static int nrfmin_send(netdev2_t *dev, const struct iovec *vector, unsigned count)
325 {
326  (void)dev;
327 
328  assert((vector != NULL) && (count > 0) && (state != STATE_OFF));
329 
330  /* wait for any ongoing transmission to finish and go into idle state */
331  while (state == STATE_TX) {}
332  go_idle();
333 
334  /* copy packet data into the transmit buffer */
335  int pos = 0;
336  for (unsigned i = 0; i < count; i++) {
337  if ((pos + vector[i].iov_len) > NRFMIN_PKT_MAX) {
338  DEBUG("[nrfmin] send: unable to do so, packet is too large!\n");
339  return -EOVERFLOW;
340  }
341  memcpy(&tx_buf.raw[pos], vector[i].iov_base, vector[i].iov_len);
342  pos += vector[i].iov_len;
343  }
344 
345  /* set output buffer and destination address */
346  nrfmin_hdr_t *hdr = (nrfmin_hdr_t *)vector[0].iov_base;
347  NRF_RADIO->PACKETPTR = (uint32_t)(&tx_buf);
348  NRF_RADIO->BASE0 = (CONF_ADDR_BASE | hdr->dst_addr);
349 
350  /* trigger the actual transmission */
351  DEBUG("[nrfmin] send: putting %i byte into the ether\n", (int)hdr->len);
352  state = STATE_TX;
353  NRF_RADIO->TASKS_TXEN = 1;
354 
355  return (int)count;
356 }

358 static int nrfmin_recv(netdev2_t *dev, void *buf, size_t len, void *info)
359 {
360  (void)dev;
361  (void)info;
362 
363  assert(state != STATE_OFF);
364 
365  int pktlen = (int)rx_buf.pkt.hdr.len;
366 
367  /* check if packet data is readable */
368  if (rx_lock || (pktlen == 0)) {
369  DEBUG("[nrfmin] recv: no packet data available\n");
370  return 0;
371  }
372 
373  if (buf == NULL) {
374  if (len > 0) {
375  /* drop packet */
376  DEBUG("[nrfmin] recv: dropping packet of length %i\n", pktlen);
377  rx_buf.pkt.hdr.len = 0;
378  goto_target_state();
379  }
380  }
381  else {
382  DEBUG("[nrfmin] recv: reading packet of length %i\n", pktlen);
383 
384  pktlen = (len < pktlen) ? len : pktlen;
385  memcpy(buf, rx_buf.raw, pktlen);
386  rx_buf.pkt.hdr.len = 0;
387  goto_target_state();
388  }
389 
390  return pktlen;
391 }

393 static int nrfmin_init(netdev2_t *dev)
394 {
395  uint8_t cpuid[CPUID_LEN];
396 
397  /* check given device descriptor */
398  assert(dev);
399 
400  /* initialize our own address from the CPU ID */
401  my_addr = 0;
402  cpuid_get(cpuid);
403  for (int i = 0; i < CPUID_LEN; i++) {
404  my_addr ^= cpuid[i] << (8 * (i & 0x01));
405  }
406 
407  /* power on the NRFs radio */
408  NRF_RADIO->POWER = 1;
409  /* load driver specific configuration */
410  NRF_RADIO->MODE = CONF_MODE;
411  /* configure variable parameters to default values */
412  NRF_RADIO->TXPOWER = NRFMIN_TXPOWER_DEFAULT;
413  NRF_RADIO->FREQUENCY = NRFMIN_CHAN_DEFAULT;
414  /* pre-configure radio addresses */
415  NRF_RADIO->PREFIX0 = CONF_ADDR_PREFIX0;
416  NRF_RADIO->BASE0 = (CONF_ADDR_BASE | my_addr);
417  NRF_RADIO->BASE1 = CONF_ADDR_BCAST;
418  /* always send from logical address 0 */
419  NRF_RADIO->TXADDRESS = 0x00UL;
420  /* and listen to logical addresses 0 and 1 */
421  NRF_RADIO->RXADDRESSES = 0x03UL;
422  /* configure data fields and packet length whitening and endianess */
423  NRF_RADIO->PCNF0 = ((CONF_S1 << RADIO_PCNF0_S1LEN_Pos) |
424  (CONF_S0 << RADIO_PCNF0_S0LEN_Pos) |
425  (CONF_LEN << RADIO_PCNF0_LFLEN_Pos));
426  NRF_RADIO->PCNF1 = ((CONF_WHITENING << RADIO_PCNF1_WHITEEN_Pos) |
427  (CONF_ENDIAN << RADIO_PCNF1_ENDIAN_Pos) |
428  (CONF_BASE_ADDR_LEN << RADIO_PCNF1_BALEN_Pos) |
429  (CONF_STATLEN << RADIO_PCNF1_STATLEN_Pos) |
430  (NRFMIN_PKT_MAX << RADIO_PCNF1_MAXLEN_Pos));
431  /* configure the CRC unit, we skip the address field as this seems to lead
432  * to wrong checksum calculation on nRF52 devices in some cases */
433  NRF_RADIO->CRCCNF = CONF_CRC_LEN | RADIO_CRCCNF_SKIPADDR_Msk;
434  NRF_RADIO->CRCPOLY = CONF_CRC_POLY;
435  NRF_RADIO->CRCINIT = CONF_CRC_INIT;
436  /* set shortcuts for more efficient transfer */
437  NRF_RADIO->SHORTS = RADIO_SHORTS_READY_START_Msk;
438  /* enable interrupts */
439  NVIC_EnableIRQ(RADIO_IRQn);
440  /* enable END interrupt */
441  NRF_RADIO->EVENTS_END = 0;
442  NRF_RADIO->INTENSET = RADIO_INTENSET_END_Msk;
443  /* put device in receive mode */
444  target_state = STATE_RX;
445  goto_target_state();
446 
447  DEBUG("[nrfmin] initialization successful\n");
448 
449  return 0;
450 }
