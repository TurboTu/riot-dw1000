29 static uint32_t si70xx_measure(si70xx_t *dev, uint8_t command)
30 {
31  uint8_t result[2];
32 
33  i2c_acquire(dev->i2c_dev);
34  i2c_write_byte(dev->i2c_dev, dev->address, command);
35  i2c_read_bytes(dev->i2c_dev, dev->address, result, 2);
36  i2c_release(dev->i2c_dev);
37 
38  /* reconstruct raw result */
39  return ((uint32_t)result[0] << 8) + (result[1] & 0xfc);
40 }

90 uint16_t si70xx_get_relative_humidity(si70xx_t *dev)
91 {
92  uint32_t raw;
93  int32_t humidity;
94 
95  /* perform measurement */
96  raw = si70xx_measure(dev, SI70XX_MEASURE_RH_HOLD);
97 
98  humidity = ((12500 * raw) / 65536) - 600;
99 
100  /* according to datasheet, values may exceed bounds, but can be clipped */
101  if (humidity < 0) {
102  return 0;
103  }
104  else if (humidity > 10000) {
105  return 10000;
106  }
107  else {
108  return (uint16_t) humidity;
109  }
110 }

135 uint64_t si70xx_get_serial(si70xx_t *dev)
136 {
137  uint8_t out[2];
138  uint8_t in_first[8] = { 0 };
139  uint8_t in_second[8] = { 0 };
140 
141  /* read the lower bytes */
142  out[0] = SI70XX_READ_ID_FIRST_A;
143  out[1] = SI70XX_READ_ID_FIRST_B;
144 
145  i2c_acquire(dev->i2c_dev);
146  i2c_write_bytes(dev->i2c_dev, dev->address, out, 2);
147  i2c_read_bytes(dev->i2c_dev, dev->address, in_first, 8);
148 
149  /* read the higher bytes */
150  out[0] = SI70XX_READ_ID_SECOND_A;
151  out[1] = SI70XX_READ_ID_SECOND_B;
152 
153  i2c_write_bytes(dev->i2c_dev, dev->address, out, 2);
154  i2c_read_bytes(dev->i2c_dev, dev->address, in_second, 8);
155  i2c_release(dev->i2c_dev);
156 
157  /* calculate the ID */
158  uint32_t id_first = ((uint32_t)in_first[0] << 24) + ((uint32_t)in_first[2] << 16) +
159  (in_first[4] << 8) + (in_first[6] << 0);
160  uint32_t id_second = ((uint32_t)in_second[0] << 24) + ((uint32_t)in_second[2] << 16) +
161  (in_second[4] << 8) + (in_second[6] << 0);
162 
163  return (((uint64_t) id_first) << 32) + id_second;
164 }
