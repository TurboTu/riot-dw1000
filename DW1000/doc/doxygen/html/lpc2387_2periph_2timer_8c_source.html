111 int timer_init(tim_t tim, unsigned long freq, timer_cb_t cb, void *arg)
112 {
113  /* get the timers base register */
114  lpc23xx_timer_t *dev = get_dev(tim);
115 
116  /* make sure the timer device is valid */
117  if (dev == NULL) {
118  return -1;
119  }
120 
121  /* save the callback */
122  isr_ctx[tim].cb = cb;
123  isr_ctx[tim].arg = arg;
124  /* enable power, config periph clock and install ISR vector */
125  pwr_clk_and_isr(tim);
126  /* reset timer configuration (sets the timer to timer mode) */
127  dev->TCR = 0;
128  dev->CTCR = 0;
129  /* configure the prescaler */
130  dev->PR = (CLOCK_PCLK / freq) - 1;
131  /* enable timer */
132  dev->TCR = 1;
133  return 0;
134 }
