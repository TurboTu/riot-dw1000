47 /* Must be <= 0x1000 because otherwise x86_map_physical_pages() might get a page out of this pool.
48  * Because static memory has the PT_G flag, flushing the TLB would cause a stale PT
49  * after calling add_pages_to_pool() in x86_map_physical_pages().
50  */
51 static char early_malloc_pool[0x1000] __attribute__((aligned(4)));
