43 void mrf24j40_reset(mrf24j40_t *dev)
44 {
45  eui64_t addr_long;
46 
47  mrf24j40_init(dev);
48 
49  /* reset options and sequence number */
50  dev->netdev.seq = 0;
51  dev->netdev.flags = 0;
52 
53  /* get an 8-byte unique ID to use as hardware address */
54  uuid_get(addr_long.uint8, IEEE802154_LONG_ADDRESS_LEN);
55  addr_long.uint8[0] &= ~(0x01);
56  addr_long.uint8[0] |= (0x02);
57  /* set short and long address */
58  mrf24j40_set_addr_long(dev, NTOHLL(addr_long.uint64.u64));
59  mrf24j40_set_addr_short(dev, NTOHS(addr_long.uint16[0].u16));
60 
61  /* set default PAN id */
62  mrf24j40_set_pan(dev, IEEE802154_DEFAULT_PANID);
63  mrf24j40_set_chan(dev, IEEE802154_DEFAULT_CHANNEL);
64 
65  /* configure Immediate Sleep and Wake-Up mode */
66  mrf24j40_reg_write_short(dev, MRF24J40_REG_WAKECON, MRF24J40_WAKECON_IMMWAKE);
67 
68  /* set default options */
69  mrf24j40_set_option(dev, IEEE802154_FCF_PAN_COMP, true);
70  mrf24j40_set_option(dev, NETDEV2_IEEE802154_SRC_MODE_LONG, true);
71  mrf24j40_set_option(dev, NETDEV2_IEEE802154_ACK_REQ, true);
72  mrf24j40_set_option(dev, MRF24J40_OPT_CSMA, true);
73  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_RX_START, false);
74  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_RX_END, true);
75 #ifdef MODULE_NETSTATS_L2
76  mrf24j40_set_option(dev, MRF24J40_OPT_TELL_TX_END, true);
77 #endif
78 
79  /* set default protocol */
80 #ifdef MODULE_GNRC_SIXLOWPAN
81  dev->netdev.proto = GNRC_NETTYPE_SIXLOWPAN;
82 #elif MODULE_GNRC
83  dev->netdev.proto = GNRC_NETTYPE_UNDEF;
84 #endif
85 
86  /* go into RX state */
87  mrf24j40_reset_tasks(dev);
88  dev->state = 0;
89  mrf24j40_set_state(dev, MRF24J40_PSEUDO_STATE_IDLE);
90  DEBUG("mrf24j40_reset(): reset complete.\n");
91 }

93 bool mrf24j40_cca(mrf24j40_t *dev)
94 {
95  uint8_t tmp_ccaedth;
96  uint8_t status;
97  uint8_t tmp_rssi;
98 
99  mrf24j40_assert_awake(dev);
100 
101  /* trigger CCA measurment */
102  /* take a look onto datasheet chapter 3.6.1 */
103  mrf24j40_reg_write_short(dev, MRF24J40_REG_BBREG6, MRF24J40_BBREG6_RSSIMODE1);
104  /* wait for result to be ready */
105  do {
106  status = mrf24j40_reg_read_short(dev, MRF24J40_REG_BBREG6);
107  } while (!(status & MRF24J40_BBREG2_RSSIRDY));
108  mrf24j40_assert_sleep(dev);
109  /* return according to measurement */
110  tmp_ccaedth = mrf24j40_reg_read_short(dev, MRF24J40_REG_CCAEDTH); /* Energy detection threshold */
111  tmp_rssi = mrf24j40_reg_read_long(dev, MRF24J40_REG_RSSI);
112  if (tmp_rssi < tmp_ccaedth) {
113  /* channel is clear */
114  return true; /* idle */
115  }
116  else {
117  /* channel is busy */
118  return false;
119  }
120 }

122 void mrf24j40_tx_prepare(mrf24j40_t *dev)
123 {
124  DEBUG("[mrf24j40] TX_Prepare, Current state: %x\n", dev->state);
125  do {
126  mrf24j40_update_tasks(dev);
127  } while (!(dev->pending & MRF24J40_TASK_TX_DONE));
128  mrf24j40_assert_awake(dev);
129  dev->pending &= ~(MRF24J40_TASK_TX_DONE);
130  dev->tx_frame_len = IEEE802154_FCS_LEN;
131 }

143 void mrf24j40_tx_exec(mrf24j40_t *dev)
144 {
145  netdev2_t *netdev = (netdev2_t *)dev;
146 
147 
148  dev->tx_frame_len = dev->tx_frame_len - IEEE802154_FCS_LEN;
149  /* write frame length field in FIFO */
150  mrf24j40_tx_normal_fifo_write(dev, MRF24J40_TX_NORMAL_FIFO + 1, &(dev->tx_frame_len), 1);
151 
152  /* write header length to FIFO address 0x00 */
153  /* from figure 3-11 datasheet: header length = 2 Bytes Frame Control
154  * + 1 Byte Seq. No.
155  * + 4 to 20 Bytes Addressing Fields
156  */
157  mrf24j40_reg_write_long(dev, MRF24J40_TX_NORMAL_FIFO, dev->header_len);
158 
159  if (dev->netdev.flags & NETDEV2_IEEE802154_ACK_REQ) {
160  mrf24j40_reg_write_short(dev, MRF24J40_REG_TXNCON, MRF24J40_TXNCON_TXNACKREQ | MRF24J40_TXNCON_TXNTRIG);
161  }
162  else {
163  mrf24j40_reg_write_short(dev, MRF24J40_REG_TXNCON, MRF24J40_TXNCON_TXNTRIG);
164  }
165  if (netdev->event_callback && (dev->netdev.flags & MRF24J40_OPT_TELL_TX_START)) {
166  netdev->event_callback(netdev, NETDEV2_EVENT_TX_STARTED);
167  }
168 }
