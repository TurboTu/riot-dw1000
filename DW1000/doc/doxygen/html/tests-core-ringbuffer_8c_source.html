53 static void run_add(void)
54 {
55  char next = 0;
56  for (unsigned iteration = 0; iteration < ITERATIONS; ++iteration) {
57  mutex_lock(&mutex);
58 
59  for (unsigned i = 0; i < BUF_SIZE; ++i) {
60  assert_avail(i);
61  assert_add_one(next, -1);
62  assert_avail(i + 1);
63  ++next;
64  }
65 
66  /* Overwrite oldest element. It should be returned to us. */
67  assert_avail(BUF_SIZE);
68  assert_add_one(next, next - BUF_SIZE);
69  assert_avail(BUF_SIZE);
70  ++next;
71 
72  thread_wakeup(pid_get);
73  mutex_unlock_and_sleep(&mutex);
74  }
75 
76  thread_wakeup(pid_get);
77 }

79 static void *run_get(void *arg)
80 {
81  (void) arg;
82 
83  char next = 0;
84  for (unsigned iteration = 0; iteration < ITERATIONS; ++iteration) {
85  ++next; /* the first element of a stride is always overwritten */
86 
87  mutex_lock(&mutex);
88 
89  for (unsigned i = BUF_SIZE; i > 0; --i) {
90  assert_avail(i);
91  assert_get_one(next);
92  assert_avail(i - 1);
93  ++next;
94  }
95 
96  assert_avail(0);
97  assert_get_one(-1);
98  assert_avail(0);
99 
100  thread_wakeup(pid_add);
101  mutex_unlock_and_sleep(&mutex);
102  }
103 
104  return NULL;
105 }
