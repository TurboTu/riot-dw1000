121 uint16_t cc2420_get_chan(cc2420_t *dev)
122 {
123  uint16_t chan;
124  uint16_t freq = cc2420_reg_read(dev, CC2420_REG_FSCTRL);

130 int cc2420_set_chan(cc2420_t *dev, uint16_t chan)
131 {
132  if ((chan < CC2420_CHAN_MIN) || (chan > CC2420_CHAN_MAX)) {
133  DEBUG("cc2420: set channel: given channel invalid\n");
134  return -ENOTSUP;
135  }
136 
137  /* calculation from http://www.ti.com/lit/ds/symlink/cc2420.pdf p.50 */
138  uint16_t freq = cc2420_reg_read(dev, CC2420_REG_FSCTRL);
139  freq &= ~CC2420_FSCTRL_FREQ_MASK;
140  freq |= (357 + (5 * (chan - 11)));
141  cc2420_reg_write(dev, CC2420_REG_FSCTRL, freq);
142 
143  dev->netdev.chan = chan;
144 
145  return CC2420_RET_CHAN_OK;
146 }

155 {

157  txpower = CC2420_TXPOWER_MAX;

164  txctrl &= ~(CC2420_TXCTRL_PA_MASK);
165  txctrl |= power_dbm_to_pa[txpower + 25];

169 int cc2420_set_option(cc2420_t *dev, uint16_t option, bool state)
170 {
171  uint16_t reg;
172 
173  /* set option field */
174  if (state) {
175  dev->options |= option;
176  /* trigger option specific actions */
177  switch (option) {
178  case CC2420_OPT_AUTOACK:
179  DEBUG("cc2420: set_opt: CC2420_OPT_AUTOACK\n");
180  reg = cc2420_reg_read(dev, CC2420_REG_MDMCTRL0);
181  reg |= CC2420_MDMCTRL0_AUTOACK;
182  cc2420_reg_write(dev, CC2420_REG_MDMCTRL0, reg);
183  break;
184 
185  case CC2420_OPT_CSMA:
186  DEBUG("cc2420: set_opt: CC2420_OPT_CSMA\n");
187  /* TODO: en/disable csma */
188  break;
189 
190  case CC2420_OPT_PROMISCUOUS:
191  DEBUG("cc2420: set_opt: CC2420_OPT_PROMISCUOUS\n");
192  /* in promisc mode, AUTO ACKs are should be disabled */
193  reg = cc2420_reg_read(dev, CC2420_REG_MDMCTRL0);
194  reg &= ~(CC2420_MDMCTRL0_AUTOACK);
195  reg &= ~(CC2420_MDMCTRL0_ADR_DECODE);
196  cc2420_reg_write(dev, CC2420_REG_MDMCTRL0, reg);
197  break;
198 
199  case CC2420_OPT_PRELOADING:
200  DEBUG("cc2420: set_opt: CC2420_OPT_PRELOADING\n");
201  break;
202 
203  case CC2420_OPT_TELL_TX_START:
204  case CC2420_OPT_TELL_TX_END:
205  case CC2420_OPT_TELL_RX_START:
206  case CC2420_OPT_TELL_RX_END:
207  DEBUG("cc2420: set_opt: TX/RX START/END\n");
208  /* TODO */
209  break;
210 
211  default:
212  return -ENOTSUP;
213  }
214  }
215  else {
216  dev->options &= ~(option);
217  /* trigger option specific actions */
218  switch (option) {
219  case CC2420_OPT_AUTOACK:
220  DEBUG("cc2420: clr_opt: CC2420_OPT_AUTOACK\n");
221  reg = cc2420_reg_read(dev, CC2420_REG_MDMCTRL0);
222  reg &= ~(CC2420_MDMCTRL0_AUTOACK);
223  cc2420_reg_write(dev, CC2420_REG_MDMCTRL0, reg);
224  break;
225 
226  case CC2420_OPT_CSMA:
227  DEBUG("cc2420: clr_opt: CC2420_OPT_CSMA\n");
228  /* TODO: en/disable csma */
229  break;
230 
231  case CC2420_OPT_PROMISCUOUS:
232  DEBUG("cc2420: clr_opt: CC2420_OPT_PROMISCUOUS\n");
233  reg = cc2420_reg_read(dev, CC2420_REG_MDMCTRL0);
234  reg |= CC2420_MDMCTRL0_ADR_DECODE;
235  /* re-enable AUTOACK only if the option was set */
236  if (dev->options & CC2420_OPT_AUTOACK) {
237  reg |= CC2420_MDMCTRL0_AUTOACK;
238  }
239  cc2420_reg_write(dev, CC2420_REG_MDMCTRL0, reg);
240  break;
241 
242  case CC2420_OPT_PRELOADING:
243  DEBUG("cc2420: clr_opt: CC2420_OPT_PRELOADING\n");
244  break;
245 
246  case CC2420_OPT_TELL_TX_START:
247  case CC2420_OPT_TELL_TX_END:
248  case CC2420_OPT_TELL_RX_START:
249  case CC2420_OPT_TELL_RX_END:
250  DEBUG("cc2420: clr_opt: TX/RX START/END\n");
251  /* TODO */
252  break;
253 
254  default:
255  return -ENOTSUP;
256  }
257  }
258  return sizeof(netopt_enable_t);
259 }

261 int cc2420_set_state(cc2420_t *dev, netopt_state_t cmd)
262 {
263  if ((cc2420_get_state(dev) == NETOPT_STATE_OFF) &&
264  (cmd != NETOPT_STATE_OFF)) {
265  cc2420_en_xosc(dev);
266  }
267  switch (cmd) {
268  case NETOPT_STATE_OFF:
269  cc2420_strobe(dev, CC2420_STROBE_XOSCOFF);
270  while (cc2420_state(dev) != CC2420_STATE_PD) {}
271  break;
272  case NETOPT_STATE_SLEEP:
273  cc2420_strobe(dev, CC2420_STROBE_RFOFF);
274  while (cc2420_state(dev) != CC2420_STATE_IDLE) {}
275  break;
276  case NETOPT_STATE_IDLE:
277  cc2420_strobe(dev, CC2420_STROBE_FLUSHRX);
278  cc2420_strobe(dev, CC2420_STROBE_RXON);
279  break;
280  case NETOPT_STATE_TX:
281  cc2420_tx_exec(dev);
282  break;
283  case NETOPT_STATE_RESET:
284  cc2420_init(dev);
285  break;
286  case NETOPT_STATE_RX:
287  default:
288  DEBUG("cc2420: set_state: called with invalid target state\n");
289  return -ENOTSUP;
290  }
291  return sizeof(netopt_state_t);
292 }

294 netopt_state_t cc2420_get_state(cc2420_t *dev)
295 {
296  uint8_t cur_state = cc2420_state(dev);
297 
298  if (cur_state == 0) {
299  return NETOPT_STATE_OFF;
300  }
301  else if (cur_state == 1) {
302  return NETOPT_STATE_SLEEP;
303  }
304  else if (((cur_state >= 32) && (cur_state <=39)) || (cur_state == 56)) {
305  return NETOPT_STATE_TX;
306  }
307  else if ((cur_state >= 3) && (cur_state <= 6)) {
308  return NETOPT_STATE_IDLE;
309  }
310  else {
311  return NETOPT_STATE_RX;
312  }
