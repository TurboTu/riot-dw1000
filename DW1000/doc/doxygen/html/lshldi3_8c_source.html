37 /*
38  * Shift an (unsigned) quad value left (logical shift left).
39  * This is the same as arithmetic shift left!
40  */
41 quad_t __lshldi3(quad_t a, qshift_t shift)
42 {
43  union uu aa;
44 
45  if (shift == 0) {
46  return a;
47  }
48 
49  aa.q = a;
50 
51  if (shift >= INT_BITS) {
52  aa.ul[H] = aa.ul[L] << (shift - INT_BITS);
53  aa.ul[L] = 0;
54  }
55  else {
56  aa.ul[H] = (aa.ul[H] << shift) | (aa.ul[L] >> (INT_BITS - shift));
57  aa.ul[L] <<= shift;
58  }
59 
60  return aa.q;
61 }
