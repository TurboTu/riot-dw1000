51 void at86rf2xx_reset(at86rf2xx_t *dev)
52 {
53  eui64_t addr_long;
54 
55  at86rf2xx_hardware_reset(dev);
56 
57  /* Reset state machine to ensure a known state */
58  at86rf2xx_reset_state_machine(dev);
59 
60  /* reset options and sequence number */
61  dev->netdev.seq = 0;
62  dev->netdev.flags = 0;
63 
64  /* get an 8-byte unique ID to use as hardware address */
65  uuid_get(addr_long.uint8, IEEE802154_LONG_ADDRESS_LEN);
66  /* make sure we mark the address as non-multicast and not globally unique */
67  addr_long.uint8[0] &= ~(0x01);
68  addr_long.uint8[0] |= (0x02);
69  /* set short and long address */
70  at86rf2xx_set_addr_long(dev, NTOHLL(addr_long.uint64.u64));
71  at86rf2xx_set_addr_short(dev, NTOHS(addr_long.uint16[0].u16));
72 
73  /* set default PAN id */
74  at86rf2xx_set_pan(dev, AT86RF2XX_DEFAULT_PANID);
75  /* set default channel */
76  at86rf2xx_set_chan(dev, AT86RF2XX_DEFAULT_CHANNEL);
77  /* set default TX power */
78  at86rf2xx_set_txpower(dev, AT86RF2XX_DEFAULT_TXPOWER);
79  /* set default options */
80  at86rf2xx_set_option(dev, AT86RF2XX_OPT_AUTOACK, true);
81  at86rf2xx_set_option(dev, AT86RF2XX_OPT_CSMA, true);
82  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_RX_START, false);
83  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_RX_END, true);
84 #ifdef MODULE_NETSTATS_L2
85  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_TX_END, true);
86 #endif
87  /* set default protocol */
88 #ifdef MODULE_GNRC_SIXLOWPAN
89  dev->netdev.proto = GNRC_NETTYPE_SIXLOWPAN;
90 #elif MODULE_GNRC
91  dev->netdev.proto = GNRC_NETTYPE_UNDEF;
92 #endif
93  /* enable safe mode (protect RX FIFO until reading data starts) */
94  at86rf2xx_reg_write(dev, AT86RF2XX_REG__TRX_CTRL_2,
95  AT86RF2XX_TRX_CTRL_2_MASK__RX_SAFE_MODE);
96 #ifdef MODULE_AT86RF212B
97  at86rf2xx_set_page(dev, 0);
98 #endif
99 
100  /* don't populate masked interrupt flags to IRQ_STATUS register */
101  uint8_t tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__TRX_CTRL_1);
102  tmp &= ~(AT86RF2XX_TRX_CTRL_1_MASK__IRQ_MASK_MODE);
103  at86rf2xx_reg_write(dev, AT86RF2XX_REG__TRX_CTRL_1, tmp);
104 
105  /* disable clock output to save power */
106  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__TRX_CTRL_0);
107  tmp &= ~(AT86RF2XX_TRX_CTRL_0_MASK__CLKM_CTRL);
108  tmp &= ~(AT86RF2XX_TRX_CTRL_0_MASK__CLKM_SHA_SEL);
109  tmp |= (AT86RF2XX_TRX_CTRL_0_CLKM_CTRL__OFF);
110  at86rf2xx_reg_write(dev, AT86RF2XX_REG__TRX_CTRL_0, tmp);
111 
112  /* enable interrupts */
113  at86rf2xx_reg_write(dev, AT86RF2XX_REG__IRQ_MASK,
114  AT86RF2XX_IRQ_STATUS_MASK__TRX_END);
115  /* clear interrupt flags */
116  at86rf2xx_reg_read(dev, AT86RF2XX_REG__IRQ_STATUS);
117 
118  /* go into RX state */
119  at86rf2xx_set_state(dev, AT86RF2XX_STATE_RX_AACK_ON);
120 
121  DEBUG("at86rf2xx_reset(): reset complete.\n");
122 }

124 size_t at86rf2xx_send(at86rf2xx_t *dev, uint8_t *data, size_t len)
125 {
126  /* check data length */
127  if (len > AT86RF2XX_MAX_PKT_LENGTH) {
128  DEBUG("[at86rf2xx] Error: data to send exceeds max packet size\n");
129  return 0;
130  }
131  at86rf2xx_tx_prepare(dev);
132  at86rf2xx_tx_load(dev, data, len, 0);
133  at86rf2xx_tx_exec(dev);
134  return len;
135 }

137 void at86rf2xx_tx_prepare(at86rf2xx_t *dev)
138 {
139  uint8_t state;
140 
141  dev->pending_tx++;
142  /* make sure ongoing transmissions are finished */
143  do {
144  state = at86rf2xx_get_status(dev);
145  } while (state == AT86RF2XX_STATE_BUSY_RX_AACK ||
146  state == AT86RF2XX_STATE_BUSY_TX_ARET);
147  if (state != AT86RF2XX_STATE_TX_ARET_ON) {
148  dev->idle_state = state;
149  }
150  at86rf2xx_set_state(dev, AT86RF2XX_STATE_TX_ARET_ON);
151  dev->tx_frame_len = IEEE802154_FCS_LEN;
152 }

154 size_t at86rf2xx_tx_load(at86rf2xx_t *dev, uint8_t *data,
155  size_t len, size_t offset)
156 {
157  dev->tx_frame_len += (uint8_t)len;
158  at86rf2xx_sram_write(dev, offset + 1, data, len);
159  return offset + len;
160 }

162 void at86rf2xx_tx_exec(at86rf2xx_t *dev)
163 {
164  netdev2_t *netdev = (netdev2_t *)dev;
165 
166  /* write frame length field in FIFO */
167  at86rf2xx_sram_write(dev, 0, &(dev->tx_frame_len), 1);
168  /* trigger sending of pre-loaded frame */
169  at86rf2xx_reg_write(dev, AT86RF2XX_REG__TRX_STATE,
170  AT86RF2XX_TRX_STATE__TX_START);
171  if (netdev->event_callback &&
172  (dev->netdev.flags & AT86RF2XX_OPT_TELL_TX_START)) {
173  netdev->event_callback(netdev, NETDEV2_EVENT_TX_STARTED);
174  }
