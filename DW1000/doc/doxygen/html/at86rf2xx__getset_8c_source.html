52  const uint8_t *dbm_to_tx_pow;

73  return i - 25;
74  }
75  }

77 }

314 void at86rf2xx_set_cca_threshold(at86rf2xx_t *dev, int8_t value)
315 {
316  /* ensure the given value is negative, since a CCA threshold > 0 is
317  just impossible: thus, any positive value given is considered
318  to be the absolute value of the actually wanted threshold */
319  if (value > 0) {
320  value = -value;
321  }
322  /* transform the dBm value in the form
323  that will fit in the AT86RF2XX_REG__CCA_THRES register */
324  value -= RSSI_BASE_VAL;
325  value >>= 1;
326  value &= AT86RF2XX_CCA_THRES_MASK__CCA_ED_THRES;
327  value |= AT86RF2XX_CCA_THRES_MASK__RSVD_HI_NIBBLE;
328  at86rf2xx_reg_write(dev, AT86RF2XX_REG__CCA_THRES, value);

331 void at86rf2xx_set_option(at86rf2xx_t *dev, uint16_t option, bool state)
332 {
333  uint8_t tmp;
334 
335  DEBUG("set option %i to %i\n", option, state);
336 
337  /* set option field */
338  if (state) {
339  dev->netdev.flags |= option;
340  /* trigger option specific actions */
341  switch (option) {
342  case AT86RF2XX_OPT_CSMA:
343  DEBUG("[at86rf2xx] opt: enabling CSMA mode" \
344  "(4 retries, min BE: 3 max BE: 5)\n");
345  /* Initialize CSMA seed with hardware address */
346  at86rf2xx_set_csma_seed(dev, dev->netdev.long_addr);
347  at86rf2xx_set_csma_max_retries(dev, 4);
348  at86rf2xx_set_csma_backoff_exp(dev, 3, 5);
349  break;
350  case AT86RF2XX_OPT_PROMISCUOUS:
351  DEBUG("[at86rf2xx] opt: enabling PROMISCUOUS mode\n");
352  /* disable auto ACKs in promiscuous mode */
353  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__CSMA_SEED_1);
354  tmp |= AT86RF2XX_CSMA_SEED_1__AACK_DIS_ACK;
355  at86rf2xx_reg_write(dev, AT86RF2XX_REG__CSMA_SEED_1, tmp);
356  /* enable promiscuous mode */
357  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__XAH_CTRL_1);
358  tmp |= AT86RF2XX_XAH_CTRL_1__AACK_PROM_MODE;
359  at86rf2xx_reg_write(dev, AT86RF2XX_REG__XAH_CTRL_1, tmp);
360  break;
361  case AT86RF2XX_OPT_AUTOACK:
362  DEBUG("[at86rf2xx] opt: enabling auto ACKs\n");
363  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__CSMA_SEED_1);
364  tmp &= ~(AT86RF2XX_CSMA_SEED_1__AACK_DIS_ACK);
365  at86rf2xx_reg_write(dev, AT86RF2XX_REG__CSMA_SEED_1, tmp);
366  break;
367  case AT86RF2XX_OPT_TELL_RX_START:
368  DEBUG("[at86rf2xx] opt: enabling SFD IRQ\n");
369  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__IRQ_MASK);
370  tmp |= AT86RF2XX_IRQ_STATUS_MASK__RX_START;
371  at86rf2xx_reg_write(dev, AT86RF2XX_REG__IRQ_MASK, tmp);
372  break;
373  default:
374  /* do nothing */
375  break;
376  }
377  }
378  else {
379  dev->netdev.flags &= ~(option);
380  /* trigger option specific actions */
381  switch (option) {
382  case AT86RF2XX_OPT_CSMA:
383  DEBUG("[at86rf2xx] opt: disabling CSMA mode\n");
384  /* setting retries to -1 means CSMA disabled */
385  at86rf2xx_set_csma_max_retries(dev, -1);
386  break;
387  case AT86RF2XX_OPT_PROMISCUOUS:
388  DEBUG("[at86rf2xx] opt: disabling PROMISCUOUS mode\n");
389  /* disable promiscuous mode */
390  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__XAH_CTRL_1);
391  tmp &= ~(AT86RF2XX_XAH_CTRL_1__AACK_PROM_MODE);
392  at86rf2xx_reg_write(dev, AT86RF2XX_REG__XAH_CTRL_1, tmp);
393  /* re-enable AUTOACK only if the option is set */
394  if (dev->netdev.flags & AT86RF2XX_OPT_AUTOACK) {
395  tmp = at86rf2xx_reg_read(dev,
396  AT86RF2XX_REG__CSMA_SEED_1);
397  tmp &= ~(AT86RF2XX_CSMA_SEED_1__AACK_DIS_ACK);
398  at86rf2xx_reg_write(dev, AT86RF2XX_REG__CSMA_SEED_1,
399  tmp);
400  }
401  break;
402  case AT86RF2XX_OPT_AUTOACK:
403  DEBUG("[at86rf2xx] opt: disabling auto ACKs\n");
404  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__CSMA_SEED_1);
405  tmp |= AT86RF2XX_CSMA_SEED_1__AACK_DIS_ACK;
406  at86rf2xx_reg_write(dev, AT86RF2XX_REG__CSMA_SEED_1, tmp);
407  break;
408  case AT86RF2XX_OPT_TELL_RX_START:
409  DEBUG("[at86rf2xx] opt: disabling SFD IRQ\n");
410  tmp = at86rf2xx_reg_read(dev, AT86RF2XX_REG__IRQ_MASK);
411  tmp &= ~AT86RF2XX_IRQ_STATUS_MASK__RX_START;
412  at86rf2xx_reg_write(dev, AT86RF2XX_REG__IRQ_MASK, tmp);
413  break;
414  default:
415  /* do nothing */
416  break;
417  }
418  }

431 static inline void _set_state(at86rf2xx_t *dev, uint8_t state, uint8_t cmd)
432 {
433  at86rf2xx_reg_write(dev, AT86RF2XX_REG__TRX_STATE, cmd);
434 
435  /* To prevent a possible race condition when changing to
436  * RX_AACK_ON state the state doesn't get read back in that
437  * case. See discussion
438  * in https://github.com/RIOT-OS/RIOT/pull/5244
439  */
440  if (state != AT86RF2XX_STATE_RX_AACK_ON) {
441  while (at86rf2xx_get_status(dev) != state);
442  }
443 
444  dev->state = state;
445 }

447 void at86rf2xx_set_state(at86rf2xx_t *dev, uint8_t state)
448 {
449  uint8_t old_state = at86rf2xx_get_status(dev);
450 
451  if (state == old_state) {
452  return;
453  }
454 
455  if (state == AT86RF2XX_STATE_FORCE_TRX_OFF) {
456  _set_state(dev, AT86RF2XX_STATE_TRX_OFF, state);
457  return;
458  }
459 
460  /* make sure there is no ongoing transmission, or state transition already
461  * in progress */
462  while (old_state == AT86RF2XX_STATE_BUSY_RX_AACK ||
463  old_state == AT86RF2XX_STATE_BUSY_TX_ARET ||
464  old_state == AT86RF2XX_STATE_IN_PROGRESS) {
465  old_state = at86rf2xx_get_status(dev);
466  }
467 
468  if (state == old_state) {
469  return;
470  }
471 
472  /* we need to go via PLL_ON if we are moving between RX_AACK_ON <-> TX_ARET_ON */
473  if ((old_state == AT86RF2XX_STATE_RX_AACK_ON &&
474  state == AT86RF2XX_STATE_TX_ARET_ON) ||
475  (old_state == AT86RF2XX_STATE_TX_ARET_ON &&
476  state == AT86RF2XX_STATE_RX_AACK_ON)) {
477  _set_state(dev, AT86RF2XX_STATE_PLL_ON, AT86RF2XX_STATE_PLL_ON);
478  }
479  /* check if we need to wake up from sleep mode */
480  else if (old_state == AT86RF2XX_STATE_SLEEP) {
481  DEBUG("at86rf2xx: waking up from sleep mode\n");
482  at86rf2xx_assert_awake(dev);
483  }
484 
485  if (state == AT86RF2XX_STATE_SLEEP) {
486  /* First go to TRX_OFF */
487  at86rf2xx_set_state(dev, AT86RF2XX_STATE_FORCE_TRX_OFF);
488  /* Discard all IRQ flags, framebuffer is lost anyway */
489  at86rf2xx_reg_read(dev, AT86RF2XX_REG__IRQ_STATUS);
490  /* Go to SLEEP mode from TRX_OFF */
491  gpio_set(dev->params.sleep_pin);
492  dev->state = state;
493  } else {
494  _set_state(dev, state, state);
495  }
496 }

499 {
