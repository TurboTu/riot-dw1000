42 uint32_t pwm_init(pwm_t dev, pwm_mode_t mode, uint32_t freq, uint16_t res)
43 {
44  uint32_t pwm_clk = freq * res; /* Desired/real pwm_clock */
45  uint32_t diva = 1; /* Candidate for 8bit divider */
46  uint32_t prea = 0; /* Candidate for clock select */
47 
48  /* check for valid device and mode (left-aligned PWM only so far) */
49  if ((dev != PWM_DEV(0)) || (mode != PWM_LEFT)) {
50  return 0;
51  }
52 
53  /* check if target frequency and resolution is applicable */
54  if (pwm_clk > CLOCK_CORECLOCK) {
55  return 0;
56  }
57 
58  /* calculate pre-scalers for targeted frequency and resolution:
59  * clk = CORECLOCK / (2 ^ prea) / diva
60  * width prea := [0, 10] and diva [1, 255]
61  */
62  while ((diva = (CLOCK_CORECLOCK / pwm_clk / (1 << prea))) > 255) {
63  ++prea;
64  }
65  /* make sure PREA does not exceed its limit */
66  if (prea > PREA_MAX) {
67  return 0;
68  }
69 
70  /* activate PWM block by enabling it's clock. */
71  PMC->PMC_PCER1 = PMC_PCER1_PID36;
72  /* disable all channels to allow CPRD updates. */
73  PWM->PWM_DIS = 0xff;
74  /* configure clock generator */
75  PWM->PWM_CLK = PWM_CLK_PREA(prea) | PWM_CLK_DIVA(diva);
76  /* remember the used resolution (for cropping inputs later) */
77  pwm_period = res - 1;
78 
79  /* setup the configured channels */
80  pwm_chan_mask = 0;
81  for (unsigned i = 0; i < PWM_CHAN_NUMOF; i++) {
82  /* configure the use pin */
83  gpio_init_mux(pwm_chan[i].pin, GPIO_MUX_B);
84  /* and setup the channel */
85  pwm_chan_mask |= (1 << pwm_chan[i].hwchan);
86  PWM->PWM_CH_NUM[pwm_chan[i].hwchan].PWM_CMR = PWM_CMR_CPRE_CLKA;
87  PWM->PWM_CH_NUM[pwm_chan[i].hwchan].PWM_CPRD = pwm_period;
88  PWM->PWM_CH_NUM[pwm_chan[i].hwchan].PWM_CDTY = 0;
89  }
90 
91  /* enable all configured channels */
92  PWM->PWM_ENA = pwm_chan_mask;
93 
94  /* and return the actual configured frequency */
95  return (CLOCK_CORECLOCK / (1 << prea) / diva / res);
96 }

110  assert((pwm == PWM_DEV(0)) && (channel < PWM_CHAN_NUMOF));
