33 adc_init(void)
34 {
35  // enable clock /Power for ADC
36  PCONP |= BIT12;
37 
38  // peripheral Clock Selection for ADC
39  PCLKSEL0 |= 0x03000000; // pclock = cclock/8
40 
41  // set A/D control register AD0CR
42  AD0CR = (0x01 << 0) | /* SEL=1, select channel 0~7 on ADC0 */
43  (0xff << 8) | /* CLKDIV = 1, ADC_CLK = PCLK/10 = 0.45 MHz */
44  (0 << 16) | /* BURST = 0, no BURST, software controlled */
45  (0 << 17) | /* CLKS = 0, 11 clocks/10 bits */
46  (1 << 21) | /* PDN = 1, normal operation */
47  (0 << 22) | /* TEST1:0 = 00 */
48  (0 << 24) | /* START = 0 A/D conversion stops */
49  (0 << 27); /* EDGE = 0 (CAP/MAT signal falling,trigger A/D conversion) */
50 }

97 uint16_t adc_read(uint8_t channel)
98 {
99 #if ENABLE_DEBUG
100  uint32_t t1, t2;
101 #endif
102  uint32_t regVal, adc_data;
103 
104  /* channel number is 0 through 7 */
105  if (channel >= ADC_NUM) {
106  channel = 0; /* reset channel number to 0 */
107  }
108 
109  /* switch channel, start A/D convert */
110  AD0CR &= 0xFFFFFF00;
111 #if ENABLE_DEBUG
112  t1 = _xtimer_now();
113 #endif
114  AD0CR |= (1 << 24) | (1 << channel);
115 
116 #if ENABLE_DEBUG
117  t2 = _xtimer_now();
118 #endif
119 
120  while (1) {
121  /* read result of A/D conversion */
122  regVal = *(volatile unsigned long *)(AD0_BASE_ADDR + ADC_OFFSET + ADC_INDEX * channel);
123 
124  /* wait until end of A/D convert */
125  if (regVal & ADC_DONE) {
126  break;
127  }
128  }
129 
130  AD0CR &= 0xF8FFFFFF; /* stop ADC now */
131 
132  if (regVal & ADC_OVERRUN) { /* save data when it's not overrun, otherwise, return zero */
133  return 0;
134  }
135 
136  adc_data = (regVal >> 6) & 0x3FF;
137  DEBUG("%s, %d: %lu\n", RIOT_FILE_RELATIVE, __LINE__, t1);
138  DEBUG("%s, %d: %lu\n", RIOT_FILE_RELATIVE, __LINE__, t2);
139  return (uint16_t) adc_data; /* return A/D conversion value */
140 }
