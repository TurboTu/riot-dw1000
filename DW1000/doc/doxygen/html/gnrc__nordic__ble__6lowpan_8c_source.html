81 static void _handle_raw_sixlowpan(ble_mac_inbuf_t *inbuf)
82 {
83  gnrc_pktsnip_t *pkt = NULL;
84 
85  pkt = gnrc_pktbuf_add(NULL, inbuf->payload,
86  inbuf->len,
87  GNRC_NETTYPE_SIXLOWPAN);
88 
89  if(!pkt) {
90  DEBUG("_handle_raw_sixlowpan(): no space left in packet buffer.\n");
91  return;
92  }
93 
94  /* create netif header */
95  gnrc_pktsnip_t *netif_hdr;
96  netif_hdr = gnrc_pktbuf_add(NULL, NULL,
97  sizeof(gnrc_netif_hdr_t) + (2 * sizeof(eui64_t)),
98  GNRC_NETTYPE_NETIF);
99 
100  if (netif_hdr == NULL) {
101  DEBUG("_handle_raw_sixlowpan(): no space left in packet buffer.\n");
102  gnrc_pktbuf_release(pkt);
103  return;
104  }
105 
106  gnrc_netif_hdr_init(netif_hdr->data, BLE_SIXLOWPAN_L2_ADDR_LEN, BLE_SIXLOWPAN_L2_ADDR_LEN);
107  gnrc_netif_hdr_set_src_addr(netif_hdr->data, inbuf->src, BLE_SIXLOWPAN_L2_ADDR_LEN);
108  gnrc_netif_hdr_set_dst_addr(netif_hdr->data, _own_mac_addr, BLE_SIXLOWPAN_L2_ADDR_LEN);
109  ((gnrc_netif_hdr_t *)netif_hdr->data)->if_pid = gnrc_nordic_ble_6lowpan_pid;
110 
111  DEBUG("_handle_raw_sixlowpan(): received packet from %02x:%02x:%02x:%02x:%02x:%02x:%02x:%02x "
112  "of length %d\n",
113  inbuf->src[0], inbuf->src[1], inbuf->src[2], inbuf->src[3], inbuf->src[4],
114  inbuf->src[5], inbuf->src[6], inbuf->src[7], inbuf->len);
115 #if defined(MODULE_OD) && ENABLE_DEBUG
116  od_hex_dump(inbuf->payload, inbuf->len, OD_WIDTH_DEFAULT);
117 #endif
118 
119  LL_APPEND(pkt, netif_hdr);
120 
121  /* throw away packet if no one is interested */
122  if (!gnrc_netapi_dispatch_receive(pkt->type, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
123  DEBUG("_handle_raw_sixlowpan: unable to forward packet of type %i\n", pkt->type);
124  gnrc_pktbuf_release(pkt);
125  }

128 static int _send(gnrc_pktsnip_t *pkt)
129 {
130  if (pkt == NULL) {
131  DEBUG("_send_ble: pkt was NULL\n");
132  return -EINVAL;
133  }
134 
135  gnrc_netif_hdr_t *netif_hdr;
136  gnrc_pktsnip_t *payload = pkt->next;
137  uint8_t *dst;
138 
139  if (ble_mac_busy_tx) {
140  DEBUG("_send(): ble_mac_busy_tx\n");
141  return -EBUSY;
142  }
143 
144  uint8_t *buf = _sendbuf;
145  unsigned len = 0;
146 
147  if (pkt->type != GNRC_NETTYPE_NETIF) {
148  DEBUG("_send_ble: first header is not generic netif header\n");
149  return -EBADMSG;
150  }
151 
152  netif_hdr = pkt->data;
153 
154  /* prepare destination address */
155  if (netif_hdr->flags &
156  (GNRC_NETIF_HDR_FLAGS_BROADCAST | GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
157  dst = NULL;
158  }
159  else {
160  dst = gnrc_netif_hdr_get_dst_addr(netif_hdr);
161  }
162 
163  /* prepare packet for sending */
164  while (payload) {
165  memcpy(buf, payload->data, payload->size);
166  len += payload->size;
167  buf += payload->size;
168  payload = payload->next;
169  }
170 
171  gnrc_pktbuf_release(pkt);
172 
173  if (ble_mac_send(dst, _sendbuf, len) == -1) {
174  return -EBADMSG;
175  }
176 
177  return 0;
178 }
179 static int _handle_get(gnrc_netapi_opt_t *_opt)
180 {
181  int res = -ENOTSUP;
182  uint8_t *value = _opt->data;
183 
184  switch (_opt->opt) {
185  case NETOPT_ACK_REQ:
186  case NETOPT_CHANNEL:
187  case NETOPT_NID:
188  case NETOPT_ADDRESS:
189  /* -ENOTSUP */
190  break;
191  case NETOPT_ADDRESS_LONG:
192  assert(_opt->data_len >= BLE_SIXLOWPAN_L2_ADDR_LEN);
193  memcpy(value, _own_mac_addr, BLE_SIXLOWPAN_L2_ADDR_LEN);
194  value[0] = IPV6_IID_FLIP_VALUE;
195  res = BLE_SIXLOWPAN_L2_ADDR_LEN;
196  break;
197  case NETOPT_ADDR_LEN:
198  case NETOPT_SRC_LEN:
199  assert(_opt->data_len == sizeof(uint16_t));
200  *((uint16_t *)value) = BLE_SIXLOWPAN_L2_ADDR_LEN;
201  res = sizeof(uint16_t);
202  break;
203 #ifdef MODULE_GNRC
204  case NETOPT_PROTO:
205  assert(_opt->data_len == sizeof(gnrc_nettype_t));
206  *((gnrc_nettype_t *)value) = GNRC_NETTYPE_SIXLOWPAN;
207  res = sizeof(gnrc_nettype_t);
208  break;
209 #endif
210 /* case NETOPT_DEVICE_TYPE:
211  assert(_opt->data_len == sizeof(uint16_t));
212  *((uint16_t *)value) = NETDEV2_TYPE_IEEE802154;
213  res = sizeof(uint16_t);
214  break;*/
215  case NETOPT_IPV6_IID:
216  memcpy(value, _own_mac_addr, BLE_SIXLOWPAN_L2_ADDR_LEN);
217  value[0] = IPV6_IID_FLIP_VALUE;
218  res = BLE_SIXLOWPAN_L2_ADDR_LEN;
219  break;
220  default:
221  break;
222  }
223  return res;
224 }

231 static void *_gnrc_nordic_ble_6lowpan_thread(void *args)
232 {
233  (void)args;
234 
235  DEBUG("gnrc_nordic_ble_6lowpan: starting thread\n");
236 
237  gnrc_nordic_ble_6lowpan_pid = thread_getpid();
238 
239  gnrc_netapi_opt_t *opt;
240  int res;
241  msg_t msg, reply, msg_queue[BLE_NETAPI_MSG_QUEUE_SIZE];
242 
243  /* setup the message queue */
244  msg_init_queue(msg_queue, BLE_NETAPI_MSG_QUEUE_SIZE);
245 
246  /* initialize BLE stack */
247  assert((unsigned)softdevice_handler_isEnabled());
248 
249  ble_stack_init();
250  ble_get_mac(_own_mac_addr);
251 
252  ble_mac_init(_ble_mac_callback);
253 
254  ble_advertising_init("RIOT BLE");
255  ble_advertising_start();
256 
257  /* register the device to the network stack*/
258  gnrc_netif_add(thread_getpid());
259 
260  /* start the event loop */
261  while (1) {
262 // DEBUG("gnrc_nordic_ble_6lowpan: waiting for incoming messages\n");
263  msg_receive(&msg);
264  /* dispatch NETDEV and NETAPI messages */
265  switch (msg.type) {
266  case BLE_EVENT_RX_DONE:
267  {
268  DEBUG("ble rx:\n");
269  _handle_raw_sixlowpan(msg.content.ptr);
270  ble_mac_busy_rx = 0;
271  break;
272  }
273  case GNRC_NETAPI_MSG_TYPE_SND:
274  DEBUG("gnrc_nordic_ble_6lowpan: GNRC_NETAPI_MSG_TYPE_SND received\n");
275  _send(msg.content.ptr);
276  break;
277  case GNRC_NETAPI_MSG_TYPE_SET:
278  /* read incoming options */
279  opt = msg.content.ptr;
280  DEBUG("gnrc_nordic_ble_6lowpan: GNRC_NETAPI_MSG_TYPE_SET received. opt=%s\n",
281  netopt2str(opt->opt));
282  /* set option for device driver */
283  res = ENOTSUP;
284  DEBUG("gnrc_nordic_ble_6lowpan: response of netdev->set: %i\n", res);
285  /* send reply to calling thread */
286  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
287  reply.content.value = (uint32_t)res;
288  msg_reply(&msg, &reply);
289  break;
290  case GNRC_NETAPI_MSG_TYPE_GET:
291  /* read incoming options */
292  opt = msg.content.ptr;
293  DEBUG("gnrc_nordic_ble_6lowpan: GNRC_NETAPI_MSG_TYPE_GET received. opt=%s\n",
294  netopt2str(opt->opt));
295  res = _handle_get(opt);
296  DEBUG("gnrc_nordic_ble_6lowpan: response of netdev->get: %i\n", res);
297  /* send reply to calling thread */
298  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
299  reply.content.value = (uint32_t)res;
300  msg_reply(&msg, &reply);
301  break;
302  default:
303  DEBUG("gnrc_nordic_ble_6lowpan: Unknown command %" PRIu16 "\n", msg.type);
304  break;
305  }
306  }
307  /* never reached */
308  return NULL;
309 }
