117 bool i2c_start(uint8_t i2c_interface)
118 {
119  // puts("i2c_start begin...\n");
120  uint32_t timeout = 0;
121  bool retVal = false;
122 
123  /*--- Issue a start condition ---*/
124  switch (i2c_interface) {
125  case I2C0:
126  I20CONSET = I2CONSET_STA; /* Set Start flag */
127  break;
128 
129  case I2C1_0:
130  case I2C1_1:
131  I21CONSET = I2CONSET_STA; /* Set Start flag */
132  break;
133 
134  case I2C2:
135  I22CONSET = I2CONSET_STA; /* Set Start flag */
136  }
137 
138  /*--- Wait until START transmitted ---*/
139  while (1) {
140  if (i2c_master_state == I2C_STARTED) {
141  retVal = true;
142  break;
143  }
144 
145  if (timeout >= MAX_TIMEOUT) {
146  puts("timeout");
147  retVal = false;
148  break;
149  }
150 
151  timeout++;
152  }
153 
154  //puts("...i2c_start ended\n");
155  return (retVal);

255 {

259 
260  switch (i2c_interface) {

262  PCLKSEL0 &= ~(BIT14 | BIT15); //clear Bits

264

267  break;
268 
269  case I2C1_0:
270  case I2C1_1:
271  PCLKSEL1 &= ~(BIT6 | BIT7);
272  PCLKSEL1 |= pclksel << 6;
273 
274  I21SCLL = prescale / 2;
275  I21SCLH = prescale / 2;
276  break;
277 
278  case I2C2:
279  PCLKSEL1 &= ~(BIT20 | BIT21);
280  PCLKSEL1 |= pclksel << 20;
281 
282  I22SCLL = prescale / 2;
283  I22SCLH = prescale / 2;

351 bool i2c_read(uint8_t i2c_interface, uint8_t slave_addr, uint8_t reg_addr,
352  uint8_t *rx_buff, uint8_t rx_buff_length)
353 {
354  i2c_clear_buffer((uint8_t *) i2c_master_buffer,
355  I2C_BUFSIZE * sizeof(uint8_t));
356  i2c_write_length = 1;
357  i2c_read_length = rx_buff_length;
358  bool successful;
359  uint8_t readIndex = 3;
360  i2c_master_buffer[0] = (slave_addr << 1) & WRITE_ENABLE_BIT_MASK;
361  i2c_master_buffer[1] = reg_addr;
362  i2c_master_buffer[2] = ((slave_addr << 1) & WRITE_ENABLE_BIT_MASK)
363  | READ_ENABLE_BIT_MASK;
364  successful = i2c_transaction(i2c_interface);
365 
366  if (successful && (rx_buff != NULL) &&
367  (rx_buff_length < (I2C_BUFSIZE - readIndex))) {
368  memcpy(rx_buff, (const uint8_t *)(i2c_master_buffer + readIndex),
369  sizeof(uint8_t) * rx_buff_length);
370  return true;
371  }
372  else {
373  return false;
374  }
375 }

382 bool i2c_write(uint8_t i2c_interface, uint8_t slave_addr, uint8_t reg_addr,
383  uint8_t *tx_buff, uint8_t tx_buff_length)
384 {
385  //puts("[i2c.c/i2cWrite]: entered\n");
386  i2c_clear_buffer((uint8_t *) i2c_master_buffer,
387  I2C_BUFSIZE * sizeof(uint8_t));
388  i2c_write_length = tx_buff_length + 1;
389  i2c_master_buffer[0] = (slave_addr << 1) & WRITE_ENABLE_BIT_MASK;
390  i2c_master_buffer[1] = reg_addr;
391 
392  if ((tx_buff != NULL) && tx_buff_length < (I2C_BUFSIZE - 2)) {
393  int32_t j = 0;

395  i2c_master_buffer[i] = tx_buff[j];
396  j++;
397  //printf("I2CMasterBuffer[%d] = %d\n", i, I2CMasterBuffer[i]);
398  }
399 
400  return i2c_transaction(i2c_interface);
401  }
402  else {
403  puts("[i2c.c/i2cWrite]: Invalid buffer or invalid write buffer size\n");
404  return false;

418

420  int32_t read_index = 0;

425  i2c_master_buffer[i] = tx_buff[i - 1];

427  }
428 
429  //enable I2C to read
430  if ((rx_buff_length > 0) && (i < I2C_BUFSIZE)) {
431  i2c_master_buffer[i] = ((slave_addr << 1) & WRITE_ENABLE_BIT_MASK)
432  | READ_ENABLE_BIT_MASK;
433  read_index = i + 1;
434  }
435 
436  bool successful = i2c_transaction(i2c_interface);
437 
438  if (successful && (rx_buff != NULL) && (rx_buff_length > 0)) {
439  memcpy(rx_buff, (const uint8_t *)(i2c_master_buffer + read_index),
440  sizeof(uint8_t) * rx_buff_length);
441  return true;
442  }
443  else {
444  return false;
445  }
446  }
447  else {
448  puts(
449  "[i2c.c/i2cRead]: the txBuff is not valid or has not a valid \
450  length value !\n");
451  return false;
452  }

460 void i2c_interface0_master_handler(void) //__irq
461 {
462  //puts("entering I2C handler function\n");
463  uint8_t state_value;
464 
465  /* this handler deals with master read and master write only */
466  state_value = I20STAT;
467 
468  //IENABLE; /* handles nested interrupt */
469  //irq_enable();
470  switch (state_value) {
471  case 0x08: /* A Start condition is issued. */
472  //puts("A Start condition is issued\n");
473  I20DAT = i2c_master_buffer[0];
474  //printf("I22DAT = %lu\n", I22DAT);
475  I20CONCLR = (I2CONCLR_SIC | I2CONCLR_STAC);
476  i2c_master_state = I2C_STARTED;
477  break;
478 
479  case 0x10: /* A repeated started is issued */
480  //puts("A repeated Start is issued\n");
481  // if ( I2CCmd == L3DG420_WHO_AM_I)
482  // {
483  // I22DAT = I2CMasterBuffer[2];
484  // }
485  I20DAT = i2c_master_buffer[2];
486  I20CONCLR = (I2CONCLR_SIC | I2CONCLR_STAC);
487  i2c_master_state = I2C_RESTARTED;
488  break;
489 
490  case 0x18: /* Regardless, it's a ACK */
491 
492  //puts("got an Ack\n");
493  if (i2c_master_state == I2C_STARTED) {
494  I20DAT = i2c_master_buffer[1 + wr_index];
495  wr_index++;
496  i2c_master_state = DATA_ACK;
497 
498  }
499 
500  I20CONCLR = I2CONCLR_SIC;
501  break;
502 
503  case 0x28: /* Data byte has been transmitted, regardless ACK or NACK */
504  case 0x30:
505 
506  //puts("Data byte has been transmitted\n");
507  if (wr_index != i2c_write_length) {
508 
509  // this should be the last one
510  I20DAT = i2c_master_buffer[1 + wr_index];
511 
512  if (wr_index != i2c_write_length) {
513  i2c_master_state = DATA_ACK;
514  }
515  else {
516  i2c_master_state = DATA_NACK;
517  I20CONSET = I2CONSET_STO; /* Set Stop flag */
518 
519  if (i2c_read_length != 0) {
520  I20CONSET = I2CONSET_STA; /* Set Repeated-start flag */
521  i2c_master_state = I2C_REPEATED_START;
522  }
523  }
524 
525  wr_index++;
526  }
527  else {
528  if (i2c_read_length != 0) {
529  I20CONSET = I2CONSET_STA; /* Set Repeated-start flag */
530  i2c_master_state = I2C_REPEATED_START;
531  }
532  else {
533  i2c_master_state = DATA_NACK;
534  I20CONSET = I2CONSET_STO; /* Set Stop flag */
535  }
536  }
537 
538  I20CONCLR = I2CONCLR_SIC;
539  break;
540 
541  case 0x40: /* Master Receive, SLA_R has been sent */
542 
543  //puts("Master Receive, SLA_R has been sent!\n");
544  if (i2c_read_length >= 2) {
545  I20CONSET = I2CONSET_AA; /* assert ACK after data is received */
546  }
547 
548  I20CONCLR = I2CONCLR_SIC;
549  break;
550 
551  // Data byte has been received, regardless following ACK or NACK
552  case 0x50:
553  case 0x58:
554  //puts("Data received\n");
555  i2c_master_buffer[3 + rd_index] = I20DAT;
556  rd_index++;
557 
558  if (rd_index < (i2c_read_length - 1)) {
559  i2c_master_state = DATA_ACK;
560  I20CONSET = I2CONSET_AA; /* assert ACK after data is received */
561  }
562  else {
563  I20CONCLR = I2CONCLR_AAC; /* NACK after data is received */
564  }
565 
566  if (rd_index == i2c_read_length) {
567  rd_index = 0;
568  i2c_master_state = DATA_NACK;
569  }
570 
571  I20CONCLR = I2CONCLR_SIC;
572  break;
573 
574  case 0x20: /* regardless, it's a NACK */
575  case 0x48:
576  I20CONCLR = I2CONCLR_SIC;
577  i2c_master_state = DATA_NACK;
578  break;
579 
580  case 0x38: /*
581  * Arbitration lost, in this example, we don't
582  * deal with multiple master situation
583  **/
584 
585  //puts("Arbritration lost!\n");
586  default:
587  I20CONCLR = I2CONCLR_SIC;
588  break;
589  }
590 
591  //IDISABLE;
592  //irq_disable();
593  //puts("leave I2C handler function\n");
594  VICVectAddr = 0; /* Acknowledge Interrupt */
595 }
