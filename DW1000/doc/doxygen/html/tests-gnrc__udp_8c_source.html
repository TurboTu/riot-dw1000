196 static void test_gnrc_udp__csum_zero(void)
197 {
198  uint8_t payload_data[] = {
199  0xFF, 0xFF, 0xFF, 0xE2,
200  };
201 
202  uint16_t checksum = 0;
203

205 
206  TEST_ASSERT_EQUAL_INT(0, status);
207  /* https://tools.ietf.org/html/rfc2460#section-8.1

211  */
212  TEST_ASSERT_EQUAL_INT(0xFFFF, checksum);

215 static void test_gnrc_udp__csum_all(void)
216 {
217  uint8_t payload_data[] = {
218  0x00, 0x00, 0xFF, 0xE2,
219  };
220 
221  for (uint32_t i = 0; i < 0x10000; i++) {
222  payload_data[0] = i >> 8;
223  payload_data[1] = i & 0xFF;
224 
225  uint16_t checksum = 0;
226 
227  int status = _compute_checksum(payload_data, sizeof(payload_data), &checksum);
228 
229  TEST_ASSERT_EQUAL_INT(0, status);
230  if (i == 0xFFFF) {
231  /* https://tools.ietf.org/html/rfc2460#section-8.1
232  * bullet 4
233  * "if that computation yields a result of zero, it must be changed
234  * to hex FFFF for placement in the UDP header."
235  */
236  TEST_ASSERT_EQUAL_INT(0xFFFF, checksum);
237  } else {
238  TEST_ASSERT_EQUAL_INT(0xFFFF - i, checksum);
239  }
240  }
