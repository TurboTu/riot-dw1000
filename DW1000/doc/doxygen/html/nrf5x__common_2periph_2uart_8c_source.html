137 void uart_write(uart_t uart, const uint8_t *data, size_t len)
138 {
139  if (uart == 0) {
140  for (size_t i = 0; i < len; i++) {
141  /* This section of the function is not thread safe:
142  - another thread may mess up with the uart at the same time.
143  In order to avoid an infinite loop in the interrupted thread,
144  the TXRDY flag must be cleared before writing the data to be
145  sent and not after. This way, the higher priority thread will
146  exit this function with the TXRDY flag set, then the interrupted
147  thread may have not transmitted his data but will still exit the
148  while loop.
149  */
150 
151  /* reset ready flag */
152  NRF_UART0->EVENTS_TXDRDY = 0;
153  /* write data into transmit register */
154  NRF_UART0->TXD = data[i];
155  /* wait for any transmission to be done */
156  while (NRF_UART0->EVENTS_TXDRDY == 0) {}
157  }
158  }
159 }
