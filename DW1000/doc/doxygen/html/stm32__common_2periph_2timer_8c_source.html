61 {
62  /* check if device is valid */
63  if (tim >= TIMER_NUMOF) {
64  return -1;
65  }
66 
67  /* remember the interrupt context */
68  isr_ctx[tim].cb = cb;
69  isr_ctx[tim].arg = arg;

71  /* enable the peripheral clock */
72  periph_clk_en(timer_config[tim].bus, timer_config[tim].rcc_mask);
73 
74  /* configure the timer as upcounter in continuous mode */
75  dev(tim)->CR1 = 0;
76  dev(tim)->CR2 = 0;
77  dev(tim)->ARR = timer_config[tim].max;
78

81  apbmul[timer_config[tim].bus]) / freq) - 1);
