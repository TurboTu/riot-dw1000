230 int pcd8544_init(pcd8544_t *dev, spi_t spi, gpio_t cs, gpio_t reset, gpio_t mode)
231 {
232  /* save pin mapping */
233  dev->spi = spi;
234  dev->cs = cs;
235  dev->reset = reset;
236  dev->mode = mode;
237  dev->inverted = 0;
238 
239  DEBUG("done setting dev members\n");
240 
241  /* initialze pins */
242  gpio_init(reset, GPIO_OUT);
243  gpio_init(mode, GPIO_OUT);
244  DEBUG("done with gpios\n");
245  /* clear CS line */
246  DEBUG("done clearing CS line\n");
247  /* initialize SPI */
248  spi_init_cs(spi, (spi_cs_t)cs);
249  DEBUG("done initializing SPI master\n");
250  /* reset display */
251  gpio_clear(reset);
252  xtimer_usleep(RESET_DELAY);
253  gpio_set(reset);
254 
255  /* clear display memory */
256  pcd8544_clear(dev);
257  /* write initialization sequence to display */
258  pcd8544_set_contrast(dev, PCD8544_DEFAULT_CONTRAST);
259  pcd8544_set_bias(dev, PCD8544_DEFAULT_BIAS);
260  pcd8544_set_tempcoef(dev, PCD8544_DEFAULT_TEMPCOEF);
261  /* enable display */
262  lock(dev);
263  _write(dev, MODE_CMD, CMD_ENABLE_H);
264  _write(dev, MODE_CMD, CMD_MODE_NORMAL);
265  done(dev);
266  return 0;
267 }

323 void pcd8544_write_c(pcd8544_t *dev, uint8_t x, uint8_t y, char c)
324 {
325  /* check position */
326  if (x >= PCD8544_COLS || y >= PCD8544_ROWS) {
327  return ;
328  }
329  /* set position */
330  lock(dev);
331  _set_x(dev, x * CHAR_WIDTH);
332  _set_y(dev, y);
333  /* write char */
334  for (int i = 0; i < CHAR_WIDTH - 1; i++) {
335  _write(dev, MODE_DTA, _ascii[c - ASCII_MIN][i]);
336  }
337  _write(dev, MODE_DTA, 0x00);
338  done(dev);
339 }
