365 {
366  struct sigaction sa;
367  int ret;
368 
369  /* update the signal mask so irq_enable()/irq_disable() will be aware */
370  if (add) {
371  _native_syscall_enter();
372  ret = sigdelset(&_native_sig_set, sig);
373  _native_syscall_leave();
374  } else {
375  _native_syscall_enter();
376  ret = sigaddset(&_native_sig_set, sig);
377  _native_syscall_leave();
378  }
379 
380  if (ret == -1) {
381  err(EXIT_FAILURE, "set_signal_handler: sigdelset");
382  }
383 
384  memset(&sa, 0, sizeof(sa));
385 
386  /* Disable other signal during execution of the handler for this signal. */
387  memcpy(&sa.sa_mask, &_native_sig_set_dint, sizeof(sa.sa_mask));
388 
389  /* restart interrupted systems call and custom signal stack */
390  sa.sa_flags = SA_RESTART | SA_ONSTACK;
391 
392  if (add) {
393  sa.sa_flags |= SA_SIGINFO; /* sa.sa_sigaction is used */
394  sa.sa_sigaction = native_isr_entry;
395  } else
396  {
397  sa.sa_handler = SIG_IGN;
398  }
399 
400  _native_syscall_enter();
401  if (sigaction(sig, &sa, NULL)) {
402  err(EXIT_FAILURE, "set_signal_handler: sigaction");
403  }
404  _native_syscall_leave();
