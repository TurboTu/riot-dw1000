53 static void sha1_hash_block(sha1_context *s)
54 {
55  uint8_t i;
56  uint32_t a, b, c, d, e, t;
57 
58  a = s->state[0];
59  b = s->state[1];
60  c = s->state[2];
61  d = s->state[3];
62  e = s->state[4];
63  for (i = 0; i < 80; i++) {
64  if (i >= 16) {
65  t = s->buffer[(i + 13) & 15] ^ s->buffer[(i + 8) & 15] ^
66  s->buffer[(i + 2) & 15] ^ s->buffer[i & 15];
67  s->buffer[i & 15] = sha1_rol32(t, 1);
68  }
69  if (i < 20) {
70  t = (d ^ (b & (c ^ d))) + SHA1_K0;
71  }
72  else if (i < 40) {
73  t = (b ^ c ^ d) + SHA1_K20;
74  }
75  else if (i < 60) {
76  t = ((b & c) | (d & (b | c))) + SHA1_K40;
77  }
78  else {
79  t = (b ^ c ^ d) + SHA1_K60;
80  }
81  t += sha1_rol32(a, 5) + e + s->buffer[i & 15];
82  e = d;
83  d = c;
84  c = sha1_rol32(b, 30);
85  b = a;
86  a = t;
87  }
88  s->state[0] += a;
89  s->state[1] += b;
90  s->state[2] += c;
91  s->state[3] += d;
92  s->state[4] += e;
93 }

125 static void sha1_pad(sha1_context *s)
126 {
127  /* Implement SHA-1 padding (fips180-2 §5.1.1) */
128 
129  /* Pad with 0x80 followed by 0x00 until the end of the block */
130  sha1_add_uncounted(s, 0x80);
131  while (s->buffer_offset != 56) {
132  sha1_add_uncounted(s, 0x00);
133  }
134 
135  /* Append length in the last 8 bytes */
136  sha1_add_uncounted(s, 0); /* We're only using 32 bit lengths */
137  sha1_add_uncounted(s, 0); /* But SHA-1 supports 64 bit lengths */
138  sha1_add_uncounted(s, 0); /* So zero pad the top bits */
139  sha1_add_uncounted(s, s->byte_count >> 29); /* Shifting to multiply by 8 */
140  sha1_add_uncounted(s, s->byte_count >> 21); /* as SHA-1 supports bitstreams as well as */
141  sha1_add_uncounted(s, s->byte_count >> 13); /* byte. */
142  sha1_add_uncounted(s, s->byte_count >> 5);
143  sha1_add_uncounted(s, s->byte_count << 3);
144 }

146 void sha1_final(sha1_context *ctx, void *digest)
147 {
148  /* Pad to complete the last block */
149  sha1_pad(ctx);
150 
151  /* Swap byte order back */
152  for (int i = 0; i < 5; i++) {
153  ctx->state[i] =
154  (((ctx->state[i]) << 24) & 0xff000000)
155  | (((ctx->state[i]) << 8) & 0x00ff0000)
156  | (((ctx->state[i]) >> 8) & 0x0000ff00)
157  | (((ctx->state[i]) >> 24) & 0x000000ff);
158  }
159 
160  /* Copy the content of the hash (20 characters) */
161  memcpy(digest, ctx->state, 20);
162 }
