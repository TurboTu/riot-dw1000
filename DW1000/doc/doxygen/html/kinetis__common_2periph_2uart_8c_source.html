94 static int init_base(uart_t uart, uint32_t baudrate)
95 {
96  KINETIS_UART *dev;
97  PORT_Type *port;
98  uint32_t clk;
99  uint16_t ubd;
100  uint8_t tx_pin = 0;
101  uint8_t rx_pin = 0;
102  uint8_t af;
103 
104  switch (uart) {
105 #if UART_0_EN
106 
107  case UART_0:
108  dev = UART_0_DEV;
109  port = UART_0_PORT;
110  clk = UART_0_CLK;
111  tx_pin = UART_0_TX_PIN;
112  rx_pin = UART_0_RX_PIN;
113  af = UART_0_AF;
114  UART_0_PORT_CLKEN();
115  UART_0_CLKEN();
116  break;
117 #endif
118 #if UART_1_EN
119 
120  case UART_1:
121  dev = UART_1_DEV;
122  port = UART_1_PORT;
123  clk = UART_1_CLK;
124  tx_pin = UART_1_TX_PIN;
125  rx_pin = UART_1_RX_PIN;
126  af = UART_1_AF;
127  UART_1_PORT_CLKEN();
128  UART_1_CLKEN();
129  break;
130 #endif
131 
132  default:
133  return UART_NODEV;
134  }
135 
136  /* configure RX and TX pins, set pin to use alternative function mode */
137  port->PCR[rx_pin] = PORT_PCR_MUX(af);
138  port->PCR[tx_pin] = PORT_PCR_MUX(af);
139 
140  /* disable transmitter and receiver */
141  dev->C2 &= ~(UART_C2_TE_MASK | UART_C2_RE_MASK);
142  /* set defaults, 8-bit mode, no parity */
143  dev->C1 = 0;
144 
145  /* calculate baudrate */
146  ubd = (uint16_t)(clk / (baudrate * 16));
147 
148  /* set baudrate */
149  dev->BDH = (uint8_t)UART_BDH_SBR(ubd >> 8);
150  dev->BDL = (uint8_t)UART_BDL_SBR(ubd);
151  kinetis_set_brfa(dev, baudrate, clk);
152 
153 #if KINETIS_UART_ADVANCED
154  /* Enable FIFO buffers */
155  dev->PFIFO |= UART_PFIFO_RXFE_MASK | UART_PFIFO_TXFE_MASK;
156  /* Set level to trigger TDRE flag whenever there is space in the TXFIFO */
157  /* FIFO size is 2^(PFIFO_TXFIFOSIZE + 1) (4, 8, 16 ...) for values != 0.
158  * TXFIFOSIZE == 0 means size = 1 (i.e. only one byte, no hardware FIFO) */
159  if ((dev->PFIFO & UART_PFIFO_TXFIFOSIZE_MASK) != 0) {
160  uint8_t txfifo_size =
161  (2 << ((dev->PFIFO & UART_PFIFO_TXFIFOSIZE_MASK) >>
162  UART_PFIFO_TXFIFOSIZE_SHIFT));
163  dev->TWFIFO = UART_TWFIFO_TXWATER(txfifo_size - 1);
164  }
165  else {
166  /* Missing hardware support */
167  dev->TWFIFO = 0;
168  }
169  /* Trigger RX interrupt when there is 1 byte or more in the RXFIFO */
170  dev->RWFIFO = 1;
171  /* Clear all hardware buffers now, this must be done whenever the FIFO
172  * enable flags are modified. */
173  dev->CFIFO = UART_CFIFO_RXFLUSH_MASK | UART_CFIFO_TXFLUSH_MASK;
174 #endif
175 
176  /* enable transmitter and receiver */
177  dev->C2 |= UART_C2_TE_MASK | UART_C2_RE_MASK;
178  return UART_OK;
179 }

181 void uart_write(uart_t uart, const uint8_t *data, size_t len)
182 {
183  KINETIS_UART *dev;
184 
185  switch (uart) {
186 #if UART_0_EN
187  case UART_0:
188  dev = UART_0_DEV;
189  break;
190 #endif
191 #if UART_1_EN
192  case UART_1:
193  dev = UART_1_DEV;
194  break;
195 #endif
196  default:
197  return;
198  }
199 
200  for (size_t i = 0; i < len; i++) {
201  while (!(dev->S1 & UART_S1_TDRE_MASK));
202  dev->D = data[i];
203  }
204 }

206 static inline void irq_handler(uart_t uartnum, KINETIS_UART *dev)
207 {
208  /*
209  * On Cortex-M0, it happens that S1 is read with LDR
210  * instruction instead of LDRB. This will read the data register
211  * at the same time and arrived byte will be lost. Maybe it's a GCC bug.
212  *
213  * Observed with: arm-none-eabi-gcc (4.8.3-8+..)
214  * It does not happen with: arm-none-eabi-gcc (4.8.3-9+11)
215  */
216 
217  if (dev->S1 & UART_S1_RDRF_MASK) {
218  /* RDRF flag will be cleared when dev-D was read */
219  uint8_t data = dev->D;
220 
221  if (config[uartnum].rx_cb != NULL) {
222  config[uartnum].rx_cb(config[uartnum].arg, data);
223  }
224  }
225 
226 #if (KINETIS_UART_ADVANCED == 0)
227  /* clear overrun flag */
228  if (dev->S1 & UART_S1_OR_MASK) {
229  dev->S1 = UART_S1_OR_MASK;
230  }
231 #endif
232 
233  cortexm_isr_end();
234 
235 }
