108 gnrc_pktsnip_t *gnrc_pktbuf_mark(gnrc_pktsnip_t *pkt, size_t size, gnrc_nettype_t type)
109 {
110  gnrc_pktsnip_t *marked_snip;
111  /* size required for chunk */
112  size_t required_new_size = (size < sizeof(_unused_t)) ?
113  _align(sizeof(_unused_t)) : _align(size);
114  void *new_data_marked;
115 
116  mutex_lock(&_mutex);
117  if ((size == 0) || (pkt == NULL) || (size > pkt->size) || (pkt->data == NULL)) {
118  DEBUG("pktbuf: size == 0 (was %u) or pkt == NULL (was %p) or "
119  "size > pkt->size (was %u) or pkt->data == NULL (was %p)\n",
120  (unsigned)size, (void *)pkt, (pkt ? (unsigned)pkt->size : 0),
121  (pkt ? pkt->data : NULL));
122  mutex_unlock(&_mutex);
123  return NULL;
124  }
125  /* create new snip descriptor for marked data */
126  marked_snip = _pktbuf_alloc(sizeof(gnrc_pktsnip_t));
127  if (marked_snip == NULL) {
128  DEBUG("pktbuf: could not reallocate marked section.\n");
129  mutex_unlock(&_mutex);
130  return NULL;
131  }
132  /* marked data would not fit _unused_t marker => move data around to allow
133  * for proper free */
134  if ((pkt->size != size) &&
135  ((size < required_new_size) || ((pkt->size - size) < sizeof(_unused_t)))) {
136  void *new_data_rest;
137  new_data_marked = _pktbuf_alloc(size);
138  if (new_data_marked == NULL) {
139  DEBUG("pktbuf: could not reallocate marked section.\n");
140  _pktbuf_free(marked_snip, sizeof(gnrc_pktsnip_t));
141  mutex_unlock(&_mutex);
142  return NULL;
143  }
144  new_data_rest = _pktbuf_alloc(pkt->size - size);
145  if (new_data_rest == NULL) {
146  DEBUG("pktbuf: could not reallocate remaining section.\n");
147  _pktbuf_free(marked_snip, sizeof(gnrc_pktsnip_t));
148  _pktbuf_free(new_data_marked, size);
149  mutex_unlock(&_mutex);
150  return NULL;
151  }
152  memcpy(new_data_marked, pkt->data, size);
153  memcpy(new_data_rest, ((uint8_t *)pkt->data) + size, pkt->size - size);
154  _pktbuf_free(pkt->data, pkt->size);
155  marked_snip->data = new_data_marked;
156  pkt->data = new_data_rest;
157  }
158  else {
159  new_data_marked = pkt->data;
160  /* if (pkt->size - size) != 0 take remainder of data, otherwise set NULL */
161  pkt->data = (pkt->size != size) ? (((uint8_t *)pkt->data) + size) :
162  NULL;
163  }
164  pkt->size -= size;
165  _set_pktsnip(marked_snip, pkt->next, new_data_marked, size, type);
166  pkt->next = marked_snip;
167  mutex_unlock(&_mutex);
168  return marked_snip;
169 }

171 int gnrc_pktbuf_realloc_data(gnrc_pktsnip_t *pkt, size_t size)
172 {
173  size_t aligned_size = (size < sizeof(_unused_t)) ?
174  _align(sizeof(_unused_t)) : _align(size);
175 
176  mutex_lock(&_mutex);
177  assert(pkt != NULL);
178  assert(((pkt->size == 0) && (pkt->data == NULL)) ||
179  ((pkt->size > 0) && (pkt->data != NULL) && _pktbuf_contains(pkt->data)));
180  /* new size and old size are equal */
181  if (size == pkt->size) {
182  /* nothing to do */
183  mutex_unlock(&_mutex);
184  return 0;
185  }
186  /* new size is 0 and data pointer isn't already NULL */
187  if ((size == 0) && (pkt->data != NULL)) {
188  /* set data pointer to NULL */
189  _pktbuf_free(pkt->data, pkt->size);
190  pkt->data = NULL;
191  }
192  /* if new size is bigger than old size */
193  else if ((size > pkt->size) || /* new size does not fit */
194  ((pkt->size - aligned_size) < sizeof(_unused_t))) { /* resulting hole would not fit marker */
195  void *new_data = _pktbuf_alloc(size);
196  if (new_data == NULL) {
197  DEBUG("pktbuf: error allocating new data section\n");
198  mutex_unlock(&_mutex);
199  return ENOMEM;
200  }
201  if (pkt->data != NULL) { /* if old data exist */
202  memcpy(new_data, pkt->data, (pkt->size < size) ? pkt->size : size);
203  }
204  _pktbuf_free(pkt->data, pkt->size);
205  pkt->data = new_data;
206  }
207  else if (_align(pkt->size) > aligned_size) {
208  _pktbuf_free(((uint8_t *)pkt->data) + aligned_size,
209  pkt->size - aligned_size);
210  }
211  pkt->size = size;
212  mutex_unlock(&_mutex);
213  return 0;
214 }

373 bool gnrc_pktbuf_is_sane(void)
374 {
375  _unused_t *ptr = _first_unused;
376 
377  /* Invariants of this implementation:
378  * - the head of _unused_t list is _first_unused
379  * - if _unused_t list is empty the packet buffer is full and _first_unused is NULL
380  * - forall ptr_in _unused_t list: &_pktbuf[0] < ptr < &_pktbuf[GNRC_PKTBUF_SIZE]
381  * - forall ptr in _unused_t list: ptr->next == NULL || ptr < ptr->next
382  * - forall ptr in _unused_t list: (ptr->next != NULL && ptr->size <= (ptr->next - ptr)) ||
383  * (ptr->next == NULL && ptr->size == (GNRC_PKTBUF_SIZE - (ptr - &_pktbuf[0])))
384  */
385 
386  while (ptr) {
387  if (&_pktbuf[0] >= (uint8_t *)ptr && (uint8_t *)ptr >= &_pktbuf[GNRC_PKTBUF_SIZE]) {
388  return false;
389  }
390  if ((ptr->next != NULL) && (ptr >= ptr->next)) {
391  return false;
392  }
393  if (((ptr->next == NULL) || (ptr->size > (size_t)((uint8_t *)(ptr->next) - (uint8_t *)ptr))) &&
394  ((ptr->next != NULL) ||
395  (ptr->size != (size_t)(GNRC_PKTBUF_SIZE - ((uint8_t *)ptr - &_pktbuf[0]))))) {
396  return false;
397  }
398  ptr = ptr->next;
399  }
400 
401  return true;
402 }

430 static void *_pktbuf_alloc(size_t size)
431 {
432  _unused_t *prev = NULL, *ptr = _first_unused;
433 
434  size = (size < sizeof(_unused_t)) ? _align(sizeof(_unused_t)) : _align(size);
435  while (ptr && (size > ptr->size)) {
436  prev = ptr;
437  ptr = ptr->next;
438  }
439  if (ptr == NULL) {
440  DEBUG("pktbuf: no space left in packet buffer\n");
441  return NULL;
442  }
443  /* _unused_t struct would fit => add new space at ptr */
444  if (sizeof(_unused_t) > (ptr->size - size)) {
445  if (prev == NULL) { /* ptr was _first_unused */
446  _first_unused = ptr->next;
447  }
448  else {
449  prev->next = ptr->next;
450  }
451  }
452  else {
453  _unused_t *new = (_unused_t *)(((uint8_t *)ptr) + size);
454 
455  if (((((uint8_t *)new) - &(_pktbuf[0])) + sizeof(_unused_t)) > GNRC_PKTBUF_SIZE) {
456  /* content of new would exceed packet buffer size so set to NULL */
457  _first_unused = NULL;
458  }
459  else if (prev == NULL) { /* ptr was _first_unused */
460  _first_unused = new;
461  }
462  else {
463  prev->next = new;
464  }
465  new->next = ptr->next;
466  new->size = ptr->size - size;
467  }
468 #ifdef DEVELHELP
469  uint16_t last_byte = (uint16_t)((((uint8_t *)ptr) + size) - &(_pktbuf[0]));
470  if (last_byte > max_byte_count) {
471  max_byte_count = last_byte;
472  }
473 #endif
474  return (void *)ptr;
475 }

491 static void _pktbuf_free(void *data, size_t size)
492 {
493  size_t bytes_at_end;
494  _unused_t *new = (_unused_t *)data, *prev = NULL, *ptr = _first_unused;
495 
496  if (!_pktbuf_contains(data)) {
497  return;
498  }
499  while (ptr && (((void *)ptr) < data)) {
500  prev = ptr;
501  ptr = ptr->next;
502  }
503  new->next = ptr;
504  new->size = (size < sizeof(_unused_t)) ? _align(sizeof(_unused_t)) : _align(size);
505  /* calculate number of bytes between new _unused_t chunk and end of packet
506  * buffer */
507  bytes_at_end = ((&_pktbuf[0] + GNRC_PKTBUF_SIZE) - (((uint8_t *)new) + new->size));
508  if (bytes_at_end < _align(sizeof(_unused_t))) {
509  /* new is very last segment and there is a little bit of memory left
510  * that wouldn't fit _unused_t (cut of in _pktbuf_alloc()) => re-add it */
511  new->size += bytes_at_end;
512  }
513  if (prev == NULL) { /* ptr was _first_unused or data before _first_unused */
514  _first_unused = new;
515  }
516  else {
517  prev->next = new;
518  if (_too_small_hole(prev, new)) {
519  new = _merge(prev, new);
520  }
521  }
522  if ((new->next != NULL) && (_too_small_hole(new, new->next))) {
523  _merge(new, new->next);
524  }
525 }
