147 void daemonize(void)
148 {
149  if ((_native_pid = real_fork()) == -1) {
150  err(EXIT_FAILURE, "daemonize: fork");
151  }
152 
153  if (_native_pid > 0) {
154  real_printf("RIOT pid: %d\n", _native_pid);
155  real_exit(EXIT_SUCCESS);
156  }

159
