175 inline static int pit_init(uint8_t dev, uint32_t freq, timer_cb_t cb, void *arg)
176 {
177  PIT_CLOCKGATE = 1;
178  /* Completely disable the module before messing with the settings */
179  PIT->MCR = PIT_MCR_MDIS_MASK;
180 
181  /* Disable IRQs to avoid race with ISR */
182  unsigned int mask = irq_disable();
183 
184  /* Clear configuration */
185  PIT->CHANNEL[pit_config[dev].count_ch].TCTRL = 0;
186 
187  /* Freeze timers during debug break, resume normal operations (clear MDIS) */
188  PIT->MCR = PIT_MCR_FRZ_MASK;
189 
190  _pit_set_cb_config(dev, cb, arg);
191 
192  /* Clear IRQ flag */
193  PIT->CHANNEL[pit_config[dev].count_ch].TFLG = PIT_TFLG_TIF_MASK;
194  /* Refactor the below lines if there are any CPUs where the PIT IRQs are not sequential */
195  NVIC_ClearPendingIRQ(PIT0_IRQn + pit_config[dev].count_ch);
196  NVIC_EnableIRQ(PIT0_IRQn + pit_config[dev].count_ch);
197 
198  /* Reset up-counter */
199  pit[dev].count = PIT_MAX_VALUE;
200  pit[dev].ldval = PIT_MAX_VALUE;
201  pit[dev].tctrl = PIT_TCTRL_CHN_MASK | PIT_TCTRL_TEN_MASK;
202  _pit_set_prescaler(pit_config[dev].prescaler_ch, freq);
203  _pit_set_counter(dev);
204 
205  irq_restore(mask);
206  return 0;
207 }

209 inline static int pit_set(uint8_t dev, uint32_t timeout)
210 {
211  const uint8_t ch = pit_config[dev].count_ch;
212  /* Disable IRQs to minimize the number of lost ticks */
213  unsigned int mask = irq_disable();
214  pit[dev].ldval = timeout;
215  pit[dev].tctrl = PIT_TCTRL_TIE_MASK | PIT_TCTRL_CHN_MASK | PIT_TCTRL_TEN_MASK;
216  /* Add the new timeout offset to the up-counter */
217  pit[dev].count += timeout;
218  if (PIT_BITBAND_TEN(ch) != 0) {
219  /* Timer is currently running */
220  uint32_t cval = PIT->CHANNEL[ch].CVAL;
221  /* Subtract if there was anything left on the counter */
222  pit[dev].count -= cval;
223  _pit_set_counter(dev);
224  }
225  irq_restore(mask);
226  return 0;
227 }

343 inline static uint32_t _rtt_get_subtick(void)
344 {
345  uint32_t tpr;
346  uint32_t tsr;
347 
348  for (int i = 0; i < 5; i++) {
349  /* Read twice to make sure we get a stable reading */
350  tpr = RTC->TPR & RTC_TPR_TPR_MASK;
351  tsr = RTC->TSR & RTC_TSR_TSR_MASK;
352 
353  if ((tsr == (RTC->TSR & RTC_TSR_TSR_MASK)) &&
354  (tpr == (RTC->TPR & RTC_TPR_TPR_MASK))) {
355  break;
356  }
357  }
358  if (tpr > TIMER_RTC_SUBTICK_MAX) {
359  /* This only happens if the RTC time compensation value has been
360  * modified to compensate for RTC drift. See Kinetis ref.manual,
361  * RTC Time Compensation Register (RTC_TCR).
362  */
363  tpr = TIMER_RTC_SUBTICK_MAX;
364  }
365 
366  return (tsr << TIMER_RTC_SUBTICK_BITS) | tpr;
367 }

379 inline static int32_t _lptmr_compute_prescaler(uint32_t freq) {
380  uint32_t prescale = 0;
381  if ((freq > LPTMR_BASE_FREQ) || (freq == 0)) {
382  /* Frequency out of range */
383  return -1;
384  }
385  while (freq < LPTMR_BASE_FREQ){
386  ++prescale;
387  freq <<= 1;
388  }
389  if (freq != LPTMR_BASE_FREQ) {
390  /* freq was not a power of two division of LPTMR_BASE_FREQ */
391  return -2;
392  }
393  if (prescale > 0) {
394  /* LPTMR_PSR_PRESCALE == 0 yields LPTMR_BASE_FREQ/2,
395  * LPTMR_PSR_PRESCALE == 1 yields LPTMR_BASE_FREQ/4 etc.. */
396  return LPTMR_PSR_PRESCALE(prescale - 1);
397  }
398  else {
399  /* Prescaler bypass enabled */
400  return LPTMR_PSR_PBYP_MASK;
401  }
402 }

430 inline static int lptmr_init(uint8_t dev, uint32_t freq, timer_cb_t cb, void *arg)
431 {
432  int32_t prescale = _lptmr_compute_prescaler(freq);
433  if (prescale < 0) {
434  return -1;
435  }
436  LPTMR_Type *hw = lptmr_config[dev].dev;
437  /* Disable IRQs to avoid race with ISR */
438  unsigned int mask = irq_disable();
439 
440  /* Turn on module clock */
441  *lptmr_config[dev].clk_gate = 1;
442  /* Completely disable the module before messing with the settings */
443  hw->CSR = 0;
444  /* select ERCLK32K as clock source for LPTMR */
445  hw->PSR = LPTMR_PSR_PCS(2) | ((uint32_t)prescale);
446 
447  /* Clear IRQ flag in case it was already set */
448  hw->CSR = LPTMR_CSR_TCF_MASK;
449  /* Enable IRQs on the counting channel */
450  /* Refactor the below lines if there are any CPUs where the LPTMR IRQs are not sequential */
451  NVIC_ClearPendingIRQ(LPTMR0_IRQn + lptmr_config[dev].index);
452  NVIC_EnableIRQ(LPTMR0_IRQn + lptmr_config[dev].index);
453 
454  _lptmr_set_cb_config(dev, cb, arg);
455 
456  /* Reset state */
457  _lptmr_update_rtt_offset(dev);
458  lptmr[dev].running = 1;
459  lptmr_clear(dev);
460 
461  irq_restore(mask);
462 
463  return 0;
464 }
