83 int gnrc_netreg_num(gnrc_nettype_t type, uint32_t demux_ctx)
84 {
85  int num = 0;

87 
88  if (_INVALID_TYPE(type)) {
89  return 0;
90  }
91 
92  entry = netreg[type];
93 
94  while (entry != NULL) {
95  if (entry->demux_ctx == demux_ctx) {
96  num++;

98 
99  entry = entry->next;
100  }
101 
102  return num;
