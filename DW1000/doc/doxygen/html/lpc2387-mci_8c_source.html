101 void Isr_MCI(void)
102 {
103  unsigned long ms;
104  unsigned char xs;
105 
106  ms = MCI_STATUS & 0x073A; /* Clear MCI interrupt status */
107  MCI_CLEAR = ms;
108 
109  xs = XferStat;
110 
111  if (ms & 0x400) { /* A block transfer completed (DataBlockEnd) */
112  if (xs & 1) { /* In card read operation */
113  if (ms & 0x100) { /* When last block is received (DataEnd), */
114  GPDMA_SOFT_BREQ = 0x10; /* Pop off remaining data in the MCIFIFO */
115  }
116 
117  unsigned char n = (XferWp + 1) % N_BUF; /* Next write buffer */
118  XferWp = n;
119 
120  if (n == XferRp) {
121  xs |= 4; /* Check block overrun */
122  }
123  }
124  else { /* In card write operation */
125  unsigned char n = (XferRp + 1) % N_BUF; /* Next read buffer */
126  XferRp = n;
127 
128  if (n == XferWp) {
129  xs |= 4; /* Check block underrun */
130  }
131  }
132  }
133  else { /* An MCI error occured (not DataBlockEnd) */
134  xs |= 8;
135  }
136 
137  XferStat = xs;
138 
139  VICVectAddr = 0;
140 }

170 static void ready_reception(unsigned int blks, unsigned int bs)
171 {
172  unsigned int n;
173  unsigned long dma_ctrl;
174 
175  /* ------ Setting up GPDMA Ch-0 ------ */
176  GPDMA_CH0_CFG &= 0xFFF80420; /* Disable ch-0 */
177  GPDMA_INT_TCCLR = 0x01; /* Clear interrupt flag */
178  dma_ctrl = 0x88492000 | (bs / 4); /* 1_000_1_0_00_010_010_010_010_************ */
179 
180  /* Create link list */
181  for (n = 0; n < N_BUF; n++) {
182  LinkList[n][0] = (unsigned long)&MCI_FIFO;
183  LinkList[n][1] = (unsigned long)DmaBuff[n];
184  LinkList[n][2] = (unsigned long)LinkList[(n + 1) % N_BUF];
185  LinkList[n][3] = dma_ctrl;
186  }
187 
188  /* Load first LLI */
189  GPDMA_CH0_SRC = LinkList[0][0];
190  GPDMA_CH0_DEST = LinkList[0][1];
191  GPDMA_CH0_LLI = LinkList[0][2];
192  GPDMA_CH0_CTRL = LinkList[0][3];
193 
194  /* Enable ch-0 */
195  GPDMA_CH0_CFG |= 0x19009; /* *************_0_0_1_1_0_010_*_0000_*_0100_1 */
196 
197  /* --------- Setting up MCI ---------- */
198 
199  XferRp = 0;
200  XferWp = 0; /* Block FIFO R/W index */
201  XferStat = 1; /* Transfer status: MCI --> Memory */
202 
203  MCI_DATA_LEN = bs * blks; /* Set total data length */
204  MCI_DATA_TMR = (unsigned long)(MCLK_RW * 0.2); /* Data timer: 0.2sec */
205  MCI_CLEAR = 0x72A; /* Clear status flags */
206  MCI_MASK0 = 0x72A; /* DataBlockEnd StartBitErr DataEnd RxOverrun DataTimeOut DataCrcFail */
207 
208  for (n = 0; bs > 1; bs >>= 1, n += 0x10);
209 
210  MCI_DATA_CTRL = n | 0xB; /* Start to receive data blocks */
211 }

293 static void power_on(void)
294 {
295  /* Enable MCI and GPDMA clock */
296  PCONP |= (3 << 28);
297 
298  /* Enable GPDMA controller with little-endian */
299  GPDMA_CH0_CFG &= 0xFFF80000; /* Disable DMA ch-0 */
300  GPDMA_CONFIG = 0x01;
301 
302  /* Select PCLK for MCI, CCLK/2 = 36MHz */
303  PCLKSEL1 = (PCLKSEL1 & 0xFCFFFFFF) | 0x02000000;
304 
305  //0.19 0.20 0.21 0.22
306  PINMODE1 &= ~((BIT6 | BIT7) | (BIT8 | BIT9) | (BIT10 | BIT11) | (BIT12 | BIT13));
307  PINMODE1 |= (BIT7) | (BIT9) | (BIT11) | (BIT13); // no resistors
308  //2.11 2.12 2.13
309  PINMODE4 &= ~((BIT22 | BIT23) | (BIT24 | BIT25) | (BIT26 | BIT27));
310  PINMODE4 |= (BIT23) | (BIT25) | (BIT27); // no resistors
311  /* Attach MCI unit to I/O pad */
312  PINSEL1 = (PINSEL1 & 0xFFFFC03F) | 0x00002A80; /* MCICLK, MCICMD, MCIDATA0, MCIPWR */
313 #if USE_4BIT
314  PINSEL4 = (PINSEL4 & 0xF03FFFFF) | 0x0A800000; /* MCIDATA1-3 */
315 #endif
316  MCI_MASK0 = 0;
317  MCI_COMMAND = 0;
318  MCI_DATA_CTRL = 0;
319 
320  // pin 0.21 high active
321  SCS |= 0x08;
322 
323  /* Register interrupt handlers for MCI,DMA event */
324  //RegisterIrq(MCI_INT, Isr_MCI, PRI_LOWEST-1);
325  install_irq(MCI_INT, Isr_MCI, 5);
326 
327  //RegisterIrq(GPDMA_INT, Isr_GPDMA, PRI_LOWEST-1);
328  install_irq(GPDMA_INT, Isr_GPDMA, 5);
329 
330 
331  /* Power-on (VCC is always tied to the socket on this board) */
332  MCI_POWER = 0x01; /* Power on */
333 
334  //for (Timer[0] = 10; Timer[0]; ) ; /* 10ms */
335  xtimer_usleep(1000);
336 
337  MCI_POWER = 0x03; /* Enable signals */
338 }

368 /*-----------------------------------------------------------------------*/
369 /* Send a command packet to the card and receive a response */
370 /*-----------------------------------------------------------------------*/
371 
372 /*
373  * @param idx Command index (bit[5..0]), ACMD flag (bit7)
374  * @param arg Command argument
375  * @param rt Expected response type. None(0), Short(1) or Long(2)
376  * @param *buff Response return buffer
377  * @return 1 when function succeeded otherwise returns 0
378  * */
379 static int send_cmd(unsigned int idx, unsigned long arg, unsigned int rt, unsigned long *buff)
380 {
381  unsigned int s, mc;
382 
383  if (idx & 0x80) { /* Send a CMD55 prior to the specified command if it is ACMD class */
384  if (!send_cmd(CMD55, (unsigned long)CardRCA << 16, 1, buff) /* When CMD55 is faild, */
385  || !(buff[0] & 0x00000020)) {
386  return 0; /* exit with error */
387  }
388  }
389 
390  idx &= 0x3F; /* Mask out ACMD flag */
391 
392  do { /* Wait while CmdActive bit is set */
393  MCI_COMMAND = 0; /* Cancel to transmit command */
394  MCI_CLEAR = 0x0C5; /* Clear status flags */
395 
396  for (s = 0; s < 10; s++) {
397  MCI_STATUS; /* Skip lock out time of command reg. */
398  }
399  }
400  while (MCI_STATUS & 0x00800);
401 
402  MCI_ARGUMENT = arg; /* Set the argument into argument register */
403  mc = 0x400 | idx; /* Enable bit + index */
404 
405  if (rt == 1) {
406  mc |= 0x040; /* Set Response bit to reveice short resp */
407  }
408 
409  if (rt > 1) {
410  mc |= 0x0C0; /* Set Response and LongResp bit to receive long resp */
411  }
412 
413  MCI_COMMAND = mc; /* Initiate command transaction */
414 
415  //Timer[1] = 100;
416  uint32_t timerstart = xtimer_now_usec();
417 
418  while (1) { /* Wait for end of the cmd/resp transaction */
419 
420  //if (!Timer[1]) return 0;
421  if ((xtimer_now_usec() - timerstart) > 10000) {
422  return 0;
423  }
424 
425  s = MCI_STATUS; /* Get the transaction status */
426 
427  if (rt == 0) {
428  if (s & 0x080) {
429  return 1; /* CmdSent */
430  }
431  }
432  else {
433  if (s & 0x040) {
434  break; /* CmdRespEnd */
435  }
436 
437  if (s & 0x001) {
438  /* CmdCrcFail */
439  if (idx == 1 || idx == 12 || idx == 41) { /* Ignore CRC error on CMD1/12/41 */
440  break;
441  }
442 
443  return 0;
444  }
445 
446  if (s & 0x004) {
447  return 0; /* CmdTimeOut */
448  }
449  }
450  }
451 
452  buff[0] = MCI_RESP0; /* Read the response words */
453 
454  if (rt == 2) {
455  buff[1] = MCI_RESP1;
456  buff[2] = MCI_RESP2;
457  buff[3] = MCI_RESP3;
458  }
459 
460  return 1; /* Return with success */
461 }

514 /*-----------------------------------------------------------------------*/
515 /* Initialize Disk Drive */
516 /*-----------------------------------------------------------------------*/
517 diskio_sta_t mci_initialize(void)
518 {
519  unsigned int cmd, n;
520  unsigned long resp[4];
521  unsigned char ty;
522 
523  if (Stat & DISKIO_STA_NODISK) {
524  return Stat; /* No card in the socket */
525  }
526 
527  power_off();
528 
529  xtimer_usleep(1000);
530 
531  power_on(); /* Force socket power on */
532  MCI_CLOCK = 0x100 | (PCLK / MCLK_ID / 2 - 1); /* Set MCICLK = MCLK_ID */
533  //for (Timer[0] = 2; Timer[0]; );
534  xtimer_usleep(250);
535 
536  send_cmd(CMD0, 0, 0, NULL); /* Enter idle state */
537  CardRCA = 0;
538 
539  /*---- Card is 'idle' state ----*/
540 
541  /* Initialization timeout of 1000 msec */
542  uint32_t start = xtimer_now_usec();
543 
544  /* SDC Ver2 */
545  if (send_cmd(CMD8, 0x1AA, 1, resp) && (resp[0] & 0xFFF) == 0x1AA) {
546  /* The card can work at vdd range of 2.7-3.6V */
547  DEBUG("SDC Ver. 2\n");
548 
549  do {
550  /* Wait while card is busy state (use ACMD41 with HCS bit) */
551  /* This loop will take a time. Insert wai_tsk(1) here for multitask envilonment. */
552  if (xtimer_now_usec() > (start + 1000000/* !Timer[0] */)) {
553  DEBUG("%s, %d: Timeout #1\n", RIOT_FILE_RELATIVE, __LINE__);
554  goto di_fail;
555  }
556  }
557  while (!send_cmd(ACMD41, 0x40FF8000, 1, resp) || !(resp[0] & 0x80000000));
558 
559  ty = (resp[0] & 0x40000000) ? CT_SD2 | CT_BLOCK : CT_SD2; /* Check CCS bit in the OCR */
560  }
561  else { /* SDC Ver1 or MMC */
562  if (send_cmd(ACMD41, 0x00FF8000, 1, resp)) {
563  DEBUG("SDC Ver. 1\n");
564  ty = CT_SD1;
565  cmd = ACMD41; /* ACMD41 is accepted -> SDC Ver1 */
566  }
567  else {
568  DEBUG("MMC\n");
569  ty = CT_MMC;
570  cmd = CMD1; /* ACMD41 is rejected -> MMC */
571  }
572 
573  do {
574  /* Wait while card is busy state (use ACMD41 or CMD1) */
575  DEBUG("%s, %d: %lX\n", RIOT_FILE_RELATIVE, __LINE__, resp[0]);
576 
577  /* This loop will take a time. Insert wai_tsk(1) here for multitask envilonment. */
578  if (xtimer_now_usec() > (start + 1000000/* !Timer[0] */)) {
579  DEBUG("now: %lu, started at: %lu\n", xtimer_now_usec(), start);
580  DEBUG("%s, %d: Timeout #2\n", RIOT_FILE_RELATIVE, __LINE__);
581  goto di_fail;
582  }
583  }
584  while (!send_cmd(cmd, 0x00FF8000, 1, resp) || !(resp[0] & 0x80000000));
585  }
586 
587  CardType = ty; /* Save card type */
588  bswap_cp(&CardInfo[32], resp); /* Save OCR */
589 
590  /*---- Card is 'ready' state ----*/
591 
592  if (!send_cmd(CMD2, 0, 2, resp)) {
593  DEBUG("%s, %d: Failed entering ident state", RIOT_FILE_RELATIVE,
594  __LINE__);
595  goto di_fail; /* Enter ident state */
596  }
597 
598  for (n = 0; n < 4; n++) {
599  bswap_cp(&CardInfo[n * 4 + 16], &resp[n]); /* Save CID */
600  }
601 
602  /*---- Card is 'ident' state ----*/
603 
604  if (ty & CT_SDC) { /* SDC: Get generated RCA and save it */
605  if (!send_cmd(CMD3, 0, 1, resp)) {
606  DEBUG("%s, %d: Failed generating RCA\n", RIOT_FILE_RELATIVE, __LINE__);
607  goto di_fail;
608  }
609 
610  CardRCA = (unsigned short)(resp[0] >> 16);
611  }
612  else { /* MMC: Assign RCA to the card */
613  if (!send_cmd(CMD3, 1 << 16, 1, resp)) {
614  goto di_fail;
615  }
616 
617  CardRCA = 1;
618  }
619 
620  /*---- Card is 'stby' state ----*/
621 
622  if (!send_cmd(CMD9, (unsigned long)CardRCA << 16, 2, resp)) { /* Get CSD and save it */
623  goto di_fail;
624  }
625 
626  for (n = 0; n < 4; n++) {
627  bswap_cp(&CardInfo[n * 4], &resp[n]);
628  }
629 
630  if (!send_cmd(CMD7, (unsigned long)CardRCA << 16, 1, resp)) { /* Select card */
631  //printf("MCI CMD7 fail\n");
632  goto di_fail;
633  }
634 
635  /*---- Card is 'tran' state ----*/
636 
637  if (!(ty & CT_BLOCK)) { /* Set data block length to 512 (for byte addressing cards) */
