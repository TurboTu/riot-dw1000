36 int lis3dh_init(lis3dh_t *dev, spi_t spi, spi_clk_t clk,
37  gpio_t cs_pin, uint8_t scale)
38 {
39  uint8_t test;
40 
41  dev->spi = spi;
42  dev->clk = clk;
43  dev->cs = cs_pin;
44  dev->scale = 0;
45 
46  /* initialize the chip select line */
47  if (spi_init_cs(dev->spi, dev->cs) != SPI_OK) {
48  return -1;
49  }
50 
51  /* test connection to the device */
52  spi_acquire(dev->spi, dev->cs, SPI_MODE, dev->clk);
53  test = spi_transfer_reg(dev->spi, dev->cs, LIS3DH_REG_WHO_AM_I, 0);
54  spi_release(dev->spi);
55  if (test != LIS3DH_WHO_AM_I_RESPONSE) {
56  /* chip is not responding correctly */
57  return -1;
58  }
59 
60  /* Clear all settings */
61  lis3dh_write_reg(dev, LIS3DH_REG_CTRL_REG1, LIS3DH_CTRL_REG1_XYZEN_MASK);
62  /* Disable HP filter */
63  lis3dh_write_reg(dev, LIS3DH_REG_CTRL_REG2, 0);
64  /* Disable INT1 interrupt sources */
65  lis3dh_write_reg(dev, LIS3DH_REG_CTRL_REG3, 0);
66  /* Set block data update and little endian, set Normal mode (LP=0, HR=1) */
67  lis3dh_write_reg(dev, LIS3DH_REG_CTRL_REG4,
68  (LIS3DH_CTRL_REG4_BDU_ENABLE |
69  LIS3DH_CTRL_REG4_BLE_LITTLE_ENDIAN |
70  LIS3DH_CTRL_REG4_HR_MASK));
71  /* Disable FIFO */
72  lis3dh_write_reg(dev, LIS3DH_REG_CTRL_REG5, 0);
73  /* Reset INT2 settings */
74  lis3dh_write_reg(dev, LIS3DH_REG_CTRL_REG6, 0);
75 
76  /* Configure scale */
77  lis3dh_set_scale(dev, scale);
78 
79  return 0;
80 }

82 int lis3dh_read_xyz(const lis3dh_t *dev, lis3dh_data_t *acc_data)
83 {
84  uint8_t i;
85  /* Set READ MULTIPLE mode */
86  static const uint8_t addr = (LIS3DH_REG_OUT_X_L | LIS3DH_SPI_READ_MASK |
87  LIS3DH_SPI_MULTI_MASK);
88 
89  /* Acquire exclusive access to the bus. */
90  spi_acquire(dev->spi, dev->cs, SPI_MODE, dev->clk);
91  /* Perform the transaction */
92  spi_transfer_regs(dev->spi, dev->cs, addr,
93  NULL, acc_data, sizeof(lis3dh_data_t));
94  /* Release the bus for other threads. */
95  spi_release(dev->spi);
96 
97  /* Scale to milli-G */
98  for (i = 0; i < 3; ++i) {
99  int32_t tmp = (int32_t)(((int16_t *)acc_data)[i]);
100  tmp *= dev->scale;
101  tmp /= 32768;
102  (((int16_t *)acc_data)[i]) = (int16_t)tmp;
103  }
104 
105  return 0;
106 }

141 int lis3dh_set_fifo(lis3dh_t *dev, const uint8_t mode, const uint8_t watermark)
142 {
143  int status;
144  uint8_t reg;
145  reg = (watermark << LIS3DH_FIFO_CTRL_REG_FTH_SHIFT)
146  & LIS3DH_FIFO_CTRL_REG_FTH_MASK;
147  reg |= mode;
148  status = lis3dh_write_reg(dev, LIS3DH_REG_FIFO_CTRL_REG, reg);
149  if (status < 0) {
150  /* communication error */
151  return status;
152  }
153  if (mode != 0x00) {
154  status = lis3dh_write_bits(dev, LIS3DH_REG_CTRL_REG5,
155  LIS3DH_CTRL_REG5_FIFO_EN_MASK, LIS3DH_CTRL_REG5_FIFO_EN_MASK);
156  } else {
157  status = lis3dh_write_bits(dev, LIS3DH_REG_CTRL_REG5,
158  LIS3DH_CTRL_REG5_FIFO_EN_MASK, 0);
159  }
160  return status;
161 }

169 int lis3dh_set_scale(lis3dh_t *dev, const uint8_t scale)
170 {
171  uint8_t scale_reg;
172  /* Sensor full range is -32768 -- +32767 (measurements are left adjusted) */
173  /* => Scale factor is scale/32768 */
174  switch (scale)
175  {
176  case 2:
177  dev->scale = 2000;
178  scale_reg = LIS3DH_CTRL_REG4_SCALE_2G;
179  break;
180  case 4:
181  dev->scale = 4000;
182  scale_reg = LIS3DH_CTRL_REG4_SCALE_4G;
183  break;
184  case 8:
185  dev->scale = 8000;
186  scale_reg = LIS3DH_CTRL_REG4_SCALE_8G;
187  break;
188  case 16:
189  dev->scale = 16000;
190  scale_reg = LIS3DH_CTRL_REG4_SCALE_16G;
191  break;
192  default:
193  return -1;
194  }
195  return lis3dh_write_bits(dev, LIS3DH_REG_CTRL_REG4,
196  LIS3DH_CTRL_REG4_FS_MASK, scale_reg);
197 }

204 int lis3dh_get_fifo_level(lis3dh_t *dev)

282 static inline int lis3dh_write_bits(const lis3dh_t *dev, const lis3dh_reg_t reg,
283  const uint8_t mask, const uint8_t values)
284 {
285  uint8_t tmp;
286 
287  if (lis3dh_read_regs(dev, reg, 1, &tmp) < 0) {
288  /* Communication error */
289  return -1;
290  }
291 
292  tmp &= ~mask;
293  tmp |= (values & mask);
294 
295  if (lis3dh_write_reg(dev, reg, tmp) < 0) {
296  /* Communication error */
297  return -1;
298  }
299 
300  return 0;
301 }
