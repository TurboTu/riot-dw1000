168 static int _write(pn532_t *dev, char *buff, unsigned len)
169 {
170  int ret = -1;
171 
172  (void)buff;
173  (void)len;
174 
175  if (dev->mode == PN532_I2C) {
176 #ifdef PN532_SUPPORT_I2C
177  i2c_acquire(dev->conf->i2c);
178  ret = i2c_write_bytes(dev->conf->i2c, PN532_I2C_ADDRESS, buff, len);
179  i2c_release(dev->conf->i2c);
180 #endif
181  }
182  else {
183 #ifdef PN532_SUPPORT_SPI
184  spi_acquire(dev->conf->spi, SPI_CS_UNDEF, SPI_MODE, SPI_CLK);
185  gpio_clear(dev->conf->nss);
186  xtimer_usleep(SPI_WRITE_DELAY_US);
187  reverse(buff, len);
188  spi_transfer_byte(dev->conf->spi, SPI_CS_UNDEF, true, SPI_DATA_WRITE);
189  spi_transfer_bytes(dev->conf->spi, SPI_CS_UNDEF, true, buff, NULL, len);
190  gpio_set(dev->conf->nss);
191  spi_release(dev->conf->spi);
192  ret = (int)len;
193 #endif
194  }
195  DEBUG("pn532: -> ");
196  PRINTBUFF(buff, len);
197  return ret;

206

234 static int send_cmd(pn532_t *dev, char *buff, unsigned len)
235 {
236  unsigned pos, checksum;
237 
238  buff[0] = 0x00;
239  buff[1] = 0x00;
240  buff[2] = 0xFF;
241 
242  len += 1;
243  if (len < 0xff) {
244  buff[3] = (char)len;
245  buff[4] = 0x00 - buff[3];
246  pos = 5;
247 
248  }
249  else {
250  buff[3] = 0xff;
251  buff[4] = 0xff;
252  buff[5] = (len >> 8) & 0xff;
253  buff[6] = (len) & 0xff;
254  buff[7] = 0x00 - buff[5] - buff[6];
255  pos = 8;
256  }
257 
258  buff[pos] = HOST_TO_PN532;
259  checksum = chksum(&buff[pos], len);
260 
261  len += pos;
262  buff[len++] = checksum;
263  buff[len++] = 0x00;
264 
265  return _write(dev, buff, len);
266 }

273 /* Returns >0 payload len (or <0 received len but not as expected) */
274 static int read_command(pn532_t *dev, char *buff, unsigned len, int expected_cmd)
275 {
276  int r;
277  unsigned j, fi, lp, lc;
278 
279  /* apply framing overhead */
280  len += 8;
281  if (len >= 0xff) {
282  len += 3;
283  }
284 
285  r = _read(dev, buff, len);
286 
287  /* Validate frame structure and CRCs
288  *
289  * Note that all offsets are shifted by one since the first byte is always
290  * 0x01. */
291  if ((r < len) || (buff[1] != 0x00) || (buff[2] != 0x00) || (buff[3] != 0xFF)) {
292  return -r;
293  }
294 
295  if (buff[4] == 0xff && buff[5] == 0xff) {
296  /* extended frame */
297  lp = buff[6] << 8 | buff[7];
298  lc = (buff[6] + buff[7] + buff[8]) & 0xff;
299  fi = 9;
300 
301  }
302  else {
303  /* normal frame */
304  lp = buff[4];
305  lc = (buff[4] + buff[5]) & 0xff;
306  fi = 6;
307  }
308 
309  if (lc != 0 || lp >= 265 || buff[fi] != PN532_TO_HOST) {
310  return -r;
311  }
312 
313  if (lp == 0) {
314  return -r;
315  }
316 
317  if (chksum(&buff[fi], lp) != buff[fi + lp]) {
318  return -r;
319  }
320 
321  if (buff[fi + 1] != expected_cmd) {
322  return -r;
323  }
324 
325  /* Move the meaningful data to the beginning of the buffer */
326  /* start copying after command byte */
327  for (j = 0, fi += 2, lp -= 2; j < lp; fi++, j++) {
328  buff[j] = buff[fi];
329  }
330 
331  DEBUG("pn532: in cmd ");
332  PRINTBUFF(buff, lp);
333 
334  return lp;
335 }

338 static int send_check_ack(pn532_t *dev, char *buff, unsigned len)
339 {
340  if (send_cmd(dev, buff, len) > 0) {
341  static char ack[] = { 0x00, 0x00, 0xff, 0x00, 0xff, 0x00 };
342 
343  wait_ready(dev);
344  if (_read(dev, buff, sizeof(ack)) != sizeof(ack) + 1) {
345  return -2;
346  }
347 
348  if (memcmp(&buff[1], ack, sizeof(ack)) != 0) {
349  return -3;
350  }
351 
352  wait_ready(dev);
353  return 0;
354  }
355 
356  return -1;
357 }

463 int pn532_get_passive_iso14443a(pn532_t *dev, nfc_iso14443a_t *out,
464  unsigned max_retries)
465 {
466  int ret = -1;
467  char buff[PN532_BUFFER_LEN];
468 
469  if (_set_act_retries(dev, buff, max_retries) == 0) {
470  ret = _list_passive_targets(dev, buff, PN532_BR_106_ISO_14443_A, 1,
471  LIST_PASSIVE_LEN_14443(1));
472  }
473 
474  if (ret > 0 && buff[0] > 0) {
475  out->target = buff[1];
476  out->sns_res = (buff[2] << 8) | buff[3];
477  out->sel_res = buff[4];
478  out->id_len = buff[5];
479  out->type = ISO14443A_UNKNOWN;
480 
481  for (int i = 0; i < out->id_len; i++) {
482  out->id[i] = buff[6 + i];
483  }
484 
485  /* try to find out the type */
486  if (out->id_len == 4) {
487  out->type = ISO14443A_MIFARE;
488  }
489  else if (out->id_len == 7) {
490  /* In the case of type 4, the first byte of RATS is the length
491  * of RATS including the length itself (6+7) */
492  if (buff[13] == ret - 13) {
493  out->type = ISO14443A_TYPE4;
494  }
495  }
496  ret = 0;
497  }
498  else {
499  ret = -1;
500  }
501 
502  return ret;

526  pn532_mifare_key_t keyid, char *key, unsigned block)
527 {
528  int ret = -1;
529  char buff[PN532_BUFFER_LEN];
530 
531  buff[BUFF_CMD_START ] = CMD_DATA_EXCHANGE;
532  buff[BUFF_DATA_START ] = card->target;
533  buff[BUFF_DATA_START + 1] = keyid;
534  buff[BUFF_DATA_START + 2] = block; /* current block */
535 
536  /*
537  * The card ID directly follows the key in the buffer
538  * The key consists of 6 bytes and starts at offset 3
539  */
540  for (int i = 0; i < 6; i++) {
541  buff[BUFF_DATA_START + 3 + i] = key[i];
542  }
543 
544  for (int i = 0; i < card->id_len; i++) {
545  buff[BUFF_DATA_START + 9 + i] = card->id[i];
546  }
547 
548  ret = send_rcv(dev, buff, 9 + card->id_len, 1);
549  if (ret == 1) {
550  ret = buff[0];
551  card->auth = 1;
552  }
553 
554  return ret;
555 }
