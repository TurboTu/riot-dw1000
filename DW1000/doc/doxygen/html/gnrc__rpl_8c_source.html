130 static void _receive(gnrc_pktsnip_t *icmpv6)
131 {
132  gnrc_pktsnip_t *ipv6, *netif;
133  ipv6_hdr_t *ipv6_hdr;
134  icmpv6_hdr_t *icmpv6_hdr;
135  kernel_pid_t iface = KERNEL_PID_UNDEF;
136 
137  assert(icmpv6 != NULL);
138 
139  ipv6 = gnrc_pktsnip_search_type(icmpv6, GNRC_NETTYPE_IPV6);
140  netif = gnrc_pktsnip_search_type(icmpv6, GNRC_NETTYPE_NETIF);
141 
142  assert(ipv6 != NULL);
143 
144  if (netif) {
145  iface = ((gnrc_netif_hdr_t *)netif->data)->if_pid;
146  }
147 
148  ipv6_hdr = (ipv6_hdr_t *)ipv6->data;
149 
150  icmpv6_hdr = (icmpv6_hdr_t *)icmpv6->data;
151  switch (icmpv6_hdr->code) {
152  case GNRC_RPL_ICMPV6_CODE_DIS:
153  DEBUG("RPL: DIS received\n");
154  gnrc_rpl_recv_DIS((gnrc_rpl_dis_t *)(icmpv6_hdr + 1), iface, &ipv6_hdr->src,
155  &ipv6_hdr->dst, byteorder_ntohs(ipv6_hdr->len));
156  break;
157  case GNRC_RPL_ICMPV6_CODE_DIO:
158  DEBUG("RPL: DIO received\n");
159  gnrc_rpl_recv_DIO((gnrc_rpl_dio_t *)(icmpv6_hdr + 1), iface, &ipv6_hdr->src,
160  &ipv6_hdr->dst, byteorder_ntohs(ipv6_hdr->len));
161  break;
162  case GNRC_RPL_ICMPV6_CODE_DAO:
163  DEBUG("RPL: DAO received\n");
164  gnrc_rpl_recv_DAO((gnrc_rpl_dao_t *)(icmpv6_hdr + 1), iface, &ipv6_hdr->src,
165  &ipv6_hdr->dst, byteorder_ntohs(ipv6_hdr->len));
166  break;
167  case GNRC_RPL_ICMPV6_CODE_DAO_ACK:
168  DEBUG("RPL: DAO-ACK received\n");
169  gnrc_rpl_recv_DAO_ACK((gnrc_rpl_dao_ack_t *)(icmpv6_hdr + 1), iface, &ipv6_hdr->src,
170  &ipv6_hdr->dst, byteorder_ntohs(ipv6_hdr->len));
171  break;
172 #ifdef MODULE_GNRC_RPL_P2P
173  case GNRC_RPL_P2P_ICMPV6_CODE_DRO:
174  DEBUG("RPL: P2P DRO received\n");
175  gnrc_pktsnip_t *icmpv6_snip = gnrc_pktbuf_add(NULL, NULL, icmpv6->size,
176  GNRC_NETTYPE_ICMPV6);
177  if (icmpv6_snip == NULL) {
178  DEBUG("RPL-P2P: cannot copy ICMPv6 packet\n");
179  break;
180  }
181 
182  memcpy(icmpv6_snip->data, icmpv6->data, icmpv6->size);
183 
184  gnrc_rpl_p2p_recv_DRO(icmpv6_snip, &ipv6_hdr->src);
185  break;
186  case GNRC_RPL_P2P_ICMPV6_CODE_DRO_ACK:
187  DEBUG("RPL: P2P DRO-ACK received\n");
188  break;
189 #endif
190  default:
191  DEBUG("RPL: Unknown ICMPV6 code received\n");
192  break;
193  }
194 
195  gnrc_pktbuf_release(icmpv6);
196 }

198 static void *_event_loop(void *args)
199 {
200  msg_t msg, reply;
201 
202  (void)args;
203  msg_init_queue(_msg_q, GNRC_RPL_MSG_QUEUE_SIZE);
204 
205  /* preinitialize ACK */
206  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
207 
208  trickle_t *trickle;
209  /* start event loop */
210  while (1) {
211  DEBUG("RPL: waiting for incoming message.\n");
212  msg_receive(&msg);
213 
214  switch (msg.type) {
215  case GNRC_RPL_MSG_TYPE_LIFETIME_UPDATE:
216  DEBUG("RPL: GNRC_RPL_MSG_TYPE_LIFETIME_UPDATE received\n");
217  _update_lifetime();
218  break;
219  case GNRC_RPL_MSG_TYPE_TRICKLE_INTERVAL:
220  DEBUG("RPL: GNRC_RPL_MSG_TYPE_TRICKLE_INTERVAL received\n");
221  trickle = msg.content.ptr;
222  if (trickle && (trickle->callback.func != NULL)) {
223  trickle_interval(trickle);
224  }
225  break;
226  case GNRC_RPL_MSG_TYPE_TRICKLE_CALLBACK:
227  DEBUG("RPL: GNRC_RPL_MSG_TYPE_TRICKLE_CALLBACK received\n");
228  trickle = msg.content.ptr;
229  if (trickle && (trickle->callback.func != NULL)) {
230  trickle_callback(trickle);
231  }
232  break;
233  case GNRC_NETAPI_MSG_TYPE_RCV:
234  DEBUG("RPL: GNRC_NETAPI_MSG_TYPE_RCV received\n");
235  _receive(msg.content.ptr);
236  break;
237  case GNRC_NETAPI_MSG_TYPE_SND:
238  break;
239  case GNRC_NETAPI_MSG_TYPE_GET:
240  case GNRC_NETAPI_MSG_TYPE_SET:
241  DEBUG("RPL: reply to unsupported get/set\n");
242  reply.content.value = -ENOTSUP;
243  msg_reply(&msg, &reply);
244  break;
245  default:
246  break;
247  }
248  }
249 
250  return NULL;
251 }

253 void _update_lifetime(void)
254 {
255  uint32_t now = xtimer_now_usec();
256  uint16_t now_sec = now / US_PER_SEC;
257 
258  gnrc_rpl_parent_t *parent;
259  gnrc_rpl_instance_t *inst;
260 
261  for (uint8_t i = 0; i < GNRC_RPL_PARENTS_NUMOF; ++i) {
262  parent = &gnrc_rpl_parents[i];
263  if (parent->state != 0) {
264  if ((int32_t)(parent->lifetime - now_sec) <= GNRC_RPL_LIFETIME_UPDATE_STEP) {
265  gnrc_rpl_dodag_t *dodag = parent->dodag;
266  gnrc_rpl_parent_remove(parent);
267  gnrc_rpl_parent_update(dodag, NULL);
268  continue;
269  }
270  else if ((int32_t)(parent->lifetime - now_sec) <= (GNRC_RPL_LIFETIME_UPDATE_STEP * 2)) {
271  gnrc_rpl_send_DIS(parent->dodag->instance, &parent->addr);
272  }
273  }
274  }
275 
276  for (int i = 0; i < GNRC_RPL_INSTANCES_NUMOF; ++i) {
277  inst = &gnrc_rpl_instances[i];
278  if (inst->state != 0) {
279  if ((inst->cleanup > 0) && (inst->dodag.parents == NULL) &&
280  (inst->dodag.my_rank == GNRC_RPL_INFINITE_RANK)) {
281  inst->cleanup -= GNRC_RPL_LIFETIME_UPDATE_STEP;
282  if (inst->cleanup <= 0) {
283  /* no parents - delete this instance and DODAG */
284  gnrc_rpl_instance_remove(inst);
285  continue;
286  }
287  }
288 
289  if (inst->dodag.dao_time > GNRC_RPL_LIFETIME_UPDATE_STEP) {
290  inst->dodag.dao_time -= GNRC_RPL_LIFETIME_UPDATE_STEP;
291  }
292  else {
293  _dao_handle_send(&inst->dodag);
294  }
295  }
296  }
297 
298 #ifdef MODULE_GNRC_RPL_P2P
299  gnrc_rpl_p2p_update();
300 #endif
301 
302  xtimer_set_msg(&_lt_timer, _lt_time, &_lt_msg, gnrc_rpl_pid);
303 }
