113 void msp430_init_dco(void)
114 {
115 #if MSP430_HAS_EXTERNAL_CRYSTAL
116  /*------------------ use external oszillator -----------------------*/
117  uint16_t i;
118 
119  /* Stop watchdog */
120  WDTCTL = WDTPW + WDTHOLD;
121 
122  //Init crystal for mclk
123  //XT2 = HF XTAL
124  BCSCTL1 = RSEL2;
125 
126  /* Wait for xtal to stabilize */
127  do {
128  IFG1 &= ~OFIFG; /* Clear oscillator fault flag */
129 
130  for (i = 0xFF; i > 0; i--); /* Time for flag to set */
131  }
132  while ((IFG1 & OFIFG) != 0); /* Oscillator fault flag still set? */
133 
134  BCSCTL2 = SELM_2 + SELS; /* MCLK und SMCLK = XT2 (safe) */
135 #else
136  int delta = __msp430_cpu_speed >> 12;
137  unsigned int oldcapture = 0;
138  unsigned int i;
139 
140  BCSCTL1 = 0xa4; /* ACLK is devided by 4. RSEL=6 no division for MCLK
141  and SSMCLK. XT2 is off. */
142 
143  /* Init FLL to desired frequency using the 32762Hz crystal */
144 #if MSP430_HAS_DCOR
145  BCSCTL2 = 0x01;
146 #else
147  BCSCTL2 = 0x00;
148 #endif
149 
150  WDTCTL = WDTPW + WDTHOLD; /* Stop WDT */
151  BCSCTL1 |= DIVA1 + DIVA0; /* ACLK = LFXT1CLK/8 */
152 
153  for (i = 0xffff; i > 0; i--); /* Delay for XTAL to settle */
154 
155  CCTL2 = CCIS0 + CM0 + CAP; /* Define CCR2, CAP, ACLK */
156  TACTL = TASSEL1 + TACLR + MC1; /* SMCLK, continous mode */
157 
158 
159  while (1) {
160  unsigned int compare;
161 
162  while ((CCTL2 & CCIFG) != CCIFG); /* Wait until capture occured! */
163 
164  CCTL2 &= ~CCIFG; /* Capture occured, clear flag */
165  compare = CCR2; /* Get current captured SMCLK */
166  compare = compare - oldcapture; /* SMCLK difference */
167  oldcapture = CCR2; /* Save current captured SMCLK */
168 
169  if (delta == compare) {
170  break; /* if equal, leave "while(1)" */
171  }
172  else if (delta < compare) { /* DCO is too fast, slow it down */
173  DCOCTL--;
174 
175  if (DCOCTL == 0xFF) { /* Did DCO role under? */
176  BCSCTL1--;
177  }
178  }
179  else { /* -> Select next lower RSEL */
180  DCOCTL++;
181 
182  if (DCOCTL == 0x00) { /* Did DCO role over? */
183  BCSCTL1++;
184  }
185 
186  /* -> Select next higher RSEL */
187  }
188  }
189 
190  CCTL2 = 0; /* Stop CCR2 function */
191  TACTL = 0; /* Stop Timer_A */
192 
193  BCSCTL1 &= ~(DIVA1 + DIVA0); /* remove /8 divisor from ACLK again */
194 #endif
195 }
