59 static int init_base(uart_t uart, uint32_t baudrate)
60 {
61  /* Calculate the BAUD value */
62  uint64_t temp1 = ((16 * ((uint64_t)baudrate)) << 32);
63  uint64_t ratio = _long_division(temp1 , UART_0_REF_F);
64  uint64_t scale = ((uint64_t)1 << 32) - ratio;
65  uint64_t baud_calculated = (65536 * scale) >> 32;
66 
67  switch (uart) {
68 #if UART_0_EN
69  case UART_0:
70  /* Enable the peripheral channel */
71  GCLK->PCHCTRL[SERCOM3_GCLK_ID_CORE].reg |= GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN_GCLK0;
72 
73  while (!(GCLK->PCHCTRL[SERCOM3_GCLK_ID_CORE].reg & GCLK_PCHCTRL_CHEN)) {
74  /* Wait for clock synchronization */
75  }
76 
77  MCLK->APBCMASK.reg |= MCLK_APBCMASK_SERCOM3;
78  /* configure PINS to input/output*/
79  UART_0_PORT.DIRSET.reg = (1 << UART_0_TX_PIN); /* tx's direction is output */
80  UART_0_PORT.PINCFG[UART_0_RX_PIN % 32].bit.INEN = true; /* buffer rx pin's value */
81 
82  /* enable PMUX for pins and set to config C. */
83  UART_0_PORT.WRCONFIG.reg = PORT_WRCONFIG_WRPINCFG \
84  | PORT_WRCONFIG_WRPMUX \
85  | PORT_WRCONFIG_PMUX(0x2) \
86  | PORT_WRCONFIG_PMUXEN \
87  | UART_0_PINS;
88 
89  UART_0_DEV.CTRLA.bit.ENABLE = 0; /* Disable to write, need to sync tho */
90  while(UART_0_DEV.SYNCBUSY.bit.ENABLE) {}
91 
92  /* set to LSB, asynchronous mode without parity, PAD0 Tx, PAD1 Rx,
93  * 16x over-sampling, internal clk */
94  UART_0_DEV.CTRLA.reg = SERCOM_USART_CTRLA_DORD \
95  | SERCOM_USART_CTRLA_FORM(0x0) \
96  | SERCOM_USART_CTRLA_SAMPA(0x0) \
97  | SERCOM_USART_CTRLA_TXPO(0x0) \
98  | SERCOM_USART_CTRLA_RXPO(0x1) \
99  | SERCOM_USART_CTRLA_SAMPR(0x0) \
100  | SERCOM_USART_CTRLA_MODE(0x1) \
101  | (UART_0_RUNSTDBY ? SERCOM_USART_CTRLA_RUNSTDBY : 0);
102 
103  /* Set baud rate */
104  UART_0_DEV.BAUD.bit.BAUD = baud_calculated;
105 
106  /* enable receiver and transmitter, one stop bit*/
107  UART_0_DEV.CTRLB.reg = (SERCOM_USART_CTRLB_RXEN | SERCOM_USART_CTRLB_TXEN);
108  while(UART_0_DEV.SYNCBUSY.bit.CTRLB) {}
109 
110  break;
111 #endif
112  default:
113  (void)baud_calculated;
114  return UART_NODEV;
115  }
116 
117  uart_poweron(uart);
118  return UART_OK;
119 }

122 {

166 static uint64_t _long_division(uint64_t n, uint64_t d)
167 {
168  int32_t i;
169  uint64_t q = 0, r = 0, bit_shift;
170  for (i = 63; i >= 0; i--) {
171  bit_shift = (uint64_t)1 << i;
172 
173  r = r << 1;
174 
175  if (n & bit_shift) {
176  r |= 0x01;
177  }
178 
179  if (r >= d) {
180  r = r - d;
181  q |= bit_shift;
182  }
183  }
184 
185  return q;
186 }
