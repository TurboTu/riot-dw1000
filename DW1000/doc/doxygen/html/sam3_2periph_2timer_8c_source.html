77 int timer_init(tim_t tim, unsigned long freq, timer_cb_t cb, void *arg)
78 {
79  /* check if device is valid */
80  if (tim >= TIMER_NUMOF) {
81  return -1;
82  }
83 
84  /* enable the device clock */
85  clk_en(tim);
86 
87  /* save callback */
88  isr_ctx[tim].cb = cb;
89  isr_ctx[tim].arg = arg;
90 
91  /* configure the timer block by connecting TIOA1 to XC0 */
92  dev(tim)->TC_BMR = TC_BMR_TC0XC0S_TIOA1;
93  /* configure and enable channel 0 to use XC0 as input */
94  dev(tim)->TC_CHANNEL[0].TC_CMR = (TC_CMR_TCCLKS_XC0 |
95  TC_CMR_WAVE | TC_CMR_EEVT_XC0);
96  dev(tim)->TC_CHANNEL[0].TC_CCR = (TC_CCR_CLKEN | TC_CCR_SWTRG);
97 
98  /* configure channel 1:
99  * - select wave mode
100  * - set input clock to TIMER_CLOCK1 (MCK/2)
101  * - reload on TC_CV == TC_RC
102  * - let TIOA2 signal be toggled when TC_CV == TC_RC
103  */
104  dev(tim)->TC_CHANNEL[1].TC_CMR = (TC_CMR_TCCLKS_TIMER_CLOCK1 | TC_CMR_WAVE |
105  TC_CMR_WAVSEL_UP_RC | TC_CMR_ACPC_TOGGLE);
106 
107  /* configure the frequency of channel 1 to freq * 4
108  *
109  * note: as channel 0 is only incremented on rising edges of TIOA1 line and
110  * channel 1 toggles this line on each timer tick, the actual frequency
111  * driving channel 0 is f_ch2 / 2 --> f_ch0/1 = (MCK / 2) / 2 / freq.
112  */
113  dev(tim)->TC_CHANNEL[1].TC_R[2] = (CLOCK_CORECLOCK / 4) / freq;
114 
115  /* start channel 1 */
116  dev(tim)->TC_CHANNEL[1].TC_CCR = (TC_CCR_CLKEN | TC_CCR_SWTRG);
117 
118  /* enable global interrupts for given timer */
119  NVIC_EnableIRQ(timer_config[tim].id_ch0);
120 
121  return 0;
122 }
