49 /* sets an entry to stale if its l2addr differs from the given one or creates it stale if it
50  * does not exist */
51 static void _stale_nc(kernel_pid_t iface, ipv6_addr_t *ipaddr, uint8_t *l2addr,
52  int l2addr_len)
53 {
54  if (l2addr_len != -ENOTSUP) {
55  gnrc_ipv6_nc_t *nc_entry = gnrc_ipv6_nc_get(iface, ipaddr);
56  if (nc_entry == NULL) {
57  gnrc_ipv6_nc_add(iface, ipaddr, l2addr, (uint16_t)l2addr_len,
58  GNRC_IPV6_NC_STATE_STALE);
59  }
60 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
61  /* unreachable set in gnrc_ndp_retrans_nbr_sol() will just be staled */
62  else if (gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_UNREACHABLE) {
63  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_STALE);
64  }
65 #endif
66  else if (((uint16_t)l2addr_len != nc_entry->l2_addr_len) ||
67  (memcmp(l2addr, nc_entry->l2_addr, l2addr_len) != 0)) {
68  /* if entry exists but l2 address differs: set */
69  nc_entry->l2_addr_len = (uint16_t)l2addr_len;
70  memcpy(nc_entry->l2_addr, l2addr, l2addr_len);
71  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_STALE);
72  }
73  }

76 void gnrc_ndp_nbr_sol_handle(kernel_pid_t iface, gnrc_pktsnip_t *pkt,
77  ipv6_hdr_t *ipv6, ndp_nbr_sol_t *nbr_sol,
78  size_t icmpv6_size)
79 {
80  uint16_t opt_offset = 0;
81  uint8_t l2src[GNRC_IPV6_NC_L2_ADDR_MAX];
82  uint8_t *buf = ((uint8_t *)nbr_sol) + sizeof(ndp_nbr_sol_t);
83  ipv6_addr_t *tgt, nbr_adv_dst;
84  gnrc_pktsnip_t *nbr_adv_opts = NULL;
85 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
86  ndp_opt_t *sl2a_opt = NULL;
87  sixlowpan_nd_opt_ar_t *ar_opt = NULL;
88 #endif
89  int sicmpv6_size = (int)icmpv6_size, l2src_len = 0;
90  DEBUG("ndp: received neighbor solicitation (src: %s, ",
91  ipv6_addr_to_str(addr_str, &ipv6->src, sizeof(addr_str)));
92  DEBUG("dst: %s, ",
93  ipv6_addr_to_str(addr_str, &ipv6->dst, sizeof(addr_str)));
94  DEBUG("tgt: %s)\n",
95  ipv6_addr_to_str(addr_str, &nbr_sol->tgt, sizeof(addr_str)));
96  /* check validity */
97  if ((ipv6->hl != 255) || (nbr_sol->code != 0) ||
98  (icmpv6_size < sizeof(ndp_nbr_sol_t)) ||
99  ipv6_addr_is_multicast(&nbr_sol->tgt) ||
100  (ipv6_addr_is_unspecified(&ipv6->src) &&
101  ipv6_addr_is_solicited_node(&ipv6->dst))) {
102  DEBUG("ndp: neighbor solicitation was invalid.\n");
103  /* icmpv6 releases */
104  return;
105  }
106  if ((tgt = gnrc_ipv6_netif_find_addr(iface, &nbr_sol->tgt)) == NULL) {
107  DEBUG("ndp: Target address is not to interface %" PRIkernel_pid "\n",
108  iface);
109  /* icmpv6 releases */
110  return;
111  }
112  sicmpv6_size -= sizeof(ndp_nbr_sol_t);
113  while (sicmpv6_size > 0) {
114  ndp_opt_t *opt = (ndp_opt_t *)(buf + opt_offset);
115  switch (opt->type) {
116  case NDP_OPT_SL2A:
117  if ((l2src_len = gnrc_ndp_internal_sl2a_opt_handle(pkt, ipv6, nbr_sol->type, opt,
118  l2src)) < 0) {
119  /* -ENOTSUP can not happen, since the function only returns this for invalid
120  * message types containing the SL2A. Neighbor solicitations are not an
121  * invalid message type for SL2A. According to that, we don't need to watch
122  * out for that here, but regardless, the source link-layer address option
123  * is invalid. */
124  return;
125  }
126 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
127  sl2a_opt = opt;
128  break;
129  case NDP_OPT_AR:
130  /* actually handling at the end of the function (see below) */
131  ar_opt = (sixlowpan_nd_opt_ar_t *)opt;
132 #endif
133  break;
134  default:
135  /* silently discard all other options */
136  break;
137  }
138  opt_offset += (opt->len * 8);
139  sicmpv6_size -= (opt->len * 8);
140 
141 #if ENABLE_DEBUG
142  if (sicmpv6_size < 0) {
143  DEBUG("ndp: Option parsing out of sync.\n");
144  }
145 #endif
146  }
147 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
148  gnrc_ipv6_netif_t *ipv6_iface = gnrc_ipv6_netif_get(iface);
149  assert(ipv6_iface != NULL);
150  if ((sl2a_opt != NULL) && (ar_opt != NULL) &&
151  (ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) &&
152  (ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER)) {
153  uint8_t status = gnrc_sixlowpan_nd_opt_ar_handle(iface, ipv6,
154  nbr_sol->type,
155  &ipv6->src, ar_opt,
156  l2src, l2src_len);
157  /* check for multihop DAD return */
158  nbr_adv_opts = gnrc_sixlowpan_nd_opt_ar_build(status, GNRC_SIXLOWPAN_ND_AR_LTIME,
159  &ar_opt->eui64, NULL);
160  if (status == 0) {
161  memcpy(&nbr_adv_dst, &ipv6->src, sizeof(ipv6_addr_t));
162  }
163  else {
164  /* see https://tools.ietf.org/html/rfc6775#section-6.5.2 */
165  eui64_t iid;
166  ieee802154_get_iid(&iid, ar_opt->eui64.uint8, sizeof(eui64_t));
167  ipv6_addr_set_aiid(&nbr_adv_dst, iid.uint8);
168  ipv6_addr_set_link_local_prefix(&nbr_adv_dst);
169  }
170  }
171  else { /* gnrc_sixlowpan_nd_opt_ar_handle updates neighbor cache */
172  _stale_nc(iface, &ipv6->src, l2src, l2src_len);
173  memcpy(&nbr_adv_dst, &ipv6->src, sizeof(ipv6_addr_t));
174  }
175 #else
176  _stale_nc(iface, &ipv6->src, l2src, l2src_len);
177  memcpy(&nbr_adv_dst, &ipv6->src, sizeof(ipv6_addr_t));
178 #endif
179  gnrc_ndp_internal_send_nbr_adv(iface, tgt, &nbr_adv_dst, ipv6_addr_is_multicast(&ipv6->dst),
180  nbr_adv_opts);
181 }

189 void gnrc_ndp_nbr_adv_handle(kernel_pid_t iface, gnrc_pktsnip_t *pkt,
190  ipv6_hdr_t *ipv6, ndp_nbr_adv_t *nbr_adv,
191  size_t icmpv6_size)
192 {
193  uint16_t opt_offset = 0;
194  uint8_t *buf = ((uint8_t *)nbr_adv) + sizeof(ndp_nbr_adv_t);
195  int l2tgt_len = 0;
196  uint8_t l2tgt[GNRC_IPV6_NC_L2_ADDR_MAX];
197  int sicmpv6_size = (int)icmpv6_size;
198  gnrc_ipv6_nc_t *nc_entry = gnrc_ipv6_nc_get(iface, &nbr_adv->tgt);
199  gnrc_pktsnip_t *netif;
200  gnrc_netif_hdr_t *netif_hdr = NULL;
201 
202  DEBUG("ndp: received neighbor advertisement (src: %s, ",
203  ipv6_addr_to_str(addr_str, &ipv6->src, sizeof(addr_str)));
204  DEBUG("dst: %s, ",
205  ipv6_addr_to_str(addr_str, &ipv6->dst, sizeof(addr_str)));
206  DEBUG("tgt: %s)\n",
207  ipv6_addr_to_str(addr_str, &nbr_adv->tgt, sizeof(addr_str)));
208 
209  /* check validity */
210  if ((ipv6->hl != 255) || (nbr_adv->code != 0) ||
211  (icmpv6_size < sizeof(ndp_nbr_adv_t)) ||
212  ipv6_addr_is_multicast(&nbr_adv->tgt)) {
213  DEBUG("ndp: neighbor advertisement was invalid.\n");
214  /* icmpv6 releases */
215  return;
216  }
217 
218  if (nc_entry == NULL) {
219  /* see https://tools.ietf.org/html/rfc4861#section-7.2.5 */
220  DEBUG("ndp: no neighbor cache entry found for advertisement's target\n");
221  /* icmpv6 releases */
222  return;
223  }
224 
225  sicmpv6_size -= sizeof(ndp_nbr_adv_t);
226 
227  while (sicmpv6_size > 0) {
228  ndp_opt_t *opt = (ndp_opt_t *)(buf + opt_offset);
229 
230  switch (opt->type) {
231  case NDP_OPT_TL2A:
232  if ((l2tgt_len = gnrc_ndp_internal_tl2a_opt_handle(pkt, ipv6, nbr_adv->type, opt, l2tgt)) < 0) {
233  /* invalid target link-layer address option */
234  return;
235  }
236  break;
237 #ifdef MODULE_GNRC_SIXLOWPAN_ND
238  case NDP_OPT_AR:
239  /* address registration option is always ignored when invalid */
240  gnrc_sixlowpan_nd_opt_ar_handle(iface, ipv6, nbr_adv->type,
241  &nbr_adv->tgt,
242  (sixlowpan_nd_opt_ar_t *)opt,
243  NULL, 0);
244  break;
245 #endif
246  default:
247  /* silently discard all other options */
248  break;
249  }
250 
251  opt_offset += (opt->len * 8);
252  sicmpv6_size -= (opt->len * 8);
253 
254 #if ENABLE_DEBUG
255  if (sicmpv6_size < 0) {
256  DEBUG("ndp: Option parsing out of sync.\n");
257  }
258 #endif
259  }
260 
261  netif = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_NETIF);
262 
263  if (netif != NULL) {
264  netif_hdr = netif->data;
265  }
266 
267  if (l2tgt_len != -ENOTSUP) {
268 #ifdef MODULE_GNRC_SIXLOWPAN_ND
269  /* check if entry wasn't removed by ARO, ideally there should not be any TL2A in here */
270  nc_entry = gnrc_ipv6_nc_get(iface, &nbr_adv->tgt);
271  if (nc_entry == NULL) {
272  return;
273  }
274 #endif
275  if (gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_INCOMPLETE) {
276  if (_pkt_has_l2addr(netif_hdr) && (l2tgt_len == 0)) {
277  /* link-layer has addresses, but no TLLAO supplied: discard silently
278  * (see https://tools.ietf.org/html/rfc4861#section-7.2.5) */
279  return;
280  }
281 
282  nc_entry->iface = iface;
283  nc_entry->l2_addr_len = l2tgt_len;
284  memcpy(nc_entry->l2_addr, l2tgt, l2tgt_len);
285 
286  if (nbr_adv->flags & NDP_NBR_ADV_FLAGS_S) {
287  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_REACHABLE);
288  }
289  else {
290  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_STALE);
291  }
292 
293  if (nbr_adv->flags & NDP_NBR_ADV_FLAGS_R) {
294  nc_entry->flags |= GNRC_IPV6_NC_IS_ROUTER;
295  }
296  else {
297  nc_entry->flags &= ~GNRC_IPV6_NC_IS_ROUTER;
298  /* TODO: update state of neighbor as router in FIB? */
299  }
300 #ifdef MODULE_GNRC_NDP_NODE
301  gnrc_pktqueue_t *queued_pkt;
302  while ((queued_pkt = gnrc_pktqueue_remove_head(&nc_entry->pkts)) != NULL) {
303  if (gnrc_netapi_send(gnrc_ipv6_pid, queued_pkt->pkt) < 1) {
304  DEBUG("ndp: unable to send queued packet\n");
305  gnrc_pktbuf_release(queued_pkt->pkt);
306  }
307  queued_pkt->pkt = NULL;
308  }
309 #endif
310  }
311  else {
312  /* first or-term: no link-layer, but nc_entry has l2addr,
313  * second or-term: different l2addr cached */
314  bool l2tgt_changed = false;
315 
316  if ((!_pkt_has_l2addr(netif_hdr)) && (l2tgt_len == 0)) {
317  /* there was previously a L2 address registered */
318  l2tgt_changed = (nc_entry->l2_addr_len != 0);
319  }
320  /* link-layer has addresses and TLLAO with different address */
321  else if (_pkt_has_l2addr(netif_hdr) && (l2tgt_len != 0)) {
322  l2tgt_changed = (!(l2tgt_len == nc_entry->l2_addr_len)) &&
323  (memcmp(nc_entry->l2_addr, l2tgt, l2tgt_len) == 0);
324  }
325 
326  if ((nbr_adv->flags & NDP_NBR_ADV_FLAGS_O) || !l2tgt_changed ||
327  (l2tgt_len == 0)) {
328  if (l2tgt_len != 0) {
329  nc_entry->iface = iface;
330  nc_entry->l2_addr_len = l2tgt_len;
331  memcpy(nc_entry->l2_addr, l2tgt, l2tgt_len);
332  }
333 
334  if (nbr_adv->flags & NDP_NBR_ADV_FLAGS_S) {
335  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_REACHABLE);
336  }
337  else if (l2tgt_changed) {
338  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_STALE);
339  }
340 
341  if (nbr_adv->flags & NDP_NBR_ADV_FLAGS_R) {
342  nc_entry->flags |= GNRC_IPV6_NC_IS_ROUTER;
343  }
344  else {
345  nc_entry->flags &= ~GNRC_IPV6_NC_IS_ROUTER;
346  /* TODO: update state of neighbor as router in FIB? */
347  }
348  }
349  else if (l2tgt_changed &&
350  gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_REACHABLE) {
351  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_STALE);
352  }
353  }
354  }
355 
356  return;
357 }

464 static inline void _set_reach_time(gnrc_ipv6_netif_t *if_entry, uint32_t mean)

498  if (nc_entry == NULL) { /* not in default router list */
499  /* create default router list entry */
500  nc_entry = gnrc_ipv6_nc_add(iface, &ipv6->src, NULL, 0,
501  GNRC_IPV6_NC_STATE_STALE | GNRC_IPV6_NC_IS_ROUTER);
502  if (nc_entry == NULL) {
503  DEBUG("ndp: error on default router list entry creation\n");
504  return;
505  }
506  }
507  else if ((nc_entry->flags & GNRC_IPV6_NC_IS_ROUTER) && (byteorder_ntohs(rtr_adv->ltime) == 0)) {
508  nc_entry->flags &= ~GNRC_IPV6_NC_IS_ROUTER;
509  }
510  else {
511  nc_entry->flags |= GNRC_IPV6_NC_IS_ROUTER;
512  }
513  /* set router life timer */
514  if (rtr_adv->ltime.u16 != 0) {
515  uint16_t ltime = byteorder_ntohs(rtr_adv->ltime);
516 #ifdef MODULE_GNRC_SIXLOWPAN_ND
517  next_rtr_sol = ltime;
518 #endif
519  xtimer_set_msg(&nc_entry->rtr_timeout, (ltime * US_PER_SEC),
520  &nc_entry->rtr_timeout_msg, thread_getpid());
521  }
522  /* set current hop limit from message if available */
523  if (rtr_adv->cur_hl != 0) {
524  if_entry->cur_hl = rtr_adv->cur_hl;
525  }
526  /* set flags from message */
527  if_entry->flags &= ~GNRC_IPV6_NETIF_FLAGS_RTR_ADV_MASK;
528  if_entry->flags |= (rtr_adv->flags << GNRC_IPV6_NETIF_FLAGS_RTR_ADV_POS) &
529  GNRC_IPV6_NETIF_FLAGS_RTR_ADV_MASK;
530  /* set reachable time from message if it is not the same as the random base
531  * value */
532  if ((rtr_adv->reach_time.u32 != 0) &&
533  (if_entry->reach_time_base != byteorder_ntohl(rtr_adv->reach_time))) {
534  _set_reach_time(if_entry, byteorder_ntohl(rtr_adv->reach_time));
535  }
536  /* set retransmission timer from message */
537  if (rtr_adv->retrans_timer.u32 != 0) {
538  if_entry->retrans_timer = byteorder_ntohl(rtr_adv->retrans_timer);
539  }
540  mutex_unlock(&if_entry->mutex);
541  sicmpv6_size -= sizeof(ndp_rtr_adv_t);
542  /* parse options */
543  while (sicmpv6_size > 0) {
544  ndp_opt_t *opt = (ndp_opt_t *)(buf + opt_offset);
545  switch (opt->type) {
546  case NDP_OPT_SL2A:
547  if ((l2src_len = gnrc_ndp_internal_sl2a_opt_handle(pkt, ipv6, rtr_adv->type, opt,
548  l2src)) < 0) {
549  /* -ENOTSUP can not happen */
550  /* invalid source link-layer address option */
551  return;
552  }
553  break;
554  case NDP_OPT_MTU:
555  if (!gnrc_ndp_internal_mtu_opt_handle(iface, rtr_adv->type, (ndp_opt_mtu_t *)opt)) {
556  /* invalid MTU option */
557  return;
558  }
559  break;
560  case NDP_OPT_PI:
561  if (!gnrc_ndp_internal_pi_opt_handle(iface, rtr_adv->type, (ndp_opt_pi_t *)opt)) {
562  /* invalid prefix information option */
563  return;
564  }
565 #ifdef MODULE_GNRC_SIXLOWPAN_ND
566  uint32_t valid_ltime = byteorder_ntohl(((ndp_opt_pi_t *)opt)->valid_ltime);
567  if ((valid_ltime != 0) && (valid_ltime < next_rtr_sol)) {
568  next_rtr_sol = valid_ltime;
569  }
570 #endif
571  break;
572 #ifdef MODULE_GNRC_SIXLOWPAN_ND
573  case NDP_OPT_6CTX:
574  if (!gnrc_sixlowpan_nd_opt_6ctx_handle(rtr_adv->type,
575  (sixlowpan_nd_opt_6ctx_t *)opt)) {
576  /* invalid 6LoWPAN context option */
577  return;
578  }
579  uint16_t ltime = byteorder_ntohs(((sixlowpan_nd_opt_6ctx_t *)opt)->ltime);
580  if ((ltime != 0) && (ltime < (next_rtr_sol / 60))) {
581  next_rtr_sol = ltime * 60;
582  }
583 
584  break;
585 #endif
586 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
587  case NDP_OPT_ABR:
588  gnrc_sixlowpan_nd_opt_abr_handle(iface, rtr_adv, icmpv6_size,
589  (sixlowpan_nd_opt_abr_t *)opt);
590  break;
591 #endif
592  }
593 
594  opt_offset += (opt->len * 8);
595  sicmpv6_size -= (opt->len * 8);
596 
597 #if ENABLE_DEBUG
598  if (sicmpv6_size < 0) {
599  DEBUG("ndp: Option parsing out of sync.\n");
600  }
601 #endif
602  }
603 #if ENABLE_DEBUG && defined(MODULE_GNRC_SIXLOWPAN_ND)
604  if ((if_entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) && (l2src_len <= 0)) {
605  DEBUG("ndp: Router advertisement did not contain any source address information\n");
606  }
607 #endif
608  _stale_nc(iface, &ipv6->src, l2src, l2src_len);
609  /* stop multicast router solicitation retransmission timer */
610  xtimer_remove(&if_entry->rtr_sol_timer);
611 #ifdef MODULE_GNRC_SIXLOWPAN_ND
612  if (if_entry->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) {
613  /* 3/4 of the time should be "well before" enough the respective timeout
614  * not to run out; see https://tools.ietf.org/html/rfc6775#section-5.4.3 */
615  next_rtr_sol *= 3;
616  next_rtr_sol = (next_rtr_sol > 4) ? (next_rtr_sol >> 2) : 1;
617  /* according to https://tools.ietf.org/html/rfc6775#section-5.3:
618  * "In all cases, the RS retransmissions are terminated when an RA is
619  * received."
620  * Hence, reset router solicitation counter and reset timer. */
621  if_entry->rtr_sol_count = 0;
622  gnrc_sixlowpan_nd_rtr_sol_reschedule(nc_entry, next_rtr_sol);
623  gnrc_ndp_internal_send_nbr_sol(nc_entry->iface, NULL, &nc_entry->ipv6_addr,
624  &nc_entry->ipv6_addr);
625  if (if_entry->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER) {
626  gnrc_ipv6_netif_set_rtr_adv(if_entry, true);
627  }
628  }
629 #endif

632 void gnrc_ndp_retrans_nbr_sol(gnrc_ipv6_nc_t *nc_entry)
633 {
634  if ((gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_INCOMPLETE) ||
635  (gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_PROBE)) {
636  if (nc_entry->probes_remaining > 1) {
637  ipv6_addr_t dst;
638 
639  DEBUG("ndp: Retransmit neighbor solicitation for %s\n",
640  ipv6_addr_to_str(addr_str, &nc_entry->ipv6_addr, sizeof(addr_str)));
641 
642  /* retransmit neighbor solicatation */
643  if (gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_INCOMPLETE) {
644  ipv6_addr_set_solicited_nodes(&dst, &nc_entry->ipv6_addr);
645  }
646  else {
647  dst.u64[0] = nc_entry->ipv6_addr.u64[0];
648  dst.u64[1] = nc_entry->ipv6_addr.u64[1];
649  }
650 
651  nc_entry->probes_remaining--;
652 
653  if (nc_entry->iface == KERNEL_PID_UNDEF) {
654  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
655  size_t ifnum = gnrc_netif_get(ifs);
656 
657  for (size_t i = 0; i < ifnum; i++) {
658  gnrc_ndp_internal_send_nbr_sol(ifs[i], NULL, &nc_entry->ipv6_addr, &dst);
659  }
660 
661  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, GNRC_NDP_RETRANS_TIMER,
662  GNRC_NDP_MSG_NBR_SOL_RETRANS, gnrc_ipv6_pid);
663  }
664  else {
665  gnrc_ipv6_netif_t *ipv6_iface = gnrc_ipv6_netif_get(nc_entry->iface);
666 
667  gnrc_ndp_internal_send_nbr_sol(nc_entry->iface, NULL, &nc_entry->ipv6_addr, &dst);
668 
669  mutex_lock(&ipv6_iface->mutex);
670  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, ipv6_iface->retrans_timer,
671  GNRC_NDP_MSG_NBR_SOL_RETRANS, gnrc_ipv6_pid);
672  mutex_unlock(&ipv6_iface->mutex);
673  }
674  }
675  else if (nc_entry->probes_remaining <= 1) {
676 
677  /* For a 6LoWPAN router entries may be set to UNREACHABLE instead
678  * of removing them, since RFC6775, section 6
679  * (https://tools.ietf.org/html/rfc6775#section-6) says: "if NUD on
680  * the router determines that the host is UNREACHABLE (based on the
681  * logic in [RFC4861]), the NCE SHOULD NOT be deleted but rather
682  * retained until the Registration Lifetime expires." However, this
683  * "SHOULD NOT" is not implemented to circumvent NCEs going into
684  * UNREACHABLE forever and in order to save some memory in the
685  * neighbor cache. */
686 
687  DEBUG("ndp: Remove nc entry %s for interface %" PRIkernel_pid "\n",
688  ipv6_addr_to_str(addr_str, &nc_entry->ipv6_addr, sizeof(addr_str)),
689  nc_entry->iface);
690 
691  gnrc_ipv6_nc_remove(nc_entry->iface, &nc_entry->ipv6_addr);
692  }
693  }
