57 int gpio_init(gpio_t pin, gpio_mode_t mode)
58 {
59  GPIO_P_TypeDef *port = _port(pin);
60  uint32_t pin_pos = _pin_pos(pin);
61 
62  /* enable power for the GPIO module */
63  CMU->HFPERCLKEN0 |= CMU_HFPERCLKEN0_GPIO;
64 
65  /* configure the mode */
66  port->MODE[pin_pos >> 3] &= ~(0xf << ((pin_pos & 0x7) * 4));
67  port->MODE[pin_pos >> 3] |= (mode << ((pin_pos & 0x7) * 4));
68  /* reset output register */
69  port->DOUTCLR = (1 << pin_pos);
70  /* if input with pull-up, set the data out register */
71  if (mode == GPIO_IN_PU) {
72  port->DOUTSET = (1 << pin_pos);
73  }
74 
75  return 0;
76 }
