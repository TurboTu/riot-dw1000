67 static void _receive(gnrc_pktsnip_t *pkt)
68 {
69  gnrc_pktsnip_t *payload;
70  uint8_t *dispatch;
71 
72  /* seize payload as a temporary variable */
73  payload = gnrc_pktbuf_start_write(pkt); /* need to duplicate since pkt->next
74  * might get replaced */
75 
76  if (payload == NULL) {
77  DEBUG("6lo: can not get write access on received packet\n");
78 #if defined(DEVELHELP) && ENABLE_DEBUG
79  gnrc_pktbuf_stats();
80 #endif
81  gnrc_pktbuf_release(pkt);
82  return;
83  }
84 
85  pkt = payload; /* reset pkt from temporary variable */
86 
87  payload = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_SIXLOWPAN);
88 
89  if ((payload == NULL) || (payload->size < 1)) {
90  DEBUG("6lo: Received packet has no 6LoWPAN payload\n");
91  gnrc_pktbuf_release(pkt);
92  return;
93  }
94 
95  dispatch = payload->data;
96 
97  if (dispatch[0] == SIXLOWPAN_UNCOMP) {
98  gnrc_pktsnip_t *sixlowpan;
99  DEBUG("6lo: received uncompressed IPv6 packet\n");
100  payload = gnrc_pktbuf_start_write(payload);
101 
102  if (payload == NULL) {
103  DEBUG("6lo: can not get write access on received packet\n");
104 #if defined(DEVELHELP) && ENABLE_DEBUG
105  gnrc_pktbuf_stats();
106 #endif
107  gnrc_pktbuf_release(pkt);
108  return;
109  }
110 
111  /* packet is uncompressed: just mark and remove the dispatch */
112  sixlowpan = gnrc_pktbuf_mark(payload, sizeof(uint8_t), GNRC_NETTYPE_SIXLOWPAN);
113 
114  if (sixlowpan == NULL) {
115  DEBUG("6lo: can not mark 6LoWPAN dispatch\n");
116  gnrc_pktbuf_release(pkt);
117  return;
118  }
119 
120  pkt = gnrc_pktbuf_remove_snip(pkt, sixlowpan);
121  payload->type = GNRC_NETTYPE_IPV6;
122  }
123 #ifdef MODULE_GNRC_SIXLOWPAN_FRAG
124  else if (sixlowpan_frag_is((sixlowpan_frag_t *)dispatch)) {
125  DEBUG("6lo: received 6LoWPAN fragment\n");
126  gnrc_sixlowpan_frag_handle_pkt(pkt);
127  return;
128  }
129 #endif
130 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC
131  else if (sixlowpan_iphc_is(dispatch)) {
132  size_t dispatch_size, nh_len;
133  gnrc_pktsnip_t *sixlowpan;
134  gnrc_pktsnip_t *dec_hdr = gnrc_pktbuf_add(NULL, NULL, sizeof(ipv6_hdr_t),
135  GNRC_NETTYPE_IPV6);
136  if ((dec_hdr == NULL) ||
137  (dispatch_size = gnrc_sixlowpan_iphc_decode(&dec_hdr, pkt, 0, 0,
138  &nh_len)) == 0) {
139  DEBUG("6lo: error on IPHC decoding\n");
140  if (dec_hdr != NULL) {
141  gnrc_pktbuf_release(dec_hdr);
142  }
143  gnrc_pktbuf_release(pkt);
144  return;
145  }
146  sixlowpan = gnrc_pktbuf_mark(pkt, dispatch_size, GNRC_NETTYPE_SIXLOWPAN);
147  if (sixlowpan == NULL) {
148  DEBUG("6lo: error on marking IPHC dispatch\n");
149  gnrc_pktbuf_release(dec_hdr);
150  gnrc_pktbuf_release(pkt);
151  return;
152  }
153 
154  /* Remove IPHC dispatches */
155  /* Insert decoded header instead */
156  pkt = gnrc_pktbuf_replace_snip(pkt, sixlowpan, dec_hdr);
157  payload->type = GNRC_NETTYPE_UNDEF;
158  }
159 #endif
160  else {
161  DEBUG("6lo: dispatch %02" PRIx8 " ... is not supported\n",
162  dispatch[0]);
163  gnrc_pktbuf_release(pkt);
164  return;
165  }
166  if (!gnrc_netapi_dispatch_receive(GNRC_NETTYPE_IPV6, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
167  DEBUG("6lo: No receivers for this packet found\n");
168  gnrc_pktbuf_release(pkt);
169  }
170 }

172 static inline bool _add_uncompr_disp(gnrc_pktsnip_t *pkt)
173 {
174  gnrc_pktsnip_t *sixlowpan;
175  uint8_t *disp;
176 
177  DEBUG("6lo: Send uncompressed\n");
178 
179  sixlowpan = gnrc_pktbuf_add(NULL, NULL, sizeof(uint8_t), GNRC_NETTYPE_SIXLOWPAN);
180 
181  if (sixlowpan == NULL) {
182  return false;
183  }
184  sixlowpan->next = pkt->next;
185  pkt->next = sixlowpan;
186  disp = sixlowpan->data;
187  disp[0] = SIXLOWPAN_UNCOMP;
188  return true;
189 }

191 static void _send(gnrc_pktsnip_t *pkt)
192 {
193  gnrc_netif_hdr_t *hdr;
194  gnrc_pktsnip_t *pkt2;
195  gnrc_sixlowpan_netif_t *iface;
196  /* datagram_size: pure IPv6 packet without 6LoWPAN dispatches or compression */
197  size_t datagram_size;
198 
199  if ((pkt == NULL) || (pkt->size < sizeof(gnrc_netif_hdr_t))) {
200  DEBUG("6lo: Sending packet has no netif header\n");
201  gnrc_pktbuf_release(pkt);
202  return;
203  }
204 
205  if ((pkt->next == NULL) || (pkt->next->type != GNRC_NETTYPE_IPV6)) {
206  DEBUG("6lo: Sending packet has no IPv6 header\n");
207  gnrc_pktbuf_release(pkt);
208  return;
209  }
210 
211  pkt2 = gnrc_pktbuf_start_write(pkt);
212 
213  if (pkt2 == NULL) {
214  DEBUG("6lo: no space left in packet buffer\n");
215  gnrc_pktbuf_release(pkt);
216  return;
217  }
218 
219  hdr = pkt2->data;
220  iface = gnrc_sixlowpan_netif_get(hdr->if_pid);
221  datagram_size = gnrc_pkt_len(pkt2->next);
222 
223  if (iface == NULL) {
224  DEBUG("6lo: Can not get 6LoWPAN specific interface information.\n");
225  gnrc_pktbuf_release(pkt);
226  return;
227  }
228 
229 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC
230  if (iface->iphc_enabled) {
231  if (!gnrc_sixlowpan_iphc_encode(pkt2)) {
232  DEBUG("6lo: error on IPHC encoding\n");
233  gnrc_pktbuf_release(pkt2);
234  return;
235  }
236  /* IPHC dispatch does not count on dispatch length since it _shortens_
237  * the datagram */
238  }
239  else {
240  if (!_add_uncompr_disp(pkt2)) {
241  /* adding uncompressed dispatch failed */
242  DEBUG("6lo: no space left in packet buffer\n");
243  gnrc_pktbuf_release(pkt2);
244  return;
245  }
246  }
247 #else
248  /* suppress clang-analyzer report about iface being not read */
249  (void) iface;
250  if (!_add_uncompr_disp(pkt2)) {
251  /* adding uncompressed dispatch failed */
252  DEBUG("6lo: no space left in packet buffer\n");
253  gnrc_pktbuf_release(pkt2);
254  return;
255  }
256 #endif
257  DEBUG("6lo: iface->max_frag_size = %" PRIu16 " for interface %"
258  PRIkernel_pid "\n", iface->max_frag_size, hdr->if_pid);
259 
260  /* IP should not send anything here if it is not a 6LoWPAN interface,
261  * so we don't need to check for NULL pointers.
262  * Note, that datagram_size cannot be used here, because the header size
263  * might be changed by IPHC. */
264  if (gnrc_pkt_len(pkt2->next) <= iface->max_frag_size) {
265  DEBUG("6lo: Send SND command for %p to %" PRIu16 "\n",
266  (void *)pkt2, hdr->if_pid);
267  if (gnrc_netapi_send(hdr->if_pid, pkt2) < 1) {
268  DEBUG("6lo: unable to send %p over %" PRIu16 "\n", (void *)pkt, hdr->if_pid);
269  gnrc_pktbuf_release(pkt2);
270  }
271 
272  return;
273  }
274 #ifdef MODULE_GNRC_SIXLOWPAN_FRAG
275  else if (fragment_msg.pkt != NULL) {
276  DEBUG("6lo: Fragmentation already ongoing. Dropping packet\n");
277  gnrc_pktbuf_release(pkt2);
278  return;
279  }
280  else if (datagram_size <= SIXLOWPAN_FRAG_MAX_LEN) {
281  DEBUG("6lo: Send fragmented (%u > %" PRIu16 ")\n",
282  (unsigned int)datagram_size, iface->max_frag_size);
283  msg_t msg;
284 
285  fragment_msg.pid = hdr->if_pid;
286  fragment_msg.pkt = pkt2;
287  fragment_msg.datagram_size = datagram_size;
288  /* Sending the first fragment has an offset==0 */
289  fragment_msg.offset = 0;
290 
291  /* set the outgoing message's fields */
292  msg.type = GNRC_SIXLOWPAN_MSG_FRAG_SND;
293  msg.content.ptr = &fragment_msg;
294  /* send message to self */
295  msg_send_to_self(&msg);
296  }
297  else {
298  DEBUG("6lo: packet too big (%u > %" PRIu16 ")\n",
299  (unsigned int)datagram_size, (uint16_t)SIXLOWPAN_FRAG_MAX_LEN);
300  gnrc_pktbuf_release(pkt2);
301  }
302 #else
303  (void) datagram_size;
304  DEBUG("6lo: packet too big (%u > %" PRIu16 ")\n",
305  (unsigned int)datagram_size, iface->max_frag_size);
306  gnrc_pktbuf_release(pkt2);
307 #endif
308 }

310 static void *_event_loop(void *args)
311 {
312  msg_t msg, reply, msg_q[GNRC_SIXLOWPAN_MSG_QUEUE_SIZE];
313  gnrc_netreg_entry_t me_reg = GNRC_NETREG_ENTRY_INIT_PID(GNRC_NETREG_DEMUX_CTX_ALL,
314  sched_active_pid);
315 
316  (void)args;
317  msg_init_queue(msg_q, GNRC_SIXLOWPAN_MSG_QUEUE_SIZE);
318 
319  /* register interest in all 6LoWPAN packets */
320  gnrc_netreg_register(GNRC_NETTYPE_SIXLOWPAN, &me_reg);
321 
322  /* preinitialize ACK */
323  reply.type = GNRC_NETAPI_MSG_TYPE_ACK;
324 
325  /* start event loop */
326  while (1) {
327  DEBUG("6lo: waiting for incoming message.\n");
328  msg_receive(&msg);
329 
330  switch (msg.type) {
331  case GNRC_NETAPI_MSG_TYPE_RCV:
332  DEBUG("6lo: GNRC_NETDEV_MSG_TYPE_RCV received\n");
333  _receive(msg.content.ptr);
334  break;
335 
336  case GNRC_NETAPI_MSG_TYPE_SND:
337  DEBUG("6lo: GNRC_NETDEV_MSG_TYPE_SND received\n");
338  _send(msg.content.ptr);
339  break;
340 
341  case GNRC_NETAPI_MSG_TYPE_GET:
342  case GNRC_NETAPI_MSG_TYPE_SET:
343  DEBUG("6lo: reply to unsupported get/set\n");
344  reply.content.value = -ENOTSUP;
345  msg_reply(&msg, &reply);
346  break;
347 #ifdef MODULE_GNRC_SIXLOWPAN_FRAG
348  case GNRC_SIXLOWPAN_MSG_FRAG_SND:
349  DEBUG("6lo: send fragmented event received\n");
350  gnrc_sixlowpan_frag_send(msg.content.ptr);
351  break;
352 #endif
353 
354  default:
355  DEBUG("6lo: operation not supported\n");
356  break;
357  }
358  }
359 
360  return NULL;
361 }
