44 uint32_t pwm_init(pwm_t pwm, pwm_mode_t mode, uint32_t freq, uint16_t res)
45 {
46  uint32_t bus_clk = periph_apb_clk(pwm_config[pwm].bus);
47 
48  /* verify parameters */
49  assert((pwm < PWM_NUMOF) && ((freq * res) < bus_clk));
50 
51  /* power on the used timer */
52  pwm_poweron(pwm);
53  /* reset configuration and CC channels */
54  dev(pwm)->CR1 = 0;
55  dev(pwm)->CR2 = 0;
56  for (int i = 0; i < TIMER_CHAN; i++) {
57  dev(pwm)->CCR[i] = 0;
58  }
59 
60  /* configure the used pins */
61  unsigned i = 0;
62  while ((i < TIMER_CHAN) && (pwm_config[pwm].chan[i].pin != GPIO_UNDEF)) {
63  gpio_init(pwm_config[pwm].chan[i].pin, GPIO_OUT);
64  gpio_init_af(pwm_config[pwm].chan[i].pin, pwm_config[pwm].af);
65  i++;
66  }
67 
68  /* configure the PWM frequency and resolution by setting the auto-reload
69  * and prescaler registers */
70  dev(pwm)->PSC = (bus_clk / (res * freq)) - 1;
71  dev(pwm)->ARR = res - 1;
72 
73  /* set PWM mode */
74  switch (mode) {
75  case PWM_LEFT:
76  dev(pwm)->CCMR1 = CCMR_LEFT;
77  dev(pwm)->CCMR2 = CCMR_LEFT;
78  break;
79  case PWM_RIGHT:
80  dev(pwm)->CCMR1 = CCMR_RIGHT;
81  dev(pwm)->CCMR2 = CCMR_RIGHT;
82  break;
83  case PWM_CENTER:
84  dev(pwm)->CCMR1 = 0;
85  dev(pwm)->CCMR2 = 0;
86  dev(pwm)->CR1 |= (TIM_CR1_CMS_0 | TIM_CR1_CMS_1);
87  break;
88  }
89 
90  /* enable PWM outputs and start PWM generation */
91 #ifdef TIM_BDTR_MOE
92  dev(pwm)->BDTR = TIM_BDTR_MOE;
93 #endif
94  dev(pwm)->CCER = (TIM_CCER_CC1E | TIM_CCER_CC2E |
95  TIM_CCER_CC3E | TIM_CCER_CC4E);
96  dev(pwm)->CR1 |= TIM_CR1_CEN;
97 
98  /* return the actual used PWM frequency */
99  return (bus_clk / (res * (dev(pwm)->PSC + 1)));
100 }

113 void pwm_set(pwm_t pwm, uint8_t channel, uint16_t value)
114 {
115  assert((pwm < PWM_NUMOF) &&
116  (channel < TIMER_CHAN) &&
117  (pwm_config[pwm].chan[channel].pin != GPIO_UNDEF));
118 
119  /* norm value to maximum possible value */
120  if (value > dev(pwm)->ARR) {
121  value = (uint16_t)dev(pwm)->ARR;
122  }
123  /* set new value */
