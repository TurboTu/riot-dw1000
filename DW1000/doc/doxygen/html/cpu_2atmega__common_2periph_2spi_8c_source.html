47 void spi_init_pins(spi_t bus)
48 {
49 
50  /* the pin configuration for this CPU is fixed:
51  * - PB3: MISO (configure as input - done automatically)
52  * - PB2: MOSI (configure as output)
53  * - PB1: SCK (configure as output)
54  * - PB0: SS (configure as output, but unused)
55  *
56  * The SS pin must be configured as output for the SPI device to work as
57  * master correctly, though we do not use it for now (as we handle the chip
58  * select externally for now)
59  */
60  DDRB |= ((1 << DDB2) | (1 << DDB1) | (1 << DDB0));
61 }

63 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
64 {
65  (void)cs;
66 
67  /* lock the bus and power on the SPI peripheral */
68  mutex_lock(&lock);
69  MEGA_PRR &= ~(1 << PRSPI);
70 
71  /* configure as master, with given mode and clock */
72  SPSR = (clk >> S2X_SHIFT);
73  SPCR = ((1 << SPE) | (1 << MSTR) | mode | (clk & CLK_MASK));
74  SPCR |= (1 << SPE);
75 
76  /* clear interrupt flag by reading SPSR and data register by reading SPDR */
77  (void)SPSR;
78  (void)SPDR;
79 
80  return SPI_OK;
81 }

104  uint8_t tmp = (out_buf) ? out_buf[i] : 0;
105  SPDR = tmp;
