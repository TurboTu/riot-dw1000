157 void gnrc_rpl_send_DIO(gnrc_rpl_instance_t *inst, ipv6_addr_t *destination)
158 {
159  if (inst == NULL) {
160  DEBUG("RPL: Error - trying to send DIO without being part of a dodag.\n");
161  return;
162  }
163 
164  gnrc_rpl_dodag_t *dodag = &inst->dodag;
165  gnrc_pktsnip_t *pkt = NULL, *tmp;
166  gnrc_rpl_dio_t *dio;
167 
168 #ifdef MODULE_GNRC_RPL_P2P
169  gnrc_rpl_p2p_ext_t *p2p_ext = gnrc_rpl_p2p_ext_get(dodag);
170  if (dodag->instance->mop == GNRC_RPL_P2P_MOP) {
171  if (!p2p_ext->for_me) {
172  if ((pkt = gnrc_rpl_p2p_rdo_build(pkt, p2p_ext)) == NULL) {
173  return;
174  }
175  }
176  dodag->dio_opts &= ~GNRC_RPL_REQ_DIO_OPT_PREFIX_INFO;
177  }
178 #endif
179 
180 #ifndef GNRC_RPL_WITHOUT_PIO
181  if (dodag->dio_opts & GNRC_RPL_REQ_DIO_OPT_PREFIX_INFO) {
182  if ((pkt = _dio_prefix_info_build(pkt, dodag)) == NULL) {
183  return;
184  }
185  }
186 #endif
187 
188  if (dodag->dio_opts & GNRC_RPL_REQ_DIO_OPT_DODAG_CONF) {
189  if ((pkt = _dio_dodag_conf_build(pkt, dodag)) == NULL) {
190  return;
191  }
192  dodag->dio_opts &= ~GNRC_RPL_REQ_DIO_OPT_DODAG_CONF;
193  }
194 
195  if ((tmp = gnrc_pktbuf_add(pkt, NULL, sizeof(gnrc_rpl_dio_t), GNRC_NETTYPE_UNDEF)) == NULL) {
196  DEBUG("RPL: Send DIO - no space left in packet buffer\n");
197  gnrc_pktbuf_release(pkt);
198  return;
199  }
200  pkt = tmp;
201  dio = pkt->data;
202  dio->instance_id = inst->id;
203  dio->version_number = dodag->version;
204  /* a leaf node announces an INFINITE_RANK */
205  dio->rank = ((dodag->node_status == GNRC_RPL_LEAF_NODE) ?
206  byteorder_htons(GNRC_RPL_INFINITE_RANK) : byteorder_htons(dodag->my_rank));
207  dio->g_mop_prf = (dodag->grounded << GNRC_RPL_GROUNDED_SHIFT) |
208  (inst->mop << GNRC_RPL_MOP_SHIFT) | dodag->prf;
209  dio->dtsn = dodag->dtsn;
210  dio->flags = 0;
211  dio->reserved = 0;
212  dio->dodag_id = dodag->dodag_id;
213 
214  if ((tmp = gnrc_icmpv6_build(pkt, ICMPV6_RPL_CTRL, GNRC_RPL_ICMPV6_CODE_DIO,
215  sizeof(icmpv6_hdr_t))) == NULL) {
216  DEBUG("RPL: Send DIO - no space left in packet buffer\n");
217  gnrc_pktbuf_release(pkt);
218  return;
219  }
220  pkt = tmp;
221 
222 #ifdef MODULE_NETSTATS_RPL
223  gnrc_rpl_netstats_tx_DIO(&gnrc_rpl_netstats, gnrc_pkt_len(pkt),
224  (destination && !ipv6_addr_is_multicast(destination)));
225 #endif
226 
227  gnrc_rpl_send(pkt, dodag->iface, NULL, destination, &dodag->dodag_id);
228 }

231 {

248  }

251  dis = (gnrc_rpl_dis_t *)(icmp + 1);
252  dis->flags = 0;
253  dis->reserved = 0;
254 
255  /* TODO add padding may be removed if packet size grows */

257 
258 #ifdef MODULE_NETSTATS_RPL

260  (destination && !ipv6_addr_is_multicast(destination)));

267  ipv6_addr_t *dst, uint16_t len)
268 {
269  /* TODO handle Solicited Information Option */
270  (void)iface;
271  (void)dis;
272  (void)len;
273 
274 #ifdef MODULE_NETSTATS_RPL
275  gnrc_rpl_netstats_rx_DIS(&gnrc_rpl_netstats, len, (dst && !ipv6_addr_is_multicast(dst)));
276 #endif
277 
278 #ifndef GNRC_RPL_WITHOUT_VALIDATION
279  if (!gnrc_rpl_validation_DIS(dis, len)) {
280  return;
281  }
282 #endif
283 
284  if (ipv6_addr_is_multicast(dst)) {
285  for (uint8_t i = 0; i < GNRC_RPL_INSTANCES_NUMOF; ++i) {
286  if ((gnrc_rpl_instances[i].state != 0)
287  /* a leaf node should only react to unicast DIS */
288  && (gnrc_rpl_instances[i].dodag.node_status != GNRC_RPL_LEAF_NODE)) {
289 #ifdef MODULE_GNRC_RPL_P2P

291  DEBUG("RPL: Not responding to DIS for P2P-RPL DODAG\n");
292  continue;
293  }
294 #endif
295  trickle_reset_timer(&(gnrc_rpl_instances[i].dodag.trickle));
296  }
297  }
298  }
299  else {
300  for (uint8_t i = 0; i < GNRC_RPL_INSTANCES_NUMOF; ++i) {
301  if (gnrc_rpl_instances[i].state != 0) {
302  gnrc_rpl_instances[i].dodag.dio_opts |= GNRC_RPL_REQ_DIO_OPT_DODAG_CONF;
303  gnrc_rpl_send_DIO(&gnrc_rpl_instances[i], src);
304  }
305  }
306  }

310 bool _parse_options(int msg_type, gnrc_rpl_instance_t *inst, gnrc_rpl_opt_t *opt, uint16_t len,
311  ipv6_addr_t *src, uint32_t *included_opts)
312 {
313  uint16_t l = 0;
314  gnrc_rpl_opt_target_t *first_target = NULL;
315  gnrc_rpl_dodag_t *dodag = &inst->dodag;
316  eui64_t iid;
317  *included_opts = 0;
318  ipv6_addr_t *me;
319 
320 #ifndef GNRC_RPL_WITHOUT_VALIDATION
321  if (!gnrc_rpl_validation_options(msg_type, inst, opt, len)) {
322  return false;
323  }
324 #else
325  (void) msg_type;
326 #endif
327 
328  while(l < len) {
329  switch(opt->type) {
330  case (GNRC_RPL_OPT_PAD1):
331  DEBUG("RPL: PAD1 option parsed\n");
332  *included_opts |= ((uint32_t) 1) << GNRC_RPL_OPT_PAD1;
333  l += 1;
334  opt = (gnrc_rpl_opt_t *) (((uint8_t *) opt) + 1);
335  continue;
336 
337  case (GNRC_RPL_OPT_PADN):
338  DEBUG("RPL: PADN option parsed\n");
339  *included_opts |= ((uint32_t) 1) << GNRC_RPL_OPT_PADN;
340  break;
341 
342  case (GNRC_RPL_OPT_DODAG_CONF):
343  DEBUG("RPL: DODAG CONF DIO option parsed\n");
344  *included_opts |= ((uint32_t) 1) << GNRC_RPL_OPT_DODAG_CONF;
345  dodag->dio_opts |= GNRC_RPL_REQ_DIO_OPT_DODAG_CONF;
346  gnrc_rpl_opt_dodag_conf_t *dc = (gnrc_rpl_opt_dodag_conf_t *) opt;
347  gnrc_rpl_of_t *of = gnrc_rpl_get_of_for_ocp(byteorder_ntohs(dc->ocp));
348  if (of != NULL) {
349  inst->of = of;
350  }
351  else {
352  DEBUG("RPL: Unsupported OCP 0x%02x\n", byteorder_ntohs(dc->ocp));
353  inst->of = gnrc_rpl_get_of_for_ocp(GNRC_RPL_DEFAULT_OCP);
354  }
355  dodag->dio_interval_doubl = dc->dio_int_doubl;
356  dodag->dio_min = dc->dio_int_min;
357  dodag->dio_redun = dc->dio_redun;
358  inst->max_rank_inc = byteorder_ntohs(dc->max_rank_inc);
359  inst->min_hop_rank_inc = byteorder_ntohs(dc->min_hop_rank_inc);
360  dodag->default_lifetime = dc->default_lifetime;
361  dodag->lifetime_unit = byteorder_ntohs(dc->lifetime_unit);
362  dodag->trickle.Imin = (1 << dodag->dio_min);
363  dodag->trickle.Imax = dodag->dio_interval_doubl;
364  dodag->trickle.k = dodag->dio_redun;
365  break;
366 
367  case (GNRC_RPL_OPT_PREFIX_INFO):
368  DEBUG("RPL: Prefix Information DIO option parsed\n");
369  *included_opts |= ((uint32_t) 1) << GNRC_RPL_OPT_PREFIX_INFO;
370 #ifndef GNRC_RPL_WITHOUT_PIO
371  dodag->dio_opts |= GNRC_RPL_REQ_DIO_OPT_PREFIX_INFO;
372 #endif
373  gnrc_rpl_opt_prefix_info_t *pi = (gnrc_rpl_opt_prefix_info_t *) opt;
374  /* check for the auto address-configuration flag */
375  if ((gnrc_netapi_get(dodag->iface, NETOPT_IPV6_IID, 0, &iid, sizeof(eui64_t)) < 0)
376  && !(pi->LAR_flags & GNRC_RPL_PREFIX_AUTO_ADDRESS_BIT)) {
377  break;
378  }
379  ipv6_addr_set_aiid(&pi->prefix, iid.uint8);
380  me = gnrc_ipv6_netif_add_addr(dodag->iface, &pi->prefix, pi->prefix_len, 0);
381  if (me) {
382  dodag->netif_addr = gnrc_ipv6_netif_addr_get(me);
383  }
384 
385  break;
386 
387  case (GNRC_RPL_OPT_TARGET):
388  DEBUG("RPL: RPL TARGET DAO option parsed\n");
389  *included_opts |= ((uint32_t) 1) << GNRC_RPL_OPT_TARGET;
390 
391  gnrc_rpl_opt_target_t *target = (gnrc_rpl_opt_target_t *) opt;
392  if (first_target == NULL) {
393  first_target = target;
394  }
395 
396  uint32_t fib_dst_flags = 0;
397 
398  if (target->prefix_length <= IPV6_ADDR_BIT_LEN) {
399  fib_dst_flags = ((uint32_t)(target->prefix_length) << FIB_FLAG_NET_PREFIX_SHIFT);
400  }
401 
402  DEBUG("RPL: adding fib entry %s/%d 0x%" PRIx32 "\n",
403  ipv6_addr_to_str(addr_str, &(target->target), sizeof(addr_str)),
404  target->prefix_length,
405  fib_dst_flags);
406 
407  fib_add_entry(&gnrc_ipv6_fib_table, dodag->iface, target->target.u8,
408  sizeof(ipv6_addr_t), fib_dst_flags, src->u8,
409  sizeof(ipv6_addr_t), FIB_FLAG_RPL_ROUTE,
410  (dodag->default_lifetime * dodag->lifetime_unit) *
411  MS_PER_SEC);
412  break;
413 
414  case (GNRC_RPL_OPT_TRANSIT):
415  DEBUG("RPL: RPL TRANSIT INFO DAO option parsed\n");
416  *included_opts |= ((uint32_t) 1) << GNRC_RPL_OPT_TRANSIT;
417  gnrc_rpl_opt_transit_t *transit = (gnrc_rpl_opt_transit_t *) opt;
418  if (first_target == NULL) {
419  DEBUG("RPL: Encountered a RPL TRANSIT DAO option without "
420  "a preceding RPL TARGET DAO option\n");
421  break;
422  }
423 
424  do {
425  DEBUG("RPL: updating fib entry %s/%d\n",
426  ipv6_addr_to_str(addr_str, &(first_target->target), sizeof(addr_str)),
427  first_target->prefix_length);
428 
429  fib_update_entry(&gnrc_ipv6_fib_table,
430  first_target->target.u8,
431  sizeof(ipv6_addr_t), src->u8,
432  sizeof(ipv6_addr_t),
433  ((transit->e_flags & GNRC_RPL_OPT_TRANSIT_E_FLAG) ?
434  0x0 : FIB_FLAG_RPL_ROUTE),
435  (transit->path_lifetime *
436  dodag->lifetime_unit * MS_PER_SEC));
437  first_target = (gnrc_rpl_opt_target_t *) (((uint8_t *) (first_target)) +
438  sizeof(gnrc_rpl_opt_t) + first_target->length);
439  }
440  while (first_target->type == GNRC_RPL_OPT_TARGET);
441 
442  first_target = NULL;
443  break;
444 
445 #ifdef MODULE_GNRC_RPL_P2P
446  case (GNRC_RPL_P2P_OPT_RDO):
447  gnrc_rpl_p2p_rdo_parse((gnrc_rpl_p2p_opt_rdo_t *) opt, gnrc_rpl_p2p_ext_get(dodag));
448  break;
449 #endif
450  }
451  l += opt->length + sizeof(gnrc_rpl_opt_t);
452  opt = (gnrc_rpl_opt_t *) (((uint8_t *) (opt + 1)) + opt->length);
453  }
454  return true;
455 }

457 void gnrc_rpl_recv_DIO(gnrc_rpl_dio_t *dio, kernel_pid_t iface, ipv6_addr_t *src, ipv6_addr_t *dst,
458  uint16_t len)
459 {
460  (void) dst;
461  gnrc_rpl_instance_t *inst = NULL;
462  gnrc_rpl_dodag_t *dodag = NULL;
463 
464 #ifdef MODULE_NETSTATS_RPL
465  gnrc_rpl_netstats_rx_DIO(&gnrc_rpl_netstats, len, (dst && !ipv6_addr_is_multicast(dst)));
466 #endif
467 
468 #ifndef GNRC_RPL_WITHOUT_VALIDATION

471  }

491  }

497  DEBUG("RPL: Joined DODAG (%s).\n",
498  ipv6_addr_to_str(addr_str, &dio->dodag_id, sizeof(addr_str)));
499
