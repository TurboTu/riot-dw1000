442 __attribute__((always_inline)) __STATIC_INLINE uint32_t __RBIT(uint32_t value)
443 {
444  uint32_t result;
445  int32_t s = 4 /*sizeof(v)*/ * 8 - 1; /* extra shift needed at end */
446 
447  result = value; /* r will be reversed bits of v; first get LSB of v */
448  for (value >>= 1U; value; value >>= 1U)
449  {
450  result <<= 1U;
451  result |= value & 1U;
452  s--;
453  }
454  result <<= s; /* shift when v's highest bits are zero */
455  return(result);
