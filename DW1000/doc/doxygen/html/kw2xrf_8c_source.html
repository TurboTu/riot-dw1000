128 void kw2xrf_set_sequence(kw2xrf_t *dev, kw2xrf_physeq_t seq)
129 {
130  uint8_t reg = 0;
131 
132  /* Only interrupt interruptable states */
133  uint8_t curr_seq = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
134  curr_seq &= (MKW2XDM_PHY_CTRL1_XCVSEQ_MASK);
135 
136  if ((curr_seq == XCVSEQ_RECEIVE) || (curr_seq == XCVSEQ_CONTINUOUS_CCA)) {
137  /* Clear all pending interrupts */
138  gpio_irq_disable(kw2xrf_gpio_int);
139 
140  /* abort any ongoing sequence */
141  DEBUG("kw2xrf_tx: abort SEQ_STATE: %x\n", kw2xrf_read_dreg(MKW2XDM_SEQ_STATE));
142  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
143  reg &= ~(MKW2XDM_PHY_CTRL1_XCVSEQ_MASK);
144  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
145 
146  /* Mask all possible interrupts */
147  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL3);
148  reg |= MKW2XDM_PHY_CTRL3_WAKE_MSK;
149  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL3, reg);
150 
151  kw2xrf_write_dreg(MKW2XDM_IRQSTS1, 0x7f);
152  kw2xrf_write_dreg(MKW2XDM_IRQSTS2, 0x03);
153  kw2xrf_write_dreg(MKW2XDM_IRQSTS3, 0xff);
154 
155  gpio_irq_enable(kw2xrf_gpio_int);
156  }
157 
158 #ifdef DEVELHELP
159  uint16_t max_tries = 0;
160 #endif
161 
162  /* Wait for all other states finish */
163  while (kw2xrf_read_dreg(MKW2XDM_SEQ_STATE)) {
164 #ifdef DEVELHELP
165  max_tries++;
166 
167  /* At 10MHz SPI-Clock, 40000 should be in the magnitue of 0.1s */
168  if (max_tries == 40000) {
169  DEBUG("kw2xrf_error: device does not finish sequence\n");
170  core_panic(PANIC_GENERAL_ERROR, "kw2xrf_error: device does not finish sequence");
171  }
172 
173 #endif
174  }
175 
176  /* For all sequences only enable SEQ-irq will be set when sequence completed */
177  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL2);
178  reg &= ~(MKW2XDM_PHY_CTRL2_SEQMSK);
179  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL2, reg);
180 
181  /* Progrmm new sequence */
182  switch (seq) {
183  case XCVSEQ_IDLE:
184  dev->state = NETOPT_STATE_SLEEP;
185  break;
186 
187  case XCVSEQ_RECEIVE:
188  dev->state = NETOPT_STATE_RX;
189  break;
190 
191  case XCVSEQ_TRANSMIT:
192  dev->state = NETOPT_STATE_TX;
193  break;
194 
195  case XCVSEQ_CCA:
196  dev->state = NETOPT_STATE_TX;
197  break;
198 
199  case XCVSEQ_TX_RX:
200  dev->state = NETOPT_STATE_TX;
201  break;
202 
203  case XCVSEQ_CONTINUOUS_CCA:
204  dev->state = NETOPT_STATE_TX;
205  break;
206 
207  default:
208  DEBUG("kw2xrf: undefined state assigned to phy\n");
209  dev->state = NETOPT_STATE_IDLE;
210  }
211 
212  /* Mapping of TX-sequences depending on AUTOACK flag */
213  /* TODO: This should only used in combination with
214  * an CSMA-MAC layer. Currently not working
215  */
216  /*if ((seq == XCVSEQ_TRANSMIT) || (seq == XCVSEQ_TX_RX)) {
217  if ((dev->option) & KW2XRF_OPT_AUTOACK) {
218  seq = XCVSEQ_TX_RX;
219  }
220  else {
221  seq = XCVSEQ_TRANSMIT;
222  }
223  }*/
224 
225  DEBUG("kw2xrf: Set sequence to %i\n", seq);
226  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
227  reg &= ~(MKW2XDM_PHY_CTRL1_XCVSEQ_MASK);
228  reg |= MKW2XDM_PHY_CTRL1_XCVSEQ(seq);
229  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
230 }

232 int kw2xrf_set_channel(kw2xrf_t *dev, uint8_t *val, size_t len)
233 {
234  /* Save old sequence to restore this state later */
235  uint8_t old_seq = kw2xrf_get_sequence();
236 
237  if (old_seq) {
238  kw2xrf_set_sequence(dev, XCVSEQ_IDLE);
239  }
240 
241  if ((val[0] < 11) || (val[0] > 26)) {
242  DEBUG("kw2xrf: Invalid channel %i set. Valid channels are 11 through 26\n", val[0]);
243  return -ENOTSUP;
244  }
245 
246  if ((len != 2) || (val[1] != 0)) {
247  DEBUG("kw2xrf: set channel failed, len: %u, val[0]:%u\n", len, val[0]);
248  return -EINVAL;
249  }
250 
251  /*
252  * Fc = 2405 + 5(k - 11) , k = 11,12,...,26
253  *
254  * Equation for PLL frequency, MKW2xD Reference Manual, p.255 :
255  * F = ((PLL_INT0 + 64) + (PLL_FRAC0/65536))32MHz
256  *
257  */
258  uint8_t tmp = val[0] - 11;
259  kw2xrf_write_dreg(MKW2XDM_PLL_INT0, MKW2XDM_PLL_INT0_VAL(pll_int_lt[tmp]));
260  kw2xrf_write_dreg(MKW2XDM_PLL_FRAC0_LSB, (uint8_t)pll_frac_lt[tmp]);
261  kw2xrf_write_dreg(MKW2XDM_PLL_FRAC0_MSB, (uint8_t)(pll_frac_lt[tmp] >> 8));
262 
263  DEBUG("kw2xrf: set channel to %u\n", val[0]);
264 
265  if (old_seq) {
266  kw2xrf_set_sequence(dev, old_seq);
267  }
268 
269  return 2;
270 }

386 int kw2xrf_init(kw2xrf_t *dev, spi_t spi, spi_clk_t spi_clk,
387  gpio_t cs_pin, gpio_t int_pin)
388 {
389  uint8_t reg = 0;
390  uint8_t tmp[2];
391  kw2xrf_gpio_int = int_pin;
392 #if CPUID_LEN
393 /* make sure that the buffer is always big enough to store a 64bit value */
394 # if CPUID_LEN < IEEE802154_LONG_ADDRESS_LEN
395  uint8_t cpuid[IEEE802154_LONG_ADDRESS_LEN];
396 # else
397  uint8_t cpuid[CPUID_LEN];
398 #endif
399  uint16_t addr_short = 0;
400  eui64_t addr_long;
401 #endif
402 
403  /* check device parameters */
404  if (dev == NULL) {
405  return -ENODEV;
406  }
407 
408  kw2xrf_spi_init(spi, spi_clk, cs_pin);
409 
410  if (kw2xrf_on(dev) != 0) {
411  core_panic(PANIC_GENERAL_ERROR, "Could not start MKW2XD radio transceiver");
412  }
413 
414  /* General initialization of interrupt sources.
415  * sets radio to idle mode with all interrupt masked
416  */
417  kw2xrf_init_interrupts(dev, kw2xrf_gpio_int);
418 
419  /* set device driver */
420  dev->driver = &kw2xrf_driver;
421  /* set default option */
422  dev->proto = KW2XRF_DEFAULT_PROTOCOL;
423  dev->option = 0;
424 
425 #if CPUID_LEN
426  /* in case CPUID_LEN < 8, fill missing bytes with zeros */
427  memset(cpuid, 0, CPUID_LEN);
428 
429  cpuid_get(cpuid);
430 
431  /* generate short hardware address if CPUID_LEN > 0 */
432  for (int i = 0; i < CPUID_LEN; i++) {
433  /* XOR each even byte of the CPUID with LSB of short address
434  and each odd byte with MSB */
435  addr_short ^= (uint16_t)(cpuid[i] << ((i & 0x01) * 8));
436  }
437 #if CPUID_LEN > IEEE802154_LONG_ADDRESS_LEN
438  for (int i = IEEE802154_LONG_ADDRESS_LEN; i < CPUID_LEN; i++) {
439  cpuid[i & 0x07] ^= cpuid[i];
440  }
441 #endif
442  /* make sure we mark the address as non-multicast and not globally unique */
443  cpuid[0] &= ~(0x01);
444  cpuid[0] |= 0x02;
445  /* copy and set long address */
446  memcpy(&addr_long, cpuid, IEEE802154_LONG_ADDRESS_LEN);
447  kw2xrf_set_addr_long(dev, NTOHLL(addr_long.uint64.u64));
448  kw2xrf_set_addr(dev, addr_short);
449 #else
450  kw2xrf_set_addr_long(dev, KW2XRF_DEFAULT_SHORT_ADDR);
451  kw2xrf_set_addr(dev, KW2XRF_DEFAULT_ADDR_LONG);
452 #endif
453 
454  /* set default TX-Power */
455  dev->tx_power = KW2XRF_DEFAULT_TX_POWER;
456  kw2xrf_set_tx_power(dev, &(dev->tx_power), sizeof(dev->tx_power));
457 
458  /* set default channel */
459  dev->radio_channel = KW2XRF_DEFAULT_CHANNEL;
460  tmp[0] = dev->radio_channel;
461  tmp[1] = 0;
462  kw2xrf_set_channel(dev, tmp, 2);
463  /* set default PAN ID */
464  kw2xrf_set_pan(dev, KW2XRF_DEFAULT_PANID);
465 
466  /* CCA Setup */
467  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL4);
468  /* Set up CCA mode 1 (RSSI threshold) */
469  reg |= MKW2XDM_PHY_CTRL4_CCATYPE(1);
470  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL4, reg);
471  DEBUG("kw2xrf: Initialized and set to channel %i and pan %i.\n",
472  KW2XRF_DEFAULT_CHANNEL, KW2XRF_DEFAULT_PANID);
473 
474  kw2xrf_set_option(dev, KW2XRF_OPT_AUTOACK, true);
475  kw2xrf_set_option(dev, KW2XRF_OPT_CSMA, true);
476 
477  /* Switch to Receive state per default after initialization */
478  kw2xrf_set_sequence(dev, XCVSEQ_RECEIVE);
479  return 0;
480 }

567

585  *((uint16_t *)value) = IEEE802154_LONG_ADDRESS_LEN;
586  }
587  else {
588  *((uint16_t *)value) = IEEE802154_SHORT_ADDRESS_LEN;
589  }
590 
591  return sizeof(uint16_t);
592 
593  case NETOPT_NID:
594  if (max_len < sizeof(uint16_t)) {
595  return -EOVERFLOW;
596  }

598  *((uint16_t *)value) = dev->radio_pan;
599  return sizeof(uint16_t);

608  }
609  else {
610  uint16_t addr = kw2xrf_get_addr_short(dev);
611  ieee802154_get_iid(value, (uint8_t *)&addr, IEEE802154_SHORT_ADDRESS_LEN);
612  }
613  return sizeof(eui64_t);
614 
615  case NETOPT_CHANNEL:
616  return kw2xrf_get_channel(dev, (uint8_t *)value, max_len);
617 
618  case NETOPT_PROTO:
619  return kw2xrf_get_proto(dev, (uint8_t *)value, max_len);
620

674 void kw2xrf_set_option(kw2xrf_t *dev, uint16_t option, bool state)
675 {
676  uint8_t reg;
677 
678  DEBUG("set option %i to %i\n", option, state);
679 
680  /* set option field */
681  if (state) {
682  dev->option |= option;
683 
684  /* trigger option specific actions */
685  switch (option) {
686  case KW2XRF_OPT_CSMA:
687  DEBUG("[kw2xrf] opt: enabling CSMA mode\n");
688  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
689  reg |= MKW2XDM_PHY_CTRL1_CCABFRTX;
690  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
691  break;
692 
693  case KW2XRF_OPT_PROMISCUOUS:
694  DEBUG("[kw2xrf] opt: enabling PROMISCUOUS mode\n");
695  /* disable auto ACKs in promiscuous mode */
696  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
697  reg &= ~(MKW2XDM_PHY_CTRL1_AUTOACK);
698  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
699  /* enable promiscuous mode */
700  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL4);
701  reg |= MKW2XDM_PHY_CTRL4_PROMISCUOUS;
702  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL4, reg);
703  break;
704 
705  case KW2XRF_OPT_AUTOACK:
706  DEBUG("[kw2xrf] opt: enabling auto ACKs\n");
707  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
708  reg |= MKW2XDM_PHY_CTRL1_AUTOACK;
709  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
710  break;
711 
712  default:
713  /* do nothing */
714  break;
715  }
716  }
717  else {
718  dev->option &= ~(option);
719 
720  /* trigger option specific actions */
721  switch (option) {
722  case KW2XRF_OPT_CSMA:
723  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
724  reg &= ~(MKW2XDM_PHY_CTRL1_CCABFRTX);
725  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
726  break;
727 
728  case KW2XRF_OPT_PROMISCUOUS:
729  /* disable promiscuous mode */
730  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL4);
731  reg &= ~(MKW2XDM_PHY_CTRL4_PROMISCUOUS);
732  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL4, reg);
733 
734  /* re-enable AUTOACK only if the option is set */
735  if (dev->option & KW2XRF_OPT_AUTOACK) {
736  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
737  reg |= MKW2XDM_PHY_CTRL1_AUTOACK;
738  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
739  }
740 
741  break;
742 
743  case KW2XRF_OPT_AUTOACK:
744  reg = kw2xrf_read_dreg(MKW2XDM_PHY_CTRL1);
745  reg &= ~(MKW2XDM_PHY_CTRL1_AUTOACK);
746  kw2xrf_write_dreg(MKW2XDM_PHY_CTRL1, reg);
747  break;
748 
749  default:
750  /* do nothing */
751  break;
752  }

916  return (len + IEEE802154_LONG_ADDRESS_LEN);

934

949

952  }
953  else if (tmp == IEEE802154_FCF_DST_ADDR_LONG) {
954  dst_len = IEEE802154_LONG_ADDRESS_LEN;
955  }
956  else if (tmp == 0) {
957  dst_len = 0;
958  }
959  else {
960  return NULL;
961  }
962 
963  /* allocate space for header */
964  snip = gnrc_pktbuf_add(NULL, NULL, sizeof(gnrc_netif_hdr_t) + src_len + dst_len,
965  GNRC_NETTYPE_NETIF);
966 
967  if (snip == NULL) {
968  return NULL;
969  }
970

972  hdr = (gnrc_netif_hdr_t *)snip->data;
973  gnrc_netif_hdr_init(hdr, src_len, dst_len);
974

976  tmp = 5 + dst_len;
977  addr = gnrc_netif_hdr_get_dst_addr(hdr);
978 
979  for (int i = 0; i < dst_len; i++) {
980  addr[i] = mhr[5 + (dst_len - i) - 1];
981  }
982  }
983  else {
984  tmp = 3;
985  }
986 
987  if (!(mhr[0] & IEEE802154_FCF_PAN_COMP)) {
988  tmp += 2;
989  }
990 
991  if (src_len > 0) {
992  addr = gnrc_netif_hdr_get_src_addr(hdr);
993 
994  for (int i = 0; i < src_len; i++) {
995  addr[i] = mhr[tmp + (src_len - i) - 1];
996  }
997  }

1002 void _receive_data(kw2xrf_t *dev)
1003 {
1004  size_t pkt_len, hdr_len;
1005  gnrc_pktsnip_t *hdr, *payload = NULL;
1006  gnrc_netif_hdr_t *netif;
1007 
1008  /* get size of the received packet */
1009  pkt_len = kw2xrf_read_dreg(MKW2XDM_RX_FRM_LEN);
1010 
1011  /* read PSDU */
1012  kw2xrf_read_fifo(dev->buf, pkt_len + 1);
1013 
1014  /* abort here already if no event callback is registered */
1015  if (!dev->event_cb) {
1016  return;
1017  }
1018 
1019  /* If RAW-mode is selected directly forward pkt, MAC does the rest */
1020  if (dev->option & KW2XRF_OPT_RAWDUMP) {
1021  payload = gnrc_pktbuf_add(NULL, NULL, pkt_len, GNRC_NETTYPE_UNDEF);
1022 
1023  if (payload == NULL) {
1024  DEBUG("kw2xf: error: unable to allocate RAW data\n");
1025  return;
1026  }
1027 
1028  payload->data = dev->buf;
1029  dev->event_cb(NETDEV_EVENT_RX_COMPLETE, payload);
1030  return;
1031  }
1032 
1033  /* get FCF field and compute 802.15.4 header length */
1034  hdr_len = _get_frame_hdr_len(dev->buf);
1035 
1036  if (hdr_len == 0) {
1037  DEBUG("kw2xrf error: unable parse incoming frame header\n");
1038  return;
1039  }
1040 
1041  /* read the rest of the header and parse the netif header from it */
1042  hdr = _make_netif_hdr(dev->buf);
1043 
1044  if (hdr == NULL) {
1045  DEBUG("kw2xrf error: unable to allocate netif header\n");
1046  return;
1047  }
1048 
1049  /* fill missing fields in netif header */
1050  netif = (gnrc_netif_hdr_t *)hdr->data;
1051  netif->if_pid = thread_getpid();
1052  netif->lqi = dev->buf[pkt_len];
1053  /* lqi and rssi are directly related to each other in the kw2x-device.
1054  * The rssi-unit is dBm and in this case always negative, nevertheless a
1055  * positive value is reported.
1056  */
1057  netif->rssi = -((netif->lqi) - 286.6) / 2.69333;
1058 
1059  payload = gnrc_pktbuf_add(hdr, (void *) & (dev->buf[hdr_len]),
1060  pkt_len - hdr_len - 2, dev->proto);
1061 
1062  if (payload == NULL) {
1063  DEBUG("kw2xrf: ERROR allocating payload in packet buffer on RX\n");
1064  gnrc_pktbuf_release(hdr);
1065  return;
1066  }
1067 
1068  dev->event_cb(NETDEV_EVENT_RX_COMPLETE, payload);
1069 }

1071 void kw2xrf_isr_event(gnrc_netdev_t *netdev, uint32_t event_type)
1072 {

1074  uint8_t irqst1 = kw2xrf_read_dreg(MKW2XDM_IRQSTS1);
1075  uint8_t irqst2 = kw2xrf_read_dreg(MKW2XDM_IRQSTS2);
1076 
1077  if ((irqst1 & MKW2XDM_IRQSTS1_RXIRQ) && (irqst1 & MKW2XDM_IRQSTS1_SEQIRQ)) {
1078  /* RX */

1080  _receive_data(dev);
1081  kw2xrf_set_sequence(dev, XCVSEQ_RECEIVE);
1082  kw2xrf_write_dreg(MKW2XDM_IRQSTS1, MKW2XDM_IRQSTS1_RXIRQ | MKW2XDM_IRQSTS1_SEQIRQ);
1083  }
1084  else if ((irqst1 & MKW2XDM_IRQSTS1_TXIRQ) && (irqst1 & MKW2XDM_IRQSTS1_SEQIRQ)) {
1085  /* TX_Complete */
1086  /* Device is automatically in Radio-idle state when TX is done */
1087  kw2xrf_set_sequence(dev, XCVSEQ_RECEIVE);
1088  DEBUG("kw2xrf: TX Complete\n");
1089  kw2xrf_write_dreg(MKW2XDM_IRQSTS1, MKW2XDM_IRQSTS1_TXIRQ | MKW2XDM_IRQSTS1_SEQIRQ);
1090  }
1091  else if ((irqst1 & MKW2XDM_IRQSTS1_CCAIRQ) && (irqst1 & MKW2XDM_IRQSTS1_SEQIRQ)) {
1092  /* TX_Started (CCA_done) */
1093  if (irqst2 & MKW2XDM_IRQSTS2_CCA) {
1094  DEBUG("kw2xrf: CCA done -> Channel busy\n");
1095  }
1096  else {
1097  DEBUG("kw2xrf: CCA done -> Channel idle\n");
1098  }
1099 
1100  kw2xrf_write_dreg(MKW2XDM_IRQSTS1, MKW2XDM_IRQSTS1_CCAIRQ | MKW2XDM_IRQSTS1_SEQIRQ);
1101  kw2xrf_set_sequence(dev, XCVSEQ_RECEIVE);
1102  }
1103  else {
1104  /* Unknown event */
1105  /* Clear all interrupts to prevent ISR-loop */
1106  kw2xrf_write_dreg(MKW2XDM_IRQSTS1, 0x7f);
1107  kw2xrf_write_dreg(MKW2XDM_IRQSTS2, 0x03);
1108  kw2xrf_write_dreg(MKW2XDM_IRQSTS3, 0xff);
1109  DEBUG("kw2xrf_isr_event: unknown Interrupt\n");
1110  kw2xrf_set_sequence(dev, XCVSEQ_RECEIVE);
1111  return;
1112  }

1115 /* TODO: Move to gnrc_ieee802.15.4 as soon as ready */
1116 int _assemble_tx_buf(kw2xrf_t *dev, gnrc_pktsnip_t *pkt)
1117 {
1118  gnrc_netif_hdr_t *hdr;
1119  int index = 4;
1120 
1121  if (dev == NULL) {
1122  return 0;
1123  }
1124 
1125  /* get netif header check address length */
1126  hdr = (gnrc_netif_hdr_t *)pkt->data;
1127 
1128  /* we are building a data frame here */
1129  dev->buf[1] = IEEE802154_FCF_TYPE_DATA;
1130  dev->buf[2] = IEEE802154_FCF_VERS_V0;
1131 
1132  /* if AUTOACK is enabled, then we also expect ACKs for this packet */
1133  if (!(hdr->flags & GNRC_NETIF_HDR_FLAGS_BROADCAST) &&
1134  !(hdr->flags & GNRC_NETIF_HDR_FLAGS_MULTICAST) &&
1135  (dev->option & KW2XRF_OPT_AUTOACK)) {
1136  dev->buf[1] |= IEEE802154_FCF_ACK_REQ;
1137  }
1138 
1139  /* set destination pan_id */
1140  dev->buf[index++] = (uint8_t)((dev->radio_pan) & 0xff);
1141  dev->buf[index++] = (uint8_t)((dev->radio_pan) >> 8);
1142 
1143  /* fill in destination address */
1144  if (hdr->flags &
1145  (GNRC_NETIF_HDR_FLAGS_BROADCAST | GNRC_NETIF_HDR_FLAGS_MULTICAST)) {
1146  dev->buf[2] |= IEEE802154_FCF_DST_ADDR_SHORT;
1147  dev->buf[index++] = 0xff;
1148  dev->buf[index++] = 0xff;
1149  }
1150  else if (hdr->dst_l2addr_len == IEEE802154_SHORT_ADDRESS_LEN) {
1151  /* set to short addressing mode */
1152  dev->buf[2] |= IEEE802154_FCF_DST_ADDR_SHORT;
1153  /* set destination address, byte order is inverted */
1154  uint8_t *dst_addr = gnrc_netif_hdr_get_dst_addr(hdr);
1155  dev->buf[index++] = dst_addr[1];
1156  dev->buf[index++] = dst_addr[0];
1157  }
1158  else if (hdr->dst_l2addr_len == IEEE802154_LONG_ADDRESS_LEN) {
1159  /* default to use long address mode for src and dst */
1160  dev->buf[2] |= IEEE802154_FCF_DST_ADDR_LONG;
1161  /* set destination address located directly after gnrc_ifhrd_t in memory */
1162  uint8_t *dst_addr = gnrc_netif_hdr_get_dst_addr(hdr);
1163  for (int i = IEEE802154_LONG_ADDRESS_LEN - 1; i >= 0; i--) {
1164  dev->buf[index++] = dst_addr[i];
1165  }
1166  }
1167  else {
1168  return 0;
1169  }
1170 
1171  /* fill in source PAN ID (if applicable */
1172  if (dev->option & KW2XRF_OPT_USE_SRC_PAN) {
1173  dev->buf[index++] = (uint8_t)((dev->radio_pan) & 0xff);
1174  dev->buf[index++] = (uint8_t)((dev->radio_pan) >> 8);
1175  }
1176  else {
1177  dev->buf[1] |= IEEE802154_FCF_PAN_COMP;
1178  }
1179 
1180  /* insert source address according to length */
1181  if (hdr->src_l2addr_len == IEEE802154_SHORT_ADDRESS_LEN) {
1182  dev->buf[2] |= IEEE802154_FCF_SRC_ADDR_SHORT;
1183  dev->buf[index++] = (uint8_t)(dev->addr_short[0]);
1184  dev->buf[index++] = (uint8_t)(dev->addr_short[1]);
1185  }
1186  else {
1187  dev->buf[2] |= IEEE802154_FCF_SRC_ADDR_LONG;
1188  memcpy(&(dev->buf[index]), dev->addr_long, IEEE802154_LONG_ADDRESS_LEN);
1189  index += IEEE802154_LONG_ADDRESS_LEN;
1190  }
1191  /* set sequence number */
1192  dev->buf[3] = dev->seq_nr++;
1193 
1194  /* return header size */
1195  return index;
1196 }
