57 static int init_base(uart_t uart, uint32_t baudrate)
58 {
59  if (uart != 0) {
60  return UART_NODEV;
61  }
62 
63  /* get the default UART for now -> TODO: enable for multiple devices */
64  msp_usart_t *dev = UART_BASE;
65 
66  /* power off and reset device */
67  uart_poweroff(uart);
68  dev->CTL = USART_CTL_SWRST;
69  /* configure to 8N1 and using the SMCLK*/
70  dev->CTL |= USART_CTL_CHAR;
71  dev->TCTL = (USART_TCTL_TXEPT | USART_TCTL_SSEL_SMCLK);
72  dev->RCTL = 0x00;
73  /* baudrate configuration */
74  uint16_t br = (uint16_t)(CLOCK_CMCLK / baudrate);
75  dev->BR0 = (uint8_t)br;
76  dev->BR1 = (uint8_t)(br >> 8);
77  /* TODO: calculate value for modulation register */
78  dev->MCTL = 0;
79  /* configure pins -> TODO: move into GPIO driver (once implemented) */
80  UART_PORT->SEL |= (UART_RX_PIN | UART_TX_PIN);
81  UART_PORT->OD |= UART_RX_PIN;
82  UART_PORT->OD &= ~(UART_TX_PIN);
83  UART_PORT->DIR |= UART_TX_PIN;
84  UART_PORT->DIR &= ~(UART_RX_PIN);
85  /* enable receiver and transmitter */
86  uart_poweron(uart);
87  /* and finally release the software reset bit */
88  dev->CTL &= ~(USART_CTL_SWRST);
89  return UART_OK;
90 }

149 static int init_base(uart_t uart, uint32_t baudrate)
150 {
151  if (uart != 0) {
152  return -1;
153  }
154 
155  /* get the default UART for now -> TODO: enable for multiple devices */
156  msp_usci_t *dev = UART_BASE;
157 
158  /* put device in reset mode while configuration is going on */
159  dev->ACTL1 = USCI_ACTL1_SWRST;
160  /* configure to UART, using SMCLK in 8N1 mode */
161  dev->ACTL1 |= USCI_ACTL1_SSEL_SMCLK;
162  dev->ACTL0 = 0;
163  dev->ASTAT = 0;
164  /* configure baudrate */
165  uint32_t base = ((CLOCK_CMCLK << 7) / baudrate);
166  uint16_t br = (uint16_t)(base >> 7);
167  uint8_t brs = (((base & 0x3f) * 8) >> 7);
168  dev->ABR0 = (uint8_t)br;
169  dev->ABR1 = (uint8_t)(br >> 8);
170  dev->AMCTL = (brs << USCI_AMCTL_BRS_SHIFT);
171  /* pin configuration -> TODO: move to GPIO driver once implemented */
172  UART_RX_PORT->SEL |= UART_RX_PIN;
173  UART_TX_PORT->SEL |= UART_TX_PIN;
174  UART_RX_PORT->DIR &= ~(UART_RX_PIN);
175  UART_TX_PORT->DIR |= UART_TX_PIN;
176  /* releasing the software reset bit starts the UART */
177  dev->ACTL1 &= ~(USCI_ACTL1_SWRST);
178  return 0;
179 }
