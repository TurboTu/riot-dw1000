58 void random_init(uint32_t s)
59 {
60  mt[0] = s;
61  for (int i = 1; i < N; ++i) {
62  mt[i] = 1812433253UL * (mt[i - 1] ^ (mt[i - 1] >> 30)) + i;
63  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
64  /* In the previous versions, MSBs of the seed affect */
65  /* only MSBs of the array mt[]. */
66  /* 2002/01/09 modified by Makoto Matsumoto */
67  }
68 
69  mti = N;
70 }

72 void random_init_by_array(uint32_t *init_key, int key_length)
73 {
74  random_init(19650218UL);
75  int i = 1;
76  int j = 0;
77  for (int k = N > key_length ? N : key_length; k; --k) {
78  mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1664525UL))
79  + init_key[j] + j; /* non linear */
80  ++i;
81  ++j;
82  if (i >= N) {
83  mt[0] = mt[N - 1];
84  i = 1;
85  }
86  if (j >= key_length) {
87  j = 0;
88  }
89  }
90  for (int k = N - 1; k; k--) {
91  mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1566083941UL)) - i; /* non linear */
92  i++;
93  if (i >= N) {
94  mt[0] = mt[N - 1];
95  i = 1;
96  }
97  }
98 
99  mt[0] = 0x80000000UL; /* MSB is 1; assuring non-zero initial array */
100 }

103 static void generate_numbers(void)
104 {
105  if (mti == MTI_UNINITIALIZED) {
106  /* if init_genrand() has not been called, a default initial seed is used */
107  random_init(5489UL);
108  }
109 
110  for (int k = 0; k < N; ++k) {
111  uint32_t y = (mt[k] & UPPER_MASK) | (mt[(k + 1) % N] & LOWER_MASK);
112  mt[k] = mt[(k + M) % N] ^ (y >> 1);
113  if (y & 1) {
114  mt[k] ^= MATRIX_A;
115  }
116  }
117 
118  mti = 0;
119 }

121 uint32_t random_uint32(void)
122 {
123  if (mti >= N) {
124  generate_numbers();
125  }
126 
127  uint32_t y = mt[mti++];
128  y ^= y >> 11;
129  y ^= (y << 7) & 0x9d2c5680UL;
130  y ^= (y << 15) & 0xefc60000UL;
131  y ^= y >> 18;
132  return y;
133 }
