53 void rtt_init(void)
54 {
55  RtcMode0 *rtcMode0 = &(RTT_DEV);
56 
57  /* Turn on power manager for RTC */
58  PM->APBAMASK.reg |= PM_APBAMASK_RTC;
59 
60  /* RTC uses External 32,768KHz Oscillator because OSC32K isn't accurate
61  * enough (p1075/1138). Also keep running in standby. */
62  SYSCTRL->XOSC32K.reg = SYSCTRL_XOSC32K_ONDEMAND |
63  SYSCTRL_XOSC32K_EN32K |
64  SYSCTRL_XOSC32K_XTALEN |
65  SYSCTRL_XOSC32K_STARTUP(6) |
66 #if RTT_RUNSTDBY
67  SYSCTRL_XOSC32K_RUNSTDBY |
68 #endif
69  SYSCTRL_XOSC32K_ENABLE;
70 
71  /* Setup clock GCLK2 with divider 1 */
72  GCLK->GENDIV.reg = GCLK_GENDIV_ID(2) | GCLK_GENDIV_DIV(1);
73  while (GCLK->STATUS.bit.SYNCBUSY) {}
74 
75  /* Enable GCLK2 with XOSC32K as source. Use divider without modification
76  * and keep running in standby. */
77  GCLK->GENCTRL.reg = GCLK_GENCTRL_ID(2) |
78  GCLK_GENCTRL_GENEN |
79 #if RTT_RUNSTDBY
80  GCLK_GENCTRL_RUNSTDBY |
81 #endif
82  GCLK_GENCTRL_SRC_XOSC32K;
83  while (GCLK->STATUS.bit.SYNCBUSY) {}
84 
85  /* Connect GCLK2 to RTC */
86  GCLK->CLKCTRL.reg = GCLK_CLKCTRL_GEN_GCLK2 |
87  GCLK_CLKCTRL_CLKEN |
88  GCLK_CLKCTRL_ID(RTC_GCLK_ID);
89  while (GCLK->STATUS.bit.SYNCBUSY) {}
90 
91  /* Disable RTC */
92  rtt_poweroff();
93 
94  /* Reset RTC */
95  rtcMode0->CTRL.bit.SWRST = 1;
96  while (rtcMode0->STATUS.bit.SYNCBUSY || rtcMode0->CTRL.bit.SWRST) {}
97 
98  /* Configure as 32bit counter with no prescaler and no clear on match compare */
99 #ifdef RIOT_TREK_DW1000_APP
100  rtcMode0->CTRL.reg = RTC_MODE0_CTRL_MODE_COUNT32 | RTC_MODE0_CTRL_PRESCALER_DIV1 | RTC_MODE0_CTRL_MATCHCLR;
101 #else
102  rtcMode0->CTRL.reg = RTC_MODE0_CTRL_MODE_COUNT32 | RTC_MODE0_CTRL_PRESCALER_DIV1;
103 #endif
104  while (rtcMode0->STATUS.bit.SYNCBUSY) {}
105 
106  /* Setup interrupt */
107  NVIC_EnableIRQ(RTT_IRQ);
108 
109  /* Enable RTC */
110  rtt_poweron();
111 }

192 void RTT_ISR(void)
193 {
194  RtcMode0 *rtcMode0 = &(RTT_DEV);
195  uint8_t status = rtcMode0->INTFLAG.reg;
196 
197  if ( (status & RTC_MODE0_INTFLAG_CMP0) && (rtt_callback.alarm_cb != NULL) ) {
198  rtt_callback.alarm_cb(rtt_callback.alarm_arg);
199  rtcMode0->INTFLAG.bit.CMP0 = 1;
200  }
201 
202  if ( (status & RTC_MODE0_INTFLAG_OVF) && (rtt_callback.overflow_cb != NULL) ) {
203  rtt_callback.overflow_cb(rtt_callback.overflow_arg);
204  rtcMode0->INTFLAG.bit.OVF = 1;
205  }
206 
207  cortexm_isr_end();
208 }
