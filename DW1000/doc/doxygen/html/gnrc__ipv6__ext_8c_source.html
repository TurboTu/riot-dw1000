34 static enum gnrc_ipv6_ext_demux_status _handle_rh(gnrc_pktsnip_t *current, gnrc_pktsnip_t *pkt)
35 {
36  gnrc_pktsnip_t *ipv6;
37  ipv6_ext_t *ext = (ipv6_ext_t *) current->data;
38  size_t current_offset;
39  ipv6_hdr_t *hdr;
40 
41  /* check seg_left early to avoid duplicating the packet */
42  if (((ipv6_ext_rh_t *)ext)->seg_left == 0) {
43  return GNRC_IPV6_EXT_OK;
44  }
45 
46  /* We cannot use `gnrc_pktbuf_start_write` since it duplicates only
47  the head. `ipv6_ext_rh_process` modifies the IPv6 header as well as
48  the extension header */
49 
50  current_offset = gnrc_pkt_len_upto(current->next, GNRC_NETTYPE_IPV6);
51 
52  if (pkt->users != 1) {
53  if ((ipv6 = gnrc_pktbuf_duplicate_upto(pkt, GNRC_NETTYPE_IPV6)) == NULL) {
54  DEBUG("ipv6: could not get a copy of pkt\n");
55  gnrc_pktbuf_release(pkt);
56  return GNRC_IPV6_EXT_ERROR;
57  }
58  pkt = ipv6;
59  hdr = ipv6->data;
60  ext = (ipv6_ext_t *)(((uint8_t *)ipv6->data) + current_offset);
61  }
62  else {
63  ipv6 = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_IPV6);
64  hdr = ipv6->data;
65  }
66 
67  switch (ipv6_ext_rh_process(hdr, (ipv6_ext_rh_t *)ext)) {
68  case EXT_RH_CODE_ERROR:
69  /* TODO: send ICMPv6 error codes */
70  gnrc_pktbuf_release(pkt);
71  return GNRC_IPV6_EXT_ERROR;
72 
73  case EXT_RH_CODE_FORWARD:
74  /* forward packet */
75  if (!gnrc_netapi_dispatch_receive(GNRC_NETTYPE_IPV6, GNRC_NETREG_DEMUX_CTX_ALL, pkt)) {
76  DEBUG("ipv6: could not dispatch packet to the ipv6 thread\n");
77  gnrc_pktbuf_release(pkt);
78  }
79  return GNRC_IPV6_EXT_FORWARDED;
80 
81  case EXT_RH_CODE_OK:
82  /* this should not happen since we checked seg_left early */
83  gnrc_pktbuf_release(pkt);
84  return GNRC_IPV6_EXT_ERROR;
85  }
86 
87  return GNRC_IPV6_EXT_OK;
88 }

100 static gnrc_pktsnip_t *_mark_extension_header(gnrc_pktsnip_t *current,
101  gnrc_pktsnip_t **pkt)
102 {
103  gnrc_pktsnip_t *tmp, *next;
104  ipv6_ext_t *ext = (ipv6_ext_t *) current->data;
105  size_t offset = ((ext->len * IPV6_EXT_LEN_UNIT) + IPV6_EXT_LEN_UNIT);
106 
107  if (current == *pkt) {
108  gnrc_pktsnip_t *ext_snip;
109  if ((tmp = gnrc_pktbuf_start_write(*pkt)) == NULL) {
110  DEBUG("ipv6: could not get a copy of pkt\n");
111  gnrc_pktbuf_release(*pkt);
112  return NULL;
113  }
114  *pkt = tmp;
115 
116  ext_snip = gnrc_pktbuf_mark(*pkt, offset, GNRC_NETTYPE_IPV6_EXT);
117  next = *pkt;
118 
119  if (ext_snip == NULL) {
120  gnrc_pktbuf_release(*pkt);
121  return NULL;
122  }
123  }
124  else {
125  /* the header is already marked */
126 
127  next = NULL;
128 
129  for (tmp = *pkt; tmp != NULL; tmp = tmp->next) {
130  if (tmp->next == current) {
131  next = tmp;
132  break;
133  }
134  }
135 
136  assert(next != NULL);
137  }
138 
139  return next;
140 }

142 static inline bool _has_valid_size(gnrc_pktsnip_t *pkt, uint8_t nh)
143 {
144  ipv6_ext_t *ext;
145 
146  if (pkt->size < sizeof(ipv6_ext_t)) {
147  return false;
148  }
149 
150  ext = pkt->data;
151 
152  switch (nh) {
153  case PROTNUM_IPV6_EXT_RH:
154  case PROTNUM_IPV6_EXT_HOPOPT:
155  case PROTNUM_IPV6_EXT_DST:
156  case PROTNUM_IPV6_EXT_FRAG:
157  case PROTNUM_IPV6_EXT_AH:
158  case PROTNUM_IPV6_EXT_ESP:
159  case PROTNUM_IPV6_EXT_MOB:
160  return ((ext->len * IPV6_EXT_LEN_UNIT) + IPV6_EXT_LEN_UNIT) <= pkt->size;
161 
162  default:
163  return true;
164  }
165 }

173 void gnrc_ipv6_ext_demux(kernel_pid_t iface,
174  gnrc_pktsnip_t *current,
175  gnrc_pktsnip_t *pkt,
176  uint8_t nh)
177 {
178  ipv6_ext_t *ext;
179 
180  while (true) {
181  ext = (ipv6_ext_t *) current->data;
182 
183  switch (nh) {
184  case PROTNUM_IPV6_EXT_RH:
185 #ifdef MODULE_GNRC_RPL_SRH
186  /* if current != pkt, size is already checked */
187  if (current == pkt && !_has_valid_size(pkt, nh)) {
188  DEBUG("ipv6_ext: invalid size\n");
189  gnrc_pktbuf_release(pkt);
190  return;
191  }
192 
193  switch (_handle_rh(current, pkt)) {
194  case GNRC_IPV6_EXT_OK:
195  /* We are the final destination. So proceeds like normal packet. */
196  nh = ext->nh;
197  DEBUG("ipv6_ext: next header = %" PRIu8 "\n", nh);
198 
199  if ((current = _mark_extension_header(current, &pkt)) == NULL) {
200  return;
201  }
202 
203  gnrc_ipv6_demux(iface, current, pkt, nh); /* demultiplex next header */
204 
205  return;
206 
207  case GNRC_IPV6_EXT_ERROR:
208  /* already released by _handle_rh, so no release here */
209  return;
210 
211  case GNRC_IPV6_EXT_FORWARDED:
212  /* the packet is forwarded and released. finish processing */
213  return;
214  }
215 
216  break;
217 #endif
218 
219  case PROTNUM_IPV6_EXT_HOPOPT:
220  case PROTNUM_IPV6_EXT_DST:
221  case PROTNUM_IPV6_EXT_FRAG:
222  case PROTNUM_IPV6_EXT_AH:
223  case PROTNUM_IPV6_EXT_ESP:
224  case PROTNUM_IPV6_EXT_MOB:
225  /* TODO: add handling of types */
226 
227  /* if current != pkt, size is already checked */
228  if (current == pkt && !_has_valid_size(pkt, nh)) {
229  DEBUG("ipv6_ext: invalid size\n");
230  gnrc_pktbuf_release(pkt);
231  return;
232  }
233 
234  nh = ext->nh;
235  DEBUG("ipv6_ext: next header = %" PRIu8 "\n", nh);
236 
237  if ((current = _mark_extension_header(current, &pkt)) == NULL) {
238  return;
239  }
240 
241  gnrc_pktbuf_hold(pkt, 1); /* don't release on next dispatch */
242  if (gnrc_netapi_dispatch_receive(GNRC_NETTYPE_IPV6, nh, pkt) == 0) {
243  gnrc_pktbuf_release(pkt);
244  }
245 
246  break;
247 
248  default:
249  gnrc_ipv6_demux(iface, current, pkt, nh); /* demultiplex next header */
250  return;
251  }
252  }
253 
254  assert(false); /* never reaches here */
255 }

257 gnrc_pktsnip_t *gnrc_ipv6_ext_build(gnrc_pktsnip_t *ipv6, gnrc_pktsnip_t *next,
258  uint8_t nh, size_t size)
259 {
260  gnrc_pktsnip_t *prev = NULL, *snip;
261  ipv6_ext_t *ext;
262 
263  if (size < IPV6_EXT_LEN_UNIT) {
264  return NULL;
265  }
266 
267  if (ipv6 != NULL) {
268  LL_SEARCH_SCALAR(ipv6, prev, next, next);
269 
270  if (prev == NULL) {
271  return NULL;
272  }
273  }
274 
275  snip = gnrc_pktbuf_add(next, NULL, size, GNRC_NETTYPE_IPV6);
276 
277  if (snip == NULL) {
278  return NULL;
279  }
280 
281  ext = snip->data;
282 
283  ext->nh = nh;
284 
285  if (size & 0x7) { /* not divisible by eight */
286  ext->len = (size / IPV6_EXT_LEN_UNIT);
287  }
288  else {
289  ext->len = (size / IPV6_EXT_LEN_UNIT) - 1;
290  }
291 
292  if (prev != NULL) {
293  prev->next = snip;
294  }
295 
296  return snip;
297 }
