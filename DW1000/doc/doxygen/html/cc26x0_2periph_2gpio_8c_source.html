33 int gpio_init(gpio_t pin, gpio_mode_t mode)
34 {
35  if ((unsigned int)pin > 31)

37 
38  /* enable GPIO clock */
39  PRCM->PDCTL0 |= PDCTL0_PERIPH_ON;
40  while(!(PRCM->PDSTAT0 & PDSTAT0_PERIPH_ON)) ;
41  PRCM->GPIOCLKGR |= 1;
42  PRCM->CLKLOADCTL |= CLKLOADCTL_LOAD;
43  while (!(PRCM->CLKLOADCTL & CLKLOADCTL_LOADDONE)) ;

52 }
