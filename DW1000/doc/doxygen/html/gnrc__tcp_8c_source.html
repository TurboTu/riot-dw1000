94 static int _gnrc_tcp_open(gnrc_tcp_tcb_t *tcb, const uint8_t *target_addr, uint16_t target_port,
95  const uint8_t *local_addr, uint16_t local_port, uint8_t passive)
96 {
97  msg_t msg; /* Message for incomming Messages */
98  msg_t connection_timeout_msg; /* Connection Timeout Message */
99  xtimer_t connection_timeout_timer; /* Connection Timeout Timer */
100  int8_t ret = 0; /* Return Value */
101 
102  /* Lock the tcb for this function call */
103  mutex_lock(&(tcb->function_lock));
104 
105  /* Connection is already connected: Return -EISCONN */
106  if (tcb->state != GNRC_TCP_FSM_STATE_CLOSED) {
107  mutex_unlock(&(tcb->function_lock));
108  return -EISCONN;
109  }
110 
111  /* Setup connection (common parts) */
112  msg_init_queue(tcb->msg_queue, GNRC_TCP_MSG_QUEUE_SIZE);
113  tcb->owner = thread_getpid();
114 
115  /* Setup passive connection */
116  if (passive){
117  /* Set Status Flags */
118  tcb->status |= GNRC_TCP_STATUS_PASSIVE;
119  if (local_addr == NULL) {
120  tcb->status |= GNRC_TCP_STATUS_ALLOW_ANY_ADDR;
121  }
122  /* If local address is specified: Copy it into tcb: only connections to this addr are ok */
123  else {
124  switch (tcb->address_family) {
125 #ifdef MODULE_GNRC_IPV6
126  case AF_INET6:
127  memcpy(tcb->local_addr, local_addr, sizeof(ipv6_addr_t));
128  break;
129 #endif
130  }
131  }
132  /* Assign Port to listen on, to tcb */
133  tcb->local_port = local_port;
134  }
135  /* Setup active connection */
136  else{
137  /* Copy Target Address and Port into tcb structure */
138  if (target_addr != NULL) {
139  switch (tcb->address_family) {
140  #ifdef MODULE_GNRC_IPV6
141  case AF_INET6:
142  memcpy(tcb->peer_addr, target_addr, sizeof(ipv6_addr_t));
143  break;
144  #endif
145  }
146  }
147  /* Copy Port Information, verfication happens in fsm */
148  tcb->local_port = local_port;
149  tcb->peer_port = target_port;
150 
151  /* Setup Timeout: If connection could not be established before */
152  /* the timer expired, the connection attempt failed */
153  connection_timeout_msg.type = MSG_TYPE_CONNECTION_TIMEOUT;
154  xtimer_set_msg(&connection_timeout_timer, GNRC_TCP_CONNECTION_TIMEOUT_DURATION,
155  &connection_timeout_msg, tcb->owner);
156  }
157 
158  /* Call FSM with Event: CALL_OPEN */
159  ret = _fsm(tcb, GNRC_TCP_FSM_EVENT_CALL_OPEN, NULL, NULL, 0);
160  if (ret == -ENOMEM) {
161  DEBUG("gnrc_tcp.c : gnrc_tcp_connect() : Out of receive buffers.\n");
162  } else if(ret == -EADDRINUSE) {
163  DEBUG("gnrc_tcp.c : gnrc_tcp_connect() : local_port is already in use.\n");
164  }
165 
166  /* Wait until a connection was established or closed */
167  while (ret >= 0 && tcb->state != GNRC_TCP_FSM_STATE_CLOSED
168  && tcb->state != GNRC_TCP_FSM_STATE_ESTABLISHED
169  && tcb->state != GNRC_TCP_FSM_STATE_CLOSE_WAIT
170  ) {
171  msg_receive(&msg);
172  switch (msg.type) {
173  case MSG_TYPE_CONNECTION_TIMEOUT:
174  DEBUG("gnrc_tcp.c : _gnrc_tcp_open() : CONNECTION_TIMEOUT\n");
175  _fsm(tcb, GNRC_TCP_FSM_EVENT_TIMEOUT_CONNECTION, NULL, NULL, 0);
176  ret = -ETIMEDOUT;
177  break;
178 
179  case MSG_TYPE_NOTIFY_USER:
180  DEBUG("gnrc_tcp.c : _gnrc_tcp_open() : MSG_TYPE_NOTIFY_USER\n");
181  break;
182 
183  default:
184  DEBUG("gnrc_tcp.c : _gnrc_tcp_open() : other message type\n");
185  }
186  }
187 
188  /* Cleanup */
189  xtimer_remove(&connection_timeout_timer);
190  if (tcb->state == GNRC_TCP_FSM_STATE_CLOSED && ret == 0) {
191  ret = -ECONNREFUSED;
192  }
193  tcb->owner = KERNEL_PID_UNDEF;
194  mutex_unlock(&(tcb->function_lock));
195  return ret;
196 }

304 ssize_t gnrc_tcp_send(gnrc_tcp_tcb_t *tcb, const void *data, const size_t len,
305  const uint32_t timeout_duration_us)
306 {
307  assert(tcb != NULL);
308  assert(data != NULL);
309 
310  msg_t msg; /* Message for incomming Messages */
311  msg_t connection_timeout_msg; /* Connection Timeout Message */
312  msg_t probe_timeout_msg; /* Probe Timeout Message */
313  msg_t user_timeout_msg; /* User Specified Timeout Message */
314  xtimer_t connection_timeout_timer; /* Connection Timeout Timer */
315  xtimer_t probe_timeout_timer; /* Probe Timeout Timer */
316  xtimer_t user_timeout_timer; /* User Specified Timeout Timer */
317  uint32_t probe_timeout_duration_us = 0; /* Probe Timeout Duration in microseconds */
318  ssize_t ret = 0; /* Return Value */
319  bool probing = false; /* True if this connection is probing */
320 
321  /* Lock the tcb for this function call */
322  mutex_lock(&(tcb->function_lock));
323 
324  /* Check if connection is in a valid state */
325  if (tcb->state != GNRC_TCP_FSM_STATE_ESTABLISHED
326  && tcb->state != GNRC_TCP_FSM_STATE_CLOSE_WAIT
327  ) {
328  mutex_unlock(&(tcb->function_lock));
329  return -ENOTCONN;
330  }
331 
332  /* Re-init message queue, take ownership. FSM can send Messages to this thread now */
333  msg_init_queue(tcb->msg_queue, GNRC_TCP_MSG_QUEUE_SIZE);
334  tcb->owner = thread_getpid();
335 
336  /* Setup Connection Timeout */
337  connection_timeout_msg.type = MSG_TYPE_CONNECTION_TIMEOUT;
338  xtimer_set_msg(&connection_timeout_timer, GNRC_TCP_CONNECTION_TIMEOUT_DURATION,
339  &connection_timeout_msg, tcb->owner);
340 
341  /* Setup User specified timeout if timeout_us is greater than zero */
342  if (timeout_duration_us > 0) {
343  user_timeout_msg.type = MSG_TYPE_USER_SPEC_TIMEOUT;
344  xtimer_set_msg(&user_timeout_timer, timeout_duration_us, &user_timeout_msg, tcb->owner);
345  }
346 
347  /* Loop until something was sent and acked */
348  while (ret == 0 || tcb->pkt_retransmit != NULL) {
349  /* Check if the connections state is closed. If so, a reset was received */
350  if (tcb->state == GNRC_TCP_FSM_STATE_CLOSED) {
351  ret = -ECONNRESET;
352  break;
353  }
354 
355  /* If the send window is closed: Setup Probing */
356  if (tcb->snd_wnd <= 0) {
357  /* If this is the first probe: Setup probing duration */
358  if (!probing) {
359  probing = true;
360  probe_timeout_duration_us = tcb->rto;
361  }
362  /* Initialize Probe Timer */
363  probe_timeout_msg.type = MSG_TYPE_PROBE_TIMEOUT;
364  xtimer_set_msg(&probe_timeout_timer, probe_timeout_duration_us, &probe_timeout_msg,
365  tcb->owner);
366  }
367 
368  /* Try to send data in case there nothing has been sent and we are not probing */
369  if (ret == 0 && !probing) {
370  ret = _fsm(tcb, GNRC_TCP_FSM_EVENT_CALL_SEND, NULL, (void *) data, len);
371  }
372 
373  /* Wait for responses */
374  msg_receive(&msg);
375  switch (msg.type) {
376  case MSG_TYPE_CONNECTION_TIMEOUT:
377  DEBUG("gnrc_tcp.c : gnrc_tcp_send() : CONNECTION_TIMEOUT\n");
378  _fsm(tcb, GNRC_TCP_FSM_EVENT_TIMEOUT_CONNECTION, NULL, NULL, 0);
379  ret = -ECONNABORTED;
380  break;
381 
382  case MSG_TYPE_USER_SPEC_TIMEOUT:
383  DEBUG("gnrc_tcp.c : gnrc_tcp_send() : USER_SPEC_TIMEOUT\n");
384  _fsm(tcb, GNRC_TCP_FSM_EVENT_CLEAR_RETRANSMIT, NULL, NULL, 0);
385  ret = -ETIMEDOUT;
386  break;
387 
388  case MSG_TYPE_PROBE_TIMEOUT:
389  DEBUG("gnrc_tcp.c : gnrc_tcp_send() : PROBE_TIMEOUT\n");
390  /* Send Probe */
391  _fsm(tcb, GNRC_TCP_FSM_EVENT_SEND_PROBE, NULL, NULL, 0);
392  probe_timeout_duration_us += probe_timeout_duration_us;
393 
394  /* Boundry check for time interval between probes */
395  if (probe_timeout_duration_us < GNRC_TCP_PROBE_LOWER_BOUND) {
396  probe_timeout_duration_us = GNRC_TCP_PROBE_LOWER_BOUND;
397  }
398  else if (probe_timeout_duration_us > GNRC_TCP_PROBE_UPPER_BOUND) {
399  probe_timeout_duration_us = GNRC_TCP_PROBE_UPPER_BOUND;
400  }
401  break;
402 
403  case MSG_TYPE_NOTIFY_USER:
404  DEBUG("gnrc_tcp.c : gnrc_tcp_send() : NOTIFY_USER\n");
405  /* Connection is alive: Reset Connection Timeout */
406  xtimer_set_msg(&connection_timeout_timer, GNRC_TCP_CONNECTION_TIMEOUT_DURATION,
407  &connection_timeout_msg, tcb->owner);
408 
409  /* If the window re-opened and we are probing: Stop it */
410  if (tcb->snd_wnd > 0 && probing) {
411  probing = false;
412  xtimer_remove(&probe_timeout_timer);
413  }
414  break;
415 
416  default:
417  DEBUG("gnrc_tcp.c : gnrc_tcp_send() : other message type\n");
418  }
419  }
420 
421  /* Cleanup */
422  xtimer_remove(&probe_timeout_timer);
423  xtimer_remove(&connection_timeout_timer);
424  xtimer_remove(&user_timeout_timer);
425  tcb->owner = KERNEL_PID_UNDEF;
426  mutex_unlock(&(tcb->function_lock));
427  return ret;
428 }

430 ssize_t gnrc_tcp_recv(gnrc_tcp_tcb_t *tcb, void *data, const size_t max_len,
431  const uint32_t timeout_duration_us)
432 {

434  assert(data != NULL);
435 
436  msg_t msg; /* Message for incomming Messages */
437  msg_t connection_timeout_msg; /* Connection Timeout Message */
438  msg_t user_timeout_msg; /* User Specified Timeout Message */
439  xtimer_t connection_timeout_timer; /* Connection Timeout Timer */
440  xtimer_t user_timeout_timer; /* User Specified Timeout Timer */
441  ssize_t ret = 0; /* Return Value */
442 
443  /* Lock the tcb for this function call */
444  mutex_lock(&(tcb->function_lock));
445 
446  /* Check if connection is in a valid state */
447  if (tcb->state != GNRC_TCP_FSM_STATE_ESTABLISHED
448  && tcb->state != GNRC_TCP_FSM_STATE_FIN_WAIT_1
449  && tcb->state != GNRC_TCP_FSM_STATE_FIN_WAIT_2
450  && tcb->state != GNRC_TCP_FSM_STATE_CLOSE_WAIT
451  ) {
452  mutex_unlock(&(tcb->function_lock));
453  return -ENOTCONN;
454  }
455 
456  /* If this call is non-blocking (timeout_duration_us == 0): Try to read data and return */
457  if (timeout_duration_us == 0) {
458  ret = _fsm(tcb, GNRC_TCP_FSM_EVENT_CALL_RECV, NULL, data, max_len);
459  if(ret == 0) {
460  ret = -EAGAIN;
461  }
462  mutex_unlock(&(tcb->function_lock));
463  return ret;
464  }
465 
466  /* If this call is blocking, setup messages and timers */
467  msg_init_queue(tcb->msg_queue, GNRC_TCP_MSG_QUEUE_SIZE);
468  tcb->owner = thread_getpid();
469 
470  /* Setup Connection Timeout */
471  connection_timeout_msg.type = MSG_TYPE_CONNECTION_TIMEOUT;
472  xtimer_set_msg(&connection_timeout_timer, GNRC_TCP_CONNECTION_TIMEOUT_DURATION,

477  xtimer_set_msg(&user_timeout_timer, timeout_duration_us, &user_timeout_msg, tcb->owner);
478 
479  /* Processing Loop */
480  while (ret == 0) {
481  /* Check if the connections state is closed. If so, a reset was received */
482  if (tcb->state == GNRC_TCP_FSM_STATE_CLOSED) {
483  ret = -ECONNRESET;
484  break;
485  }
486 
487  /* Try to read available data */
488  ret = _fsm(tcb, GNRC_TCP_FSM_EVENT_CALL_RECV, NULL, data, max_len);
489 
490  /* If there was no data: Wait for next packet or until the timeout fires */
491  if (ret <= 0) {
492  msg_receive(&msg);
493  switch (msg.type) {
494  case MSG_TYPE_CONNECTION_TIMEOUT:
495  DEBUG("gnrc_tcp.c : gnrc_tcp_recv() : CONNECTION_TIMEOUT\n");
496  _fsm(tcb, GNRC_TCP_FSM_EVENT_TIMEOUT_CONNECTION, NULL, NULL, 0);
497  ret = -ECONNABORTED;
498  break;
499 
500  case MSG_TYPE_USER_SPEC_TIMEOUT:
501  DEBUG("gnrc_tcp.c : gnrc_tcp_send() : USER_SPEC_TIMEOUT\n");
502  _fsm(tcb, GNRC_TCP_FSM_EVENT_CLEAR_RETRANSMIT, NULL, NULL, 0);
503  ret = -ETIMEDOUT;
504  break;
505 
506  case MSG_TYPE_NOTIFY_USER:
507  DEBUG("gnrc_tcp.c : gnrc_tcp_recv() : NOTIFY_USER\n");
508  break;
509 
510  default:
511  DEBUG("gnrc_tcp.c : gnrc_tcp_recv() : other message type\n");
512  }
513  }
514  }
515 
516  /* Cleanup */
517  xtimer_remove(&connection_timeout_timer);
518  xtimer_remove(&user_timeout_timer);
519  tcb->owner = KERNEL_PID_UNDEF;
520  mutex_unlock(&(tcb->function_lock));
