35 void trickle_interval(trickle_t *trickle)
36 {
37  uint32_t max_interval;
38 
39  trickle->I = trickle->I * 2;
40  max_interval = trickle->Imin << trickle->Imax;
41 
42  if ((trickle->I == 0) || (trickle->I > max_interval)) {
43  trickle->I = max_interval;
44  }
45 
46  DEBUG("trickle: I == %" PRIu32 "\n", trickle->I);
47 
48  trickle->c = 0;
49  trickle->t = (trickle->I / 2) + random_uint32_range(0, (trickle->I / 2) + 1);
