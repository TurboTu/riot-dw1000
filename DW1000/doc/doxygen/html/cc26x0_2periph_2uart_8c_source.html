46 int uart_init(uart_t uart, uint32_t baudrate, uart_rx_cb_t rx_cb, void *arg)
47 {
48  /* make sure the uart device is valid */
49  if (uart != 0) {
50  return UART_NODEV;
51  }
52 
53  /* enable clocks: serial power domain and UART */
54  PRCM->PDCTL0SERIAL = 1;
55  while (!(PRCM->PDSTAT0 & PDSTAT0_SERIAL_ON)) ;
56  uart_poweron(uart);
57 
58  /* disable and reset the UART */
59  UART->CTL = 0;
60 
61  /* save context */
62  ctx[0].rx_cb = rx_cb;
63  ctx[0].arg = arg;
64 
65  /* configure pins */
66  IOC->CFG[UART_TX_PIN] = IOCFG_PORTID_UART0_TX;
67  IOC->CFG[UART_RX_PIN] = (IOCFG_PORTID_UART0_RX | IOCFG_INPUT_ENABLE);
68 #if UART_HW_FLOW_CONTROL
69  IOC->CFG[UART_RTS_PIN] = IOCFG_PORTID_UART0_RTS;
70  IOC->CFG[UART_CTS_PIN] = (IOCFG_PORTID_UART0_CTS | IOCFG_INPUT_ENABLE);
71 #endif
72 
73  /* calculate baud-rate */
74  uint32_t tmp = (CLOCK_CORECLOCK * 4);
75  tmp += (baudrate / 2);
76  tmp /= baudrate;
77  UART->IBRD = (tmp >> FRAC_BITS);
78  UART->FBRD = (tmp & FRAC_MASK);
79 
80  /* configure line to 8N1 mode, LRCH must be written after IBRD and FBRD! */
81  UART->LCRH = UART_LCRH_WLEN_8;
82 
83  /* enable the RX interrupt */
84  UART->IMSC = UART_IMSC_RXIM;
85  NVIC_EnableIRQ(UART0_IRQN);
86 
87  /* start the UART */
88  UART->CTL = ENABLE_MASK;
89 
90  return UART_OK;
91 }
