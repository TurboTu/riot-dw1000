47 int cc2420_init(cc2420_t *dev)
48 {
49  uint16_t reg;
50  uint8_t addr[8];
51 
52  /* reset options and sequence number */
53  dev->netdev.seq = 0;
54  dev->netdev.flags = 0;
55 
56  /* set default address, channel, PAN ID, and TX power */
57  uuid_get(addr, sizeof(addr));
58  /* make sure we mark the address as non-multicast and not globally unique */
59  addr[0] &= ~(0x01);
60  addr[0] |= 0x02;
61  cc2420_set_addr_short(dev, &addr[6]);
62  cc2420_set_addr_long(dev, addr);
63  cc2420_set_pan(dev, CC2420_PANID_DEFAULT);
64  cc2420_set_chan(dev, CC2420_CHAN_DEFAULT);
65  cc2420_set_txpower(dev, CC2420_TXPOWER_DEFAULT);
66 
67  /* set default options */
68  cc2420_set_option(dev, CC2420_OPT_AUTOACK, true);
69  cc2420_set_option(dev, CC2420_OPT_CSMA, true);
70  cc2420_set_option(dev, CC2420_OPT_TELL_TX_START, true);
71  cc2420_set_option(dev, CC2420_OPT_TELL_RX_END, true);
72 
73 #ifdef MODULE_NETSTATS_L2
74  cc2420_set_option(dev, CC2420_OPT_TELL_RX_END, true);
75 #endif
76  /* set default protocol*/
77 #ifdef MODULE_GNRC_SIXLOWPAN
78  dev->netdev.proto = GNRC_NETTYPE_SIXLOWPAN;
79 #elif MODULE_GNRC
80  dev->netdev.proto = GNRC_NETTYPE_UNDEF;
81 #endif
82 
83  /* change default RX bandpass filter to 1.3uA (as recommended) */
84  reg = cc2420_reg_read(dev, CC2420_REG_RXCTRL1);
85  reg |= CC2420_RXCTRL1_RXBPF_LOCUR;
86  cc2420_reg_write(dev, CC2420_REG_RXCTRL1, reg);
87 
88  /* set the FIFOP threshold to maximum. */
89  cc2420_reg_write(dev, CC2420_REG_IOCFG0, CC2420_PKT_MAXLEN);
90 
91  /* turn off "Security enable" (page 33). */
92  reg = cc2420_reg_read(dev, CC2420_REG_SECCTRL0);
93  reg &= ~CC2420_SECCTRL0_RXFIFO_PROT;
94  cc2420_reg_write(dev, CC2420_REG_SECCTRL0, reg);
95 
96  /* set preamble length to 3 leading zero byte */
97  /* and turn on hardware CRC generation */
98  reg = cc2420_reg_read(dev, CC2420_REG_MDMCTRL0);
99  reg &= ~(CC2420_MDMCTRL0_PREAMBLE_M);
100  reg |= CC2420_MDMCTRL0_PREAMBLE_3B;
101  reg |= CC2420_MDMCTRL0_AUTOCRC;
102  cc2420_reg_write(dev, CC2420_REG_MDMCTRL0, reg);
103 
104  /* go into RX state */
105  cc2420_set_state(dev, CC2420_GOTO_RX);
106 
107  return 0;
108 }

128 size_t cc2420_tx_prepare(cc2420_t *dev, const struct iovec *data, unsigned count)
129 {
130  size_t pkt_len = 2; /* include the FCS (frame check sequence) */
131 
132  /* wait for any ongoing transmissions to be finished */
133  DEBUG("cc2420: tx_exec: waiting for any ongoing transmission\n");
134  while (cc2420_get_state(dev) & NETOPT_STATE_TX) {}
135 
136  /* get and check the length of the packet */
137  for (unsigned i = 0; i < count; i++) {
138  pkt_len += data[i].iov_len;
139  }
140  if (pkt_len >= CC2420_PKT_MAXLEN) {
141  DEBUG("cc2420: tx_prep: unable to send, pkt too large\n");
142  return 0;
143  }
144 
145  /* flush TX FIFO and write new packet to it */
146  cc2420_strobe(dev, CC2420_STROBE_FLUSHTX);
147  /* push packet length to TX FIFO */
148  cc2420_fifo_write(dev, (uint8_t *)&pkt_len, 1);
149  /* push packet to TX FIFO */
150  for (int i = 0; i < count; i++) {
151  cc2420_fifo_write(dev, (uint8_t *)data[i].iov_base, data[i].iov_len);
152  }
153  DEBUG("cc2420: tx_prep: loaded %i byte into the TX FIFO\n", (int)pkt_len);
154 
155  return pkt_len;
156 }

158 void cc2420_tx_exec(cc2420_t *dev)
159 {
160  /* trigger the transmission */
161  if (dev->options & CC2420_OPT_TELL_TX_START) {
162  dev->netdev.netdev.event_callback(&dev->netdev.netdev,
163  NETDEV2_EVENT_TX_STARTED);
164  }
165  DEBUG("cc2420: tx_exec: TX_START\n");
166  if (dev->options & CC2420_OPT_CSMA) {
167  DEBUG("cc2420: tx_exec: triggering TX with CCA\n");
168  cc2420_strobe(dev, CC2420_STROBE_TXONCCA);
169  }
170  else {
171  DEBUG("cc2420: tx_exec: triggering TX without CCA\n");
172  cc2420_strobe(dev, CC2420_STROBE_TXON);
173  }
174 }

176 int cc2420_rx(cc2420_t *dev, uint8_t *buf, size_t max_len, void *info)
177 {
178  uint8_t len;
179  uint8_t crc_corr;
180 
181  /* without a provided buffer, only readout the length and return it */
182  if (buf == NULL) {
183  /* get the packet length (without dropping it) (first byte in RX FIFO) */
184  cc2420_ram_read(dev, CC2420_RAM_RXFIFO, &len, 1);
185  len -= 2; /* subtract RSSI and FCF */
186  DEBUG("cc2420: recv: packet of length %i in RX FIFO\n", (int)len);
187  }
188  else {
189  /* read length byte */
190  cc2420_fifo_read(dev, &len, 1);
191  len -= 2; /* subtract RSSI and FCF */
192 
193  /* if a buffer is given, read (and drop) the packet */
194  len = (len > max_len) ? max_len : len;
195 
196  /* read fifo contents */
197  DEBUG("cc2420: recv: reading %i byte of the packet\n", (int)len);
198  cc2420_fifo_read(dev, buf, len);
199 
200  uint8_t rssi;
201  cc2420_fifo_read(dev, &rssi, 1);
202  DEBUG("cc2420: recv: RSSI is %i\n", (int)rssi);
203 
204  /* fetch and check if CRC_OK bit (MSB) is set */
205  cc2420_fifo_read(dev, &crc_corr, 1);
206  if (!(crc_corr & 0x80)) {
207  DEBUG("cc2420: recv: CRC_OK bit not set, dropping packet\n");
208  /* drop the corrupted frame from the RXFIFO */
209  len = 0;
210  }
211 
212  /* finally flush the FIFO */
213  cc2420_strobe(dev, CC2420_STROBE_FLUSHRX);
214  cc2420_strobe(dev, CC2420_STROBE_FLUSHRX);
215  }
216 
217  return (int)len;
218 }
