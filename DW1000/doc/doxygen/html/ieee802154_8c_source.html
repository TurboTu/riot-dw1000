24 size_t ieee802154_set_frame_hdr(uint8_t *buf, const uint8_t *src, size_t src_len,
25  const uint8_t *dst, size_t dst_len,
26  le_uint16_t src_pan, le_uint16_t dst_pan,
27  uint8_t flags, uint8_t seq)
28 {
29  int pos = 3; /* 0-1: FCS, 2: seq */
30  uint8_t type = (flags & IEEE802154_FCF_TYPE_MASK);
31 
32  buf[0] = flags;
33  buf[1] = IEEE802154_FCF_VERS_V1;
34 
35  if (((src_len != 0) && (src == NULL)) ||
36  ((dst_len != 0) && (dst == NULL))) {
37  return 0;
38  }
39 
40  /* Frame type is not beacon or ACK, but both address modes are zero */
41  if ((type != IEEE802154_FCF_TYPE_BEACON) && (type != IEEE802154_FCF_TYPE_ACK) &&
42  (src_len == 0) && (dst_len == 0)) {
43  return 0;
44  }
45 
46  /* set sequence number */
47  buf[2] = seq;
48 
49  if (dst_len != 0) {
50  buf[pos++] = dst_pan.u8[0];
51  buf[pos++] = dst_pan.u8[1];
52  }
53 
54  /* fill in destination address */
55  switch (dst_len) {
56  case 0:
57  buf[1] |= IEEE802154_FCF_DST_ADDR_VOID;
58  break;
59  case 2:
60  if (memcmp(dst, ieee802154_addr_bcast,
61  sizeof(ieee802154_addr_bcast)) == 0) {
62  /* do not request ACKs for broadcast address */
63  buf[0] &= ~IEEE802154_FCF_ACK_REQ;
64  }
65  buf[1] |= IEEE802154_FCF_DST_ADDR_SHORT;
66  buf[pos++] = dst[1];
67  buf[pos++] = dst[0];
68  break;
69  case 8:
70  buf[1] |= IEEE802154_FCF_DST_ADDR_LONG;
71  for (int i = 7; i >= 0; i--) {
72  buf[pos++] = dst[i];
73  }
74  break;
75  default:
76  return 0;
77  }
78 
79  /* fill in source PAN ID (if applicable) */
80  if (src_len != 0) {
81  if ((dst_len != 0) && (src_pan.u16 == dst_pan.u16)) {
82  buf[0] |= IEEE802154_FCF_PAN_COMP;
83  }
84  else {
85  /* (little endian) */
86  buf[pos++] = src_pan.u8[0];
87  buf[pos++] = src_pan.u8[1];
88  }
89  }
90 
91  /* fill in source address */
92  switch (src_len) {
93  case 0:
94  buf[1] |= IEEE802154_FCF_SRC_ADDR_VOID;
95  break;
96  case 2:
97  buf[1] |= IEEE802154_FCF_SRC_ADDR_SHORT;
98  buf[pos++] = src[1];
99  buf[pos++] = src[0];
100  break;
101  case 8:
102  buf[1] |= IEEE802154_FCF_SRC_ADDR_LONG;
103  for (int i = 7; i >= 0; i--) {
104  buf[pos++] = src[i];
105  }
106  break;
107  default:
108  return 0;
109  }
110 
111  /* return actual header length */
112  return pos;
113 }

115 size_t ieee802154_get_frame_hdr_len(const uint8_t *mhr)
116 {
117  /* TODO: include security header implications */
118  uint8_t tmp;
119  size_t len = 3; /* 2 byte FCF, 1 byte sequence number */
120 
121  /* figure out address sizes */
122  tmp = (mhr[1] & IEEE802154_FCF_DST_ADDR_MASK);
123  if (tmp == IEEE802154_FCF_DST_ADDR_SHORT) {
124  len += 4; /* 2 byte dst PAN + 2 byte dst short address */
125  }
126  else if (tmp == IEEE802154_FCF_DST_ADDR_LONG) {
127  len += 10; /* 2 byte dst PAN + 2 byte dst long address */
128  }
129  else if (tmp != IEEE802154_FCF_DST_ADDR_VOID) {
130  return 0;
131  }
132  else if (mhr[0] & IEEE802154_FCF_PAN_COMP) {
133  /* PAN compression, but no destination address => illegal state */
134  return 0;
135  }
136  tmp = (mhr[1] & IEEE802154_FCF_SRC_ADDR_MASK);
137  if (tmp == IEEE802154_FCF_SRC_ADDR_VOID) {
138  return len;
139  }
140  else {
141  if (!(mhr[0] & IEEE802154_FCF_PAN_COMP)) {
142  len += 2;
143  }
144  if (tmp == IEEE802154_FCF_SRC_ADDR_SHORT) {
145  return len + 2;
146  }
147  else if (tmp == IEEE802154_FCF_SRC_ADDR_LONG) {
148  return len + 8;
149  }
150  }
151  return 0;
152 }

154 int ieee802154_get_src(const uint8_t *mhr, uint8_t *src, le_uint16_t *src_pan)
155 {
156  int offset = 3; /* FCF: 0-1, Seq: 2 */
157  uint8_t tmp;
158 
159  assert((src != NULL) && (src_pan != NULL));
160  tmp = mhr[1] & IEEE802154_FCF_DST_ADDR_MASK;
161  if (tmp == IEEE802154_FCF_DST_ADDR_SHORT) {
162  if (mhr[0] & IEEE802154_FCF_PAN_COMP) {
163  src_pan->u8[0] = mhr[offset];
164  src_pan->u8[1] = mhr[offset + 1];
165  }
166  offset += 4;
167  }
168  else if (tmp == IEEE802154_FCF_DST_ADDR_LONG) {
169  if (mhr[0] & IEEE802154_FCF_PAN_COMP) {
170  src_pan->u8[0] = mhr[offset];
171  src_pan->u8[1] = mhr[offset + 1];
172  }
173  offset += 10;
174  }
175  else if (tmp != IEEE802154_FCF_DST_ADDR_VOID) {
176  return -EINVAL;
177  }
178  else if (mhr[0] & IEEE802154_FCF_PAN_COMP) {
179  /* PAN compression, but no destination address => illegal state */
180  return -EINVAL;
181  }
182 
183  tmp = mhr[1] & IEEE802154_FCF_SRC_ADDR_MASK;
184  if (tmp != IEEE802154_FCF_SRC_ADDR_VOID) {
185  if (!(mhr[0] & IEEE802154_FCF_PAN_COMP)) {
186  src_pan->u8[0] = mhr[offset++];
187  src_pan->u8[1] = mhr[offset++];
188  }
189  }
190  if (tmp == IEEE802154_FCF_SRC_ADDR_SHORT) {
191  /* read src PAN and address in little endian */
192  src[1] = mhr[offset++];
193  src[0] = mhr[offset++];
194  return 2;
195  }
196  else if (tmp == IEEE802154_FCF_SRC_ADDR_LONG) {
197  /* read src PAN and address in little endian */
198  for (int i = 7; i >= 0; i--) {
199  src[i] = mhr[offset++];
200  }
201  return 8;
202  }
203  else if (tmp != IEEE802154_FCF_SRC_ADDR_VOID) {
204  return -EINVAL;
205  }
206 
207  return 0;
208 }
