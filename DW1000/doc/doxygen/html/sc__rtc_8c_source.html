39 static int dow(int year, int month, int day)
40 {
41  /* calculate the day of week using Tøndering's algorithm */
42  static int t[] = {0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4};
43  year -= month < 3;
44  return (year + year/4 - year/100 + year/400 + t[month-1] + day) % 7;
