150 static uint8_t tcs37727_trim_gain(tcs37727_t *dev, int rawc)
151 {
152  uint8_t reg_again = 0;
153  int val_again = dev->again;
154 
155  if (dev->initialized == false) {
156  return -1;
157  }
158 
159  if (rawc < TCS37727_AG_THRESHOLD_LOW) {
160  switch (val_again) {
161  case 1:
162  reg_again = TCS37727_CONTROL_AGAIN_4;
163  val_again = 4;
164  break;
165 
166  case 4:
167  reg_again = TCS37727_CONTROL_AGAIN_16;
168  val_again = 16;
169  break;
170 
171  case 16:
172  reg_again = TCS37727_CONTROL_AGAIN_60;
173  val_again = 60;
174  break;
175 
176  case 60:
177  default:
178  return -1;
179  }
180  }
181  else if (rawc > TCS37727_AG_THRESHOLD_HIGH) {
182  switch (val_again) {
183  case 60:
184  reg_again = TCS37727_CONTROL_AGAIN_16;
185  val_again = 16;
186  break;
187 
188  case 16:
189  reg_again = TCS37727_CONTROL_AGAIN_4;
190  val_again = 4;
191  break;
192 
193  case 4:
194  reg_again = TCS37727_CONTROL_AGAIN_1;
195  val_again = 1;
196  break;
197 
198  case 1:
199  default:
200  return -1;
201  }
202  }
203  else {
204  return 0;
205  }
206 
207  i2c_acquire(dev->i2c);
208  uint8_t reg = 0;
209  if (i2c_read_reg(dev->i2c, dev->addr, TCS37727_CONTROL, &reg) != 1) {
210  i2c_release(dev->i2c);
211  return -2;
212  }
213  reg &= ~TCS37727_CONTROL_AGAIN_MASK;
214  reg |= reg_again;
215  if (i2c_write_reg(dev->i2c, dev->addr, TCS37727_CONTROL, reg) != 1) {
216  i2c_release(dev->i2c);
217  return -2;
218  }
219  i2c_release(dev->i2c);
220  dev->again = val_again;
221 
222  return 0;
223 }

225 int tcs37727_read(tcs37727_t *dev, tcs37727_data_t *data)
226 {
227  uint8_t buf[8];
228 
229  if (dev->initialized == false) {
230  return -1;
231  }
232 
233  i2c_acquire(dev->i2c);
234 
235  if (i2c_read_regs(dev->i2c, dev->addr,
236  (TCS37727_INC_TRANS | TCS37727_CDATA), buf, 8) != 8) {
237  i2c_release(dev->i2c);
238  return -1;
239  }
240 
241  i2c_release(dev->i2c);
242 
243  int32_t tmpc = ((uint16_t)buf[1] << 8) | buf[0];
244  int32_t tmpr = ((uint16_t)buf[3] << 8) | buf[2];
245  int32_t tmpg = ((uint16_t)buf[5] << 8) | buf[4];
246  int32_t tmpb = ((uint16_t)buf[7] << 8) | buf[6];
247  DEBUG("rawr: %"PRIi32" rawg %"PRIi32" rawb %"PRIi32" rawc %"PRIi32"\n",
248  tmpr, tmpg, tmpb, tmpc);
249 
250  /* Remove IR component as described in the DN40. */
251  int32_t ir = (tmpr + tmpg + tmpb - tmpc) >> 1;
252  tmpr -= ir;
253  tmpg -= ir;
254  tmpb -= ir;
255 
256  /* Color temperature calculation as described in the DN40. */
257  int32_t ct = (CT_COEF_IF * tmpb) / tmpr + CT_OFFSET_IF;
258 
259  /* Lux calculation as described in the DN40. */
260  int32_t gi = R_COEF_IF * tmpr + G_COEF_IF * tmpg + B_COEF_IF * tmpb;
261  /* TODO: add Glass Attenuation Factor GA compensation */
262  int32_t cpl = (dev->atime_us * dev->again) / DGF_IF;
263  int32_t lux = gi / cpl;
264 
265  /* Autogain */
266  tcs37727_trim_gain(dev, tmpc);
267 
268  data->red = (tmpr < 0) ? 0 : (tmpr * 1000) / cpl;
269  data->green = (tmpg < 0) ? 0 : (tmpg * 1000) / cpl;
270  data->blue = (tmpb < 0) ? 0 : (tmpb * 1000) / cpl;
271  data->clear = (tmpb < 0) ? 0 : (tmpc * 1000) / cpl;
272  data->lux = (lux < 0) ? 0 : lux;
273  data->ct = (ct < 0) ? 0 : ct;
274 
275  return 0;
276 }
