28 void gnrc_ndp_router_set_router(gnrc_ipv6_netif_t *iface, bool enable)
29 {
30  if (enable && !(iface->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER)) {
31  gnrc_ipv6_netif_add_addr(iface->pid, &ipv6_addr_all_routers_link_local,
32  128, GNRC_IPV6_NETIF_ADDR_FLAGS_NON_UNICAST);
33  mutex_lock(&iface->mutex);
34  iface->flags |= GNRC_IPV6_NETIF_FLAGS_ROUTER;
35  iface->max_adv_int = GNRC_IPV6_NETIF_DEFAULT_MAX_ADV_INT;
36  iface->min_adv_int = GNRC_IPV6_NETIF_DEFAULT_MIN_ADV_INT;
37  iface->adv_ltime = GNRC_IPV6_NETIF_DEFAULT_ROUTER_LTIME;
38  mutex_unlock(&iface->mutex);
39  gnrc_ndp_router_set_rtr_adv(iface, enable);
40  }
41  else if (!enable && (iface->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER)) {
42  gnrc_ipv6_netif_remove_addr(iface->pid, (ipv6_addr_t *)&ipv6_addr_all_routers_link_local);
43  gnrc_ndp_router_set_rtr_adv(iface, enable);
44  }

47 void gnrc_ndp_router_set_rtr_adv(gnrc_ipv6_netif_t *iface, bool enable)
48 {
49  if (enable && !(iface->flags & GNRC_IPV6_NETIF_FLAGS_RTR_ADV)) {
50  mutex_lock(&iface->mutex);
51  iface->flags |= GNRC_IPV6_NETIF_FLAGS_RTR_ADV;
52  iface->rtr_adv_count = GNRC_NDP_MAX_INIT_RTR_ADV_NUMOF;
53  mutex_unlock(&iface->mutex);
54  _send_rtr_adv(iface, NULL);
55  }
56  else if (!enable && (iface->flags & GNRC_IPV6_NETIF_FLAGS_RTR_ADV)) {
57  mutex_lock(&iface->mutex);
58  iface->rtr_adv_count = GNRC_NDP_MAX_FIN_RTR_ADV_NUMOF;
59  iface->flags &= ~GNRC_IPV6_NETIF_FLAGS_RTR_ADV;

66 #ifdef MODULE_GNRC_NDP_HOST

84 static void _send_rtr_adv(gnrc_ipv6_netif_t *iface, ipv6_addr_t *dst)
85 {
86  bool fin;
87  uint32_t interval;
88 
89  mutex_lock(&iface->mutex);
90  fin = (iface->adv_ltime == 0);
91  assert((iface->min_adv_int != 0) && (iface->max_adv_int != 0));
92  interval = random_uint32_range(iface->min_adv_int, iface->max_adv_int);
93  if (!fin && !((iface->flags | GNRC_IPV6_NETIF_FLAGS_ROUTER) &&
94  (iface->flags | GNRC_IPV6_NETIF_FLAGS_RTR_ADV))) {
95  DEBUG("ndp rtr: interface %" PRIkernel_pid " is not an advertising interface\n",
96  iface->pid);
97  return;
98  }
99  if (iface->rtr_adv_count > 1) { /* regard for off-by-one error */
100  iface->rtr_adv_count--;
101  if (!fin && (interval > GNRC_NDP_MAX_INIT_RTR_ADV_INT)) {
102  interval = GNRC_NDP_MAX_INIT_RTR_ADV_INT;
103  }
104  }
105  if (!fin || (iface->rtr_adv_count > 1)) { /* regard for off-by-one-error */
106  /* reset timer for next router advertisement */
107  xtimer_remove(&iface->rtr_adv_timer);
108  iface->rtr_adv_msg.type = GNRC_NDP_MSG_RTR_ADV_RETRANS;
109  iface->rtr_adv_msg.content.ptr = iface;
110  xtimer_set_msg(&iface->rtr_adv_timer, interval * US_PER_SEC, &iface->rtr_adv_msg,
111  gnrc_ipv6_pid);
112  }
113  mutex_unlock(&iface->mutex);
114  for (int i = 0; i < GNRC_IPV6_NETIF_ADDR_NUMOF; i++) {
115  ipv6_addr_t *src = &iface->addrs[i].addr;
116 
117  if (!ipv6_addr_is_unspecified(src) && ipv6_addr_is_link_local(src) &&
118  !gnrc_ipv6_netif_addr_is_non_unicast(src)) {
119  /* send one for every link local address (ideally there is only one) */
120  gnrc_ndp_internal_send_rtr_adv(iface->pid, src, dst, fin);
121  }
122  }
123 }
