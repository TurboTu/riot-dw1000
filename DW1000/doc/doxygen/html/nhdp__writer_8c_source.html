137 void nhdp_writer_add_addr(struct rfc5444_writer *wr, nhdp_addr_t *addr,
138  enum rfc5444_addrtlv_iana type, uint8_t value,
139  uint16_t metric_in, uint16_t metric_out)
140 {
141  struct rfc5444_writer_address *wr_addr;
142  struct netaddr n_addr;
143 
144  netaddr_from_nhdp_address(&n_addr, addr);
145 
146  switch (type) {
147  case RFC5444_ADDRTLV_LOCAL_IF:
148  /* Address is mandatory for every sub-msg (if message is splitted) */
149  wr_addr = rfc5444_writer_add_address(wr, _nhdp_message_content_provider.creator,
150  &n_addr, true);
151  break;
152 
153  case RFC5444_ADDRTLV_LINK_STATUS:
154  /* Fall through */
155 
156  case RFC5444_ADDRTLV_OTHER_NEIGHB:
157  /* Address only has to be included in one sub-msg (if message is splitted) */
158  wr_addr = rfc5444_writer_add_address(wr, _nhdp_message_content_provider.creator,
159  &n_addr, false);
160  break;
161 
162  default:
163  /* Unknown type, extend switch if other types are allowed */
164  return;
165  }
166 
167  rfc5444_writer_add_addrtlv(wr, wr_addr, &_nhdp_addrtlvs[type],
168  &value, sizeof(uint8_t), false);
169 
170  /* Add LINK_METRIC TLV if necessary */
171  if ((NHDP_METRIC == NHDP_LMT_DAT) && (metric_in != NHDP_METRIC_UNKNOWN)) {
172  switch(type) {
173  case RFC5444_ADDRTLV_LINK_STATUS:
174  metric_in |= NHDP_KD_LM_INC;
175  metric_in |= (metric_in == metric_out) ? NHDP_KD_LM_OUT : 0x00;
176  break;
177  case RFC5444_ADDRTLV_OTHER_NEIGHB:
178  metric_in |= NHDP_KD_NM_INC;
179  metric_in |= (metric_in == metric_out) ? NHDP_KD_NM_OUT : 0x00;
180  break;
181  default:
182  /* Other types are not used and therefore no address tlv is added */
183  return;
184  }
185  rfc5444_writer_add_addrtlv(wr, wr_addr, &_nhdp_addrtlvs[RFC5444_ADDRTLV_LINK_METRIC],
186  &metric_in, sizeof(metric_in), true);
187  }
188 }

210 static void _nhdp_add_message_tlvs_cb(struct rfc5444_writer *wr)
211 {
212  uint8_t validity_time, interval_time;
213  /* Convert validity time and interval time to milliseconds */
214  uint64_t val_tmp = (uint64_t) nhdp_wr_curr_if_entry->validity_time.seconds * MS_PER_SEC
215  + (nhdp_wr_curr_if_entry->validity_time.microseconds / 1000ULL);
216  uint64_t int_tmp = (uint64_t) nhdp_wr_curr_if_entry->hello_interval.seconds * MS_PER_SEC
217  + (nhdp_wr_curr_if_entry->hello_interval.microseconds / 1000ULL);
218 
219  /* Add validity time (mandatory) and interval time to msg */
220  validity_time = rfc5444_timetlv_encode(val_tmp);
221  interval_time = rfc5444_timetlv_encode(int_tmp);
222  rfc5444_writer_add_messagetlv(wr, RFC5444_MSGTLV_VALIDITY_TIME, 0, &validity_time,
223  sizeof(validity_time));
224  rfc5444_writer_add_messagetlv(wr, RFC5444_MSGTLV_INTERVAL_TIME, 0, &interval_time,
225  sizeof(interval_time));
226 }

254 static void netaddr_from_nhdp_address(struct netaddr *target, nhdp_addr_t *n_addr)
255 {
256  memset(target->_addr, 0, NETADDR_MAX_LENGTH);
257  memcpy(target->_addr, n_addr->addr, n_addr->addr_size);
258 
259  switch (n_addr->addr_type) {
260  case AF_CC110X:
261  target->_prefix_len = 8u;
262  target->_type = AF_CC110X;
263  break;
264 
265  case AF_INET:
266  target->_prefix_len = 32u;
267  target->_type = AF_INET;
268  break;
269 
270  case AF_INET6:
271  /* Fall-through */
272 
273  default:
274  target->_prefix_len = 128u;
275  target->_type = AF_INET6;
276  break;
277  }
