35 void rtc_init(void)
36 {
37  PCONP |= BIT9;
38  RTC_AMR = 0xff; /* disable alarm irq */
39  RTC_CIIR = 0; /* disable increase irq */
40  RTC_CISS = 0; /* disable subsecond irq */
41 
42  INTWAKE |= BIT15; /* rtc irq wakes up mcu from power down */
43 
44  RTC_CCR = CCR_CLKSRC; /* Clock from external 32 kHz Osc. */
45 
46  /* initialize clock with valid unix compatible values
47  * If RTC_YEAR contains an value larger unix time_t we must reset. */
48  if (RTC_YEAR > 2037) {
49  _rtc_set(0);
50  }
51 
52  DEBUG("%2lu.%2lu.%4lu %2lu:%2lu:%2lu\n",
53  RTC_DOM, RTC_MONTH, RTC_YEAR, RTC_HOUR, RTC_MIN, RTC_SEC);
54 }

148 void rtc_poweron(void)
149 {
150  PCONP |= BIT9;
151  RTC_ILR = (ILR_RTSSF | ILR_RTCCIF | ILR_RTCALF); /* clear interrupt flags */
152  RTC_CCR |= CCR_CLKEN; /* enable clock */
153  install_irq(RTC_INT, &RTC_IRQHandler, IRQP_RTC); /* install interrupt handler */
154 }

164 void RTC_IRQHandler(void)
165 {
166  if (RTC_ILR & ILR_RTSSF) {
167  /* sub second interrupt (does not need flag-clearing) */
168  }
169  else if (RTC_ILR & ILR_RTCCIF) {
170  /* counter increase interrupt */
171  }
172  else if (RTC_ILR & ILR_RTCALF) {
173  RTC_ILR |= ILR_RTCALF;
174  RTC_AMR = 0xff; /* disable alarm irq */
175  if (_cb) {
176  _cb(_cb_arg);
177  }
178  DEBUG("Ring\n");
179  }
180 
181  VICVectAddr = 0; /* Acknowledge Interrupt */
182 }
