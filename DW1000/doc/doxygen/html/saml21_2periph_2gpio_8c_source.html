90 int gpio_init(gpio_t pin, gpio_mode_t mode)
91 {
92  PortGroup* port = _port(pin);
93  int pin_pos = _pin_pos(pin);
94  int pin_mask = _pin_mask(pin);
95 
96  /* make sure pin mode is applicable */
97  if (mode > 0x7) {
98  return -1;
99  }
100 
101  /* set pin direction */
102  if (mode & 0x2) {
103  port->DIRCLR.reg = pin_mask;
104  }
105  else {
106  port->DIRSET.reg = pin_mask;
107  }
108 
109  /* configure the pin cfg and clear output register */
110  port->PINCFG[pin_pos].reg = (mode & MODE_PINCFG_MASK);
111  port->OUTCLR.reg = pin_mask;
112 
113  /* and set pull-up/pull-down if applicable */
114  if (mode == 0x7) {
115  port->OUTSET.reg = pin_mask;
116  }
117 
118  return 0;
119 }

121 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
122  gpio_cb_t cb, void *arg)
123 {
124  int exti = _exti(pin);
125 
126  /* make sure EIC channel is valid */
127  if (exti == -1) {
128  return -1;
129  }
130 
131  /* save callback */
132  gpio_config[exti].cb = cb;
133  gpio_config[exti].arg = arg;
134  /* configure pin as input and set MUX to peripheral function A */
135  gpio_init(pin, mode);
136  gpio_init_mux(pin, GPIO_MUX_A);
137  /* enable clocks for the EIC module */
138  MCLK->APBAMASK.reg |= MCLK_APBAMASK_EIC;
139  GCLK->PCHCTRL[EIC_GCLK_ID].reg = GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN_GCLK0;
140  /* configure the active flank */
141  EIC->CONFIG[exti >> 3].reg &= ~(0xf << ((exti & 0x7) * 4));
142  EIC->CONFIG[exti >> 3].reg |= (flank << ((exti & 0x7) * 4));
143  /* enable the global EIC interrupt */
144  NVIC_EnableIRQ(EIC_IRQn);
145  /*Enable pin interrupt */
146  EIC->INTFLAG.reg = (1 << exti);
147  EIC->INTENSET.reg = (1 << exti);
148  /* enable the EIC module*/
149  EIC->CTRLA.reg = EIC_CTRLA_ENABLE;
150  while (EIC->SYNCBUSY.reg & EIC_SYNCBUSY_ENABLE) {}
151  return 0;
152 }
