95 int gpio_init(gpio_t pin, gpio_mode_t mode)
96 {
97  const uint8_t port_num = _port_num(pin);
98  const uint32_t port_addr = _port_base[port_num];
99  const uint8_t pin_num = _pin_num(pin);
100  const uint32_t sysctl_port_base = _sysctl_port_base[port_num];
101  const unsigned long pin_bit = 1 << pin_num;
102 
103  DEBUG("Init GPIO: port %c, %d\n", 'A' + port_num, pin_num);
104  DEBUG("Sysctl %" PRIx32 "\n", sysctl_port_base);
105 
106  ROM_SysCtlPeripheralEnable(sysctl_port_base);
107 
108  HWREG(port_addr+GPIO_LOCK_R_OFF) = GPIO_LOCK_KEY;
109  HWREG(port_addr+GPIO_CR_R_OFF) |= pin_bit;
110  HWREG(port_addr+GPIO_DEN_R_OFF) |= pin_bit;
111  HWREG(port_addr+GPIO_LOCK_R_OFF) = 0;
112 
113  ROM_GPIOPadConfigSet(port_addr, pin_bit,
114  GPIO_STRENGTH_2MA, TYPE(mode));
115  ROM_GPIODirModeSet(port_addr, pin_bit, MODE(mode));
116 
117  return 0;
118 }

174  gpio_config[port_num][pin_num].arg = arg;

176  DEBUG("init int pin:%d, int num %d, port addr %" PRIx32 "\n",
177  pin_num, int_num, port_addr);
178 
179  ROM_GPIODirModeSet(port_addr, 1<<pin_num, GPIO_DIR_MODE_IN);
180  ROM_GPIOPadConfigSet(port_addr, 1<<pin_num,
181  GPIO_STRENGTH_2MA, TYPE(mode));
182 
183  ROM_IntMasterDisable();
184 
185  HWREG(icr_reg_addr) = pin_bit;
186 
187  ROM_GPIOIntTypeSet(port_addr, pin_bit, flank);
188 
189  HWREG(port_addr+GPIO_LOCK_R_OFF) = GPIO_LOCK_KEY;
190  HWREG(port_addr+GPIO_CR_R_OFF) |= pin_bit;
191  HWREG(port_addr+GPIO_DEN_R_OFF) |= pin_bit;
192  HWREG(port_addr+GPIO_LOCK_R_OFF) = 0;
