24 int gnrc_conn_recvfrom(conn_t *conn, void *data, size_t max_len, void *addr, size_t *addr_len,
25  uint16_t *port)
26 {
27  msg_t msg;
28  int timeout = 3;
29  while ((timeout--) > 0) {
30  gnrc_pktsnip_t *pkt, *l3hdr;
31  size_t size = 0;
32  msg_receive(&msg);
33  switch (msg.type) {
34  case GNRC_NETAPI_MSG_TYPE_RCV:
35  pkt = msg.content.ptr;
36  if (pkt->size > max_len) {
37  return -ENOMEM;
38  }
39  l3hdr = gnrc_pktsnip_search_type(pkt, conn->l3_type);
40  if (l3hdr == NULL) {
41  msg_send_to_self(&msg); /* requeue invalid messages */
42  continue;
43  }
44 #if defined(MODULE_CONN_UDP) || defined(MODULE_CONN_TCP)
45  if ((conn->l4_type != GNRC_NETTYPE_UNDEF) && (port != NULL)) {
46  gnrc_pktsnip_t *l4hdr;
47  l4hdr = gnrc_pktsnip_search_type(pkt, conn->l4_type);
48  if (l4hdr == NULL) {
49  msg_send_to_self(&msg); /* requeue invalid messages */
50  continue;
51  }
52  *port = byteorder_ntohs(((udp_hdr_t *)l4hdr->data)->src_port);
53  }
54 #endif /* defined(MODULE_CONN_UDP) */
55  if (addr != NULL) {
56  memcpy(addr, &((ipv6_hdr_t *)l3hdr->data)->src, sizeof(ipv6_addr_t));
57  *addr_len = sizeof(ipv6_addr_t);
58  }
59  memcpy(data, pkt->data, pkt->size);
60  size = pkt->size;
61  gnrc_pktbuf_release(pkt);
62  return (int)size;
63  default:
64  (void)port;
65  msg_send_to_self(&msg); /* requeue invalid messages */
66  break;
67  }
68  }
69  return -ETIMEDOUT;
70 }
