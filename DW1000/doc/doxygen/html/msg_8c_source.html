78 static int _msg_send(msg_t *m, kernel_pid_t target_pid, bool block, unsigned state)
79 {
80 #ifdef DEVELHELP
81  if (!pid_is_valid(target_pid)) {
82  DEBUG("msg_send(): target_pid is invalid, continuing anyways\n");
83  }
84 #endif /* DEVELHELP */
85 
86  thread_t *target = (thread_t*) sched_threads[target_pid];
87 
88  m->sender_pid = sched_active_pid;
89 
90  if (target == NULL) {
91  DEBUG("msg_send(): target thread does not exist\n");
92  irq_restore(state);
93  return -1;
94  }
95 
96  thread_t *me = (thread_t *) sched_active_thread;
97 
98  DEBUG("msg_send() %s:%i: Sending from %" PRIkernel_pid " to %" PRIkernel_pid
99  ". block=%i src->state=%i target->state=%i\n", RIOT_FILE_RELATIVE,
100  __LINE__, sched_active_pid, target_pid,
101  block, me->status, target->status);
102 
103  if (target->status != STATUS_RECEIVE_BLOCKED) {
104  DEBUG("msg_send() %s:%i: Target %" PRIkernel_pid " is not RECEIVE_BLOCKED.\n",
105  RIOT_FILE_RELATIVE, __LINE__, target_pid);
106 
107  if (queue_msg(target, m)) {
108  DEBUG("msg_send() %s:%i: Target %" PRIkernel_pid
109  " has a msg_queue. Queueing message.\n", RIOT_FILE_RELATIVE,
110  __LINE__, target_pid);
111  irq_restore(state);
112  if (me->status == STATUS_REPLY_BLOCKED) {
113  thread_yield_higher();
114  }
115  return 1;
116  }
117 
118  if (!block) {
119  DEBUG("msg_send: %" PRIkernel_pid ": Receiver not waiting, block=%u\n",
120  me->pid, block);
121  irq_restore(state);
122  return 0;
123  }
124 
125  DEBUG("msg_send: %" PRIkernel_pid ": going send blocked.\n",
126  me->pid);
127 
128  me->wait_data = (void*) m;
129 
130  int newstatus;
131 
132  if (me->status == STATUS_REPLY_BLOCKED) {
133  newstatus = STATUS_REPLY_BLOCKED;
134  }
135  else {
136  newstatus = STATUS_SEND_BLOCKED;
137  }
138 
139  sched_set_status((thread_t*) me, newstatus);
140 
141  thread_add_to_list(&(target->msg_waiters), me);
142 
143  irq_restore(state);
144  thread_yield_higher();
145 
146  DEBUG("msg_send: %" PRIkernel_pid ": Back from send block.\n",
147  me->pid);
148  }
149  else {
150  DEBUG("msg_send: %" PRIkernel_pid ": Direct msg copy from %"
151  PRIkernel_pid " to %" PRIkernel_pid ".\n",
152  me->pid, thread_getpid(), target_pid);
153  /* copy msg to target */
154  msg_t *target_message = (msg_t*) target->wait_data;
155  *target_message = *m;
156  sched_set_status(target, STATUS_PENDING);
157 
158  irq_restore(state);
159  thread_yield_higher();
160  }
161 
162  return 1;
163 }

211 int msg_send_receive(msg_t *m, msg_t *reply, kernel_pid_t target_pid)
212 {
213  assert(sched_active_pid != target_pid);
214  unsigned state = irq_disable();
215  thread_t *me = (thread_t*) sched_threads[sched_active_pid];
216  sched_set_status(me, STATUS_REPLY_BLOCKED);
217  me->wait_data = (void*) reply;
218 
219  /* we re-use (abuse) reply for sending, because wait_data might be
220  * overwritten if the target is not in RECEIVE_BLOCKED */
221  *reply = *m;
222  /* msg_send blocks until reply received */
223  return _msg_send(reply, target_pid, true, state);
224 }

280 static int _msg_receive(msg_t *m, int block)
281 {
282  unsigned state = irq_disable();
283  DEBUG("_msg_receive: %" PRIkernel_pid ": _msg_receive.\n",
284  sched_active_thread->pid);
285 
286  thread_t *me = (thread_t*) sched_threads[sched_active_pid];
287 
288  int queue_index = -1;
289 
290  if (me->msg_array) {
291  queue_index = cib_get(&(me->msg_queue));
292  }
293 
294  /* no message, fail */
295  if ((!block) && ((!me->msg_waiters.next) && (queue_index == -1))) {
296  irq_restore(state);
297  return -1;
298  }
299 
300  if (queue_index >= 0) {
301  DEBUG("_msg_receive: %" PRIkernel_pid ": _msg_receive(): We've got a queued message.\n",
302  sched_active_thread->pid);
303  *m = me->msg_array[queue_index];
304  }
305  else {
306  me->wait_data = (void *) m;
307  }
308 
309  list_node_t *next = list_remove_head(&me->msg_waiters);
310 
311  if (next == NULL) {
312  DEBUG("_msg_receive: %" PRIkernel_pid ": _msg_receive(): No thread in waiting list.\n",
313  sched_active_thread->pid);
314 
315  if (queue_index < 0) {
316  DEBUG("_msg_receive(): %" PRIkernel_pid ": No msg in queue. Going blocked.\n",
317  sched_active_thread->pid);
318  sched_set_status(me, STATUS_RECEIVE_BLOCKED);
319 
320  irq_restore(state);
321  thread_yield_higher();
322 
323  /* sender copied message */
324  }
325  else {
326  irq_restore(state);
327  }
328 
329  return 1;
330  }
331  else {
332  DEBUG("_msg_receive: %" PRIkernel_pid ": _msg_receive(): Waking up waiting thread.\n",
333  sched_active_thread->pid);
334 
335  thread_t *sender = container_of((clist_node_t*)next, thread_t, rq_entry);
336 
337  if (queue_index >= 0) {
338  /* We've already got a message from the queue. As there is a
339  * waiter, take it's message into the just freed queue space.
340  */
341  m = &(me->msg_array[cib_put(&(me->msg_queue))]);
342  }
343 
344  /* copy msg */
345  msg_t *sender_msg = (msg_t*) sender->wait_data;
346  *m = *sender_msg;
347 
348  /* remove sender from queue */
349  uint16_t sender_prio = THREAD_PRIORITY_IDLE;
350  if (sender->status != STATUS_REPLY_BLOCKED) {
351  sender->wait_data = NULL;
352  sched_set_status(sender, STATUS_PENDING);
353  sender_prio = sender->priority;
354  }
355 
356  irq_restore(state);
357  if (sender_prio < THREAD_PRIORITY_IDLE) {
358  sched_switch(sender_prio);
359  }
360  return 1;
361  }
362 
363  DEBUG("This should have never been reached!\n");
364 }
