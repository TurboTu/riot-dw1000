38 int _mutex_lock(mutex_t *mutex, int blocking)
39 {
40  unsigned irqstate = irq_disable();
41 
42  DEBUG("PID[%" PRIkernel_pid "]: Mutex in use.\n", sched_active_pid);
43 
44  if (mutex->queue.next == NULL) {
45  /* mutex is unlocked. */
46  mutex->queue.next = MUTEX_LOCKED;
47  DEBUG("PID[%" PRIkernel_pid "]: mutex_wait early out.\n",
48  sched_active_pid);
49  irq_restore(irqstate);
50  return 1;
51  }
52  else if (blocking) {
53  thread_t *me = (thread_t*)sched_active_thread;
54  DEBUG("PID[%" PRIkernel_pid "]: Adding node to mutex queue: prio: %"
55  PRIu32 "\n", sched_active_pid, (uint32_t)me->priority);
56  sched_set_status(me, STATUS_MUTEX_BLOCKED);
57  if (mutex->queue.next == MUTEX_LOCKED) {
58  mutex->queue.next = (list_node_t*)&me->rq_entry;
59  mutex->queue.next->next = NULL;
60  }
61  else {
62  thread_add_to_list(&mutex->queue, me);
63  }
64  irq_restore(irqstate);
65  thread_yield_higher();
66  /* We were woken up by scheduler. Waker removed us from queue.
67  * We have the mutex now. */
68  return 1;
69  }
70  else {
71  irq_restore(irqstate);
72  return 0;
73  }
74 }

77 {

79 
80  DEBUG("mutex_unlock(): queue.next: 0x%08x pid: %" PRIkernel_pid "\n",
81  (unsigned)mutex->queue.next, sched_active_pid);
82 
83  if (mutex->queue.next == NULL) {
84  /* the mutex was not locked */
85  irq_restore(irqstate);
86  return;
87  }
88 
89  if (mutex->queue.next == MUTEX_LOCKED) {
90  mutex->queue.next = NULL;
91  /* the mutex was locked and no thread was waiting for it */
92  irq_restore(irqstate);
93  return;
94  }

97

99
