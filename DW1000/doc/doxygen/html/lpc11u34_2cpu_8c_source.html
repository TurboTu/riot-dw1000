39 void clk_init(void)
40 {
41  volatile uint32_t i;
42 
43  /* Power-up System Osc */
44  LPC_SYSCON->PDRUNCFG &= ~(1 << 5);
45  LPC_SYSCON->SYSOSCCTRL = SYSOSCCTRL_Val;
46  for (i = 0; i < 200; i++) __NOP();
47 
48  /* Select PLL Input*/
49  LPC_SYSCON->SYSPLLCLKSEL = SYSPLLCLKSEL_Val;
50  /* Update clock source */
51  LPC_SYSCON->SYSPLLCLKUEN = 0x01;
52  /* Toggle update register */
53  LPC_SYSCON->SYSPLLCLKUEN = 0x00;
54  LPC_SYSCON->SYSPLLCLKUEN = 0x01;
55  /* Wait until updated */
56  while (!(LPC_SYSCON->SYSPLLCLKUEN & 0x01));
57  /* Main Clock is PLL Out */
58  LPC_SYSCON->SYSPLLCTRL = SYSPLLCTRL_Val;
59  /* Power-up SYSPLL */
60  LPC_SYSCON->PDRUNCFG &= ~(1 << 7);
61  /* Wait Until PLL Locked */
62  while (!(LPC_SYSCON->SYSPLLSTAT & 0x01));
63 
64  /* Select PLL Clock Output */
65  LPC_SYSCON->MAINCLKSEL = MAINCLKSEL_Val;
66  /* Update MCLK Clock Source */
67  LPC_SYSCON->MAINCLKUEN = 0x01;
68  /* Toggle Update Register */
69  LPC_SYSCON->MAINCLKUEN = 0x00;
70  LPC_SYSCON->MAINCLKUEN = 0x01;
71  /* Wait Until Updated */
72  while (!(LPC_SYSCON->MAINCLKUEN & 0x01));
73 
74  LPC_SYSCON->SYSAHBCLKDIV = SYSAHBCLKDIV_Val;
75 
76  /* Power-up USB PHY */
77  LPC_SYSCON->PDRUNCFG &= ~(1 << 10);
78  /* Power-up USB PLL */
79  LPC_SYSCON->PDRUNCFG &= ~(1 << 8);
