33 void cpuid_get(void *id)
34 {
35  uint32_t cpuid[4]; /* 128 bits sam3 unique indentifier */
36 
37  /*
38  Reading SAM3 Unique Identifier process (P 19.3.3.8):
39  - Send the Start Read unique Identifier command (STUI) by writing the Flash Command Register with the
40  STUI command.
41  - When the Unique Identifier is ready to be read, the FRDY bit in the Flash Programming Status Register
42  (EEFC_FSR) falls
43  - The Unique Identifier is located in the first 128 bits of the Flash memory mapping. So, at the address
44  0x400000-0x40008F.
45  - To stop the Unique Identifier mode, the user needs to send the Stop Read unique Identifier command (SPUI)
46  by writing the Flash Command Register with the SPUI command.
47  - When the Stop read Unique Identifier command (SPUI) has been performed, the FRDY bit in the Flash
48  Programming Status Register (EEFC_FSR) rises. If an interrupt was enabled by setting the FRDY bit in
49  EEFC_FMR, the interrupt line of the NVIC is activated.
50  */
51 
52  /* Send STUI */
53  EFC1->EEFC_FCR = EEFC_FCR_FKEY(EFC_KEY) | EFC_FCMD_STUI;
54 
55  /* Wait for FRDY bit falls */
56  while (EFC1->EEFC_FSR & EEFC_FSR_FRDY) {}
57 
58  /* Read UID */
59  cpuid[0] = *(uint32_t *)IFLASH1_ADDR;
60  cpuid[1] = *(uint32_t *)(IFLASH1_ADDR + 4);
61  cpuid[2] = *(uint32_t *)(IFLASH1_ADDR + 8);
62  cpuid[3] = *(uint32_t *)(IFLASH1_ADDR + 12);
63 
64  /* Send SPUI */
65  EFC1->EEFC_FCR = EEFC_FCR_FKEY(EFC_KEY) | EFC_FCMD_SPUI ;
66 
67  /* Wait for FRDY bit rises. */
68  while (0 == (EFC1->EEFC_FSR & EEFC_FSR_FRDY)) {}
69 
70  memcpy(id, (void*)cpuid, CPUID_LEN);
71 }
