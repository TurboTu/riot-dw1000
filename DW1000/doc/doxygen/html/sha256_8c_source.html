60 /*
61  * Encode a length len/4 vector of (uint32_t) into a length len vector of
62  * (unsigned char) in big-endian form. Assumes len is a multiple of 4.
63  */
64 static void be32enc_vect(void *dst_, const void *src_, size_t len)
65 {
66  if ((uintptr_t)dst_ % sizeof(uint32_t) == 0 &&
67  (uintptr_t)src_ % sizeof(uint32_t) == 0) {
68  uint32_t *dst = dst_;
69  const uint32_t *src = src_;
70  for (size_t i = 0; i < len / 4; i++) {
71  dst[i] = __builtin_bswap32(src[i]);
72  }
73  }
74  else {
75  uint8_t *dst = dst_;
76  const uint8_t *src = src_;
77  for (size_t i = 0; i < len; i += 4) {
78  dst[i] = src[i + 3];
79  dst[i + 1] = src[i + 2];
80  dst[i + 2] = src[i + 1];
81  dst[i + 3] = src[i];
82  }
83  }
84 }

123 /*
124  * SHA256 block compression function. The 256-bit state is transformed via
125  * the 512-bit input block to produce a new state.
126  */
127 static void sha256_transform(uint32_t *state, const unsigned char block[64])
128 {
129  uint32_t W[64];
130  uint32_t S[8];
131 
132  /* 1. Prepare message schedule W. */
133  be32dec_vect(W, block, 64);
134  for (int i = 16; i < 64; i++) {
135  W[i] = s1(W[i - 2]) + W[i - 7] + s0(W[i - 15]) + W[i - 16];
136  }
137 
138  /* 2. Initialize working variables. */
139  memcpy(S, state, 32);
140 
141  /* 3. Mix. */
142  for (int i = 0; i < 64; ++i) {
143  uint32_t e = S[(68 - i) % 8], f = S[(69 - i) % 8];
144  uint32_t g = S[(70 - i) % 8], h = S[(71 - i) % 8];
145  uint32_t t0 = h + S1(e) + Ch(e, f, g) + W[i] + K[i];
146 
147  uint32_t a = S[(64 - i) % 8], b = S[(65 - i) % 8];
148  uint32_t c = S[(66 - i) % 8], d = S[(67 - i) % 8];
149  uint32_t t1 = S0(a) + Maj(a, b, c);
150 
151  S[(67 - i) % 8] = d + t0;
152  S[(71 - i) % 8] = t0 + t1;
153  }
154 
155  /* 4. Mix local working variables into global state */
156  for (int i = 0; i < 8; i++) {
157  state[i] += S[i];
158  }
159 }

168 /* Add padding and terminating bit-count. */
169 static void sha256_pad(sha256_context_t *ctx)
170 {
171  /*
172  * Convert length to a vector of bytes -- we do this now rather
173  * than later because the length will change after we pad.
174  */
175  unsigned char len[8];
176 
177  be32enc_vect(len, ctx->count, 8);
178 
179  /* Add 1--64 bytes so that the resulting length is 56 mod 64 */
180  uint32_t r = (ctx->count[1] >> 3) & 0x3f;
181  uint32_t plen = (r < 56) ? (56 - r) : (120 - r);
182  sha256_update(ctx, PAD, (size_t) plen);
183 
184  /* Add the terminating bit-count */
185  sha256_update(ctx, len, 8);
186 }

205 /* Add bytes into the hash */
206 void sha256_update(sha256_context_t *ctx, const void *data, size_t len)
207 {
208  /* Number of bytes left in the buffer from previous updates */
209  uint32_t r = (ctx->count[1] >> 3) & 0x3f;
210 
211  /* Convert the length into a number of bits */
212  uint32_t bitlen1 = ((uint32_t) len) << 3;
213  uint32_t bitlen0 = ((uint32_t) len) >> 29;
214 
215  /* Update number of bits */
216  if ((ctx->count[1] += bitlen1) < bitlen1) {
217  ctx->count[0]++;
218  }
219 
220  ctx->count[0] += bitlen0;
221 
222  /* Handle the case where we don't need to perform any transforms */
223  if (len < 64 - r) {
224  memcpy(&ctx->buf[r], data, len);
225  return;
226  }
227 
228  /* Finish the current block */
229  const unsigned char *src = data;
230 
231  memcpy(&ctx->buf[r], src, 64 - r);
232  sha256_transform(ctx->state, ctx->buf);
233  src += 64 - r;
234  len -= 64 - r;
235 
236  /* Perform complete blocks */
237  while (len >= 64) {
238  sha256_transform(ctx->state, src);
239  src += 64;
240  len -= 64;
241  }
242 
243  /* Copy left over data into buffer */
244  memcpy(ctx->buf, src, len);
245 }

247 /*
248  * SHA-256 finalization. Pads the input data, exports the hash value,
249  * and clears the context state.
250  */
251 void sha256_final(sha256_context_t *ctx, void *dst)
252 {
253  /* Add padding */
254  sha256_pad(ctx);
255 
256  /* Write the hash */
257  be32enc_vect(dst, ctx->state, 32);
258 
259  /* Clear the context state */
260  memset((void *) ctx, 0, sizeof(*ctx));
261 }

279 const void *hmac_sha256(const void *key, size_t key_length,
280  const void *data, size_t len, void *digest)
281 {
282  unsigned char k[SHA256_INTERNAL_BLOCK_SIZE];
283 
284  memset((void *)k, 0x00, SHA256_INTERNAL_BLOCK_SIZE);
285 
286  if (key_length > SHA256_INTERNAL_BLOCK_SIZE) {
287  sha256(key, key_length, k);
288  }
289  else {
290  memcpy((void *)k, key, key_length);
291  }
292 
293  /*
294  * create the inner and outer keypads
295  * rising hamming distance enforcing i_* and o_* are distinct
296  * in at least one bit
297  */
298  unsigned char o_key_pad[SHA256_INTERNAL_BLOCK_SIZE];
299  unsigned char i_key_pad[SHA256_INTERNAL_BLOCK_SIZE];
300 
301  for (size_t i = 0; i < SHA256_INTERNAL_BLOCK_SIZE; ++i) {
302  o_key_pad[i] = 0x5c ^ k[i];
303  i_key_pad[i] = 0x36 ^ k[i];
304  }
305 
306  /*
307  * Create the inner hash
308  * tmp = hash(i_key_pad CONCAT message)
309  */
310  sha256_context_t c;
311  unsigned char tmp[SHA256_DIGEST_LENGTH];
312 
313  sha256_init(&c);
314  sha256_update(&c, i_key_pad, SHA256_INTERNAL_BLOCK_SIZE);
315  sha256_update(&c, data, len);
316  sha256_final(&c, tmp);
317 
318  static unsigned char m[SHA256_DIGEST_LENGTH];
319 
320  if (digest == NULL) {
321  digest = m;
322  }
323 
324  /*
325  * Create the outer hash
326  * result = hash(o_key_pad CONCAT tmp)
327  */
328  sha256_init(&c);
329  sha256_update(&c, o_key_pad, SHA256_INTERNAL_BLOCK_SIZE);
330  sha256_update(&c, tmp, SHA256_DIGEST_LENGTH);
331  sha256_final(&c, digest);
332 
333  return digest;
334 }

417  sha256_inplace(tmp_element);

441  sha256_inplace(tmp_element);
