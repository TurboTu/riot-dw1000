61 int gpio_init(gpio_t pin, gpio_mode_t mode)
62 {
63  /* check if mode is valid */
64  if (mode == MODE_NOTSUP) {
65  return -1;
66  }
67 
68 
69  /* disable any alternate function and any eventual interrupts */
70  gpio(pin)->IE &= ~pin_mask(pin);
71  gpio(pin)->AFSEL &= ~pin_mask(pin);
72  /* configure pull configuration */
73  IOC->OVER[pp_num(pin)] = mode;
74 
75  /* set pin direction */
76  if (mode == IOC_OVERRIDE_OE) {
77  gpio(pin)->DIR |= pin_mask(pin);
78  }
79  else {
80  gpio(pin)->DIR &= ~pin_mask(pin);
81  }
82  /* clear pin */
83  gpio(pin)->DATA &= ~pin_mask(pin);
84 
85  return 0;
86 }

88 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
89  gpio_cb_t cb, void *arg)
90 {
91  if (gpio_init(pin, mode) != 0) {
92  return -1;
93  }
94 
95  /* store the callback information for later: */
96  isr_ctx[port_num(pin)][pin_num(pin)].cb = cb;
97  isr_ctx[port_num(pin)][pin_num(pin)].arg = arg;
98 
99  /* enable power-up interrupts for this GPIO port: */
100  SYS_CTRL->IWE |= port_num(pin);
101 
102  /* configure the active flank(s) */
103  gpio(pin)->IS &= ~pin_mask(pin);
104  switch(flank) {
105  case GPIO_FALLING:
106  gpio(pin)->IBE &= ~pin_mask(pin);
107  gpio(pin)->IEV &= ~pin_mask(pin);
108  gpio(pin)->P_EDGE_CTRL |= (1 << pp_num(pin));
109  break;
110  case GPIO_RISING:
111  gpio(pin)->IBE &= ~pin_mask(pin);
112  gpio(pin)->IEV |= pin_mask(pin);
113  gpio(pin)->P_EDGE_CTRL &= (1 << pp_num(pin));
114  break;
115  case GPIO_BOTH:
116  gpio(pin)->IBE |= pin_mask(pin);
117  break;
118  default:
119  return -1;
120  }
121 
122  /* reset interrupt status */
123  gpio(pin)->IC = pin_mask(pin);
124  gpio(pin)->PI_IEN |= (1 << pp_num(pin));
125  /* enable global interrupt for the selected GPIO port */
126  NVIC_EnableIRQ(GPIO_PORT_A_IRQn + port_num(pin));
127  /* unmask pin interrupt */
128  gpio(pin)->IE |= pin_mask(pin);
129 
130  return 0;
131 }

174 {
