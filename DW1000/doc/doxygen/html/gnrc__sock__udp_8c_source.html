39 static bool _dyn_port_used(uint16_t port)
40 {
41 #ifdef MODULE_GNRC_SOCK_CHECK_REUSE
42  for (sock_udp_t *ptr = _udp_socks; ptr != NULL;
43  ptr = (sock_udp_t *)ptr->reg.next) {
44  bool spec_addr = false;
45  for (unsigned i = 0; i < sizeof(ptr->local.addr); i++) {
46  const uint8_t *const p = (uint8_t *)&ptr->local.addr;
47  if (p[i] != 0) {
48  spec_addr = true;
49  }
50  }
51  if (spec_addr) {
52  continue;
53  }
54  if (ptr->local.port == port) {
55  /* port already in use by another sock */
56  return true;
57  }
58  }
59 #else
60  (void) port;
61 #endif /* MODULE_GNRC_SOCK_CHECK_REUSE */
62  return false;
63 }

70 static uint16_t _get_dyn_port(sock_udp_t *sock)
71 {
72  uint16_t port;
73  unsigned count = GNRC_SOCK_DYN_PORTRANGE_NUM;
74  do {
75  port = GNRC_SOCK_DYN_PORTRANGE_MIN +
76  (_dyn_port_next * GNRC_SOCK_DYN_PORTRANGE_OFF) % GNRC_SOCK_DYN_PORTRANGE_NUM;
77  _dyn_port_next++;
78  if ((sock == NULL) || (sock->flags & SOCK_FLAGS_REUSE_EP) ||
79  !_dyn_port_used(port)) {
80  return port;
81  }
82  --count;
83  } while (count > 0);
84  return GNRC_SOCK_DYN_PORTRANGE_ERR;
85 }

87 int sock_udp_create(sock_udp_t *sock, const sock_udp_ep_t *local,
88  const sock_udp_ep_t *remote, uint16_t flags)
89 {
90  assert(sock);
91  assert(local == NULL || local->port != 0);
92  assert(remote == NULL || remote->port != 0);
93  if ((local != NULL) && (remote != NULL) &&
94  (local->netif != SOCK_ADDR_ANY_NETIF) &&
95  (remote->netif != SOCK_ADDR_ANY_NETIF) &&
96  (local->netif != remote->netif)) {
97  return -EINVAL;
98  }
99  memset(&sock->local, 0, sizeof(sock_udp_ep_t));
100  if (local != NULL) {
101 #ifdef MODULE_GNRC_SOCK_CHECK_REUSE
102  if (!(flags & SOCK_FLAGS_REUSE_EP)) {
103  for (sock_udp_t *ptr = _udp_socks; ptr != NULL;
104  ptr = (sock_udp_t *)ptr->reg.next) {
105  if (memcmp(&ptr->local, local, sizeof(sock_udp_ep_t)) == 0) {
106  return -EADDRINUSE;
107  }
108  }
109  }
110  /* prepend to current socks */
111  sock->reg.next = (gnrc_sock_reg_t *)_udp_socks;
112  _udp_socks = sock;
113 #endif
114  if (gnrc_af_not_supported(local->family)) {
115  return -EAFNOSUPPORT;
116  }
117  memcpy(&sock->local, local, sizeof(sock_udp_ep_t));
118  }
119  memset(&sock->remote, 0, sizeof(sock_udp_ep_t));
120  if (remote != NULL) {
121  if (gnrc_af_not_supported(remote->family)) {
122  return -EAFNOSUPPORT;
123  }
124  if (gnrc_ep_addr_any((const sock_ip_ep_t *)remote)) {
125  return -EINVAL;
126  }
127  memcpy(&sock->remote, remote, sizeof(sock_udp_ep_t));
128  }
129  if (local != NULL) {
130  /* listen only with local given */
131  gnrc_sock_create(&sock->reg, GNRC_NETTYPE_UDP, local->port);
132  }
133  sock->flags = flags;
134  return 0;
135 }

169 ssize_t sock_udp_recv(sock_udp_t *sock, void *data, size_t max_len,
170  uint32_t timeout, sock_udp_ep_t *remote)
171 {
172  gnrc_pktsnip_t *pkt, *udp;
173  udp_hdr_t *hdr;
174  sock_ip_ep_t tmp;
175  int res;
176 
177  assert((sock != NULL) && (data != NULL) && (max_len > 0));
178  if (sock->local.family == AF_UNSPEC) {
179  return -EADDRNOTAVAIL;
180  }
181  tmp.family = sock->local.family;
182  res = gnrc_sock_recv((gnrc_sock_reg_t *)sock, &pkt, timeout, &tmp);
183  if (res < 0) {
184  return res;
185  }
186  if (pkt->size > max_len) {
187  gnrc_pktbuf_release(pkt);
188  return -ENOBUFS;
189  }
190  udp = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_UDP);
191  assert(udp);
192  hdr = udp->data;
193  if (remote != NULL) {
194  /* return remote to possibly block if wrong remote */
195  memcpy(remote, &tmp, sizeof(tmp));
196  remote->port = byteorder_ntohs(hdr->src_port);
197  }
198  if ((sock->remote.family != AF_UNSPEC) && /* check remote end-point if set */
199  ((sock->remote.port != byteorder_ntohs(hdr->src_port)) ||
200  /* We only have IPv6 for now, so just comparing the whole end point
201  * should suffice */
202  ((memcmp(&sock->remote.addr, &ipv6_addr_unspecified,
203  sizeof(ipv6_addr_t)) != 0) &&
204  (memcmp(&sock->remote.addr, &tmp.addr, sizeof(ipv6_addr_t)) != 0)))) {
205  gnrc_pktbuf_release(pkt);
206  return -EPROTO;
207  }
208  memcpy(data, pkt->data, pkt->size);
209  gnrc_pktbuf_release(pkt);
210  return (int)pkt->size;
211 }

213 ssize_t sock_udp_send(sock_udp_t *sock, const void *data, size_t len,
214  const sock_udp_ep_t *remote)
215 {
216  int res;
217  gnrc_pktsnip_t *payload, *pkt;
218  uint16_t src_port = 0, dst_port;
219  sock_ip_ep_t local;
220  sock_ip_ep_t *rem;
221 
222  assert((sock != NULL) || (remote != NULL));
223  assert((len == 0) || (data != NULL)); /* (len != 0) => (data != NULL) */
224 
225  if (remote != NULL) {
226  if (remote->port == 0) {
227  return -EINVAL;
228  }
229  else if (gnrc_ep_addr_any((const sock_ip_ep_t *)remote)) {
230  return -EINVAL;
231  }
232  else if (gnrc_af_not_supported(remote->family)) {
233  return -EAFNOSUPPORT;
234  }
235  else if ((sock != NULL) &&
236  (sock->local.netif != SOCK_ADDR_ANY_NETIF) &&
237  (remote->netif != SOCK_ADDR_ANY_NETIF) &&
238  (sock->local.netif != remote->netif)) {
239  return -EINVAL;
240  }
241  }
242  else if (sock->remote.family == AF_UNSPEC) {
243  return -ENOTCONN;
244  }
245  /* compiler evaluates lazily so this isn't a redundundant check and cppcheck
246  * is being weird here anyways */
247  /* cppcheck-suppress nullPointerRedundantCheck */
248  /* cppcheck-suppress nullPointer */
249  if ((sock == NULL) || (sock->local.family == AF_UNSPEC)) {
250  /* no sock or sock currently unbound */
251  memset(&local, 0, sizeof(local));
252  if ((src_port = _get_dyn_port(sock)) == GNRC_SOCK_DYN_PORTRANGE_ERR) {
253  return -EINVAL;
254  }
255  if (sock != NULL) {
256  /* bind sock object implicitly */
257  sock->local.port = src_port;
258  if (remote == NULL) {
259  sock->local.family = sock->remote.family;
260  }
261  else {
262  sock->local.family = remote->family;
263  }
264  gnrc_sock_create(&sock->reg, GNRC_NETTYPE_UDP, src_port);
265 #ifdef MODULE_GNRC_SOCK_CHECK_REUSE
266  /* prepend to current socks */
267  sock->reg.next = (gnrc_sock_reg_t *)_udp_socks;
268  _udp_socks = sock;
269 #endif /* MODULE_GNRC_SOCK_CHECK_REUSE */
270  }
271  }
272  else {
273  src_port = sock->local.port;
274  memcpy(&local, &sock->local, sizeof(local));
275  }
276  /* sock can't be NULL at this point */
277  if (remote == NULL) {
278  rem = (sock_ip_ep_t *)&sock->remote;
279  dst_port = sock->remote.port;
280  }
281  else {
282  rem = (sock_ip_ep_t *)remote;
283  dst_port = remote->port;
284  }
285  /* check for matching address families in local and remote */
286  if (local.family == AF_UNSPEC) {
287  local.family = rem->family;
288  }
289  else if (local.family != rem->family) {
290  return -EINVAL;
291  }
292  /* generate payload and header snips */
293  payload = gnrc_pktbuf_add(NULL, (void *)data, len, GNRC_NETTYPE_UNDEF);
294  if (payload == NULL) {
295  return -ENOMEM;
296  }
297  pkt = gnrc_udp_hdr_build(payload, src_port, dst_port);
298  if (pkt == NULL) {
299  gnrc_pktbuf_release(payload);
300  return -ENOMEM;
301  }
302  res = gnrc_sock_send(pkt, &local, rem, PROTNUM_UDP);
303  if (res > 0) {
304  res -= sizeof(udp_hdr_t);
305  }
306  return res;
307 }
