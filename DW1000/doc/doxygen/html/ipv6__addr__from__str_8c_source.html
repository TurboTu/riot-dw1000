42 {

44 #ifdef MODULE_IPV4_ADDR
45  const char *curtok = addr;
46 #endif
47  uint32_t val = 0;
48  char ch;
49  uint8_t saw_xdigit = 0;
50  uint8_t i = 0;
51 
52  if ((result == NULL) || (addr == NULL)) {
53  return NULL;
54  }
55 
56  ipv6_addr_set_unspecified(result);
57 
58  /* Leading :: requires some special handling. */
59  if (*addr == ':') {
60  if (*++addr != ':') {
61  return NULL;
62  }
63  }
64 
65  while ((ch = *addr++) != '\0') {
66  const char *pch;
67  const char *xdigits;
68 
69  if ((pch = strchr((xdigits = HEX_L), ch)) == NULL) {
70  pch = strchr((xdigits = HEX_U), ch);
71  }
72 
73  if (pch != NULL) {
74  val <<= 4;
75  val |= (pch - xdigits);
76 
77  if (val > 0xffff) {
78  return NULL;
79  }
80 
81  saw_xdigit = 1;
82  continue;
83  }
84 
85  if (ch == ':') {
86 #ifdef MODULE_IPV4_ADDR
87  curtok = addr;
88 #endif
89 
90  if (!saw_xdigit) {
91  if (colonp != NULL) {
92  return NULL;
93  }
94 
95  colonp = &(result->u8[i]);
96  continue;
97  }
98 
99  if (i > sizeof(ipv6_addr_t)) {
100  return NULL;
101  }
102 
103  result->u8[i++] = (uint8_t)(val >> 8) & 0xff;
104  result->u8[i++] = (uint8_t) val & 0xff;
105  saw_xdigit = 0;
106  val = 0;
107  continue;
108  }
109 
110 #ifdef MODULE_IPV4_ADDR
111  if (ch == '.' && (i <= sizeof(ipv6_addr_t)) &&
112  ipv4_addr_from_str((ipv4_addr_t *)(&(result->u8[i])),
113  curtok) != NULL) {
114  i += sizeof(ipv4_addr_t);
115  saw_xdigit = 0;
116  break; /* '\0' was seen by ipv4_addr_from_str(). */
117  }
118 #endif
119 
120  return NULL;
121  }
122 
123  if (saw_xdigit) {
124  if (i + sizeof(uint16_t) > sizeof(ipv6_addr_t)) {
125  return NULL;
126  }
127 
128  result->u8[i++] = (uint8_t)(val >> 8) & 0xff;
129  result->u8[i++] = (uint8_t) val & 0xff;
130  }
131 
132  if (colonp != NULL) {
133  /*
134  * Since some memmove()'s erroneously fail to handle
135  * overlapping regions, we'll do the shift by hand.
136  */
137  const int32_t n = &(result->u8[i++]) - colonp;
138 
139  for (int32_t j = 1; j <= n; j++) {
140  result->u8[sizeof(ipv6_addr_t) - j] = colonp[n - j];
141  colonp[n - j] = 0;
142  }
143 
144  i = sizeof(ipv6_addr_t);
145  }
146 
147  if (i != sizeof(ipv6_addr_t)) {
148  return NULL;
149  }
150

152 }
