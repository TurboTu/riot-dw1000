71 void rbuf_add(gnrc_netif_hdr_t *netif_hdr, gnrc_pktsnip_t *pkt,
72  size_t frag_size, size_t offset)
73 {
74  rbuf_t *entry;
75  /* cppcheck is clearly wrong here */
76  /* cppcheck-suppress variableScope */
77  unsigned int data_offset = 0;
78  size_t original_size = frag_size;
79  sixlowpan_frag_t *frag = pkt->data;
80  rbuf_int_t *ptr;
81  uint8_t *data = ((uint8_t *)pkt->data) + sizeof(sixlowpan_frag_t);
82 
83  _rbuf_gc();
84  entry = _rbuf_get(gnrc_netif_hdr_get_src_addr(netif_hdr), netif_hdr->src_l2addr_len,
85  gnrc_netif_hdr_get_dst_addr(netif_hdr), netif_hdr->dst_l2addr_len,
86  byteorder_ntohs(frag->disp_size) & SIXLOWPAN_FRAG_SIZE_MASK,
87  byteorder_ntohs(frag->tag));
88 
89  if (entry == NULL) {
90  DEBUG("6lo rbuf: reassembly buffer full.\n");
91  return;
92  }
93 
94  ptr = entry->ints;
95 
96  /* dispatches in the first fragment are ignored */
97  if (offset == 0) {
98  if (data[0] == SIXLOWPAN_UNCOMP) {
99  data++; /* skip 6LoWPAN dispatch */
100  frag_size--;
101  }
102 #ifdef MODULE_GNRC_SIXLOWPAN_IPHC
103  else if (sixlowpan_iphc_is(data)) {
104  size_t iphc_len, nh_len = 0;
105  iphc_len = gnrc_sixlowpan_iphc_decode(&entry->pkt, pkt, entry->pkt->size,
106  sizeof(sixlowpan_frag_t), &nh_len);
107  if (iphc_len == 0) {
108  DEBUG("6lo rfrag: could not decode IPHC dispatch\n");
109  gnrc_pktbuf_release(entry->pkt);
110  _rbuf_rem(entry);
111  return;
112  }
113  data += iphc_len; /* take remaining data as data */
114  frag_size -= iphc_len; /* and reduce frag size by IPHC dispatch length */
115  /* but add IPv6 header + next header lengths */
116  frag_size += sizeof(ipv6_hdr_t) + nh_len;
117  /* start copying after IPv6 header and next headers */
118  data_offset += sizeof(ipv6_hdr_t) + nh_len;
119  }
120 #endif
121  }
122  else {
123  data++; /* FRAGN header is one byte longer (offset) */
124  }
125 
126  if ((offset + frag_size) > entry->pkt->size) {
127  DEBUG("6lo rfrag: fragment too big for resulting datagram, discarding datagram\n");
128  gnrc_pktbuf_release(entry->pkt);
129  _rbuf_rem(entry);
130  return;
131  }
132 
133  /* If the fragment overlaps another fragment and differs in either the size
134  * or the offset of the overlapped fragment, discards the datagram
135  * https://tools.ietf.org/html/rfc4944#section-5.3 */
136  while (ptr != NULL) {
137  if (_rbuf_int_overlap_partially(ptr, offset, offset + frag_size - 1)) {
138  DEBUG("6lo rfrag: overlapping intervals, discarding datagram\n");
139  gnrc_pktbuf_release(entry->pkt);
140  _rbuf_rem(entry);
141 
142  /* "A fresh reassembly may be commenced with the most recently
143  * received link fragment"
144  * https://tools.ietf.org/html/rfc4944#section-5.3 */
145  rbuf_add(netif_hdr, pkt, original_size, offset);
146 
147  return;
148  }
149 
150  ptr = ptr->next;
151  }
152 
153  if (_rbuf_update_ints(entry, offset, frag_size)) {
154  DEBUG("6lo rbuf: add fragment data\n");
155  entry->cur_size += (uint16_t)frag_size;
156  memcpy(((uint8_t *)entry->pkt->data) + offset + data_offset, data,
157  frag_size - data_offset);
158  }
159 
160  if (entry->cur_size == entry->pkt->size) {
161  gnrc_pktsnip_t *netif = gnrc_netif_hdr_build(entry->src, entry->src_len,
162  entry->dst, entry->dst_len);
163 
164  if (netif == NULL) {
165  DEBUG("6lo rbuf: error allocating netif header\n");
166  gnrc_pktbuf_release(entry->pkt);
167  _rbuf_rem(entry);
168  return;
169  }
170 
171  /* copy the transmit information of the latest fragment into the newly
172  * created header to have some link_layer information. The link_layer
173  * info of the previous fragments is discarded.
174  */
175  gnrc_netif_hdr_t *new_netif_hdr = netif->data;
176  new_netif_hdr->if_pid = netif_hdr->if_pid;
177  new_netif_hdr->flags = netif_hdr->flags;
178  new_netif_hdr->lqi = netif_hdr->lqi;
179  new_netif_hdr->rssi = netif_hdr->rssi;
180  LL_APPEND(entry->pkt, netif);
181 
182  if (!gnrc_netapi_dispatch_receive(GNRC_NETTYPE_IPV6, GNRC_NETREG_DEMUX_CTX_ALL,
183  entry->pkt)) {
184  DEBUG("6lo rbuf: No receivers for this packet found\n");
185  gnrc_pktbuf_release(entry->pkt);
186  }
187 
188  _rbuf_rem(entry);
189  }
190 }

192 static inline bool _rbuf_int_overlap_partially(rbuf_int_t *i, uint16_t start, uint16_t end)
193 {
194  /* start and ends are both inclusive, so using <= for both */
195  return ((i->start <= end) && (start <= i->end)) && /* overlaps */
196  ((start != i->start) || (end != i->end)); /* not identical */
197 }

210 static void _rbuf_rem(rbuf_t *entry)
211 {
212  while (entry->ints != NULL) {
213  rbuf_int_t *next = entry->ints->next;
214 
215  entry->ints->start = 0;
216  entry->ints->end = 0;
217  entry->ints->next = NULL;
218  entry->ints = next;
219  }
220 
221  entry->pkt = NULL;
222 }

224 static bool _rbuf_update_ints(rbuf_t *entry, uint16_t offset, size_t frag_size)
225 {
226  rbuf_int_t *new;
227  uint16_t end = (uint16_t)(offset + frag_size - 1);
228 
229  new = _rbuf_int_get_free();
230 
231  if (new == NULL) {
232  DEBUG("6lo rfrag: no space left in rbuf interval buffer.\n");
233  return false;
234  }
235 
236  new->start = offset;
237  new->end = end;
238 
239  DEBUG("6lo rfrag: add interval (%" PRIu16 ", %" PRIu16 ") to entry (%s, ",
240  new->start, new->end, gnrc_netif_addr_to_str(l2addr_str,
241  sizeof(l2addr_str), entry->src, entry->src_len));
242  DEBUG("%s, %u, %u)\n", gnrc_netif_addr_to_str(l2addr_str,
243  sizeof(l2addr_str), entry->dst, entry->dst_len),
244  (unsigned)entry->pkt->size, entry->tag);
245 
246  LL_PREPEND(entry->ints, new);
247 
248  return true;
249 }

251 static void _rbuf_gc(void)
252 {
253  uint32_t now_usec = xtimer_now_usec();
254  unsigned int i;
255 
256  for (i = 0; i < RBUF_SIZE; i++) {
257  /* since pkt occupies pktbuf, aggressivly collect garbage */
258  if ((rbuf[i].pkt != NULL) &&
259  ((now_usec - rbuf[i].arrival) > RBUF_TIMEOUT)) {
260  DEBUG("6lo rfrag: entry (%s, ", gnrc_netif_addr_to_str(l2addr_str,
261  sizeof(l2addr_str), rbuf[i].src, rbuf[i].src_len));
262  DEBUG("%s, %u, %u) timed out\n",
263  gnrc_netif_addr_to_str(l2addr_str, sizeof(l2addr_str), rbuf[i].dst,
264  rbuf[i].dst_len),
265  (unsigned)rbuf[i].pkt->size, rbuf[i].tag);
266 
267  gnrc_pktbuf_release(rbuf[i].pkt);
268  _rbuf_rem(&(rbuf[i]));
269  }

273 static rbuf_t *_rbuf_get(const void *src, size_t src_len,
274  const void *dst, size_t dst_len,
275  size_t size, uint16_t tag)
276 {
277  rbuf_t *res = NULL, *oldest = NULL;
278  uint32_t now_usec = xtimer_now_usec();
279 
280  for (unsigned int i = 0; i < RBUF_SIZE; i++) {
281  /* check first if entry already available */
282  if ((rbuf[i].pkt != NULL) && (rbuf[i].pkt->size == size) &&
283  (rbuf[i].tag == tag) && (rbuf[i].src_len == src_len) &&
284  (rbuf[i].dst_len == dst_len) &&
285  (memcmp(rbuf[i].src, src, src_len) == 0) &&
286  (memcmp(rbuf[i].dst, dst, dst_len) == 0)) {
287  DEBUG("6lo rfrag: entry %p (%s, ", (void *)(&rbuf[i]),
288  gnrc_netif_addr_to_str(l2addr_str, sizeof(l2addr_str),
289  rbuf[i].src, rbuf[i].src_len));
290  DEBUG("%s, %u, %u) found\n",
291  gnrc_netif_addr_to_str(l2addr_str, sizeof(l2addr_str),
292  rbuf[i].dst, rbuf[i].dst_len),
293  (unsigned)rbuf[i].pkt->size, rbuf[i].tag);
294  rbuf[i].arrival = now_usec;
295  return &(rbuf[i]);
296  }
297 
298  /* if there is a free spot: remember it */
299  if ((res == NULL) && (rbuf[i].pkt == NULL)) {
300  res = &(rbuf[i]);
301  }
302 
303  /* remember oldest slot */
304  /* note that xtimer_now will overflow in ~1.2 hours */
305  if ((oldest == NULL) || (oldest->arrival - rbuf[i].arrival < UINT32_MAX / 2)) {
306  oldest = &(rbuf[i]);
307  }
308  }
309 
310  /* entry not in buffer and no empty spot found */
311  if (res == NULL) {
312  assert(oldest != NULL);
313  assert(oldest->pkt != NULL); /* if oldest->pkt == NULL, res must not be NULL */
314  DEBUG("6lo rfrag: reassembly buffer full, remove oldest entry\n");
315  gnrc_pktbuf_release(oldest->pkt);
316  _rbuf_rem(oldest);
317  res = oldest;
318  }
319 
320  /* now we have an empty spot */
321 
322  res->pkt = gnrc_pktbuf_add(NULL, NULL, size, GNRC_NETTYPE_IPV6);
323  if (res->pkt == NULL) {
324  DEBUG("6lo rfrag: can not allocate reassembly buffer space.\n");
325  return NULL;
326  }
327 
328  *((uint64_t *)res->pkt->data) = 0; /* clean first few bytes for later
329  * look-ups */
330  res->arrival = now_usec;
331  memcpy(res->src, src, src_len);
332  memcpy(res->dst, dst, dst_len);
333  res->src_len = src_len;
334  res->dst_len = dst_len;
335  res->tag = tag;
336  res->cur_size = 0;
337 
338  DEBUG("6lo rfrag: entry %p (%s, ", (void *)res,
339  gnrc_netif_addr_to_str(l2addr_str, sizeof(l2addr_str), res->src,
340  res->src_len));
341  DEBUG("%s, %u, %u) created\n",
342  gnrc_netif_addr_to_str(l2addr_str, sizeof(l2addr_str), res->dst,
343  res->dst_len), (unsigned)res->pkt->size,
344  res->tag);
345 
346  return res;
347 }
