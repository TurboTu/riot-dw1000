59 static int init_base(uart_t uart, uint32_t baudrate)
60 {
61  switch (uart) {
62 #if UART_0_EN
63  case UART_0:
64  /* this implementation only supports 115200 baud */
65  if (baudrate != 115200) {
66  return UART_NOBAUD;
67  }
68 
69  /* select and configure the pin for RX */
70  UART_0_RX_PINSEL &= ~0x07;
71  UART_0_RX_PINSEL |= (UART_0_AF);
72  /* select and configure the pin for TX */
73  UART_0_TX_PINSEL &= ~0x07;
74  UART_0_TX_PINSEL |= (UART_0_AF);
75 
76  /* power on UART device and select peripheral clock */
77  UART_0_CLKEN();
78  UART_0_CLKSEL();
79  /* set mode to 8N1 and enable access to divisor latch */
80  UART_0_DEV->LCR = ((0x3 << 0) | (1 << 7)) | (3 << 4);
81  /* set baud rate registers (fixed for now) */
82  UART_0_DEV->DLM = 0;
83  UART_0_DEV->DLL = 17;
84  UART_0_DEV->FDR |= (8) | (15 << 4);
85  /* disable access to divisor latch */
86  UART_0_DEV->LCR &= ~0x80;
87  /* enable FIFOs */
88  UART_0_DEV->FCR = (1 << 0) | (1 << 1) | (1 << 2) | (2 << 6);
89  break;
90 #endif
91  default:
92  return UART_NODEV;
93  }
94 
95  return UART_OK;
