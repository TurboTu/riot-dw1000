65 bool cc2538_channel_clear(void)
66 {
67  if (RFCORE->XREG_FSMSTAT0bits.FSM_FFCTRL_STATE == FSM_STATE_IDLE) {
68  bool result;
69  cc2538_on();
70  RFCORE_WAIT_UNTIL(RFCORE->XREG_RSSISTATbits.RSSI_VALID);
71  result = BOOLEAN(RFCORE->XREG_FSMSTAT1bits.CCA);
72  cc2538_off();
73  return result;
74  }
75  else {
76  RFCORE_WAIT_UNTIL(RFCORE->XREG_RSSISTATbits.RSSI_VALID);
77  return BOOLEAN(RFCORE->XREG_FSMSTAT1bits.CCA);
78  }
79 }

81 void cc2538_init(void)
82 {
83  const init_pair_t *pair;
84 
85  for (pair = init_table; pair->reg_addr != NULL; pair++) {
86  *pair->reg_addr = pair->value;
87  }
88 
89  cc2538_set_tx_power(CC2538_RF_POWER_DEFAULT);
90  cc2538_set_chan(CC2538_RF_CHANNEL_DEFAULT);
91  cc2538_set_pan(CC2538_RF_PANID_DEFAULT);
92  cc2538_set_addr_long(cc2538_get_eui64_primary());
93 
94  /* Select the observable signals (maximum of three) */
95  RFCORE_XREG_RFC_OBS_CTRL0 = tx_active;
96  RFCORE_XREG_RFC_OBS_CTRL1 = rx_active;
97  RFCORE_XREG_RFC_OBS_CTRL2 = ffctrl_fifo;
98 
99  /* Select output pins for the three observable signals */
100 #ifdef BOARD_OPENMOTE_CC2538
101  CCTEST_OBSSEL0 = 0; /* PC0 = USB_SEL */
102  CCTEST_OBSSEL1 = 0; /* PC1 = N/C */
103  CCTEST_OBSSEL2 = 0; /* PC2 = N/C */
104  CCTEST_OBSSEL3 = 0; /* PC3 = USER_BUTTON */
105  CCTEST_OBSSEL4 = OBSSEL_EN | rfc_obs_sig0; /* PC4 = RED_LED */
106  CCTEST_OBSSEL5 = 0; /* PC5 = ORANGE_LED */
107  CCTEST_OBSSEL6 = OBSSEL_EN | rfc_obs_sig1; /* PC6 = YELLOW_LED */
108  CCTEST_OBSSEL7 = OBSSEL_EN | rfc_obs_sig2; /* PC7 = GREEN_LED */
109 #else
110  /* Assume BOARD_CC2538DK (or similar). */
111  CCTEST_OBSSEL0 = OBSSEL_EN | rfc_obs_sig0; /* PC0 = LED_1 (red) */
112  CCTEST_OBSSEL1 = OBSSEL_EN | rfc_obs_sig1; /* PC1 = LED_2 (yellow) */
113  CCTEST_OBSSEL2 = OBSSEL_EN | rfc_obs_sig2; /* PC2 = LED_3 (green) */
114  CCTEST_OBSSEL3 = 0; /* PC3 = LED_4 (red) */
115  CCTEST_OBSSEL4 = 0; /* PC4 = BTN_L */
116  CCTEST_OBSSEL5 = 0; /* PC5 = BTN_R */
117  CCTEST_OBSSEL6 = 0; /* PC6 = BTN_UP */
118  CCTEST_OBSSEL7 = 0; /* PC7 = BTN_DN */
119 #endif /* BOARD_OPENMOTE_CC2538 */
120 
121  if (SYS_CTRL->I_MAP) {
122  NVIC_SetPriority(RF_RXTX_ALT_IRQn, RADIO_IRQ_PRIO);
123  NVIC_EnableIRQ(RF_RXTX_ALT_IRQn);
124 
125  NVIC_SetPriority(RF_ERR_ALT_IRQn, RADIO_IRQ_PRIO);
126  NVIC_EnableIRQ(RF_ERR_ALT_IRQn);
127  }
128  else {
129  NVIC_SetPriority(RF_RXTX_IRQn, RADIO_IRQ_PRIO);
130  NVIC_EnableIRQ(RF_RXTX_IRQn);
131 
132  NVIC_SetPriority(RF_ERR_IRQn, RADIO_IRQ_PRIO);
133  NVIC_EnableIRQ(RF_ERR_IRQn);
134  }
135 
136  /* Flush the receive and transmit FIFOs */
137  RFCORE_SFR_RFST = ISFLUSHTX;
138  RFCORE_SFR_RFST = ISFLUSHRX;
139  /* Disable/filter l2 Acks */
140  RFCORE_XREG_FRMFILT1 &= ~CC2538_ACCEPT_FT_2_ACK;
141  cc2538_on();
142 }

149 void cc2538_off(void)
150 {
151  /* Wait for ongoing TX to complete (e.g. this could be an outgoing ACK) */
152  RFCORE_WAIT_UNTIL(RFCORE->XREG_FSMSTAT1bits.TX_ACTIVE == 0);
153 
154  /* Flush RX FIFO */
155  RFCORE_SFR_RFST = ISFLUSHRX;
156 
157  /* Don't turn off if we are off as this will trigger a Strobe Error */
158  if (RFCORE_XREG_RXENABLE != 0) {
159  RFCORE_SFR_RFST = ISRFOFF;
160  }
161 }

163 bool cc2538_on(void)
164 {
165  /* Flush RX FIFO */
166  RFCORE_SFR_RFST = ISFLUSHRX;
167 
168  /* Enable/calibrate RX */
169  RFCORE_SFR_RFST = ISRXON;
170 
171  return true;
172 }
