102 bool _check_packet(const ipv6_addr_t *src, const ipv6_addr_t *dst,
103  uint8_t proto, void *data, size_t data_len,
104  uint16_t netif)
105 {
106  gnrc_pktsnip_t *pkt, *ipv6;
107  ipv6_hdr_t *ipv6_hdr;
108  msg_t msg;
109 
110  msg_receive(&msg);
111  if (msg.type != GNRC_NETAPI_MSG_TYPE_SND) {
112  return false;
113  }
114  pkt = msg.content.ptr;
115  if (netif != SOCK_ADDR_ANY_NETIF) {
116  gnrc_netif_hdr_t *netif_hdr;
117 
118  if (pkt->type != GNRC_NETTYPE_NETIF) {
119  return _res(pkt, false);
120  }
121  netif_hdr = pkt->data;
122  if (netif_hdr->if_pid != netif) {
123  return _res(pkt, false);
124  }
125  ipv6 = pkt->next;
126  }
127  else {
128  ipv6 = pkt;
129  }
130  if (ipv6->type != GNRC_NETTYPE_IPV6) {
131  return _res(pkt, false);
132  }
133  ipv6_hdr = ipv6->data;
134  return _res(pkt, (memcmp(src, &ipv6_hdr->src, sizeof(ipv6_addr_t)) == 0) &&
135  (memcmp(dst, &ipv6_hdr->dst, sizeof(ipv6_addr_t)) == 0) &&
136  (ipv6_hdr->nh == proto) &&
137  (ipv6->next != NULL) &&
138  (data_len == ipv6->next->size) &&
139  (memcmp(data, ipv6->next->data, data_len) == 0));
140 }
