334 void od(const void *data, size_t data_len, uint8_t width, uint16_t flags)
335 {
336  char address_format[5];
337  uint8_t date_length = _length(flags);
338  char bytes_format[_log10(date_length) + 7];
339 
340  if (data_len == 0) {
341  return;
342  }
343 
344  _address_format(address_format, flags);
345  _bytes_format(bytes_format, flags);
346 
347  if (width == 0) {
348  width = OD_WIDTH_DEFAULT;
349  }
350 
351  if (width < date_length) {
352  width = 1;
353  }
354  else {
355  width = (width / date_length);
356  }
357 
358  if (data_len % date_length) {
359  data_len = (data_len / date_length) + 1;
360  }
361  else {
362  data_len = data_len / date_length;
363  }
364 
365  if ((flags & OD_FLAGS_ADDRESS_MASK) != OD_FLAGS_ADDRESS_NONE) {
366  printf(address_format, 0);
367  }
368 
369  for (size_t i = 0; i < data_len; i++) {
370  _print_date(data, i, bytes_format, date_length, flags);
371 
372  if ((((i + 1) % width) == 0) || i == (data_len - 1)) {
373  printf("\n");
374 
375  if (i != (data_len - 1)) {
376  if ((flags & OD_FLAGS_ADDRESS_MASK) != OD_FLAGS_ADDRESS_NONE) {
377  printf(address_format, date_length * (i + 1));
378  }
379  }
380  }
381  }
382 }
