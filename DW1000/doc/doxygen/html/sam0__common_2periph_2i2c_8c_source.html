71 int i2c_init_master(i2c_t dev, i2c_speed_t speed)
72 {
73  SercomI2cm *I2CSercom = 0;
74  gpio_t pin_scl = 0;
75  gpio_t pin_sda = 0;
76  gpio_mux_t mux;
77  uint32_t clock_source_speed = 0;
78  uint8_t sercom_gclk_id = 0;
79  uint8_t sercom_gclk_id_slow = 0;
80 
81  uint32_t timeout_counter = 0;
82  int32_t tmp_baud;
83 
84  switch (dev) {
85 #if I2C_0_EN
86  case I2C_0:
87  I2CSercom = &I2C_0_DEV;
88  pin_sda = I2C_0_SDA;
89  pin_scl = I2C_0_SCL;
90  mux = I2C_0_MUX;
91  clock_source_speed = CLOCK_CORECLOCK;
92  sercom_gclk_id = I2C_0_GCLK_ID;
93  sercom_gclk_id_slow = I2C_0_GCLK_ID_SLOW;
94  break;
95 #endif
96  default:
97  DEBUG("I2C FALSE VALUE\n");
98  return -1;
99  }
100 
101  /* DISABLE I2C MASTER */
102  i2c_poweroff(dev);
103 
104  /* Reset I2C */
105  I2CSercom->CTRLA.reg = SERCOM_I2CS_CTRLA_SWRST;
106  while (I2CSercom->SYNCBUSY.reg & SERCOM_I2CM_SYNCBUSY_MASK) {}
107 
108  /* Turn on power manager for sercom */
109  PM->APBCMASK.reg |= (PM_APBCMASK_SERCOM0 << (sercom_gclk_id - GCLK_CLKCTRL_ID_SERCOM0_CORE_Val));
110 
111  /* I2C using CLK GEN 0 */
112  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN |
113  GCLK_CLKCTRL_GEN_GCLK0 |
114  GCLK_CLKCTRL_ID(sercom_gclk_id));
115  while (GCLK->STATUS.bit.SYNCBUSY) {}
116 
117  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN |
118  GCLK_CLKCTRL_GEN_GCLK0 |
119  GCLK_CLKCTRL_ID(sercom_gclk_id_slow));
120  while (GCLK->STATUS.bit.SYNCBUSY) {}
121 
122 
123  /* Check if module is enabled. */
124  if (I2CSercom->CTRLA.reg & SERCOM_I2CM_CTRLA_ENABLE) {
125  DEBUG("STATUS_ERR_DENIED\n");
126  return -3;
127  }
128  /* Check if reset is in progress. */
129  if (I2CSercom->CTRLA.reg & SERCOM_I2CM_CTRLA_SWRST) {
130  DEBUG("STATUS_BUSY\n");
131  return -3;
132  }
133 
134  /************ SERCOM PAD0 - SDA and SERCOM PAD1 - SCL *************/
135  gpio_init_mux(pin_sda, mux);
136  gpio_init_mux(pin_scl, mux);
137 
138  /* I2C CONFIGURATION */
139  while (I2CSercom->SYNCBUSY.reg & SERCOM_I2CM_SYNCBUSY_MASK) {}
140 
141  /* Set sercom module to operate in I2C master mode. */
142  I2CSercom->CTRLA.reg = SERCOM_I2CM_CTRLA_MODE_I2C_MASTER;
143 
144  /* Enable Smart Mode (ACK is sent when DATA.DATA is read) */
145  I2CSercom->CTRLB.reg = SERCOM_I2CM_CTRLB_SMEN;
146 
147  /* Find and set baudrate. Read speed configuration. Set transfer
148  * speed: SERCOM_I2CM_CTRLA_SPEED(0): Standard-mode (Sm) up to 100
149  * kHz and Fast-mode (Fm) up to 400 kHz */
150  switch (speed) {
151  case I2C_SPEED_NORMAL:
152  tmp_baud = (int32_t)(((clock_source_speed + (2 * (100000)) - 1) / (2 * (100000))) - 5);
153  if (tmp_baud < 255 && tmp_baud > 0) {
154  I2CSercom->CTRLA.reg |= SERCOM_I2CM_CTRLA_SPEED(0);
155  I2CSercom->BAUD.reg = SERCOM_I2CM_BAUD_BAUD(tmp_baud);
156  }
157  break;
158  case I2C_SPEED_FAST:
159  tmp_baud = (int32_t)(((clock_source_speed + (2 * (400000)) - 1) / (2 * (400000))) - 5);
160  if (tmp_baud < 255 && tmp_baud > 0) {
161  I2CSercom->CTRLA.reg |= SERCOM_I2CM_CTRLA_SPEED(0);
162  I2CSercom->BAUD.reg = SERCOM_I2CM_BAUD_BAUD(tmp_baud);
163  }
164  break;
165  case I2C_SPEED_HIGH:
166  tmp_baud = (int32_t)(((clock_source_speed + (2 * (3400000)) - 1) / (2 * (3400000))) - 1);
167  if (tmp_baud < 255 && tmp_baud > 0) {
168  I2CSercom->CTRLA.reg |= SERCOM_I2CM_CTRLA_SPEED(2);
169  I2CSercom->BAUD.reg = SERCOM_I2CM_BAUD_HSBAUD(tmp_baud);
170  }
171  break;
172  default:
173  DEBUG("BAD BAUDRATE\n");
174  return -2;
175  }
176 
177  /* ENABLE I2C MASTER */
178  i2c_poweron(dev);
179 
180  /* Start timeout if bus state is unknown. */
181  while ((I2CSercom->STATUS.reg & SERCOM_I2CM_STATUS_BUSSTATE_Msk) == BUSSTATE_UNKNOWN) {
182  if (timeout_counter++ >= SAMD21_I2C_TIMEOUT) {
183  /* Timeout, force bus state to idle. */
184  I2CSercom->STATUS.reg = BUSSTATE_IDLE;
185  }
186  }
187  return 0;
188 }

381 static int _start(SercomI2cm *dev, uint8_t address, uint8_t rw_flag)
382 {
383  /* Wait for hardware module to sync */
384  DEBUG("Wait for device to be ready\n");
385  while (dev->SYNCBUSY.reg & SERCOM_I2CM_SYNCBUSY_MASK) {}
386 
387  /* Set action to ACK. */
388  dev->CTRLB.reg &= ~SERCOM_I2CM_CTRLB_ACKACT;
389 
390  /* Send Start | Address | Write/Read */
391  DEBUG("Generate start condition by sending address\n");
392  dev->ADDR.reg = (address << 1) | rw_flag | (0 << SERCOM_I2CM_ADDR_HS_Pos);
393 
394  /* Wait for response on bus. */
395  if (rw_flag == I2C_FLAG_READ) {
396  /* Some devices (e.g. SHT2x) can hold the bus while preparing the reply */
397  if (_wait_for_response(dev, 100 * SAMD21_I2C_TIMEOUT) < 0)
398  return -1;
399  }
400  else {
401  if (_wait_for_response(dev, SAMD21_I2C_TIMEOUT) < 0)
402  return -1;
403  }
404 
405  /* Check for address response error unless previous error is detected. */
406  /* Check for error and ignore bus-error; workaround for BUSSTATE
407  * stuck in BUSY */
408  if (dev->INTFLAG.reg & SERCOM_I2CM_INTFLAG_SB) {
409  /* Clear write interrupt flag */
410  dev->INTFLAG.reg = SERCOM_I2CM_INTFLAG_SB;
411  /* Check arbitration. */
412  if (dev->STATUS.reg & SERCOM_I2CM_STATUS_ARBLOST) {
413  DEBUG("STATUS_ERR_PACKET_COLLISION\n");
414  return -2;
415  }
416  }
417  /* Check that slave responded with ack. */
418  else if (dev->STATUS.reg & SERCOM_I2CM_STATUS_RXNACK) {
419  /* Slave busy. Issue ack and stop command. */
420  dev->CTRLB.reg |= SERCOM_I2CM_CTRLB_CMD(3);
421  DEBUG("STATUS_ERR_BAD_ADDRESS\n");
422  return -3;
423  }
424  return 0;
425 }

427 static inline int _write(SercomI2cm *dev, const uint8_t *data, int length)
428 {
429  uint16_t tmp_data_length = length;
430  uint16_t buffer_counter = 0;
431 
432  /* Write data buffer until the end. */
433  DEBUG("Looping through bytes\n");
434  while (tmp_data_length--) {
435  /* Check that bus ownership is not lost. */
436  if ((dev->STATUS.reg & SERCOM_I2CM_STATUS_BUSSTATE_Msk) != BUSSTATE_OWNER) {
437  DEBUG("STATUS_ERR_PACKET_COLLISION\n");
438  return -2;
439  }
440 
441  /* Wait for hardware module to sync */
442  while (dev->SYNCBUSY.reg & SERCOM_I2CM_SYNCBUSY_MASK) {}
443 
444  DEBUG("Written byte #%i to data reg, now waiting for DR to be empty again\n", buffer_counter);
445  dev->DATA.reg = data[buffer_counter++];
446 
447  /* Wait for response on bus. */
448  if (_wait_for_response(dev, SAMD21_I2C_TIMEOUT) < 0)
449  return -1;
450 
451  /* Check for NACK from slave. */
452  if (dev->STATUS.reg & SERCOM_I2CM_STATUS_RXNACK) {
453  DEBUG("STATUS_ERR_OVERFLOW\n");
454  return -4;
455  }
456  }
457  return 0;
458 }

460 static inline int _read(SercomI2cm *dev, uint8_t *data, int length)
461 {
462  uint8_t count = 0;
463 
464  /* Set action to ack. */
465  dev->CTRLB.reg &= ~SERCOM_I2CM_CTRLB_ACKACT;
466 
467  /* Read data buffer. */
468  while (count != length) {
469  /* Check that bus ownership is not lost. */
470  if ((dev->STATUS.reg & SERCOM_I2CM_STATUS_BUSSTATE_Msk) != BUSSTATE_OWNER) {
471  DEBUG("STATUS_ERR_PACKET_COLLISION\n");
472  return -2;
473  }
474 
475  /* Wait for hardware module to sync */
476  while (dev->SYNCBUSY.reg & SERCOM_I2CM_SYNCBUSY_MASK) {}
477  /* Save data to buffer. */
478  data[count] = dev->DATA.reg;
479 
480  /* Wait for response on bus. */
481  if (_wait_for_response(dev, SAMD21_I2C_TIMEOUT) < 0)
482  return -1;
483 
484  count++;
485  }
486  /* Send NACK before STOP */
487  dev->CTRLB.reg |= SERCOM_I2CM_CTRLB_ACKACT;
488  return 0;
489 }

503 {
