68 static int _init(netdev2_t *netdev)
69 {
70  at86rf2xx_t *dev = (at86rf2xx_t *)netdev;
71 
72  /* initialise GPIOs */
73  spi_init_cs(dev->params.spi, dev->params.cs_pin);
74  gpio_init(dev->params.sleep_pin, GPIO_OUT);
75  gpio_clear(dev->params.sleep_pin);
76  gpio_init(dev->params.reset_pin, GPIO_OUT);
77  gpio_set(dev->params.reset_pin);
78  gpio_init_int(dev->params.int_pin, GPIO_IN, GPIO_RISING, _irq_handler, dev);
79 
80  /* make sure device is not sleeping, so we can query part number */
81  at86rf2xx_assert_awake(dev);
82 
83  /* test if the SPI is set up correctly and the device is responding */
84  if (at86rf2xx_reg_read(dev, AT86RF2XX_REG__PART_NUM) !=
85  AT86RF2XX_PARTNUM) {
86  DEBUG("[at86rf2xx] error: unable to read correct part number\n");
87  return -1;
88  }
89 
90 #ifdef MODULE_NETSTATS_L2
91  memset(&netdev->stats, 0, sizeof(netstats_t));
92 #endif
93  /* reset device to default values and put it into RX state */
94  at86rf2xx_reset(dev);
95 
96  return 0;
97 }

99 static int _send(netdev2_t *netdev, const struct iovec *vector, unsigned count)
100 {
101  at86rf2xx_t *dev = (at86rf2xx_t *)netdev;
102  const struct iovec *ptr = vector;
103  size_t len = 0;
104 
105  at86rf2xx_tx_prepare(dev);
106 
107  /* load packet data into FIFO */
108  for (unsigned i = 0; i < count; i++, ptr++) {
109  /* current packet data + FCS too long */
110  if ((len + ptr->iov_len + 2) > AT86RF2XX_MAX_PKT_LENGTH) {
111  DEBUG("[at86rf2xx] error: packet too large (%u byte) to be send\n",
112  (unsigned)len + 2);
113  return -EOVERFLOW;
114  }
115 #ifdef MODULE_NETSTATS_L2
116  netdev->stats.tx_bytes += len;
117 #endif
118  len = at86rf2xx_tx_load(dev, ptr->iov_base, ptr->iov_len, len);
119  }
120 
121  /* send data out directly if pre-loading id disabled */
122  if (!(dev->netdev.flags & AT86RF2XX_OPT_PRELOADING)) {
123  at86rf2xx_tx_exec(dev);
124  }
125  /* return the number of bytes that were actually send out */
126  return (int)len;
127 }

129 static int _recv(netdev2_t *netdev, void *buf, size_t len, void *info)
130 {
131  at86rf2xx_t *dev = (at86rf2xx_t *)netdev;
132  uint8_t phr;
133  size_t pkt_len;
134 
135  /* frame buffer protection will be unlocked as soon as at86rf2xx_fb_stop()
136  * is called*/
137  at86rf2xx_fb_start(dev);
138 
139  /* get the size of the received packet */
140  at86rf2xx_fb_read(dev, &phr, 1);
141 
142  /* ignore MSB (refer p.80) and substract length of FCS field */
143  pkt_len = (phr & 0x7f) - 2;
144 
145  /* just return length when buf == NULL */
146  if (buf == NULL) {
147  at86rf2xx_fb_stop(dev);
148  return pkt_len;
149  }
150  /* not enough space in buf */
151  if (pkt_len > len) {
152  at86rf2xx_fb_stop(dev);
153  return -ENOBUFS;
154  }
155  #ifdef MODULE_NETSTATS_L2
156  netdev->stats.rx_count++;
157  netdev->stats.rx_bytes += pkt_len;
158  #endif
159  /* copy payload */
160  at86rf2xx_fb_read(dev, (uint8_t *)buf, pkt_len);
161 
162  /* Ignore FCS but advance fb read - we must give a temporary buffer here,
163  * as we are not allowed to issue SPI transfers without any buffer */
164  uint8_t tmp[2];
165  at86rf2xx_fb_read(dev, tmp, 2);
166  (void)tmp;
167 
168  if (info != NULL) {
169  netdev2_ieee802154_rx_info_t *radio_info = info;
170  at86rf2xx_fb_read(dev, &(radio_info->lqi), 1);
171 #ifndef MODULE_AT86RF231
172  at86rf2xx_fb_read(dev, &(radio_info->rssi), 1);
173  at86rf2xx_fb_stop(dev);
174 #else
175  at86rf2xx_fb_stop(dev);
176  radio_info->rssi = at86rf2xx_reg_read(dev, AT86RF2XX_REG__PHY_ED_LEVEL);
177 #endif
178  }
179  else {
180  at86rf2xx_fb_stop(dev);
181  }
182 
183  return pkt_len;
184 }

187 {

225 static int _get(netdev2_t *netdev, netopt_t opt, void *val, size_t max_len)
226 {
227  at86rf2xx_t *dev = (at86rf2xx_t *) netdev;
228 
229  if (netdev == NULL) {
230  return -ENODEV;
231  }
232 
233  /* getting these options doesn't require the transceiver to be responsive */
234  switch (opt) {
235  case NETOPT_CHANNEL_PAGE:
236  assert(max_len >= sizeof(uint16_t));
237  ((uint8_t *)val)[1] = 0;
238  ((uint8_t *)val)[0] = at86rf2xx_get_page(dev);
239  return sizeof(uint16_t);
240 
241  case NETOPT_MAX_PACKET_SIZE:
242  assert(max_len >= sizeof(int16_t));
243  *((uint16_t *)val) = AT86RF2XX_MAX_PKT_LENGTH - _MAX_MHR_OVERHEAD;
244  return sizeof(uint16_t);
245 
246  case NETOPT_STATE:
247  assert(max_len >= sizeof(netopt_state_t));
248  *((netopt_state_t *)val) = _get_state(dev);
249  return sizeof(netopt_state_t);
250 
251  case NETOPT_PRELOADING:
252  if (dev->netdev.flags & AT86RF2XX_OPT_PRELOADING) {
253  *((netopt_enable_t *)val) = NETOPT_ENABLE;
254  }
255  else {
256  *((netopt_enable_t *)val) = NETOPT_DISABLE;
257  }
258  return sizeof(netopt_enable_t);
259 
260  case NETOPT_PROMISCUOUSMODE:
261  if (dev->netdev.flags & AT86RF2XX_OPT_PROMISCUOUS) {
262  *((netopt_enable_t *)val) = NETOPT_ENABLE;
263  }
264  else {
265  *((netopt_enable_t *)val) = NETOPT_DISABLE;
266  }
267  return sizeof(netopt_enable_t);
268 
269  case NETOPT_RX_START_IRQ:
270  *((netopt_enable_t *)val) =
271  !!(dev->netdev.flags & AT86RF2XX_OPT_TELL_RX_START);
272  return sizeof(netopt_enable_t);
273 
274  case NETOPT_RX_END_IRQ:
275  *((netopt_enable_t *)val) =
276  !!(dev->netdev.flags & AT86RF2XX_OPT_TELL_RX_END);
277  return sizeof(netopt_enable_t);
278 
279  case NETOPT_TX_START_IRQ:
280  *((netopt_enable_t *)val) =
281  !!(dev->netdev.flags & AT86RF2XX_OPT_TELL_TX_START);
282  return sizeof(netopt_enable_t);
283 
284  case NETOPT_TX_END_IRQ:
285  *((netopt_enable_t *)val) =
286  !!(dev->netdev.flags & AT86RF2XX_OPT_TELL_TX_END);
287  return sizeof(netopt_enable_t);
288 
289  case NETOPT_CSMA:
290  *((netopt_enable_t *)val) =
291  !!(dev->netdev.flags & AT86RF2XX_OPT_CSMA);
292  return sizeof(netopt_enable_t);
293 
294  default:
295  /* Can still be handled in second switch */
296  break;
297  }
298 
299  int res;
300 
301  if (((res = netdev2_ieee802154_get((netdev2_ieee802154_t *)netdev, opt, val,
302  max_len)) >= 0) || (res != -ENOTSUP)) {
303  return res;
304  }
305 
306  uint8_t old_state = at86rf2xx_get_status(dev);
307  res = 0;
308 
309  /* temporarily wake up if sleeping */
310  if (old_state == AT86RF2XX_STATE_SLEEP) {
311  at86rf2xx_assert_awake(dev);
312  }
313 
314  /* these options require the transceiver to be not sleeping*/
315  switch (opt) {
316  case NETOPT_TX_POWER:
317  assert(max_len >= sizeof(int16_t));
318  *((uint16_t *)val) = at86rf2xx_get_txpower(dev);
319  res = sizeof(uint16_t);
320  break;
321 
322  case NETOPT_RETRANS:
323  assert(max_len >= sizeof(uint8_t));
324  *((uint8_t *)val) = at86rf2xx_get_max_retries(dev);
325  res = sizeof(uint8_t);
326  break;
327 
328  case NETOPT_CSMA_RETRIES:
329  assert(max_len >= sizeof(uint8_t));
330  *((uint8_t *)val) = at86rf2xx_get_csma_max_retries(dev);
331  res = sizeof(uint8_t);
332  break;
333 
334  case NETOPT_CCA_THRESHOLD:
335  assert(max_len >= sizeof(int8_t));
336  *((int8_t *)val) = at86rf2xx_get_cca_threshold(dev);
337  res = sizeof(int8_t);
338  break;
339 
340  default:
341  res = -ENOTSUP;
342  }
343 
344  /* go back to sleep if were sleeping */
345  if (old_state == AT86RF2XX_STATE_SLEEP) {
346  at86rf2xx_set_state(dev, AT86RF2XX_STATE_SLEEP);
347  }
348 
349  return res;
350 }

352 static int _set(netdev2_t *netdev, netopt_t opt, void *val, size_t len)
353 {
354  at86rf2xx_t *dev = (at86rf2xx_t *) netdev;
355  uint8_t old_state = at86rf2xx_get_status(dev);
356  int res = -ENOTSUP;
357 
358  if (dev == NULL) {
359  return -ENODEV;
360  }
361 
362  /* temporarily wake up if sleeping */
363  if (old_state == AT86RF2XX_STATE_SLEEP) {
364  at86rf2xx_assert_awake(dev);
365  }
366 
367  switch (opt) {
368  case NETOPT_ADDRESS:
369  assert(len <= sizeof(uint16_t));
370  at86rf2xx_set_addr_short(dev, *((uint16_t *)val));
371  /* don't set res to set netdev2_ieee802154_t::short_addr */
372  break;
373  case NETOPT_ADDRESS_LONG:
374  assert(len <= sizeof(uint64_t));
375  at86rf2xx_set_addr_long(dev, *((uint64_t *)val));
376  /* don't set res to set netdev2_ieee802154_t::long_addr */
377  break;
378  case NETOPT_NID:
379  assert(len <= sizeof(uint16_t));
380  at86rf2xx_set_pan(dev, *((uint16_t *)val));
381  /* don't set res to set netdev2_ieee802154_t::pan */
382  break;
383  case NETOPT_CHANNEL:
384  assert(len != sizeof(uint8_t));
385  uint8_t chan = ((uint8_t *)val)[0];
386  if (chan < AT86RF2XX_MIN_CHANNEL ||
387  chan > AT86RF2XX_MAX_CHANNEL) {
388  res = -EINVAL;
389  break;
390  }
391  at86rf2xx_set_chan(dev, chan);
392  /* don't set res to set netdev2_ieee802154_t::chan */
393  break;
394 
395  case NETOPT_CHANNEL_PAGE:
396  assert(len != sizeof(uint8_t));
397  uint8_t page = ((uint8_t *)val)[0];
398 #ifdef MODULE_AT86RF212B
399  if ((page != 0) && (page != 2)) {
400  res = -EINVAL;
401  }
402  else {
403  at86rf2xx_set_page(dev, page);
404  res = sizeof(uint8_t);
405  }
406 #else
407  /* rf23x only supports page 0, no need to configure anything in the driver. */
408  if (page != 0) {
409  res = -EINVAL;
410  }
411  else {
412  res = sizeof(uint8_t);
413  }
414 #endif
415  break;
416 
417  case NETOPT_TX_POWER:
418  assert(len <= sizeof(int16_t));
419  at86rf2xx_set_txpower(dev, *((int16_t *)val));
420  res = sizeof(uint16_t);
421  break;
422 
423  case NETOPT_STATE:
424  assert(len <= sizeof(netopt_state_t));
425  res = _set_state(dev, *((netopt_state_t *)val));
426  break;
427 
428  case NETOPT_AUTOACK:
429  at86rf2xx_set_option(dev, AT86RF2XX_OPT_AUTOACK,
430  ((bool *)val)[0]);
431  /* don't set res to set netdev2_ieee802154_t::flags */
432  break;
433 
434  case NETOPT_RETRANS:
435  assert(len <= sizeof(uint8_t));
436  at86rf2xx_set_max_retries(dev, *((uint8_t *)val));
437  res = sizeof(uint8_t);
438  break;
439 
440  case NETOPT_PRELOADING:
441  at86rf2xx_set_option(dev, AT86RF2XX_OPT_PRELOADING,
442  ((bool *)val)[0]);
443  res = sizeof(netopt_enable_t);
444  break;
445 
446  case NETOPT_PROMISCUOUSMODE:
447  at86rf2xx_set_option(dev, AT86RF2XX_OPT_PROMISCUOUS,
448  ((bool *)val)[0]);
449  res = sizeof(netopt_enable_t);
450  break;
451 
452  case NETOPT_RX_START_IRQ:
453  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_RX_START,
454  ((bool *)val)[0]);
455  res = sizeof(netopt_enable_t);
456  break;
457 
458  case NETOPT_RX_END_IRQ:
459  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_RX_END,
460  ((bool *)val)[0]);
461  res = sizeof(netopt_enable_t);
462  break;
463 
464  case NETOPT_TX_START_IRQ:
465  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_TX_START,
466  ((bool *)val)[0]);
467  res = sizeof(netopt_enable_t);
468  break;
469 
470  case NETOPT_TX_END_IRQ:
471  at86rf2xx_set_option(dev, AT86RF2XX_OPT_TELL_TX_END,
472  ((bool *)val)[0]);
473  res = sizeof(netopt_enable_t);
474  break;
475 
476  case NETOPT_CSMA:
477  at86rf2xx_set_option(dev, AT86RF2XX_OPT_CSMA,
478  ((bool *)val)[0]);
479  res = sizeof(netopt_enable_t);
480  break;
481 
482  case NETOPT_CSMA_RETRIES:
483  assert(len <= sizeof(uint8_t));
484  if( !(dev->netdev.flags & AT86RF2XX_OPT_CSMA ||
485  (*((uint8_t *)val) > 5)) ) {
486  /* If CSMA is disabled, don't allow setting retries */
487  res = -EINVAL;
488  }
489  else {
490  at86rf2xx_set_csma_max_retries(dev, *((uint8_t *)val));
491  res = sizeof(uint8_t);
492  }
493  break;
494 
495  case NETOPT_CCA_THRESHOLD:
496  assert(len <= sizeof(int8_t));
497  at86rf2xx_set_cca_threshold(dev, *((int8_t *)val));
498  res = sizeof(int8_t);
499  break;
500 
501  default:
502  break;
503  }
504 
505  /* go back to sleep if were sleeping and state hasn't been changed */
506  if ((old_state == AT86RF2XX_STATE_SLEEP) &&
507  (opt != NETOPT_STATE)) {
508  at86rf2xx_set_state(dev, AT86RF2XX_STATE_SLEEP);
509  }
510 
511  if (res == -ENOTSUP) {
512  res = netdev2_ieee802154_set((netdev2_ieee802154_t *)netdev, opt,
513  val, len);
514  }
515 
516  return res;
517 }

519 static void _isr(netdev2_t *netdev)
520 {
521  at86rf2xx_t *dev = (at86rf2xx_t *) netdev;
522  uint8_t irq_mask;
523  uint8_t state;
524  uint8_t trac_status;
525 
526  /* If transceiver is sleeping register access is impossible and frames are
527  * lost anyway, so return immediately.
528  */
529  state = at86rf2xx_get_status(dev);
530  if (state == AT86RF2XX_STATE_SLEEP) {
531  return;
532  }
533 
534  /* read (consume) device status */
535  irq_mask = at86rf2xx_reg_read(dev, AT86RF2XX_REG__IRQ_STATUS);
536 
537  trac_status = at86rf2xx_reg_read(dev, AT86RF2XX_REG__TRX_STATE) &
538  AT86RF2XX_TRX_STATE_MASK__TRAC;
539 
540  if (irq_mask & AT86RF2XX_IRQ_STATUS_MASK__RX_START) {
541  netdev->event_callback(netdev, NETDEV2_EVENT_RX_STARTED);
542  DEBUG("[at86rf2xx] EVT - RX_START\n");
543  }
544 
545  if (irq_mask & AT86RF2XX_IRQ_STATUS_MASK__TRX_END) {
546  if (state == AT86RF2XX_STATE_RX_AACK_ON ||
547  state == AT86RF2XX_STATE_BUSY_RX_AACK) {
548  DEBUG("[at86rf2xx] EVT - RX_END\n");
549  if (!(dev->netdev.flags & AT86RF2XX_OPT_TELL_RX_END)) {
550  return;
551  }
552  netdev->event_callback(netdev, NETDEV2_EVENT_RX_COMPLETE);
553  }
554  else if (state == AT86RF2XX_STATE_TX_ARET_ON ||
555  state == AT86RF2XX_STATE_BUSY_TX_ARET) {
556  /* check for more pending TX calls and return to idle state if
557  * there are none */
558  assert(dev->pending_tx != 0);
559  if ((--dev->pending_tx) == 0) {
560  at86rf2xx_set_state(dev, dev->idle_state);
561  DEBUG("[at86rf2xx] return to state 0x%x\n", dev->idle_state);
562  }
563 
564  DEBUG("[at86rf2xx] EVT - TX_END\n");
565 
566  if (netdev->event_callback && (dev->netdev.flags & AT86RF2XX_OPT_TELL_TX_END)) {
567  switch (trac_status) {
568  case AT86RF2XX_TRX_STATE__TRAC_SUCCESS:
569  case AT86RF2XX_TRX_STATE__TRAC_SUCCESS_DATA_PENDING:
570  netdev->event_callback(netdev, NETDEV2_EVENT_TX_COMPLETE);
571  DEBUG("[at86rf2xx] TX SUCCESS\n");
572  break;
573  case AT86RF2XX_TRX_STATE__TRAC_NO_ACK:
574  netdev->event_callback(netdev, NETDEV2_EVENT_TX_NOACK);
575  DEBUG("[at86rf2xx] TX NO_ACK\n");
576  break;
577  case AT86RF2XX_TRX_STATE__TRAC_CHANNEL_ACCESS_FAILURE:
578  netdev->event_callback(netdev, NETDEV2_EVENT_TX_MEDIUM_BUSY);
579  DEBUG("[at86rf2xx] TX_CHANNEL_ACCESS_FAILURE\n");
580  break;
581  default:
582  DEBUG("[at86rf2xx] Unhandled TRAC_STATUS: %d\n",
583  trac_status >> 5);
584  }
585  }
586  }
587  }
