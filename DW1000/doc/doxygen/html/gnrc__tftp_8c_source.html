453 int _tftp_server(tftp_context_t *ctxt)
454 {
455  msg_t msg;
456  bool active = true;
457  gnrc_netreg_entry_t entry = GNRC_NETREG_ENTRY_INIT_PID(GNRC_TFTP_DEFAULT_DST_PORT,
458  sched_active_pid);
459 
460  while (active) {
461  int ret = TS_BUSY;
462  bool got_client = false;
463 
464  /* register the servers main listening port */
465  if (gnrc_netreg_register(GNRC_NETTYPE_UDP, &entry)) {
466  DEBUG("tftp: error starting server.\n");
467  return TS_FAILED;
468  }
469 
470  /* main processing loop */
471  while (ret == TS_BUSY) {
472  /* wait for a message */
473  msg_receive(&msg);
474 
475  /* check if the server stop message has been received */
476  if (msg.type == TFTP_STOP_SERVER_MSG) {
477  active = false;
478  ret = TS_FAILED;
479  break;
480  }
481  else {
482  /* continue normal server opration */
483  DEBUG("tftp: message incoming\n");
484  ret = _tftp_state_processes(ctxt, &msg);
485 
486  /* release packet if we received one */
487  if (msg.type == GNRC_NETAPI_MSG_TYPE_RCV) {
488  gnrc_pktbuf_release(msg.content.ptr);
489  }
490  }
491 
492  /* if we just accepted a client, disable the server main listening port */
493  if (!got_client) {
494  gnrc_netreg_unregister(GNRC_NETTYPE_UDP, &entry);
495 
496  if (ret == TS_BUSY) {
497  got_client = true;
498  DEBUG("tftp: connection established\n");
499  }
500  }
501  }
502 
503  /* remove any stall timers */
504  xtimer_remove(&(ctxt->timer));
505 
506  /* if the server transfer has finished, unregister the client dst port */
507  gnrc_netreg_unregister(GNRC_NETTYPE_UDP, &(ctxt->entry));
508  DEBUG("tftp: connection terminated\n");
509  }
510 
511  /* unregister our UDP listener on this thread */
512  gnrc_netreg_unregister(GNRC_NETTYPE_UDP, &entry);
513 
514  return 0;
515 }

517 int _tftp_do_client_transfer(tftp_context_t *ctxt)
518 {
519  msg_t msg;
520  tftp_state ret = TS_BUSY;
521 
522  /* register our DNS response listener */
523  gnrc_netreg_entry_t entry = GNRC_NETREG_ENTRY_INIT_PID(ctxt->src_port,
524  sched_active_pid);
525 
526  if (gnrc_netreg_register(GNRC_NETTYPE_UDP, &entry)) {
527  DEBUG("tftp: error starting server.\n");
528  return TS_FAILED;
529  }
530 
531  /* try to start the TFTP transfer */
532  ret = _tftp_state_processes(ctxt, NULL);
533  if (ret != TS_BUSY) {
534  DEBUG("tftp: transfer failed\n");
535  /* if the start failed return */
536  return ret;
537  }
538 
539  /* main processing loop */
540  while (ret == TS_BUSY) {
541  /* wait for a message */
542  msg_receive(&msg);
543  DEBUG("tftp: message received\n");
544  ret = _tftp_state_processes(ctxt, &msg);
545 
546  /* release packet if we received one */
547  if (msg.type == GNRC_NETAPI_MSG_TYPE_RCV) {
548  gnrc_pktbuf_release(msg.content.ptr);
549  }
550  }
551 
552  /* unregister our UDP listener on this thread */
553  gnrc_netreg_unregister(GNRC_NETTYPE_UDP, &entry);
554 
555  return ret;
556 }

558 tftp_state _tftp_state_processes(tftp_context_t *ctxt, msg_t *m)
559 {
560  gnrc_pktsnip_t *outbuf = gnrc_pktbuf_add(NULL, NULL, TFTP_DEFAULT_DATA_SIZE,
561  GNRC_NETTYPE_UNDEF);
562 
563  /* check if this is an client start */
564  if (!m) {
565  DEBUG("tftp: starting transaction as client\n");
566  return _tftp_send_start(ctxt, outbuf);
567  }
568  else if (m->type == TFTP_TIMEOUT_MSG) {
569  DEBUG("tftp: timeout occured\n");
570  if (++(ctxt->retries) > GNRC_TFTP_MAX_RETRIES) {
571  /* transfer failed due to lost peer */
572  DEBUG("tftp: peer lost\n");
573  gnrc_pktbuf_release(outbuf);
574  return TS_FAILED;
575  }
576 
577  /* increase the timeout for congestion control */
578  ctxt->block_timeout <<= 1;
579 
580  /* the send message timed out, re-sending */
581  if (ctxt->dst_port == GNRC_TFTP_DEFAULT_DST_PORT) {
582  DEBUG("tftp: sending timed out, re-sending\n");
583  /* we are still negotiating resent, start */
584  return _tftp_send_start(ctxt, outbuf);
585  }
586  else {
587  DEBUG("tftp: last data or ack packet lost, resending\n");
588  /* we are sending / receiving data */
589  /* if we are reading resent the ACK, if writing the DATA */
590  return _tftp_send_dack(ctxt, outbuf, (ctxt->op == TO_RRQ) ? TO_ACK : TO_DATA);
591  }
592  }
593  else if (m->type != GNRC_NETAPI_MSG_TYPE_RCV) {
594  DEBUG("tftp: unknown message\n");
595  gnrc_pktbuf_release(outbuf);
596  return TS_BUSY;
597  }
598 
599  gnrc_pktsnip_t *pkt = m->content.ptr;
600 
601  gnrc_pktsnip_t *tmp;
602  tmp = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_UDP);
603  udp_hdr_t *udp = (udp_hdr_t *)tmp->data;
604 
605  tmp = gnrc_pktsnip_search_type(pkt, GNRC_NETTYPE_IPV6);
606  ipv6_hdr_t *ip = (ipv6_hdr_t *)tmp->data;
607  uint8_t *data = (uint8_t *)pkt->data;
608 
609  xtimer_remove(&(ctxt->timer));
610 
611  switch (_tftp_parse_type(data)) {
612  case TO_RRQ:
613  case TO_WRQ: {
614  if (byteorder_ntohs(udp->dst_port) != GNRC_TFTP_DEFAULT_DST_PORT) {
615  /* not a valid start packet */
616  DEBUG("tftp: incoming packet on port %d dropped\n", byteorder_ntohs(udp->dst_port));
617  gnrc_pktbuf_release(outbuf);
618  return TS_FAILED;
619  }
620 
621  /* reinitialize the context with the current client */
622  tftp_opcodes_t op = _tftp_parse_type(data);
623  _tftp_init_ctxt(&(ip->src), NULL, op, TTM_ASCII, CT_SERVER,
624  ctxt->start_cb, ctxt->stop_cb, ctxt->data_cb,
625  ctxt->enable_options, ctxt);
626 
627  /* get the context of the client */
628  ctxt->dst_port = byteorder_ntohs(udp->src_port);
629  DEBUG("tftp: client's port is %" PRIu16 "\n", ctxt->dst_port);
630 
631  int offset = _tftp_decode_start(ctxt, data, outbuf);
632  DEBUG("tftp: offset after decode start = %i\n", offset);
633  if (offset < 0) {
634  DEBUG("tftp: there is no data?\n");
635  gnrc_pktbuf_release(outbuf);
636  return TS_FAILED;
637  }
638 
639  /* register a listener for the UDP port */
640  gnrc_netreg_entry_init_pid(&(ctxt->entry), ctxt->src_port,
641  sched_active_pid);
642  gnrc_netreg_register(GNRC_NETTYPE_UDP, &(ctxt->entry));
643 
644  /* try to decode the options */
645  tftp_state state;
646  tftp_opcodes_t opcode;
647  if (ctxt->enable_options &&
648  _tftp_decode_options(ctxt, pkt, offset) > offset) {
649  DEBUG("tftp: send option ACK\n");
650 
651  /* the client send the TFTP options */
652  opcode = TO_OACK;
653  }
654  else {
655  DEBUG("tftp: send normal ACK\n");
656 
657  /* the client didn't send options, use ACK and set defaults */
658  _tftp_set_default_options(ctxt);
659 
660  /* send the first data block */
661  if (ctxt->op == TO_RRQ) {
662  ++(ctxt->block_nr);
663  opcode = TO_DATA;
664  }
665  else {
666  opcode = TO_ACK;
667  }
668  }
669 
670  /* validate if the application accepts the action, mode, filename and transfer_size */
671  tftp_action_t action = (ctxt->op == TO_RRQ) ? TFTP_READ : TFTP_WRITE;
672  if (!ctxt->start_cb(action, ctxt->mode, ctxt->file_name, &(ctxt->transfer_size))) {
673  _tftp_send_error(ctxt, outbuf, TE_ACCESS, "Blocked by user application");
674  DEBUG("tftp: callback not able to handle mode\n");
675  return TS_FAILED;
676  }
677 
678  /* the client send the TFTP options */
679  state = _tftp_send_dack(ctxt, outbuf, opcode);
680 
681  /* check if the client negotiation was successful */
682  if (state != TS_BUSY) {
683  DEBUG("tftp: not able to send ACK\n");
684  }
685  return state;
686  } break;
687 
688  case TO_DATA: {
689  /* try to process the data */
690  int proc = _tftp_process_data(ctxt, pkt);
691  if (proc < 0) {
692  DEBUG("tftp: data not accepted\n");
693  /* the data is not accepted return */
694  gnrc_pktbuf_release(outbuf);
695  return TS_BUSY;
696  }
697 
698  /* check if this is the first block */
699  if (!ctxt->block_nr
700  && ctxt->dst_port == GNRC_TFTP_DEFAULT_DST_PORT
701  && !ctxt->use_options) {
702  /* no OACK received, restore default TFTP parameters */
703  _tftp_set_default_options(ctxt);
704  DEBUG("tftp: restore default TFTP parameters\n");
705 
706  /* switch the destination port to the src port of the server */
707  ctxt->dst_port = byteorder_ntohs(udp->src_port);
708  }
709 
710  /* wait for the next data block */
711  DEBUG("tftp: wait for the next data block\n");
712  ++(ctxt->block_nr);
713  _tftp_send_dack(ctxt, outbuf, TO_ACK);
714 
715  /* check if the data transfer has finished */
716  if (proc < ctxt->block_size) {
717  DEBUG("tftp: transfer finished\n");
718 
719  if (ctxt->stop_cb) {
720  ctxt->stop_cb(TFTP_SUCCESS, NULL);
721  }
722 
723  return TS_FINISHED;
724  }
725 
726  return TS_BUSY;
727  } break;
728 
729  case TO_ACK: {
730  /* validate if this is the ACK we are waiting for */
731  if (!_tftp_validate_ack(ctxt, data)) {
732  /* invalid packet ACK, drop */
733  gnrc_pktbuf_release(outbuf);
734  return TS_BUSY;
735  }
736 
737  /* check if the write action is finished */
738  if (ctxt->write_finished) {
739  gnrc_pktbuf_release(outbuf);
740 
741  if (ctxt->stop_cb) {
742  ctxt->stop_cb(TFTP_SUCCESS, NULL);
743  }
744 
745  return TS_FINISHED;
746  }
747 
748  /* check if this is the first ACK */
749  if (!ctxt->block_nr && ctxt->dst_port != byteorder_ntohs(udp->src_port)) {
750  /* no OACK received restore default TFTP parameters */
751  _tftp_set_default_options(ctxt);
752 
753  /* switch the destination port to the src port of the server */
754  ctxt->dst_port = byteorder_ntohs(udp->src_port);
755  }
756 
757  /* send the next data block */
758  ++(ctxt->block_nr);
759 
760  return _tftp_send_dack(ctxt, outbuf, TO_DATA);
761  } break;
762 
763  case TO_ERROR: {
764  tftp_err_codes_t err;
765  const char *err_msg;
766 
767  /* decode the received error */
768  _tftp_decode_error(data, &err, &err_msg);
769 
770  /* inform the user application about the error */
771  if (ctxt->stop_cb) {
772  ctxt->stop_cb(TFTP_PEER_ERROR, err_msg);
773  }
774 
775  DEBUG("tftp: ERROR: %s\n", err_msg);
776  gnrc_pktbuf_release(outbuf);
777  return TS_FAILED;
778  } break;
779 
780  case TO_OACK: {
781  /* only allow one OACK to be received */
782  if (ctxt->dst_port != byteorder_ntohs(udp->src_port)) {
783  DEBUG("tftp: TO_OACK received\n");
784 
785  /* decode the options */
786  _tftp_decode_options(ctxt, pkt, 0);
787 
788  /* take the new source port */
789  ctxt->dst_port = byteorder_ntohs(udp->src_port);
790 
791  /* we must send block one to finish the negotiation in send mode */
792  if (ctxt->op == TO_WRQ) {
793  ++(ctxt->block_nr);
794  }
795  }
796  else {
797  DEBUG("tftp: dropping double TO_OACK\n");
798  }
799 
800  return _tftp_send_dack(ctxt, outbuf, (ctxt->op == TO_WRQ) ? TO_DATA : TO_ACK);
801  } break;
802  }
803 
804  gnrc_pktbuf_release(outbuf);
805  return TS_FAILED;
806 }

926 tftp_state _tftp_send(gnrc_pktsnip_t *buf, tftp_context_t *ctxt, size_t len)
927 {
928  network_uint16_t src_port, dst_port;
929  gnrc_pktsnip_t *udp, *ip;
930 
931  assert(len <= TFTP_DEFAULT_DATA_SIZE);
932 
933  /* down-size the packet to it's used size */
934  if (len > TFTP_DEFAULT_DATA_SIZE) {
935  DEBUG("tftp: can't reallocate to bigger packet, buffer overflowed\n");
936  gnrc_pktbuf_release(buf);
937 
938  if (ctxt->stop_cb) {
939  ctxt->stop_cb(TFTP_INTERN_ERROR, "buffer overflowed");
940  }
941 
942  return TS_FAILED;
943  }
944  else if (gnrc_pktbuf_realloc_data(buf, len) != 0) {
945  assert(false);
946 
947  DEBUG("tftp: failed to reallocate data snippet\n");
948  gnrc_pktbuf_release(buf);
949 
950  /* inform the user that we can't reallocate */
951  if (ctxt->stop_cb) {
952  ctxt->stop_cb(TFTP_INTERN_ERROR, "no reallocate");
953  }
954 
955  return TS_FAILED;
956  }
957 
958  /* allocate UDP header, set source port := destination port */
959  src_port.u16 = ctxt->src_port;
960  dst_port.u16 = ctxt->dst_port;
961  udp = gnrc_udp_hdr_build(buf, src_port.u16, dst_port.u16);
962  if (udp == NULL) {
963  DEBUG("tftp: error unable to allocate UDP header\n");
964  gnrc_pktbuf_release(buf);
965 
966  if (ctxt->stop_cb) {
967  ctxt->stop_cb(TFTP_INTERN_ERROR, "no udp allocate");
968  }
969 
970  return TS_FAILED;
971  }
972 
973  /* allocate IPv6 header */
974  ip = gnrc_ipv6_hdr_build(udp, NULL, &(ctxt->peer));
975  if (ip == NULL) {
976  DEBUG("tftp: error unable to allocate IPv6 header\n");
977  gnrc_pktbuf_release(udp);
978 
979  if (ctxt->stop_cb) {
980  ctxt->stop_cb(TFTP_INTERN_ERROR, "no ip allocate");
981  }
982 
983  return TS_FAILED;
984  }
985 
986  /* send packet */
987  if (gnrc_netapi_dispatch_send(GNRC_NETTYPE_UDP, GNRC_NETREG_DEMUX_CTX_ALL,
988  ip) == 0) {
989  /* if send failed inform the user */
990  DEBUG("tftp: error unable to locate UDP thread\n");
991  gnrc_pktbuf_release(ip);
992 
993  if (ctxt->stop_cb) {
994  ctxt->stop_cb(TFTP_INTERN_ERROR, "no dispatch send");
995  }
996 
997  return TS_FAILED;
998  }
999 
1000  /* only set timeout if enabled for this block */
1001  if (ctxt->block_timeout) {
1002  ctxt->timer_msg.type = TFTP_TIMEOUT_MSG;
1003  xtimer_set_msg(&(ctxt->timer), ctxt->block_timeout, &(ctxt->timer_msg), thread_getpid());
1004  DEBUG("tftp: set timeout %" PRIu32 " ms\n", ctxt->block_timeout / US_PER_MS);
1005  }
1006 
1007  return TS_BUSY;
1008 }
