174 int gpio_init(gpio_t pin, gpio_mode_t mode)
175 {
176  /* set pin to analog mode while configuring it */
177  gpio_init_port(pin, GPIO_AF_ANALOG);
178 
179  /* set pin direction */
180  if (mode & MODE_OUT) {
181  gpio(pin)->PDDR |= (1 << pin_num(pin));
182  gpio(pin)->PCOR = (1 << pin_num(pin));
183  }
184  else {
185  gpio(pin)->PDDR &= ~(1 << pin_num(pin));
186  }
187 
188  /* enable GPIO function */
189  port(pin)->PCR[pin_num(pin)] = (GPIO_AF_GPIO | (mode & MODE_PCR_MASK));
190  return 0;
191 }

193 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
194  gpio_cb_t cb, void *arg)
195 {
196  if (gpio_init(pin, mode) < 0) {
197  return -1;
198  }
199 
200  /* try go grab a free spot in the context array */
201  int ctx_num = get_free_ctx();
202  if (ctx_num < 0) {
203  return -1;
204  }
205 
206  /* save interrupt context */
207  isr_ctx[ctx_num].cb = cb;
208  isr_ctx[ctx_num].arg = arg;
209  isr_ctx[ctx_num].state = flank;
210  write_map(port_num(pin), pin_num(pin), ctx_num);
211 
212  /* clear interrupt flags */
213  port(pin)->ISFR &= ~(1 << pin_num(pin));
214  /* enable global port interrupts in the NVIC */
215  NVIC_EnableIRQ(PORTA_IRQn + port_num(pin));
216  /* finally, enable the interrupt for the select pin */
217  port(pin)->PCR[pin_num(pin)] |= flank;
218  return 0;
219 }

221 void gpio_init_port(gpio_t pin, uint32_t pcr)
222 {
223  /* enable PORT clock in case it was not active before */
224  clk_en(pin);
225 
226  /* if the given interrupt was previously configured as interrupt source, we
227  * need to free its interrupt context. We to this only after we
228  * re-configured the pin in case an event is happening just in between... */
229  uint32_t isr_state = port(pin)->PCR[pin_num(pin)];
230  /* set new PCR value */
231  port(pin)->PCR[pin_num(pin)] = pcr;
232  /* and clear the interrupt context if needed */
233  if (isr_state & PORT_PCR_IRQC_MASK) {
234  ctx_clear(port_num(pin), pin_num(pin));
235  }
236 }

286 static inline void irq_handler(PORT_Type *port, int port_num)
287 {
288  /* take interrupt flags only from pins which interrupt is enabled */
289  uint32_t status = port->ISFR;
290 
291  for (int i = 0; i < 32; i++) {
292  if ((status & (1 << i)) && (port->PCR[i] & PORT_PCR_IRQC_MASK)) {
293  port->ISFR = (1 << i);
294  int ctx = get_ctx(port_num, i);
295  isr_ctx[ctx].cb(isr_ctx[ctx].arg);
296  }
297  }
298  cortexm_isr_end();
299 }
