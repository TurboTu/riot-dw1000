83 int i2c_init_master(i2c_t dev, i2c_speed_t speed)
84 {
85  I2C_TypeDef *i2c;
86  gpio_t pin_scl, pin_sda;
87  int ccr;
88 
89  /* read speed configuration */
90  switch (speed) {
91  case I2C_SPEED_NORMAL:
92  ccr = I2C_APBCLK / 200000;
93  break;
94  case I2C_SPEED_FAST:
95  ccr = I2C_APBCLK / 800000;
96  break;
97  default:
98  return -2;
99  }
100 
101  /* read static device configuration */
102  switch (dev) {
103 #if I2C_0_EN
104  case I2C_0:
105  i2c = I2C_0_DEV;
106  pin_scl = I2C_0_SCL_PIN;
107  pin_sda = I2C_0_SDA_PIN;
108  I2C_0_CLKEN();
109  NVIC_SetPriority(I2C_0_ERR_IRQ, I2C_IRQ_PRIO);
110  NVIC_EnableIRQ(I2C_0_ERR_IRQ);
111  break;
112 #endif
113 #if I2C_1_EN
114  case I2C_1:
115  i2c = I2C_1_DEV;
116  pin_scl = I2C_1_SCL_PIN;
117  pin_sda = I2C_1_SDA_PIN;
118  I2C_1_CLKEN();
119  NVIC_SetPriority(I2C_1_ERR_IRQ, I2C_IRQ_PRIO);
120  NVIC_EnableIRQ(I2C_1_ERR_IRQ);
121  break;
122 #endif
123  default:
124  return -1;
125  }
126 
127  /* disable peripheral */
128  i2c->CR1 &= ~I2C_CR1_PE;
129 
130  /* configure pins */
131  _pin_config(pin_scl, pin_sda);
132  /* configure device */
133  _i2c_init(i2c, ccr);
134 
135  /* make sure the analog filters don't hang -> see errata sheet 2.14.7 */
136  if (i2c->SR2 & I2C_SR2_BUSY) {
137  DEBUG("LINE BUSY AFTER RESET -> toggle pins now\n");
138  /* disable peripheral */
139  i2c->CR1 &= ~I2C_CR1_PE;
140  /* re-run pin config to toggle and re-configure pins */
141  _pin_config(pin_scl, pin_sda);
142  /* make peripheral soft reset */
143  i2c->CR1 |= I2C_CR1_SWRST;
144  i2c->CR1 &= ~I2C_CR1_SWRST;
145  /* enable device */
146  _i2c_init(i2c, ccr);
147  }
148 
149  return 0;
150 }

152 static void _i2c_init(I2C_TypeDef *i2c, int ccr)
153 {
154  /* disable device and set ACK bit */
155  i2c->CR1 = I2C_CR1_ACK;
156  /* configure I2C clock and enable error interrupts */
157  i2c->CR2 = (I2C_APBCLK / 1000000) | I2C_CR2_ITERREN;
158  i2c->CCR = ccr;
159  i2c->TRISE = (I2C_APBCLK / 1000000) + 1;
160  /* configure device */
161  i2c->OAR1 = 0; /* makes sure we are in 7-bit address mode */
162  /* enable device */
163  i2c->CR1 |= I2C_CR1_PE;
164 }

207 int i2c_read_bytes(i2c_t dev, uint8_t address, void *data, int length)
208 {
209  int i = 0;
210  I2C_TypeDef *i2c;
211  uint8_t *my_data = data;
212 
213  switch (dev) {
214 #if I2C_0_EN
215  case I2C_0:
216  i2c = I2C_0_DEV;
217  break;
218 #endif
219 #if I2C_1_EN
220  case I2C_1:
221  i2c = I2C_1_DEV;
222  break;
223 #endif
224  default:
225  return -1;
226  }
227 
228  DEBUG("Send Slave address and wait for ADDR == 1\n");
229  _start(i2c, address, I2C_FLAG_READ, &err_flag[dev]);
230 
231  if (err_flag[dev]) {
232  return -2;
233  }
234 
235  DEBUG("Clear ADDR\n");
236  _clear_addr(i2c);
237 
238  switch (length) {
239  case 1:
240  break;
241 
242  case 2:
243  DEBUG("Set POS and ACK bit\n");
244  i2c->CR1 |= (I2C_CR1_POS | I2C_CR1_ACK);
245  DEBUG("Crit block: clear ACK flag\n");
246  i2c->CR1 &= ~(I2C_CR1_ACK);
247  DEBUG("Wait for transfer to be completed\n");
248  while (!(i2c->SR1 & I2C_SR1_BTF)) {}
249 
250  break;
251 
252  default:
253  i2c->CR1 |= (I2C_CR1_ACK);
254 
255  while (i < (length - 3)) {
256  DEBUG("Wait until byte was received\n");
257  while (!(i2c->SR1 & I2C_SR1_RXNE)) {}
258  DEBUG("Copy byte from DR\n");
259  my_data[i++] = i2c->DR;
260  }
261 
262  DEBUG("Reading the last 3 bytes, waiting for BTF flag\n");
263  while (!(i2c->SR1 & I2C_SR1_BTF)) {}
264 
265  DEBUG("Read N-3 byte\n");
266  my_data[i++] = i2c->DR;
267  }
268 
269  DEBUG("Clear ACK\n");
270  i2c->CR1 &= ~(I2C_CR1_ACK);
271 
272  DEBUG("Setting STOP=1\n");
273  i2c->CR1 |= (I2C_CR1_STOP);
274 
275  while (i < length) {
276  DEBUG("Wait for RXNE == 1\n");
277  while (!(i2c->SR1 & I2C_SR1_RXNE)) {}
278 
279  DEBUG("Read byte\n");
280  my_data[i++] = i2c->DR;
281  }
282 
283  DEBUG("wait for STOP bit to be cleared again\n");
284  while (i2c->CR1 & I2C_CR1_STOP) {}
285 
286  DEBUG("reset POS = 0 and ACK = 1\n");
287  i2c->CR1 &= ~(I2C_CR1_POS);
288  i2c->CR1 |= (I2C_CR1_ACK);
289 
290  return length;
291 }

454 static void _start(I2C_TypeDef *dev, uint8_t address, uint8_t rw_flag, uint8_t *err)
455 {
456  /* flag that there's no error (yet) */
457  *err = 0x00;
458  /* wait for device to be ready */
459  DEBUG("Wait for device to be ready\n");
460  while (dev->SR2 & I2C_SR2_BUSY) {}
461  /* generate start condition */
462  DEBUG("Generate start condition\n");
463  dev->CR1 |= I2C_CR1_START;
464  DEBUG("Wait for SB flag to be set\n");
465  while (!(dev->SR1 & I2C_SR1_SB)) {}
466  /* send address and read/write flag */
467  DEBUG("Send address\n");
468  dev->DR = (address << 1) | rw_flag;
469  /* clear ADDR flag by reading first SR1 and then SR2 */
470  DEBUG("Wait for ADDR flag to be set\n");
471  while (!(dev->SR1 & I2C_SR1_ADDR) && !(*err)) {}
472 }

481 {

483  for (int i = 0; i < length && !(*err); i++) {
484  /* write data to data register */
485  dev->DR = data[i];
486  DEBUG("Written %i byte to data reg, now waiting for DR to be empty again\n", i);
487  /* wait for transfer to finish */
488  while (!(dev->SR1 & I2C_SR1_TXE) && !(*err)) {}
489  DEBUG("DR is now empty again\n");
490  }

495 {

504 static inline void i2c_irq_handler(i2c_t i2c_dev, I2C_TypeDef *dev)
505 {
506  unsigned volatile state = dev->SR1;
507 
508  /* record and clear errors */

515  DEBUG("OVR\n");

517  if (state & I2C_SR1_AF) {
518  DEBUG("AF\n");

530  DEBUG("TIMEOUT\n");
531  }
532  if (state & I2C_SR1_SMBALERT) {
533  DEBUG("SMBALERT\n");
