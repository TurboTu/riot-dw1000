46 {
47  /* make sure given timer is valid */
48  if (tim >= TIMER_NUMOF) {

50  }
51 
52  /* enable the times clock */
53  PRCM->GPTCLKGR |= (1 << timer_config[tim].num);
54  PRCM->CLKLOADCTL = CLKLOADCTL_LOAD;
55  while (!(PRCM->CLKLOADCTL & CLKLOADCTL_LOADDONE)) {}
56 
57  /* disable (and reset) timer */
58  dev(tim)->CTL = 0;
59 
60  /* save context */
61  ctx[tim].cb = cb;
62  ctx[tim].arg = arg;

64  /* configure timer to 16-bit, periodic, up-counting */
65  dev(tim)->CFG = GPT_CFG_16T;
66  dev(tim)->TAMR = (GPT_TXMR_TXMR_PERIODIC | GPT_TXMR_TXCDIR_UP | GPT_TXMR_TXMIE);

72  dev(tim)->CTL = GPT_CTL_TAEN;
