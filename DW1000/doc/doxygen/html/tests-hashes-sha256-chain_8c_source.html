32 static void test_sha256_hash_chain(void)
33 {
34  const char strSeed[] = "My cool secret seed, you'll never guess it ;) 12345";
35  static unsigned char tail_hash_chain_element[SHA256_DIGEST_LENGTH];
36 
37  /* we produce a sha256-chain of 257 elements */
38  size_t elements = 257;
39 
40  memset(tail_hash_chain_element, 0, SHA256_DIGEST_LENGTH);
41  TEST_ASSERT(sha256_chain((unsigned char*)strSeed, strlen(strSeed),
42  elements, tail_hash_chain_element) != NULL);
43 
44  /* we check if the first element is part of the chain */
45  unsigned char element_one[SHA256_DIGEST_LENGTH];
46  sha256((unsigned char*)strSeed, strlen(strSeed), element_one);
47  TEST_ASSERT(sha256_chain_verify_element(element_one, 0,
48  tail_hash_chain_element, elements) == 0);
49 
50  /* now we check if the test fails if the index is wrong */
51  TEST_ASSERT(sha256_chain_verify_element(element_one, 2,
52  tail_hash_chain_element, elements) == 1);
53 
54  /* now we check if other elements are also part of the chain */
55  unsigned char tmp_element[SHA256_DIGEST_LENGTH];
56  memcpy((void*)tmp_element, (void*)element_one, SHA256_DIGEST_LENGTH);
57 
58  /* since we know the seed we build every element and test if its in the chain */
59  for (size_t i = 1; i < elements; ++i) {
60  sha256_context_t ctx;
61  sha256_init(&ctx);
62  sha256_update(&ctx, tmp_element, SHA256_DIGEST_LENGTH);
63  sha256_final(&ctx, tmp_element);
64 
65  TEST_ASSERT(sha256_chain_verify_element(tmp_element, i,
66  tail_hash_chain_element, elements) == 0);
67  }
68 }
