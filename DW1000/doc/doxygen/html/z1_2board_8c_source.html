29 static void z1_ports_init(void)
30 {
31  /* Port 1:
32  * P1.0 is not assigned by default
33  * P1.1 is the bootstrap-loader (BSL) TX pin -> input, special function, default to GND
34  * THIS PIN MUST *NEVER* BE USED IN NORMAL EXECUTION, SINCE IT INTERFERES WITH UART0 !!!
35  * P1.2 receives the FIFOP interrupt from CC2420 -> input, GPIO, default to GND
36  * P1.3 receives the FIFO/GIO0 interrupt from CC2420 -> input, GPIO, default to GND
37  * P1.4 receives the CCA/GIO1 signal from CC2420 -> input, GPIO, default to GND
38  * P1.5 is wired to Vcc -> input, GPIO, default to Vcc
39  * P1.6 receives interrupt INT1 from accelerometer -> input, GPIO, default to GND
40  * P1.7 receives interrupt INT2 from accelerometer -> input, GPIO, default to GND
41  */
42  P1SEL = 0x02; /* Port1 Select: 00000010 = 0x02 */
43  P1OUT = 0x20; /* Port1 Output: 00100000 = 0x20 */
44  P1DIR = 0x00; /* Port1 Direction: 00000000 = 0x00 */
45 
46  /* Port 2:
47  * P2.0 is not assigned by default
48  * P2.1 is not assigned by default
49  * P2.2 is the bootstrap-loader (BSL) RX pin -> input, special function, default to GND
50  * THIS PIN MUST *NEVER* BE USED IN NORMAL EXECUTION, SINCE IT INTERFERES WITH UART0 !!!
51  * P2.3 is not assigned by default
52  * P2.4 is used as an active-low output to the BSL and USB interface -> output, GPIO, default to Vcc
53  * P2.5 is connected to the active-low "user interrupt" button -> input, GPIO, default to Vcc
54  * P2.6 is not assigned by default
55  * P2.7 receives the ALERT interrupt from TMP102 -> input, GPIO, default to Vcc
56  */
57  P2SEL = 0x04; /* Port2 Select: 00000100 = 0x04 */
58  P2OUT = 0xB0; /* Port2 Output: 10110000 = 0xB0 */
59  P2DIR = 0xD0; /* Port2 Direction: 11010000 = 0xD0 */
60 
61  /* Port 3:
62  * P3.0 is the radio (CC2420) active-low chip select -> output, GPIO, default to Vcc
63  * P3.1 is SPI's MOSI pin -> output, special function, default to GND
64  * P3.2 is SPI's MISO pin -> input, special function, default to GND
65  * P3.3 is SPI's CLK pin -> output, special function, default to GND
66  * P3.4 is USCI0 UART TX pin -> output, special function, default to Vcc
67  * P3.5 is USCI0 UART RX pin -> input, special function, default to Vcc
68  * P3.6 is USCI1 UART TX pin -> output, special function, default to GND
69  * P3.7 is USCI1 UART RX pin -> input, special function, default to GND
70  * NOTES :
71  * - Z1 only uses the USCI0 SPI channel
72  * - UART0 is connected to the micro-USB port (via the CP2102 chip)
73  */
74  P3SEL = 0xFE; /* Port3 Select: 11111110 = 0xFE */
75  P3OUT = 0x31; /* Port3 Output: 00110001 = 0x31 */
76  P3DIR = 0x5B; /* Port3 Direction: 01011011 = 0x5B */
77 
78  /* Port 4:
79  * P4.0 is not assigned by default (but controls a LED in Z1 "starter pack") -> output, GPIO, default to Vcc
80  * P4.1 receives the SFD interrupt from CC2420, that can be captured by TimerB -> input, special function, default to GND
81  * P4.2 is not assigned by default (but controls a LED in Z1 "starter pack") -> output, GPIO, default to Vcc
82  * P4.3 is not assigned by default (but controls a buzzer in Z1 "starter pack") -> output, GPIO, default to GND
83  * P4.4 is the Flash chip (M25P16) active-low chip select -> output, GPIO, default to Vcc
84  * P4.5 is the radio (CC2420) active-high VREG enable line -> output, GPIO, default to GND
85  * P4.6 is the radio (CC2420) active-low RESET line -> output, GPIO, default to Vcc
86  * P4.7 is not assigned by default (but controls a LED in Z1 "starter pack") -> output, GPIO, default to Vcc
87  */
88  P4SEL = 0x02; /* Port4 Select: 00000010 = 0x02 */
89  P4OUT = 0xD5; /* Port4 Output: 11010101 = 0xD5 */
90  P4DIR = 0xFD; /* Port4 Direction: 11111101 = 0xFD */
91 
92  /* Port 5:
93  * I2C, and GPIO (LEDs, flash)
94  * P5.0 controls TMP102 power (active high) -> output, GPIO, default to GND
95  * P5.1 is I2C's SDA (data) pin -> input (by default/changeable), special function, default to GND
96  * P5.2 is I2C's SCL (clock) pin -> output, special function, default to GND
97  * P5.3 is not assigned by default

103  * - Z1 only uses the USCI1 I2C channel

123 void msp430_init_dco(void)
124 {
125 #ifdef CALIBRATE_MSP430_DCO
126 #define DELTA (F_CPU / (F_RC_OSCILLATOR / 8))
127  /* This code taken from the FU Berlin sources and reformatted. */
128  unsigned int oldcapture = 0;
129  unsigned int i;
130 
131  /* 10100100 = XT2 is off, ACLK divided by 4, RSELx=4 */
132  BCSCTL1 = XT2OFF | DIVA_3 | RSEL2;
133 
134  /* Init undivided DCO with internal resistor for MCLK and SMCLK
135  * DCO = 32762Hz -> FLL = 2,4576 MHz */
136  BCSCTL2 = 0x00;
137 
138  BCSCTL1 |= DIVA1 + DIVA0; /* ACLK = LFXT1CLK/8 */
139 
140  for (i = 0xFFFF; i > 0; i--) { /* Delay for XTAL to settle */
141  __nop();
142  }
143 
144  CCTL2 = CCIS0 + CM0 + CAP; /* Define CCR2, CAP, ACLK */
145  TACTL = TASSEL1 + TACLR + MC1; /* SMCLK, continous mode */
146 
147  while (1) {
148  unsigned int compare;
149 
150  while ((CCTL2 & CCIFG) != CCIFG); /* Wait until capture occured!*/
151 
152  CCTL2 &= ~CCIFG; /* Capture occured, clear flag */
153  compare = CCR2; /* Get current captured SMCLK */
154  compare = compare - oldcapture; /* SMCLK difference */
155  oldcapture = CCR2; /* Save current captured SMCLK */
156 
157  if (DELTA == compare) {
158  break; /* if equal, leave "while (1)" */
159  }
160  else if (DELTA < compare) { /* DCO is too fast, slow it down */
161  DCOCTL--;
162 
163  if (DCOCTL == 0xFF) { /* Did DCO role under? */
164  BCSCTL1--;
165  }
166  }
167  else { /* -> Select next lower RSEL */
168  DCOCTL++;
169 
170  if (DCOCTL == 0x00) { /* Did DCO role over? */
171  BCSCTL1++;
172  } /* -> Select next higher RSEL */
173  }
174  }
175 
176  CCTL2 = 0; /* Stop CCR2 function */
177  TACTL = 0; /* Stop Timer_A */
178 
179  BCSCTL1 &= ~(DIVA1 + DIVA0); /* remove divisor from ACLK again */
180 
181  /*
182  * On a MSP430F2617 (as on a Z1), for a 8 MHz target frequency,
183  * we normally obtain these values:
184  * DCOCTL == 0x9a (i.e.: DCOx == 4 && MODx == 26)
185  * [the MODx field is the most prone to variation]
186  * BCSCTL1 == 0x0d (i.e.: RSELx == 13)
187  */
188 #else
189  /* default values for quick start-up */
190  DCOCTL = 0x00; /* avoid possible temporary overclocking... */
191  BCSCTL1 = 0x8d; /* as seen in Contiki code */
192  DCOCTL = 0x88; /* as seen in Contiki code */
193 #endif
194 
195  /* Other clock configuration */
196  BCSCTL1 |= XT2OFF; /* XT2 not connected on Z1 */
197  BCSCTL2 = 0; /* get MCLK and SMCLK from DCO, without divisor */
198  BCSCTL3 = XCAP_1; /* default value for LFXT1 capacitor and frequency */
199 }

208 
209  /* disable watchdog timer */
