47 {

50  cc110x_strobe(dev, CC110X_SIDLE); /* Switch to IDLE (should already be)... */

68 static void _rx_read_data(cc110x_t *dev, void(*callback)(void*), void*arg)
69 {
70  int fifo = cc110x_get_reg_robust(dev, 0xfb);
71 
72  if (fifo & 0x80) {
73  DEBUG("%s:%s:%u rx overflow\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
74  _rx_abort(dev);
75  return;
76  }
77 
78  if (!fifo) {
79  gpio_irq_enable(dev->params.gdo2);
80  return;
81  }
82 
83  cc110x_pkt_buf_t *pkt_buf = &dev->pkt_buf;
84  if (!pkt_buf->pos) {
85  pkt_buf->pos = 1;
86  pkt_buf->packet.length = cc110x_read_reg(dev, CC110X_RXFIFO);
87 
88  /* Possible packet received, RX -> IDLE (0.1 us) */
89  dev->cc110x_statistic.packets_in++;
90  }
91 
92  int left = pkt_buf->packet.length+1 - pkt_buf->pos;
93 
94  /* if the fifo doesn't contain the rest of the packet,
95  * leav at least one byte as per spec sheet. */
96  int to_read = (fifo < left) ? (fifo-1) : fifo;
97  if (to_read > left) {
98  to_read = left;
99  }
100 
101  if (to_read) {
102  cc110x_readburst_reg(dev, CC110X_RXFIFO,
103  ((char *)&pkt_buf->packet)+pkt_buf->pos, to_read);
104  pkt_buf->pos += to_read;
105  }
106 
107  if (to_read == left) {
108  uint8_t status[2];
109  /* full packet received. */
110  /* Read the 2 appended status bytes (status[0] = RSSI, status[1] = LQI) */
111  cc110x_readburst_reg(dev, CC110X_RXFIFO, (char *)status, 2);
112 
113  /* Store RSSI value of packet */
114  pkt_buf->rssi = status[I_RSSI];
115 
116  /* Bit 0-6 of LQI indicates the link quality (LQI) */
117  pkt_buf->lqi = status[I_LQI] & LQI_EST;
118 
119  /* MSB of LQI is the CRC_OK bit */
120  int crc_ok = (status[I_LQI] & CRC_OK) >> 7;
121 
122  if (crc_ok) {
123  LOG_DEBUG("cc110x: received packet from=%u to=%u payload "
124  "len=%u\n",
125  (unsigned)pkt_buf->packet.phy_src,
126  (unsigned)pkt_buf->packet.address,
127  pkt_buf->packet.length-3);
128  /* let someone know that we've got a packet */
129  callback(arg);
130 
131  cc110x_switch_to_rx(dev);
132  }
133  else {
134  DEBUG("%s:%s:%u crc-error\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
135  dev->cc110x_statistic.packets_in_crc_fail++;
136  _rx_abort(dev);
137  }
138  }
139 }

141 static void _rx_continue(cc110x_t *dev, void(*callback)(void*), void*arg)
142 {
143 
144  if (dev->radio_state != RADIO_RX_BUSY) {
145  DEBUG("%s:%s:%u _rx_continue in invalid state\n", RIOT_FILE_RELATIVE,
146  __func__, __LINE__);
147  _rx_abort(dev);
148  return;
149  }
150 
151  gpio_irq_disable(dev->params.gdo2);
152 
153  do {
154  _rx_read_data(dev, callback, arg);
155  }
156  while (gpio_read(dev->params.gdo2));
157 }

164 static void _tx_continue(cc110x_t *dev)
165 {
166  gpio_irq_disable(dev->params.gdo2);
167 
168  cc110x_pkt_t *pkt = &dev->pkt_buf.packet;
169  int size = pkt->length + 1;
170  int left = size - dev->pkt_buf.pos;
171 
172  if (!left) {
173  dev->cc110x_statistic.raw_packets_out++;
174 
175  LOG_DEBUG("cc110x: packet successfully sent.\n");
176 
177  cc110x_switch_to_rx(dev);
178  return;
179  }
180 
181  int fifo = 64 - cc110x_get_reg_robust(dev, 0xfa);
182 
183  if (fifo & 0x80) {
184  DEBUG("%s:%s:%u tx underflow!\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
185  _tx_abort(dev);
186  return;
187  }
188 
189  if (!fifo) {
190  DEBUG("%s:%s:%u fifo full!?\n", RIOT_FILE_RELATIVE, __func__, __LINE__);
191  _tx_abort(dev);
192  return;
193  }
194 
195  int to_send = left > fifo ? fifo : left;
196 
197  /* Write packet into TX FIFO */
198  cc110x_writeburst_reg(dev, CC110X_TXFIFO, ((char *)pkt)+dev->pkt_buf.pos, to_send);
199  dev->pkt_buf.pos += to_send;
200 
201  if (left == size) {
202  /* Switch to TX mode */
203  cc110x_strobe(dev, CC110X_STX);
204  }
205 
206  if (to_send < left) {
207  /* set GDO2 to 0x2 -> will deassert at TX FIFO below threshold */
208  gpio_irq_enable(dev->params.gdo2);
209  cc110x_write_reg(dev, CC110X_IOCFG2, 0x02);
210  }
211  else {
212  /* set GDO2 to 0x6 -> will deassert at packet end */
213  cc110x_write_reg(dev, CC110X_IOCFG2, 0x06);
214  gpio_irq_enable(dev->params.gdo2);
215  }
216 }

218 void cc110x_isr_handler(cc110x_t *dev, void(*callback)(void*), void*arg)
219 {
220  switch (dev->radio_state) {
221  case RADIO_RX:
222  if (gpio_read(dev->params.gdo2)) {
223  _rx_start(dev);
224  }
225  else {
226  DEBUG("cc110x_isr_handler((): isr handled too slow?\n");
227  _rx_abort(dev);
228  }
229  break;
230  case RADIO_RX_BUSY:
231  _rx_continue(dev, callback, arg);
232  break;
233  case RADIO_TX_BUSY:
234  if (!gpio_read(dev->params.gdo2)) {
235  _tx_continue(dev);
236  }
237  else {
238  DEBUG("cc110x_isr_handler() RADIO_TX_BUSY + GDO2\n");
239  }
240  break;
241  default:
242  DEBUG("%s:%s:%u: unhandled mode\n", RIOT_FILE_RELATIVE,
243  __func__, __LINE__);
244  }
245 }

247 int cc110x_send(cc110x_t *dev, cc110x_pkt_t *packet)
248 {
249  DEBUG("cc110x: snd pkt to %u payload_length=%u\n",
250  (unsigned)packet->address, (unsigned)packet->length-3);
251  uint8_t size;
252 
253  switch (dev->radio_state) {
254  case RADIO_RX_BUSY:
255  case RADIO_TX_BUSY:
256  DEBUG("cc110x: invalid state for sending: %s\n",
257  cc110x_state_to_text(dev->radio_state));
258  return -EAGAIN;
259  }
260 
261  /*
262  * Number of bytes to send is:
263  * length of phy payload (packet->length)
264  * + size of length field (1 byte)
265  */
266  size = packet->length + 1;
267 
268  if (size > CC110X_PACKET_LENGTH) {
269  DEBUG("%s:%s:%u trying to send oversized packet\n",
270  RIOT_FILE_RELATIVE, __func__, __LINE__);
271  return -ENOSPC;
272  }
273 
274  /* set source address */
275  packet->phy_src = dev->radio_address;
276 
277  /* Disable RX interrupt */
278  gpio_irq_disable(dev->params.gdo2);
279  dev->radio_state = RADIO_TX_BUSY;
280 
281 #ifdef MODULE_CC110X_HOOKS
282  cc110x_hook_tx();
283 #endif
284 
285  cc110x_write_reg(dev, CC110X_IOCFG2, 0x02);
286 
287  /* Put CC110x in IDLE mode to flush the FIFO */
288  cc110x_strobe(dev, CC110X_SIDLE);
289  /* Flush TX FIFO to be sure it is empty */
290  cc110x_strobe(dev, CC110X_SFTX);
291 
292  memcpy((char*)&dev->pkt_buf.packet, packet, size);
293  dev->pkt_buf.pos = 0;
294 
295  _tx_continue(dev);
296 
297  return size;
298 }
