112 int gpio_init(gpio_t pin, gpio_mode_t mode)
113 {
114  switch (mode) {
115  case GPIO_OUT:
116  _SFR_MEM8(_ddr_addr(pin)) |= (1 << _pin_num(pin));
117  break;
118  case GPIO_IN:
119  _SFR_MEM8(_ddr_addr(pin)) &= ~(1 << _pin_num(pin));
120  _SFR_MEM8(_port_addr(pin)) &= ~(1 << _pin_num(pin));
121  break;
122  case GPIO_IN_PU:
123  _SFR_MEM8(_port_addr(pin)) |= (1 << _pin_num(pin));
124  break;
125  default:
126  return -1;
127  }
128 
129  return 0;
130 }

132 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
133  gpio_cb_t cb, void *arg)
134 {
135  uint8_t pin_num = _pin_num(pin);
136 
137  if ((_port_num(pin) == PORT_D && pin_num > 3)
138 #if defined (PORTE)
139  || (_port_num(pin) == PORT_E && pin_num < 4)
140 #endif
141  || ((mode != GPIO_IN) && (mode != GPIO_IN_PU))) {
142  return -1;
143  }
144 
145  gpio_init(pin, mode);
146 
147  /* clear global interrupt flag */
148  cli();
149 
150  EIMSK |= (1 << pin_num);
151 
152  /* configure the flank */
153  switch (flank) {
154  case GPIO_RISING:
155  if (pin_num < 4) {
156  EICRA |= (3 << pin_num * 2);
157  }
158 #if defined(EICRB)
159  else {
160  EICRB |= (3 << (pin_num * 2) % 4);
161  }
162 #endif
163  break;
164  case GPIO_FALLING:
165  if (pin_num < 4) {
166  EICRA |= (2 << pin_num * 2);
167  }
168 #if defined(EICRB)
169  else {
170  EICRB |= (2 << (pin_num * 2) % 4);
171  }
172 #endif
173  break;
174  case GPIO_BOTH:
175  if (pin_num < 4) {
176  EICRA |= (1 << pin_num * 2);
177  }
178 #if defined(EICRB)
179  else {
180  EICRB |= (1 << (pin_num * 2) % 4);
181  }
182 #endif
183  break;
184  default:
185  return -1;
186  };
187 
188  /* set callback */
189  config[pin_num].cb = cb;
190  config[pin_num].arg = arg;
191 
192  /* set global interrupt flag */
193  sei();
194 
195  return 0;
196 }
