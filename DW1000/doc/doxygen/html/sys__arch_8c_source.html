77 u32_t sys_arch_sem_wait(sys_sem_t *sem, u32_t count)
78 {
79  LWIP_ASSERT("invalid semaphor", sys_sem_valid(sem));
80  if (count != 0) {
81  uint64_t stop, start;
82  start = xtimer_now_usec64();
83  int res = sema_wait_timed((sema_t *)sem, count * US_PER_MS);
84  stop = xtimer_now_usec64() - start;
85  if (res == -ETIMEDOUT) {
86  return SYS_ARCH_TIMEOUT;
87  }
88  return (u32_t)(stop / US_PER_MS);
89  }
90  else {
91  sema_wait_timed((sema_t *)sem, 0);
92  return 0;
93  }

135 u32_t sys_arch_mbox_fetch(sys_mbox_t *mbox, void **msg, u32_t timeout)
136 {
137  msg_t m;
138  xtimer_t timer = { .callback = _mbox_timeout, .arg = &mbox->mbox };
139  uint64_t start, stop;
140 
141  start = xtimer_now_usec64();
142  if (timeout > 0) {
143  uint64_t u_timeout = (timeout * US_PER_MS);
144  _xtimer_set64(&timer, (uint32_t)u_timeout, (uint32_t)(u_timeout >> 32));
145  }
146  mbox_get(&mbox->mbox, &m);
147  stop = xtimer_now_usec64();
148  xtimer_remove(&timer); /* in case timer did not time out */
149  switch (m.type) {
150  case _MSG_SUCCESS:
151  *msg = m.content.ptr;
152  return (u32_t)((stop - start) / US_PER_MS);
153  case _MSG_TIMEOUT:
154  break;
155  default: /* should not happen */
156  LWIP_ASSERT("invalid message received", false);
157  break;
158  }
159  return SYS_ARCH_TIMEOUT;
160 }
