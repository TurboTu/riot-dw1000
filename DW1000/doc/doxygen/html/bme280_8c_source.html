68 int bme280_init(bme280_t* dev, const bme280_params_t* params)
69 {
70  uint8_t chip_id;
71 
72  dev->params = *params;
73 
74  /* Initialize I2C interface */
75  if (i2c_init_master(dev->params.i2c_dev, I2C_SPEED_NORMAL)) {
76  DEBUG("[Error] I2C device not enabled\n");
77  return BME280_ERR_I2C;
78  }
79 
80  /* Read chip ID */
81  chip_id = read_u8_reg(dev, BME280_CHIP_ID_REG);
82  if (chip_id != BME280_CHIP_ID) {
83  DEBUG("[Error] Did not detect a BME280 at address %02x (%02X != %02X)\n",
84  dev->params.i2c_addr, chip_id, BME280_CHIP_ID);
85  return BME280_ERR_NODEV;
86  }
87 
88  /* Read compensation data, 0x88..0x9F, 0xA1, 0xE1..0xE7 */
89  if (read_calibration_data(dev)) {
90  DEBUG("[Error] Could not read calibration data\n");
91  return BME280_ERR_NOCAL;
92  }
93 
94  return BME280_OK;
95 }

97 /*
98  * Returns temperature in DegC, resolution is 0.01 DegC.
99  * t_fine carries fine temperature as global value
100  */
101 int16_t bme280_read_temperature(bme280_t* dev)
102 {
103  if (do_measurement(dev) < 0) {
104  return INT16_MIN;
105  }
106 
107  bme280_calibration_t *cal = &dev->calibration; /* helper variable */
108 
109  /* Read the uncompensated temperature */
110  int32_t adc_T = (((uint32_t)measurement_regs[3 + 0]) << 12) |
111  (((uint32_t)measurement_regs[3 + 1]) << 4) |
112  ((((uint32_t)measurement_regs[3 + 2]) >> 4) & 0x0F);
113 
114  /*
115  * Compensate the temperature value.
116  * The following is code from Bosch's BME280_driver bme280_compensate_temperature_int32()
117  * The variable names and the many defines have been modified to make the code
118  * more readable.
119  */
120  int32_t var1;
121  int32_t var2;
122 
123  var1 = ((((adc_T >> 3) - ((int32_t)cal->dig_T1 << 1))) * ((int32_t)cal->dig_T2)) >> 11;
124  var2 = (((((adc_T >> 4) - ((int32_t)cal->dig_T1)) * ((adc_T >> 4) - ((int32_t)cal->dig_T1))) >> 12) *
125  ((int32_t)cal->dig_T3)) >> 14;
126 
127  /* calculate t_fine (used for pressure and humidity too) */
128  t_fine = var1 + var2;
129 
130  return (t_fine * 5 + 128) >> 8;
131 }

133 /*
134  * Returns pressure in Pa
135  */
136 uint32_t bme280_read_pressure(bme280_t *dev)
137 {
138  bme280_calibration_t *cal = &dev->calibration; /* helper variable */
139 
140  /* Read the uncompensated pressure */
141  int32_t adc_P = (((uint32_t)measurement_regs[0 + 0]) << 12) |
142  (((uint32_t)measurement_regs[0 + 1]) << 4) |
143  ((((uint32_t)measurement_regs[0 + 2]) >> 4) & 0x0F);
144 
145  int64_t var1;
146  int64_t var2;
147  int64_t p_acc;
148 
149  /*
150  * Compensate the pressure value.
151  * The following is code from Bosch's BME280_driver bme280_compensate_pressure_int64()
152  * The variable names and the many defines have been modified to make the code
153  * more readable.
154  */
155  var1 = ((int64_t)t_fine) - 128000;
156  var2 = var1 * var1 * (int64_t)cal->dig_P6;
157  var2 = var2 + ((var1 * (int64_t)cal->dig_P5) << 17);
158  var2 = var2 + (((int64_t)cal->dig_P4) << 35);
159  var1 = ((var1 * var1 * (int64_t)cal->dig_P3) >> 8) + ((var1 * (int64_t)cal->dig_P2) << 12);
160  var1 = (((((int64_t)1) << 47) + var1)) * ((int64_t)cal->dig_P1) >> 33;
161  /* Avoid division by zero */
162  if (var1 == 0) {
163  return UINT32_MAX;
164  }
165 
166  p_acc = 1048576 - adc_P;
167  p_acc = (((p_acc << 31) - var2) * 3125) / var1;
168  var1 = (((int64_t)cal->dig_P9) * (p_acc >> 13) * (p_acc >> 13)) >> 25;
169  var2 = (((int64_t)cal->dig_P8) * p_acc) >> 19;
170  p_acc = ((p_acc + var1 + var2) >> 8) + (((int64_t)cal->dig_P7) << 4);
171 
172  return p_acc >> 8;
173 }

175 uint16_t bme280_read_humidity(bme280_t *dev)
176 {
177  bme280_calibration_t *cal = &dev->calibration; /* helper variable */
178 
179  /* Read the uncompensated pressure */
180  int32_t adc_H = (((uint32_t)measurement_regs[6 + 0]) << 8) |
181  (((uint32_t)measurement_regs[6 + 1]));
182 
183  /*
184  * Compensate the humidity value.
185  * The following is code from Bosch's BME280_driver bme280_compensate_humidity_int32()
186  * The variable names and the many defines have been modified to make the code
187  * more readable.
188  * The value is first computed as a value in %rH as unsigned 32bit integer
189  * in Q22.10 format(22 integer 10 fractional bits).
190  */
191  int32_t var1;
192 
193  /* calculate x1*/
194  var1 = (t_fine - ((int32_t)76800));
195  /* calculate x1*/
196  var1 = (((((adc_H << 14) - (((int32_t)cal->dig_H4) << 20) - (((int32_t)cal->dig_H5) * var1)) +
197  ((int32_t)16384)) >> 15) *
198  (((((((var1 * ((int32_t)cal->dig_H6)) >> 10) *
199  (((var1 * ((int32_t)cal->dig_H3)) >> 11) + ((int32_t)32768))) >> 10) +
200  ((int32_t)2097152)) * ((int32_t)cal->dig_H2) + 8192) >> 14));
201  var1 = (var1 - (((((var1 >> 15) * (var1 >> 15)) >> 7) * ((int32_t)cal->dig_H1)) >> 4));
202  var1 = (var1 < 0 ? 0 : var1);
203  var1 = (var1 > 419430400 ? 419430400 : var1);
204  /* First multiply to avoid losing the accuracy after the shift by ten */
205  return (100 * ((uint32_t)var1 >> 12)) >> 10;
206 }

214 static int read_calibration_data(bme280_t* dev)
215 {
216  uint8_t buffer[128]; /* 128 should be enough to read all calibration bytes */
217  int nr_bytes;
218  int nr_bytes_to_read = (BME280_DIG_H6_REG - BME280_DIG_T1_LSB_REG) + 1;
219  uint8_t offset = 0x88;
220 
221  memset(buffer, 0, sizeof(buffer));
222  nr_bytes = i2c_read_regs(dev->params.i2c_dev, dev->params.i2c_addr, offset,
223  buffer, nr_bytes_to_read);
224  if (nr_bytes != nr_bytes_to_read) {
225  LOG_ERROR("Unable to read calibration data\n");
226  return -1;
227  }
228  DUMP_BUFFER("Raw Calibration Data", buffer, nr_bytes);
229 
230  /* All little endian */
231  dev->calibration.dig_T1 = get_uint16_le(buffer, BME280_DIG_T1_LSB_REG - offset);
232  dev->calibration.dig_T2 = get_int16_le(buffer, BME280_DIG_T2_LSB_REG - offset);
233  dev->calibration.dig_T3 = get_int16_le(buffer, BME280_DIG_T3_LSB_REG - offset);
234 
235  dev->calibration.dig_P1 = get_uint16_le(buffer, BME280_DIG_P1_LSB_REG - offset);
236  dev->calibration.dig_P2 = get_int16_le(buffer, BME280_DIG_P2_LSB_REG - offset);
237  dev->calibration.dig_P3 = get_int16_le(buffer, BME280_DIG_P3_LSB_REG - offset);
238  dev->calibration.dig_P4 = get_int16_le(buffer, BME280_DIG_P4_LSB_REG - offset);
239  dev->calibration.dig_P5 = get_int16_le(buffer, BME280_DIG_P5_LSB_REG - offset);
240  dev->calibration.dig_P6 = get_int16_le(buffer, BME280_DIG_P6_LSB_REG - offset);
241  dev->calibration.dig_P7 = get_int16_le(buffer, BME280_DIG_P7_LSB_REG - offset);
242  dev->calibration.dig_P8 = get_int16_le(buffer, BME280_DIG_P8_LSB_REG - offset);
243  dev->calibration.dig_P9 = get_int16_le(buffer, BME280_DIG_P9_LSB_REG - offset);
244 
245  dev->calibration.dig_H1 = buffer[BME280_DIG_H1_REG - offset];
246  dev->calibration.dig_H2 = get_int16_le(buffer, BME280_DIG_H2_LSB_REG - offset);
247  dev->calibration.dig_H3 = buffer[BME280_DIG_H3_REG - offset];
248  dev->calibration.dig_H4 = (((int16_t)buffer[BME280_DIG_H4_MSB_REG - offset]) << 4) +
249  (buffer[BME280_DIG_H4_H5_REG - offset] & 0x0F);
250  dev->calibration.dig_H5 = (((int16_t)buffer[BME280_DIG_H5_MSB_REG - offset]) << 4) +
251  ((buffer[BME280_DIG_H4_H5_REG - offset] & 0xF0) >> 4);
252  dev->calibration.dig_H6 = buffer[BME280_DIG_H6_REG - offset];
253 
254  DEBUG("[INFO] Chip ID = 0x%02X\n", buffer[BME280_CHIP_ID_REG - offset]);
255 
256  /* Config is only be writable in sleep mode */
257  (void)i2c_write_reg(dev->params.i2c_dev, dev->params.i2c_addr,
258  BME280_CTRL_MEAS_REG, 0);
259 
260  uint8_t b;
261 
262  /* Config Register */
263  /* spi3w_en unused */
264  b = ((dev->params.t_sb & 7) << 5) | ((dev->params.filter & 7) << 2);
265  write_u8_reg(dev, BME280_CONFIG_REG, b);
266 
267  /*
268  * Note from the datasheet about ctrl_hum: "Changes to this register only become effective
269  * after a write operation to "ctrl_meas".
270  * So, set ctrl_hum first.
271  */
272  b = dev->params.humid_oversample & 7;
273  write_u8_reg(dev, BME280_CTRL_HUMIDITY_REG, b);
274 
275  b = ((dev->params.temp_oversample & 7) << 5) |
276  ((dev->params.press_oversample & 7) << 2) |
277  (dev->params.run_mode & 3);
278  write_u8_reg(dev, BME280_CTRL_MEAS_REG, b);
279 
280  return 0;
281 }

286 static int do_measurement(bme280_t* dev)
287 {
288  /*
289  * If settings has FORCED mode, then the device go to sleep after
290  * it finished the measurement. To read again we have to set the
291  * run_mode back to FORCED.
292  */
293  uint8_t ctrl_meas = get_ctrl_meas(dev);
294  uint8_t run_mode = ctrl_meas & 3;
295  if (run_mode != dev->params.run_mode) {
296  /* Set the run_mode back to what we want. */
297  ctrl_meas &= ~3;
298  ctrl_meas |= dev->params.run_mode;
299  write_u8_reg(dev, BME280_CTRL_MEAS_REG, ctrl_meas);
300 
301  /* Wait for measurement ready? */
302  size_t count = 0;
303  while (count < 10 && (get_status(dev) & 0x08) != 0) {
304  ++count;
305  }
306  /* What to do when measuring is still on? */
307  }
308  int nr_bytes;
309  int nr_bytes_to_read = sizeof(measurement_regs);
310  uint8_t offset = BME280_PRESSURE_MSB_REG;
311 
312  nr_bytes = i2c_read_regs(dev->params.i2c_dev, dev->params.i2c_addr,
313  offset, measurement_regs, nr_bytes_to_read);
314  if (nr_bytes != nr_bytes_to_read) {
315  LOG_ERROR("Unable to read temperature data\n");
316  return -1;
317  }
318  DUMP_BUFFER("Raw Sensor Data", measurement_regs, nr_bytes);
319 
320  return 0;
321 }
