303 static gnrc_rpl_parent_t *_gnrc_rpl_find_preferred_parent(gnrc_rpl_dodag_t *dodag)
304 {
305  gnrc_rpl_parent_t *old_best = dodag->parents;
306  gnrc_rpl_parent_t *new_best = old_best;
307  uint16_t old_rank = dodag->my_rank;
308  gnrc_rpl_parent_t *elt, *tmp;
309 
310  if (dodag->parents == NULL) {
311  return NULL;
312  }
313 
314  LL_FOREACH(dodag->parents, elt) {
315  new_best = dodag->instance->of->which_parent(new_best, elt);
316  }
317 
318  if (new_best->rank == GNRC_RPL_INFINITE_RANK) {
319  return NULL;
320  }
321 
322  if (new_best != old_best) {
323  LL_DELETE(dodag->parents, new_best);
324  LL_PREPEND(dodag->parents, new_best);
325  /* no-path DAOs only for the storing mode */
326  if ((dodag->instance->mop == GNRC_RPL_MOP_STORING_MODE_NO_MC) ||
327  (dodag->instance->mop == GNRC_RPL_MOP_STORING_MODE_MC)) {
328  gnrc_rpl_send_DAO(dodag->instance, &old_best->addr, 0);
329  gnrc_rpl_delay_dao(dodag);
330  }
331 
332 #ifdef MODULE_GNRC_RPL_P2P
333  if (dodag->instance->mop != GNRC_RPL_P2P_MOP) {
334 #endif
335  fib_add_entry(&gnrc_ipv6_fib_table,
336  dodag->iface,
337  (uint8_t *) ipv6_addr_unspecified.u8,
338  sizeof(ipv6_addr_t),
339  0x00,
340  dodag->parents->addr.u8,
341  sizeof(ipv6_addr_t),
342  FIB_FLAG_RPL_ROUTE,
343  (dodag->default_lifetime * dodag->lifetime_unit) * MS_PER_SEC);
344 #ifdef MODULE_GNRC_RPL_P2P
345  }
346 #endif
347 
348  }
349 
350  dodag->my_rank = dodag->instance->of->calc_rank(dodag->parents, 0);
351  if (dodag->my_rank != old_rank) {
352  trickle_reset_timer(&dodag->trickle);
353  }
354 
355  LL_FOREACH_SAFE(dodag->parents, elt, tmp) {
356  if (DAGRANK(dodag->my_rank, dodag->instance->min_hop_rank_inc)
357  <= DAGRANK(elt->rank, dodag->instance->min_hop_rank_inc)) {
358  gnrc_rpl_parent_remove(elt);
359  }
360  }
361 
362  return dodag->parents;
363 }
