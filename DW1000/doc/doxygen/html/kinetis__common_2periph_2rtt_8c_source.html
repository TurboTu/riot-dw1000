53 void rtt_init(void)
54 {
55  RTC_Type *rtt = RTT_DEV;
56 
57  RTT_UNLOCK();
58  /* Reset RTC */
59  rtt->CR = RTC_CR_SWR_MASK;
60  rtt->CR = 0;
61 
62  if (rtt->SR & RTC_SR_TIF_MASK) {
63  /* Clear TIF by writing TSR. */
64  rtt->TSR = 0;
65  }
66 
67  /* Enable RTC oscillator and non-supervisor mode accesses. */
68  /* Enable load capacitance as configured by periph_conf.h */
69  rtt->CR = RTC_CR_OSCE_MASK | RTC_CR_SUP_MASK | RTC_LOAD_CAP_BITS;
70 
71  /* Clear TAF by writing TAR. */
72  rtt->TAR = 0xffffff42;
73 
74  /* Disable all RTC interrupts. */
75  rtt->IER = 0;
76 
77  rtt_poweron();
78 }

96 uint32_t rtt_get_counter(void)
97 {
98  RTC_Type *rtt = RTT_DEV;
99  uint32_t t;
100  for (int i = 0; i < 3; i++) {
101  /* Read twice to make sure we get a stable reading */
102  t = rtt->TSR;
103 
104  if (t == rtt->TSR) {

107  }
108  /* Fallback if we are not getting stable readings */

177 void RTT_ISR(void)
178 {
179  RTC_Type *rtt = RTT_DEV;
180 
181  if (rtt->SR & RTC_SR_TAF_MASK) {
182  if (rtt_callback.alarm_cb != NULL) {
183  /* Disable Timer Alarm Interrupt */
184  rtt->IER &= ~RTC_IER_TAIE_MASK;
185  rtt_callback.alarm_cb(rtt_callback.alarm_arg);
186  }
187  }
188 
189  if (rtt->SR & RTC_SR_TOF_MASK) {
190  if (rtt_callback.overflow_cb != NULL) {
191  rtt_callback.overflow_cb(rtt_callback.overflow_arg);
192  }
193  }
194 
195  cortexm_isr_end();
