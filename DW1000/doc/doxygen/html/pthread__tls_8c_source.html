74 static tls_data_t *get_specific(pthread_key_t key)
75 {
76  pthread_t self_id = pthread_self();
77  if (self_id == 0) {
78  DEBUG("ERROR called pthread_self() returned 0 in \"%s\"!\n", __func__);
79  return NULL;
80  }
81 
82  tls_data_t **tls = __pthread_get_tls_head(self_id);
83  tls_data_t *prev, *specific = find_specific(tls, key, &prev);
84 
85  /* Did the datum already exist? */
86  if (specific) {
87  if (prev) {
88  /* Move the datum to the front for a faster next lookup. */
89  /* Let's pretend that we have a totally degenerated splay tree. ;-) */
90  prev->next = specific->next;
91  specific->next = *tls;
92  *tls = specific;
93  }
94  return specific;
95  }
96 
97  /* Allocate new datum. */
98  specific = malloc(sizeof (*specific));
99  if (specific) {
100  specific->key = key;
101  specific->next = *tls;
102  specific->value = NULL;
103  *tls = specific;
104  }
105  else {
106  DEBUG("ERROR out of memory in %s!\n", __func__);
107  }
108  return specific;
109 }

182 {

184 
185  /* Calling the dtor could cause another pthread_exit(), so we dehead and free defore calling it. */
186  mutex_lock(&tls_mutex);

188  *tls = specific->next;

191  free(specific);
