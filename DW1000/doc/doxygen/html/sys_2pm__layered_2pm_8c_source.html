49 void pm_set_lowest(void)
50 {
51  pm_blocker_t blocker = (pm_blocker_t) pm_blocker;
52  unsigned mode = PM_NUM_MODES;
53  while (mode) {
54  if (blocker.val_u8[mode-1]) {
55  break;
56  }
57  mode--;
58  }
59 
60  /* set lowest mode if blocker is still the same */
61  unsigned state = irq_disable();
62  if (blocker.val_u32 == pm_blocker.val_u32) {
63  DEBUG("pm: setting mode %u\n", mode);
64  pm_set(mode);
65  }
66  else {
67  DEBUG("pm: mode block changed\n");
68  }
69  irq_restore(state);
