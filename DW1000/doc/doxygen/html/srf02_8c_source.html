60 int srf02_init(srf02_t *dev, i2c_t i2c, uint8_t addr)
61 {
62  dev->i2c = i2c;
63  dev->addr = (addr >> 1); /* internally we right align the 7-bit addr */
64  uint8_t rev;
65 
66  /* Acquire exclusive access to the bus. */
67  i2c_acquire(dev->i2c);
68  /* initialize i2c interface */
69  if (i2c_init_master(dev->i2c, BUS_SPEED) < 0) {
70  DEBUG("[srf02] error initializing I2C bus\n");
71  return -1;
72  }
73  /* try to read the software revision (read the CMD reg) from the device */
74  i2c_read_reg(i2c, dev->addr, REG_CMD, &rev);
75  if (rev == 0 || rev == 255) {
76  i2c_release(dev->i2c);
77  DEBUG("[srf02] error reading the devices software revision\n");
78  return -1;
79  } else {
80  DEBUG("[srf02] software revision: 0x%02x\n", rev);
81  }
82  /* Release the bus for other threads. */
83  i2c_release(dev->i2c);
84 
85  DEBUG("[srf02] initialization successful\n");
86  return 0;
87 }

98 uint16_t srf02_read(srf02_t *dev)
99 {
100  uint8_t res[2];
101 
102  /* read the results */
103  i2c_acquire(dev->i2c);
104  i2c_read_regs(dev->i2c, dev->addr, REG_HIGH, res, 2);
105  i2c_release(dev->i2c);
106  DEBUG("[srf02] result - high: 0x%02x low: 0x%02x\n", res[0], res[1]);
107 
108  /* compile result - TODO: fix for different host byte order other than LE */
109  return ((((uint16_t)res[0]) << 8) | (res[1] & 0xff));
110 }

122 void srf02_set_addr(srf02_t *dev, uint8_t new_addr)
123 {
124  /* get access to the bus */
125  i2c_acquire(dev->i2c);
126 
127  DEBUG("[srf02] reprogramming device address to 0x%02x\n", (int)new_addr);
128 
129  /* write the new address, for this we need to follow a certain sequence */
130  i2c_write_reg(dev->i2c, dev->addr, REG_CMD, CMD_ADDR_SEQ1);
131  i2c_write_reg(dev->i2c, dev->addr, REG_CMD, CMD_ADDR_SEQ2);
132  i2c_write_reg(dev->i2c, dev->addr, REG_CMD, CMD_ADDR_SEQ3);
133  i2c_write_reg(dev->i2c, dev->addr, REG_CMD, new_addr);
134  dev->addr = (new_addr >> 1);
135 
136  /* release the bus */
137  i2c_release(dev->i2c);
