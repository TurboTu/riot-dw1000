67 int i2c_init_master(i2c_t dev, i2c_speed_t speed)
68 {
69  int ccr;
70 
71  if ((unsigned int)dev >= I2C_NUMOF) {
72  return -1;
73  }
74 
75  /* read speed configuration */
76  switch (speed) {
77  case I2C_SPEED_NORMAL:
78  ccr = I2C_APBCLK / 200000;
79  break;
80 
81  case I2C_SPEED_FAST:
82  ccr = I2C_APBCLK / 800000;
83  break;
84 
85  default:
86  return -2;
87  }
88  I2C_TypeDef *i2c = i2c_config[dev].dev;
89 
90  /* enable I2C clock */
91  i2c_poweron(dev);
92 
93  /* set IRQn priority */
94  NVIC_SetPriority(i2c_config[dev].er_irqn, I2C_IRQ_PRIO);
95 
96  /* enable IRQn */
97  NVIC_EnableIRQ(i2c_config[dev].er_irqn);
98 
99  /* configure pins */
100  gpio_init(i2c_config[dev].scl, i2c_config[dev].pin_mode);
101  gpio_init_af(i2c_config[dev].scl, i2c_config[dev].af);
102  gpio_init(i2c_config[dev].sda, i2c_config[dev].pin_mode);
103  gpio_init_af(i2c_config[dev].sda, i2c_config[dev].af);
104 
105  /* configure device */
106  _i2c_init(i2c, ccr);
107 
108  return 0;
109 }

111 static void _i2c_init(I2C_TypeDef *i2c, int ccr)
112 {
113  /* disable device and set ACK bit */
114  i2c->CR1 = I2C_CR1_ACK;
115  /* configure I2C clock */
116  i2c->CR2 = (I2C_APBCLK / 1000000) | I2C_CR2_ITERREN;
117  i2c->CCR = ccr;
118  i2c->TRISE = (I2C_APBCLK / 1000000) + 1;
119  /* configure device */
120  i2c->OAR1 = 0; /* makes sure we are in 7-bit address mode */
121  /* enable device */
122  i2c->CR1 |= I2C_CR1_PE;
123 }

148 int i2c_read_bytes(i2c_t dev, uint8_t address, void *data, int length)
149 {
150  unsigned int state;
151  int i = 0;
152  uint8_t *my_data = data;
153 
154  if ((unsigned int)dev >= I2C_NUMOF) {
155  return -1;
156  }
157 
158  I2C_TypeDef *i2c = i2c_config[dev].dev;
159  switch (length) {
160  case 1:
161  DEBUG("Send Slave address and wait for ADDR == 1\n");
162  _start(i2c, address, I2C_FLAG_READ);
163 
164  DEBUG("Set ACK = 0\n");
165  i2c->CR1 &= ~(I2C_CR1_ACK);
166 
167  DEBUG("Clear ADDR and set STOP = 1\n");
168  state = irq_disable();
169  _clear_addr(i2c);
170  i2c->CR1 |= (I2C_CR1_STOP);
171  irq_restore(state);
172 
173  DEBUG("Wait for RXNE == 1\n");
174 
175  while (!(i2c->SR1 & I2C_SR1_RXNE)) {}
176 
177  DEBUG("Read received data\n");
178  *my_data = i2c->DR;
179 
180  /* wait until STOP is cleared by hardware */
181  while (i2c->CR1 & I2C_CR1_STOP) {}
182 
183  /* reset ACK to be able to receive new data */
184  i2c->CR1 |= (I2C_CR1_ACK);
185  break;
186 
187  case 2:
188  DEBUG("Send Slave address and wait for ADDR == 1\n");
189  _start(i2c, address, I2C_FLAG_READ);
190  DEBUG("Set POS bit\n");
191  i2c->CR1 |= (I2C_CR1_POS | I2C_CR1_ACK);
192  DEBUG("Crit block: Clear ADDR bit and clear ACK flag\n");
193  state = irq_disable();
194  _clear_addr(i2c);
195  i2c->CR1 &= ~(I2C_CR1_ACK);
196  irq_restore(state);
197 
198  DEBUG("Wait for transfer to be completed\n");
199 
200  while (!(i2c->SR1 & I2C_SR1_BTF)) {}
201 
202  DEBUG("Crit block: set STOP and read first byte\n");
203  state = irq_disable();
204  i2c->CR1 |= (I2C_CR1_STOP);
205  my_data[0] = i2c->DR;
206  irq_restore(state);
207 
208  DEBUG("read second byte\n");
209  my_data[1] = i2c->DR;
210 
211  DEBUG("wait for STOP bit to be cleared again\n");
212 
213  while (i2c->CR1 & I2C_CR1_STOP) {}
214 
215  DEBUG("reset POS = 0 and ACK = 1\n");
216  i2c->CR1 &= ~(I2C_CR1_POS);
217  i2c->CR1 |= (I2C_CR1_ACK);
218  break;
219 
220  default:
221  DEBUG("Send Slave address and wait for ADDR == 1\n");
222  _start(i2c, address, I2C_FLAG_READ);
223  _clear_addr(i2c);
224 
225  while (i < (length - 3)) {
226  DEBUG("Wait until byte was received\n");
227 
228  while (!(i2c->SR1 & I2C_SR1_RXNE)) {}
229 
230  DEBUG("Copy byte from DR\n");
231  my_data[i++] = i2c->DR;
232  }
233 
234  DEBUG("Reading the last 3 bytes, waiting for BTF flag\n");
235 
236  while (!(i2c->SR1 & I2C_SR1_BTF)) {}
237 
238  DEBUG("Disable ACK\n");
239  i2c->CR1 &= ~(I2C_CR1_ACK);
240 
241  DEBUG("Crit block: set STOP and read N-2 byte\n");
242  state = irq_disable();
243  my_data[i++] = i2c->DR;
244  i2c->CR1 |= (I2C_CR1_STOP);
245  irq_restore(state);
246 
247  DEBUG("Read N-1 byte\n");
248  my_data[i++] = i2c->DR;
249 
250  while (!(i2c->SR1 & I2C_SR1_RXNE)) {}
251 
252  DEBUG("Read last byte\n");
253 
254  my_data[i++] = i2c->DR;
255 
256  DEBUG("wait for STOP bit to be cleared again\n");
257 
258  while (i2c->CR1 & I2C_CR1_STOP) {}
259 
260  DEBUG("reset POS = 0 and ACK = 1\n");
261  i2c->CR1 &= ~(I2C_CR1_POS);
262  i2c->CR1 |= (I2C_CR1_ACK);
263  }
264 
265  return length;
266 }

359 static void _start(I2C_TypeDef *i2c, uint8_t address, uint8_t rw_flag)
360 {
361  /* wait for device to be ready */
362  DEBUG("Wait for device to be ready\n");
363 
364  while (i2c->SR2 & I2C_SR2_BUSY) {}
365 
366  /* generate start condition */
367  DEBUG("Generate start condition\n");
368  i2c->CR1 |= I2C_CR1_START;
369  DEBUG("Wait for SB flag to be set\n");
370 
371  while (!(i2c->SR1 & I2C_SR1_SB)) {}
372 
373  /* send address and read/write flag */
374  DEBUG("Send address\n");
375  i2c->DR = (address << 1) | rw_flag;
376  /* clear ADDR flag by reading first SR1 and then SR2 */
377  DEBUG("Wait for ADDR flag to be set\n");
378 
379  while (!(i2c->SR1 & I2C_SR1_ADDR)) {}
380 }
