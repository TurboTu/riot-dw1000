45 uint8_t ipv6_addr_match_prefix(const ipv6_addr_t *a, const ipv6_addr_t *b)
46 {
47  uint8_t prefix_len = 0;
48 
49  if ((a == NULL) || (b == NULL)) {
50  return 0;
51  }
52 
53  if (a == b) {
54  return 128;
55  }
56 
57  for (int i = 0; i < 16; i++) {
58  /* if bytes are equal add 8 */
59  if (a->u8[i] == b->u8[i]) {
60  prefix_len += 8;
61  }
62  else {
63  uint8_t xor = (a->u8[i] ^ b->u8[i]);
64 
65  /* while bits from byte equal add 1 */
66  for (int j = 0; j < 8; j++) {
67  if ((xor & 0x80) == 0) {
68  prefix_len++;
69  xor = xor << 1;
70  }
71  else {
72  break;
73  }
74  }
75 
76  break;
77  }
78  }
79 
80  return prefix_len;
81 }

83 void ipv6_addr_init_prefix(ipv6_addr_t *out, const ipv6_addr_t *prefix,
84  uint8_t bits)
85 {
86  uint8_t bytes;
87 
88  if (bits > 128) {
89  bits = 128;
90  }
91 
92  bytes = bits / 8;
93 
94  memcpy(out, prefix, bytes);
95 
96  if (bits % 8) {
97  uint8_t mask = 0xff << (8 - (bits - (bytes * 8)));
98 
99  out->u8[bytes] &= ~mask;
100  out->u8[bytes] |= (prefix->u8[bytes] & mask);
101  }

104 void ipv6_addr_init_iid(ipv6_addr_t *out, const uint8_t *iid, uint8_t bits)
105 {
106  uint8_t unaligned_bits, bytes, pos;
107 
108  if (bits > 128) {
109  bits = 128;
110  }
111 
112  unaligned_bits = bits % 8;
113  bytes = bits / 8;
114  pos = (IPV6_ADDR_BIT_LEN / 8) - bytes;
115 
116  if (unaligned_bits) {
117  uint8_t mask = 0xff << unaligned_bits;
118  out->u8[pos - 1] &= mask;
119  out->u8[pos - 1] |= (*iid & ~mask);
120  iid++;
121  }
122 
123  memcpy(&(out->u8[pos]), iid, bytes);
124 }
