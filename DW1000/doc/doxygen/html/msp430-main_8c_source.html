127 /*
128  * Allocate memory from the heap. Check that we don't collide with the
129  * stack right now (some other routine might later). A watchdog might
130  * be used to check if cur_break and the stack pointer meet during
131  * runtime.
132  */
133 void *sbrk(int incr)
134 {
135  char *stack_pointer;
136 
137  asmv("mov r1, %0" : "=r"(stack_pointer));
138  stack_pointer -= STACK_EXTRA;
139 
140  if (incr > (stack_pointer - cur_break)) {
141  return (void *) - 1; /* ENOMEM */
142  }
143 
144  void *old_break = cur_break;
145  cur_break += incr;
146  /*
147  * If the stack was never here then [old_break .. cur_break] should
148  * be filled with zeros.
149  */
150  return old_break;
151 }

154  * Mask all interrupts that can be masked.

156 int
157 splhigh_(void)
158 {
159  /* Clear the GIE (General Interrupt Enable) flag. */
160  int sr;
161  asmv("mov r2, %0" : "=r"(sr));
162  asmv("bic %0, r2" : : "i"(GIE));
163  asmv("nop");
164  return sr & GIE; /* Ignore other sr bits. */
165 }
