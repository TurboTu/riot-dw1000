31 void isr_rfcoreerr(void)
32 {
33  uint_fast8_t flags;
34 
35  /* Latch and clear the interrupt status */
36  flags = RFCORE_SFR_RFERRF;
37  RFCORE_SFR_RFERRF = 0;
38 
39  /* These conditions shouldn't happen if the driver is working properly */
40  RFCORE_ASSERT(NOT(flags & RXUNDERF));
41  RFCORE_ASSERT(NOT(flags & TXOVERF));
42  RFCORE_ASSERT(NOT(flags & TXUNDERF));
43 
44  /* Fail in case of miscallaneous unexpected error conditions */
45  RFCORE_ASSERT(NOT(flags & STROBE_ERR));
46 
47  DEBUG("%s(): RFERRF=0x%02x.\n", __FUNCTION__, flags);
48 
49  /* Flush the receive FIFO in case of a receive overflow */
50  if (flags & RXOVERF) {
51  DEBUG("%s(): RXOVERF detected!\n", __FUNCTION__);
52  RFCORE_FLUSH_RECEIVE_FIFO();
53  }
54 
55  /* Flush the receive FIFO in case of a receive overflow */
56  if (flags & NLOCK) {
57  /* The frequency synthesizer failed to achieve lock after time-out, or
58  * lock is lost during reception. The receiver must be restarted to clear
59  * this error situation. */
60  DEBUG("%s(): NLOCK detected!\n", __FUNCTION__);
61  RFCORE_FLUSH_RECEIVE_FIFO();
62  }
63 }
