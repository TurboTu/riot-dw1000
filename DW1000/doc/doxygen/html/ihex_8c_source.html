120 int
121 parse_hex_line(char *line)
122 {
123  int addr, code, num;
124  int sum, len, cksum, i;
125  char *ptr;
126 
127  num = 0;
128 
129  if (line[0] != ':') {
130  return 0;
131  }
132 
133  if (strlen(line) < 11) {
134  return 0;
135  }
136 
137  ptr = line + 1;
138 
139  if (!sscanf(ptr, "%02x", &len)) {
140  return 0;
141  }
142 
143  ptr += 2;
144 
145  if (strlen(line) < (11 + (len * 2))) {
146  return 0;
147  }
148 
149  if (!sscanf(ptr, "%04x", &addr)) {
150  return 0;
151  }
152 
153  ptr += 4;
154 
155  /* printf("Line: length=%d Addr=%d\n", len, addr); */
156  if (!sscanf(ptr, "%02x", &code)) {
157  return 0;
158  }
159 
160  if (addr + extended_addr + len >= MAX_MEMORY_SIZE) {
161  return 0;
162  }
163 
164  ptr += 2;
165  sum = (len & 255) + ((addr >> 8) & 255) + (addr & 255) + (code & 255);
166 
167  if (code != 0) {
168  if (code == 1) {
169  end_record_seen = 1;
170  return 1;
171  }
172 
173  if (code == 2 && len == 2) {
174  if (!sscanf(ptr, "%04x", &i)) {
175  return 1;
176  }
177 
178  ptr += 4;
179  sum += ((i >> 8) & 255) + (i & 255);
180 
181  if (!sscanf(ptr, "%02x", &cksum)) {
182  return 1;
183  }
184 
185  if (((sum & 255) + (cksum & 255)) & 255) {
186  return 1;
187  }
188 
189  extended_addr = i << 4;
190  //printf("ext addr = %05X\n", extended_addr);
191  }
192 
193  if (code == 4 && len == 2) {
194  if (!sscanf(ptr, "%04x", &i)) {
195  return 1;
196  }
197 
198  ptr += 4;
199  sum += ((i >> 8) & 255) + (i & 255);
200 
201  if (!sscanf(ptr, "%02x", &cksum)) {
202  return 1;
203  }
204 
205  if (((sum & 255) + (cksum & 255)) & 255) {
206  return 1;
207  }
208 
209  extended_addr = i << 16;
210  //printf("ext addr = %08X\n", extended_addr);
211  }
212 
213  return 1; // non-data line
214  }
215 
216  byte_count += len;
217 
218  while (num != len) {
219  if (sscanf(ptr, "%02x", &i) != 1) {
220  return 0;
221  }
222 
223  i &= 255;
224  firmware_image[addr + extended_addr + num] = i;
225  firmware_mask[addr + extended_addr + num] = 1;
226  ptr += 2;
227  sum += i;
228  (num)++;
229 
230  if (num >= 256) {
231  return 0;
232  }
233  }
234 
235  if (!sscanf(ptr, "%02x", &cksum)) {
236  return 0;
237  }
238 
239  if (((sum & 255) + (cksum & 255)) & 255) {
240  return 0; /* checksum error */
241  }
242 
243  return 1;
244 }
