75 int gpio_init(gpio_t pin, gpio_mode_t mode)
76 {
77  GPIO_TypeDef *port = _port(pin);
78  int pin_num = _pin_num(pin);
79 
80  /* open-drain output with pull-up is not supported */
81  if (mode == GPIO_OD_PU) {
82  return -1;
83  }
84 
85  /* enable the clock for the selected port */
86  periph_clk_en(APB2, (RCC_APB2ENR_IOPAEN << _port_num(pin)));
87 
88  /* set pin mode */
89  port->CR[pin_num >> 3] &= ~(0xf << ((pin_num & 0x7) * 4));
90  port->CR[pin_num >> 3] |= ((mode & MODE_MASK) << ((pin_num & 0x7) * 4));
91  /* set initial state of output register */
92  port->BRR = (1 << pin_num);
93  port->BSRR = ((mode >> ODR_POS) << pin_num);
94 
95  return 0; /* all OK */
96 }

98 int gpio_init_int(gpio_t pin, gpio_mode_t mode, gpio_flank_t flank,
99  gpio_cb_t cb, void *arg)
100 {
101  int pin_num = _pin_num(pin);
102 
103  /* disable interrupts on the channel we want to edit (just in case) */
104  EXTI->IMR &= ~(1 << pin_num);
105  /* configure pin as input */
106  gpio_init(pin, mode);
107  /* set callback */
108  exti_ctx[pin_num].cb = cb;
109  exti_ctx[pin_num].arg = arg;
110  /* enable alternate function clock for the GPIO module */
111  periph_clk_en(APB2, RCC_APB2ENR_AFIOEN);
112  /* configure the EXTI channel */
113  AFIO->EXTICR[pin_num >> 2] &= ~(0xf << ((pin_num & 0x3) * 4));
114  AFIO->EXTICR[pin_num >> 2] |= (_port_num(pin) << ((pin_num & 0x3) * 4));
115  /* configure the active flank */
116  EXTI->RTSR &= ~(1 << pin_num);
117  EXTI->RTSR |= ((flank & 0x1) << pin_num);
118  EXTI->FTSR &= ~(1 << pin_num);
119  EXTI->FTSR |= ((flank >> 1) << pin_num);
120  /* active global interrupt for the selected port */
121  if (pin_num < 5) {
122  NVIC_EnableIRQ(EXTI0_IRQn + pin_num);
123  }
124  else if (pin_num < 10) {
125  NVIC_EnableIRQ(EXTI9_5_IRQn);
126  }
127  else {
128  NVIC_EnableIRQ(EXTI15_10_IRQn);
129  }
130  /* clear event mask */
131  EXTI->EMR &= ~(1 << pin_num);
132  /* unmask the pins interrupt channel */
133  EXTI->IMR |= (1 << pin_num);
134  return 0;
135 }
