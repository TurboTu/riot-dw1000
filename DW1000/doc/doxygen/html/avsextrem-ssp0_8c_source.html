76 uint8_t SSP0Prepare(uint8_t chip, uint8_t datasize, uint8_t cpol, uint8_t cpha,
77  uint16_t freq)
78 {
79  switch (chip) {
80  case BMA180_INTERN:
81  case SMB380_ACC: {
82 #if USE_CS
83  PINSEL3 |= BIT8 | BIT9 | BIT10 | BIT11 | BIT14 | BIT15 | BIT16 |
84  BIT17; //P1.20 1.21 1.23 1.24
85 #else
86  // 1.20 1.23 1.24 are not configured as SSEL0
87  PINSEL3 |= BIT8 | BIT9 | BIT14 | BIT15 | BIT16 | BIT17;
88 #endif
89  break;
90  }
91 
92  case BMA180_EXTERN:
93  case L3G_EXTERN:
94  case NANOPAN:
95  case ACAMDMS: {
96 #if USE_CS
97  PINSEL0 |= BIT31;
98  PINSEL1 |= BIT1 | BIT3 | BIT5; // P0.15 0.16 0.17 0.18
99 #else
100  // Turn on NanoPAN
101  PINSEL0 |= BIT31;
102  // 0.15 0.17 0.18 are not configured as SSEL0
103  PINSEL1 |= BIT3 | BIT5;
104 #endif
105  break;
106  }
107 
108  case NORDIC: {
109  PINSEL0 |= BIT31;
110  PINSEL1 |= BIT3 | BIT5; // 0.15 0.17 0.18 SSEL0 (No)
111  break;
112  }
113 
114  default: {
115  printf("wrong CHIP selected\n");
116  return 0;
117  }
118  }
119 
120  uint32_t SSP0CR0tmp = 0;
121 
122  switch (datasize) {
123  case 4:
124  SSP0CR0tmp = BIT0 | BIT1;
125  break;
126 
127  case 5:
128  SSP0CR0tmp = BIT2;
129  break;
130 
131  case 6:
132  SSP0CR0tmp = BIT0 | BIT2;
133  break;
134 
135  case 7:
136  SSP0CR0tmp = BIT1 | BIT2;
137  break;
138 
139  case 8:
140  SSP0CR0tmp = BIT0 | BIT1 | BIT2;
141  break;
142 
143  case 9:
144  SSP0CR0tmp = BIT3;
145  break;
146 
147  case 10:
148  SSP0CR0tmp = BIT0 | BIT3;
149  break;
150 
151  case 11:
152  SSP0CR0tmp = BIT1 | BIT3;
153  break;
154 
155  case 12:
156  SSP0CR0tmp = BIT0 | BIT1 | BIT3;
157  break;
158 
159  case 13:
160  SSP0CR0tmp = BIT2 | BIT3;
161  break;
162 
163  case 14:
164  SSP0CR0tmp = BIT0 | BIT2 | BIT3;
165  break;
166 
167  case 15:
168  SSP0CR0tmp = BIT1 | BIT2 | BIT3;
169  break;
170 
171  case 16:
172  SSP0CR0tmp = BIT0 | BIT1 | BIT2 | BIT3;
173  break;
174 
175  default:
176  return 0;
177  }
178 
179  if (cpol) {
180  SSP0CR0tmp |= BIT6;
181  }
182 
183  if (cpha) {
184  SSP0CR0tmp |= BIT7;
185  }
186 
187  SSP0CR1 = 0x00; // SSP0 disabled
188 
189  // Setting xx-Bit Datasize, CPOL and CPHA
190  SSP0CR0 = SSP0CR0tmp;
191 
192  // Clock Setup
193  uint32_t pclksel;
194  uint32_t cpsr;
195  lpc2387_pclk_scale(CLOCK_CORECLOCK / 1000, freq, &pclksel, &cpsr);
196  PCLKSEL1 &= ~(BIT10 | BIT11); // CCLK to PCLK divider ???
197  PCLKSEL1 |= pclksel << 10;
198  SSP0CPSR = cpsr;
199 
200  // Enable
201  SSP0CR1 |= BIT1; // SSP0 enabled
202 
203  uint32_t dummy;
204 
205  for (uint32_t i = 0; i < FIFOSIZE; i++) {
206  dummy = SSP0DR; // clear the RxFIFO
207  }
208 
209  /* to suppress unused-but-set-variable */
210  (void) dummy;
211 
212  return 1;
213 }

215 uint8_t SSP0Unprepare(uint8_t chip)
216 {
217  switch (chip) {
218  case BMA180_INTERN:
219  case SMB380_ACC: {
220  // Turn off Acceleration Sensor
221  PINSEL3 &= ~(BIT8 | BIT9 | BIT10 | BIT11 | BIT14 | BIT15 | BIT16 |
222  BIT17);
223  FIO1DIR |= BIT20 | BIT21 | BIT24;
224  FIO1DIR &= ~BIT23; // MISO as Input
225  FIO1SET = BIT20 | BIT24; /*
226  * CLK + SSEL + MOSI GPIO as Output
227  * TODO: depends on CPOL+CPHA Settings
228  */
229  FIO1CLR = BIT21;
230  break;
231  }
232 
233  case BMA180_EXTERN:
234  case L3G_EXTERN:
235  case NORDIC:
236  case NANOPAN:
237  case ACAMDMS: {
238  // Turn off Nanopan (Pins to GPIO)
239  PINSEL0 &= ~(BIT30 | BIT31); //CLK to GPIO
240  PINSEL1 &= ~(BIT0 | BIT1 | BIT2 | BIT3 | BIT4 | BIT5);
241  FIO0DIR |= BIT15 | BIT16 | BIT18; //CLK + SSEL + MOSI GPIO as Output
242  FIO0DIR &= ~BIT17; // MISO as Input
243  FIO0SET = BIT15 | BIT16; /*
244  * CLK + SSEL + MOSI GPIO as Output
245  * TODO: depends on CPOL+CPHA Settings
246  */
247  FIO0CLR = BIT18;
248  break;
249  }
250 
251  default: {
252  printf("wrong CHIP selected\n");
253  return 0;
254  }
255  }
256 
257  return 1;
258 }
