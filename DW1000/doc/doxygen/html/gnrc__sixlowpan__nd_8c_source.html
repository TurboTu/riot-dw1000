64 void gnrc_sixlowpan_nd_mc_rtr_sol(gnrc_ipv6_netif_t *iface)
65 {
66  uint32_t interval;
67  assert(iface->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN);
68  mutex_lock(&iface->mutex);
69  if (iface->rtr_sol_count < GNRC_NDP_MAX_RTR_SOL_NUMOF) {
70  DEBUG("6lo nd: retransmit multicast rtr sol in 10 sec\n");
71  iface->rtr_sol_count++;
72  interval = GNRC_SIXLOWPAN_ND_RTR_SOL_INT;
73  }
74  else {
75  unsigned int exp = (unsigned int)(iface->rtr_sol_count - GNRC_NDP_MAX_RTR_SOL_NUMOF);
76  interval = _binary_exp_backoff(1, exp);
77  if ((1U << exp) < GNRC_SIXLOWPAN_ND_MAX_RTR_SOL_INT) {
78  /* XXX Not sure if this is the correct interpretation of the truncation described in
79  * https://tools.ietf.org/html/rfc6775#section-5.3. In every source I've read the
80  * truncating value was the exponent, not the target value, so I'm very confused
81  * about this sentencing. Anyway, since 60 sec is a maximum value this should only
82  * affect the energy consumption of the implementation by sending the next RS too fast
83  * but not its interoperability. */
84  iface->rtr_sol_count++;
85  }
86  /* RFC6775, section 5.3 (https://tools.ietf.org/html/rfc6775#section-5.3)
87  * states that router solicitation should be sent slower after the
88  * initial 3 retransmissions (i.e. >= 10 secondes) and truncate "the
89  * increase of the retransmission timer at 60 seconds". */
90  if (interval < GNRC_SIXLOWPAN_ND_RTR_SOL_INT) {
91  interval = GNRC_SIXLOWPAN_ND_RTR_SOL_INT;
92  }
93  else if (interval > GNRC_SIXLOWPAN_ND_MAX_RTR_SOL_INT) {
94  interval = GNRC_SIXLOWPAN_ND_MAX_RTR_SOL_INT;
95  }
96 
97  DEBUG("6lo nd: retransmit multicast rtr sol in %" PRIu32 " sec\n", interval);
98  }
99  _rtr_sol_reschedule(iface, interval);
100  mutex_unlock(&iface->mutex);
101  gnrc_ndp_internal_send_rtr_sol(iface->pid, NULL);
102 }

105 {
106  assert(gnrc_ipv6_netif_get(nce->iface)->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN);
107  /* neighbor is not a router anymore */
108  if (!(nce->flags & GNRC_IPV6_NC_IS_ROUTER) || ipv6_addr_is_unspecified(&nce->ipv6_addr)) {
109  /* and there are no routers anymore */

121 kernel_pid_t gnrc_sixlowpan_nd_next_hop_l2addr(uint8_t *l2addr, uint8_t *l2addr_len,
122  kernel_pid_t iface, ipv6_addr_t *dst)
123 {
124  ipv6_addr_t *next_hop = NULL;
125  gnrc_ipv6_nc_t *nc_entry = NULL;
126 
127 #ifdef MODULE_FIB
128  kernel_pid_t fib_iface;
129  ipv6_addr_t next_hop_actual; /* FIB copies address into this variable */
130  /* don't look-up link local addresses in FIB */
131  if ((next_hop == NULL) && !ipv6_addr_is_link_local(dst)) {
132  size_t next_hop_size = sizeof(ipv6_addr_t);
133  uint32_t next_hop_flags = 0;
134  if ((next_hop == NULL) &&
135  (fib_get_next_hop(&gnrc_ipv6_fib_table, &fib_iface, next_hop_actual.u8, &next_hop_size,
136  &next_hop_flags, (uint8_t *)dst,
137  sizeof(ipv6_addr_t), 0) >= 0) &&
138  (next_hop_size == sizeof(ipv6_addr_t))) {
139  next_hop = &next_hop_actual;
140  }
141  }
142 #endif
143 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
144  /* next hop determination: https://tools.ietf.org/html/rfc6775#section-6.5.4 */
145  nc_entry = gnrc_ipv6_nc_get(iface, dst);
146 #ifdef MODULE_FIB
147  if ((next_hop != NULL) && (nc_entry == NULL)) {
148  nc_entry = gnrc_ipv6_nc_get(fib_iface, dst);
149  }
150 #endif
151  /* if NCE found */
152  if (nc_entry != NULL) {
153  gnrc_ipv6_netif_t *ipv6_if = gnrc_ipv6_netif_get(nc_entry->iface);
154  /* and interface is not 6LoWPAN */
155  if (!((ipv6_if == NULL) ||
156  (ipv6_if->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN)) ||
157  /* or entry is registered */
158  (gnrc_ipv6_nc_get_type(nc_entry) == GNRC_IPV6_NC_TYPE_REGISTERED)) {
159  next_hop = dst;
160  }
161  }
162 #endif
163  /* next hop determination according to: https://tools.ietf.org/html/rfc6775#section-5.6 */
164  if ((next_hop == NULL) && ipv6_addr_is_link_local(dst)) { /* prefix is "on-link" */
165  /* multicast is not handled here anyway so we don't need to check that */
166  next_hop = dst;
167  }
168  else if (next_hop == NULL) { /* prefix is off-link */
169  next_hop = gnrc_ndp_internal_default_router();
170  }
171 
172  /* no routers found */
173  if (next_hop == NULL) {
174  return KERNEL_PID_UNDEF;
175  }
176 
177  /* address resolution of next_hop: https://tools.ietf.org/html/rfc6775#section-5.7 */
178  if ((nc_entry == NULL) || (next_hop != dst)) {
179  /* get if not gotten from previous check */
180  nc_entry = gnrc_ipv6_nc_get(iface, next_hop);
181  }
182  /* If a (non-tentative) NCE for this destination exist, we can use even for
183  * link-local addresses. This should be only the case for 6LBRs. */
184  if ((ipv6_addr_is_link_local(next_hop)) &&
185  ((nc_entry == NULL) ||
186  (gnrc_ipv6_nc_get_type(nc_entry) == GNRC_IPV6_NC_TYPE_TENTATIVE))) {
187 /* in case of a border router there is no sensible way for address resolution
188  * if the interface is not given */
189 #ifdef MODULE_GNRC_SIXLOWPAN_ND_BORDER_ROUTER
190  /* if no interface is specified it is impossible to resolve the
191  * link-layer address for a link-local address on a 6LBR */
192  if (iface == KERNEL_PID_UNDEF) {
193  return KERNEL_PID_UNDEF;
194  }
195 #endif
196  kernel_pid_t ifs[GNRC_NETIF_NUMOF];
197  size_t ifnum = gnrc_netif_get(ifs);
198  /* we don't need address resolution, the EUI-64 is in next_hop's IID */
199  *l2addr_len = sizeof(eui64_t);
200  memcpy(l2addr, &next_hop->u8[8], sizeof(eui64_t));
201  _revert_iid(l2addr);
202  if (iface == KERNEL_PID_UNDEF) {
203  for (unsigned i = 0; i < ifnum; i++) {
204  gnrc_ipv6_netif_t *ipv6_if = gnrc_ipv6_netif_get(ifs[i]);
205  if ((ipv6_if != NULL) && (ipv6_if->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN)) {
206  /* always take the first 6LoWPAN interface we can find */
207  return ifs[i];
208  }
209  }
210  }
211  return iface;
212  }
213  if ((gnrc_ipv6_nc_get_type(nc_entry) == GNRC_IPV6_NC_TYPE_TENTATIVE)) {
214  return KERNEL_PID_UNDEF;
215  }
216  else {
217  if (gnrc_ipv6_nc_get_state(nc_entry) == GNRC_IPV6_NC_STATE_STALE) {
218  gnrc_ndp_internal_set_state(nc_entry, GNRC_IPV6_NC_STATE_DELAY);
219  }
220  }
221  return gnrc_ipv6_nc_get_l2_addr(l2addr, l2addr_len, nc_entry);
222 }

252 uint8_t gnrc_sixlowpan_nd_opt_ar_handle(kernel_pid_t iface, ipv6_hdr_t *ipv6,
253  uint8_t icmpv6_type, ipv6_addr_t *addr,
254  sixlowpan_nd_opt_ar_t *ar_opt,
255  uint8_t *sl2a, size_t sl2a_len)
256 {
257  eui64_t eui64;
258  gnrc_ipv6_netif_t *ipv6_iface;
259  gnrc_ipv6_nc_t *nc_entry;
260  uint8_t status = 0;
261  (void)sl2a;
262  (void)sl2a_len;
263  if (ar_opt->len != SIXLOWPAN_ND_OPT_AR_LEN) {
264  /* discard silently: see https://tools.ietf.org/html/rfc6775#section-5.5.2 */
265  return 0;
266  }
267  if (gnrc_netapi_get(iface, NETOPT_ADDRESS_LONG, 0, &eui64,
268  sizeof(eui64)) < 0) {
269  /* discard silently: see https://tools.ietf.org/html/rfc6775#section-5.5.2 */
270  return 0;
271  }
272  ipv6_iface = gnrc_ipv6_netif_get(iface);
273  nc_entry = gnrc_ipv6_nc_get(iface, addr);
274  switch (icmpv6_type) {
275  case ICMPV6_NBR_ADV:
276  if (!(ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN)) {
277  DEBUG("6lo nd: interface not a 6LoWPAN interface\n");
278  return 0;
279  }
280  if (eui64.uint64.u64 != ar_opt->eui64.uint64.u64) {
281  /* discard silently: see https://tools.ietf.org/html/rfc6775#section-5.5.2 */
282  return 0;
283  }
284  /* we expect the sender to be already in neighbor cache, if not we
285  * ignore it */
286  if (nc_entry == NULL) {
287  DEBUG("6lo nd: sending router not in neighbor cache\n");
288  return 0;
289  }
290  switch (ar_opt->status) {
291  case SIXLOWPAN_ND_STATUS_SUCCESS:
292  DEBUG("6lo nd: address registration successful\n");
293  mutex_lock(&ipv6_iface->mutex);
294  /* reschedule 1 minute before lifetime expires */
295  gnrc_ndp_internal_reset_nbr_sol_timer(nc_entry, US_PER_SEC * 60 *
296  (uint32_t)(byteorder_ntohs(ar_opt->ltime)
297  -1),
298  GNRC_NDP_MSG_NBR_SOL_RETRANS,
299  gnrc_ipv6_pid);
300  mutex_unlock(&ipv6_iface->mutex);
301  break;
302  case SIXLOWPAN_ND_STATUS_DUP:
303  DEBUG("6lo nd: address registration determined duplicated\n");
304  /* TODO: handle DAD failed case */
305  gnrc_ipv6_netif_remove_addr(iface, &ipv6->dst);
306  /* address should not be used anymore */
307  break;
308  case SIXLOWPAN_ND_STATUS_NC_FULL:
309  DEBUG("6lo nd: neighbor cache on router is full\n");
310  gnrc_ipv6_nc_remove(iface, &ipv6->src);
311  /* try to find another router */
312  gnrc_sixlowpan_nd_init(ipv6_iface);
313  break;
314  default:
315  DEBUG("6lo nd: unknown status for registration received\n");
316  break;
317  }
318  break;
319 #ifdef MODULE_GNRC_SIXLOWPAN_ND_ROUTER
320  case ICMPV6_NBR_SOL:
321  if (!(ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_SIXLOWPAN) &&
322  !(ipv6_iface->flags & GNRC_IPV6_NETIF_FLAGS_ROUTER)) {
323  DEBUG("6lo nd: interface not a 6LoWPAN or forwarding interface\n");
324  return 0;
325  }
326  if ((ar_opt->status != 0) ||
327  ipv6_addr_is_unspecified(&ipv6->src)) {
328  /* discard silently */
329  return 0;
330  }
331  /* TODO multihop DAD */
332  if ((nc_entry != NULL) &&
333  ((gnrc_ipv6_nc_get_type(nc_entry) == GNRC_IPV6_NC_TYPE_REGISTERED) ||
334  (gnrc_ipv6_nc_get_type(nc_entry) == GNRC_IPV6_NC_TYPE_TENTATIVE)) &&
335  ((nc_entry->eui64.uint64.u64 != 0) &&
336  (ar_opt->eui64.uint64.u64 != nc_entry->eui64.uint64.u64))) {
337  /* there is already another node with this address */
338  DEBUG("6lo nd: duplicate address detected\n");
339  status = SIXLOWPAN_ND_STATUS_DUP;
340  }
341  else if ((nc_entry != NULL) && (ar_opt->ltime.u16 == 0)) {
342  gnrc_ipv6_nc_remove(iface, &ipv6->src);
343  /* TODO, notify routing protocol */
344  }
345  else if (ar_opt->ltime.u16 != 0) {
346  /* TODO: multihop DAD behavior */
347  uint16_t reg_ltime;
348  if (nc_entry == NULL) {
349  if ((nc_entry = gnrc_ipv6_nc_add(iface, &ipv6->src, sl2a, sl2a_len,
350  GNRC_IPV6_NC_STATE_STALE)) == NULL) {
351  DEBUG("6lo nd: neighbor cache is full\n");
352  return SIXLOWPAN_ND_STATUS_NC_FULL;
353  }
354  nc_entry->eui64 = ar_opt->eui64;
355  }
356  nc_entry->flags &= ~GNRC_IPV6_NC_TYPE_MASK;
357  nc_entry->flags |= GNRC_IPV6_NC_TYPE_REGISTERED;
358  reg_ltime = byteorder_ntohs(ar_opt->ltime);
359  /* TODO: notify routing protocol */
360  xtimer_set_msg(&nc_entry->type_timeout, (reg_ltime * 60 * US_PER_SEC),
361  &nc_entry->type_timeout_msg, gnrc_ipv6_pid);
362  }
363  break;
364 #endif
365  default:
366  break;
367  }
368 
369  return status;
370 }
