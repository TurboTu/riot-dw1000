64 int spi_acquire(spi_t bus, spi_cs_t cs, spi_mode_t mode, spi_clk_t clk)
65 {
66  /* lock bus */
67  mutex_lock(&locks[bus]);
68  /* enable SPI device clock */
69  PMC->PMC_PCER0 |= (1 << spi_config[bus].id);
70  /* set mode and speed */
71  dev(bus)->SPI_CSR[0] = (SPI_CSR_SCBR(CLOCK_CORECLOCK / clk) | mode);
72  dev(bus)->SPI_MR = (SPI_MR_MSTR | SPI_MR_MODFDIS);
73  dev(bus)->SPI_CR = SPI_CR_SPIEN;
74 
75  return SPI_OK;
76 }
