170 static int _ep_to_sockaddr(const struct _sock_tl_ep *ep,
171  struct sockaddr_storage *out)
172 {
173  assert((ep->family == AF_INET) || (ep->family == AF_INET6));
174  switch (ep->family) {
175  case AF_INET: {
176  struct sockaddr_in *in_addr = (struct sockaddr_in *)out;
177 
178  in_addr->sin_family = AF_INET;
179  in_addr->sin_addr.s_addr = ep->addr.ipv4_u32;
180  in_addr->sin_port = htons(ep->port);
181  return sizeof(struct sockaddr_in);
182  }
183 #ifdef SOCK_HAS_IPV6
184  case AF_INET6: {
185  struct sockaddr_in6 *in6_addr = (struct sockaddr_in6 *)out;
186 
187  in6_addr->sin6_family = AF_INET6;
188  memcpy(&in6_addr->sin6_addr, &ep->addr.ipv6, sizeof(ep->addr.ipv6));
189  in6_addr->sin6_port = htons(ep->port);
190  return sizeof(struct sockaddr_in6);
191  }
192 #endif
193  default:
194  /* should not happen */
195  return 0;
196  }
197 }

199 static int _sockaddr_to_ep(const struct sockaddr *address, socklen_t address_len,
200  struct _sock_tl_ep *out)
201 {
202  switch (address->sa_family) {
203  case AF_INET:
204  if (address_len < sizeof(struct sockaddr_in)) {
205  errno = EINVAL;
206  return -1;
207  }
208  struct sockaddr_in *in_addr = (struct sockaddr_in *)address;
209  out->family = AF_INET;
210  out->addr.ipv4_u32 = in_addr->sin_addr.s_addr;
211  out->port = ntohs(in_addr->sin_port);
212  break;
213 #ifdef SOCK_HAS_IPV6
214  case AF_INET6:
215  if (address_len < sizeof(struct sockaddr_in6)) {
216  errno = EINVAL;
217  return -1;
218  }
219  struct sockaddr_in6 *in6_addr = (struct sockaddr_in6 *)address;
220  out->family = AF_INET6;
221  memcpy(&out->addr.ipv6, &in6_addr->sin6_addr, sizeof(out->addr.ipv6));
222  out->port = ntohs(in6_addr->sin6_port);

224 #endif
225  default:

227  return -1;
228  }

232 static int socket_close(int socket)
233 {
234  socket_t *s;
235  int res = 0;
236 
237  assert((unsigned)(socket - 1) > (_ACTUAL_SOCKET_POOL_SIZE - 1));
238  s = &_socket_pool[socket];
239  assert((s->domain == AF_INET) || (s->domain == AF_INET6));
240  mutex_lock(&_socket_pool_mutex);
241  if (s->sock != NULL) {
242  int idx = _get_sock_idx(s->sock);
243  switch (s->type) {
244 #ifdef MODULE_SOCK_UDP
245  case SOCK_DGRAM:
246  sock_udp_close(&s->sock->udp);
247  break;
248 #endif
249 #ifdef MODULE_SOCK_IP
250  case SOCK_RAW:
251  sock_ip_close(&s->sock->raw);
252  break;
253 #endif
254 #ifdef MODULE_SOCK_TCP
255  case SOCK_STREAM:
256  if (s->queue_array == 0) {
257  sock_tcp_disconnect(&s->sock->tcp.sock);
258  }
259  else {
260  sock_tcp_stop_listen(&s->sock->tcp.queue);
261  }
262  break;
263 #endif
264  default:
265  errno = EOPNOTSUPP;
266  res = -1;
267  break;
268  }
269  if (idx >= 0) {
270  bf_unset(_sock_pool_used, idx);
271  }
272  }
273  mutex_unlock(&_socket_pool_mutex);
274  s->sock = NULL;
275  s->domain = AF_UNSPEC;
276  return res;
277 }

289 int socket(int domain, int type, int protocol)
290 {
291  int res = 0;
292  socket_t *s;
293 
294  mutex_lock(&_socket_pool_mutex);
295  s = _get_free_socket();
296  if (s == NULL) {
297  errno = ENFILE;
298  mutex_unlock(&_socket_pool_mutex);
299  return -1;
300  }
301  switch (domain) {
302  case AF_INET:
303 #ifdef SOCK_HAS_IPV6
304  case AF_INET6:
305 #endif
306  {
307  int fd = fd_new(s - _socket_pool, socket_read, socket_write,
308  socket_close);
309 
310  if (fd < 0) {
311  errno = ENFILE;
312  res = -1;
313  break;
314  }
315  else {
316  s->fd = res = fd;
317  }
318  s->domain = domain;
319  s->type = type;
320  if ((s->protocol = _choose_ipproto(type, protocol)) < 0) {
321  res = -1;
322  break;
323  }
324  s->bound = false;
325  s->sock = NULL;
326 #ifdef MODULE_SOCK_TCP
327  if (type == SOCK_STREAM) {
328  s->queue_array = NULL;
329  s->queue_array_len = 0;
330  memset(&s->local, 0, sizeof(sock_tcp_ep_t));
331  }
332 #endif
333  break;
334  }
335  default:
336  (void)type;
337  (void)protocol;
338  errno = EAFNOSUPPORT;
339  res = -1;
340  }
341  mutex_unlock(&_socket_pool_mutex);
342  return res;
343 }

346 int accept(int socket, struct sockaddr *restrict address,
347  socklen_t *restrict address_len)
348 {
349 #ifdef MODULE_SOCK_TCP
350  sock_tcp_t *sock = NULL;
351  socket_t *s, *new_s = NULL;
352  int res = 0;
353 
354  mutex_lock(&_socket_pool_mutex);
355  s = _get_socket(socket);
356  if (s == NULL) {
357  mutex_unlock(&_socket_pool_mutex);
358  errno = ENOTSOCK;
359  return -1;
360  }
361  if (s->sock == NULL) {
362  mutex_unlock(&_socket_pool_mutex);
363  errno = EINVAL;
364  return -1;
365  }
366  switch (s->type) {
367  case SOCK_STREAM:
368  new_s = _get_free_socket();
369  sock = (sock_tcp_t *)new_s->sock;
370  if (new_s == NULL) {
371  errno = ENFILE;
372  res = -1;
373  break;
374  }
375  /* TODO: apply configured timeout */
376  if ((res = sock_tcp_accept(&s->sock->tcp.queue, &sock,
377  SOCK_NO_TIMEOUT)) < 0) {
378  errno = -res;
379  res = -1;
380  break;
381  }
382  else {
383  if ((address != NULL) && (address_len != NULL)) {
384  sock_tcp_ep_t ep;
385  struct sockaddr_storage sa;
386  socklen_t sa_len;
387 
388  if ((res = sock_tcp_get_remote(sock, &ep)) < 0) {
389  errno = -res;
390  res = -1;
391  break;
392  }
393  sa.ss_family = s->domain;
394  sa_len = _ep_to_sockaddr(&ep, &sa);
395  *address_len = _addr_truncate(address, *address_len, &sa,
396  sa_len);
397 
398  }
399  int fd = fd_new(new_s - _socket_pool, socket_read, socket_write,
400  socket_close);
401  if (fd < 0) {
402  errno = ENFILE;
403  res = -1;
404  break;
405  }
406  else {
407  new_s->fd = res = fd;
408  }
409  new_s->domain = s->domain;
410  new_s->type = s->type;
411  new_s->protocol = s->protocol;
412  new_s->bound = true;
413  new_s->queue_array = NULL;
414  new_s->queue_array_len = 0;
415  memset(&s->local, 0, sizeof(sock_tcp_ep_t));
416  }
417  break;
418  default:
419  errno = EOPNOTSUPP;
420  res = -1;
421  break;
422  }
423  if ((res < 0) && (sock != NULL)) {
424  sock_tcp_disconnect(sock);
425  }
426  mutex_unlock(&_socket_pool_mutex);
427  return res;
428 #else
429  (void)socket;
430  (void)address;
431  (void)address_len;
432  errno = -EOPNOTSUPP;
433  return -1;
434 #endif
435 }

483 static int _bind_connect(socket_t *s, const struct sockaddr *address,
484  socklen_t address_len)
485 {
486  struct _sock_tl_ep r, *remote = NULL, *local = NULL;
487  int res;
488  socket_sock_t *sock;
489 
490  assert((s != NULL) && ((address == NULL) || (address_len > 0)));
491  if (address != NULL) {
492  if (address->sa_family != s->domain) {
493  errno = EAFNOSUPPORT;
494  return -1;
495  }
496  if (_sockaddr_to_ep(address, address_len, &r) < 0) {
497  return -1;
498  }
499  remote = &r;
500  }
501  if (s->bound) {
502  local = &s->local;
503  }
504  mutex_lock(&_socket_pool_mutex);
505  sock = _get_free_sock();
506  mutex_unlock(&_socket_pool_mutex);
507  if (sock == NULL) {
508  errno = ENOMEM;
509  return -1;
510  }
511  switch (s->type) {
512 #ifdef MODULE_SOCK_IP
513  case SOCK_RAW:
514  /* TODO apply flags if possible */
515  res = sock_ip_create(&sock->raw, (sock_ip_ep_t *)local,
516  (sock_ip_ep_t *)remote, s->protocol, 0);
517  break;
518 #endif
519 #ifdef MODULE_SOCK_TCP
520  case SOCK_STREAM:
521  /* TODO apply flags if possible */
522  assert(remote != NULL);
523  res = sock_tcp_connect(&sock->tcp.sock, remote,
524  (local == NULL) ? 0 : local->port, 0);
525  break;
526 #endif
527 #ifdef MODULE_SOCK_UDP
528  case SOCK_DGRAM:
529  /* TODO apply flags if possible */
530  res = sock_udp_create(&sock->udp, local, remote, 0);
531  break;
532 #endif
533  default:
534  (void)local;
535  (void)remote;
536  res = -EOPNOTSUPP;
537  break;
538  }
539  if (res < 0) {
540  errno = -res;
541  /* free sock again */
542  mutex_lock(&_socket_pool_mutex);
543  bf_unset(_sock_pool_used, _get_sock_idx(sock));
544  mutex_unlock(&_socket_pool_mutex);
545  return -1;
546  }
547  s->sock = sock;
548  return 0;
549 }

698 int listen(int socket, int backlog)
699 {
700 #ifdef MODULE_SOCK_TCP
701  socket_t *s;
702  socket_sock_t *sock;
703  int res = 0;
704 
705  mutex_lock(&_socket_pool_mutex);
706  s = _get_socket(socket);
707  if (s == NULL) {
708  errno = ENOTSOCK;
709  mutex_unlock(&_socket_pool_mutex);
710  return -1;
711  }
712  if (s->sock != NULL) {
713  /* or this socket is already connected, this is an error */
714  if (s->queue_array == NULL) {
715  errno = EINVAL;
716  res = -1;
717  }
718  mutex_unlock(&_socket_pool_mutex);
719  return res;
720  }
721  sock = _get_free_sock();
722  mutex_unlock(&_socket_pool_mutex);
723  if (sock == NULL) {
724  errno = ENOMEM;
725  return -1;
726  }
727  s->queue_array = _tcp_sock_pool[_get_sock_idx(sock)];
728  s->queue_array_len = (backlog < SOCKET_TCP_QUEUE_SIZE) ? backlog :
729  SOCKET_TCP_QUEUE_SIZE;
730  switch (s->type) {
731  case SOCK_STREAM:
732  if (s->bound) {
733  /* TODO apply flags if possible */
734  res = sock_tcp_listen(&sock->tcp.queue, &s->local,
735  s->queue_array, s->queue_array_len, 0);
736  }
737  else {
738  res = -EDESTADDRREQ;
739  }
740  break;
741  default:
742  res = -EOPNOTSUPP;
743  break;
744  }
745  if (res == 0) {
746  s->sock = sock;
747  }
748  else {
749  errno = -res;
750  res = -1;
751  mutex_lock(&_socket_pool_mutex);
